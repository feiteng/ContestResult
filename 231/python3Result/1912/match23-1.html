class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        mod = 10 ** 9 + 7
        G = defaultdict(list)
        for a, b, w in edges:
            G[a].append((b, w))
            G[b].append((a, w))
        hp = [(0, n)]
        dic = {n: 0}
        while hp:
            step, i = heappop(hp)
            for j, w in G[i]:
                
if j not in dic or dic[j] > step + w:
                    dic[j] = step + w
                    heappush(hp, (dic[j], j))
        G2 = defaultdict(list)
        for a, b, _ in edges:
            
if dic[a] > dic[b]:
                G2[a].append(b)
            elif dic[b] > dic[a]:
                G2[b].append(a)
        
        @lru_cache(None)
        def dfs(i):
            if i == n: return 1
            ans = 0
            for j in G2[i]:
                ans += dfs(j)
                ans %= mod
            return ans
        
        return dfs(1)
