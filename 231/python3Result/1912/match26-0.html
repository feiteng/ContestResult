class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        graph = defaultdict(list)
        for u, v, w in edges:
            graph[u].append([v, w])
            graph[v].append([u, w])
            
        distance = [float('inf')] * (n + 1)
        pq = [(0, n)]
        distance[n] = 0
        
        while pq:
            d, u = heapq.heappop(pq)
            if d > distance[u]:
                continue
            for nei, w in graph[u]:
                if distance[nei] > d + w:
                    distance[nei] = d + w
                    heapq.heappush(pq, [d + w, nei])
        
        MOD = 10 ** 9 + 7
        
        @lru_cache(None)
        def dfs(node):
            if node == n:
                return 1
            ans = 0
            for nei, w in graph[node]:
                if distance[node] > distance[nei]:
                    ans += dfs(nei)
                    ans %= MOD
            return ans
        
        return dfs(1)
