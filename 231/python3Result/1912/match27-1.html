class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        MOD = int(1e9) + 7
        g = [{} for _ in range(n)]
        for x, y, w in edges:
            x -= 1
            y -= 1
            g[x][y] = g[y][x] = w
        dist = [float(inf)] * n
        dist[n-1] = 0
        h 
= [(0, n-1)]
        while h:
            d, i = heapq.heappop(h)
            if dist[i] != d:
                continue
            for ch in g[i]:
                nd = d + g[i]
[ch]
                if dist[ch] > nd:
                    dist[ch] = nd
                    heapq.heappush(h, (nd, ch))
        
        @lru_cache(None)
        def f(idx):
            if idx == n-1:
                return 1
            ret = 0
            for ch in g[idx]:
                if dist[ch] < dist[idx]:
                    ret += f(ch)
            return ret % MOD
        
        return f(0)
