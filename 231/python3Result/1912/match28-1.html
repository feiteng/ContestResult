class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        adj
=[[] for _ in range(n)]
        for u,v,w in edges:
            u-=1
            v-=1
            adj[u].append((v,w))
            adj[v].append((u,w))
            
        distance=[10**10]*n
        distance[n-1]=0
        visited=[0]*n
        h=[(0,n-1)]
        while h:
            _,node=heappop(h)
            if visited[node]:
                continue
            visited[node]=1
            for c,w in adj[node]:
                if distance[node]+w<distance[c]:
                    distance[c]=distance[node]+w
                    heappush(h,(distance
[c],c))
                    
        @lru_cache(None)       
        def npath(node):
            if node==n-1:
                return 1
            ret=0
            for c,_ in adj[node]:
                if distance[c]<distance[node]:
                    ret+=npath(c)
            return ret
        
        return npath(0)%(10**9+7)
