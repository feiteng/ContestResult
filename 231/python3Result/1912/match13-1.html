import collections

import functools
class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        d=collections.defaultdict(list)
        for u,v,wght in edges:
            d[u].append([v,wght])
            d[v].append([u,wght])
        dis=[float("inf") for _ in range(n+1)]
        dis[n]=0
        cur=[n]
        while cur:
            temp=[]
            for c in cur:
                for u, weight in d[c]:
                    if dis[u]>dis[c]+weight:  # update 
                        temp.append(u)
                        dis[u]=dis[c]+weight
            cur=list(set(
temp))
        # dfs 
        # memorize everything
        @functools.lru_cache(None)
        def dfs(cur):
            if cur==n: return 1 
            res=0
            for u, w in d[cur]:
                if dis[u]<dis[cur]:
                    res+=dfs(u)
            return res
        return dfs(1)%(10**9+7)
