class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        links=defaultdict
(list)
        for u,v,w in edges:
            links[u].append((v,w))
            links[v].append((u,w))
            
        h = [(0, n)] #dis,node
        visited = {}
        while h and len(visited) < n:
            weight, cur 
= heappop(h)
            if cur in visited:
                continue
            visited[cur] = weight
            for i, w in links[cur]:
                if w > 0 and i not in visited:
                    heappush(h, (weight + w, i))
        
        @cache
        def find(root):
            if root==n:
                return 1
            ret=0
            for v,_ in links[root]:
                if visited[v]<visited[root]:
                    ret+=find(v)
            return ret
        
        ans=find(1)
        find.cache_clear()
        return ans%(int(1e9+7))
