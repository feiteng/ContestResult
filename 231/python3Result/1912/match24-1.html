class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        d = [-1]*(n+1)
        G = [[]for _ in range(n+1)]
        for u, v, w in edges:
            G[u].append((v, w))
            G[v].append((u, w))
        h = [(
0, n)]
        while h:
            w, u = heapq.heappop(h)
            if d[u]!=-1: continue
            d[u] = w
            for v, w1 in G[u]:
                if d[v]== -1:
                    heapq.heappush(h, (w+
w1, v))
        @lru_cache(None)
        def dfs(u, limit):
            if u==1: return 1
            res = 0
            for v, _ in G[u]:
                if d[v]>limit: res += dfs(v, d[v])
            return res
        return dfs(n, 0) % (10**9+7)
