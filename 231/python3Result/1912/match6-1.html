class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        
        MOD 
= 10 ** 9 + 7
        
        graph = collections.defaultdict(list)
        for f, t, v in edges:
            graph[f].append((t, v))
            graph[t].append((f, v))
            
        visited = set([n])
        heap = []
        for nxt, v in graph[n]:
            heapq.heappush(heap, (v, nxt))
        d = collections.defaultdict(int)
        d[n] = 0
        
        while heap:
            l, node = heapq.heappop(heap)
            if node in visited:
                continue 
            else:
                visited.add(
node)
            d[node] = l 
            for nxt, length in graph[node]:
                if nxt not in visited: 
                    heapq.heappush(heap, (l + length, nxt))
        
        @functools.lru_cache(None)
        def dp(i = 1):
            
if i == n:
                return 1
            
            ans = 0
            for nxt, _ in graph[i]:
                if d[i] > d[nxt]:
                    ans += dp(nxt)
            return ans 
    
        return dp() % MOD
