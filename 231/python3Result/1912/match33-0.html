class Solution:
    def countRestrictedPaths(self, n: int, edges: List[List[int]]) -> int:
        # use dijkstras to calculate distance to n
        graph = defaultdict(dict)
        
        for a, b, w in edges:
            graph[a][b] = w
            graph[b][a] = w
        
        distToEnd = [float('inf')] * (n + 1)
        
        q = [(0, n)]
        
        while q:
            dist, node = heapq.heappop(q)
            if dist >= distToEnd[node]:
                continue
            
            distToEnd[node] = dist
            for neigh in graph[node]:
                heapq.heappush(q, (dist + graph[node][neigh], neigh))
        
        # Now we do a dfs from 1
        @lru_cache(None)
        def dfs(node):
            if node == n:
                return 1
            result = 0
            for neigh in graph[node]:
                if distToEnd[neigh] >= distToEnd[node]:
                    continue
                result += dfs(neigh)
            return result
        
        return dfs(1) % (10 ** 9 + 7)
