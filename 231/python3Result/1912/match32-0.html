class Solution:
    def countRestrictedPaths(self, n, edges):

        def Dijkstra(n, edges, start):
            dist = [float('inf')] * n
            dist[start - 1] = 0
            adj_list = defaultdict(list)
            
            for a, b, w in edges:
                adj_list[a-1].append((b-1, w))
                adj_list[b-1].append((a-1, w))
            
            heap = [(0, start - 1)]

            while heap:
                (min_dist, min_dist_id) = heappop(heap)
                for neibh, weight in adj_list[min_dist_id]:
                    candidate = dist[min_dist_id] + weight
                    if candidate < dist[neibh]:
                        dist[neibh] = candidate
                        heappush(heap, (weight + min_dist, neibh))
                        
            return dist

        dist = Dijkstra(n, edges, n)

        graph = defaultdict(list)

        for x, y, w in edges:
            graph[x-1].append(y-1)
            graph[y-1].append(x-1)

        #print("@@@", dist, graph)

        @lru_cache(None)
        def dfs(i):
            #print("!!!!", i)
            if i == n-1: return 1
            ans = 0
            for neib in graph[i]:
                if dist[i] > dist[neib]:
                    ans += dfs(neib)

            return ans

        return dfs(0) % (10**9 + 7)
