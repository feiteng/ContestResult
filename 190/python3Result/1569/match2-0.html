class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        
        n = len(nums1)
        m = len(nums2)
        
        mat_sum = [[None for _ in range(m)] for _ in range(n)]
        
        mat_sum[0][0] = nums1[0]*nums2[0]
        
        for i in range(1, n):
            mat_sum[i][0] = max(mat_sum[i-1][0], nums1[i]*nums2[0])
            
        for i in range(1, m):
            mat_sum[0][i] = max(mat_sum[0][i-1], nums1[0]*nums2[i])
            
        for i in range(1, n):
            for j in range(1, m):
                mat_sum[i][j] = max(mat_sum[i-1][j],
                                    mat_sum[i][j-1],
                                    mat_sum[i-1][j-1] + nums1[i]*nums2[j],
                                    nums1[i]*nums2[j])
                
        return mat_sum[n-1][m-1]
