class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        """
        dp[i][j] stands for the maximum dot product of nums1[: i] and nums2[: j], then
        dp[i][j] = max(dp[i][j - 1], dp[i - 1][j], max(0, dp[i - 1][j - 1]) + nums1[i] * nums2[j])
        """
        n = len(nums1)
        m = len(nums2)
        dp = [[float('-inf')] * (m + 1) for _ in range(n + 1)]
        res = float('-inf')
        for row in range(1, n + 1):
            for col in range(1, m + 1):
                dp[row][col] = max(dp[row][col - 1], dp[row - 1][col], max(0, dp[row - 1][col - 1]) + nums1[row - 1] * nums2[col - 1])
                res = max(res, dp[row][col])
        return res
