class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        len1 = len(nums1)
        len2 = len(nums2)
        dp = [[0 for _ in range(len2)] for _ in range(len1)]
        dp[len1-1][len2-1] = nums1[len1-1]*nums2[len2-1]
        for i in range(len1-2, -1, -1):
            dp[i][len2-1] = max(dp[i+1][len2-1], nums1[i]*nums2[len2-1])
        for j in range(len2-2, -1, -1):
            dp[len1-1][j] = max(dp[len1-1][j+1], nums1[len1-1]*nums2[j])
        
        for i in range(len1-2, -1, -1):
            for j in range(len2-2, -1, -1):
                dp[i][j] = max(dp[i+1][j], dp[i][j+1], dp[i+1][j+1] + nums1[i]*nums2[j], nums1[i]*nums2[j])
                # dp[i][j] = max(dp[i+1][j], dp[i][j+1], nums1[i]*nums2[j])
                # for k in range(j, len2-1):
                    # dp[i][j] = max(dp[i][j], dp[i+1][k+1] + nums1[i]*nums2[k])
        return dp[0][0]
