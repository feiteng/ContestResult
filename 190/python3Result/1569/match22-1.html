class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        len1, len2 = len(nums1), len(nums2)
        max_dots = [[float('-inf')] * len2 for _ in range(len1)]
        
                
        for i in range(len1 - 1, -1, -1):
            for j in range(len2 - 1, -1, -1):
                cans = []
                
                if i + 1 < len1:
                    cans.append(max_dots[i + 1][j])
                if j + 1 < len2:
                    cans.append(max_dots[i][j + 1])
                
                if i + 1 < len1 and j + 1 < len2:
                    cans.append(max_dots[i + 1][j + 1] + nums1[i] * nums2[j])

                cans.append(nums1[i] * nums2[j])
                    
                max_dots[i][j] = max(cans)
        
        return max_dots[0][0]
