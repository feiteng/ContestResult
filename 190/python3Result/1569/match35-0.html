class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        dp = [[0] * len(nums1) for i in range(len(nums2))]
        dp[0][0] = nums1[0] * nums2[0]
        for j in range(1, len(nums1)):
            dp[0][j] = max(nums1[j] * nums2[0], dp[0][j - 1])
        for i in range(1, len(nums2)):
            dp[i][0] = max(nums1[0] * nums2[i], dp[i - 1][0])
        
        for i in range(1, len(nums2)):
            for j in range(1, len(nums1)):
                dp[i][j] = max(dp[i - 1][j - 1] + nums1[j] * nums2[i],
                               nums1[j] * nums2[i],
                               dp[i - 1][j],
                               dp[i][j - 1])
        # print(dp)
        return dp[len(nums2) - 1][len(nums1) - 1]
