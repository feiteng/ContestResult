class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        def dfs(id1, id2, n1, n2, memo):
            if (id1 == n1 or id2 == n2):
                return 0, 0
            elif (id1 == n1-1 and id2 == n2-1):
                return nums1[-1] * nums2[-1],1
            
            tup = tuple((id1,id2))
            if (tup in memo) :
                return memo[tup], 1

            pod1 = nums1[id1]*nums2[id2];
            pod2,count2 = dfs(id1+1, id2+1, n1, n2, memo)    
            pod3,count3 = dfs(id1+1, id2, n1, n2, memo)    
            pod4,count4 = dfs(id1, id2+1, n1, n2, memo)   
            max_prod = pod1
            if count2 > 0:
                max_prod = max(max_prod, pod1 + pod2, pod2)
                
            if count3 > 0:
                max_prod = max(max_prod, pod3)

            if count4 > 0:
                max_prod = max(max_prod, pod4)
                
            memo[tup] = max_prod
            return max_prod,1
        memo = {}
        prod, count = dfs(0,0, len(nums1), len(nums2), memo)           
        return prod
