class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        R = len(nums1)
        C = len(nums2)
        
        tbl = [[0 for c in range(C)] for r in range(R)]
        tbl[0][0] = nums1[0] * nums2[0]
        for c in range(1, C):
            tbl[0][c] = max(tbl[0][c-1], nums1[0]*nums2[c])
            
        for r in range(1, R):
            tbl[r][0] = max(tbl[r-1][0], nums1[r]*nums2[0])
            
        for r in range(1,R):
            for c in range(1,C):
                tbl[r][c] = max(tbl[r-1][c], tbl[r][c-1], tbl[r-1][c-1]+nums1[r]*nums2[c], nums1[r]*nums2[c])
                
        #print(tbl)        
        
        return tbl[r][c]
