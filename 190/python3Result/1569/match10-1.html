class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        @functools.lru_cache(None)
        def helper(idx1, idx2):
            if idx1 == len(nums1) or idx2 == len(nums2):
                return float("-inf")

            ans = float("-inf")
            ans = max(ans, max(helper(idx1 + 1, idx2 + 1), 0) + nums1[idx1] * nums2[idx2],
                      helper(idx1, idx2 + 1), helper(idx1 + 1, idx2))
            return ans

        return helper(0, 0)
