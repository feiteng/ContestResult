class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = map(len, (nums1, nums2))
        if not (n1 and n2):
            return 0
        res = nums1[0] * nums2[0]
        dp = [[float('-inf')] * (1 + n2) for _ in range(1 + n1)]

        for i in range(n1):
            for j in range(n2):
                tmp = nums1[i] * nums2[j]
                dp[i][j] = max(tmp, tmp + dp[i - 1][j - 1], dp[i][j - 1], dp[i - 1][j])
                res = max(res, dp[i][j])
        return res
