class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        M = len(nums1)
        N = len(nums2)
        
        
        dp = [[0] * N for _ in range(0, M)]
        
        dp[0][0] = nums1[0] * nums2[0]
        for n in range(1, N):
            dp[0][n] = max(nums1[0]*nums2[n], dp[0][n-1])
        
        for m in range(1, M):
            dp[m][0] = max(nums1[m]*nums2[0], dp[m-1][0])
            for n in range(1, N):
                dp[m][n] = max(nums1[m]*nums2[n], nums1[m]*nums2[n] + dp[m-1][n-1], dp[m-1][n], dp[m][n-1])
        
        
        return dp[-1][-1]
