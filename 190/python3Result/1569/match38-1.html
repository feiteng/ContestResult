        """
        dp[i][j] = max(dp[i-1][j], dp[i][j-1], dp[i-1][j-1] + nums[i][j])
        """
        m = len(nums1) 
        n = len(nums2)
        dp = [[float('-inf')] * n for _ in range(m)]
        res = float('-inf')
        for i in range(m):
            if i == 0:
                dp[0][0] = nums1[0]* nums2[0]
            else:
                dp[i][0] = max(dp[i-1][0], nums1[i] * nums2[0])
            res = max(dp[i][0], res) 
        for j in range(n):
            if j == 0:
                dp[0][j] = nums1[0]* nums2[0]
            else:
                dp[0][j] = max(dp[0][j-1], nums1[0] * nums2[j])
            res = max(dp[0][j], res) 
        
for i in range(1, m):
            for j in range(1, n):
                dp[i][j] = max(dp[i-1][j], dp[i][j-1], max(0,dp[i-1][j-1]) + nums1[i]* nums2[j])
                res = max(dp[i][j], res) 
        return res
