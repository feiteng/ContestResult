class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        dim1, dim2 = len(nums1), len(nums2)
        dp = [[0] * dim2 for _ in range(dim1)]
        dp[0][0] = nums1[0] * nums2[0]
        
        for i in range(1, dim1):
            dp[i][0] = max(dp[i-1][0], nums1[i] * nums2[0])
            # print(dp[i][0])
        for j in range(1, dim2):
            dp[0][j] = max(dp[0][j-1], nums2[j] * nums1[0])
            # print(dp[0][j])
        for i in range(1, dim1):
            for j in range(1, dim2):
                dp[i][j] = max(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]+nums1[i]*nums2[j], dp[i-1][j-1], nums1[i]*nums2[j])

                # print(dp[i][j])
        return dp[-1][-1]
