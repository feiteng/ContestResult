class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        m = [[-10e9] * (len(nums1) + 1) for _ in range(len(nums2) + 1)]
        
        for x in range(1, len(nums2) + 1):
            for y in range(1, len(nums1) + 1):
                #print(x, y)
                m[x][y] = max(nums1[y-1] * nums2[x-1], nums1[y-1] * nums2[x-1] + m[x-1][y-1], m[x-1][y], m[x][y-1])
                    
        return m[len(nums2)][len(nums1)]
