class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:

        def dfs(idx1, idx2, n1, n2, memo):
            if (idx1 == n1 or idx2 == n2):
                return 0,0
            elif (idx1 == n1-1 and idx2 == n2-1):
                return nums1[-1]*nums2[-1],1
        
            tup = tuple((idx1, idx2))
            if (tup in memo):
                return memo[tup],1
            
            prod1 = nums1[idx1]*nums2[idx2]
            prod2,count2 = dfs(idx1+1, idx2+1, n1, n2, memo)
            prod3,count3 = dfs(idx1+1, idx2, n1, n2, memo)
            prod4,count4 = dfs(idx1, idx2+1, n1, n2, memo)
            
            max_prod = prod1
            if count2 > 0:
                max_prod = max(max_prod, prod1+prod2, prod2)
            if (count3 > 0):
                max_prod = max(max_prod, prod3)
            if (count4 > 0):
                max_prod = max(max_prod, prod4)
            
            memo[tup] = max_prod
            return max_prod,1
            
        memo = {}
        prod,count = dfs(0,0,len(nums1), len(nums2), memo)
        return prod
