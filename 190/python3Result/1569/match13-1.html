class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        p1,p2 = map(len,(nums1,nums2))
        if not(p1 and p2):
            return 0
        res = nums1[0]*nums2[0]
        
        
        
        dp = [[float('-inf')]*(1+p2) for _ in range(1+p1)]
        
        for i in range(p1):
            for j in range(p2):
                temp = nums1[i] *nums2[j]
                
                
                dp[i][j]  = max(temp,temp + dp[i-1][j-1],dp[i][j-1],dp[i-1][j])
                res = max(res,dp[i][j])
        return res
