class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = len(nums1), len(nums2)
        dp = [[float('-inf')] * (n2 + 1) for _ in range(n1 + 1)]
        for i in range(n1):
            for j in range(n2):
                dp[i + 1][j + 1] = max(nums1[i] * nums2[j], dp[i][j] + max(0, nums1[i] * nums2[j]), dp[i + 1][j], dp[i][j + 1])
        # print(dp)
        return dp[-1][-1]
