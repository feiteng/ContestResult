class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        l1, l2 = len(nums1), len(nums2)
        dp = [[float('-inf')] * l2  for _ in range(l1)]
        for idx in range(l1 - 1, -1, -1):
            for idx2 in range(l2 - 1, -1, -1):
                cands = []
                if idx < l1 - 1:
                    cands.append(dp[idx + 1][idx2])
                if idx2 < l2 - 1:
                    cands.append(dp[idx][idx2 + 1])
                if idx < l1 - 1 and idx2 < l2 - 1:
                    cands.append(dp[idx + 1][idx2 + 1] + nums1[idx] * nums2[idx2])
                cands.append(nums1[idx] * nums2[idx2])
                dp[idx][idx2] = max(cands)
        return dp[0][0]
