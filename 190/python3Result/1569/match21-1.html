class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = len(nums1), len(nums2)
        dp = [[0 for i in range(n2)] for j in range(n1)]
        dp[0][0] = nums1[0] * nums2[0]
        for c in range(1, n2):
            dp[0][c] = max(dp[0][c-1], nums1[0] * nums2[c])
        for r in range(1, n1):
            dp[r][0] = max(dp[r-1][0], nums1[r] * nums2[0])
        # print(dp)
        for r in range(1, n1):
            for c in range(1, n2):
                dp[r][c] = max([dp[r-1][c], dp[r][c-1], dp[r-1][c-1] + nums1[r]*nums2[c], nums1[r]*nums2[c]])
        # print(dp)
        return dp[n1-1][n2-1]
