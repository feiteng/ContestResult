class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        m,n = len(nums1), len(nums2)
        dp = [ [0 for _ in range(n)] for _ in range(m)]
        
        for i in range(m):
            for j in range(n):
                
                dp[i][j] = max(dp[i][j-1] if j>0 else -99999, (dp[i-1][j-1] if i>0 and j>0 and dp[i-1][j-1]> 0 else 0)+ nums1[i]*nums2[j], dp[i-1][j] if i>0 else -99999)
        # print(dp)
        return dp[m-1][n-1]
