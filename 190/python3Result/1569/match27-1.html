class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        n = len(nums1)
        m = len(nums2)
        dp = [[0]*m for _ in range(n)]
        dp[n-1][m-1] = nums1[n-1]*nums2[m-1]
        for j in range(m-2, -1, -1):
            dp[n-1][j] = max(dp[n-1][j+1], nums1[n-1]*nums2[j])
        for i in range(n-2, -1, -1):
            dp[i][m-1] = max(dp[i+1][m-1], nums1[i]*nums2[m-1])
        for i in range(n-2, -1, -1):
            for j in range(m-2, -1, -1):
                dp[i][j] = max(dp
[i+1][j], dp[i][j+1], max(dp[i+1][j+1],0) + nums1[i]*nums2[j])
                
#        for i in range(n):
#            print(f'dp[{i}] = {dp[i]}')
        return dp[0][0]
