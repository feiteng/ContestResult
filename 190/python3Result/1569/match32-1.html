class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        n1, n2 = len(nums1), len(nums2)
        dp = [[float('-inf') for _ in range(n2 + 1)] for _ in range(n1 + 1)]
        for i in range(n1):
            for j in range(n2):
                p = nums1[i] * nums2[j]
                dp
[i + 1][j + 1] = max(dp[i + 1][j + 1], dp[i][j + 1], dp[i + 1][j], dp[i][j], p, dp[i][j] + p)
        return dp[n1][n2]
