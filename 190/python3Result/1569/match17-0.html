class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        m, n = len(nums1), len(nums2)
        F = [[0 for i in range(n)] for j in range(m)]
        F[0][0] = nums1[0] * nums2[0]
        for i in range(1, m):
            F[i][0] = max(F[i - 1][0], nums1[i] * nums2[0])
        for i in range(1, n):
            F[0][i] = max(F[0][i - 1], nums2[i] * nums1[0])
        for i in range(1, m):
            for j in range(1, n):
                F[i][j]  = max(F[i][j - 1], F[i - 1][j], max(0, F[i - 1][j - 1]) + nums1[i] * nums2[j])
        #print(F[1][4])
        return F[-1][-1]
