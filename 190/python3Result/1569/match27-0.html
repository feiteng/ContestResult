class Solution:
    def maxDotProduct(self, nums1: List[int], nums2: List[int]) -> int:
        m = len(nums1)
        n = len(nums2)
        f = [[0] * n for i in range(m)]
        f[0][0] = nums1[0] * nums2[0]
        for i in range(1, m):
            f[i][0] = max(f[i-1][0], nums1[i] * nums2[0])
        for j in range(1, n):
            f[0][j] = max(f[0][j-1], nums1[0] * nums2[j])
        for i in range(1, m):
            for j in range(1, n):
                f[i][j] = max(max(f[i-1][j-1], 0) + nums1[i] * nums2[j], max(f[i-1][j], f[i][j-1]))
        return f[m-1][n-1]
