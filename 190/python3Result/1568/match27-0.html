# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        
        if not root:
            return 0
        
        self._count = 0
        info = collections.defaultdict(int)
        info[root.val] = 1
        self._dfs(root, info)
        return self._count
    
    def _dfs(self, node, info):
        if not node:
            return
        if not node.left and not node.right and self._check(info):
            self._count += 1
            return
        if node.left:
            info[node.left.val] += 1
            self._dfs(node.left, info)
            info[node.left.val] -= 1
        if node.right:
            info[node.right.val] += 1
            self._dfs(node.right, info)
            info[node.right.val] -= 1
    
    def _check(self, info):
        one_count = 0
        for val, count in info.items():
            if count % 2:
                one_count += 1
                if one_count > 1:
                    return False
        return True
