# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        
        #print("---")
        def visit(root, path, out):
            if not root:
                return
            
            if not root.left and not root.right:
                #print(cur, root.val)
                path[root.val] ^= 1
                
                #print(path)
                a = sum([path[i] for i in range(1, 10)])
                if a == 1 or a == 0:
                    out[0] += 1
                    
                path[root.val] ^= 1
                return
            
            path[root.val] ^= 1

            visit(root.left, path, out)
            visit(root.right, path, out)

            path[root.val] ^= 1
            
        path = [0] * 10
        out = [0]
        visit(root, path, out)
        return out[0]
