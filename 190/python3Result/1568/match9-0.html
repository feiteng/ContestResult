# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        bucket = [0] * 10
        self.single = 0
        self.res = 0
        def dfs(root):
            # if root is None:
                # if self.single <= 1:
                #     self.res += 1
                # return
            bucket[root.val] += 1
            # print(self.single)
            self.single += 1 if bucket[root.val] % 2 == 1 else -1
            # print(root.val, bucket, self.single, self.res)
            if root.left is None and root.right is None and self.single <= 1:
                self.res += 1
            if root.left:
                dfs(root.left)
            if root.right:
                dfs(root.right)
            bucket[root.val] -= 1
            self.single += 1 if bucket[root.val] % 2 == 1 else -1
        dfs(root)
        return self.res
