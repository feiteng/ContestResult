# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
p = set()

def flop(v):
    if v not in p:
        p.add(v)
    else:
        p.remove(v)

def count(node):
    flop(node.val)

    ct = 0
    if node.left == None and node.right == None:
        if len(p) == 0 or len(p) == 1:
            ct += 1
    else:
        if node.left != None:
            ct += count(node.left)
        if node.right != None:
            ct += count(node.right)

    flop(node.val)
    return ct

class Solution:
    def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        return count(root)
