class Solution:
    def pseudoPalindromicPaths(self, root: TreeNode) -> int:
        self.ans = 0
        cur = set()
        def helper(root: TreeNode):
            if not root: return 
            
            if root.val in cur: cur.remove(root.val)
            else: cur.add(root.val)
            
            if not root.left and not root.right and len(cur) <= 1:
                self.ans += 1
            helper(root.left)
            helper(root.right)
            
            if root.val in cur: cur.remove(root.val)
            else: cur.add(root.val)
        helper(root)
        return self.ans
