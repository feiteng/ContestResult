class Solution:
    def __init__(self):
        self.res = 0
        self.cnt = [0] * 9
    def pseudoPalindromicPaths(self, root: TreeNode) -> int:
        
def dfs(root: TreeNode):
            if root==None:
                return
            self.cnt[root.val-1] += 1
            if root.left==None and root.right==None:
                oddCnt = 0
                for i in range(9):
                    if self.cnt[i] % 2 ==1: oddCnt+=1
                if oddCnt<=1:self.res += 1
            dfs(root.left)
            dfs(root.right)
            self.cnt[root.val-1] -= 1
        dfs(root)
        return self.res
