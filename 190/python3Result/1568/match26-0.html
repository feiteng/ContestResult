# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

from collections import defaultdict

class Solution:
    
    result = 0
    odd_count = 0
    
    def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        
        count_map = defaultdict(int)
        
        def helper(node):
            count_map[node.val] += 1
            if count_map[node.val] % 2 == 1:
                self.odd_count += 1
            else:
                self.odd_count -= 1
                
            if not node.left and not node.right:
                if self.odd_count < 2:
                    self.result += 1
            
            else:
                
                if node.left:
                    helper(node.left)

                if node.right:
                    helper(node.right)
                
            count_map[node.val] -= 1
            if count_map[node.val] % 2 == 1:
                self.odd_count += 1
            else:
                self.odd_count -= 1
            
        
        if root:
            helper(root)
            
        return self.result
