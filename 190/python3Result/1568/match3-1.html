# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    
def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        self.answer = 0
        def recursive(node, count, single_set):
            if node == None:
                return
            if node.val in single_set:
                single_set.remove(node.val)
            else:
                single_set.add(node.val)
            if node.left == None and node.right == None:
                if count % 2 == 1 and len(single_set) <= 1:
                    self.answer += 1
                
if count % 2 == 0 and len(single_set) == 0:
                    self.answer += 1
            recursive(node.left, count+1, single_set)
            recursive(node.right, count+1, single_set)
            if node.val in single_set:
                single_set.remove(node.val)
            else:
                single_set.add(node.val)
        recursive(root, 1, set())
        return self.answer
