# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    ans = 0
    def pseudoPalindromicPaths (self, root: TreeNode) -> int:
        self.ans = 0
        
        def dfs(node, counts):
            
            #print(counts)
            if not node:
                return

            if node.val not in counts:
                counts.add(node.val)
            else:
                counts.remove(node.val)

            if not node.left and not node.right:
             #   print('none')
                if len(counts) <= 1:
                    self.ans += 1    
                
                
            dfs(node.left, counts)
            dfs(node.right, counts)
            
            if node.val not in counts:
                counts.add(node.val)
            else:
                counts.remove(node.val)

        dfs(root, set())
        return self.ans
