class Solution:
	def pseudoPalindromicPaths (self, root: TreeNode) -> int:
		res = 0
		count = [0]*10
		def check(count): return sum(1 for c in count if c % 2 == 1) < 2
		
def tour(root):
			nonlocal res
			count[root.val] += 1
			if not root.left and not root.right: 
				res += check(count)
			else:
				if root.left: tour(root.left)
				if root.right: tour(root.right)
			count[root.val] -= 1

		tour(root)
		return res
