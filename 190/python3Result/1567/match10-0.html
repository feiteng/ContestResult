class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        v='aeiou'
        ans=0
        t=0
        for i,j in enumerate(s):
            if j in v:
                ans+=1
            if i>=k and s[i-k] in v:
                ans-=1
            t=max(t,ans)
        return t
