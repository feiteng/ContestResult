class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        a = ['a','e','i','o','u']
        now = 0
        for c in s[:k]:
            if c in a:
                now+=1
        res = now
        for i in range(k,len(s)):
            if s[i-k] in a:
                now-=1
            if s[i] in a:
                now+=1
            res = max(now,res)
        return res
