class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        v = set(['a','e','i','o','u'])
        r = 0
        c = 0
        for i in range(len(s)):
            if s[i] in v:
                c += 1
            if i-k>=0 and s[i-k] in v:
                c -= 1
            r = max(c,r)
        return r
