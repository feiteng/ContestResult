class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        V = set(['a','e','i','o','u'])
        curr = 0
        for i in range(k):
            if s[i] in V:
                curr += 1
        ans = curr
        for i in range(k,len(s)):
            if s[i] in V:
                curr += 1
            if s[i-k] in V:
                curr -= 1
            ans = max(ans,curr)
        return ans
