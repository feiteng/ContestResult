class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vset = set(['a', 'e', 'i', 'o', 'u'])
        # first pass
        cur_num = 0 
        for i in range(k):
            if s[i] in vset:
                cur_num += 1
        max_num = cur_num
        # search end one
        for i in range(k, len(s)):
            if s[i] in vset:
                cur_num += 1
            if s[i - k] in vset:
                cur_num -= 1
            max_num = max(max_num, cur_num)
        return max_num
