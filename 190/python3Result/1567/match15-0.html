class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        lst = ['a','e','i','o','u']
        start = 0
        for i in s[:k]:
            if i in lst:
                start+=1
        res = start
        for i in range(k,len(s)):
            if s[i] in lst:
                start+=1
            if s[i-k] in lst:
                start-=1
            res = max(res,start)
        return res
