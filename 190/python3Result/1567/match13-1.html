class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowels = set(['a','e','i','o','u'])
        res = 0
        cur = 0
        for i in range(len(s)):
            if i>=k and s[i-k] in vowels:
                cur-=1
            if s[i] in vowels:
                cur+=1
            res = max(res,cur)
        return res
