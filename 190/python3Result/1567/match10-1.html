class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        max_c = 0
        cnt = 0
        voe = 'aeiou'
        for i, c in enumerate(s):
            if c in voe:
                cnt += 1
            if i >= k and s[i - k] in voe:
                cnt -= 1
            max_c = max(max_c, cnt)
        return max_c
