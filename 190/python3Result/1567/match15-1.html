class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        
        vs = {'a', 'e', 'i', 'o', 'u'}
        
        cur = 0
        for c in s[:k]:
            if c in vs:
                cur += 1
        res = cur
        
        for i in range(k, len(s)):
            if s[i] in vs:
                cur += 1
            if s[i-k] in vs:
                cur -= 1
            res = max(res, cur)
        return res
