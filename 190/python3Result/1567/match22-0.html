class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        cnt = 0
        # vowels in size k
        for i in range(k):
            if s[i] in "aeiou":
                cnt += 1
        res = cnt
        for i in range(k, len(s)):
            if s[i-k] in "aeiou":
                cnt -= 1
            if s[i] in "aeiou":
                cnt += 1
            res = max(res, cnt)
        
        return res
