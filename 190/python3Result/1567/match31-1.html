class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowels = {"a","e","i","o","u"}

        cnt = 0
        for i in range(min(k,len(s))):
            if s[i] in vowels: cnt += 1
        if k >= len(s):
            return cnt     
        
        res 
= cnt
        for i in range(len(s)-k):
            if s[i] in vowels: cnt -= 1
            if s[i+k] in vowels: cnt += 1
            #print(cnt)
            res = max(cnt,res)
        
        return res
