class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        ss = set("aeiou")
        ans, t = 0, 0
        for i in range(k):
            if s[i] in ss:
                t += 1
        ans = t
        for i in range(k, len(s)):
            if s[i] in ss:
                t += 1
            if s[i - k] in ss:
                t -= 1
            ans = max(ans, t)
        return ans
