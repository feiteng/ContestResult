class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        a = { "a", "e", "i", "o", "u" };
        q = 0
        for i in range(min(k, len(s))):
            if s[i] in a:
                q += 1
        m = q
        for i in range(k, len(s)):
            if s[i] in a:
                q += 1
            if s[i - k] in a:
                q -= 1
            m = max(m, q)
        return m
