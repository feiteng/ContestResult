class Solution(object):
    def isPrefixOfWord(self, sentence, searchWord):
        """
        :type sentence: str
        :type searchWord: str
        :rtype: int
        """
        sen = sentence.split()
        for i,j in enumerate(sen):
            if len(j) >= len(searchWord) and j[:len(searchWord)] == searchWord:
                return i+1
        return -1
