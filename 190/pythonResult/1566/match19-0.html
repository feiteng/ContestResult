class Solution(object):
    def isPrefixOfWord(self, sentence, searchWord):
        """
        :type sentence: str
        :type searchWord: str
        :rtype: int
        """
        words = sentence.split(' ')
        for i, word in enumerate(words):
            if len(word) >= len(searchWord) and word[:len(searchWord)] == searchWord:
                return i+1
        return -1
