class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        vowel = set(["a", "e", "i", "o", "u"])
        ans = 0
        for i in range(k):
            if s[i] in vowel:
                ans += 1
        if ans == k:
            return ans
        ret = ans
        for i in range(k, len(s)):
            if s[i] in vowel:
                ans += 1
            if s[i - k] in vowel:
                ans -= 1
            ret = max(ret, ans)
        return ret
