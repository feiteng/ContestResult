class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        dic = {"a":1,"e":1,"i":1,"o":1,"u":1}
        vowel = 0
        for i in range(k):
            if s[i] in dic: vowel += 1
        
        ans = vowel
        for i in range(k,len(s)):
            if s[i] in dic: vowel += 1
            if s[i-k] in dic: vowel -= 1
            ans = max(ans,vowel)
        return ans
