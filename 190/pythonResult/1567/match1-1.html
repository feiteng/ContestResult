class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        d = ('a','e','i','o','u')
        ref = 0
        for i in range(k):
            if s[i] in d:
                ref+=1
        ans = ref
        for i in range(k,len(s)):
            #print(i,' s[i] is ',s[i],' i-k ',s[i-k])
            if s[i] in d:
                ref+=1
            if s[i-k] in d:
                ref-=1
            ans = max(ans,ref)
        return ans
