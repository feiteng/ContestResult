class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        l
=len(s)
        t='aiueo'
        res=0
        for i in range(k):
            if s[i] in t:
                res+=1
        m=res
        for i in range(l-k):
            if s[i] in t:
                res-=1
            if s[i+k] in t:
                res+=1
            m=max(m,res)
        return m
