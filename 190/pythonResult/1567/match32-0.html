class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        start = 0
        end = 0
        vowels = set(['a', 'e', 'i', 'o', 'u'])
        count = 0
        for i in xrange(k):
            if s[i] in vowels:
                count+=1
        
        ans = count
        for i in xrange(k, len(s)):
            if s[i] in vowels:
                count+=1
            if s[i-k] in vowels:
                count-=1
            ans = max(ans, count)
        return ans
