class Solution(object):
    def maxVowels(self, s, k):
        vowel = set(['a', 'e', 'i', 'o', 'u'])
        res = 0
        for i in range(k):
            if s[i] in vowel:
                res += 1
        start, end = 0, k
        length = len(s)
        maxLen = res
        
while end < length:
            if s[start] in vowel:
                res -= 1
            if s[end] in vowel:
                res += 1
            maxLen = max(res, maxLen)
            end += 1
            start += 1
        return maxLen
            
            
        """
        :type s: str
        :type k: int
        :rtype: int
        """
