class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        n = len(s)
        vowels 
= {"a","e","i","o","u"}
        cnt = 0
        for i in range(k):
            if s[i] in vowels:
                cnt+=1
        res = cnt
        for i in range
(k, n):
            if s[i] in vowels:
                cnt+=1
            if s[i-k] in vowels:
                cnt-=1
            res = max(res, cnt)
        return res
