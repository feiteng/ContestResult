class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        vowel = set(['a','e','i','o','u'])
        t = 0
        for i in range(k):
            if s[i] in vowel:
                t += 1
        out = t
        for i in range(k,len(s)):
            if s[i] in vowel:
                t += 1
            if s[i-k] in vowel:
                t -= 1
            out = max(out,t)
        return out
