class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        vow = 'aeiou'
        running = 0
        for i in range(k):
            if s[i] in vow:
                running += 1
        ans = running
        for j in range(k,len(s)):
            if s[j] in vow:
                running += 1
            if s[j-k] in vow:
                running -= 1
            ans = max(ans,running)
        return ans
