class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        vls = set(['a', 'e', 'i', 'o', 'u'])
        
        count = 0
        for l in range(k):
            if s[l] in vls:
                count += 1
        max_count = count
        
        i, j, n = 0, k, len(s)
        while i < n - k + 1 and j < n:
            if s[j] in vls:
                count += 1
            if s[i] in vls:
                count -= 1
            max_count = max(max_count, count)
            
            i += 1
            j += 1
        
        return max_count
