class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        vowels = set(['a','e','i','o','u'])
        
        res = 0
        for i in range(k):
            if s[i] in vowels:
                res += 1
                
        _max 
= res
        
        for i in range(k, len(s)):
            if s[i-k] in vowels:
                res -= 1
            if s[i] in vowels:
                res += 1
            _max = max(res, _max)
        
        return _max
