class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        vowels = set(["a","e","i","o","u"])
        count = 0
        for i in xrange(k):
            if s[i] in vowels:
                count += 1
        n = len(s)
        res = count;
        for i in xrange(k, n):
            if s[i] in vowels:
                count += 1
            if s[i - k] in vowels:
                count -= 1
            res = max(res, count)
        return res;
