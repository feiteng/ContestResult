class Solution(object):
    def maxVowels(self, s, k):
        """
        :type s: str
        :type k: int
        :rtype: int
        """
        n = len(s)
        vowels = ['a','e','i','o','u']
        curr = 0
        for idx in range(k):
            if s[idx] in vowels: curr+=1
        out = curr
        for idx in range(n-k):
            if s[idx+k] in vowels: curr+=1
            if s[idx] in vowels: curr-=1
            out = max(out, curr)
        return out
