# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if root == None:
            return 0
        self
.result = 0
        arr = [0]*10
        self.search(root,arr)
        return self.result
        
        
    def search(self,root, arr):
        if root == None:
            return
        arr[root.val]+=1
        if root.left == None and root.right == None:
            # print arr
            count = 0
            valid = True
            for i in range(10):
                if arr[i]%2 == 1:
                    count+=1
                    if count>1:
                        valid = False
                        break
            if valid:
                self.result+=1
        else:
            self.search(root.left, arr)
            self.search(root.right, arr)
        arr[root.val]-=1
        return
