# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        f = [0]*9
        self.out = 0
        def isPP(freq):
            n=0
            for count in freq:
                if count%2 == 1: n+=1 
            # print freq, n
            return (n <= 1)
                
        def paths(node, freq):
            freq[node.val-1] += 1
            if node.left is None and node.right is None:
                self.out += isPP(freq)
                freq[node.val-1] -= 1
                return
            if node.left:
                paths(node.left, freq)
            if node.right:
                paths(node.right, freq)
            freq[node.val-1] -= 1
            return
        paths(root, f)
        return self.out
