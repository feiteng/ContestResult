# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        
        
        def isPalindromic(c):
            count = 0
            for _, val in c.items():
                if val%2==1:
                    count+=1
                if count > 1:
                    return False
            return True
        
        def dfs(root, c):
            if not root:
                return 0
            c[root.val]+=1
            ans = 0
            if not root.left and not root.right:
                if isPalindromic(c):
                    ans += 1
            else:
                if root.left:
                    ans += dfs(root.left, c)
                if root.right:
                    ans += dfs(root.right, c)
            c[root.val]-=1
            if c[root.val] == 0:
                del c[root.val]
            return ans
        
        return dfs(root, collections.Counter())
