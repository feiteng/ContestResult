# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        ct = [0] * 9
        self.ret = 0
        
        def isPali():
            single = False
            for i in ct:
                if i % 2:
                    if single:  return False
                    single = True
            return True
        
        def dfs(node):
            ct[node.val - 1] += 1
            if not (node.left or node.right) and isPali():
                self.ret += 1
            if node.left:
                dfs(node.left)
            if node.right:
                dfs(node.right)
            ct[node.val - 1] -= 1
        
        dfs(root)
        return self.ret
