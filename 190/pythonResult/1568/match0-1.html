# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
  def pseudoPalindromicPaths (self, root):
    digitCount = [0 for _ in range(10)]
    
    def isPossible(digitCount):
      odd = 0
      for num in digitCount:
        odd += num % 2
        
      return odd <= 1
    
    self.ans = 0
    def func(root):
      if not root:
        return
      
      digitCount
[root.val] += 1
      if not root.left and not root.right:
        self.ans += isPossible(digitCount)
        #print(digitCount)
        digitCount[root.val] -= 1
        return
      
      if root.left:
        func(root.left)
      
      if root.right:
        func(root.right)
      
      digitCount[root.val] -= 1
    
    func(root)
    return self.ans
