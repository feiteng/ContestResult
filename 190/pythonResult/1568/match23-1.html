# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.res = 0
        
        def helper(nums):
            cnt = 0
            for v in nums:
                if v % 2 == 1:
                    cnt += 1
            return cnt <= 1
        
        def dfs(cur, pre):
            if not cur:
                self.res += helper(pre)
            else:
                tmp = pre[:]
                tmp
[cur.val] += 1
                if cur.left:
                    dfs(cur.left, tmp)
                if cur.right:
                    dfs(cur.right, tmp)
                if cur.left is None and cur.right is None:
                    self.res += helper(tmp)
                    
        
        dfs(root, [0 for _ in range(10)])
        return self.res
