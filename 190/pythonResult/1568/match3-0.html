# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        cur = [0] * 9
        self.res = 0
        
        def dfs(node, path):
            if not node:
                return
            if not node.left and not node.right:
                path[node.val - 1] += 1
                odd = 0
                for i in range(9):
                    if path[i] % 2==1:
                        odd += 1
                if odd==0 or odd==1:
                    self.res += 1
                return
            path[node.val-1] += 1
            dfs(node.left, path[:])
            dfs(node.right, path[:])
        dfs(root, cur[:])
        return self.res
