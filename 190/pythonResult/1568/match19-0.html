# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        
        def checkPal(count):
            # print(count)
            odd = 0
            for k in count.keys():
                if count[k] % 2 == 1:
                    odd += 1
                if odd == 2:
                    return False
            return odd < 2
        
        def dfs(root, counter):
            if not root:
                return
            counter[root.val] += 1
            if not root.left and not root.right:
                # print(root.val)
                if checkPal(counter):
                    self.ans += 1
            
            if root.left:    
                dfs(root.left, counter)
            if root.right:
                dfs(root.right, counter)

        dfs(root, counter)
        return self.ans
