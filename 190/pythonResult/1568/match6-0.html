# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        def countPath(root,curSingle):
            if not root:
                return 0
            
            if root.val in curSingle:
                curSingle.discard(root.val)
            else:
                curSingle.add(root.val)
                
            if root.left == None and root.right == None:
                if len(curSingle) <= 1:
                    return 1
                else:
                    return 0
            else:
                return countPath(root.left,curSingle.copy())+countPath(root.right,curSingle.copy())
        
        s = set()
        return countPath(root,s)
