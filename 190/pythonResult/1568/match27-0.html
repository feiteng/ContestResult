class Solution(object):
    def pseudoPalindromicPaths (self, root):
        def dfs(node, count, length):
            if node:
                count[node.val] ^= 1
                length += 1
                #print("!", count, node.left is node.right is None)
                if length & 1 == sum(count) and (node.left is node.right is None):
                    self.ans += 1
                dfs(node.left, count, length)
                dfs(node.right, count, length)
                count[node.val ]^=1
                
        
        self.ans = 0
        count = [0] * 10
        dfs(root, count, 0)
        return self.ans
