# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        def isPalindrome(table):
            used = False
            for i in range(len(table)):
                if table[i] % 2 == 1:
                    if used:
                        return False
                    else:
                        used = 
True
            return True
        
        def dfs(root, table):
            if root is None:
                return
            table[root.val - 1] += 1
            if root.left is None and root.right is None:
                if isPalindrome(table):
                    self.ans += 1
            dfs(root.left, table)
            dfs(root.right, table)
            table[root.val - 1] -= 1
            
        table = [0] * 9
        dfs(root, table)
        return self.ans
