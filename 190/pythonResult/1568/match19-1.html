# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.res = 0
        
        def is_p(arr):
            odd = 0
            for num in arr:
                if num%2==1:
                    odd
+=1
                if odd>1:
                    return False
            return True
        
        def dfs(root, counts):
            if not root:
                return
            counts[root.val]+=1
            if not root.left and not root.right:
                self.res += is_p(counts)
            # print (root.val, counts)
            dfs(root.left, counts)
            dfs(root.right,counts)
            counts[root.val]-=1
            
        counts = [0 for i in range(10)]
        dfs(root, counts)
        
        return self.res
