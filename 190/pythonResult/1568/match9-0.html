# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.cnt = 0
        counts = [0] * 10
        self.count(root, counts)
        return self.cnt
    def count(self, cur_node, counts):
        if cur_node == None:
            return
        counts[cur_node.val] += 1
        if cur_node.left == None and cur_node.right == None:
            if self.is_palin(counts):
                self.cnt += 1
        else:
            if cur_node.left:
                self.count(cur_node.left, counts)
            if cur_node.right:
                self.count(cur_node.right, counts)
        counts[cur_node.val] -= 1
        
    def is_palin(self, counts):
        num_odd = 0
        for count in counts:
            num_odd += count % 2
        return num_odd <= 1
