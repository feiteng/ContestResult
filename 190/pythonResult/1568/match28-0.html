# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        def helper(cur, node):
            if not node: return
            cur[node.val] += 1
            if not node.left and not node.right:
                if sum(1 for v in cur.values() if v % 2 == 1) <= 1:
                    self.ans += 1
            else:
                helper(cur, node.left)
                helper(cur, node.right)
            cur[node.val] -= 1
        helper(collections.defaultdict(int), root)
        return self.ans
