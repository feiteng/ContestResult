# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        path_map = {} 
        result = [0]
        if root is None:
            return result[0]
        self.dfs(root, path_map, result)
        return result[0]
        
    def dfs(self, root, path_map, result):
        if root.val not in path_map:
            path_map[root.val] = 0
        path_map
[root.val] += 1
        if root.left is None and root.right is None:
            if self.is_palindromic(path_map):
                result[0] +=
 1
        if root.left is not None:
            self.dfs(root.left, path_map, result)
        if root.right is not None:
            self.dfs(root.right, path_map, result)
        path_map[root.val] -= 1
        
        
    def is_palindromic(self, path_map):
        odd_num = 0
        for num in path_map:
            if path_map[num] % 2 == 1:
                odd_num += 1
        if odd_num > 1:
            return False
        return True
