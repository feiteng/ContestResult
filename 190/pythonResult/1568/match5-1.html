# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        
        def isPseudoPalindromic(L):
            l_dict = {}
            for i in L:
                if i in l_dict:
                    l_dict[i] += 1
                else:
                    l_dict[i] = 1
            return sum([i%2 for i in l_dict.values()])<= 1
        path = []
        queue = [(root, [root.val])]
        while queue:
            current, val_list 
= queue.pop(0)
            if current.left:
                queue.append((current.left, val_list + [current.left.val ]))
            if current.right:
                queue.append((current.right, val_list + [current.right.val ]))
            if not current.left and not current.right:
                path.append(val_list)
        
        return sum(isPseudoPalindromic(i) for i in path)
