# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        C = [0] * 10
        self.ans = 0
        def dfs(root):
            C[root.val] += 1
            if root.left == None and root.right == None:
                check = 0
                for i in range(1, 10):
                    check += (C[i] % 2)
                if check <= 1:
                    self.ans += 1
            else:
                if root.left != None:
                    dfs(root.left)
                if root.right != None:
                    dfs(root.right)
            C[root.val] -= 1
        if root != None:
            dfs(root)
        return self.ans
