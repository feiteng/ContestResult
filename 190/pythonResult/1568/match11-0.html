# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.path = {}
        self.res = 0
        for i in range(1,10):
            self.path[i] = 0
        self.dfs(root)
        return self.res
        
    def isPalin(self):
        cnt = 0
        for i in range(1,10):
            if self.path[i] % 2 == 1:
                cnt += 1
        return cnt <= 1
        
    def dfs(self, node):
        self.path[node.val] += 1
        if node.left == None and node.right == None:
            if self.isPalin():
                self.res += 1
            self.path[node.val] -= 1
            return
        if node.left != None:
            self.dfs(node.left)
        
        if node.right != None:
            self.dfs(node.right)
        self.path[node.val] -= 1
