class Solution(object):
    def pseudoPalindromicPaths (self, root):
        hs = set([])
        count = [0]
        self.dfs(root, hs, count)
        return count[0]
    def dfs(self, node, hs, count):
        if node is None:
            return
        
        if node.val not in hs:
            hs.add(node.val)    
        else:
            hs.remove(node.val)
        if node.left is None and node.right is None :
            if len(hs) <= 1:        
                count[0] +=1
            
        self.dfs(node.left, set([e for e in hs]), count)
        
        self.dfs(node.right, set([e for e in hs]), count)
