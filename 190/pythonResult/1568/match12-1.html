# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.odds = 0
        self.mp = {}
        self.result = 0
        self.traverse(root)
        return self.result
        
    def traverse(self, root):
        self.mp[root.val] = self.mp.get(root.val, 0) + 1
        
if self.mp[root.val] % 2:
            self.odds += 1
        else:
            self.odds -= 1
        if not root.left and not root.right and self.odds <= 1:
            self.result += 1
        if root.left:
            self.traverse(root.left)
        if root.right:
            self.traverse(root.right)
        self.mp[root.val] -= 1
        if self.mp[root.val] % 2:
            self.odds += 1
        else:
            self.odds -= 1
