# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        count = [0]*10
        self.res = 0
        def dfs(root, count):
            if root is None:
                return 
            count[root.val] += 1
            if root.left is None and root.right is None:
                temp = 0
                for i in range(1,10):
                    if count[i]%2 == 1:
                        temp += 1
                if temp <= 1:
                    self.res += 1
            dfs(root.left, count)
            dfs(root.right, count)
            count[root.val] -= 1
        dfs(root, count)
        return self.res
