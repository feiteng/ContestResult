# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        status = {}
        self.odds = 0
        self.even = 0

        def dfs(cur,depth):
            if cur.val not in status:
                status[cur.val] = 0
            status[cur.val] +=1
            v = status[cur.val]
            if v == 1:
                self.odds += 1
            else:
                if v % 2 == 0:
                    self.odds -= 1
                    self.even += 1
                else:
                    self.odds += 1
                    self.even -= 1
            if cur.left is None and cur.right is None:
                if self.odds == 0 or (self.odds == 1 and depth % 2 == 1):
                    self
.ans += 1
                    #print (cur.val, self.odds, self.even, depth)
            else:
                if cur.left:
                    dfs(cur.left,depth+1)
                if cur.right:
                    dfs(cur.right,depth+1)
            status[cur.val] -= 1
            if v == 1:
                self.odds -= 1
            else:
                if v % 2 == 0:
                    self.odds += 1
                    self.even -= 1
                else:
                    self.odds -= 1
                    self.even += 1
        dfs(root,1)
        return self.ans
