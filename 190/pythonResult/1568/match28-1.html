# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.res = 0
        def dfs(cur, pre):
            if not cur:
                return
            pre[cur.val] += 1
            if not cur.left and not cur.right:
                odd = 0
                for k in pre:
                    if pre[k] % 2 == 1:
                        odd += 1
                if odd <= 1:
                    self.res += 1
                pre[cur.val] -= 1
                return
            
            dfs(cur.left, pre)
            dfs(cur.right, pre)
            pre[cur.val] -= 1
        dfs(root, collections.defaultdict(int))
        return self.res
