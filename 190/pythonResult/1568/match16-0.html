# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        def check_pali(arr):
            s = sum(arr)
            if s%2 == 0:
                for a in arr:
                    if a%2 == 1:
                        return False
                return True
            else:
                odd_num = 0
                for a in arr:
                    if a%2 == 1:
                        odd_num +=1
                        if odd_num >=2:
                            return False
                return True
        
        self.ans = 0
        def trav(node, arr):
            if not node:
                return
            if not node.left and not node.right:
                arr[node.val]+=1
                if check_pali(arr):
                    self.ans +=1
                arr[node.val] -=1
                return
            arr[node.val] +=1
            if node.left:
                trav(node.left, arr)
            if node.right:
                trav(node.right, arr)
            arr[node.val] -=1
        
        trav(root, [0]*10)
        return self.ans
