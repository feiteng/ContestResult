# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans=0
        def func(root,arr):
            if not root:
                return
            elif not root.left and not root.right:
                arr[root.val]+=1
                res=0
                for x in arr:
                    if x%2!=0:
                        res
+=1
                if res<=1:
                    self.ans+=1
                return
            arr[root.val]+=1
            func(root.left,arr[:])
            func(root.right,arr[:])
        func(root,[0]*10)
        return self.ans
