# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        def dfs(node, so_far):
            if not node:
                return
            
            if not node.left and not node.right:
                so_far[node.val] += 1
                odds = 0
                for i in xrange(1, 10):
                    if so_far[i] % 2:
                        odds += 1
                if odds == 1 or odds == 0:
                    self.ans += 1
                so_far[node.val] -= 1
                return
            so_far[node.val] += 1
            dfs(node.left, so_far)
            dfs(node.right, so_far)
            so_far[node.val] -= 1
        
        so_far = [0] * 10
        dfs(root, so_far)
        return self.ans
