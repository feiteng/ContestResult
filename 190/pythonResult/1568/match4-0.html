# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def check(self):
        odd = 0 
        for f in self.freq:
            if f%2 == 1:
                 odd += 1
        if odd > 1:
            return
        else:
            self.ans += 1
    def dfs(self, root):
        if root == None:
            return
        self.freq[root.val] += 1
        if root.left == None and root.right == None:
            self.check()
        
        self.dfs(root.left)
        self.dfs(root.right)
        
        self.freq[root.val] -= 1
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        self.freq = [0]*10
        
        self.dfs(root)
        return self.ans
