# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.m_Path = 0
        if not root:
            return 0
        self.DFS(root, set())
        return self.m_Path
        
    def DFS(self, root, helpSet):
        helpSet = set(helpSet)
        
if root.val in helpSet:
            helpSet.remove(root.val)
        else:
            helpSet.add(root.val)
        if not root.left and not root.right:
            if len(helpSet) <= 1:
                self.m_Path += 1
            return
        
        newTuple = tuple(helpSet)
        if root.left:
            self.DFS(root.left, newTuple)
        if root.right:
            self.DFS(root.right, newTuple)
