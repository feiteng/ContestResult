# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        dit = collections.defaultdict(int)
        tot 
= [0]
        def dfs(root):
            if not root:
                return 
            dit[root.val]+=1
            if not root.left and not root.right:
                # check if it has no more than 1 odd
                odd = 0
                for k in dit:
                    if dit[k]%2 == 1:
                        if odd == 0:
                            odd = 1
                        else:
                            dit[root.val]-=1
                            return
                tot[0]+=1
            else:
                dfs(root.left)
                dfs(root.right)
            
            dit[root.val]-=1
            
        dfs(root)
        return tot[0]
