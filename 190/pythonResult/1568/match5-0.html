# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        # 得到根结点到叶子节点的所有路径
        # 判断各个路径是否为回文路径
        
        if not root:
            return 0
        
        queue = [(root, [root.val])]
        
        paths = []
        
        while queue:
            tmp = []
            while queue:
                (node, path) = queue.pop(0)
                if node.left:
                    tmp.append((node.left, path+[node.left.val]))
                if node.right:
                    tmp.append((node.right, path+[node.right.val]))
                if not node.left and not node.right:
                    paths.append(path)
            queue = tmp
            
        # print paths
        return [self.check(path) for path in paths].count(True)
    
    def check(self, lyst):
        flag = True
        for _ in range(1, 10):
            if lyst.count(_) % 2 != 0:
                if flag:
                    flag = False
                else:
                    return False
        return True
