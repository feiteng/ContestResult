# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
import collections
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.ans = 0
        def isPalindrom(now):
            odds = 0
            for x in now:
                if now[x] % 2 == 1:
                    odds += 1
            return odds <= 1
        def dfs(root, counter):
            if root == None:
                return collections.defaultdict(int)
            if root.left == None and root.right == None:
                counter[root.val] += 1
                if isPalindrom(counter):
                    self
.ans += 1
                counter[root.val] -= 1
                return
            counter[root.val] += 1
            dfs(root.left, counter)
            dfs(root.right, counter)
            counter[root.val] -= 1
            return
        dfs(root, collections.defaultdict(int))
        return self.ans
