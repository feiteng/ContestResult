# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def pseudoPalindromicPaths (self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        nums = [False]*10
        
        def isPseudoPalindromic(nums):
            count = 0;
            for num in nums:
                if num:
                    count += 1
                if count > 1:
                    return False;
            return True
                
            
        def countPseudoPalindromic(node, nums):
            res = 0
            nums[node.val] = not nums[node.val]
            
if not node.left and not node.right:
                if isPseudoPalindromic(nums):
                    res = 1
            else:
                if node.left:
                    res += countPseudoPalindromic(node.left, nums)
                if node.right:
                    res += countPseudoPalindromic(node.right, nums)
                
            nums[node.val] = not nums[node.val]
            return res
        
        return countPseudoPalindromic(root, nums)
