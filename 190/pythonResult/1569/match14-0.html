class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        d = [[-1000000 for i in range(len(nums2) + 1)] for j in range(len(nums1) + 1)]
        for i in range(len(nums1)):
            for j in range(len(nums2)):
                d[i + 1][j + 1] = max(
                    d[i + 1][j], 
                    d[i][j + 1], 
                    d[i][j] + nums1[i] * nums2[j], 
                    nums1[i] * nums2[j]
                )
        return d[len(nums1)][len(nums2)]
