class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        # because the max dot product is additive,
        # let dp[i][j] be the max dot ending at i and j
        # dp[i][j] = max( dp[i-1][j],    # skip i-1
        #                 dp[i][j-1],    # skip j-1
        #                 dp[i-1][j-1] + A[i-1]*B[j-1]) # use (i-1)*(j-1)
        
        # build dp map
        # 0th index represents empty string i.e. A[:0]
        
        # Note that the subsequence cannot be non-empty so you cannot prime as 0.
        
        # first check i
        dp = [ [-float('inf')] * (len(nums2)+1) for _ in range( len(nums1) + 1 ) ]
        
        for i in range(1, len(dp) ):
            for j in range(1, len(dp[0]) ):
                dp
[i][j] = max( dp[i-1][j], dp[i][j-1], dp[i-1][j-1] + nums1[i-1]*nums2[j-1],
                                nums1[i-1]*nums2[j-1]
                              ) 
        
        return dp[-1][-1]
