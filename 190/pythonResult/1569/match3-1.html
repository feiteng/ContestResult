class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        s, t = len(nums1), len(nums2)
        
        dp = [[0 for i in range(t)] for j in range
(s)]
        
        dp[0][0] = nums1[0]*nums2[0]
        
        for i in range(1,s):
            dp[i][0] = max(dp[i-1][0], nums2[0]*nums1[i])
        
        for i in range(1,t):
            dp[0][i] = max(dp[0][i-1], nums1[0]*nums2[i])
        
        for i in range(1,s):
            for j in range(1,t):
                dp[i][j] = max(dp[i-1][j-1]+nums1[i]*nums2[j],dp[i][j-1],dp[i-1][j],nums1[i]*nums2[j])
        
        return dp[-1][-1]
