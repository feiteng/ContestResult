class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        m = len(nums1)
        n = len(nums2)
        dp = [[0 for j in range(n)] for i in range(m)]
        dp[0][0] = max(nums1[0]*nums2[0],0)
        
        
        for i in range(1,m):
            dp[i][0] = max(dp[i-1][0], nums1[i]*nums2[0])
        for j in range(1,n):
            dp[0][j] = max(dp[0][j-1], nums1[0]*nums2[j])
         
        
for i in range(1,m):
            for j in range(1,n):
                dp[i][j] = max(dp[i-1][j-1],dp[i][j-1],dp[i-1][j],dp[i-1][j-1]+nums1[i]*nums2[j])
                
        if dp[-1][-1]<=0:
            if sum(nums1)<0:  t1 = max(nums1)
            else: t1 = min(nums1)
            if sum(nums2)<0:  t2 = max(nums2)
            else: t2 = min(nums2)
            return t1*t2
                
        return dp[-1][-1]
