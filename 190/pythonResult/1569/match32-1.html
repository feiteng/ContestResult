class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        dp = [[0 for i in range(len(nums2))] for j in range(len(nums1))]
        
        
        for i in range(len(nums2)):
            if i == 0:
                dp[0][i] = nums1[0] * nums2[0]
            else:
                dp[0][i] = max(nums1[0] * nums2[i], dp[0][i-1])
        
        for i in range(len(nums1)):
            if i == 0:
                dp[i][0] = nums1[0] * nums2[0]
            else:
                dp[i][0] = max(nums2[0] * nums1[i], dp[i-1][0])
        
        
        
for i in range(1, len(nums1)):
            for j in range(1, len(nums2)):
                dp[i][j] = max(dp[i-1][j-1] + nums1[i]*nums2[j], dp[i-1][j], dp[i][j-1], nums1[i]*nums2[j])
        
        return dp[-1][-1]
