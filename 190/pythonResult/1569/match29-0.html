class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        m = len(nums1)
        n = len(nums2)
        dp = [[float('-inf') for i in range(n + 1)] for j in range(m + 1)]
        for i in range(1, m + 1):
            if i == 1:
                dp[i][1] = nums1[0] * nums2[0]
            else:
                dp[i][1] = max(dp[i - 1][1], nums1[i - 1] * nums2[0])
        for i in range(1, n + 1):
            if i == 1:
                dp[1][i] = nums1[0] * nums2[0]
            else:
                dp[1][i] = max(dp[1][i - 1], nums1[0] * nums2[i - 1])
        for i in range(2, m + 1):
            for j in range(2, n + 1):
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1] + nums1[i - 1] * nums2[j - 1], dp[i - 1][j - 1], nums1[i - 1] * nums2[j - 1])
        #print dp
        return dp[m][n]
