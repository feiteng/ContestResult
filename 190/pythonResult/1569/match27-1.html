class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        n1 = len(nums1)
        n2 = len(nums2)
        dp = [[float('-inf')]*n2 for _ in range(n1)]
        res = float('-inf') 
        for i in range(n1):
            for j in range
(n2):
                if i==0 and j==0:
                    dp[i][j] = nums1[i]*nums2[j]
                elif i==0:
                    dp[i][j] = max(dp[i][j-1], nums1[i]*nums2[j])
                elif j ==0:
                    dp[i][j] = max(dp[i-1][j], nums1[i]*nums2[j])
                else:
                    dp[i][j] = max(dp[i][j], nums1[i]*nums2[j], dp[i-1][j-1] + nums1[i]*nums2[j], dp[i-1][j], dp[i][j-1])
                res = max(res, dp[i][j])
        return res
