class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        inf = -(1<<31)
        N1, N2 = len(nums1), len(nums2)
        N = min(N1, N2)
        dp = [[inf]*N2 for _ in range(N1)]
        dp[0][0] = nums1[0]*nums2[0]
        for i in range(1, N2):
            dp[0][i] = max(nums1[0]*nums2[i], dp[0][i-1])
        for i in range(1, N1):
            dp[i][0] = max(nums1[i]*nums2[0], dp[i-1][0])
        for i in range(1, N1):
            for j in range(1, N2):
                dp[i][j] = max(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]+nums1[i]*nums2[j], nums1[i]*nums2[j])
        return dp[N1-1][N2-1]
