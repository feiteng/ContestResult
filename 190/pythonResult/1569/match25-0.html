class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        import sys
        m = len(nums1)
        n = len(nums2)
        dp = [[-sys.maxint for _ in range(n)] for _ in range(m)]
        dp[0][0] = nums1[0]*nums2[0]
        for j in range(1, n):
            dp[0][j] = max(nums1[0]*nums2[j], dp[0][j-1])
        for i in range(1, m):
            dp[i][0] = max(nums1[i]*nums2[0], dp[i-1][0])
        
        for i in range(1, m):
            for j in range(1, n):
                dp[i][j] = max(dp[i-1][j-1], dp[i-1][j-1] + nums1[i]*nums2[j], nums1[i]*nums2[j], dp[i][j-1], dp[i-1][j])
        return dp[m-1][n-1]
