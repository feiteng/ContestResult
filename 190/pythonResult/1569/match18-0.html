class Solution(object):
    def maxDotProduct(self, nums1, nums2):
        """
        :type nums1: List[int]
        :type nums2: List[int]
        :rtype: int
        """
        m,n=len(nums1),len(nums2)
        s=[[0]*n for _ in range(m)]
        s[0][0]=nums1[0]*nums2[0]
        for j in range(1,n): s[0][j]=max(s[0][j-1],nums1[0]*nums2[j])
        for i in range(1,m):
            s[i][0]=max(s[i-1][0],nums1[i]*nums2[0])
            for j in range(1,n):
                s[i][j]=max(s[i-1][j],s[i][j-1],nums1[i]*nums2[j]+max(s[i-1][j-1],0))
        return s[-1][-1]
