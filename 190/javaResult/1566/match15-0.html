class Solution {
    public int isPrefixOfWord(String sentence, String searchWord) {
        String[] words = sentence.split(" ");
        int word_index=1;
        for(String word : words) {
            if(word.startsWith(searchWord)) {
                return word_index;
            }
            word_index++;
        }
        return -1;
    }
}
