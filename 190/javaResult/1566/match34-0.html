class Solution {
   public int isPrefixOfWord(String sentence, String searchWord) {
        String[] tokens = sentence.split(" ");
        int ans = -1;
        for(int i = 0; i < tokens.length; i++) {
            if(tokens[i].startsWith(searchWord)) {
                ans = i + 1;
                return ans;
            }
        }
        return ans;
    }

}
