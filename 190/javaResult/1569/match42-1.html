class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int M = nums1.length;
        int N = nums2.length;
        
        int [][]DP = new int[M][N];
        
        DP[0][0] = nums1[0] * nums2[0];
        for(int i=1;i<M;i++) {
            DP[i][0] = Math.max(DP[i-1][0], nums1[i]*nums2[0]);
        }
        
        for(int i=1;i<N;i++) {
            DP[0][i] = Math.max(DP[0][i-1], nums1[0]*nums2[i]);
        }
        
        for (int i=1;i<M;i++) {
            for (int j=1;j<N;j++) {
                DP[i][j] = Math.max(DP[i-1][j], DP[i][j-1]);
                DP[i][j] = Math.max(DP[i][j], 
Math.max(nums1[i]*nums2[j], nums1[i]*nums2[j] + DP[i-1][j-1]));
            }
        }
        
        return DP[M-1][N-1];
    }
}
