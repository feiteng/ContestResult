class Solution {
    public int maxDotProduct(int[] n1, int[] n2) {
        int len1 = n1.length;
        int len2 = n2.length;
        int tt = Integer.MIN_VALUE;
        int[][] arr = new int[len1 + 1][len2 + 1];
        // for(int i = 0; i < len1; i++){
        //     Arrays.fill(arr[i], Integer.MIN_VALUE);
        // }

        // arr[len1 - 1][len2 - 1] = n1[len1 - 1] * n2[len2 - 2];
        for(int i = len1 - 1; i >= 0; i--){
            for(int j = len2 - 1; j >= 0 ; j--){
                // if(i == len1 - 1 && j == len2 - 1) continue;
                int temp = Integer.MIN_VALUE;
                temp = Math.max(temp, arr[i + 1][j]);
                for(int k = j; k < len2; k++){
                    temp = Math.max(temp, n1[i] * n2[k] + arr[i + 1][k + 1]);
                    tt = Math.max(tt, n1[i] * n2[k]);
                }
                arr[i][j] = temp;
            }
        }
        
//         for(int i = 0; i <= len1; i++){
//             for(int j = 0; j <= len2; j++){
                
//                 if(arr[i][j] != Integer.MIN_VALUE) System.out.print(arr[i][j] + " ");
//                 else System.out.print("null ");
//             }
//             System.out.println();
//         }
        if(tt < 0) return tt;
        return arr[0][0];
    }

}
