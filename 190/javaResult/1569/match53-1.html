class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        //dp[i][j] = dp[i-1][j-1]+nums1[i]*nums2[j], dp[i-1][j], dp[i][j-1], dp[i-1][j-1]
        int m = nums1.length, n = nums2.length;
        int[][] dp = new int[m+1][n+1];
        
        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = Integer.MIN_VALUE;
                
if (i == 1 && j == 1) {
                    dp[i][j] = nums1[i-1]*nums2[j-1];
                } else if (i == 1) {
                    dp[i][j] = Math.max(nums1[i-1]*nums2[j-1], dp[i][j-1]);
                } else if (j == 1) {
                    dp[i][j] = Math.max(dp[i-1][j], nums1[i-1]*nums2[j-1]);
                } else {
                    dp[i][j] = Math.max(dp[i-1][j-1], dp[i-1][j-1]+nums1[i-1]*nums2[j-1]);
                    dp[i][j] = Math.max(dp[i][j], nums1[i-1]*nums2[j-1]);
                    dp[i][j] = Math.max(dp[i][j], Math.max(dp[i][j-1], dp[i-1][j]));
                
}
            }
        }
        return dp[m][n];
    }
}
