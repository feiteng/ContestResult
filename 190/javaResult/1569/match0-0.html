class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int r = nums1.length;
        int c = nums2.length;
        int dp[][] = new int[r][c];
        
        for(int i=0;i<r;i++){
            for(int j=0;j<c;j++){
                
                if(i==0 && j==0){
                    dp[i][j] = nums1[i]*nums2[j];
                }
                
                else if(i==0){
                    dp[i][j] = Math.max(nums1[i]*nums2[j],dp[i][j-1]);
                }
                
                else if(j==0){
                    dp[i][j] = Math.max(nums1[i]*nums2[j],dp[i-1][j]);
                }
                else{
                    int inc = Math.max(nums1[i] * nums2[j] + dp[i-1][j-1],nums1[i] * nums2[j]);
                    int exc = Math.max(dp[i-1][j],dp[i][j-1]);
                    dp[i][j] = Math.max(inc,exc);
                }
                
                //System.out.print(dp[i][j] + " ");
            }
            //System.out.println();
            
        }
        
        return dp[r-1][c-1];
        
    }
}
