class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        //DP
        int r = nums1.length;
        int c = nums2.length;
        int[][] m = new int[r][c];
        m[0][0] = nums1[0] * nums2[0];
        for (int i = 1; i < r; i++) {
            m[i][0] = Math.max(nums1[i] * nums2[0], m[i - 1][0]);
        }
        for (int j = 1; j < c; j++) {
            m[0][j] = Math.max(nums1[0] * nums2[j], m[0][j - 1]);
        }
        for (int i = 1; i < r; i++) {
            for (int j = 1; j < c; j++) {
                m[i][j] = Math.max(Math.max(m[i - 1][j - 1] + nums1[i] * nums2[j], nums1[i] * nums2[j]), Math.max(m[i - 1][j], m[i][j - 1]));
            }
        }
        return m[r - 1][c - 1];
    }
}
