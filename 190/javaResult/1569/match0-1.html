class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int m = nums1.length;
    int n = nums2.length;
    int res[][] = new int[m][n];
    for(int i=0;i<m;i++){
      for(int j=0;j<n;j++){
        if(i==0 && j==0){
          res[i][j] = nums1[i]*nums2[j];
        }

        else if(i==0){
          res[i][j] = Math.max(nums1[i]*nums2[j],res[i][j-1]);
        }

        else if(j==0){
          res[i][j] = Math.max(nums1[i]*nums2[j],res[i-1][j]);
        }
        else{
          int inc = Math.max(nums1[i] * nums2[j] + res[i-1][j-1],nums1[i] * nums2[j]);
          int exc = Math.max(res[i-1][j],res[i][j-1]);
          res[i][j] = Math.max(inc,exc);
        }
      }
    }
    return res[m-1][n-1];
    }
}
