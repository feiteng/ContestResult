class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int n = nums1.length, m = nums2.length;
        int dp[][] = new int[n][m];
        dp[0][0] = nums1[0] * nums2[0];
        
        for(int i = 1; i < n; ++i) {
            dp[i][0] = Math.max(dp[i - 1][0], nums1[i] * nums2[0]);
        }
        
        for(int j = 1; j < m; ++j) {
            dp[0][j] = Math.max(dp[0][j - 1], nums2[j] * nums1[0]);
        }
        
        for (int i = 1; i < n; i++) {
            for (int j = 1; j < m; j++) {
                dp[i][j] = Math.max((Math.max(0, dp[i - 1][j - 1]) + (nums2[j] * nums1[i])), Math.max(dp[i][j - 1], dp[i - 1][j]));
            }
        }

        return dp[n - 1][m - 1];
    }
}
