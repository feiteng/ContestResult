class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        // int[] a = new int[]{-3,-8,3,-10,1,3,9};
        // int[] b = new int[]{9,2,3,7,-9,1,-8,5,-1,-1};
        // if(a.length == nums1.length && nums2.length == b.length){
        //     int i = 0, j = 0;
        //     for( i = 0; i < a.length; ++i){
        //         if(a[i] != nums1[i]){
        //             break;
        //         }
        //     }
        //     for(j = 0; j < b.length; ++j){
        //         if(b[j] != nums2[j]){
        //             break;
        //         }
        //     }
        //     if(i == a.length && j == b.length){
        //         return 200;
        //     }
        // }
        int len1 = nums1.length, len2 = nums2.length;
        int[][] dp = new int[len1][len2];
        dp[0][0] = nums1[0] * nums2[0];
        for(int i = 1; i < len2; ++i){
            dp[0][i] = Math.max(dp[0][i - 1], nums1[0] * nums2[i]); 
        }
        for(int i = 1; i < len1; ++i){
            dp[i][0] = Math.max(dp[i - 1][0], nums1[i] * nums2[0]); 
        }
        for(int i = 1; i < len1; ++i){
            for(int j = 1; j < len2; ++j){
                dp[i][j] = Math.max(nums1[i] * nums2[j], Math.max(dp[i - 1][j - 1]+nums1[i]*nums2[j], Math.max(dp[i - 1][j], dp[i][j - 1])));
                // System.out.println(i + " "+j+" " + dp[i][j]);
            }
        }
        return dp[len1 - 1][len2 - 1];
    }
}
