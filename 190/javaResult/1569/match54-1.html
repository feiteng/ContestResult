class Solution {
    public int maxDotProduct(int[] nums1, int[] nums2) {
        int len1 = nums1.length, len2 = nums2.length;
        int[][] dp = new int[len2][len1]; dp[0][0] = nums2[0] * nums1[0];
        for (int i = 1; i < len2; i++) { dp[i][0] = Math.max(dp[i-1][0], nums2[i] * nums1[0]); }
        for (int j = 1; j < len1; j++) { dp[0][j] = Math.max(dp[0][j-1], nums2[0] * nums1[j]); }
        for (int i = 1; i <len2; i++) {
            for (int j = 1; j < len1; j++) {
                int a = nums2[i] * nums1[j];
                a = Math.max(a, a + dp[i-1][j-1]);
                int b = 
Math.max(dp[i-1][j], dp[i][j-1]);
                dp[i][j] = Math.max(a, b);
            }
        }
        /* for (int i = 0; i < len2; i++) {
            for (int j = 0; j < len1; j++) {
                System.out.print(dp[i][j] + ", ");
            }
            System.out.println();
        } */
        return dp[len2-1][len1-1];
    }
}
