class Solution {
    private boolean isVowel(char ch) {
        return ch =='a' || ch =='e' || ch =='i' || ch =='o' || ch =='u';
    }
    public int maxVowels(String s, int k) {
        int sum = 0;
        for(int i=0;i<k;i++) {
            if(isVowel(s.charAt(i))) {
                sum++;
            }
        }
        
        int max = sum;
        for(int i=k;i<s.length();i++) {
            if(isVowel(s.charAt(i-k))) {
                sum--;
            }
            if(isVowel(s.charAt(i))) {
                sum++;
            }
            max= Math.max(max, sum);
        }
        return max;
    }
}
