class Solution {
    public int maxVowels(String s, int k) {
        HashSet<Character> vow = new HashSet<>();
        vow.add('a'); vow.add('e'); vow.add('i'); vow.add('o'); vow.add('u');
        int vowels = 0;
        int best = 0;
        for (int i = 0; i < k; i++) {
            if (vow.contains(s.charAt(i))) vowels++;
        }
        best = vowels;
        
for (int i = k; i < s.length(); i++) {
            if (vow.contains(s.charAt(i))) vowels++;
            if (vow.contains(s.charAt(i-k))) vowels--;
            best = Math.max(vowels, best);
        }
        return best;
    }
}
