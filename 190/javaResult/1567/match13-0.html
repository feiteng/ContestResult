class Solution {
    public int maxVowels(String s, int k) {
        int res = 0;
        
        int count = 0;
        int end = 0;
        String v = "aeiou";
        for (int i = 0; i < s.length(); i++) {
            if (v.indexOf(s.charAt(i)) != -1) {
                count++;
            }
            if (i + 1 - end > k) {
                if (v.indexOf(s.charAt(end++)) != -1) {
                    count--;
                }
            }
            res = Math.max(res, count);
        }
        
        return res;
    }
}
