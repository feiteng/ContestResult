class Solution {
    public int maxVowels(String s, int k) {
        int ans = 0;
        String vowels = "aeiou";
        for (
int i = 0, cnt = 0; i < s.length(); ++i) {
            if (vowels.contains(s.substring(i, i + 1))) {
                ++cnt; 
            }
            if (i >= k && vowels.contains(s.substring(i - k, i - k + 1))) {
                --cnt;
            }
            ans = Math.max(cnt, ans);
        }
        return ans;        
    }
}
