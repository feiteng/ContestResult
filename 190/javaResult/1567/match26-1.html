class Solution {
    public int maxVowels(String s, int k) {
        HashSet<Character> v = new HashSet();
        v.add('a');
        v.add('e');
        v.add('i');
        v.add('o');
        v.add('u');
        int res = 0;
        int cur = 0;
        for(int i=0; i<k; i++){
            if(v.contains(s.charAt(i))){
                cur++;
            }
        }
        res = cur;
        for(int hi=k; hi<s.length(); hi++){
            int lo = hi-k;
            
if(v.contains(s.charAt(hi))){
                cur++;
            }
            if(v.contains(s.charAt(lo))){
                cur--;
            }
            res = Math.max(res, cur);
        }
        return res;
    }
}
