class Solution {
    public int maxVowels(String s, int k) {
        int b = 0;
        int cnt = 0;
        int ans = 0;
        String vowel = "aeiou";
        for (int i = 0; i < s.length(); ++i) {
            if (vowel.indexOf(s.charAt(i)) != -1) {
                cnt += 1;
            }
            if (i - b + 1 > k) {
                if (vowel.indexOf(s.charAt(b)) != -1) {
                    cnt -= 1;
                }
                b += 1;
            }
            ans = Math.max(ans, cnt);
        }
        
        return ans;
    }
}
