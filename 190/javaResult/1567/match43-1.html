class Solution {
    public int maxVowels(String s, int k) {
    	
    	int ans=0,num=0;
    	for(int i=0;i<k;i++) {
    		if(check(s.charAt(i)))
    			num++;
    	}
    	ans=num;
    	for
(int i=k;i<s.length();i++) {
    		if(check(s.charAt(i)))
    			num++;
    		if(check(s.charAt(i-k)))
    			num--;
    		ans=Math.max(ans, num);
    	}
    	return ans;
    }
    
    private boolean check(char c) {
    	if(c=='a' || c=='e' || c=='i' || c=='o' || c=='u')
    		return true;
    	return false;
    }
}
