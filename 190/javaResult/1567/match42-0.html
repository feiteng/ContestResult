class Solution {
public int maxVowels(String s, int k) {
        int res = 0, cur = 0;
        int i = 0;
        for(; i < k; i++) {
            if("aeiou".indexOf(s.charAt(i)) != -1) {
                cur++;
            }
        }
        res = cur;
        for(; i < s.length(); i++) {
            if("aeiou".indexOf(s.charAt(i)) != -1) {
                cur++;
            }
            if("aeiou".indexOf(s.charAt(i-k)) != -1) {
                cur--;
            }
            res = Math.max(cur, res);
        }
        return res;
    }
}
