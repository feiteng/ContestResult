class Solution {
    public int maxVowels(String s, int k) {
        int ans = 0;
        int count = 0;
        for(int i = 0 ; i < k ; i++){
            if(isVowel(s.charAt(i))){
                count++;
            }
        }
        ans = Math.max(ans , count);
        for(int i = k ; i < s.length() ;i++){
            if(isVowel(s.charAt(i))){
                count++;
            }
            if(isVowel(s.charAt(i-k))){
                count--;
            }
            ans = Math.max(ans , count);
        }
        return ans;
    }
    boolean isVowel(char c){
        if(c == 'a' || c == 'e' || c== 'i' || c == 'o' || c == 'u'){
            return true;
        }
        return false;
    }
}
