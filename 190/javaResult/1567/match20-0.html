class Solution {
    public int maxVowels(String s, int k) {
        char[] ch = s.toCharArray();
        int ans = 0, res = 0;
        for (int i = 0; i < ch.length; ++i) {
            if (i < k) {
                if (judge(ch[i]))
                    ans++;
                res = Math.max(res, ans);
                continue;
            } 
            if(judge(ch[i]))
                ans++;
            if(judge(ch[i - k]))
                ans--;
            res = Math.max(res, ans);
        }
        return res;
    }
    
    private boolean judge(char c) {
        if (c == 'a' || c =='e' || c == 'i' || c == 'o' || c == 'u')
            return true;
        return false;
    } 
}
