class Solution {
    public int maxVowels(String s, int k) {
        char[] cs = s.toCharArray();
        int cnt = 0; 
        for (int i = 0; i < k; i++) {
            if (cs[i] == 'a' || cs[i] == 'e' || cs[i] == 'i' || cs[i] == 'o' || cs[i] == 'u') {
                cnt++;
            }
        }
        int ret = cnt; 
        for (int i = k; i < cs.length; i++) {
            if (cs[i-k] == 'a' || cs[i-k] == 'e' || cs[i-k] == 'i' || cs[i-k] == 'o' || cs[i-k] == 'u') {
                cnt--;
            }
            if (cs[i] == 'a' || cs[i] == 'e' || cs[i] == 'i' || cs[i] == 'o' || cs[i] == 'u') {
                cnt++;
            }
            ret =  Math.max(ret, cnt);
            
        }
        return ret;
    }
}
