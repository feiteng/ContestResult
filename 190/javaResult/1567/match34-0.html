class Solution {
    public int maxVowels(String s, int k) {
        int count = 0;
        int res = 0;
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        for(int i = 0; i < k; i++){
            if(set.contains(s.charAt(i))){
                count ++;
            }
        }
        res = count;
        
        for(int i = k; i < s.length(); i++){
            if(set.contains(s.charAt(i-k))){
                count--;
            }
            if(set.contains(s.charAt(i))){
                count++;
            }
            res = Math.max(count, res);
        }
        return res;
    }
}
