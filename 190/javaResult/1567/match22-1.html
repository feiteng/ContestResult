class Solution {
    public int maxVowels(String s, int k) {
        int i = 0;
        int j = 0;
        int max = 0;
        int count = 0;
        while (j < s.length()) {
            while (j < s.length() && j - i + 1 <= k) {
                if (s.charAt(j) == 'a' || 
                    s.charAt(j) == 'e' || 
                    s.charAt(j) == 'i' || 
                    s.charAt(j) == 'o' || 
                    s.charAt(j) == 'u') {
                    count++;
                }
                j++;
            }
            max = Math.max(max, count);
            if (i < j && (s.charAt(i) == 'a' || 
                    s.charAt(i) == 'e' || 
                    s.charAt(i) == 'i' || 
                    s.charAt(i) == 'o' || 
                    s.charAt(i) == 'u')) {
                count--;
            }
            i++;
        }
        return max;
    }
}
