class Solution {
    public int maxVowels(String s, int k) {
        int c = 0;
        int max = 0;
        for(int i=0;i<s.length();i++) {
            char ch = s.charAt(i);
            if (ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u') {
                c++;
            }
            int old = i-k;
            if (old>=0) {
                ch = s.charAt(i-k);
                if (ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u') {
                    c--;
                }                
            }
            max = Math.max(c, max);
        }
        return max;
    }
}
