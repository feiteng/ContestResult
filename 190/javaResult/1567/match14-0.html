class Solution {
    public int maxVowels(String s, int k) {
        HashSet<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        int all = 0;
        for (int i = 0; i < k; i++) {
            if (set.contains(s.charAt(i))) {
                all++;
            }
        }
        int res = all;
        for (int i = 0; i + k < s.length(); i++) {
            if (set.contains(s.charAt(i))) {
                all--;
            }
            if (set.contains(s.charAt(i + k))) {
                all++;
            }
            res = Math.max(res, all);
        }
        return res;
    }
}
