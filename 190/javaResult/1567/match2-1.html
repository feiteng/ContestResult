class Solution {
    
    public int maxVowels(String s, int k) {
        int l = 0;
        Set<Character> V = new HashSet<>();
        V.add('a');
        V.add('e');
        V.add('i');
        V.add('o');
        V.add('u');
        
        for (int i = 0; i < k; i++) {
            if (V.contains(s.charAt(i))) {
                l++;
            }
        }
        
        int max = l;
        
        for (int i = k; i < s.length(); i++) {
            if (V.contains(s.charAt(i))) {
                l++;
            }
            if (V.contains(s.charAt(i-k))) {
                l--;
            }
            
            max = Math.max(max, l);
        }
        
        return max;
    }
}
