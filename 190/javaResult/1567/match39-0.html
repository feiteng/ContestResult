class Solution {
    public int maxVowels(String s, int k) {
        HashSet<Character> vow=new HashSet<>();
        vow.add('a');vow.add('e');vow.add('i');vow.add('o');vow.add('u');
        int pq=0,max=0;
        for(int i=0;i<k;i++){
            if(vow.contains(s.charAt(i)))
                pq++;
        }
        max=Math.max(max,pq);
        for(int i=k;i<s.length();i++){
            if(vow.contains(s.charAt(i-k)))
                pq--;
            if(vow.contains(s.charAt(i)))
                pq++;
            max=Math.max(max,pq);
        }
        return max;
    }
}
