class Solution {
    public int maxVowels(String s, int k) {
        int count = 0, max = 0;
        Set<Character> vo = new HashSet<>(Arrays.asList('a', 'e', 'i', 'o', 'u'));
        for (int i = 0; i < s.length(); ++i) {
            if (vo.contains(s.charAt(i))) {
                ++count;
            }
            if (i >= k && vo.contains(s.charAt(i - k))) {
                --count;
            }
            
            max = Math.max(max, count);
        }
        
        return max;
    }
}
