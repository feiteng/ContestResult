class Solution {
    public int maxVowels(String s, int k) {
        HashSet<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        int count = 0;
        char[] arr = s.toCharArray();
        int n = arr.length;
        
        for (int i = 0; i < k; i++) {
            if (set.contains(arr[i])) {
                count++;
            }
        }
        
        int res = count;
        for (int i = k; i < n; i++) {
            if (set.contains(arr[i])) {
                count++;
            }
            if (set.contains(arr[i - k])) {
                count--;
            }
            res = Math.max(res, count);
        }
        return res;
    }
}
