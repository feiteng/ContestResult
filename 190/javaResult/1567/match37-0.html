class Solution {
    public int maxVowels(String s, int k) {
        String ref = "aeiou";
        int num =0;
    
        for(int i=0;i<k;i++){
            if(ref.indexOf(s.charAt(i))>=0){
                num++;
            }
        }
        int max = num;
        
        for(int i=1;i<=s.length()-k;i++){
            if(ref.indexOf(s.charAt(i-1))>=0){
                num--;
            }
            if(ref.indexOf(s.charAt(i+k-1))>=0){
                num++;
            }
            max= Math.max(max,num);
        }
        
        return max;
    }
}
