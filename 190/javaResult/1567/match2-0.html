class Solution {

    public int maxVowels(String s, int k) {
        int res = 0;
        Set<Character> vow = new HashSet<>();
        vow.add('a');
        vow.add('e');
        vow.add('i');
        vow.add('o');
        vow.add('u');
        for (int i = 0; i < k; i++) {
            if (vow.contains(s.charAt(i))){
                res++;
            }
        }
        int ans = res;
        for (int i = 0; i < s.length() - k; i++) {
            if (vow.contains(s.charAt(i))){
                res--;
            }
            if (vow.contains(s.charAt(i+k))) {
                res++;
            }
            ans = Math.max(ans, res);
        }

        return ans;
    }
}
