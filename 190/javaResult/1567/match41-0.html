class Solution {
    public int maxVowels(String s, int k) {
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        char[] cs = s.toCharArray();
        int res = 0;
        int cur = 0;
        for (int i=0; i<cs.length; ++i) {
            char c = cs[i];
            if (set.contains(c)) {
                cur++;
            }
            
            if (i >= k) {
                char prev = cs[i-k];
                if (set.contains(prev)) {
                    cur--;
                }
            }
            res = Math.max(res, cur);
        }
        return res;
    }
}
