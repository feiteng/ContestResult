import java.util.*;


class Solution {
    public int maxVowels(String s, int k) {

        int windowCount = 0;
        for(int idx = 0; idx < k; idx++) {
            if (isVowel(s.charAt(idx))) { windowCount++; }
        }
        int max = windowCount;
        final int len = s.length();
        for(int idx = k; idx < len; idx++) {
            if (isVowel(s.charAt(idx))) { windowCount++; }
            if (isVowel(s.charAt(idx-k))) { windowCount--; }
            max = Math.max(max, windowCount);
        }
        return max;
    }

    private boolean isVowel(char a) {
        return a=='a' || a=='e' || a=='i' || a=='o' || a=='u';
    }
}
