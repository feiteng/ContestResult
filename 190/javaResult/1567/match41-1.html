class Solution {
    public int maxVowels(String s, int k) {
        Set<Character> set = new HashSet();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        char[] chs = s.toCharArray();
        int cnt = 0;
        
int max = 0;
        int n = chs.length;
        for(int i=0;i<n;i++){
            char ch = chs[i];
            if(set.contains(ch)){
                cnt ++;
            }
            if(i>=k){
                char pre = chs[i-k];
                if(set.contains(pre)){
                    cnt --;
                }
            }
            max = Math.max(max, cnt);
        }
        return max;
    }
}
