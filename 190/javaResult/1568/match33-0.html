/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] res = new int[1];
        int[] counts = new int[10];
        findPath(root, res, counts);
        return res[0];
    }
    
    private void findPath(TreeNode root, int[] res, int[] counts) {
        if (root == null) {
            return;
        }
        counts[root.val]++;
        if (root.left == null && root.right == null) {
            if (oddCount(counts) <= 1) {
                res[0]++;
            }
            counts[root.val]--;
            return;
        }

        findPath(root.left, res, counts);
        findPath(root.right, res, counts);
        counts[root.val]--;
    }
    
    private int oddCount(int[] counts) {
        int res = 0;
        for (int i = 1; i <= 9; i++) {
            if (counts[i] % 2 != 0) {
                res++;
            }
        }
        return res;
    }
}
