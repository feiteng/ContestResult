/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int ans = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] cnts = new int[9];
        help(root, cnts);
        return ans;
    }
    
    public void help(TreeNode node, int[] cnts){
        if(node == null) return;
        int key = --node.val;
        cnts[key]++;
        
if(node.left == null && node.right == null){
            if(isPseudoPalindromic(cnts)){
                ans++;
            }
            cnts[key]--;
            return;
        }
        if(node.left != null){
            help(node.left, cnts);
        }
        if(node.right != null){
            help(node.right, cnts);
        }
        cnts[key]--;
    }
    private boolean isPseudoPalindromic(int[] cnts){
        int oddCnt = 0;
        for(int i=0; i<9; i++){
            if((cnts[i] & 1)!=0){
                oddCnt++;
            }
        }
        return oddCnt<2;
    }
}
