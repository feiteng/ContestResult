/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        List<List<Integer>> result = new ArrayList<>();
        check(root, new ArrayList<>(), result);
        int num = 0;
        for (List<Integer> list: result) {
            if (isValid(list)) {
                num ++;
            }
        }
        return num;
    }
    private void check(TreeNode root, List<Integer> list, List<List<Integer>> result) {
        if (root == null) {
            return;
        }
        list.add(root.val);
        if (root.left == null && root.right == null) {
            result.add(new ArrayList<>(list));
        }
        check(root.left, list, result);
        list.remove(list.size() - 1);
        list.add(root.val);
        check(root.right, list, result);
        list.remove(list.size() - 1);
    }
    private boolean isValid(List<Integer> list) {
        if (list.size() == 0) {
            return false;
        }
        Map<Integer,Integer> map = new HashMap<>();
        int count = 0;
        for (int elem: list) {
            map.put(elem, map.getOrDefault(elem, 0) + 1);
        }
        for (int key: map.keySet()) {
            if (map.get(key) % 2 == 1) {
                count ++;
                if (count > 1) {
                    return false;
                }
            }
        }
        return true;
    }
}
