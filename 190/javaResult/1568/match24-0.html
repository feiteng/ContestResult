/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    
    int cnt = 0;
    
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] bucket = new int[10];
        helper(root, bucket);
        return cnt;
    }
    
    private void helper(TreeNode cur, int[] bucket) {
        if (cur == null) return;
        bucket[cur.val]++;
        if (cur.left == null && cur.right == null && checkPalPer(bucket)) cnt++;
        helper(cur.left, bucket);
        helper(cur.right, bucket);
        bucket[cur.val]--;
    }
    
    private boolean checkPalPer(int[] bucket) {
        int target = 0;
        for (int i = 0; i < 10; i++) {
            if (bucket[i] % 2 == 1) target++;
            if (target > 1) return false;
        }
        return true;
    }
}
