/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int RES = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] bucket = new int[10];
        helper(root, bucket);
        return RES;
    }
    
    public void helper(TreeNode root, int[] bucket){
        if(root == null) return;
        bucket[root.val]++;
        if(root.left == null && root.right == null){
            if(isPlin(bucket)){
                RES++;
            }
        }
        helper(root.left, bucket);
        helper(root.right, bucket);
        bucket[root.val]++;
        
    }
    
    public boolean isPlin(int[] bucket){
        int count = 0;
        for(int i: bucket){
            if(i % 2 != 0){
                count++;
            }
        }
        return count < 2;
    }
}
