/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int count = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] c = new int[10];
        check(root, c);
        return count;
    }
    
    void check (TreeNode root, int[] c) {
        if (root == null) return;
        
        c[root.val]++;
        if (root.left == null && root.right == null) {
            if (palindrome(c)) ++count;
        
}
        check(root.left, c);
        check(root.right, c);
        c[root.val]--;
    }
    
    boolean palindrome(int[] v) {
        int odds = 0;
        for (int i = 1; i < v.length; ++i) {
            if (v[i] % 2 == 1) ++odds;
            if (odds > 1) return false;
        }
        
        return true;
    }
}
