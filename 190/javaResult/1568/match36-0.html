/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int[] array = new int[10];
    int res = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        if(root == null) {
            return 0;
        }
        dfsHelper(root);
        return res;
    }
    public void dfsHelper(TreeNode root) {
        if(root == null) {
            return;
        }
        if(root.left == null && root.right == null) {
            array[root.val] ++;
            if(palindromeCheck(array)) {
                res ++;
            }
            array[root.val] --;
            return;
        }
        array[root.val] ++;
        dfsHelper(root.left);
        dfsHelper(root.right);
        array[root.val] --;
    }
    public boolean palindromeCheck(int[] array) {
        int curr = 0;
        for(int i = 1; i < array.length; i ++) {
            if(array[i] % 2 == 1) {
                curr ++;
            }
        }
        if(curr > 1) {
            return false;
        }
        return true;
    }
}
