/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int count;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] map = new int[10];
        pseudoPalindromicPathsRecur(root, map);
        return count;
    }
    
    public void pseudoPalindromicPathsRecur (TreeNode root, int[] map) {
        if (root == null)
            return;
        if (root.left == null && root.right == null) {
             map[root.val]++;
            if (checkPalinMap(map))
                count++
;
            map[root.val]--;
            return;
        }
        
        map[root.val]++;
        //System.out.println(root.val);
        pseudoPalindromicPathsRecur(root.left, map);
        pseudoPalindromicPathsRecur(root.right, map);
        map[root.val]--;
    }
    
    public boolean checkPalinMap(int[] map) {
        int countOdds = 0;
        for (int i = 1; i <= 9; i++) {
            if (map[i]%2 == 1)
                countOdds++;
        }
        return countOdds <= 1;
    }
}
