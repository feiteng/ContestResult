/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        if (root == null) return 0;
        dfs(root, new int[10]);
        return ret;
    }
    private int ret = 0;
    private void dfs(TreeNode node, int[] cnt) {
        if (node.left == null && node.right == null) {
            int odd = 0;
            cnt[node.val] += 1;
            for (int i = 1; i <= 9; i++) {
                if (cnt[i] % 2 == 1) odd += 1;
            }
            if (odd <= 1) ret += 1;
            cnt[node.val] -= 1;
            return;
        }
        cnt[node.val] += 1;
        if (node.left != null) dfs(node.left, cnt);
        if (node.right != null) dfs(node.right, cnt);
        cnt[node.val] -= 1;
    }
}
