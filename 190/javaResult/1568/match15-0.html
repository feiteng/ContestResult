/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int num = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int [] f = new int[10];
        dfs(root,f);
        return num;
    }
    
    public void dfs(TreeNode node, int[] f){
        int val = node.val;
        f[val]++;
        if(node.left==null && node.right==null){
            if(isP(f))num++;
            return;
        }
        if(node.left!=null){
            dfs(node.left,f);
            f[node.left.val]--;
        }
        if(node.right!=null){
            dfs(node.right,f);
            f[node.right.val]--;
        }
    
    }
    
    public boolean isP(int []f){
        int n = 0;
        for(int i=1;i<10;i++){
            if(f[i]%2!=0)n++;
        }
        if(n<2)return true;
        return false;
    }
    
}
