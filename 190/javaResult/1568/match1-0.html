/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int res = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        check(root, new HashSet<>());
        return res;
    }
    
    private void check(TreeNode node, Set<Integer> set){
        if(node == null) return;
        
        int val = node.val;
        if(set.contains(val)) set.remove(val);
        else set.add(val);
        // System.out.println(node.val + "   " + set.size());
        if(node.left == null && node.right == null){
            if(set.size() <= 1) res++;
            return;
        }
        

        check(node.left, new HashSet<>(set));
        check(node.right, new HashSet<>(set));
    }
}
