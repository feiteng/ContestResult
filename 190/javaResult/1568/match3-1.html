/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    
    private int res;
    
    public int pseudoPalindromicPaths (TreeNode root) {
        
        int[] stats = new int[10];
        helper(root, stats);
        return res;
    }
    
    private void helper(TreeNode node, int[] stats) {
        
        ++stats[node.val];
        
        
if (node.left == null && node.right == null)
            if (check(stats)) ++res;
        
        if (node.left != null)
            helper(node.left, stats);
        if (node.right != null)
            helper(node.right, stats);
        
        --stats[node.val];
    }
    
    private boolean check(int[] stats) {
        
        int odd = 0;
        for (int num : stats) {
            if ((num % 2) == 1) ++odd;
            if (odd > 1) return false;
        }
        
        return true;
    }
}
