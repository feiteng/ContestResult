/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] res = new int[1];
        check(root, new int[10], res);
        return res[0];
    }
    
    void check(TreeNode n, int[] counts, int[] res) {
        if (n == null) return;
        counts[n.val]++;
        if (n.left == null && n.right == null) {
            if (isPalindrom(counts)) {
                res[0]++;
            }
        } else {
            check(n.left, counts, res);
            check(n.right, counts, res);
        }
        counts[n.val]--;
    }
    
    boolean isPalindrom(int[] counts) {
        boolean foundOdd = false;
        for (int c : counts) {
            if (c % 2 != 0) {
                if (foundOdd) {
                    return false;   
                } else {
                    foundOdd = true;
                }
            }
        }
        return true;
    }
}
