/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] A = new int[9];
        dfs(root, A);
        return count;
    }
    
    private void dfs(TreeNode root, int[] A) {
        if (root == null)
            return;
        
        A[root.val - 1]++;
        if (root.left == null && root.right == null)
            if (judge(A))
                count++;
        
        if (root.left != null)
            dfs(root.left, A);
        if (root.right != null)
            dfs(root.right, A);
        A[root.val - 1]--;
    }
    
    private boolean judge(int[] A) {
        int odd = 0;
        for (int a: A) {
            if ((a & 1) == 1)
                odd++;
            if (odd == 2)
                return false;
        }
        return true;
    }
}
