/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int ans = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] map = new int[9];
        map[root.val - 1]++;  
        dfs(root, map);
        return ans;
    }
    
    public void dfs(TreeNode root, int[] map) {  
        
if (root.left == null && root.right == null) {
            if (check(map)) ans++;
            return;
        }
        if (root.left != null) {
            map[root.left.val - 1]++;
            dfs(root.left, map);  
            map[root.left.val - 1]--;
        }
        
        if (root.right != null) {
            map[root.right.val - 1]++;
            dfs(root.right, map); 
            map[root.right.val - 1]--; 
        }
           
    }
    
    public boolean check(int[] map) {
        int sum = 0;
        for (int i = 0; i < map.length; i++) {
            sum += map[i] % 2;
        
}
        if (sum == 0 || sum == 1) return true;
        return false;
    }
}
