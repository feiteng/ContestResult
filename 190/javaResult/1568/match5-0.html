/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        helper(new HashSet<>(), root);
        return count;
    }

    private void helper(Set<Integer> set, TreeNode root) {
        if(set.contains(root.val)) {
            set.remove(root.val);
        } else {
            set.add(root.val);
        }
        if(root.left == null && root.right == null) {
            if(set.size() <= 1) {
                count++;
            }
            return;
        }
        if(root.left != null) {
            helper(new HashSet<>(set), root.left);
        }
        if(root.right != null) {
            helper(new HashSet<>(set), root.right);
        }
    }
}
