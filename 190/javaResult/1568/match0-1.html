/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    
    int ans = 0;
    
    public void func(TreeNode root, int[] arr) {
        if(root == null) {
            return;
        }
        arr[root.val]++;
        if(root.left == null && root.right == null) {
            int odd = 0;
            for(int i = 0; i < arr.length; i++) {
                if(arr[i] % 2 == 1) {
                    odd++;
                }
            }
            if(odd < 2) {
                ans++;
            }
            arr[root.val]--;
            return;
        }
        func(root.left, arr);
        func(root.right, arr);
        arr[root.val]--;
    }
    
    public int pseudoPalindromicPaths (TreeNode root) {
        func(root, new int[10]);
        return ans;
    }
}
