/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        // Capture node counts, make sure they're even at the leaf (except for one node)
        int[] countSoFar = new int[10];
        check(root, countSoFar);
        return count;
    }
    
    private void check(TreeNode root, int[] countSoFar) {
        if (root == null) return;
        int cur = root.val;
        if (root.left == null && root.right == null) {
            countSoFar[cur]++;
            if (isValid(countSoFar)) {
                count++;
            }
            countSoFar[cur]--;
            return;
        }
        countSoFar[cur]++;
        check(root.left, countSoFar);
        check(root.right, countSoFar);
        countSoFar[cur]--;
    }
    
    private boolean isValid(int[] counts) {
        int oddCount = 0;
        for (int num : counts) {
            if (num % 2 != 0) oddCount++;
        }
        return oddCount <= 1;
    }
}
