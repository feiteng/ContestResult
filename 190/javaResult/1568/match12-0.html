/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
     int r = 0;

    public int pseudoPalindromicPaths(TreeNode root) {
        int[] count = new int[10];
        dfs(root, count);
        return r;
    }

    private void dfs(TreeNode root, int[] count) {
        if (root == null) {
            return;
        }
        count[root.val]++;
        if (root.left == null && root.right == null) {
            int odds = 0;
            for (int i = 1; i <= 9; i++) {
                if (count[i] % 2 == 1) {
                    odds++;
                }
            }
            if (odds <= 1) {
                r++;
            }
            count[root.val]--;
            return;
        }
        if (root.left != null) {
            dfs(root.left, count);
        }
        if (root.right != null) {
            dfs(root.right, count);
        }
        count[root.val]--;
    }
}
