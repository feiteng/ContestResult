/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int ans = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        if (root.left == null && root.right == null)
        {
            return 1;
        }
        int[] hash = new int[10];
        dfs(root, hash);
        return ans;
    }
    void dfs(TreeNode root, int[] hash)
    {
        hash[root.val]++;
        if (root.left == null && root.right == null)
        {
            if (ok(hash))
            {
                ans++;
            }
            hash[root.val]--;
            return;
        }
        
        if (root.left != null)
        {
            
            dfs(root.left, hash);
            //hash[root.val]--;
        }
        
        if (root.right != null)
        {
            
            dfs(root.right, hash);
            //hash[root.val]--;
        }
        hash[root.val]--;
    }
    boolean ok(int[] hash)
    {
        int cnt = 0;
        for(int i = 1; i <= 9; i++)
        {
            if (hash[i] % 2 == 1)
            {
                cnt++;
            }
        }
        return (cnt <= 1);
    }
}
