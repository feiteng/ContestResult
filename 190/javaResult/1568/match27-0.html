/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
  public int pseudoPalindromicPaths (TreeNode root) {
    int[] paths = new int[11];
    return traverse(root, paths);
  }

  private int traverse(TreeNode node, int[] paths) {
    if (node.left == null && node.right == null) {
      paths[node.val]++;
      int oddNum = 0;
      for (int i = 1; i < 10; i++) {
        if (paths[i] % 2 == 1) oddNum++;
      }
      paths[node.val]--;
      return oddNum > 1 ? 0 : 1;
    }

    int res = 0;
    paths[node.val]++;
    if (node.left != null) {
      res += traverse(node.left, paths);
    }
    if (node.right != null) {
      res += traverse(node.right, paths);
    }
    paths[node.val]--;
    return res;
  }
}
