/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
  int[] cnt   = new int[10];
    int   total = 0;

    public int pseudoPalindromicPaths(TreeNode root) {
        if (root == null || (root.left == null && root.right == null)) {
            return 1;
        }
        cnt[root.val]++;
        helper(root.left);
        
helper(root.right);
        return total;
    }

    private void helper(TreeNode node) {
        if (node == null) {
            return;
        }
        cnt[node.val]++;
        if (node.left == null && node.right == null) {
            
int oddCnt = 0;
            for (int i = 0; i < cnt.length; i++) {
                if ((cnt[i] & 1) != 0) {
                    oddCnt++;
                }
            }
            if (oddCnt <= 1) {
                total++;
            
}
            cnt[node.val]--;
            return;
        }
        helper(node.left);
        helper(node.right);
        cnt[node.val]--;
    }
}
