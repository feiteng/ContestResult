/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] count = new int[10];
        count[root.val]++;
        return ppp(root, count);
    }
    
    private int ppp(TreeNode root, int[] count) {
        if (root.left == null && root.right == null) return psuedo(count)
;
        int res = 0;
        if (root.left != null) {
            count[root.left.val]++;
            res += ppp(root.left,count);
            count[root.left.val]--;
        }
        if (root.right != null) {
            count[root.right.val]++;
            res += ppp(root.right,count);
            count[root.right.val]--;
        }
        return res;
    }
    
    
    
    private int psuedo(int[] count) 
{
        int odd = 0;
        for (int c : count) {
            if (c % 2 == 1) odd++;
        }
        return odd <= 1 ? 1 : 0;
    }
}
