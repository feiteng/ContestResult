/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    private int totalCnt;
    public int pseudoPalindromicPaths (TreeNode root) {
        totalCnt = 0;
        int[] counts = new int[9];
        helper(root, counts);
        
        
return totalCnt;
    }
    
    private void helper(TreeNode root, int[] counts) {
        if (root == null) {
            return;
        }
        
        counts[root.val - 1]++;
        
if (root.left == null && root.right == null) {
            // System.out.println(Arrays.toString(counts));
            if (isPalidrome(counts)) {
                totalCnt++;
            }
        }
        else {
            helper(root.left, counts);
            helper(root.right, counts);
        }       
        counts[root.val - 1]--;
    }
    
    private boolean isPalidrome(int[] counts) {
        boolean isOdd = false;
        for (int count : counts) {
            if (count % 2 == 1) {
                if (isOdd) {
                    return false;
                }
                isOdd = true;
            }
        }
        
        return true;
    }
}
