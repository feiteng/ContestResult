/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count = 0;
    
public int pseudoPalindromicPaths (TreeNode root) {
        int[] vals = new int[10];
        traversal(root, vals);
        return count;
    }
    
    public void traversal(TreeNode root, int[] vals) {
        if (root == null) return;
        vals[root.val]++;
        if (root.left == null && root.right == null) {
            int odd = 0;
            for (int i = 0; i<=9; i++) {
                if (vals[i] == 1 || vals[i]%2 == 1) odd++;
            
}
            if (odd<=1) count++;
        }
        traversal(root.left, vals);
        traversal(root.right, vals);
        vals[root.val]--;
    }
}
