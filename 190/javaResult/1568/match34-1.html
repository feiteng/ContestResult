/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int count = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        int [] dcount = new int[9];
        helper(root, dcount);
        return count;
    }
    
    public void helper(TreeNode root, int[] dcount) {
        if (root == null) {
            return;
        }
        
        // root is not null, check if root is leaf node
        if (root.left == null && root.right == null) {
            // add value of root to dcount
            dcount[root.val-1]++;
            if (isPalindromic(dcount)) 
{
                count++;
            }
            dcount[root.val-1]--;
            return;
        }
        //root intermediate node
        // add its current value and go down recursively
        dcount[root.val-1]++;
        if (root.left != null) {
            helper(root.left, dcount);
        }
        if (root.right != null) {
            helper(root.right, dcount);
        }
        dcount[root.val-1]--;
    }
    
    public boolean isPalindromic(int[] dcount) {
        
int odd = 0;
        for (int i=0; i < dcount.length; i++) {
            if (dcount[i] % 2 == 1) {
                odd++;
            }
        }
        return odd < 2;
    }
    
}
