/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    int ans = 0;
    
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] prof = new int[10];
        
        dfsCheck(root, prof);
            
        return ans;
    }
    
    void dfsCheck(TreeNode node, int[] prof) {
        if (node == null) {
            return;
        
}
        
        ++prof[node.val];
        
        if (node.left == null && node.right == null) {
            if (check(prof)) {
                ++ans;
            }
        }
        
        dfsCheck(node.left, prof);
        dfsCheck(node.right, prof);
        --prof[node.val];
    }
    
    boolean check(int[] prof) {
        int oddCnt = 0;
        for (int x: prof) {
            if (x % 2 == 1) {
                ++oddCnt;
            }
        }
        
        return oddCnt <= 1;
    }
}
