/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public int[] cnt = new int[10];
    public int ans = 0;
    public int pseudoPalindromicPaths (TreeNode root) {
        dfs(root);
        return ans;
    }
    
    public void dfs(TreeNode r){
        cnt[r.val]++;
        if(r.left == null && r.right == null){
            if(check()) ans++;
            cnt[r.val]--;
            return;
        
}
        if(r.left != null) dfs(r.left);
        if(r.right != null) dfs(r.right);
        cnt[r.val]--;
    }
    
    public boolean check()
{
        boolean f = false;
        for(int i = 0; i < 10; i++){
            if(cnt[i]%2 != 0){
                if(f){
                    return false;
                } else {
                    f = true;
                }
            }
        }
        return true;
    }
}
