/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {

	int res = 0;
	public int pseudoPalindromicPaths (TreeNode root) {
		int[] valCount = new int[10];
		traverse(root,valCount);
		return res;

	}
	
	private void traverse (TreeNode root, int[] valCount) {
		if (root.left == null && root.right == null) {
			valCount[root.val]++;
			int oddCount = 0;
			for (int i = 1; i <= 9; i++) {
				if (valCount[i] % 2 != 0) {
					oddCount++;
				}
			}
			if (oddCount <= 1) {
				res++;
			}
			valCount[root.val]--;
			return;
		}
		if (root.left != null) {
			valCount[root.val]++;
			traverse(root.left,valCount);
			valCount[root.val]--;
		}
		if (root.right != null) {
			valCount[root.val]++;
			traverse(root.right,valCount);
			valCount[root.val]--;
		}
	}
}
