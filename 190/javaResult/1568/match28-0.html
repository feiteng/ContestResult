/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public int pseudoPalindromicPaths (TreeNode root) {
        int[] res = new int[11];
        dfs(root, res);
        return res[10];
    }
    public void dfs(TreeNode root, int[] cnts) {
        if (root == null)
            return;
        cnts[root.val]++;
        if (root.left == null && root.right == null) {
            // System.out.println(root.val+" "+Arrays.toString(cnts));
            int numOdd = 0;
            for (int i = 0; i < 10; i++) {
                if(cnts[i]%2==1)
                    numOdd++;
                if (numOdd>1)
                    break;
            }
            if (numOdd <= 1)
                cnts[10]++;
        }
        dfs(root.left, cnts);
        dfs(root.right, cnts);
        cnts[root.val]--;
    }
}
