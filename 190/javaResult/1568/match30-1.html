/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    private int count;
    public int pseudoPalindromicPaths (TreeNode root) {
        if (root == null) {
            return 0;
        }
        count = 0;
        int[] a = new int[10];
        a[root.val] += 1;
        dfs(root, a);
        return count;
    }
    private void dfs(TreeNode node, int[] a) {
        if (node.left == null && node.right == null) {
            if (isPP(a)) 
{
                ++count;
            }
            return;
        }
        if (node.left != null) {
            a[node.left.val] += 1;
            dfs(node.left, a);
            a[node.left.val] -= 1;
        }
        if (node.right != null) {
            a[node.right.val] += 1;
            dfs(node.right, a);
            a[node.right.val] -= 1;
        }
    }
    
    private boolean isPP(int[] a) {
        
int oddCount = 0;
        for (int i : a) {
            if (i % 2 != 0) {
                ++oddCount;
            }
        }
        return oddCount <= 1;
    }
}
