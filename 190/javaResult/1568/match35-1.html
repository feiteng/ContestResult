class Solution {
        public int pseudoPalindromicPaths(TreeNode root) {
            int[] r = new int[1];

            pseudoPalindromicPaths(root, r, new int[9]);

            return r[0];
        }

        void pseudoPalindromicPaths(TreeNode root, int[] r, int[] count) {
            if (root == null) return;

            count[root.val - 1]++;

            if (root.left == null && root.right == null) {
                r[0] += c(count);
            
} else {
                pseudoPalindromicPaths(root.left, r, count);
                pseudoPalindromicPaths(root.right, r, count);        
            }


            count[root.val - 1]--;


        }

        int c(int[] count) {
            int r = 0;
            for (int i : count) {
                if (i % 2 != 0) {
                    if (r > 0)
                        return 0
;
                    r++;
                }
            }
            return 1;
        }


    }
