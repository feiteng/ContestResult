/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int count = 0;
    
public int pseudoPalindromicPaths (TreeNode root) {
        if (root == null) return 0;
        traverse(root, new int[10]);
        return count;
    }
    
    
private void traverse(TreeNode node, int[] nums) {
        nums[node.val]++;
        if (node.left == null && node.right == null) {
            int c = 0;
            for (int i = 1; i < 10; i++) {
                if (nums[i] % 2 == 1) c++;
            }
            if (c < 2) count++;
        
}
        if (node.left != null) traverse(node.left, nums);
        if (node.right != null) traverse(node.right, nums);
        nums[node.val]--;
    }
}
