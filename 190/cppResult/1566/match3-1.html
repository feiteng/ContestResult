class Solution {
public:
    int isPrefixOfWord(string sentence, string s) {
        stringstream ss;
        ss<<sentence;
        string cur;
        int ind=0;
        while(ss>>cur) {
            if(cur.size()>=s.size() && cur.substr(0, s.size())==s) return ind+1;
            ind++;
        }
        return -1;
    }
};
