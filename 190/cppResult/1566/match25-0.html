class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        // int ans = -1;
        stringstream input(sentence);
        string tmp;
        int cnt = 1;
        while(getline(input, tmp, ' ')) {
            if (tmp.find(searchWord) == 0) {
                return cnt;
            }
            cnt++;
        }
        return -1;
    }
};
