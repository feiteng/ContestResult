class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        stringstream ss(sentence); 

        string word; 

        int cnt = 1;
        while(getline(ss, word, ' ')) { 
            if (word.substr(0, min(word.size(),searchWord.size())) == searchWord) {
                return cnt;
            }
            cnt ++;
        } 
        
        return -1;
    }
};
