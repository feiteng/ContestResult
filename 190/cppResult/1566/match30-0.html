class Solution {
public:
  int isPrefixOfWord(string t, string s) {
    stringstream ss;
    ss << t;
    int k=1;
    while(ss >> t) {
      if(t.size()>=s.size()&&t.substr(0,s.size())==s) return k;
      k++;
    }
    return -1;
  }
