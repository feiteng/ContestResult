class Solution {
public:
    int isPrefixOfWord(string sentence, string searchWord) {
        istringstream is(sentence);
        int index = 0;
        string tmp;
        int N = searchWord.size();
        while (is >> tmp) {
            index++;
            if (N > tmp.size()) continue;
            if (tmp.substr(0, N) == searchWord) return index;
        }
        return -1;
        
    }
};
