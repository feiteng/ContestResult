class Solution {
public:
    // Use a sliding window
    bool ifVowel(char c) {
        return (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u');
    }
    
    int maxVowels(string s, int k) {
        int answer = 0, temp = 0;
        for (int i = 0; i < k; ++i) {
            if (ifVowel(s[i])) temp++;
        }
        answer = temp;
        for (int i = k; i < s.size(); ++i) {
            if (ifVowel(s[i])) temp++;
            if (ifVowel(s[i-k])) temp--;
            answer = max(answer, temp);
        }
        return answer;
    }
};
