bool is(char c)
{
    if(c=='a'||c=='e'||c=='i'||c=='o'||c=='u') return true ;
    
    return false ;
}
class Solution {
public:
    int maxVowels(string s, int k) {
        int cnt=0 ;
        int ans=0 ;
        int n=s.length() ;
        
        for(int i=0;i<k;i++)
        {
            if(is(s[i])) cnt++ ;
        }
        ans=cnt ;
        
        for(int i=k;i<n;i++)
        {
            if(is(s[i])) cnt++ ;
            
            if(is(s[i-k])) cnt-- ;
            
            ans=max(ans,cnt) ;
            
        }
        
        return ans ;
        
        
    }
};
