class Solution {
public:
    int maxVowels(string s, int k) {
        int cnt = 0;
        int l = s.size();
        int a = 0;
        for(int i=0; i<k; i++){
            if(s[i] == 'a' || s[i] == 'o' || s[i] =='e' || s[i] == 'i' || s[i] == 'u')
                cnt++;
        }
        a = cnt;
        for(int i=k; i<l; i++){
            if(s[i-k] == 'a' || s[i-k] == 'o' || s[i-k] =='e' || s[i-k] == 'i' || s[i-k] == 'u')
                cnt--;
            if(s[i] == 'a' || s[i] == 'o' || s[i] =='e' || s[i] == 'i' || s[i] == 'u')
                cnt++;
             a = max(a, cnt);
        }
        return a;
    }
};
