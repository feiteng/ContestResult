class Solution {
public:
    bool is(char c)
    {
        return c=='a'||c=='e'||c=='i'||c=='o'||c=='u';
    }
    int maxVowels(string s, int k) {
        int len=s.size();
        int cnt=0;
        for(int i=0;i<k;i++)
        {
            if(is(s[i]))cnt++;
        }
        int ans=cnt;
        for(int i=k;i<len;i++)
        {
            if(is(s[i]))cnt++;
            if(is(s[i-k]))cnt--;
            ans=max(ans,cnt);
        }
        return ans;
    }
};
