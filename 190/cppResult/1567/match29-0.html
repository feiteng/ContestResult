class Solution {
public:
    int maxVowels(string s, int k) {
        
        int max_ans = 0;
        int count = 0;
        
        unordered_set<char> S = {'a', 'e', 'i', 'o', 'u'};
        
        for(int i = 0; i < s.length(); i++){
            
            if(i - k >= 0 && S.find(s[i - k]) != S.end())
                count = count - 1;
            
            if(S.find(s[i]) != S.end())
                count++;
            
            max_ans = max(max_ans, count);
        }
        
        return max_ans;
    }
};
