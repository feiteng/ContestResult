class Solution {
public:
    bool is_vowel(char c){
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
    int maxVowels(string s, int k) {
        int max_count = 0, count = 0;
        for(int i = 0; i < s.size(); i++){
            if(is_vowel(s[i])) count++;
            if(i >= k && is_vowel(s[i - k])) count--;
            max_count = max(max_count , count);
        }
        return max_count;
    }
};
