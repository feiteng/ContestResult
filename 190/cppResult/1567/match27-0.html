class Solution {
public:
    int maxVowels(string s, int k) {
        int vowelCount = 0;
        for (int i = 0; i < k; ++i) {
            if (isVowel(s[i])) {
                vowelCount++;
            }
        }
        int maxVowelCount = vowelCount;
        for (int i = 1; i + k <= s.size(); ++i) {
            if (isVowel(s[i-1]) && !isVowel(s[i+k-1])) {
                --vowelCount;
            }
            if (!isVowel(s[i-1]) && isVowel(s[i+k-1])) {
                ++vowelCount;
            }
            maxVowelCount = max(maxVowelCount, vowelCount);
        }
        return maxVowelCount;
    }
    bool isVowel(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
};
