class Solution {
    bool isVowel(char c) {
        return (c == 'a' || 
               c == 'e' ||
               c == 'i' ||
               c == 'o' ||
               c == 'u' );
    }
public:
    int maxVowels(string s, int k) {
        int ans = 0;
        int cnt = 0;
        for (int i = 0; i < s.size(); i++) {
            if (isVowel(s[i])) {
                cnt ++;
            }
            if (i-k>=0 && isVowel(s[i-k])) {
                cnt --;
            }
            ans = max(ans, cnt);
        }
        return ans;
    }
};
