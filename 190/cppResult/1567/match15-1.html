class Solution {
public:
    int count(vector<int>& buf) {
        return buf['a'-'a'] + buf['e'-'a'] + buf['i'-'a'] + buf['o'-'a'] + buf['u'-'a'];
    }
    
    int maxVowels(string s, int k) {
        vector<int> buf(26, 0);
        for (int i = 0; i < k; i++) {
            buf[s[i]-'a']++;
        }
        int vowelCount = count(buf);
        for (int i = k; i < s.size(); i++) {
            buf[s[i]-'a']++;
            buf[s[i-k]-'a']--;
            vowelCount = max(vowelCount, count(buf));
        }
        return vowelCount;
    }
};
