class Solution {
public:
    int maxVowels(string s, int k) {
        int ans =0;
        for(int i=0;i<k;i++)
        {
            if(s[i]=='a' || s[i]=='e' || s[i]=='u' || s[i]=='o' || s[i]=='i' )
                ans++;
        }
        int temp = ans;
        // cout << temp << endl;
        for(int i=k;i<s.length();i++)
        {
            if(s[i]=='a' || s[i]=='e' || s[i]=='u' || s[i]=='o' || s[i]=='i' )
                temp++;
            if(s[i-k]=='a' || s[i-k]=='e' || s[i-k]=='u' || s[i-k]=='o' || s[i-k]=='i' )
                temp--;
            // cout << temp << endl;
            ans=max(ans,temp);
        }
        return ans;
        
    }
};
