class Solution {
public:
    int maxVowels(string s, int k) {
        int res = 0;
        int cnt = 0;
        unordered_set<char> st = {'a','e','i','o','u'};
        for(int i = 0; i < s.size() ; i++){
            if(i >= k && (st.find(s[i-k])!=st.end()))cnt--;
            if(st.find(s[i])!=st.end())cnt++;
            res = max(cnt, res);
        }
        return res;
    }
};
