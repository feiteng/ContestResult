class Solution {
public:
    int maxVowels(string s, int k) {
        unordered_set<char> vowels = {'a','e','i','o','u'};
        int vc = 0;
        for(int i = 0; i < k; i++)
        {
            if(vowels.count(s[i]))
            {
                vc++;
            }
        }
        
        int best = vc;
        for(int i = k; i < s.length(); i++)
        {
            if(vowels.count(s[i-k]))
            {
                vc--;
            }
            if(vowels.count(s[i]))
            {
                vc++;
            }
            best = max(best, vc);
        }
        
        return best;
    }
};
