class Solution {
public:
    bool vowel(char ch) {
        return (ch == 'a' || ch == 'i' || ch == 'e' || ch == 'o' || ch == 'u');
    }
    
    int maxVowels(string s, int k) {
        int mx = 0;
        int cur = 0;
        for (int i = 0; i < k; i++) {
            if (vowel(s[i])) cur++;
        }
        mx = cur;
        for (int i = k; i < s.size(); i++) {
            if (vowel(s[i])) cur++;
            if (vowel(s[i - k])) cur--;
            mx = max(mx, cur);
        }
        return mx;
    }
};
