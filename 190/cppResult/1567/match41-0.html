class Solution {
public:
    bool isVowel(char ch) {
        return (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u');
    }
    
    int maxVowels(string s, int k) {
        int vowels = 0, ans = 0;
        for(int i = 0; i < k; i++) {
            if(isVowel(s[i])) vowels++;
        }
        ans = max(ans, vowels);
        for(int i = k; i < s.length(); i++) {
            if(isVowel(s[i - k])) vowels--;
            if(isVowel(s[i])) vowels++;
            ans = max(ans, vowels);
        }
        return ans;
    }
};
