class Solution {
public:
    int maxVowels(string s, int k) {
        int vow = 0;
        for(int i=0;i<k;i++){
            if(isVowel(s[i])){
                vow++;
            }
        }
        int mxvow = vow;
        for(int i=k;i<s.length();i++){
            if(isVowel(s[i])){
                vow++;
            }
            if(isVowel(s[i-k])){
                vow--;
            }
            mxvow = max(mxvow,vow);
        }
        return mxvow;
    }
    
    bool isVowel(char a){
        return (a =='a' || a=='e' || a=='i' || a=='o' || a=='u');
    }
};
