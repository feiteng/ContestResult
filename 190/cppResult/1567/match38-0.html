class Solution {
public:
    int isvowel(char c){
        if(c=='a' || c=='e' || c=='i' || c=='o' || c=='u')
            return 1;
        return 0;
    }
    
    int maxVowels(string s, int k) {
        int n=s.length();
        
        int ans=0;
        int cnt=0;
        
        for(int i=0;i<k;i++){
            if(isvowel(s[i]))
                cnt++;
        }
        
        ans=cnt;
        
        for(int i=k;i<n;i++){
            if(isvowel(s[i]))
                cnt++;
            if(isvowel(s[i-k]))
                cnt--;
            
            ans = max(ans,cnt);
        }
        
        return ans;
    }
};
