class Solution {
public:
    int maxVowels(string s, int k) {
        int ans = 0;
        int vowels = 0;
        for (int i = 0; i < k; i++) {
            if (isVowel(s[i])) vowels++;
        }

        ans = vowels;
        for (int i = k; i < s.size(); i++) {
            if (isVowel(s[i])) vowels++;
            if (isVowel(s[i-k])) vowels--;
            ans = max(ans, vowels);
        }
        return ans;
    }
    
    bool isVowel(char c) {
        return c == 'a' || c == 'i' || c == 'u' || c == 'e' || c == 'o'; 
    }
};
