class Solution {
public:
    int judge(char t)
    {
        if(t=='a'||t=='e'||t=='i'||t=='o'||t=='u')
            return 1;
        return 0;
    }
    int maxVowels(string s, int k) {
        int n=s.size();
        if(n<k) return 0;

        int ans=0;
        int now=0;
        for(int i=0;i<k;i++)
        {
            if(judge(s[i]))
                now++;
        }
        ans=now;
        for(int i=k;i<n;i++)
        {
            if(judge(s[i])) now++;
            if(judge(s[i-k])) now--;
            ans=max(ans,now);
        }
        return ans;
    }
};
