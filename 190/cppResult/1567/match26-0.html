class Solution {
public:
    int maxVowels(string& s, int k) {
        int res=0;
        int n=s.size();
        for (int i=0; i<k; ++i){
            if (s[i]=='i' || s[i]=='a' || s[i]=='o' || s[i]=='u' || s[i]=='e') res++;
        }
        int a=res;
        for (int i=k; i<n; i++){
            if (s[i]=='i' || s[i]=='a' || s[i]=='o' || s[i]=='u' || s[i]=='e') res++;
            if (s[i-k]=='i' || s[i-k]=='a' || s[i-k]=='o' || s[i-k]=='u' || s[i-k]=='e') res--;
            a=max(a, res);
        }
        return a;
    }
};
