class Solution {
public:
    bool IsVowel(char c)
    {
        return (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u');
    }
    int maxVowels(string s, int k) {
        int maxv = 0;
        int v = 0;
        for(int i=0; i<k;i++){
            if(IsVowel(s[i])) v++;
        }
        maxv = v;
        for(int i=k;i<s.size();i++){
            if(IsVowel(s[i-k])) v--;
            if(IsVowel(s[i])) v++;
            maxv = max(maxv, v);
        }
        return maxv;
    }
};
