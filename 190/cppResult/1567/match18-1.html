class Solution {
    bool isVowel(char ch) {
        return ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u';
    }
public:
    int maxVowels(string s, int k) {
        int ans = 0;
        int temp = 0;
        
        for (int i = 0; i < s.length(); i++) {
            if (isVowel(s[i])) {
                temp++;
            }
            if (i - k >= 0 && isVowel(s[i-k])) {
                temp--;
            }
            ans = max(ans, temp);
        }
        
        return ans;
    }
};
