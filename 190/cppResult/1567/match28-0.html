class Solution {
public:
    int maxVowels(string s, int k) {
        
        int max_num_vow=0;
        int curr_num=0;
        
        for(int i=0;i<k;++i)
        {
            if(isVowel(s[i]))
                curr_num++;
        }
        
        max_num_vow=max(max_num_vow, curr_num);
        
        
        for(int i=k;i<s.size();++i)
        {
            
            if(isVowel(s[i]))
                curr_num++;
            
            if(isVowel(s[i-k]))
                curr_num--;
            
            max_num_vow=max(max_num_vow, curr_num);
            
        }
        
        return max_num_vow;
        
        
    }
    
    bool isVowel(char s)
    {
        return s=='a' || s=='e' || s=='i' || s=='o' || s=='u';
    }
};
