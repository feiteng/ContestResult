/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int pseudoPalindromicPaths (TreeNode* root) {
        dv.assign(9, 0);
        rs = 0;
        mysol(root);
        return rs;
    }

    void mysol(TreeNode* root)
    {
        if(root==NULL)
            return;
        if(root->left==NULL&&root->right==NULL)
        {
            dv[root->val-1]++;
            int sg = 0;
            for(int i=0; i<9; i++)
                if(dv[i]%2==1)
                    sg++;
            if(sg<2)
                rs++;
            dv[root->val-1]--;
            return;
        }
        dv[root->val-1]++;
        mysol(root->left);
        mysol(root->right);
        dv[root->val-1]--;
    }
    vector<int> dv;
    int rs;
};
