/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int ans = 0;
    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> m(10, 0);
        dfs(root, m, 0);
        
        return ans;
    }
    
    void dfs(TreeNode* root, vector<int>& m, int count) {
        
        if (root->left == NULL && root->right == NULL) {
            m[root->val]++;
            count++;
            // cout << root->val << endl;
            // for (int i = 1; i <= 9; i++) {
            //     cout << m[i] << " ";
            // }
            // cout << endl;
            if (count % 2) {
                int odd = 0;
                for (int i = 1; i <= 9; i++) {
                    if (m[i]%2) odd++;
                }
                if (odd <= 1) {
                    ans++;
                }
            } else {
                int odd = 0;
                for (int i = 1; i <= 9; i++) {
                    if (m[i]%2) odd++;
                }
                if (odd == 0) {
                    ans++;
                }
            }
            m[root->val]--;
            count --;
            return;
        }
        
        m[root->val]++;
        if (root->left)
        dfs(root->left, m, count+1);
        
        if (root->right)
        dfs(root->right, m, count+1);
        
        
        m[root->val]--;
    }
};
