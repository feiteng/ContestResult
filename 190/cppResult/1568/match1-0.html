/**
 * Definition for a binary tree node.
 * xxruct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int ans;
    int o=1;
    int z=0;
    set<int>xx;
    void solve(TreeNode* root)
    {
        if(root==NULL)
            return;
        
        else if(root->left==NULL && root->right==NULL)
        {
            int rr=root->val;
            if(xx.find(rr)==xx.end())
                xx.insert(rr);
            else
                xx.erase(rr);
            if(xx.size()==z || xx.size()==o)
            {
                ans+=o;
            }
            if(xx.find(rr)==xx.end())
                xx.insert(rr);
            else
                xx.erase(rr);
            return;
        }
        int pp=root->val;
        if(xx.find(pp)==xx.end())
            xx.insert(pp);
        else
            xx.erase(pp);
        
        solve(root->left);
        solve(root->right);
        
        if(xx.find(pp)==xx.end())
            xx.insert(pp);
        else
            xx.erase(pp);
        
    }
    int pseudoPalindromicPaths (TreeNode* root) {
        if(root!=NULL && root->left==NULL && root->right==NULL)
            return o;
        
        ans=0;
        xx.insert(root->val);
        solve(root->left);
        
        xx.clear();
        
        xx.insert(root->val);
        solve(root->right);
        
        return ans;
    }
};
