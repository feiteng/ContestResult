/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    
    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> counter(10, 0);
        return count(root, counter);
    }
    
    int count(TreeNode* node, vector<int> &counter) {
        if(!node)
            return 0;
        counter[node->val]++;
        if(!node->left && !node->right) {
            int odd = 0;
            for(int i = 1; i < 10; i++) {
                if(counter[i] % 2)
                    odd++;
            }
            counter[node->val]--;
            if(odd > 1)
                return 0;
            return 1;
        }
        
        int res = count(node->left, counter) + count(node->right, counter);
        counter[node->val]--;
        return res;
    }
    
};
