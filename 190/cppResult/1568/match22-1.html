/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int ans = 0;

    void g(TreeNode* node, vector<int> f) {
        if (!node) return;
        
        f[node->val]++;
        
        if (!node->left && !node->right) {
            // only one odd
            int odd = 0;
            for (int i = 0; i < 10; i++) {
                if (f[i] % 2) odd++;
            }
            if (odd <= 1) ans++;
            
            return;
        }
        
        g(node->left, f);
        
        g(node->right, f);
        
        
    }
    int pseudoPalindromicPaths (TreeNode* root) {
        g(root, vector<int>(10));
        return ans;
    }
};
