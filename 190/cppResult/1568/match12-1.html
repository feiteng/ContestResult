/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int pseudoPalindromicPaths (TreeNode* root) {
      
      int ans = 0;
      vector<int> v(10);
      helper (root, ans, v);
      return ans;
    }

    void helper(TreeNode* r, int& ans, vector<int>& v) {
      if (r == nullptr) return;
      
      v[r->val]++;
      
      if (r->left == nullptr && r->right == nullptr) {
        int cnt = 0;
        for (int j = 0; j < v.size(); j++) {
        //  cout << v[j] << " ";
          if (v[j] % 2) cnt++;
        }
        //cout << "cnt =" << cnt;

        //cout << endl;
        if (cnt == 0 || cnt == 1) ans++;
        v[r->val]--;
        
        return;
      }
      
      helper(r->left, ans, v);
      helper(r->right, ans, v);
      v[r->val]--;
    }
};
