/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:

    int findans(TreeNode* root, vector<int> &count)
    {
        if(!root)
            return 0;
        count[root->val]++;
        if(!root->left && !root->right)
        {    
            int check = 0;
            for(int i = 1; i <= 9; i++)
            {
                if((count[i])&1)
                    check++;
            }
            count[root->val]--;
            if(check > 1)
                return 0;
            //cout<<root->val<<endl;
            return 1;
        }
        int ans = findans(root->left, count) + findans(root->right, count);
        count[root->val]--;
        return ans;
    }
    
    int pseudoPalindromicPaths (TreeNode* root) {
        if(!root)
            return 0;
        vector<int> count;
        count.assign(10, 0);
        return findans(root, count);
    }
};
