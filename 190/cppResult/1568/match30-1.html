/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

class Solution {
public:
    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> arr(10,0);
        int res = 0;
        checkPalindrome(root, arr, res);
        return res;
    }
    
    
    void checkPalindrome(TreeNode * root, vector<int> & arr, int & counter){
        if(root->left==NULL && root->right==NULL){
            arr[root->val]++;

            int count = 0;
            for(int i=0;i<10;i++){
                if(arr[i]&1) count++;
            }
            if(count<=1) counter++;
            arr[root->val]--;
        }
        
        if(root->left!=NULL){
            arr[root->val]++;
            checkPalindrome(root->left,arr,counter);
            arr[root->val]--;
        }
        if(root->right!=NULL){
            arr[root->val]++;
            checkPalindrome(root->right,arr,counter);
            arr[root->val]--;
        }
    }
};
