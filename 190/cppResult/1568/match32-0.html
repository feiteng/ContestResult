/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int total = 0;
    bool is(vector<int> &v) {
        int t = 0;
        int odd = 0;
        for (int i = 1; i <= 9; ++i) {
            t += v[i];
            if (v[i] & 1) {
                odd++;
            }
        }
        
        if (((t&1) && odd == 1) || (!(t&1) && odd == 0)) {
            return true;
        }
        
        return false;
    }
    
    void dfs(TreeNode *root, vector<int> num) {
        if (root == NULL) {
            return;
        }    
        
        num[root->val]++;
        if (root->left == NULL && root->right == NULL) {
            if (is(num)) {
                total++;
            }
            return;
        }
        
        dfs(root->left, num);
        dfs(root->right, num);
    }
    
    int pseudoPalindromicPaths (TreeNode* root) {
        total = 0;
        vector<int> v(10, 0);
        dfs(root, v);
        return total;
    }
};
