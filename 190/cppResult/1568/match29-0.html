/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    int cnt[10]{0},ans=0;
public:
    int pseudoPalindromicPaths (TreeNode* root) {
        dfs(root);
        return ans;
    }
    
    void dfs(TreeNode* p) {
        if(!p) return;
        ++cnt[p->val];
        if(!p->left && !p->right){
            int odd=0;
            for(int i=0;i<10;i++){
                if(cnt[i]&1) odd++;
            }
            if(odd<=1) ans++;
            --cnt[p->val];
            return;
        }
        dfs(p->left);
        dfs(p->right);
        --cnt[p->val];
    }
};
