/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:

    int ans=0,cnts[10]={0};
    void dfs(TreeNode* R){
        if(!R) return;
        cnts[R->val]++;
        if(R&&!R->left&&!R->right){
            int cnt=0;
            for(int i=1;i<=9;i++){
                if(cnts[i]%2==1) cnt++;
            }
            if(cnt<=1) ans++;
            cnts[R->val]--;
            return;
        }
        dfs(R->left);
        dfs(R->right);
        cnts[R->val]--;
    }
    int pseudoPalindromicPaths (TreeNode* root) {
        dfs(root);
        return ans;
    }
};
