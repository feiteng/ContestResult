/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
	int res;
    int pseudoPalindromicPaths (TreeNode* root) {
        res = 0;
        int mask = 0;
        if(!root) return 0;
        dfs(root, mask);

        return res;
    }
    void dfs(TreeNode *root, int mask) {
    	int v = root->val;
    	mask ^= (1 << v);
    	cout << mask << endl;
    	if(!root->left && !root->right) {
    		if(countBit(mask) <= 1) {
                res++; 
            }
    		return;
    	}
    	if(root->left)
	    	dfs(root->left, mask);
    	if(root->right)
    		dfs(root->right, mask);

    }

    int countBit(int n) {
    	int c = 0;
    	while(n) {
    		n = n&(n-1);
    		c++;
    	}
    	return c;
    }
};
