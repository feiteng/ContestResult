/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> cur(10, 0);
        int count = 0;
        dfs(root, cur, count);
        return count;
    }
    
    void dfs(TreeNode* root, vector<int>& cur, int& count) {
        if (!root) return;
        
        ++cur[root->val];
        dfs(root->left, cur, count);
        dfs(root->right, cur, count);
        if (!root->left && !root->right) {
            int odd = 0;
            for (int i = 1; i <= 9 && odd < 2; ++i) {
                if (cur[i] % 2) ++odd;
            }
            if (odd < 2) ++count;
        }
        --cur[root->val];
    }
};
