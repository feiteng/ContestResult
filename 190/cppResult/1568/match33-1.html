class Solution {
public:
    int ans = 0;
    
    int pseudoPalindromicPaths (TreeNode* root) {
        vector<int> freq(10);
        dfs(root, freq);
        return ans;
    }
    
    void dfs(TreeNode* node, vector<int>& freq) {

        freq[node->val]++;
        
        if (node->left == node->right) {  // leaf
            int cntOdd = 0;
            for (int i = 1; i <= 9; i++) {
                if (freq[i] % 2) cntOdd++;
            }
            if (cntOdd < 2) ans++;
        }
        
        if (node->left) dfs(node->left, freq);
        if (node->right) dfs(node->right, freq);
        
        freq[node->val]--;
    }
};
