class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        vector<vector<int>> dp(nums1.size(), vector<int>(nums2.size(), 0));
        dp[0][0] = nums1[0]*nums2[0];
        for (int r = 1; r < nums1.size(); ++r)
            dp[r][0] = max(nums1[r]*nums2[0], dp[r-1][0]);
        for (int c = 1; c < nums2.size(); ++c)
            dp[0][c] = max(nums1[0]*nums2[c], dp[0][c-1]);
        for (int r = 1; r < nums1.size(); ++r) {
            for (int c = 1; c < nums2.size(); ++c) {
                dp[r][c] = max({nums1[r]*nums2[c] + max(dp[r-1][c-1], 0), dp[r-1][c-1], dp[r-1][c], dp[r][c-1]});
            }
        }
        // for (int r = 0; r < nums1.size(); ++r) {
        //     for (int c = 0; c < nums2.size(); ++c) {
        //         cout << dp[r][c] << " ";
        //     }
        //     cout << endl;
        // }
        return dp.back().back();
    }
};
