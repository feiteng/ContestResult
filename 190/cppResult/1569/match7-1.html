class Solution {
public:
    int maxDotProduct(vector<int>& n1, vector<int>& n2) {
        int n = n1.size();
        int m = n2.size();
        
        vector<vector<int>> dp(n, vector<int>(m, 0));

        dp[0][0] = n1[0] * n2[0];
        for (int i = 1; i < n; ++i) {
            dp[i][0] = max(n1[i] * n2[0], dp[i - 1][0]);
        }
        for (int j = 1; j < m; ++j) {
            dp[0][j] = max(n1[0] * n2[j], dp[0][j - 1]);
        }
        for (int i = 1; i < n; ++i) {
            for (int j = 1; j < m; ++j) {
                dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
                dp[i][j] = max(dp[i][j], n1[i] * n2[j] + max(dp[i - 1][j - 1], 0));
                
                // for (int k = 0; k < i; ++k) {
                //     int rem = k > 0 ? max(dp[k - 1][j - 1], 0) : 0;
                //     dp[i][j] = max(dp[i][j], n1[k] * n2[j] + rem);
                // }
                // for (int k = 0; k < j; ++k) {
                //     int rem = k > 0 ? max(dp[i - 1][k - 1], 0) : 0;
                //     dp[i][j] = max(dp[i][j], n1[i] * n2[k] + rem);
                // }
                
            }
        }
        return dp.back().back();
    }
};
