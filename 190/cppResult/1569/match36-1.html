class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size(), n = nums2.size();
        vector<vector<int>> dp(m, vector<int>(n, 0)), mv(m, vector<int>(n, 0));
        dp[0][0] = nums1[0] * nums2[0];
        mv[0][0] = dp[0][0];
        for (int i = 1; i < n; i++) {
            dp[0][i] = nums1[0] * nums2[i];
            mv[0][i] = max(mv[0][i - 1], dp[0][i]);
        }
        for (int i = 1; i < m; i++) {
            dp[i][0] = nums1[i] * nums2[0];
            mv[i][0] = max(mv[i - 1][0], dp[i][0]);
        }

        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[i][j] = max(0, mv[i - 1][j - 1]) + nums1[i] * nums2[j];
                mv[i][j] = max(dp[i][j], max(mv[i][j - 1], mv[i - 1][j]));
            }
        }
        
        // for (auto& v : dp) {
        //     for (int i : v) {
        //         cout << i << " ";
        //     }
        //     cout << endl;
        // }
        // cout << endl;
        // for (auto& v : mv) {
        //     for (int i : v) {
        //         cout << i << " ";
        //     }
        //     cout << endl;
        // }
        return mv[m - 1][n - 1];
    }
};
