class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        
        int len1 = nums1.size(), len2 = nums2.size();
        
        vector < vector < int > > dp(len1, vector < int > (len2, 0));
        
        bool isallneg1 = true, isallpos1 = true, isallneg2 = true, isallpos2 = true;
        
        for(int i = 0; i < nums1.size(); i++){
            if(nums1[i] >= 0)
                isallneg1 = false;
            if(nums1[i] <= 0)
                isallpos1 = false;
        }
        
        for(int i = 0; i < nums2.size(); i++){
            if(nums2[i] >= 0)
                isallneg2 = false;
            if(nums2[i] <= 0)
                isallpos2 = false;
        }
        
        if((isallneg1 && isallpos2) || (isallneg2 && isallpos1))
            dp[0][0] = nums1[0] * nums2[0];
        else

            dp[0][0] = max(nums1[0]*nums2[0], 0);
        
        for(int i = 1; i < len1; i++)
            dp[i][0] = max(nums1[i]*nums2[0], dp[i-1][0]);
        
        for(int i = 1; i < len2; i++)
            dp[0][i] = max(nums1[0]*nums2[i], dp[0][i-1]);
        
        for(int i = 1; i < len1; i++)
            for(int j = 1; j < len2; j++)
                dp[i][j] = max(nums1[i]*nums2[j] + dp[i-1][j-1], max(dp[i-1][j], dp[i][j-1]));
        
        return dp[len1-1][len2-1];
    }
};
