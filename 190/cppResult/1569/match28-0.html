class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size();
        int m = nums2.size();
        vector<vector<int>> dp(n+1, vector<int>(m+1, 0));
        dp[1][1] = nums1[0] * nums2[0];
        for(int j = 1; j < m; j++) {
            dp[1][j+1] = max(dp[1][j], nums1[0]* nums2[j]);
        }
        for(int i = 1; i < n; i++) {
            dp[i+1][1] = max(dp[i][1], nums1[i]*nums2[0]);
        }
        
        // int ans = INT_MIN;
        for(int i = 1; i < n; i++) {
            for(int j = 1; j < m; j++) {
                int mul = nums1[i] * nums2[j];
                dp[i+1][j+1] = max({dp[i+1][j], dp[i][j+1], dp[i][j]+mul, mul});                
            }
        }
        int ans = INT_MIN;
        for(int i = 1; i <= n; i++) {
            for(int j = 1; j <= m; j++) {
                //cout << dp[i][j] << " ";
                ans = max(ans, dp[i][j]);
            }
            //cout << endl;
        }
        return ans;
    }
};
