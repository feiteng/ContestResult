class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size();
        int m = nums2.size();
        vector<vector<int>> P (n + 1, vector<int>(m + 1, -1000000000));
        P[0][0] = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                P[i + 1][j + 1] = max(P[i][j] + nums1[i] * nums2[j], max(P[i][j + 1], P[i + 1][j]));
                P[i + 1][j + 1] = max(P[i + 1][j + 1], nums1[i] * nums2[j]);
            }
        }
        return P[n][m];
    }
};
