class Solution {
public:
    int maxDotProduct(vector<int>& a, vector<int>& b) {
        int n = a.size();
        int m = b.size();
        vector<vector<int>> dp(n, vector<int>(m));
        dp[0][0] = a[0]*b[0];
        for (int i = 1; i < n; i++) {
            dp[i][0] = max(dp[i-1][0], a[i]*b[0]);
        }
        for (int j = 1; j < m; j++) {
            dp[0][j] = max(dp[0][j-1], a[0]*b[j]);
        }
        for (int i = 1; i < n; i++) {
            for (int j = 1; j < m; j++) {
                int best = max(dp[i-1][j], dp[i][j-1]);
                best = max(best, a[i]*b[j]);
                best = max(best, a[i]*b[j]+dp[i-1][j-1]);
                dp[i][j] = best;
                //cout << best << " ";
            }
            //cout << "\n";
        }
        return dp[n-1][m-1];
    }
};
