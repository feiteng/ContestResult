class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n1=nums1.size();
        int n2=nums2.size();
        int dp[n1][n2];
        memset(dp,0,sizeof(dp));
        dp[0][0]=nums1[0]*nums2[0];
        for(int i=1;i<n2;++i){
            dp[0][i]=max(dp[0][i-1],nums1[0]*nums2[i]);
        }
        for(int i=1;i<n1;++i){
            dp[i][0]=max(dp[i-1][0],nums1[i]*nums2[0]);
        }
        for(int i=1;i<n1;++i){
            for(int j=1;j<n2;++j){
                dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
                dp[i][j]=max(dp[i][j],nums1[i]*nums2[j]+max(0,dp[i-1][j-1]));
            }
        }
        return dp[n1-1][n2-1];
    }
};
