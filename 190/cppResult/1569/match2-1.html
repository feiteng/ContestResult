class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int colsize=nums1.size();
        int rowsize=nums2.size();
        vector<vector<int> > dp(colsize,vector<int>(rowsize,INT_MIN));
        dp[0][0]=nums1[0]*nums2[0];
        for(int i=1;i<colsize;++i){
            dp[i][0]=max(dp[i-1][0],nums1[i]*nums2[0]);
        }
        for(int j=1;j<rowsize;++j){
            dp[0][j]=max(dp[0][j-1],nums1[0]*nums2[j]);
        }
        for(int i=1;i<colsize;++i){
            for(int j=1;j<rowsize;++j){
                dp[i][j]=max(max(max(dp[i-1][j],dp[i][j-1]),max(dp[i-1][j-1],dp[i-1][j-1]+nums1[i]*nums2[j])),nums1[i]*nums2[j]);
            }
        }
        // for(int i=0;i<colsize;++i){
        //     for(int j=0;j<rowsize;++j){
        //         cout << dp[i][j] << " ";
        //     }cout << endl;
        // }
        return dp[colsize-1][rowsize-1];
    }
};
