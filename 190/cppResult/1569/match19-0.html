class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size(), n = nums2.size(), curr = 0;
        vector<vector<int>> dp(m,vector<int> (n,0));
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                curr = nums1[i]*nums2[j];
                if (i==0&&j==0) dp[i][j] = curr;
                else if (i==0)  dp[i][j] = max(dp[i][j-1],curr);
                else if (j==0)  dp[i][j] = max(dp[i-1][j],curr);
                else    dp[i][j] = max(max(max(dp[i-1][j-1],0)+curr,dp[i-1][j]),dp[i][j-1]);
            }
        }
        return dp[m-1][n-1];
    }
};
