class Solution {
public:
    int maxDotProduct(vector<int>& V, vector<int>& W) {
        int m = V.size();
        int n = W.size();
        
        vector<vector<int>> B(m, vector<int>(n, 0));
        B[0][0] = V[0]*W[0];
        
        for(int i = 1; i < m; i++)
        {
            B[i][0] = max(B[i-1][0], V[i]*W[0]);
        }
        for(int j = 1; j < n; j++)
        {
            B[0][j] = max(B[0][j-1], V[0]*W[j]);
        }
        
        for(int i = 1; i < m; i++)
        {
            for(int j = 1; j < n; j++)
            {
                int t = max(B[i-1][j], B[i][j-1]);
                t = max(t, B[i-1][j-1] + V[i]*W[j]);
                t = max(t, V[i]*W[j]);
                B[i][j] = t;
            }
        }
        
        return B[m-1][n-1];
    }
};
