class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int len1 = nums1.size(), len2 = nums2.size();
        int dp[len1 + 1][len2 + 1];
        fill ((int*)dp, (int*)dp + (len1 + 1) * (len2 + 1), INT_MIN);
        dp[0][0] = 0;
        for (int i = 0; i < len1; ++i){
            for (int j = 0; j < len2; ++j)
                dp[i + 1][j + 1] = max(max(0, dp[i][j]) + nums1[i] * nums2[j], max(dp[i + 1][j], dp[i][j + 1]));
        }
        return dp[len1][len2];
    }
};
