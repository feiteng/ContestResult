class Solution {
public:
    int maxDotProduct(vector<int>& a, vector<int>& b) {
        int n = a.size(), m = b.size(), ans = a[0] * b[0];
        vector<vector<int>> f(n, vector<int>(m));
        
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {

                if (i == 0 && j == 0)
                    f[i][j] = a[0] * b[0];
                else if (i == 0)
                    f[i][j] = max(a[i] * b[j], f[i][j-1]);
                else if (j == 0)
                    f[i][j] = max(a[i] * b[j], f[i-1][j]);
                else
                    f[i][j] = max(a[i] * b[j] + max(f[i-1][j-1], 0), max(f[i][j-1], f[i-1][j]));
            }
        }
        return f[n-1][m-1];
    }
};
