class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int M = nums1.size(), N = nums2.size();
        vector<vector<int>> dp(M + 1, vector<int>(N + 1, INT_MIN));

        for(int i = 1; i <= M; i++){
            for(int j = 1; j <= N; j++)
                dp[i][j] = max(max(dp[i - 1][j - 1], 0) + nums1[i - 1] * nums2[j - 1], max(dp[i][j - 1], dp[i - 1][j]));
        }
        return dp[M][N];
    }
};
