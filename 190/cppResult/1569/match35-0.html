class Solution {
public:
    int maxDotProduct(vector<int>& n1, vector<int>& n2) {
        int n = n1.size(), m = n2.size();
        vector<vector<int>> arr(n+1, vector<int>(m+1,0));
        for(int i=0; i<=m; i++)
            arr[0][1] = 0;
        for(int i=0; i<n; i++)
            arr[i][0] = 0;
        for(int i=1; i<=n; i++) {
            for(int j=1; j<=m; j++)
                arr[i][j] = n1[i-1] * n2[j-1];
        }
        
        int res = -1e9;
        for(int i=1; i<=n; i++) {
            for(int j=1; j<=m; j++) {
                res = max(res, arr[i][j]+arr[i-1][j-1]);
                arr[i][j] = max(arr[i][j]+arr[i-1][j-1], max(arr[i][j], max(arr[i-1][j], arr[i][j-1])));
                // cout << arr[i][j] << " ";
            }
            // cout << endl;
        }
        
        return res;
    }
};
