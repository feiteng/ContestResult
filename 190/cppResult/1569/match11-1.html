class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int len1 = nums1.size();
        int len2 = nums2.size();
        vector<vector<int>> dp(len1 + 1, vector<int>(len2 + 1, 0));
        //vector<vector<int>> info1(len1, {INT_MAX, 0, 0, INT_MIN});
        //for (int i = 0; i < )
        //int ans = nums1[0] * nums2[0];
        for (int i = 0; i < len1; i++) {
            for (int j = 0; j < len2; j++) {
                if (i == 0 && j == 0) {
                    dp[i + 1][j + 1] = nums1[0] * nums2[0];
                }
                else if (i == 0) {
                    dp[i + 1][j + 1] = max(dp[i + 1][j], nums1[0] * nums2[j]);
                }
                else if (j == 0) {
                    dp[i + 1][j + 1] = max(dp[i][j + 1], nums1[i] * nums2[0]);
                }
                else {
                    dp[i + 1][j + 1] = max(dp[i][j], max(dp[i + 1][j], max(dp[i][j + 1], max(0, dp[i][j]) + nums1[i] * nums2[j])));
                }
                //cout << i << "," << j << "," << dp[i + 1][j + 1] << endl;
                //for (int k = 0; k <= i; k++) {
                //    cur_max = max(cur_max, dp[k][j] + nums2[j] * nums1[k]);
                    //if (cur_max == 0) cout << "1: " << i << "," << j << "," << k << endl;
                //}
                //if (i > 0) {
                //    cur_max = max(cur_max, dp[i][j + 1]);
                //}
                //for (int k = 0; k <= j; k++) {
                //    cur_max = max(cur_max, dp[i][k] + nums1[i] * nums2[k]);
                    //if (cur_max == 0) cout << "2: " << i << "," << j << "," << k << endl;
                //}
            }
        }
        return dp[len1][len2];
    }
};
