class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n = nums1.size();
        int m = nums2.size();
        vector<vector<int>> dp(m, vector<int>(n, INT_MIN));
        for(int i=m-1;i>=0;i--){
            for(int j=n-1;j>=0;j--){
                // cout<<"calculating for "<<i<<" "<<j<<endl;
                if(i == m-1 && j == n-1){
                    dp[i][j] = nums1[j] * nums2[i];
                }
                else if(i==m-1){
                    dp[i][j] = max(nums2[i] * nums1[j] , dp[i][j+1]);
                }
                else if(j==n-1){
                    dp[i][j] = max(nums2[i] * nums1[j], dp[i+1][j]);
                }
                else{
        // /;            cout<<"res will be max of "<<dp[i][j+1]<<" "<<dp[i+1][j]<<" "<<(nums1[j]*num2[i])
                    dp[i][j] = max(nums1[j]* nums2[i] , max(max(dp[i][j+1], dp[i+1][j]), max(dp[i+1][j+1], (nums1[j] * nums2[i]) + dp[i+1][j+1])));
                }
            }
        }
        // for(int i=0;i<m;i++){
        //     for(int j=0;j<n;j++){
        //         cout<<dp[i][j]<<" ";
        //     }
        //     cout<<endl;
        // }
        return dp[0][0];
    }
};
