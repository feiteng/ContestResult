class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n1 = nums1.size(), n2 = nums2.size();
        vector<vector<int>> dp(n1 + 1, vector<int>(n2 + 1, 0));
        
        int ans = nums1[0] * nums2[0];
        for (int i = 0; i < n1; ++i) {
            for (int j = 0; j < n2; ++j) {
                dp[i + 1][j + 1] = max(nums1[i] * nums2[j] + dp[i][j], max(dp[i + 1][j], dp[i][j + 1]));
                ans = max(ans, nums1[i] * nums2[j] + dp[i][j]);
            }
        }
        
        return ans;
    }
};
