class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int l1 = (int)nums1.size(), l2 = (int)nums2.size();
    vector<vector<int> > dp(l1 + 1, vector<int>(l2 + 1, -500000000));
    int ans = 0x80000000;
    for(int i = 1; i <= l1; i++)
        for(int j = 1; j <= l2; j++)
        {
            dp[i][j] = nums1[i - 1] * nums2[j - 1];
            dp[i][j] = max(dp[i][j], dp[i - 1][j - 1] + nums1[i - 1] * nums2[j - 1]);
            dp[i][j] = max(dp[i][j], max(dp[i - 1][j], dp[i][j - 1]));
            ans = max(ans, dp[i][j]);
        }
    return ans;
    }
};
