class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int M = nums1.size(), N = nums2.size();
        vector<vector<int>> dp(M + 1, vector<int>(N + 1, 0));
        int res = INT_MIN;
        int cur = INT_MIN;
        for (int i = 0; i < M; ++i) {
            cur = max(nums1[i] * nums2[0], cur);
            dp[i + 1][1] = cur;
            res = max(dp[i + 1][1], res);
        }
        cur = INT_MIN;
        for (int j = 0; j < N; ++j) {
            cur = max(nums1[0] * nums2[j], cur);
            dp[1][j + 1] = cur;
            res = max(dp[1][j + 1], res);
        }
        for (int i = 1; i < M; ++i) {
            for (int j = 1; j < N; ++j) {
                dp[i + 1][j + 1] = max(max(dp[i][j + 1], dp[i + 1][j]), max(dp[i][j], 0) + nums1[i] * nums2[j]);
                res = max(dp[i + 1][j + 1], res);
            }
        }
        return res;
    }
};
