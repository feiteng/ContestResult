class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size(), n = nums2.size(), i, j;
        vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
        dp[0][0] = nums1[0]*nums2[0];
        for (i = 1; i < m; ++i) dp[i][0] = max(dp[i - 1][0], nums1[i]*nums2[0]);
        for (j = 1; j < n; ++j) dp[0][j] = max(dp[0][j - 1], nums1[0]*nums2[j]);
        for (i = 1; i < m; ++i) {
            // if (i == 0) dp[0][0] = nums1[0]*nums2[0];
            // else dp[i][0] = max(dp[i - 1][0], nums1[i]*nums2[0]);
            for (j = 1; j < n; ++j) {
                // if (i == 0) dp[i][j] = max(dp[i][j - 1], nums1[0]*nums2[j]);
                dp[i][j] = max(max(dp[i - 1][j], dp[i][j - 1]), max(0, dp[i - 1][j - 1]) + nums1[i]*nums2[j]);
            }
        }
        return dp[m - 1][n - 1];
    }
};
