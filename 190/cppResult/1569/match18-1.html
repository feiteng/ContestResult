class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size(), n = nums2.size();
        vector<vector<int>> dp(m, vector<int>(n, 0));
        dp[0][0] = nums1[0]*nums2[0];
        
        for (int i = 1; i < m; i++) {
            dp[i][0] = max(nums1[i]*nums2[0], dp[i-1][0]);
        }
        for (int i = 1; i < n; i++) {
            dp[0][i] = max(nums1[0]*nums2[i], dp[0][i-1]);
        }
        
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[i][j] = nums1[i]*nums2[j] + dp[i-1][j-1];
                dp[i][j] = max(dp[i][j], nums1[i]*nums2[j]);
                dp[i][j] = max(dp[i][j], dp[i-1][j]);
                dp[i][j] = max(dp[i][j], dp[i][j-1]);
            }
        }
        
        // for (int i = 0;i < m; i++) {
        //     for (int j = 0; j < n; j++) {
        //         cout<<dp[i][j]<<" ";
        //     }
        //     cout<<endl;
        // }
        
        return dp[m-1][n-1];
    }
};
