class Solution {
public:
    //dp[i][j] = max(dp[i-1][j-1] + a[i] * b[j], dp[i-1][j], dp[i][j-1]);
    int maxDotProduct(vector<int>& a, vector<int>& b) {
        int n = a.size(), m = b.size();
        vector<vector<int>>dp(n+1, vector<int>(m+1));
        int ans = 0;
        bool f1 = true, f2 = true;
        for(int i = 1; i < n; ++i){
            if(a[i] * a[i-1] < 0) f1 = false; 
        }
        for(int i = 1; i < m; ++i){
            if(b[i] * b[i-1] < 0) f2 = false;
        }
        if(f1 && f2 && a[0] * b[0] <= 0){
            int max1 = -1e9, max2 = -1e9, min1 = 1e9, min2 = 1e9;
            for(int x : a) max1 = max(x, max1), min1 = min(min1, x);
            for(int x : b) max2 = max(x, max2), min2 = min(min2, x);
            if(a[0] < 0) ans = max1 * min2;
            else ans = min1 * max2;
            return ans;
        }
        for(int i = 1; i <= n; ++i){
            for(int j = 1; j <= m; ++j){
                dp[i][j] = max(dp[i-1][j-1] + a[i-1] * b[j-1], max(dp[i-1][j], dp[i][j-1]));
                ans = max(ans, dp[i][j]);
            }
        }
        return ans;
    }
};
