class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int n1=nums1.size(), n2=nums2.size();
        vector<vector<int>> dp(n1+10, vector<int>(n2+10,-1000000));
        // for(int j=1;j<=n2;j++){
        //     dp[0][j]=0;
        // }
        // for(int i=1;i<=n1;i++){
        //     dp[i][0]=0;
        // }
        dp[0][0]=0;
        // dp[1][1]=nums1[0]*nums2[0];
        for(int i=0;i<n1;i++){
            for(int j=0;j<n2;j++){
                dp[i+1][j+1]=nums1[i]*nums2[j];
                dp[i+1][j+1]=max(dp[i+1][j+1],dp[i][j]+nums1[i]*nums2[j]);
                dp[i+1][j+1]=max(dp[i+1][j+1],dp[i][j+1]);
                dp[i+1][j+1]=max(dp[i+1][j+1],dp[i+1][j]);
                // cout<<dp[i+1][j+1]<<" ";
            }
            // cout<<endl;
        }
        return dp[n1][n2];
    }
};
