class Solution {
public:
    int maxDotProduct(vector<int>& nums1, vector<int>& nums2) {
        int m = nums1.size();
        int n = nums2.size();
        vector<vector<int>> dp(m, vector<int>(n, 0));
        int res = INT_MIN;
        if (m == 0 || n == 0) {
            return res;
        }
        dp[0][0] = nums1[0]*nums2[0];
        res = max(res, dp[0][0]);
        for (int i = 1; i < m; i++) {
            dp[i][0] = max(dp[i-1][0], nums1[i]*nums2[0]);
            res = max(res, dp[i][0]);
        }
        for (int j = 1; j < n; j++) {
            dp[0][j] = max(dp[0][j-1], nums1[0]*nums2[j]);
            res = max(res, dp[0][j]);
        }

        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                dp[i][j] = max(dp[i-1][j], max(dp[i][j-1], dp[i-1][j-1]));
                dp[i][j] = max(dp[i][j], max(dp[i-1][j-1] + nums1[i]*nums2[j], nums1[i]*nums2[j]));
                res = max(res, dp[i][j]);
            }
        }
        return res;
    }
};
