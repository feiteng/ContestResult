class Solution {
    public int numSteps(String s) {
        int ans = 0;
        int index = s.length() - 1;
        char[] ca = s.toCharArray();
        while (index > 0) {
            ans++;
            if (ca[index] == '0') {
                index--;
            } else {
                while (index >= 0 && ca[index] == '1') {
                    ans++;
                    index--;
                }
                if (index >= 0) {
                    ca[index] = '1';
                } 
            }
        }
        
        return ans;
    }
}
