class Solution {
    public int numSteps(String s) {
        return sln1(s);
    }
    
    
private int sln1(String s){
        int n = s.length();
        int count = 0;
        int carry = 0;
        for(int i = n-1;i>0;i--){
            int curr =s.charAt(i) -'0' + carry ;
            if(curr == 0){
                count++;
                carry = 0;
            }else if(curr == 1){
                count +=2;
                carry = 1;
            }else{
                carry = 1;
                count++;
            
}
        }
        
        if(carry == 1) count+=1;
        
        return count;
    }
}
