class Solution {
    public int numSteps(String s) {
        int res = 0;
        StringBuilder sb = new StringBuilder(s);
        while (sb.length() != 1) {
            int len = sb.length();
            if (sb.charAt(len - 1) == '0') {
                res++;
                sb.deleteCharAt(len - 1);
            } else {
                int from = len - 1;
                while (from >= 0 && sb.charAt(from) == '1') {
                    from--;
                }
                if (from == -1) {
                    res += len - from;
                    break;
                } else {
                    sb.setCharAt(from, '1');
                    res += len - from;
                    sb.delete(from + 1, len);
                }
            }
        }
        return res;
    }
}
