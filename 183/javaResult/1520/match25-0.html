class Solution {
    public int numSteps(String s) {
        char[] chrs = s.toCharArray();
        int index = s.length()-1;
        int cnt = 0;
        while (index > 0){
            if (chrs[index] == '0'){
                cnt++;
                index--;
            }
            else{
                while (index >= 0 && chrs[index] == '1'){
                    index--;
                    cnt++;
                }
                if (index >= 0){
                    chrs[index] = '1';
                }
                cnt++;
            }
        }
        return cnt;
    }
}
