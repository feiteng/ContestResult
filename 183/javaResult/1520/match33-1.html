class Solution {
    public int numSteps(String s) {
        if (s.equals("1")) {
            return 0;
        }
        
        
int result = 0;
        
        while (!s.equals("1")) {
            //System.out.println(s);
            if (s.charAt(s.length() - 1) == '0') {
                s = divide(s);
            } else {
                s = addOne(s);
            }
            result++;
        }
        return result;
    }
    
    private String addOne(String s) {
        String result = new String();
        
        int carry = 1;
        
        for (int i = s.length() - 1; i >= 0; i--) {
            int sum = carry;
            
            int count = s.charAt(i) - '0';
            sum += count;
            int add = sum % 2;
            carry = sum / 2;
            if (add == 1) {
                result = '1' + result;
            } else {
                result = '0' + result;
            }
            
            
        }
        
        if (carry == 1) {
            result = '1' + result;
        }
        
        return result;
    }
    
    private String divide(String s) {
        return s.substring(0, s.length() - 1);
    }
}
