class Solution {
    public int numSteps(String s) {
        int res = 0;
        for(int i = s.length() - 1; i > 0; i--){
            if(s.charAt(i) == '0') res++;
            else{
                res += 2;
                s = change(s, i);
                if(s.charAt(0) == '*') return res - 1 + Integer.valueOf(s.substring(1));
            }
        }
        return res;
    }
    
    private String change(String s, int idx){
        s = s.substring(0, idx + 1);
        StringBuilder sb = new StringBuilder(s);
        for(int i = sb.length() - 1; i >= 0; i--){
            if(sb.charAt(i) == '1'){
                if(i == 0) return "*" + sb.length();
                sb.setCharAt(i, '0');
            }else{
                sb.setCharAt(i, '1');
                return sb.toString();
            }
        }
        return sb.toString();
    }
}
