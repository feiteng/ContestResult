class Solution {
    public int numSteps(String s) {
        int len=s.length();
        char[] sArr=s.toCharArray();
        int res=0;
        while(len>1){
            if(sArr[len-1]=='0')
                len--;
            else{
                int val=len-1;
                while(val>=0 && sArr[val]=='1'){
                    sArr[val]='0';
                    val--;
                }
                if(val>=0)
                    sArr[val]='1';
                else{
                    sArr[0]=1;
                    //len--;
                    res++;
                }
                    
            }
            res++;
        }
    return res;
    }
}
