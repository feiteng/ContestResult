class Solution {
      public int numSteps(String s) {
        char arr[] = s.toCharArray();
        int sum[] = new int[arr.length * 2];
        for(int i=0; i < arr.length; i++) {
            sum[i] = arr[arr.length - i - 1] - '0';
        }
        int len = arr.length;
        int cur = 0;
        int ans = 0;
        while (cur < len - 1) {
           // System.out.println("pos: "+ cur +" "+ len);
            if(sum[cur] == 0) {
                ans++;
                cur++;
            } else {
                ans++;
                int p = cur;
                sum[cur]++;
                while (sum[p] > 1) {
                    sum[p+1] += sum[p] / 2;
                    sum[p] %= 2;
                    p++;
                }
                if(p + 1 > len) len = p+1;
            }
        }

        return ans;
    }

}
