class Solution {
    public int numSteps(String s) {
        char[] data = s.toCharArray();
        int idx = data.length - 1;
        int count = 0;
        
        while (idx >= 0) {
            char curr = data[idx];
            if (curr == '0') {
                idx--;
            }
            else {
                if (idx == 0) break;
                int ptr = idx;
                while (ptr >= 0 && data[ptr] == '1') {
                    data[ptr] = '0';
                    ptr--;
                
}
                if (ptr >= 0) data[ptr] = '1';
            }
            count++;
        }
        return count;
    }
}
