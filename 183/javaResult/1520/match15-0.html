class Solution {
public int numSteps(String s) {
	    	char[] t = s.toCharArray();
	    	int k = t.length-1;
	    	int res = 0;
	    	while(k > 0) {
	    		if(t[k] == '0') {
	    			res++; k--;
	    		}
	    		else {
	    			res++;
	    			while(k >= 0 && t[k] == '1') {
	    				res++; k--;
	    			}
	    			if(k >= 0) t[k] = '1';
	    		}
	    	}
	    	return res;
	    }
}
