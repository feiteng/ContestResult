class Solution {
    public int numSteps(String s) {
        
int step = 0;
        
        while(s.length() != 1){
            if(s.charAt(s.length()-1) == '0') {
                s = s.substring(0,
s.length()-1);
                step+=1;
            } else {
                s = add(s);
                step+=1;
            }
        }
        
        return step;
    }
    
    public String add(String s){
        if(s.equals("1")) return "10";
        if(s.charAt(s.length()-1) == '0') return s.substring(0,s.length()-1)+'1';
        return add(s.substring(0,s.length()-1))+'0';
    }

}
