class Solution {
    public int numSteps(String s) {
        StringBuilder sb = new StringBuilder(s);
        int iter = 0;
        while(sb.length() != 1){
            int last = sb.charAt(sb.length() - 1) - '0';
            if (last == 1){
                sb = addOne(sb);
            }else{
                sb = sb.deleteCharAt(sb.length()-1);
            }
            iter += 1;
            
            //System.out.println(sb); 
        }
        return iter;
    }
    
    
    private StringBuilder addOne(StringBuilder sb){
        int upper = 0;
        int idx = sb.length() - 1;
        while (idx >= 0 && sb.charAt(idx) == '1'){
            sb.setCharAt(idx,'0');
            idx--;
        }
        if (idx < 0) sb.insert(0, '1');
        else sb.setCharAt(idx,'1');
        return sb;
    }
}
