class Solution {
    public int numSteps(String s) {
        int n = s.length();
        int step = 0;
        int val = 0;
        for (int i = n - 1; i >= 1; --i) {
            int v = s.charAt(i) - '0';
            val += v;
            if (val == 2) {
                step += 1;
                val = 1;
            } else if (val == 1) {
                step += 2;
                val = 1;
            } else {
                step += 1;
            }
        }
        if (val == 1) step++;
        return step;
    }
}
