class Solution {
    public int numSteps(String s) {
        int st = 0;
        while(!s.equals("1"))
        {
            s = update(s);
            st++;
        }
        return st;
    }
    
    String update(String str)
    {
        int n = str.length();
        if(str.charAt(n - 1) == '0') return str.substring(0, n - 1);
        String re = "";
        
        int c = 1;
        
        for(int i = n - 1; i >= 0; i--)
        {
            int sum = str.charAt(i) - '0' + c;
            re += "" + sum % 2;
            c = sum / 2;            
        }
        
        if(c > 0) re += "1";
        
        return new StringBuilder(re).reverse().toString();
    }
}
