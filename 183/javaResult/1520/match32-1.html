class Solution {
    public int numSteps(String s) {
        
int count = 0;
        while (!s.equals("1")) {
            if (s.charAt(s.length()-1) == '0') {
                s = divide(s);
            } else {
                s = addOne(s);
            }
            // System.out.println(s);
            count++;
        }
        return count;
    }
    
    private String addOne(String s) {
        int ind = s.length()-1;
        char[] arr = s.toCharArray();
        boolean flag = false;
        for (int i = arr.length -1; i>= 0; i--) {
            if (arr[i] == '0') {
                arr[i] = '1';
                flag = true;
                break;
            } else {
                arr[i] = '0';
            }
        }
        if (!flag) {
            return "1" + new String(arr);
        }
        return new String(arr);
    }
    
    private String divide(String s) {
        return s.substring(0, s.length()-1);
    }
}
