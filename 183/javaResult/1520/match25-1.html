class Solution {
    public int numSteps(String s) {
        int n=s.length();
        char[] chars = s.toCharArray();
        int i=n-1;
        
int result=0;
        while(i>0){
            if(chars[i]=='0'){
                result++;
                i--;
            }else{
                result++;
                while(i>=0&&chars[i]=='1'){
                    result++;
                    i--;
                }
                if(i>0){
                    chars[i]='1';
                }
            }
        }
        return result;
    }
}
