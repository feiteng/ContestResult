class Solution {
    static Comparator cpt = new Comparator(){
        @Override
        public int compare(Object o1, Object o2){
            Pair a1 = (Pair)o1, a2 = (Pair)o2;
            return Integer.compare(a2.n, a1.n);
        }
    };
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<Pair> q = new PriorityQueue(cpt);
        int total = a + b + c, cc = 0;
        char last = ' ';
        if(a > 0)q.add(new Pair(a, 'a'));
        if(b > 0)q.add(new Pair(b, 'b'));
        if(c > 0)q.add(new Pair(c, 'c'));
        StringBuilder stb = new StringBuilder();
        while(!q.isEmpty()){
            Pair p = q.remove(), p2 = null;
            if(cc == 2 && last == p.c){
                if(q.isEmpty()) break;
                p2 = p;
                p = q.remove();
            }
            stb.append(p.c);
            p.n--;
            cc = (last == p.c) ? cc + 1 : 1;
            last = p.c;
            if(p.n > 0){
                q.add(p);
            }
            if(p2 != null){
                q.add(p2);
            }
        }
        return stb.toString();
    }
}

class Pair{
    Pair(int n, char c){
        this.n = n;
        this.c = c;
    }
    int n;
    char c;
}
