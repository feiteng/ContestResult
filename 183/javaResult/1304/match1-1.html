class Solution {
    public String longestDiverseString(int a, int b, int c) {
        int max = Math.max(a, Math.max(b, c));
        int total = a + b + c;
        if(max - 2 <= (total - max) * 2) {
            return build(a, b, c);
        }
        
        StringBuilder sb = new StringBuilder();
        while(a + b + c != 0){
            max = Math.max(a, Math.max(b, c));
            char temp = '0';
            if(a == max){
                sb.append("aa");
                a -= 2;
                if(b > c){
                    b--;
                    sb.append("b");
                }else{
                    if(c == 0) return sb.toString();
                    c--;
                    sb.append("c");
                }
            }else if(b == max){
                sb.append("bb");
                b -= 2;
                if(a > c){
                    a--;
                    sb.append("a");
                }else{
                    if(c == 0) return sb.toString();
                    c--;
                    sb.append("c");
                }
            }else{
                sb.append("cc");
                c -= 2;
                if(b > a){
                    b--;
                    sb.append("b");
                }else{
                    if(a == 0) return sb.toString();
                    a--;
                    sb.append("a");
                }
            }
        }
        return sb.toString();
    }
    
    private String build(int a, int b, int c){
        StringBuilder sb = new StringBuilder();
        while(a + b + c != 0){
            int max = Math.max(a, Math.max(b, c));
            char temp = '0';
            if(max == a) temp = 'a';
            else if(max == b) temp = 'b';
            else temp = 'c';
            
            if(temp == 'a'){
                if(sb.length() == 0 || (a >= 2 && sb.charAt(sb.length() - 1) != temp)){
                    sb.append("aa");
                    a -= 2;
                }
                else{
                    sb.append("a");
                    a--;
                }
                
                if(b > c){
                    sb.append("b");
                    b--;
                }else{
                    if(c == 0) return sb.toString();
                    else{
                        sb.append("c");
                        c--;
                    }
                }
            }else if(temp == 'b'){
                if(sb.length() == 0 || (b >= 2 && sb.charAt(sb.length() - 1) != temp)){
                    sb.append("bb");
                    b -= 2;
                }
                else{
                    sb.append("b");
                    b--;
                }
                
                if(a > c){
                    sb.append("a");
                    a--;
                }else{
                    if(c == 0) return sb.toString();
                    else{
                        sb.append("c");
                        c--;
                    }
                }
            }else{
                if(sb.length() == 0 || (c >= 2 && sb.charAt(sb.length() - 1) != temp)){
                    sb.append("cc");
                    c -= 2;
                }
                else{
                    sb.append("c");
                    c--;
                }
                
                if(b > a){
                    sb.append("b");
                    b--;
                }else{
                    if(a == 0) return sb.toString();
                    else{
                        sb.append("a");
                        a--;
                    }
                }
            }
        }
        return sb.toString();
    }
}
