class Solution {
    class Elem {
        int count;
        char c;
        
        Elem(int _count, char _c) {
            count = _count;
            c = _c;
        }
    }
    
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<Elem> pq = new PriorityQueue<>((ea,eb)->(eb.count - ea.count));
        if (a > 0) pq.offer(new Elem(a, 'a'));
        if (b > 0) pq.offer(new Elem(b, 'b'));
        if (c > 0) pq.offer(new Elem(c, 'c'));
        
        StringBuilder sb = new StringBuilder();
        while (pq.size() > 0) {
            //1:
            Elem e = pq.poll();
            if (e.count > 0) {
                sb.append(e.c);
                e.count -= 1;
            }
            if (e.count > 0) {
                sb.append(e.c);
                e.count -= 1;
            }
            
            //2:
            if (pq.size() == 0) {
                break;
            }
            if (e.count >= pq.peek().count) {
                Elem ee = pq.poll();
                sb.append(ee.c);
                ee.count -= 1;
                if (ee.count > 0) pq.offer(ee);
            }
            if (e.count > 0) pq.offer(e);
            
        }
        return sb.toString();
    }
}
