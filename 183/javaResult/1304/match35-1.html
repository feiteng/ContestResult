class Solution {
    public String longestDiverseString(int a, int b, int c) {
        int[] arr = new int[]{a, b, c};
        StringBuilder res = new StringBuilder();
        int prevIndex = -1;
        while (true) 
{
            int maxIndex = -1, maxCount = 0;
            for (int i = 0; i < arr.length; i++) {
                if (i != prevIndex && arr[i] > maxCount) {
                    maxIndex = i;
                    maxCount = arr[i];
                }
            }
            if (maxIndex < 0 || arr[maxIndex] <= 0) {
                break;
            }
            int count = arr[maxIndex];
            if (count == 1 || (prevIndex >= 0 && arr[maxIndex] < arr[prevIndex])) {
                res.append((char)('a' + maxIndex));
                arr[maxIndex]--;
            } else {
                res.append((char)('a' + maxIndex)).append((char)('a' + maxIndex));
                arr[maxIndex] -= 2;
            }
            prevIndex = maxIndex;
        }
        return res.toString();
    }
}
