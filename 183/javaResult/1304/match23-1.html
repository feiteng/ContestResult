class Solution {
    public String longestDiverseString(int a, int b, int c) {
        
PriorityQueue<int[]> pq = new PriorityQueue<>((x, y) -> y[0] - x[0]);  // int[2]  0 - count, 1 - digit
        pq.offer(new int[]{a, 0});
        pq.offer(new int[]{b, 1});
        pq.offer(new int[]{c, 2});
        
        int prev = -1;
        StringBuilder sb = new StringBuilder();
        while (!pq.isEmpty()) 
{
            int[] cur = pq.poll();
            if (cur[0] == 0) {
                break;
            }
            if (cur[1] == prev) {
                int[] cur2 = pq.poll();
                if (cur2[0] == 0) {
                    break;
                } else {
                    sb.append((char)(cur2[1]+'a'));
                    prev = cur2[1];
                    cur2[0]--;
                    pq.offer(cur2);
                    pq.offer(cur);
                }
            } else {
                int count = 0;
                while (cur[0] > 0 && count < 2) {
                    sb.append((char)(cur[1]+'a'));
                    prev = cur[1];
                    cur[0]--;
                    count++;
                }
                pq.offer(cur);
            
}
        }
        return sb.toString();
    }
}
