class Solution {
    public String longestDiverseString(int a, int b, int c) {
        Queue<Node> pq = new PriorityQueue<>(Comparator.comparing(Node::getCount).reversed());
        if (a > 0) {
            pq.add(new Node('a', a));
        }
        if (b > 0) {
            pq.add(new Node('b', b));
        }
        if (c > 0) {
            pq.add(new Node('c', c));
        }
        StringBuilder sb = new StringBuilder();
        while (!pq.isEmpty()) {
            Node node = pq.remove();
            int len = sb.length();
            if (len < 2 || sb.charAt(len - 1) != node.getLetter() || sb.charAt(len - 2) != node.getLetter()) {
                sb.append(node.getLetter());
                node.setCount(node.getCount() - 1);
                if (node.getCount() > 0) {
                    pq.add(node);
                }
            } else if (!pq.isEmpty()) {
                Node node2 = pq.remove();
                sb.append(node2.getLetter());
                node2.setCount(node2.getCount() - 1);
                if (node2.getCount() > 0) {
                    pq.add(node2);
                }
                pq.add(node);
            }
        }
        return sb.toString();
    }

    class Node {
        char letter;
        int count;

        public Node(char letter, int count) {
            this.letter = letter;
            this.count = count;
        }

        public char getLetter() {
            return letter;
        }

        public void setLetter(char letter) {
            this.letter = letter;
        }

        public int getCount() {
            return count;
        }

        public void setCount(int count) {
            this.count = count;
        }
    }
}
