class Solution {
    public String longestDiverseString(int a, int b, int c) {
        String res = "";
        
PriorityQueue<int[]> pq = new PriorityQueue<int[]>((A,B)-> B[0]-A[0]);
        if(a>0)pq.offer(new int[]{a,0});
        if(b>0)pq.offer(new int[]{b,1});
        if(c>0)pq.offer(new int[]{c,2});
        boolean add = true;
        
while(pq.size()>0 && add){
            int[] cur = pq.poll();
            char ch = (char)('a'+cur[1]);
            int size = res.length();
            if(size<2||!(res.charAt(size-1)==ch && res.charAt(size-2)==ch)){
                cur[0]--;
                res += ch;
                if(cur[0]>0){
                    pq.offer(cur);
                }
                add = true;
            }else{
                if(pq.size()>0){
                    add = true;
                    int[] next = pq.poll();
                    char nch = (char)(next[1]+'a');
                    res += nch;
                    next[0]--;
                    if(next[0]>0){
                        pq.offer(next);
                    }
                }else{
                    add = false;
                }
                pq.offer(cur);
            }
        }
        
        return res;
    }
}
