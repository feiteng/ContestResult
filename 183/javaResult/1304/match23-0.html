class Solution {
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb=new StringBuilder();
        //0=a,1=b,2=c
        int prefix=-1;
        int prefixAmount=0;
        PriorityQueue<int[]>priorityQueue=new PriorityQueue<>((o1,o2)->o2[0]-o1[0]);
        priorityQueue.add(new int[]{a,0});
        priorityQueue.add(new int[]{b,1});
        priorityQueue.add(new int[]{c,2});
        while (true){
            int[] maxChar=priorityQueue.poll();
            if(maxChar[0]==0){
                break;
            }
            if(maxChar[1]!=prefix){
                prefix=maxChar[1];
                prefixAmount=1;
                sb.append((char)('a'+maxChar[1]));
                priorityQueue.add(new int[]{maxChar[0]-1,maxChar[1]});
            }else {
                if(prefixAmount<=1){
                    prefixAmount++;
                    sb.append((char)('a'+maxChar[1]));
                    priorityQueue.add(new int[]{maxChar[0]-1,maxChar[1]});
                }else {
                    int[] midChar=priorityQueue.poll();
                    if(midChar[0]==0){
                        break;
                    }
                    prefix=midChar[1];
                    prefixAmount=1;
                    sb.append((char)('a'+midChar[1]));
                    priorityQueue.add(maxChar);
                    priorityQueue.add(new int[]{midChar[0]-1,midChar[1]});
                }
            }
        }
        return sb.toString();
    }
}
