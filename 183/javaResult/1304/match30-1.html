class Solution {
    
    public String longestDiverseString(int a, int b, int c) {
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((i, j) -> (j[1] - i[1]));
        
        
if(a > 0) pq.offer(new int[]{0, a});
        if(b > 0) pq.offer(new int[]{1, b});
        if(c > 0) pq.offer(new int[]{2, c});
        
        StringBuilder sb = new StringBuilder();
        
        while(!pq.isEmpty()) {
            
            List<int[]> list = new ArrayList<>();
            
            int[] p1 = pq.poll();
            if(sb.length() < 2 || sb.charAt(sb.length() - 1) != (char)('a' + p1[0]) 
               || sb.charAt(sb.length() - 2) != (char)('a' + p1[0])) {
                sb.append((char)('a' + p1[0]));
                p1[1]--;
                if(p1[1] > 0) pq.offer(p1);
            } else {
                if(pq.isEmpty()) break;
                
                int[] p2 = pq.poll();
                sb.append((char)('a' + p2[0]));
                p2[1]--;
                if(p2[1] > 0) pq.offer(p2);
                
pq.offer(p1);
            }
            
        }
        return sb.toString();
    }
}
