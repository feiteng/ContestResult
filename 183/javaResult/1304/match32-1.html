class Solution {
    public String longestDiverseString(int a, int b, int c) {
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((m, n) -> (n[1] - m[1]));
        
        if (a > 0) pq.offer(new int[]{'a', a});
        if (b > 0) pq.offer(new int[]{'b', b});
        if (c > 0) pq.offer(new int[]{'c', c});
        
        StringBuilder sb = new StringBuilder();
        
        while (pq.size() >= 2) {
            int[] arr1 = pq.poll(), arr2 = pq.poll();
            
            if (sb.length() == 0 || arr1[0] != sb.charAt(sb.length() - 1)) {
                if (arr1[1] >= 2) {
                    sb.append("" + (char)arr1[0]).append("" + (char)arr1[0]);
                    if (arr1[1] - 2 > 0)
                        pq.offer(new int[]{arr1[0], arr1[1] - 2});
                }
                else {
                    sb.append("" + (char)arr1[0]);
                    if (arr1[1] - 1 > 0)
                        pq.offer(new int[]{arr1[0], arr1[1] - 1});
                }
            }
            if (arr1[1] - arr2[1] <= 1) {
                if (arr2[1] >= 2) {
                    sb.append("" + (char)arr2[0]).append("" + (char)arr2[0]);
                    if (arr2[1] - 2 > 0)
                        pq.offer(new int[]{arr2[0], arr2[1] - 2});
                }
                else {
                    sb.append("" + (char)arr2[0]);
                    if (arr2[1] - 1 > 0)
                        pq.offer(new int[]{arr2[0], arr2[1] - 1});
                }
            } else {
                sb.append("" + (char)arr2[0]);
                if (arr2[1] - 1 > 0)
                    pq.offer(new int[]{arr2[0], arr2[1] - 1});
            }
        }
        
        if (pq.size() == 1 && pq.peek()[0] != sb.charAt(sb.length() - 1)) {
            int[] arr1 = pq.poll();
            if (arr1[1] >= 2) 
                sb.append("" + (char)arr1[0]).append("" + (char)arr1[0]);
            else
                sb.append("" + (char)arr1[0]);
        }
        
        return sb.toString();
    }
}
