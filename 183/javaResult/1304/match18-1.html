class Solution {
    public String longestDiverseString(int a, int b, int c) {
        
PriorityQueue<int[]> heap = new PriorityQueue<>((x,y)->(y[1] - x[1]));
        if (a > 0) heap.offer(new int[]{0, a});
        if (b > 0) heap.offer(new int[]{1, b});
        if (c > 0) heap.offer(new int[]{2, c});
        StringBuilder sb = new StringBuilder();
        while (!heap.isEmpty()) {
            int[] p = heap.poll();
            int n = sb.length();
            if (n >= 2 && sb.charAt(n-1) == (char)(p[0] + 'a') && sb.charAt(n-2) == (char)(p[0] + 'a')) {
                if (heap.isEmpty()) break;
                else {
                    int[] q = heap.poll();
                    sb.append((char)(q[0] + 'a'));
                    q[1]--;
                    if (q[1] > 0)
                        heap.offer(q);
                }
            }
            sb.append((char)(p[0] + 'a'));
            p[1]--;
            
if (p[1] > 0)
                heap.offer(p);
        }
        return sb.toString();
    }
}
