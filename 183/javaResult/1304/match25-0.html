class Solution {
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb = new StringBuilder();
        PriorityQueue<Data> pq = new PriorityQueue<>();
        if(a!=0){
            pq.add(new Data('a', a));
        }
        if(b!=0){
            pq.add(new Data('b',b));
        }
        if(c!=0){
            pq.add(new Data('c',c));
        }
        while(!pq.isEmpty()){
            Data top = pq.poll();
            sb.append(top.x);
            top.cnt = top.cnt - 1;
            if(top.cnt > 0){
                sb.append(top.x);
                top.cnt = top.cnt - 1;
            }
            if(pq.isEmpty())
                break;
            Data next = pq.poll();
            sb.append(next.x);
            next.cnt = next.cnt - 1;
            if(next.cnt!=0 && next.cnt== top.cnt){
                sb.append(next.x);
                next.cnt = next.cnt - 1;
            }
            if(top.cnt!=0)
                pq.add(top);
            if(next.cnt!=0)
                pq.add(next);  
        }
        return sb.toString();
    }
    
    
}

class Data implements Comparable<Data>{
    char x;
    int cnt;
    Data(char x, int cnt){
        this.x = x;
        this.cnt = cnt;
    }
    public int compareTo(Data b){
        return Integer.compare(b.cnt, this.cnt);
    }
}
