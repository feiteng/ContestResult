class Ele {
    char letter;
    int count;
    public Ele(char l, int c) {
        letter = l;
        count = c;
    }
}
class Solution {
    
public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<Ele> order = new PriorityQueue<>((e1, e2) -> e2.count - e1.count);
        if (a > 0)
            order.offer(new Ele('a', a));
        if (b > 0)
            order.offer(new Ele('b', b));
        if (c > 0)
            order.offer(new Ele('c', c));
        
        StringBuilder res = new StringBuilder();
        while (!order.isEmpty()) {
            Ele largest = order.poll();
            if (dup(res, largest.letter)) {
                if (order.isEmpty()) return res.toString();
                Ele sec = order.poll();
                res.append(sec.letter);
                if (sec.count > 1) {
                    order.offer(new Ele(sec.letter, sec.count - 1));
                }
                order.offer(new Ele(largest.letter, largest.count));
                
            } else {
                res.append(largest.letter);
                if (largest.count > 1) {
                    order.offer(new Ele(largest.letter, largest.count - 1));
                }
            }
        }
        return res.toString();
    }
    
    private boolean dup(StringBuilder sb, char c) {
        return sb.length() > 1 && sb.charAt(sb.length() - 1) == c && sb.charAt(sb.length() - 2) == c;
    }
}
