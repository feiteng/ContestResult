class Solution {
    
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb = new StringBuilder();
        char last = '0';
        boolean valid = true;
        if(a == 0 && b == 0 && c == 0) { valid = false; }
        while(valid) {
            if(last == 'a') {
                if(b == 0 && c == 0) { valid = false; break; }
                if(b >= c) {
                    if(b >= 2 && b >= 2 * (a + c)) { 
sb.append("bb"); b -= 2; }
                    else { sb.append("b"); b--; }
                    last = 'b';
                }
                else {
                    if(c >= 2 && c >= 2 * (a + b)) { sb.append("cc"); c -= 2; }
                    else { sb.append("c"); c--; }
                    last = 'c';
                }
            }
            else if(last == 'b') {
                if(a == 0 && c == 0) { valid = false; break; }
                if(a >= c) {
                    if(a >= 2 && a >= 2 * (b + c)) { sb.append("aa"); a -= 2; }
                    else { sb.append("a"); a--; }
                    last = 'a';
                }
                else {
                    if(c >= 2 && c >= 2 * (a + b)) { sb.append("cc"); c -= 2; }
                    else { sb.append("c"); c--; }
                    last = 'c';
                }
            }
            else if(last == 'c') {
                if(a == 0 && b == 0) { valid = false; break; }
                if(a >= b) {
                    if(a >= 2 && a >= 2 * (b + c)) { sb.append("aa"); a -= 2; }
                    else { sb.append("a"); a--; }
                    last = 'a';
                }
                else {
                    if(b >= 2 && b >= 2 * (a + c)) { sb.append("bb"); b -= 2; }
                    else { sb.append("b"); b--; }
                    last = 'b';
                }
            }
            else {
                if(a >= b && a >= c) {
                    if(a >= 2 && a >= 2 * (b + c)) { 
sb.append("aa"); a -= 2; }
                    else { sb.append("a"); a--; }
                    last = 'a';
                }
                else if(b >= a && b >= c) {
                    if(b >= 2 && b >= 2 * (a + c)) { sb.append("bb"); b -= 2; 
}
                    else { sb.append("b"); b--; }
                    last = 'b';
                }
                else if(c >= a && c >= b) {
                    if(c >= 2 && c >= 2 * (a + b)) { sb.append("cc"); c -= 2; }
                    else { sb.append("c"); c--; }
                    
last = 'c';
                }
            }
        }
        return sb.toString();
    }

}
