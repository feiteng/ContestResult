import java.util.PriorityQueue;

class Solution {
  public static void main(String args[]) {
    Solution sol = new Solution();
    System.out.println(sol.longestDiverseString(2, 2, 1));
  }
  public String longestDiverseString(int a, int b, int c) {
    StringBuilder sb = new StringBuilder();
    PriorityQueue<CharCount> pq = new PriorityQueue<>(3, (c1, c2) -> c2.count - c1.count);
    if (a > 0) {
      pq.add(new CharCount('a', a));
    }
    if (b > 0) {
      pq.add(new CharCount('b', b));
    }
    if (c > 0) {
      pq.add(new CharCount('c', c));
    }
   
    int count = 0;
    char prevChar = 'z';
    while (!pq.isEmpty()) {
      CharCount next = pq.poll();
      if (next.ch == prevChar && count >= 2) {
        // potential termination.
        if (pq.isEmpty()) {
          return sb.toString();
        }
        CharCount temp = pq.poll();
        pq.offer(next);
        next = temp;
      }
      
      if (prevChar == next.ch) {
        count++;
      } else {
        count = 1;
        prevChar = next.ch;
      }
      next.count--;
      if (next.count > 0) {
        pq.offer(next);
      }
      sb.append(next.ch);
    }
    return sb.toString();
  }
  
  static class CharCount {
    char ch;
    int count;
    
    public CharCount(char ch, int count) {
      this.ch = ch;
      this.count = count;
    }
  }
}
