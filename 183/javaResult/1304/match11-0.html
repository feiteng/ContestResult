class Solution {
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((m, n) -> (n[1] - m[1]));
        if (a > 0) pq.add(new int[]{0, a});
        if (b > 0) pq.add(new int[]{1, b});
        if (c > 0) pq.add(new int[]{2, c});
        StringBuilder sb = new StringBuilder();
        boolean[] sat = new boolean[3];
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            char f = (char)('a' + cur[0]);
            int len = sb.length();
            if (len >= 2 && sb.charAt(len - 1) == f && sb.charAt(len - 2) == f ) {
                if (pq.isEmpty()) {
                    return sb.toString();
                }
                int[] next = pq.poll();
                sat[next[0]] = true;
                pq.add(cur);
                char n = (char)('a' + next[0]);
                sb.append(n);
                if (--next[1] > 0)  pq.add(next);
            } else {
                sb.append(f);
                if (--cur[1] > 0)  pq.add(cur);
                sat[cur[0]] = true;
            }
        }
        return sb.toString();
    }
}
