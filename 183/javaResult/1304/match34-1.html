class Solution {
        public String longestDiverseString(int a, int b, int c) 
{
            if (a > b && a > c) {
                return biggest('a', a, 'b', b, 'c', c);
            } else if (b > c) {
                return biggest('b', b, 'a', a, 'c', c);
            
}
            return biggest('c', c, 'b', b, 'a', a);
        }
        public String biggest(char biggest_char, int biggest_int, char x_c, int x, char y_c, int y) {
            StringBuilder str = new StringBuilder();
            while (biggest_int > x + y && x + y > 0) {
                str.append(biggest_char);
                str.append(biggest_char);
                biggest_int -= 2;
                
if (x > y) {
                    str.append(x_c);
                    x--;
                } else {
                    str.append(y_c);
                    y--;
                }
            }
            if (x+y == 0) {
                for (int i = 0; i < 2 && i < biggest_int; i++) {
                    str.append(biggest_char);
                }
                return str.toString();
            }
            int biggest_turns = 0;
            while (biggest_int + x + y > 0) 
{
                if (biggest_turns != 2 && biggest_int > x && biggest_int > y) {
                    str.append(biggest_char);
                    biggest_int--;
                    biggest_turns++;
                } else if (x > y) {
                    str.append(x_c);
                    x--;
                    biggest_turns = 0;
                } else {
                    str.append(y_c);
                    y--;
                    biggest_turns = 0;
                }
            }
            return str.toString();
        }
    }
