class Solution {
    public String longestDiverseString(int a, int b, int c) {
       StringBuilder res=new StringBuilder();
       PriorityQueue<int[]> pq=new PriorityQueue<int[]>((x,y)->y[1]-x[1]);
        if(a>0)
        pq.add(new int[]{0,a});
        if(b>0)
        pq.add(new int[]{1,b});
        if(c>0)
        pq.add(new int[]{2,c});
        int prev=-1;
        int count=0;
        while(!pq.isEmpty()){
            int[] val=pq.poll();
            if(count==2 && val[0]==prev){
                if(pq.isEmpty())
                    return res.toString();
                int[] otherVal=pq.poll();
                res.append((char)(otherVal[0]+'a'));
                prev=otherVal[0];
                count=1;
                otherVal[1]--;
                if(otherVal[1]!=0)
                    pq.add(otherVal);
            }
            
            
            res.append((char)(val[0]+'a'));
            if(prev!=val[0]){
                prev=val[0];
                count=1;
            }else
                count++;
            val[1]--;
            if(val[1]!=0)
                pq.add(val);
        }
        return res.toString();
    }
}
