class Solution {
    public String longestDiverseString(int a, int b, int c) {
        int[] ct = {a, b, c};
        if (a > (b + c + 1) * 2) ct[0] = (b + c + 1) * 2;
        if (b > (a + c + 1) * 2) ct[1] = (a + c + 1) * 2;
        if (c > (a + b + 1) * 2) ct[2] = (a + b + 1) * 2;
        return build(ct);
    }
    
    public String build(int[] ct) {
        StringBuilder sb = new StringBuilder();
        int len = ct[0] + ct[1] + ct[2];
        char[] arr = {'a', 'b', 'c'};

        while (sb.length() < len) {
            int n = sb.length();
            if (sb.length() >= 2 && sb.charAt(n-1) == sb.charAt(n-2)) {
                char last = sb.charAt(n-1);
                int max = -1;
                char cur = 'd';
                for (int i = 0; i < 3; i++) {
                    if (arr[i] == last) continue;
                    if (max == -1) {
                        cur = arr[i];
                        max = i;
                    } else {
                        if (ct[i] > ct[max]) {
                            max = i;
                            cur = arr[i];
                        }
                    }
                }
                sb.append(cur);
                ct[max]--;
            } else {
                int max = 0;
                char cur = 'a';
                for (int i = 1; i < 3; i++) {
                    if (ct[i] > ct[max]) {
                        max = i;
                        cur = arr[i];
                    }
                }
                sb.append(cur);
                ct[max]--;
            }
        }
        return sb.toString();
    }
}
