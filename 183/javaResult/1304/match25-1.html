class Solution {
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<MultiCharacter> pq = new PriorityQueue<>((x, y) -> -Integer.compare(x.count, y.count));
        
        
if (a > 0) {
            pq.offer(new MultiCharacter('a', a));
        }
        if (b > 0) {
            pq.offer(new MultiCharacter('b', b));
        }
        if (c > 0) {
            pq.offer(new MultiCharacter('c', c));
        }
        StringBuilder res = new StringBuilder();
        
        char lastChar = '*';
        int lastCharCnt = 1;
        while (!pq.isEmpty()) {
            MultiCharacter cand1 = pq.poll();
            
            if (lastChar == cand1.ch) {
                ++lastCharCnt;
            }
            
            if (lastCharCnt == 3) {
                if (pq.isEmpty()) break;
                MultiCharacter cand2 = pq.poll();
                res.append(cand2.ch);
                if (--cand2.count > 0) {
                    pq.offer(cand2);
                }
                pq.offer(cand1);
                lastChar = cand2.ch;
                lastCharCnt = 1;
            } else {
                res.append(cand1.ch);
                if (--cand1.count > 0) {
                    pq.offer(cand1);
                }
                lastChar = cand1.ch;
            }
        }
        
        return res.toString();
    }
    
    
static class MultiCharacter {
        public char ch;
        public int count;
        
        public MultiCharacter(char ch, int count) {
            this.ch = ch;
            this.count = count;
        }
    }
}
