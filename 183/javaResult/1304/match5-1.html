class Solution {
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<Info> pq = new PriorityQueue<>((k,t) -> t.num - k.num);
        
if(a != 0) {
            pq.add(new Info('a', a));
        }
        if(b != 0) {
            pq.add(new Info('b', b));
        }
        if(c != 0) {
            pq.add(new Info('c', c));
        }

        StringBuilder sb = new StringBuilder();
        while(pq.size() > 0) {
            Info first = pq.poll();
            int len = sb.length();
            if(len < 2 || sb.charAt(len - 1) != first.ch || sb.charAt(len - 2) != first.ch) {
                sb.append(first.ch);
                if(first.num != 1)
                    pq.add(new Info(first.ch, first.num - 1));
            } else {
                if(pq.size() == 0)
                    break;
                Info second = pq.poll();
                sb.append(second.ch);
                if(second.num != 1)
                    pq.add(new Info(second.ch, second.num - 1));
                pq.add(first);
            
}
        }
        return sb.toString();
    }
}

class Info {
    char ch;
    int num;

    public Info(char ch, int num) {
        this.ch = ch;
        this.num = num;
    }
}
