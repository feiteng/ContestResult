class Solution {
    private static class Pair {
        char c;
        int count;
        
        Pair(char c, int count){
            this.c = c;
            this.count = count;
        }
        
        @Override
        public String toString() {
            return "[" + c + ", " + count + "]";
        }
    
}
    
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<Pair> pq = new PriorityQueue<>((q,w) -> w.count - q.count);
        if(a > 0) {
            pq.add(new Pair('a', a));
        }
        
        if(b > 0) {
            pq.add(new Pair('b', b));
        }
         
        if(c > 0) {
            pq.add(new Pair('c', c));
        }
        
        StringBuilder sb = new StringBuilder();
        
        while(!pq.isEmpty()) {
            Pair highest = pq.poll();
            int highAdd = Math.min(highest.count, 2);
            
            for(int i = 0; i < highAdd; i++) {
                sb.append(highest.c);
            }
            
            
if(pq.isEmpty()) {
                return sb.toString();
            }
            
            Pair second = pq.poll();
            int secondAdd = 1;
            
            if(second.count > highest.count - 2) {
                secondAdd = Math.min(second.count, 2);
            }
            
            for(int i = 0; i < secondAdd; i++) {
                sb.append(second.c);
            }
            
            if(highest.count - highAdd > 0) {
                pq.add(new Pair(highest.c, highest.count - highAdd));
            }
            
            if(second.count - secondAdd > 0) {
                pq.add(new Pair(second.c, second.count - secondAdd));
            
}
            
            //System.out.println(highest + " " + second);
            //System.out.println(sb.toString());
        }
        
        return sb.toString();
    }
}
