class Solution {
    class pair {
        char c;
        int n;
        public pair(char c, int n) {
            this.c = c;
            this.n = n;
        }
    }
    public String longestDiverseString(int a, int b, int c) {
        PriorityQueue<pair> queue = new PriorityQueue<>((cc, dd) -> (dd.n == cc.n ? cc.c - dd.c : dd.n - cc.n));
        int[] counts = new int[]{a, b, c};
        for (int i = 0; i < 3; i++) {
            if (counts[i] == 0) {
                continue;
            }    
            pair p = new pair((char) ('a' + i), counts[i]);
            queue.offer(p);
        }
        StringBuilder sb = new StringBuilder();
        while (!queue.isEmpty()) {
            pair p = queue.poll();
            if (sb.length() >= 2 && sb.charAt(sb.length() - 1) == p.c && sb.charAt(sb.length() - 2) == p.c) {
                if (queue.isEmpty()) {
                    return sb.toString();
                }
                pair pp = queue.poll();
                sb.append(pp.c);
                pp.n--;
                if (pp.n > 0) {
                    queue.offer(pp);   
                }
                queue.offer(p);
            } else {
                sb.append(p.c);
                p.n--;
                if (p.n > 0) {
                    queue.offer(p);   
                }
            }
        }
        return sb.toString();
    }
}
