class Solution {
    public String longestDiverseString(int a, int b, int c) {
        Queue<int[]> q = new PriorityQueue<>((u, v)->(v[0] - u[0]));
        if(a > 0) q.offer(new int[]{a, 0});
        if(b > 0) q.offer(new int[]{b, 1});
        if(c > 0) q.offer(new int[]{c, 2});
        
        String str = "";
        
        while(!q.isEmpty())
        {
            int[] cur = q.poll();
            if(cur[0] == 0) continue;
            if(str.length() >= 2)
            {
                int n = str.length();
                int p1 = str.charAt(n - 1) - 'a', p2 = str.charAt(n - 2) - 'a';
                if(p1 == p2 && p1 == cur[1])
                {
                    if(q.isEmpty()) return str;
                    int[] next = q.poll();
                    str += (char)('a' + next[1]);
                    next[0]--;
                    if(next[0] > 0) q.offer(next);
                    q.offer(cur);
                }
                else
                {
                    str += (char)('a' + cur[1]);    
                    cur[0]--;
                    if(cur[0] > 0) q.offer(cur);
                }
            }
            else
            {
                str += (char)('a' + cur[1]);    
                cur[0]--;
                if(cur[0] > 0) q.offer(cur);
            }

        }
        
        return str;
    }
}
