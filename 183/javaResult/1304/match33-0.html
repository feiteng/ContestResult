class Solution {
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb = new StringBuilder();
        if (a > 2 * (b + c + 1)) {
            a = 2 * (b + c + 1);
        } else if (b > 2 * (a + c + 1)) {
            b = 2 * (a + c + 1);
        } else if (c > 2 * (a + b + 1)) {
            c = 2 * (a + b + 1);
        }
        char last = '0';
        while (a >= 0 && b >= 0 && c >= 0 && a + b + c != 0) {
            if (a > b && a > c && last != 'a') {
                if (a > 1) {
                    sb.append('a');
                    sb.append('a');
                    a -= 2;
                } else {
                    sb.append('a');
                    --a;
                }
                last = 'a';
            } else if (b > a && b > c && last != 'b') {
                if (b > 1) {
                    sb.append('b');
                    sb.append('b');
                    b -= 2;
                } else {
                    sb.append('b');
                    --b;
                }
                last = 'b';
            } else if (c > a && c > b && last != 'c') {
                if (c > 1) {
                    sb.append('c');
                    sb.append('c');
                    c -= 2;
                } else {
                    sb.append('c');
                    --c;
                }
                last = 'c';
            } else {
                if (last != 'a' && a > 0) {
                    sb.append('a');
                    --a;
                    last = 'a';
                } else if (last != 'b' && b > 0) {
                    sb.append('b');
                    --b;
                    last = 'b';
                } else if (last != 'c' && c > 0) {
                    sb.append('c');
                    --c;
                    last = 'c';
                }
            }
        }
        return new String(sb);
    }
    
}
