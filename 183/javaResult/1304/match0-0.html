class Solution {
    
    public String longestDiverseString(int a, int b, int c) {
        StringBuilder sb = new StringBuilder();
        PriorityQueue<int[]> pq = new PriorityQueue<>((x, y) -> y[1] - x[1]);
        Queue<int[]> queue = new LinkedList<>();
        if (a > 0) pq.offer(new int[]{0, a});
        if (b > 0) pq.offer(new int[]{1, b});
        if (c > 0) pq.offer(new int[]{2, c});
        // step1: construct
        while (pq.size() > 0) {
            // append new chars
            int[] cur = pq.poll();
            if (cur[1] >= 1) {
                sb.append(helper(cur[0]));
                cur[1] -= 1;
            }
            // update
            if (queue.size() > 0) {
                pq.offer(queue.poll());
            }
            if (cur[1] > 0) {
                queue.offer(cur);
            }
        }
        // step2: insert
        int[] count = new int[3];
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            count[cur[0]] = cur[1];
        }
        while (!queue.isEmpty()) {
            int[] cur = queue.poll();
            count[cur[0]] = cur[1];
        }
        int i = 0;
        while (i < sb.length()) {
            if (count[sb.charAt(i) - 'a'] > 0) {
                count[sb.charAt(i) - 'a']--;
                sb.insert(i, sb.charAt(i));
                i++;
            }
            i++;
        }
        return sb.toString();
    }
    
    private char helper(int i) {
        if (i == 0) return 'a';
        if (i == 1) return 'b';
        return 'c';
    }
}
