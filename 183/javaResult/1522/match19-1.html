class Solution {
    public String stoneGameIII(int[] a) {
        int n = a.length;
        int[] f = new int[n+1];
        for(int i=n-1;i>=0;i--){
            f[i] = a[i] - f[i+1];
            
if(i+2<=n){
                f[i] = Math.max(f[i],a[i] + a[i+1] - f[i+2]);
            }
            if(i+3<=n){
                f[i] = Math.max(f[i],a[i] + a[i+1]+ a[i+2] - f[i+3]);
            }
        
}
        if(f[0]==0) return "Tie";
        if(f[0]>0) return "Alice";
        return "Bob";
    }
}
