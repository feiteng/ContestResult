class Solution {
    
    int n;
    int[] memo;
    
    public String stoneGameIII(int[] stones) {
        n = stones.length;
        memo = new int[n];
        Arrays.fill(memo, Integer.MIN_VALUE);
        int sum = 0;
        for (int s : stones) {
            sum += s;
        }
        int alice = check(stones, 0, sum);
        int bob = sum - alice;
        //System.out.println(String.format("%s %s %s", sum, alice, bob));
        return alice > bob ? "Alice" : (
alice == bob ? "Tie" : "Bob");
    }
    
    private int check(int[] stones, int cur, int remaining) {
        int res = Integer.MIN_VALUE;
        if (cur == n) {
            return 0;
        }
        if (memo[cur] != Integer.MIN_VALUE) {
            return memo[cur];
        }
        int sum = 0;
        
for (int i = cur; i < Math.min(n, cur + 3); i++) {
            sum += stones[i];
            res = Math.max(res, remaining - check(stones, i + 1, remaining - sum));
        }
        memo[cur] = res;
        return res;
    }
}
