class Solution {
    public String stoneGameIII(int[] stoneValue) {
    int sum = 0;
    for (int v : stoneValue) {
      sum += v;
    }

    String res = "";
    int alice = findMax(stoneValue);
    if (2 * alice > sum) {
      res = "Alice";
    } else if (2 * alice < sum) {
      res = "Bob";
    } else {
      res = "Tie";
    }
    return res;
  }

  int[] dp;
  int[] stones;
  int[] pref;
  int n;
  int findMax(int[] stones) {
    this.n = stones.length;
    this.stones = stones;
    pref = new int[1 + n];
    dp = new int[1 + n];
    for (int i = 1; i <= n; i++) {
      pref[i] = pref[i - 1] + stones[i - 1];
    }
    Arrays.fill(dp, Integer.MIN_VALUE);
    return rec(0);
  }

  int rec(int start) {
    if (start >= n) {
      return 0;
    }
    if (dp[start] != Integer.MIN_VALUE) {
      return dp[start];
    }
    int res = Integer.MIN_VALUE;
    for (int i = 0; i < 3; i++) {
      // take [start, start + i]
      int sum = pref[Math.min(n, start + i + 1)] - pref[start];
      int min = Integer.MAX_VALUE;
      for (int j = 0; j < 3; j++) {
        // oppo take [start + i + 1, start + i + 1 + j]
        min = Math.min(min, rec(start + i + j + 2));
      }
      res = Math.max(res, sum + min);
    }
    dp[start] = res;
    return res;
  }
}
