class Solution {
    public String stoneGameIII(int[] stoneValue) {
		int INF = 1 << 30;
		int n = stoneValue.length;

		int sum = 0;
		int[] dp = new int[n + 1];

		for (int i = n - 1; i >= 0; i--) {
			sum += stoneValue[i];
			dp[i] = -INF;
			if (i + 1 <= n) {
				dp[i] = Math.max(dp[i], sum - dp[i + 1]);
			}
			if (i + 2 <= n) {
				dp[i] = Math.max(dp[i], sum - dp[i + 2]);
			}
			if (i + 3 <= n) {
				dp[i] = Math.max(dp[i], sum - dp[i + 3]);
			}
		}

		int alice = dp[0];
		int bob = sum - dp[0];

		if (alice > bob) {
			return "Alice";
		} else if (bob > alice) {
			return "Bob";
		} else {
			return "Tie";
		}
	}
}
