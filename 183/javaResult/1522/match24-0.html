class Solution {
    int[] arr;
    HashMap<Integer, Integer> memo;
    public String stoneGameIII(int[] stoneValue) {
        
        int total = 0;
        memo = new HashMap<>();
        // for (int n: stoneValue) total+=n;
        arr = new int[stoneValue.length];
        total += stoneValue[stoneValue.length-1];
        arr[stoneValue.length-1] = total;
        for (int i= stoneValue.length-2;i>=0;i--){
            arr[i] = arr[i+1] + stoneValue[i];
            total += stoneValue[i];
        }
        
        int aliceScore = get(stoneValue, 0, 0);
        // System.out.println(aliceScore);
        // System.out.println(total);
        // System.out.println(Arrays.toString(arr));
        
        if (aliceScore > total-aliceScore) return "Alice";
        else if (aliceScore < total- aliceScore) return "Bob";
        return "Tie";
    }
    
    private int get(int[] a, int ind, int cur){
        if (ind==a.length) return cur;
        if (memo.containsKey(ind)) return cur+memo.get(ind);
        int curTotal = arr[ind];
        
        int res = Integer.MAX_VALUE;
        int accu=0;
        for (int i=ind;i<Math.min(a.length, ind+3);i++){
            accu = a[ind];
            res = Math.min(res, get(a, i+1, 0));
        }
        memo.put(ind, curTotal-res);
        return cur + curTotal-res;
    }
}
