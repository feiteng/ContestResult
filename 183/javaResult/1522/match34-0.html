class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int[] sum = new int[stoneValue.length + 1];
        for (int i = stoneValue.length - 1; i >= 0; i--) {
            sum[i] = sum[i + 1] + stoneValue[i];
        }
        int a = max(stoneValue, sum, 0, new HashMap<>());
        int b = sum[0] - a;
        if (a > b) {
            return "Alice";
        } else if (a == b) {
            return "Tie";
        } else {
            return "Bob";
        }
    }

    private int max(int[] stoneValue, int[] sum, int start, Map<Integer, Integer> cache) {
        if (start >= stoneValue.length) {
            return 0;
        }
        if (cache.containsKey(start)) {
            return cache.get(start);
        }
        int value = stoneValue[start] + sum[start + 1] - max(stoneValue, sum, start + 1, cache);
        if (start + 1 < stoneValue.length) {
            value = Math.max(value, stoneValue[start] + stoneValue[start + 1] + sum[start + 2] - max(stoneValue, sum, start + 2, cache));
        }
        if (start + 2 < stoneValue.length) {
            value = Math.max(value, stoneValue[start] + stoneValue[start + 1] + stoneValue[start + 2] + sum[start + 3] - max(stoneValue, sum, start + 3, cache));
        }
        cache.put(start, value);
        return value;
    }
}
