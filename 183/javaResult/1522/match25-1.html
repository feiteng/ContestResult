class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int[] dp = new int[stoneValue.length];
        Arrays.fill(dp, Integer.MIN_VALUE);
        
int result = solve(stoneValue, 0, dp);
        // System.out.println(result);
        if(result == 0) return "Tie";
        else if(result > 0) return "Alice";
        else return "Bob";
    }
    
    int solve(int[] stoneValue, int index, int[] dp) {
        if(index >= stoneValue.length) return 0;
        if(dp[index] != Integer.MIN_VALUE) return dp[index];
        int pickSum = 0;
        int output = Integer.MIN_VALUE;
        for(int i = 0; i < 3; i++) {
            if(index + i >= stoneValue.length) break;
            pickSum += stoneValue[index + i];
            int diff = pickSum - solve(stoneValue, index + i + 1, dp);
            output = Math.max(diff, output);
        }
        dp[index] = output;
        return output;
    }
    
}
