class Solution {
   int[] dp;

    public String stoneGameIII(int[] a) {
        dp = new int[a.length];
        Arrays.fill(dp, -1);
        int rt = dos(a, 0);
        if (rt > 0) {
            return "Alice";
        } else if (rt < 0) {
            return "Bob";
        } else {
            return "Tie";
        }
    }

    // 1: this user always win
    // gap is a-b
    private int dos(int[] a, int i) {
        if (i == a.length) {
            return 0;
        }
        if (dp[i] != -1) {
            return dp[i];
        }
        int max = Integer.MIN_VALUE;
        int accu = 0;
        for (int j = i; j < i + 3 && j < a.length; j++) {
            accu += a[j];
            int later = dos(a, j + 1);
            int cur = -later + accu;
            max = Math.max(max, cur);
        }
        dp[i] = max;
        return max;
    }
}
