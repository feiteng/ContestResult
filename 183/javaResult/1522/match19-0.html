class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[] dp = new int[n+1];
        dp[n] = 0;
        for (int i = n-1; i>=0; i--) {
            int max = Integer.MIN_VALUE;
            if (i+1 <= n) {
                max = Math.max(max, stoneValue[i] - dp[i+1]);
            }
            if (i+2 <= n) {
                max = Math.max(max, stoneValue[i]+stoneValue[i+1] - dp[i+2]);
            }
            if (i+3 <= n) {
                max = Math.max(max, stoneValue[i]+stoneValue[i+1]+stoneValue[i+2] - dp[i+3]);
            }
            dp[i] = max;
        }
        
        /*for (int i = 0; i < n; i++) {
            System.out.print(dp[i] + " ");
        }*/
        if (dp[0] > 0) return "Alice";
        if (dp[0] < 0) return "Bob";
        return "Tie";
    }
}
