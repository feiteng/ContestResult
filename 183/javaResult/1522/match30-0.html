class Solution {
    public String stoneGameIII(int[] stoneValue) {
        final int[] dp = new int[stoneValue.length];
        dp[dp.length - 1] = stoneValue[dp.length - 1];
        
        for(int i = dp.length - 2; i >= 0; i --) {
            //take 1
            int best = stoneValue[i] - dp[i + 1];
            
            //take 2
            if(i <= dp.length - 2) {
                int take2 = stoneValue[i] + stoneValue[i + 1];
                if(i < dp.length - 2) take2 -= dp[i + 2];
                best = Math.max(best, take2);
            }
            
            //take 3
            if(i <= dp.length - 3) {
                int take3 = stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2];
                if(i < dp.length - 3) take3 -= dp[i + 3];
                best = Math.max(best, take3);
            }
            
            dp[i] = best;
        }
        
        if(dp[0] > 0) {
            return "Alice";
        } else if(dp[0] < 0) {
            return "Bob";
        } else {
            return "Tie";
        }
    }
}
