class Solution {
    int inf = Integer.MIN_VALUE / 2;
    public String stoneGameIII(int[] sv) {
        int[] dp = new int[sv.length];
        Arrays.fill(dp, inf);
        int diff = dfs(0, sv, dp);
        if (diff == 0) return "Tie";
        return diff > 0 ? "Alice" : "Bob";
    }
    int dfs(int idx, int[] s, int[] dp) {
        if (idx == s.length) return 0;
        if (dp[idx] != inf) return dp[idx];
        int sum = 0, ans = inf;
        for (int i = idx; i < Math.min(s.length, idx + 3); i++) {
            sum += s[i];
            ans = Math.max(ans, sum - dfs(i + 1, s, dp));
        }
        return dp[idx] = ans;
    }
}
