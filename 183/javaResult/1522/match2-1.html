class Solution {
    public String stoneGameIII(int[] stone) {
        int l =  stone.length;
        int sum = stone[l - 1];
        int[] dp = new int[l]; // sum of Alice optimally
        dp[l - 1] = stone[l - 1];
        for (int i = l - 2; i >= 0; i--){
            dp[i] = -9999999;
            sum += stone[i];
            for (int d = 0; d <= 2; d++){
                int bob = (i + d + 1 < l) ? dp[i + d + 1] : 0;
                dp[i] = Math.max(dp[i], sum - bob);
            }
        }
        System.out.println(dp[0]);
        int r = sum - 2 * dp[0];
        if (r == 0){
            return "Tie";
        }
        else if (r > 0){
            return "Bob";
        }
        else{
            return "Alice";
        }
     }
}
