class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int sum = 0, len = stoneValue.length;
        int[] suffix = new int[len + 1];
        for (int i = 0; i < len; i ++) {
            sum += stoneValue[i];
            suffix[len - i - 1] = suffix[len - i] + stoneValue[len - i - 1];
        }
        
        // dp[i][j], max value player j can get from last i stone.
        // 0 : A, 1 : B;
        int[][] dp = new int[len + 1][2]; 
        
        dp[len - 1][0] = stoneValue[len - 1];
        dp[len - 1][1] = stoneValue[len - 1];
        
        for (int i = len - 2; i >= 0; i --) {
            dp[i][0] = stoneValue[i] + suffix[i + 1] - dp[i + 1][1];
            dp[i][1] = stoneValue[i] + suffix[i + 1] - dp[i + 1][0];
            if (i + 1 < len) {
                dp[i][0] = Math.max(dp[i][0], stoneValue[i] + stoneValue[i + 1] + suffix[i + 2] - dp[i + 2][1]);
                dp[i][1] = Math.max(dp[i][1], stoneValue[i] + stoneValue[i + 1] + suffix[i + 2] - dp[i + 2][0]);
            }
            
            if (i + 2 < len) {
                dp[i][0] = Math.max(dp[i][0], stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] + 
                                              suffix[i + 3] - dp[i + 3][1]);
                dp[i][1] = Math.max(dp[i][1], stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] + 
                                              suffix[i + 3] - dp[i + 3][0]);
            }
        }
        
        int alice = dp[0][0], bob = sum - alice;
        
        if (alice == bob) {
            return "Tie";
        } else if (alice > bob) {
            return "Alice";
        } else {
            return "Bob";
        }
        
    }
}
