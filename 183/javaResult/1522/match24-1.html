class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int sum = 0;
        for(int num : stoneValue)
            sum += num;
        Map<Integer, Integer> map = new HashMap<>();
        int res = Integer.MAX_VALUE;
        int temp = sum;
        for(int i = 0; i < 3 && i < stoneValue.length; i++){
            temp -= stoneValue[i];
            res = Math.min(res, helper(stoneValue, i + 1, temp, map));
        }
        //System.out.println(res);
        
if(sum - res > res)
            return "Alice";
        else if(sum - res < res)
            return "Bob";
        return "Tie";
    }
    
    public int helper(int[] nums, int pos, int sum, Map<Integer, Integer> map)
{
        if(pos == nums.length)
            return 0;
        if(map.containsKey(pos))
            return map.get(pos);
        int res = Integer.MIN_VALUE;
        int temp = sum;
        for(int i = 0; i < 3 && pos + i < nums.length; i++){
            temp -= nums[pos + i];
            
res = Math.max(res, sum - helper(nums, pos + i + 1, temp, map));
        }
        map.put(pos, res);
        //System.out.println(pos + " " + res);
        return res;
    }
}
