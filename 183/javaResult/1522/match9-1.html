class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[] scores = new int[n];
        //int sum = 0;
        for(int i = n-1; i>= 0; i --){
            //sum += stoneValue[i];
            int sum = 0;
            int max = Integer.MIN_VALUE;
            for(int j = 0; i + j < n && j < 3; j++){
                sum += stoneValue[i+j];
                max = Math.max(max, sum - (i+j+1 >= n? 0: scores[i+j+1]));
            }
            
scores[i] = max;
        }
        return scores[0] >0 ? "Alice": scores[0] == 0?"Tie":"Bob";
    }
}
