class Solution {
    public String stoneGameIII(int[] st) {
        int n = st.length;
        int sum = 0;
        for(int k : st) sum += k;
        int alice = dfs(st, 0, 0, sum);
        int bob = sum - alice;
        if(alice == bob) return "Tie";
        if(alice > bob) return "Alice";
        return "Bob";
    }
    
    Integer[][] f = new Integer[50010][2];
    
    int dfs(int[] ary, int pos, int player, int rest)
    {
        int n = ary.length;
        if(pos >= n) return rest;
        if(f[pos][player] != null) return f[pos][player];   
        int a = ary[pos], b = pos + 1 < n ? ary[pos + 1] : 0,
            c = pos + 2 < n ? ary[pos + 2] : 0;
        
        int ans = 1 << 20; 
        
        ans = Math.min(ans, dfs(ary, pos + 1, 1 - player, rest - a));
        ans = Math.min(ans, dfs(ary, pos + 2, 1 - player, rest - a - b));
        ans = Math.min(ans, dfs(ary, pos + 3, 1 - player, rest - a - b - c));
        
        return f[pos][player] = rest - ans;
    }
}
