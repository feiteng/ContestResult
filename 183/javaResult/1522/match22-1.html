class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int s = stoneValue.length;
        int[] memo = new int[s+1];
        memo[s - 1] = stoneValue[s-1];
        for(int i = s-2; i >= 0; i--){
            int p = stoneValue[i] - memo[i+1];
            
if(i+1 < s){
                p = Math.max(p, stoneValue[i] + stoneValue[i+1] - memo[i+2]);
            }
            if(i+2 < s){
                p = Math.max(p, stoneValue[i] + stoneValue[i+1] + stoneValue[i+2] - memo[i+3]);
            }
            memo[i] = p;
        }
        System.out.println(Arrays.toString(memo));
        
if(memo[0] > 0){
            return "Alice";
        }else if(memo[0] < 0){
            return "Bob";
        }else{
            return "Tie";
        }
    }
}
