class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int sum = 0;
        for (int num : stoneValue) {
            sum += num;
        }

        final int[] dp = new int[stoneValue.length];
        int recurse = recurse(stoneValue, 0, stoneValue.length - 1, dp);
        if(2*recurse == sum) {
            return "Tie";
        }
        return 2*recurse > sum? "Alice":"Bob";
    }

    private static int recurse(int[] nums, int start, int end, int[] dp) {
        if (start > end) {
            return 0;
        }

        if (dp[start] != 0) {
            return dp[start];
        }

        int one = nums[start] + Math.min(recurse(nums, start + 2, end, dp),
                                         Math.min(recurse(nums, start + 3, end, dp),
                                                  recurse(nums, start + 4, end, dp)));
        int two = Integer.MIN_VALUE;
        int three = Integer.MIN_VALUE;
        if (start + 1 <= end) {
            two = nums[start] + nums[start + 1] + Math.min(recurse(nums, start + 3, end, dp),
                                                           Math.min(recurse(nums, start + 4, end, dp),
                                                                    recurse(nums, start + 5, end, dp)));
        }
        if (start + 2 <= end) {
            three = nums[start] + nums[start + 1] + nums[start + 2] + Math.min(
                    recurse(nums, start + 4, end, dp),
                    Math.min(recurse(nums, start + 5, end, dp),
                             recurse(nums, start + 6, end, dp)));
        }

        return dp[start] = Math.max(one, Math.max(two, three));
    }
}
