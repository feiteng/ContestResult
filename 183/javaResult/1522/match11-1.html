class Solution {
    public String stoneGameIII(int[] stones) {
        int sum = 0;
        for (int s : stones) {
            sum += s;
        }
        int max = getMax(stones, 0, sum, new Integer[stones.length]);
        
if (max > sum - max) {
            return "Alice";
        } else if (max == sum - max) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
    private int getMax(int[] stones, int start, int sum, Integer[] dp) {
        if (start == stones.length) {
            return 0;
        }
        if (dp[start] != null) {
            return dp[start];
        
}
        int res = sum - getMax(stones, start + 1, sum - stones[start], dp);
        if (start < stones.length - 1) {
            res = Math.max(res, sum - getMax(stones, start + 2, sum - stones[start] - stones[start + 1], dp));
        }
        if (start < stones.length - 2) {
            res = Math.max(res, sum - getMax(stones, start + 3, sum - stones[start] - stones[start + 1] - stones[start + 2], dp));
        }
        dp[start] = res;
        return res;
    }
}
