class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int[] firstWin = new int[stoneValue.length + 1];
        for (int i=stoneValue.length -1;i>-1;i--) {
            firstWin[i] = stoneValue[i] - firstWin[i+1];
            for (int j=1;j<3;j++) {
                if (i+j>=stoneValue.length)break;
                int tmp =0;
                for(int k=0;k<=j;k++){
                    tmp+=stoneValue[i+k];
                }
                tmp -=firstWin[i+j+1];
                
firstWin[i] = Math.max(firstWin[i], tmp);
            }
        }
//        for (int i : firstWin) {
//            System.out.println(i);
//        }
        if (firstWin[0] > 0) {
            return "Alice";
        } else if(firstWin[0] == 0) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
}
