class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int[] bobScores = new int[stoneValue.length];
        int[] aliceScores = new int[stoneValue.length];
        bobScores[stoneValue.length - 1] = stoneValue[stoneValue.length - 1];
        
aliceScores[stoneValue.length - 1] = stoneValue[stoneValue.length - 1];
        for(int i = stoneValue.length - 2; i >=0; i--) {
            //alice's best
            int aliceBest = stoneValue[i] - bobScores[i+1];
            if(i < stoneValue.length - 2) {
                int takeTwo = stoneValue[i] + stoneValue[i+1] - bobScores[i+2];
                if(takeTwo > aliceBest) {
                    aliceBest = takeTwo;
                }
                if(i < stoneValue.length - 3) {
                    int takeThree = stoneValue[i] + stoneValue[i+1] + stoneValue[i+2] - bobScores[i+3];
                    if(takeThree > aliceBest) {
                        aliceBest = takeThree;
                    }
                } else {
                    int takeThree = stoneValue[i] + stoneValue[i+1] + stoneValue[i+2];
                    if(takeThree > aliceBest) {
                        aliceBest = takeThree;
                    }
                }
            } else {
                int takeTwo = stoneValue[i] + stoneValue[i+1];
                if(takeTwo > aliceBest) {
                    aliceBest = takeTwo;
                }
            }
            aliceScores[i] = aliceBest;
            
            int bobBest = stoneValue[i] - aliceScores[i+1];
            if(i < stoneValue.length - 2) {
                int takeTwo = stoneValue[i] + stoneValue[i+1] - aliceScores[i+2];
                if(takeTwo > bobBest) {
                    bobBest = takeTwo;
                }
                if(i < stoneValue.length - 3) {
                    int takeThree = stoneValue[i] + stoneValue[i+1] + stoneValue[i+2] - aliceScores[i+3];
                    if(takeThree > bobBest) {
                        bobBest = takeThree;
                    }
                } else {
                    int takeThree = stoneValue[i] + stoneValue[i+1] + stoneValue[i+2];
                    if(takeThree > bobBest) {
                        bobBest = takeThree;
                    }
                }
            } else {
                int takeTwo = stoneValue[i] + stoneValue[i+1];
                if(takeTwo > bobBest) {
                    bobBest = takeTwo;
                }
            
}
            bobScores[i] = bobBest;
        }
        if(aliceScores[0] > 0) {
            return "Alice";
        } else if(aliceScores[0] == 0) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
}
