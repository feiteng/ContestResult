class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int sum = 0;
        for(int num : stoneValue){
            sum += num;
        }
        
        int[] dp = new int[stoneValue.length];
        Arrays.fill(dp, Integer.MIN_VALUE);
        int max = search(stoneValue, dp, 0);
        
        if(max == sum - max){
            return "Tie";
        }else if(max > sum - max){
            return "Alice";
        }else{
            return "Bob";
        }
    }
    
    private int search(int[] nums, int[] dp, 
int start){
        if(start >= nums.length){
            return 0;
        }
        if(dp[start] != Integer.MIN_VALUE){
            return dp[start];
        }
        
        // try to take some stones
        int max = Integer.MIN_VALUE;
        int sum = 0;
        for(int i=start;i<Math.min(nums.length, start+3);i++){
            sum += nums[i];
            
            // find the remaining value
            
int remaining = Integer.MAX_VALUE;
            for(int j = i+1;j<=i+3;j++){
                remaining = Math.min(remaining, search(nums, dp, j+1));
            }
            max = Math.max(max, sum + remaining);
        }
        dp[start] = max;
        return max;
    }
}
