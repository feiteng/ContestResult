class Solution {

  public String stoneGameIII(int[] sv) {
    Integer alice[] = new Integer[sv.length];
    Integer bob[] = new Integer[sv.length];
    int s = 0;
    int[] psum = new int[sv.length];
    for (int v : sv) {
      s += v;
    }
    psum[sv.length - 1] = sv[sv.length - 1];
    for (int i = sv.length - 2; i >= 0; i--) {
      psum[i] = psum[i + 1] + sv[i];
    }
    alice(sv, alice, bob, 0, psum);
    return s % 2 == 0 && alice[0] == s / 2 ? "Tie" : alice[0] > s - alice[0] ? "Alice" : "Bob";
  }

  private int alice(int[] sv, Integer[] alice, Integer[] bob, int index, int[] s) {
    if (index >= alice.length) {
      return 0;
    } else if (alice[index] == null) {
      alice[index] = Integer.MIN_VALUE;
      alice[index] = Math.max(alice[index], bob(sv, alice, bob, index + 1, s) + get(sv, index));
      alice[index] = Math.max(alice[index],
          bob(sv, alice, bob, index + 2, s) + get(sv, index) + get(sv, index + 1));
      alice[index] = Math.max(alice[index],
          bob(sv, alice, bob, index + 3, s) + get(sv, index) + get(sv, index + 1) + get(sv, index + 2));
    }
    return s[index] - alice[index];
  }

  private int bob(int[] sv, Integer[] alice, Integer[] bob, int index, int[] sum) {
    if (index >= bob.length) {
      return 0;
    } else if (bob[index] == null) {
      bob[index] = Integer.MIN_VALUE;
      bob[index] = Math.max(bob[index], alice(sv, alice, bob, index + 1, sum) + get(sv, index));
      bob[index] = Math.max(bob[index],
          alice(sv, alice, bob, index + 2, sum) + get(sv, index) + get(sv, index + 1));
      bob[index] = Math.max(bob[index],
          alice(sv, alice, bob, index + 3, sum) + get(sv, index) + get(sv, index + 1) + get(sv, index + 2));
    }
    return sum[index] - bob[index];
  }

  private int get(int[] stoneValue, int index) {
    if (index >= stoneValue.length) {
      return 0;
    } else {
      return stoneValue[index];
    }
  }
}
