class Solution {
    int[] prefixSum;
    int[] dp;
    public String stoneGameIII(int[] stoneValue) {
        prefixSum=new int[stoneValue.length+1];
        int sum=0;
        dp=new int[stoneValue.length+1];
        Arrays.fill(dp,Integer.MIN_VALUE);
        for(int i=1;i<=stoneValue.length;i++){
            sum+=stoneValue[i-1];
            prefixSum[i]=sum;
        }
        int val=dfs(stoneValue,0);
        //System.out.println(val);
        if(sum-val<val)
            return "Alice";
        else if(sum-val>val)
            return "Bob";
        else
            return "Tie";
    }
    
    public int dfs(int[] nums,int start){
        int sum=0;
        if(dp[start]!=Integer.MIN_VALUE)
            return dp[start];
        int max=Integer.MIN_VALUE;
        for(int i=start;i<Math.min(start+3,nums.length);i++){
            sum+=nums[i];
            max=Math.max(max,sum+prefixSum[nums.length]-prefixSum[i+1]-dfs(nums,i+1));
        }
        dp[start]=max;
        return max==Integer.MIN_VALUE?0:max;
    }
}
