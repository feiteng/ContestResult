class Solution {
    
int[] vals;
    int n;
    int[] dp;
    int min=Integer.MIN_VALUE;
    public String stoneGameIII(int[] stoneValue) {
        vals=stoneValue;
        n=vals.length;
        dp=new int[n];
        for(int i=0;i<n;i++){
            dp[i]=min;
        }
        int result=dfs(0);
        return result>0?"Alice":(result==0?"Tie":"Bob");
    
}
    
    public int dfs(int start){
        if(start==n){
            return 0;
        }
        if(dp[start]!=min){
            return dp[start];
        }
        int m=Math.min(start+3,n);
        int sum=0;
        
int res=min;
        for(int i=start;i<m;i++){
            sum+=vals[i];
            res=Math.max(res,sum-dfs(i+1));
        }
        dp[start]=res;
        return res;
    }
}
