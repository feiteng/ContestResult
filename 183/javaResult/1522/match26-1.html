class Solution {
    public String stoneGameIII(int[] stone) {
        int n = stone.length;
        Integer[] cache = new Integer[n];
        
int ans = helper(stone, 0, cache);
        if (ans > 0) return "Alice";
        else if (ans < 0) return "Bob";
        else return "Tie";
    }
    
    private int helper(int[] stone, int idx, Integer[] cache) {
        if (idx >= stone.length) return 0;
        if (cache[idx] != null) return cache[idx];
        int sum = 0;
        
int score = Integer.MIN_VALUE;
        for (int i = idx; i <= Math.min(stone.length - 1, idx + 2); i++) {
            sum += stone[i];
            score = Math.max(score, sum - helper(stone, i+1, cache));
        }
        cache[idx] = score;
        return score;
    }
    
}
