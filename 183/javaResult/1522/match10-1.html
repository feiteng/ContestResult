	class Solution {
	    public String stoneGameIII(int[] a) {
	        int n = a.length;
	        long[][] dp = new long[2][n+1];
	        for(int
 i = n-1;i >= 0;i--){
	        	long ls = 0;
	        	long v0 = Long.MIN_VALUE;
	        	long v1 = Long.MAX_VALUE;
	        	for(int j = 1;j <= 3 && i+j <= n;j++){
	        		ls += a[i+j-1];
	        		v0 = Math.max(v0, dp[1][i+j] + ls);
	        		v1 = Math.min(v1, dp[0][i+j] - ls);
	        	}
	        	dp[0][i] = v0;
	        	dp[1][i] = v1;
	        }
	        return dp[0][0] == 0 ? "Tie" : dp[0][0] > 0 ? "Alice" : "Bob";
	    }
	}
