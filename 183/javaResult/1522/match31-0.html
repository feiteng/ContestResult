class Solution {
	    public String stoneGameIII(int[] stoneValue) {
	    	arr = stoneValue;
	        n = arr.length;
	        dp = new int[n+1];
	        Arrays.fill(dp, minv);
	        int res = calc(0);
	        if(res > 0) return "Alice";
	        else if(res < 0) return "Bob";
	        else return "Tie";
	    }
	    int n;
	    int minv = -1000000000;
	    int[] arr, dp;	    
	    public int calc(int i) {
	    	if(dp[i] != minv) return dp[i];
	    	else if(i == n) return dp[i] = 0;
	    	else {
	    		int a = minv, b = minv, c = minv;
	    		if(i+1 <= n) a = arr[i] - calc(i+1);
	    		if(i+2 <= n) b = arr[i] + arr[i+1] - calc(i+2);
	    		if(i+3 <= n) c = arr[i] + arr[i+1] + arr[i+2] - calc(i+3);
	    		return dp[i] = Math.max(a, Math.max(b, c));
	    	}
	    }
	    
	}
