class Solution {
    public String stoneGameIII(int[] stoneValue) {
        int n = stoneValue.length;
        int[][] dp = new int[n + 1][2];
        dp[n - 1][0] = stoneValue[n - 1];
        for (int i = n - 2; i >= 0; --i) {
            int max = Integer.MIN_VALUE, sum = 0, min = Integer.MAX_VALUE;
            for (int j = i; j < n && j < i + 3; ++j) {
                min = Math.min(min, dp[j + 1][0]);
                sum += stoneValue[j];
                max = Math.max(max, sum + dp[j + 1][1]);
            }
            dp[i][0] = max;
            dp[i][1] = min;
        }
        int total = 0;
        for (int v: stoneValue) total += v;
        if (dp[0][0] > total - dp[0][0]) return "Alice";
        if (dp[0][0] < total - dp[0][0]) return "Bob";
        return "Tie";
    }
}
