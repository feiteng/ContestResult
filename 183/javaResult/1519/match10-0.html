class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        List<Integer> res= new ArrayList<>();
        int total= 0;
        for(int n:nums){
            total+=n;
        }
        Arrays.sort(nums);
        int temp=0;
        for(int i=nums.length-1;i>=0;i--){
            temp+=nums[i];
            res.add(nums[i]);
            if(temp>(total-temp)) break;
        }
        return res;
    }
}
