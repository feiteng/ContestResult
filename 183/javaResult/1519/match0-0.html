class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        List<Integer> result = new ArrayList<>();
        
        int sum = 0;
        
        for (int num : nums) {
            sum += num;
        }
        
        int subSum = 0;
        
        for (int i = nums.length - 1; i >= 0; i--) {
            subSum += nums[i];
            result.add(nums[i]);
            
            if (subSum > sum - subSum) {
                return result;
            }
        }
        
        return result;
    }
}
