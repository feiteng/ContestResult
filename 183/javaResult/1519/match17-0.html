class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Queue<Integer> q = new PriorityQueue<>(Collections.reverseOrder());
        List<Integer> list = new ArrayList<>();
        int sum = 0;
        for(int k : nums)
        {
            sum += k;
            q.offer(k);
        }
        int pop = 0;
        while(pop <= sum / 2)
        {
            int p = q.poll();
            pop += p;
            list.add(p);
        }
        return list;
    }
}
