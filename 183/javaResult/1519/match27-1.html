class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        List<Integer> res = new ArrayList<>();
        int sum = 0;
        for(int x : nums) sum += x
;
        int acc = 0;
        for(int i = nums.length - 1; i >= 0; --i) {
            acc += nums[i];
            res.add(nums[i]);
            if(acc * 2 > sum) break;
        }
        return res;
    }
}
