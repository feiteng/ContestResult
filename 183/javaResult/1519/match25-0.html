class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        int sum = 0;
        for(int i : nums){
            sum += i;
        }
        
        Arrays.sort(nums);
        
        List<Integer> res = new ArrayList<>();
        int cur = 0;
        for(int loc = nums.length - 1; loc >= 0; loc--){
            int i = nums[loc];
            cur += i;
            res.add(i);
            if(cur > sum - cur){
                break;
            }
        }
        
        return res;
    }
}
