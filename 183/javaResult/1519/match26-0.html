class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        int sum = 0;
        for (int n : nums) sum += n;
        Arrays.sort(nums);
        // Collections.sort(nums, Collections.reverseOrder());
        List<Integer> res = new ArrayList<>();
        int curr = 0;
        for (int i = nums.length - 1;i >= 0;i--) {
            curr += nums[i];
            res.add(nums[i]);
            if (curr > sum - curr) return res;
        }
        return res;
    }
}
