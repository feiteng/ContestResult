class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for (int n : nums) {
            sum += n;
        }
        int t = 0;
        List<Integer> res = new LinkedList<>();
        for (int i = nums.length - 1; i >= 0; i--) {
            int n = nums[i];
            t += n;
            res.add(n);
            if (t > sum - t) {
                break;
            }
        }
        return res;
    }
}
