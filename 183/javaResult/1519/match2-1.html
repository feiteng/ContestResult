class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        
		List<Integer> list = new LinkedList<>();
		Arrays.sort(nums);
		int allSum = 0;
		for(int n:nums) {
			allSum += n;
		}
		int sum = 0;
		for(int i=nums.length-1;i>=0;i--) {
			sum += nums[i];
			list.add(nums[i]);
			if(sum*2>allSum) {
				return list;
			}
		}
		return list;
	
    }
}
