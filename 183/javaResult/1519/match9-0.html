class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for(int x : nums) {
            sum += x;
        }
        List<Integer> result = new ArrayList<>();
        int cur = 0;
        for(int i = nums.length-1; i >= 0; i--) {
            int x = nums[i];
            cur += x;
            sum -= x;
            result.add(x);
            if(cur > sum) {
                break;
            }
        }
        return result;
    }
}
