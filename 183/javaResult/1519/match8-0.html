class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
    int sum = 0;
    for (int v : nums) {
      sum += v;
    }
    List<Integer> res = new ArrayList<>();
    int part = 0;
    for (int i = nums.length - 1; i >= 0; i--) {
      part += nums[i];
      res.add(nums[i]);
      if (2 * part > sum) {
        break;
      }
    }
    return res;
    }
}
