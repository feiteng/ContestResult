import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;


class Solution {
	public List<Integer> minSubsequence(int[] nums) {
		Arrays.sort(nums);
		int sum = 0;
		for (int i : nums) {
			sum += i;
		}
		int cur = 0;
		List<Integer> ret = new ArrayList<Integer>();
		for (int i = nums.length - 1; i >= 0 && cur <= sum - cur; i--) {
			cur += nums[i];
			ret.add(nums[i]);
		}
		return ret;

	}
}
