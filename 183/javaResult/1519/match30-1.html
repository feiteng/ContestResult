class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        List<Integer> res = new ArrayList<>();
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        int cnt = 0;
        for (int i = nums.length - 1;i >= 0;i--) {
            cnt += nums[i];
            res.add(nums[i]);
            if (cnt > sum - cnt) break;
        }
        return res;
    }
}
