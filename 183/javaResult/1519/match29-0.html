class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        int sum = 0;
        for (int i : nums){
            sum += i;
        }
        
        List<Integer> res = new ArrayList<>();
        Arrays.sort(nums);
        int n = 0;
        int target = sum/2;
        for (int i = nums.length-1; i >= 0; i--){
            n+=nums[i];
            res.add(nums[i]);
            if (n > target){
                break;
            }
        }
        return res;
    }
    

}
