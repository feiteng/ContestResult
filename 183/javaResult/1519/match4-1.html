class Solution {
  public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for(int a:nums) {
            sum += a;
        }
        int x = 0;
        List<Integer> ans = new ArrayList<>();
        for(int i =nums.length-1;i>=0;i--) {
            x += nums[i];
            ans.add(nums[i]);
            if (x * 2 > sum) {
                return ans;
            }
        }
        return ans;
    }

}
