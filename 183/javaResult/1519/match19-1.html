class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for(int i =0;i<nums.length;i++){
            sum += nums[i];
        }
        
        int running = 0;
        List<Integer> list = new ArrayList<>();
        for(int i =nums.length-1;i>=0;i--){
            running += nums[i];
            
list.add(nums[i]);
            if(running > sum - running){
                break;
            }
        }
        return list;
    }
}
