class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for (int num: nums) {
            sum += num;
        }
        
        
List<Integer> result = new ArrayList<>();
        int total = 0;
        for (int i = nums.length - 1; i >= 0; --i) {
            total += nums[i];
            result.add(nums[i]);
            if (total * 2 > sum) {
                break;
            }
        }
        return result;
    }
}
