class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        List<Integer> result = new ArrayList<>();
        int total = 0;
        for(int i = 0; i < nums.length; i++) {
            total += nums[i];
        }
        int currentTotal = 0;
        for(int i = nums.length - 1; i >= 0; i--) {
            currentTotal += nums[i];
            total -= nums[i];
            result.add(nums[i]);
            if(currentTotal > total) {
                return result;
            }
        }
        return result;
    }
}
