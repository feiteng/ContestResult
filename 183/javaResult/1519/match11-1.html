class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for (int num : nums) {
            sum += num; 
        }
        sum = sum / 2 + 1;
        List<Integer> res = new ArrayList<>();
        int cur = 0;
        for (int i = nums.length - 1; i >= 0; i--) {
            res.add(nums[i]);
            cur += nums[i];
            if (cur >= sum) {
                break;
            }
        }
        return res;
    }
}
