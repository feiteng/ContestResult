class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        Arrays.sort(nums);
        int sum = 0;
        for(int x : nums){
            sum += x;
        }
        int save = 0;
        List<Integer> ans = new ArrayList();
        for(int i = nums.length - 1; i >= 0 && save * 2 <= sum; i--){
            ans.add(nums[i]);
            save += nums[i];
        }
        return ans;
    }
}
