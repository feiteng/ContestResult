class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        int sum=0;
        for(int num:nums){
            sum+=num;
        }
        List<Integer> ret=new ArrayList<>();
        int cur=0;
        Arrays.sort(nums);
        
int n=nums.length;
        for(int i=n-1;i>=0;i--){
            ret.add(nums[i]);
            cur+=nums[i];
            if(2*cur>sum){
                break;
            }
        }
        return ret;
    }
}
