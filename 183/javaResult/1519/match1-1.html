class Solution {
    public List<Integer> minSubsequence(int[] sorted) {
        Arrays.sort(sorted);
        int tot = 0;
        for (int n : sorted) tot += n;
        int cur = 0;
        List<Integer> res = new ArrayList<>();
        for (int i = sorted.length - 1; i >= 0; i--) {
            cur += sorted[i];
            res.add(sorted[i]);
            if (cur > tot - cur) return res;
        }
        return res;
    }
}
