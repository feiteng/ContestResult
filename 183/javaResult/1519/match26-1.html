class Solution {
    public List<Integer> minSubsequence(int[] nums) {
        int sum = 0;
        for (int v : nums) sum += v;
        
        Arrays.sort(nums);
        
        List<Integer> ans = new ArrayList<>();
        int N = nums.length;
        
int sum2 = 0;
        for (int i = N - 1; i >= 0; i--) {
            sum2 += nums[i];
            ans.add(nums[i]);
            
            if (sum2 > sum - sum2) return ans;
        }
        
        return ans;
    }
}
