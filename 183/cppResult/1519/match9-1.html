class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        
        int tot = 0, sum = 0;
        for (auto x : nums)
            tot += x;
        
        vector<int> ans;
        for (int i = nums.size() - 1; i >= 0; i--)
        {
            ans.push_back(nums[i]);
            sum += nums[i];
            if (sum > tot - sum)
                break;
                
        }
        
        return ans;
        
    }
};
