class Solution {
public:
    vector<int> minSubsequence(vector<int>& a) {
      int n = a.size();
      sort(a.begin(), a.end());
      int s = 0;
      for (int i = 0; i < n; i++) {
        s += a[i];
      }
      int t = 0;
      vector<int> ans;
      for(int i = n - 1; i >= 0; i--) {
        t += a[i];
        ans.push_back(a[i]);
        if(t * 2 > s) {
          break;
        }
      }
      return ans;
    }
};
