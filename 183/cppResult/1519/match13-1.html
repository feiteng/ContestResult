class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.rbegin(), nums.rend());
        
        int total = 0;
        for (int x : nums) total += x;
        
        int cur = 0;
        vector<int> ret;
        for (int x : nums) {
            cur += x;
            ret.push_back(x);
            total -= x;
            if (cur > total) break;
        }
        return ret;
    }
};
