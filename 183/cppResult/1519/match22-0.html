class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.rbegin(),nums.rend());
        vector<int> ans;
        int sum=0,co=0;
        for(int i=0;i<nums.size();i++)
            sum+=nums[i];
        for(int i=0;i<nums.size();i++)
        {
            co+=nums[i];
            ans.push_back(nums[i]);
            if(co>sum-co)
                break;
        }
        return ans;
    }
};
