class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end(), greater<int>());
        int tot = 0;
        for (int num : nums) tot += num;
        vector<int> ret;
        int cur = 0;
        for (int num : nums) {
            ret.push_back(num);
            cur += num;
            if (cur*2 > tot) break;
        }
        return ret;
        
            
        
    }
};
