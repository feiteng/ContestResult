class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int sum=0;
        for(int x:nums)sum+=x;
        sort(nums.rbegin(),nums.rend());
        vector<int>ans;
        int s=0;
        for(int i=0;i<nums.size();i++){
            ans.push_back(nums[i]);
            s+=nums[i];
            if(s>sum/2)break;
        }
        return ans;
    }
};
