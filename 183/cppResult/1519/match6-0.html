class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end(), greater<>());
        
        int sum=0;
        for (auto& i:nums)
            sum+=i;
        
        vector<int> ans;
        int total=0;
        for (int i=0;i<nums.size();i++) {
            ans.push_back(nums[i]);
            total+=nums[i];
            if (total>sum/2)
                break;
        }
        
        return ans;
    }
};
