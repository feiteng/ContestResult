class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int n=nums.size();
        int sum=0;
        for(int i=0;i<n;++i)
            sum+=nums[i];
        sort(nums.begin(),nums.end());
        vector<int> ans;
        int tmpSum=0;
        for(int i=n-1;i>=0;--i)
        {
            tmpSum+=nums[i];
            ans.push_back(nums[i]);
            if(tmpSum*2>sum)
                break;
        }
        return ans;
    }
};
