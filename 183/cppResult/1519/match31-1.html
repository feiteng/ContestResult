class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.rbegin(), nums.rend());
        int sum = 0;
        for(int i = 0; i < nums.size(); i++)
        {
            sum += nums[i];
        }
        
        int sub = 0;
        int k = -1;
        vector<int> ans;

        for(int i = 0; i < nums.size(); i++)
        {
            ans.push_back(nums[i]);
            sub += nums[i];
            if(sub > sum - sub)
            {
                break;
            }
        }
        
        return ans;
    }
};
