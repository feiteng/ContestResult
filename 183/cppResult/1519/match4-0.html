class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int s = 0;
        for (auto x : nums) s += x;
        vector<int> v;
        int t = 0;
        for (int i = nums.size() - 1; i >= 0; --i) {
            v.push_back(nums[i]);
            t += nums[i];
            if (t * 2 > s) break;
        }
        return v;
    }
};
