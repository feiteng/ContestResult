class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end(), greater<int>());
        int sum = accumulate(nums.begin(), nums.end(), 0);
        int i = 0;
        int preSum = 0;
        while (i < nums.size() && preSum*2 <= sum) {
            preSum += nums[i];
            i++;
        }
        return vector<int>(nums.begin(), nums.begin()+i);
    }
};
