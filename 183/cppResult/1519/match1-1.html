class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int acc_sum = accumulate(nums.begin(), nums.end(), 0);
        sort(nums.begin(), nums.end());
        int cur_sum = 0;
        vector<int> res;
        
        for (int i=nums.size()-1;i>=0;i--) {
            cur_sum += nums[i];
            res.push_back(nums[i]);
            if (cur_sum > acc_sum - cur_sum)
                break;
        }
        return res;
    }
};
