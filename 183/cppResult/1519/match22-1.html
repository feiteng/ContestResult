class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        
        sort(nums.begin(), nums.end());
        vector<int> res;
        
        int sumVal = 0, currVal = 0;
        
        for (int& num : nums) sumVal += num;
        

        for (int i = nums.size() - 1; i >= 0; --i) {
            currVal += nums[i];
            res.push_back(nums[i]);
            if (currVal > sumVal - currVal) break;
        }
        return res;
    }
    
};
