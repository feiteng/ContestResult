class Solution {
public:
    vector<int> minSubsequence(vector<int>& a) {
        sort(a.begin(),a.end(),greater<int>());
        int sum=0;
        for(int i=0;i<a.size();i++)
            sum+=a[i];
        int x=0;
        vector<int> v;
        for(int i=0;i<a.size();i++){
            x+=a[i];
            sum-=a[i];
            v.push_back(a[i]);
            if(x>sum)
                break;
        }
        return v;
    }
};
