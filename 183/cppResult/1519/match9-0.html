class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int sum = 0, ps = 0;
        for (auto i: nums) sum += i;
        vector<int> ret;
        for (int i = nums.size() - 1; i >= 0; --i) {
            ps += nums[i];
            ret.push_back(nums[i]);
            if (ps > sum / 2) break;
        }
        return ret;
    }
};
