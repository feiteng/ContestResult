class Solution {
public:
    vector<int> minSubsequence(vector<int>& a) {
        int s = 0;
        for (int i : a) {
            s += i;
        }
        sort(a.begin(), a.end());
        reverse(a.begin(), a.end());
        vector <int> ret;
        int cur = 0;
        for (int i = 0; i < a.size(); i++) {
            ret.push_back(a[i]);
            cur += a[i];
            if (cur > s - cur) {
                break;
            }
        }
        return ret;
    }
};
