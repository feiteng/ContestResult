class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int sum = 0;
        for (int num : nums) sum += num;
        vector<int> res;
        sort(nums.begin(), nums.end());
        int cnt = 0;
        for (int i = nums.size()-1; i>=0; --i) {
            cnt += nums[i];
            res.push_back(nums[i]);
            if (cnt>sum/2) break;
        }
        
        return res;
    }
};
