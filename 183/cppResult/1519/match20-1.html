class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int ss = 0;
        for (int i = 0; i < nums.size(); i++) {
            ss += nums[i];
        }
        int s = 0;
        vector<int> ans;

        for (int i = nums.size()-1; i >= 0; i--) {
            s += nums[i];
            ans.push_back(nums[i]);
            if (s > ss/2) {
                break;
            }
        }
        return ans;
    }
};
