class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        int len = nums.size();
        sort(nums.begin(), nums.end());
        int sum = 0;
        for(int i=0; i<len; i++){
            sum += nums[i];
        }
        vector<int> res;
        int cur = 0;
        for(int i=nums.size()-1; i>=0; i--){
            res.push_back(nums[i]);
            cur+=nums[i];
            if(cur>sum-cur)
                break;
        }
        return res;
    }
};
