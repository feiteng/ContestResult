class Solution {
public:
    vector<int> minSubsequence(vector<int>& A) {
        int sum = accumulate(A.begin(), A.end(), 0);
        sort(A.rbegin(), A.rend());
        vector<int> ans;
        int cnt =0;
        for(int i : A) {
            cnt += i;
            ans.push_back(i);
            if(cnt > sum - cnt)
                break;
        }
        return ans;
    }
};
