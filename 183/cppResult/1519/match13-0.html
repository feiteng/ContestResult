class Solution {
public:
    vector<int> minSubsequence(vector<int>& a) {
        sort(a.begin(), a.end(), greater<>());
        
        int s = 0;
        for(int x : a)
            s+=x;
        
        int s1 = 0;
        vector<int> ans;
        for(int x:a){
            s1 += x;
            s -= x;
            ans.push_back(x);
            if(s1>s)
                break;
        }
        
        return ans;
    }
};
