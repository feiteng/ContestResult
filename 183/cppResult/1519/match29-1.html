using LL = long long;



class Solution {
public:
    vector<int> minSubsequence(vector<int>& nums) {
        sort(nums.begin(), nums.end(), greater<int>());
        LL s = 0;
        for(int i : nums) {
            s += i;
        }
        vector<int> ans;

        LL cur = 0;
        for(int i : nums) {
            cur += i;
            ans.push_back(i);
            if(cur > s - cur) {
                break;
            }
        }
        return ans;
    }
};
