class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int len = stoneValue.size();
        vector<int> first(len+1, 0);
        vector<int> second(len+1, 0);
        int sum = 0;

        for(int i=len-1;i>=0;i--){
            sum += stoneValue[i];
            int res = INT_MIN;
            int cur = 0;
            for(int k=1;k<=3;k++){
                if(i+k-1<len){
                    cur += stoneValue[i+k-1];
                    if(cur+second[i+k] > res){
                        res = cur+second[i+k];
                    }
                }
            }
            first[i] = res;
            second[i] = sum - res;
        }
        int t = first[0] - second[0];

        if(t>0){
            return "Alice";
        } else if(t==0){
            return "Tie";
        } else {
            return "Bob";
        }
    }
};
