class Solution {
public:
    string stoneGameIII(vector<int>& a) {
        int n = a.size();
        vector<int> s(n+1);
        for (int i = 1; i <= n; ++ i)
            s[i] = s[i-1]+a[i-1];
        const int INF = 1000000000;
        vector<int> f(n+2, -INF);
        f[n+1] = 0;
        for (int i = n; i >= 1; -- i)
        {
            f[i] = -INF;
            for (int j = 1; j <= 3 && i+j <= n+1; ++ j)
            {

                f[i] = max(f[i], s[i+j-1]-s[i-1]-f[i+j]);
            }
        }
        if (f[1] > 0)
            return "Alice";
        else if (f[1] < 0)
            return "Bob";
        else
            return "Tie";
    }
};
