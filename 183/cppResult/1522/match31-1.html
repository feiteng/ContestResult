class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> dp(n + 1, INT_MIN);
        

        dp[n] = 0;
        for (int i = n - 1; i >= 0; i--) {
            int tmp_gain = 0;
            for (int k = i, cnt = 1; k < n && cnt <= 3; k++, cnt++) {

                tmp_gain += stoneValue[k];
                dp[i] = max(dp[i], tmp_gain - dp[k + 1]);
            }
        }
        
        // for (int i = 0; i < n; i++)
        //     cout << dp[i] << " ";
        // cout << endl;
        
        if (dp[0] > 0) return "Alice";
        else if (dp[0] == 0) return "Tie";
        else return "Bob";
    }
};
