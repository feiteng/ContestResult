class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        const int n = stoneValue.size();
        vector<int> dp(n + 1);

        for (int i = n - 1; i >= 0; --i) {
            dp[i] = INT_MIN;
            int val = 0;
            for (int j = i; j < i + 3 && j < n; ++j) {
                val += stoneValue[j];
                dp[i] = max(dp[i], val - dp[j + 1]);
            }
        }
        if (dp[0] < 0) {
            return "Bob";
        } else if (dp[0] > 0) {
            return "Alice";
        }
        return "Tie";
    }
};
