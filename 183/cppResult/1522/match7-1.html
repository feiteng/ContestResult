class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> sum(n + 1, 0);
        for (int i = n - 1; i >= 0; i--) {
            sum[i] = sum[i + 1] + stoneValue[i];
        }
        
        vector<int> d(n + 1, 0);
        d[n - 1] = stoneValue[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            d[i] = stoneValue[i] + sum[i + 1] - d[i + 1];
            if (i + 1 <= n - 1) {
                d[i] = max(d[i], stoneValue[i] + stoneValue[i + 1] + sum[i + 2] - d[i + 2]);
            }
            if (i + 2 <= n - 1) {
                d[i] = max(d[i], stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] + sum[i + 3] - d[i + 3]);
            }
        }
        
        cout << d[0] << setw(4) << sum[0] << endl;

        if (d[0] == (sum[0] - d[0])) {
            return "Tie";
        } else if (d[0] >= (sum[0] - d[0])) {
            return "Alice";
        } else {
            return "Bob";
        }
    }
};
