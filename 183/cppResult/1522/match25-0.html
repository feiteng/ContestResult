class Solution {
public:

    string stoneGameIII(vector<int>& in) {
        int len = in.size();
        vector<int> dp(len, 0);
        for(int i=len-1;i>=0;i--)
        {
            int cur = INT_MIN;
            int sum = 0;
            for(int j=0;j<3;j++)
            {
                if(i+j>=len) break;
                sum+=in[i+j];
                cur = max(cur , sum - (i+j+1 >=len? 0:dp[i+j+1]) );
            }
            dp[i] = cur;
            //cout<<i<<" "<<cur<<endl;
        }
        return dp[0] == 0 ? "Tie":dp[0]>0?"Alice":"Bob";
    }
};
