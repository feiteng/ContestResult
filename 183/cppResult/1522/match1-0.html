class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        //if (stoneValue.size()<4) return "Alice";
        vector<int>f(0);
        int len=stoneValue.size();
        for (int i=0;i<=len+10;i++) f.push_back(0);
        for (int i=0;i<=10;i++) stoneValue.push_back(0);
        //for (int i=1;i<=3;i++)
        //{
        //    f[len-i]=f[len-i+1]+stoneValue[len-i];
        //}
        for (int i=len-1;i>=0;i--)
        {
            int Max=-2147483647;
            for (int j=1;j<=3;j++)
            {
                int sum=0;
                for (int k=0;k<j;k++) sum+=stoneValue[i+k];
                //cout<<"i: "<<i<<"    j:"<<j<<"   sum:"<<sum<<endl;
                int Min=0x7fffffff;
                for (int k=1;k<=3;k++)
                {
                    int sumB=0;
                    for (int l=0;l<k;l++) sumB+=stoneValue[i+j+l];
                    //cout<<"sumB:"<<sumB<<endl;
                    if (f[i+j+k]-sumB<Min) Min=f[i+j+k]-sumB;
                }
                //cout<<"Min: "<<Min<<endl;
                if (sum+Min>Max) Max=sum+Min;
            }
            f[i]=Max;
        }
        if (f[0]>0) return "Alice";
        else if (f[0]==0) return "Tie";
        else return "Bob";
    }
};
