class Solution {
  public:
      unordered_map<int,int> MEMORESET; 
      int TLQS[50010]={0};
      string stoneGameIII(vector<int>& stoneValue) {
          int n = stoneValue.size();
          TLQS[n-1] = stoneValue[n-1];
          for(int i=n-2; i>=0; --i)
              TLQS[i] = (TLQS[i+1]+stoneValue[i]);
          int MX= SOLVE( stoneValue, 0 );
          int mi=TLQS[0] - MX;
          if (MX>mi) return "Alice";
          else if (mi>MX) return "Bob";
          return "Tie";
      }
      
      int SOLVE(vector<int>& stoneValue, int i){
          if (i>=stoneValue.size()) return 0;
          if (MEMORESET.count(i)) return MEMORESET[i];
          int cur = stoneValue[i];
          int MX = cur +  TLQS[i+1] - SOLVE(stoneValue, i+1);
          int n = stoneValue.size();
          int u = min(i+2, n-1);
          for(int j=i+1; j<=u; ++j){
              cur += stoneValue[j];
              MX = max(MX, cur + TLQS[j+1] - SOLVE(stoneValue, j+1) );
          }
          return MEMORESET[i] = MX;
      }
  };
