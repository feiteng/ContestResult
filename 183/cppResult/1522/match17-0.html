class Solution {
public:
    string stoneGameIII(vector<int>& stone) {
        vector<int> dp(stone.size() + 1);
        reverse(stone.begin(), stone.end());
        
        for (int i=0;i<stone.size();++i) {
            int idx = i + 1;
            int best = stone[i] - dp[idx - 1];
            if (idx >= 2) {
                best = max(best, stone[i] + stone[i-1] - dp[idx-2]);
            }
            if (idx >= 3) {
                best = max(best, stone[i] + stone[i-1] + stone[i-2] - dp[idx-3]);
            }
            dp[idx] = best;
        }
        if (dp.back() == 0) return "Tie";
        else if (dp.back() > 0) return "Alice";
        else return "Bob";
        
    }
};
