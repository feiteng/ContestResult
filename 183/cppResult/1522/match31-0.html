class Solution {
    
    int best[50005] = {0};
    int left[50005] = {0};
        
public:
    string stoneGameIII(vector<int>& s) {
        for (int i=s.size()-1; i>=0; i--) {
            if (i==s.size()-1)
                left[i] = s[i];
            else
                left[i] = s[i] + left[i+1];
        }
        
        for (int i=0; i<s.size(); i++)
            best[i] = INT_MIN;
        
        for (int i=s.size()-1; i>=0; i--) {
            int now = 0;
            for (int j=0; j<3; j++) {
                if (i + j >= s.size()) break;
                
                now += s[i+j];
                best[i] = max(best[i], now + left[i+j+1] - best[i+j+1]);
            }
        }
        
        if (best[0] * 2 == left[0])
            return "Tie";
        else if (best[0] * 2 > left[0])
            return "Alice";
        else
            return "Bob";
    }
};
