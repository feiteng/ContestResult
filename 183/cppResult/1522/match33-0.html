const int INF=1e9;

class Solution {
    vector<int> S;
    int N;
    
    vector<int> res;
    int solve(int x)
    {
        assert(x<=N);
        if(x==N) return 0;
        
        if(res[x]!=-INF) return res[x];
        
        int ans=-INF;
        int sum=0;
        for(int i=x+1; i<=min(N, x+3); i++)
        {
            sum+=S[i-1];
            ans=max(ans, sum-solve(i));
        }
        
        res[x]=ans;
        return ans;
    }
    
public:
    string stoneGameIII(vector<int>& stoneValue) {
        S=stoneValue;
        N=S.size();
        
        res=vector<int>(N+1, -INF);
    
        int res=solve(0);
        //for(int s: S) res+=s;
        
        if(res>0) return "Alice";
        else if(res==0) return "Tie";
        else return "Bob";
        //return res;
    }
};
