class Solution {
public:
    string stoneGameIII(vector<int>& s) {
        int n = s.size();
        int total = accumulate(begin(s), end(s), 0);
        vector<int> dp(n + 1, INT_MIN);
        dp[n-1] = s[n-1];
        
        for (int i=n - 2; i>=0; --i)
        {
            int t = 0;
            for (int j=0; j<3 && i + j<n; ++j)
            {
                t += s[i + j];
                dp[i] = max(dp[i], t - (dp[i + j + 1] == INT_MIN? 0: dp[i + j + 1]));
            }
        }
        // for (auto x: dp)
        //     cout<<x<<" ";
        if (dp[0] == 0) return "Tie";
        else if (dp[0] > 0) return "Alice";
        else return "Bob";
    }
};
