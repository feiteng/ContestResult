class Solution {
public:
    int dp[50005];
    int pref[50005];
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        stoneValue.resize(n + 1);
        for (int i = n; i >= 1; i--) stoneValue[i] = stoneValue[i - 1];
        stoneValue[0] = 0;
        for (int i = 1; i <= n; i++) dp[i] = -1e9;
        for (int i = n; i >= 1; i--) pref[i] = pref[i + 1] + stoneValue[i];
        for (int i = n; i >= 1; i--) {
            for (int j = 1; j <= 3; j++) {
                dp[i] = max(dp[i], pref[i] - dp[i + j]);
            }
        }
        if (2 * dp[1] > pref[1]) return "Alice";
        else if (2 * dp[1] == pref[1]) return "Tie";
        else return "Bob";
    }
};
