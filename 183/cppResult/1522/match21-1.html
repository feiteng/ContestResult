class Solution {
 public:
  string stoneGameIII(vector<int>& stoneValue) {
    int n = stoneValue.size();
    vector<int> dp(n + 1, INT_MIN);
    dp[n] = 0;
    int suffsum = 0;
    for (int i = n - 1; i >= 0; i--) {
      suffsum += stoneValue[i];
      for (int j = 1; j <= 3; j++) {
        if (i + j <= n)
          dp[i] = max(dp[i], suffsum - dp[i + j]);
      }
    }
    int sum = 0;
    for (int val : stoneValue) 
      sum += val;
    bool can_tie = false;
    int a = dp[0], b = sum - a;

    if (a > b) {
      return "Alice";
    } else if (a == b) {
      return "Tie";
    } else {
      return "Bob";
    }
  }
};
