class Solution {
public:
    int totl;
    vector<int> DP;
    int find(vector<int> &A,int ind,int rem_sum)
    {
        if(ind>=A.size())
            return 0;
        if(DP[ind]!=-1)
            return DP[ind];
        int sum=0,ans=INT_MIN,i;
        for(i=ind;i<A.size() && i-ind<3;i++)
        {
            sum+=A[i];
            ans=max(ans,rem_sum-find(A,i+1,rem_sum-sum));
        }
        return DP[ind]=ans;
    }
    string stoneGameIII(vector<int>& A) 
    {
        int n=A.size(),i;
        DP.resize(n,-1);
        totl=0;
        for(i=0;i<n;i++)
            totl+=A[i];
        int cnt=find(A,0,totl);
        cout<<cnt<<endl;
        if(cnt>totl-cnt)
            return "Alice";
        if(cnt<totl-cnt)
            return "Bob";
        return "Tie";
    }
};
