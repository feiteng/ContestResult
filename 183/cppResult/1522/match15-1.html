class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        for (int i = 1; i < n; ++i)
        {
            stoneValue[i] += stoneValue[i - 1];
        }
        stoneValue.insert(stoneValue.begin(), 0);

        vector<int> f(n + 2, -50000012);
        f[n + 1] = 0;
        for (int i = n + 1; i >= 1; --i)
        {
            if (i - 1 >= 1) f[i - 1] = max(f[i - 1], stoneValue[i - 1] - stoneValue[i - 2] - f[i]);
            if (i - 2 >= 1) f[i - 2] = max(f[i - 2], stoneValue[i - 1] - stoneValue[i - 3] - f[i]);
            if (i - 3 >= 1) f[i - 3] = max(f[i - 3], stoneValue[i - 1] - stoneValue[i - 4] - f[i]);
        }
        if (f[1] == 0) return "Tie";
        if (f[1] > 0) return "Alice";
        return "Bob";
    }
};
