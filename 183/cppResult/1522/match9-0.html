class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> dp(n+1, INT_MIN);
        vector<int> suffixSum(n+1, 0);

        for (int i = n-1; i >= 0; --i) {
            suffixSum[i] = suffixSum[i+1] + stoneValue[i];
        }
        
        dp[n] = 0;
        for (int i = n-1; i >= 0; --i) {
            // cout << "i = " << i << endl;
            for (int k = i, tot = 0; k < min(n, i+3); ++k) {
                // cout << "k = " << k << endl;
                tot += stoneValue[k];
                dp[i] = max(dp[i], tot + suffixSum[k+1] - dp[k+1]);
            }
        }
        if (dp[0]*2 > suffixSum[0])
            return "Alice";
        if (dp[0]*2 < suffixSum[0])
            return "Bob";
        return "Tie";
    }
};
