class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> S(n+1, 0);
        for (int i = n-1; i >= 0; --i) {
            S[i] = S[i+1] + stoneValue[i];
        }
        vector<int> dp(n+1, 0);
        dp[n-1] = stoneValue[n-1];
        for (int i = n-2; i >= 0; --i) {
            int v1 = numeric_limits<int>::min();
            int v2 = numeric_limits<int>::min();
            int v3 = numeric_limits<int>::min();
            if (i + 1 <= n) {
                v1 = stoneValue[i] + (S[i+1] - dp[i+1]);
            }
            if (i + 2 <= n) {
                v2 = stoneValue[i] + stoneValue[i+1] + (S[i+2] - dp[i+2]);
            }
            if (i + 3 <= n) {
                v3 = stoneValue[i] + stoneValue[i+1] + stoneValue[i+2] + (S[i+3] - dp[i+3]);
            }
            dp[i] = max(v1, max(v2, v3));
        }
        int a = dp[0];
        int b = S[0] - a;
        
        cout << "A: " << a << endl;
        cout << "B: " << b << endl;
        
        if (a > b) {
            return "Alice";
        }
        else if (a < b) {
            return "Bob";
        }
        else {
            return "Tie";
        }
        
    }
};
