class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<vector<int>> dp(n + 1, vector<int>(2, 0));
        for (int i = n - 1; i >= 0; --i) {
            int cur = stoneValue[i];
            dp[i][0] = stoneValue[i] + dp[i + 1][1];
            dp[i][1] = dp[i + 1][0];
            for (int j = 1; j < 3 && i + j < n; ++j) {
                cur += stoneValue[i + j];
                if (dp[i][0] < cur + dp[i + j + 1][1]) {
                    dp[i][0] = cur + dp[i + j + 1][1];
                    dp[i][1] = dp[i + j + 1][0];
                }
            }
        }
        if (dp[0][0] > dp[0][1]) return "Alice";
        if (dp[0][0] < dp[0][1]) return "Bob";
        return "Tie";
    }
};
