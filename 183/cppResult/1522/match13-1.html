class Solution {
public:
    int n;
    int dp[2][50015]; //dp[0:alice]
    int a[50015];
    string stoneGameIII(vector<int>& stoneValue) {
        n = stoneValue.size();
        memset(a,0,sizeof(a));
        memset(dp,0,sizeof(dp));

        a[0]=0;
        for(int i=1;i<=n;i++) a[i]=a[i-1]+stoneValue[i-1];
        for(int i=n+1;i<=n+5;i++) a[i]=a[n];
        for(int i=n;i>=1;i--){
            dp[0][i]=INT_MIN;
            for(int j=1;j<=3;j++) 
                dp[0][i]=max(dp[0][i],dp[1][i+j]+a[i+j-1]-a[i-1]);
            dp[1][i]=INT_MAX;

            for(int j=1;j<=3;j++){
                dp[1][i]=min(dp[1][i],dp[0][i+j]-a[i+j-1]+a[i-1]);
                //printf("dp[%d][%d]=%d\n",1,i,dp[1][i]);
            }
        }
        //printf("%d\n",dp[1][3]);
        if(dp[0][1]>0) return "Alice";
        else if(dp[0][1]<0) return "Bob";
        else return "Tie";
    }
};
