class Solution {
public:
    int dp[50010];
    int dfs(int pos, int n, vector<int>& stone, vector<int>& suffix) {
        if (pos >= n) return 0;
        if (dp[pos] != -0x3f3f3f3f) return dp[pos];
        
        int ret = -0x3f3f3f3f, tmp = 0;
        for (int i = pos; i <= min(pos + 2, n-1); ++i) {
            tmp += stone[i];
            int s = (i + 1 >= n ? 0 : suffix[i+1]) - dfs(i+1, n, stone, suffix);

            int cur = tmp + s;
            ret = max(ret, cur);
        }
        return dp[pos] = ret;
    }
    string stoneGameIII(vector<int>& stone) {
        int sum = 0, n = stone.size();
        for (int i = 0; i < n; ++i) dp[i] = -0x3f3f3f3f;
        for (auto x: stone) sum += x;
        vector<int> suffix(n, 0);
        suffix[n-1] = stone[n-1];
        for (int i = n-2; i >= 0; --i) suffix[i] = suffix[i+1] + stone[i];
        
        int Max = dfs(0, n, stone, suffix);
        if (Max > sum - Max) return "Alice";
        else if (Max < sum - Max) return "Bob";
        else return "Tie";
    }
};
