class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        constexpr int INF = 1e9 + 7;
        int n = stoneValue.size();
        vector<int> dp(n + 1, -INF);
        dp[n] = 0;
        for (int i = n - 1; i >= 0; --i) {
            int cur = 0;
            for (int j = 0; j < 3 and i + j < n; ++j) {
                cur += stoneValue[i + j];
                dp[i] = max(dp[i], cur - dp[i + j + 1]);
            }
        }
        if (dp[0] == 0) {
            return "Tie";
        } else if (dp[0] < 0) {
            return "Bob";
        } else {
            return "Alice";
        }
        // cout << dp[0] << ' ' << total << '\n';
        return "";
    }
};
