class Solution {

    vector<int> dp;
    int assist(vector<int>& stoneValue, int start)
    {
        if (start >= stoneValue.size()) return 0;
        if (dp[start] != INT_MIN) return dp[start];
        int maxVal = INT_MIN;
        int sum = 0;
        for (int i = 0; i < 3; i++)
        {
            if (start + i >= stoneValue.size()) break;
            sum += stoneValue[start + i];
            maxVal = max(maxVal, sum - (assist(stoneValue, start + i+1)));
        }
        dp[start] = maxVal;
        return maxVal;
    }
public:
    string stoneGameIII(vector<int>& stoneValue) {

        int n = stoneValue.size();
        dp = vector<int>(n, INT_MIN);
        int val = assist(stoneValue, 0);
        if (val == 0) return "Tie";
        else if (val > 0) return "Alice";
        else return "Bob";
    }
};
