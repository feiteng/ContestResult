class Solution {
public:
    string stoneGameIII(vector<int>& a) {
		int n=a.size();
		int dp[50005];
		for(int i=0;i<=n+1;++i) dp[i]=-1e9;
		dp[n]=a[n-1];dp[n+1]=0;
		for (int i=n-1;i>=1;--i)
		{
			dp[i]=a[i-1]-dp[i+1];
			if (i+2<=n+1) dp[i]=max(dp[i],a[i-1]+a[i]-dp[i+2]);
			if (i+3<=n+1) dp[i]=max(dp[i],a[i-1]+a[i]+a[i+1]-dp[i+3]);
		}
		if (dp[1]==0) return "Tie";
		if (dp[1]>0) return "Alice";
		else return "Bob";
    }
};
