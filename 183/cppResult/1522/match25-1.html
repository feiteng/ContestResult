class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int len = stoneValue.size();
        vector<int> dp;
        dp.push_back(0);
        dp.push_back(stoneValue[len-1]);
        for (int i = 2; i <= len; i++) {
            int tmp_best = -100000000;
            int tmp_cur = 0;
            for (int j = 0; j < min(3, i); j++) {
                tmp_cur += stoneValue[len-i+j];
                if (tmp_cur - dp[i-j-1] > tmp_best) {
                    tmp_best = tmp_cur - dp[i-j-1];
                }
            }
            dp.push_back(tmp_best);
        }
        if (dp[len] > 0) return "Alice";
        if (dp[len] < 0) return "Bob";
        return "Tie";
    }
};
