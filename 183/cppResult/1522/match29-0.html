class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> dp1(n, INT_MIN);
        vector<int> dp2(n, INT_MAX);
        dp1.push_back(0), dp2.push_back(0);
        dp1[n - 1] = stoneValue[n - 1];
        dp2[n - 1] = -stoneValue[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            for (int j = i; j < i + 3 && j < n; j++) {
                int mys = 0;
                for (int k = i; k <= j; k++) mys += stoneValue[k];
                dp1[i] = max(dp1[i], dp2[j + 1] + mys);
            }
            for (int j = i; j < i + 3 && j < n; j++) {
                int mys = 0;
                for (int k = i; k <= j; k++) mys += stoneValue[k];
                dp2[i] = min(dp2[i], dp1[j + 1] - mys);
            }
        }
        if (dp1[0] == 0) return "Tie";
        else if (dp1[0] > 0) return "Alice";
        else return "Bob";       
    }
};
