class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        vector<int> dp(stoneValue.size(), 0);
        dp[stoneValue.size() - 1] = stoneValue.back();
        for (int i = stoneValue.size() - 2; i >= 0; i--) {
            dp[i] = stoneValue[i] - dp[i + 1];
            if (i == stoneValue.size() - 2) {
                dp[i] = max(dp[i], stoneValue[i] + stoneValue[i + 1]);
            } else if (i == stoneValue.size() - 3) {
                dp[i] = max(dp[i], stoneValue[i] + stoneValue[i + 1] - dp[i + 2]);
                dp[i] = max(dp[i], stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2]);
            } else {
                dp[i] = max(dp[i], stoneValue[i] + stoneValue[i + 1] - dp[i + 2]);
                dp[i] = max(dp[i], stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - dp[i + 3]);
            }
        }
        
        if (dp[0] > 0) {
            return "Alice";
        } else if (dp[0] < 0) {
            return "Bob";
        } else {
            return "Tie";
        }
    }
};
