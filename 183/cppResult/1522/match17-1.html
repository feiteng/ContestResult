class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        const int N = stoneValue.size();
        vector<int> dp(N, INT_MIN);
        dp.back() = stoneValue.back();


        for (int i = N - 2; i >= 0; --i) {
            int a = stoneValue[i] - dp[i + 1];

            int b = INT_MIN;
            if (i <= N - 2) {
                b = stoneValue[i] + stoneValue[i + 1];
                if (i < N - 2)
                    b -= dp[i + 2];
            }

            int c = INT_MIN;
            if (i <= N - 3) {
                c = stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2];
                if (i < N - 3)

                    c -= dp[i + 3];
            }

            dp[i] = max(max(a, b), c);
        }

        if (dp[0] > 0)
            return "Alice";
        else if (dp[0] < 0)
            return "Bob";
        else
            return "Tie";
    }
