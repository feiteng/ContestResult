class Solution {
public:
    string stoneGameIII(vector<int>& a) {
        reverse(a.begin(),a.end());
        const int n=a.size();
        vector<int> dp(n+1,-1e9);
        dp[0]=0;
        for(int i=1;i<=n;++i)
        {
            if(i>=1)
                dp[i]=max(dp[i],a[i-1]-dp[i-1]);
            if(i>=2)
                dp[i]=max(dp[i],a[i-2]+a[i-1]-dp[i-2]);
            if(i>=3)
                dp[i]=max(dp[i],a[i-3]+a[i-1]+a[i-2]-dp[i-3]);
        }
        
        if(dp[n]>0)
            return "Alice";
        if(dp[n]<0)
            return "Bob";
        return "Tie";
    }
};
