class Solution {
public:

    vector<int> dp;
    int f(int i, vector<int>& s)
    {
        if(i >= s.size())
            return 0;
        if(dp[i] != -1)
            return dp[i];
        int a = s[i] - f(i + 1, s);
        int b = 0, c = 0;
        if(i + 1 < s.size())
        {
            b = s[i] + s[i + 1] - f(i + 2, s);
            a = max(a, b);
        }
        if(i + 2 < s.size())
        {
            c = s[i] + s[i + 1] + s[i + 2] - f(i + 3, s);
            a = max(a, c);
        }
        dp[i] = a;
        return a;
    }

    string stoneGameIII(vector<int>& s) {
        int n = s.size();
        dp = vector<int> (n + 1, -1);
        int res = f(0, s);
        if(res > 0)
            return "Alice";
        else if(res < 0)
            return "Bob";
        else
            return "Tie";
    }
};
