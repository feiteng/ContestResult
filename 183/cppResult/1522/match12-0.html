class Solution {
public:
    
    int helper(vector<int>& v, int idx, vector<int>& vals, vector<int>& seen) {
        if (idx == v.size()) return 0;
        
        if (seen[idx]) return vals[idx];
        
        int ans = v[idx] - helper(v, idx+1, vals, seen);
        if (idx+1 < v.size()) {
            ans = max(ans, v[idx]+v[idx+1] - helper(v, idx+2, vals, seen));
        }
        if (idx+2 < v.size()) {
            ans = max(ans, v[idx]+v[idx+1]+v[idx+2] - helper(v, idx+3, vals, seen));
        }
        
        seen[idx] = true;
        vals[idx] = ans;
        return ans;
    }
    
    string stoneGameIII(vector<int>& v) {
        int n = v.size();
        vector<int> vals(n,0);
        vector<int> seen(n,false);
        int ans = helper(v, 0, vals, seen);
        if (ans == 0) {
            return "Tie";
        } else if (ans > 0) {
            return "Alice";
        } else {
            return "Bob";
        }
    }
};
