class Solution {
public:
    vector<int> sum;
    vector<int> dp;
    string stoneGameIII(vector<int>& stoneValue) {
        sum = vector<int>(stoneValue.size() + 1, 0);
        dp = vector<int>(stoneValue.size() + 1, 0);
        for (int i = stoneValue.size() - 1;i >= 0;i--) {
            sum[i] = stoneValue[i] + sum[i + 1];

            int ret = -0x7fffffff;
            int total = 0;
            for (int j = 0;j < 3 && i + j < stoneValue.size();j++) {
                total += stoneValue[i + j];
                ret = max(sum[i + j + 1] - dp[i + j + 1] + total, ret);
            }
            dp[i] = ret;
        }
        if (dp[0] > sum[0] - dp[0]) {
            return "Alice";
        } else if (dp[0] == sum[0] - dp[0]) {
            return "Tie";
        } else {
            return "Bob";
        }
    }
};
