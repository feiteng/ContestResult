int inf = 1e9;


class Solution {
public:
    string stoneGameIII(vector<int>& a) {
      int n = a.size();

      int s = 0;
      vector<int> dp(n + 1, -inf);


      dp[n] = 0;
      for(int i = n - 1; i >= 0; i--) {
          s += a[i];
          for(int j = 1; j <= 3 && i + j <= n; j++) {
            dp[i] = max(dp[i], s - dp[i + j]);
          }
      }
      if(dp[0] == s - dp[0]) return "Tie";
      else if (dp[0] > s - dp[0]) return "Alice";
      else return "Bob";
    }
};
