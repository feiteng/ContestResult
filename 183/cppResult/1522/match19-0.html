class Solution {
public:
    string stoneGameIII(vector<int>& a) {
        int n = a.size();
        vector<int> dp(n + 1);
        for (int i = n - 1; i >= 0; --i) {
            int cur = 0;
            dp[i] = -(1 << 30);
            for (int k = 0; k < 3 && i + k < n; ++k) {
                cur += a[i + k];
                dp[i] = max(dp[i], cur - dp[i + k + 1]);
            }
        }
        if (dp[0] > 0) return "Alice";
        else if (dp[0] < 0) return "Bob";
        return "Tie";
    }
};
