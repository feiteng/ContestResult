class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        reverse(stoneValue.begin(), stoneValue.end());
        vector<int> table(n, 0);
        for (int i = 0; i < n; i++) {
            int cursum = 0;
            table[i] = INT_MIN;
            for (int j = 1; j <= min(3, i + 1); j++) {
                cursum += stoneValue[i - j + 1];
                int gain = cursum;
                if (i - j >= 0) gain -= table[i - j];
                table[i] = max(table[i], gain);
            }
        }
//        for (int i = 0; i < n; i++) cout << table[i] << "\n";
        if (table[n - 1] > 0) return "Alice";
        if (table[n - 1] < 0) return "Bob";
        return "Tie";
    }
};
