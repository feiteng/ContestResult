class Solution {
public:
    const static int INF = 1e9;
    string stoneGameIII(vector<int>& stoneValue) {
        int n = stoneValue.size();
        vector<int> dp(n + 1, -INF);
        dp[0] = 0;
        int s = 0;
        for (int i = 1; i <= n; ++i) {
            s += stoneValue[n - i];
            for (int k = 1; k <= 3 && i - k >= 0; ++k) {
                dp[i] = max(dp[i], s - dp[i - k]);
            }
        }
        if (2 * dp[n] > s) return "Alice";
        if (2 * dp[n] < s) return "Bob";
        return "Tie";
    }
};
