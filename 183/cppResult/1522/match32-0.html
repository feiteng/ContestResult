class Solution {
 public:
  string stoneGameIII(vector<int>& a) {
    int size = a.size();
    vector<int> dp(size + 10, 0);
    for (int i = size - 1; i >= 0; i--) {
      dp[i] = a[i] - dp[i+1];
      if (i + 1 < size) {
        dp[i] = max(dp[i], a[i] + a[i+1] - dp[i+2]);
      }
      if (i + 2 < size) {
        dp[i] = max(dp[i], a[i] + a[i+1] + a[i+2] - dp[i+3]);
      }
    }
    if (dp[0] > 0) {
      return "Alice";
    } else if (dp[0] == 0) {
      return "Tie";
    } else {
      return "Bob";
    }
  }
};
