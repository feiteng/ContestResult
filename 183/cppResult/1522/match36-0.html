class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        reverse(stoneValue.begin(), stoneValue.end());
        int n=stoneValue.size();
        vector<int> dp(n+1,-0x7ffffff), sum(n+1,0);
        for (int i=0;i<n;i++)
            sum[i+1]=sum[i]+stoneValue[i];
        dp[0]=0;
        dp[1]=stoneValue[0];
        
        for (int i=2;i<=n;i++) {
            int t=stoneValue[i-1];
            // cout<<i<<endl;
            for (int j=i-1;j>=max(0, i-3);j--) {
                // cout<<j<<'-'<<t<<' ';
                dp[i]=max(dp[i], sum[j]-dp[j]+t);
                if (j)
                    t+=stoneValue[j-1];
            }
            // cout<<endl;
        }
        
        // for (int i=0;i<=n;i++)
        //     cout<<dp[i]<<' ';
        
        if (dp[n]>sum[n]-dp[n])
            return "Alice";
        
        else if (dp[n]==sum[n]-dp[n])
            return "Tie";
        
        else // (dp[n-1]<sum[n]/2)
            return "Bob";
    }
};
