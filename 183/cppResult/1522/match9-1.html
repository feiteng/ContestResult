class Solution {
public:
    string stoneGameIII(vector<int>& nums) {
        int len = nums.size();
        vector<int> dp(len+1,INT_MIN);
        vector<int> suff(len+1);
        for(int i = len-1;i>=0;i--)suff[i]+=suff[i+1]+nums[i];
        dp[len]=0;
        for(int i = len-1;i>=0;i--){
            int sum = 0;

            for(int j = 0;j<3&&(i+j)<len;++j){
                sum+=nums[i+j];
                dp[i]=max(dp[i],sum+suff[i+j+1]-dp[i+j+1]);
            }
        }
        if(dp[0]*2==suff[0])return "Tie";
        if(dp[0]*2>suff[0])return "Alice";
        return "Bob";
    }
};
