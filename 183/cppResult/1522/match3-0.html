class Solution {
public:
    string stoneGameIII(vector<int>& a) {
        int n = a.size();
        reverse(a.begin(),a.end());
        vector<int> pres(n+1,0);
        for(int i=1;i<=n;i++){
            pres[i]=pres[i-1]+a[i-1];
        }
        
        int NIL = 1e9+7;
        vector<int> dp(n+1,-NIL);
        dp[0] = 0;
        for(int i=1;i<=n;i++){
            int ts = 0;
            for(int j=i-1;j>=0 && j>=(i-3);j--){
                ts+=a[j];
                dp[i] = max(dp[i],pres[j]-dp[j]+ts);
            }
        }
        
        if(dp[n] < pres[n]-dp[n]){
            return "Bob";
        }
        if(dp[n] == pres[n]-dp[n]){
            return "Tie";
        }
        return "Alice";
    }
};
