class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int n=stoneValue.size();
        vector<int> tot(n+1,0);
        for (int i=n-1;i>=0;--i){
            tot[i]=tot[i+1]+stoneValue[i];
        }
        vector<int> Alice(n+1,0);
        vector<int> Bob(n+1,0);

        Alice[n-1]=stoneValue[n-1];
        Bob[n-1]=stoneValue[n-1];
        for (int i=n-2;i>=0;--i){
            Alice[i]=stoneValue[i]+tot[i+1]-Bob[i+1];
            if (i+2<=n) Alice[i]=max(Alice[i],stoneValue[i]+stoneValue[i+1]+tot[i+2]-Bob[i+2]);
            if (i+3<=n) Alice[i]=max(Alice[i],stoneValue[i]+stoneValue[i+1]+stoneValue[i+2]+tot[i+3]-Bob[i+3]);
            Bob[i]=stoneValue[i]+tot[i+1]-Alice[i+1];
            if (i+2<=n) Bob[i]=max(Bob[i],stoneValue[i]+stoneValue[i+1]+tot[i+2]-Alice[i+2]);
            if (i+3<=n) Bob[i]=max(Bob[i],stoneValue[i]+stoneValue[i+1]+stoneValue[i+2]+tot[i+3]-Alice[i+3]);
        }
        if (Alice[0]*2>tot[0]) return "Alice";
        else if (Alice[0]*2<tot[0]) return "Bob";
        else return "Tie";
    }
};
