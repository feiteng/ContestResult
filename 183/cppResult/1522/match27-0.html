class Solution {
public:
    string stoneGameIII(vector<int>& values) {
        int n = values.size();
        vector<int> post(n + 1, 0);
        for (int i = n - 1; i >= 0; i--) {
            post[i] = values[i] + post[i + 1];
        }
        vector<vector<int>> dp(2, vector<int>(n + 1, -1005 * 50000));
        dp[0][n] = 0;
        dp[1][n] = 0;
        for (int i = n - 1; i >= 0; i--) {
            int v = 0;
            for (int j = 0; j < 3 && i + j < n; j++) {
                int idx = i + j;
                v += values[idx];
                dp[0][i] = max(dp[0][i], v + post[idx + 1] - dp[1][idx + 1]);
                dp[1][i] = max(dp[1][i], v + post[idx + 1] - dp[0][idx + 1]);
            }
            // cout << dp[0][i] << " " << dp[1][i] << " " << i << endl;
        }
        int res = dp[0][0];
        int total = post[0];
        if (res > total - res) return "Alice";
        else if (res < total - res) return "Bob";
        else return "Tie";
    }
};
