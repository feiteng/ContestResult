class Solution {
public:
    string stoneGameIII(vector<int>& stoneValue) {
        int N=stoneValue.size();
        vector<vector<int>> dp(N+1,vector<int>(2,0));
        for(int i=N-1;i>=0;i--){
            int r=0;
            dp[i][0]=INT_MIN;
            dp[i][1]=INT_MAX;
            for(int j=i;j<=N-1&&j-i<=2;j++){
                r+=stoneValue[j];
                dp[i][0]=max(dp[i][0],r+dp[j+1][1]);
                dp[i][1]=min(dp[i][1],dp[j+1][0]);
            }
        }
        if(dp[0][0]>dp[0][1]) return "Alice";
        else if(dp[0][0]<dp[0][1]) return "Bob";
        else return "Tie";
    }
};
