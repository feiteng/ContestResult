class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int,char>> pq;
        if(a > 0) pq.push({a,'a'});
        if(b > 0) pq.push({b,'b'});
        if(c > 0) pq.push({c,'c'});
        string s = "";
        while(!pq.empty())
        {
            auto top = pq.top(); pq.pop();
            int n = s.size();
            if(top.first == 0) continue;
            if(n < 2 || (s[n-1] != top.second && top.first > 0) || (s[n-2]!=top.second && top.first > 0))
            {
                s = s.append(1,top.second);
                if(--top.first > 0) pq.push(top);
                //cout << top.first << "," << top.second << endl;
            }
            else
            {
                if(pq.empty()) return s;
                auto top2 = pq.top(); pq.pop();
                if(top2.first > 0)
                    s = s.append(1,top2.second);
                if(--top2.first > 0) pq.push(top2);
                cout << top2.first << top2.second << endl;
                pq.push(top);
            }
        }
        return s;
    }
};
