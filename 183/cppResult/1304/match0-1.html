class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        auto cmp = [](pair<int,char>&a,pair<int,char>&b){
            return a.first<b.first;
        };
        priority_queue<pair<int,char>, vector<pair<int,char>>, decltype(cmp)> q(cmp);
        if (a>0)
            q.push({a,'a'});
        if (b>0)
            q.push({b,'b'});
        if ( c>0 )
            q.push({c,'c'});
        string res;
        while( q.size()>0 ){
            int t = res.size();
             vector<pair<int,char>> tmp;
            auto cur = q.top(); q.pop();
            if ( res.size()>1 && cur.second == res.back() && (res[t-1]==res[t-2]) ){
                tmp.push_back(cur);
                if ( q.size()==0 ) break;
                cur = q.top(); q.pop();
            }
            res += cur.second;
            if (--cur.first>0)
                q.push(cur);
            for(auto&z:tmp)
                q.push(z);
        }
        return res;
    }
};
