class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res = "";
        int ca = 0, cb = 0, cc = 0;
        while (a + b + c > 0) {
            if (a >= b && a >= c) {
                if (ca < 2) {
                    res += "a";
                    --a;
                    ++ca;
                    cb = 0;
                    cc = 0;
                } else if (b > c) {
                    res += "b";
                    --b;
                    ca = 0;
                    cb = 1;
                    cc = 0;
                } else if (c > b) {
                    res += "c";
                    --c;
                    ca = 0;
                    cb = 0;
                    cc = 1;
                } else if (b > 0) {
                    res += "b";
                    --b;
                    ca = 0;
                    cb = 1;
                    cc = 0;
                } else {
                    return res;

                }
            } else if (b >= c && b >= a) {
                if (cb < 2) {
                    res += "b";
                    --b;
                    ++cb;
                    ca = 0;
                    cc = 0;
                } else if (a > c) {
                    res += "a";
                    --a;
                    ca = 1;
                    cb = 0;
                    cc = 0;
                } else if (c > a) {

                    res += "c";
                    --c;
                    ca = 0;
                    cb = 0;
                    cc = 1;
                } else if (a > 0) {
                    res += "a";
                    --a;
                    ca = 1;
                    cb = 0;
                    cc = 0;
                } else {
                    return res;

                }
            } else {
                if (cc < 2) {
                    res += "c";
                    --c;
                    ++cc;
                    cb = 0;
                    ca = 0;
                } else if (b > a) {
                    res += "b";
                    --b;
                    ca = 0;
                    cb = 1;
                    cc = 0;
                } else if (a > b) {
                    res += "a";
                    --a;
                    ca = 1;
                    cb = 0;
                    cc = 0;
                } else if (b > 0) {
                    res += "b";
                    --b;
                    ca = 0;
                    cb = 1;
                    cc = 0;
                } else {
                    return res;
                }
            }
        }
        return res;
    }
};
