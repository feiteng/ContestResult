class Solution {
public:
    static bool cmo(pair<int,char> &p1,pair<int,char> &p2){
        return p1.first < p2.first;
    }
    
    string longestDiverseString(int a, int b, int c) {
        string ans = "";
        vector<pair<int,char>> v(3);
        v[0] = make_pair(a,'a');
        v[1] = make_pair(b,'b');
        v[2] = make_pair(c,'c');
        
        while(v[2].first>0 || v[1].first>0 || v[0].first>0){
            int l = ans.size();
            char d = '#';
            if(ans.size()>=2 && ans[l-2]==ans[l-1]) d = ans[l-1];
            sort(v.begin(),v.end(),cmo);
            
      //      cout << d << " " << v[2].first << " " << v[1].first << " " << v[0].first << endl;
            
            if(v[2].second != d && v[2].first>0) {ans+=v[2].second;v[2].first--;}
            else if(v[1].second != d && v[1].first>0) {ans+=v[1].second;v[1].first--;}
            else if(v[0].second != d && v[0].first>0) {ans+=v[0].second;v[0].first--;}
            else break;
        }
        
        return ans;
    }
};
