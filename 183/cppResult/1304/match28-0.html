class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int,char>> pq;
        if(a > 0)pq.push({a,'a'});
        if(b > 0)pq.push({b,'b'});
        if(c > 0)pq.push({c,'c'});
        string s = "";
        while(!pq.empty()){
            pair<int,char> x1 = pq.top();
            pq.pop();
            int len = s.size();
            if(len - 2 >= 0 && s[len - 2] == s[len - 1] && s[len - 1] == x1.second){
                if(pq.empty())return s;
                pair<int,char> x2 = pq.top();
                pq.pop();
                s += x2.second;
                if(--x2.first > 0){
                    pq.push(x2);
                }
                pq.push(x1);
            }else{
                s += x1.second;
                if(--x1.first > 0){
                    pq.push(x1);
                }
            }
        }
        return s;
        
    }
};
