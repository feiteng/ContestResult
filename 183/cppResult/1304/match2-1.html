class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ret;
        using pii = pair<int, int>;
        vector<pii> v = {{a, 0} , {b, 1} , {c, 2}};
        sort(v.begin(), v.end());
        int last = -1;
        while(true) {
            string pre = ret;
            for(int i = 2; i >= 0; i--) {
                if(v[i].first > 0 && v[i].second != last) {
                    if( i != 2) {
                        ret.push_back('a' + v[i].second);
                        v[i].first --;
                    }
                    else if( v[i].first - v[i-1].first >= 2)  {
                        ret.push_back('a' + v[i].second); 
                        ret.push_back('a' + v[i].second);
                        v[i].first -= 2;
                    }

                    else {
                        ret.push_back('a' + v[i].second);
                        v[i].first --;
                    }
                    last = v[i].second;
                    break;
                }
            }
            if(pre == ret) return ret;
            sort(v.begin(), v.end());
            
        }
        return "-1";
    }
};
