class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string s("");
        int x=2, y=2, z=2;
        while(true){
            char d='d';
            int tmp=-1;
            if(x>0 && tmp<a && a>0){
                d='a';
                tmp=a;
            }
            if(y>0 && tmp<b && b>0){
                d='b';
                tmp=b;
            }
            if(z>0 && tmp<c && c>0){
                d='c';
                tmp=c;
            }
            
            if(d=='a'){
                y=z=2;
                x--;
                a--;
                s+=d;
            }
            else if(d=='b'){
                x=z=2;
                y--;
                b--;
                s+=d;
            }
            else  if(d=='c'){
                x=y=2;
                z--;
                c--;
                s+=d;
            }
            else
                break;
        }
        
        return s;
    }
};
