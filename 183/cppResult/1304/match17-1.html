class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string s;
        int i = 0;
        while (a + b + c) {

            if (a >= b && a >= c) {
                if (i >= 2 && s[i - 1] == 'a' && s[i - 2] == 'a') {
                    if (b && b >= c) {
                        s += 'b';
                        --b;
                    } else if (c && c >= b) {
                        s += 'c';
                        --c;
                    } else {
                        break;
                    }
                } else {
                    s += 'a';
                    --a;
                }
            } else if (b >= a && b >= c) {

                if (i >= 2 && s[i - 1] == 'b' && s[i - 2] == 'b') {
                    if (a && a >= c) {
                        s += 'a';
                        --a;
                    } else if (c && c >= a) {
                        s += 'c';
                        --c;
                    } else {
                        break;
                    }
                } else {
                    s += 'b';
                    --b;
                }
            } else if (c >= a && c >= b) {

                if (i >= 2 && s[i - 1] == 'c' && s[i - 2] == 'c') {
                    if (b && b >= a) {
                        s += 'b';
                        --b;
                    } else if (a && a >= b) {
                        s += 'a';
                        --a;
                    } else {
                        break;
                    }
                } else {
                    s += 'c';
                    --c;
                }
            }
            ++i;
        }
        return s;
    }
};
