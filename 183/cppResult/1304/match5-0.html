class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string r = "";
        int t1 = 0, t2 = 0, t3 = 0;        
        while (true) {
            if (t1 < 2 && t2 < 2 && t3 < 2) {
                if (a >= b && a >= c && a) {
                    t1++;
                    a--;
                    t2 = 0;
                    t3 = 0;
                    r += "a";
                } else if (b >= a && b >= c && b) {
                    t2++;
                    b--;
                    t1 = 0;
                    t3 = 0;
                    r += "b";
                } else if (c >= a && c >= b && c) {
                    t3++;
                    c--;
                    t1 = 0;
                    t2 = 0;
                    r += "c";
                } else {
                    break;
                }
            } else if (t1 == 2) {
                if (b >= c && b) {
                    t1 = 0;
                    t2++;
                    b--;
                    t3 = 0;
                    r += "b";
                } else if (c >= b && c) {
                    t1 = 0;
                    t2 = 0;
                    c--;
                    t3++;
                    r += "c";
                } else {
                    break;
                }
            } else if (t2 == 2) {
                if (a >= c && a) {
                    t1++;
                    t2 = 0;
                    a--;
                    t3 = 0;
                    r += "a";
                } else if (c >= a && c) {
                    t1 = 0;
                    t2 = 0;
                    c--;
                    t3++;
                    r += "c";
                } else {
                    break;
                }
            } else {
                if (a >= b && a) {
                    t1++;
                    t2 = 0;
                    a--;
                    t3 = 0;
                    r += "a";
                } else if (b >= a && b) {
                    t1 = 0;
                    t2++;
                    b--;
                    t3 = 0;
                    r += "b";
                } else {
                    break;
                }
            }
        }
        return r;
    }
};
