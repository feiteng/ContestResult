class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        // always append longest char
        vector<vector<int>> p = {{a, 0}, {b, 1}, {c, 2}};
        sort(p.begin(), p.end());
        string ret;
        
        while (p[2][0] > 0) {
            // cannot add largest
            if (!ret.empty() && p[2][1] + 'a' == ret.back()) {
                if (p[1][0] == 0) break;
                ret += string(1, p[1][1] + 'a');
                p[1][0] -= 1;
            } else {
                ret += string(min(2, p[2][0]), p[2][1] + 'a');
                p[2][0] -= min(2, p[2][0]);
            }
            sort(p.begin(), p.end());
        }
        return ret;
    }
};
