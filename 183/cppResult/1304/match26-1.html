class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        vector<pair<int, char>> pairs = {{a, 'a'}, {b, 'b'}, {c, 'c'}};
        sort(pairs.begin(), pairs.end());
        string r;
        
        if (pairs[2].first > 1) {
            r = string(2, pairs[2].second);
            pairs[2].first -= 2;
        } else {
            r = string(1, pairs[2].second);
            pairs[2].first -= 1;
        }
        
        sort(pairs.begin(), pairs.end());
        
        while(pairs[2].first > 0) {
            if (pairs[2].second == r.back()) {
                if (pairs[1].first == 0) {
                    break;
                }
                

                r += string(1, pairs[1].second);
                --pairs[1].first;
            } else {
                r += string(min(2, pairs[2].first), pairs[2].second);
                pairs[2].first -= min(2, pairs[2].first);
            }
            sort(pairs.begin(), pairs.end());
        }
        return r;
    }
};
