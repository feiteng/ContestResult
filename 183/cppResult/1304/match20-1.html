class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        set<pair<int, char>> count;
        if (a > 0)
            count.insert(make_pair(-a, 'a'));
        if (b > 0)
            count.insert(make_pair(-b, 'b'));
        if (c > 0)
            count.insert(make_pair(-c, 'c'));
        
        string res;
        
        while (!count.empty()) {
            auto top = *count.begin();
            count.erase(count.begin());
            if (res.size() < 2 || res[res.size()-2] != res[res.size()-1] || res[res.size()-1] != top.second) {
                res.push_back(top.second);
                top.first+=1;
                if (top.first != 0)
                    count.insert(top);
            }
            else if (!count.empty()) {
                auto t = *count.begin();
                count.erase(count.begin());
                res.push_back(t.second);
                t.first++;
                if (t.first != 0)
                    count.insert(t);
                count.insert(top);
            }
            else
                break;
            //cout << res << endl;
        }
        return res;
    }
};
