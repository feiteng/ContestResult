class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res = "";
        map<char, int> mp;
        mp['a'] = a, mp['b'] = b, mp['c'] = c;
        while(mp['a']!=0 || mp['b']!=0 || mp['c']!=0)
        {
            if(mp['a']>=mp['b'] && mp['a']>=mp['c']) 
            {
                if(res.size()>1 && res[res.size()-1]=='a' && res[res.size()-2]=='a')
                {
                    if(mp['b'] == 0 && mp['c'] == 0) break;
                    if(mp['b']>mp['c']) res += 'b', mp['b']--;
                    else res += 'c', mp['c']--;
                }
                else res += 'a', mp['a']--;
            }
            else if(mp['b']>=mp['a'] && mp['b']>=mp['c'])
            {
                if(res.size()>1 && res[res.size()-1]=='b' && res[res.size()-2]=='b')
                {
                    if(mp['a'] == 0 && mp['c'] == 0) break;
                    if(mp['a']>mp['c']) res += 'a', mp['a']--;
                    else res += 'c', mp['c']--;
                }
                else res += 'b', mp['b']--;
            }
            else if(mp['c']>=mp['a'] && mp['c']>=mp['b'])
            {
                if(res.size()>1 && res[res.size()-1]=='c' && res[res.size()-2]=='c')
                {
                    if(mp['a'] == 0 && mp['b'] == 0) break;
                    if(mp['a']>mp['b']) res += 'a', mp['a']--;
                    else res += 'b', mp['b']--;
                }
                else res += 'c', mp['c']--;
            }
        }
        return res;
    }
};
