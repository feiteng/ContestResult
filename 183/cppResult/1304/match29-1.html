class Solution {

public:
    string longestDiverseString(int a, int b, int c) {

        string ans = "";
        vector<int> cnt = {a,b,c}; 
        while(ans.length() < a+b+c) {
            int index = -1;
            int  val = 0;
            for(int i = 0; i<=2; i++){
                if(   !((ans.length()>=2) && (ans.back() == 'a'+i)
                  && (ans[ans.length()-2] == ans.back()) ) ){
                    if(val < cnt[i])
                    {
                        val = cnt[i];
                        index = i;
                    }
                }
            }
            //cout << ans << " " << index << endl;
            if(index == -1)
                return ans;
            ans+='a'+index;
            --cnt[index];
        }
        return ans;
    }
};
