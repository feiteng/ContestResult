class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ans = "";
        int l = 0;
        while (a || b || c) {
            if (a && a >= b && a >= c) {
                if (l >= 2 && ans[l - 1] == 'a' && ans[l - 2] == 'a') {
                    if (b && b >= c) {
                        ans.append("b");
                        b--;
                    }
                    else if (c) {
                        ans.append("c");
                        c--;
                    }
                    else break;

                }
                else {
                    ans.append("a");
                    a--;
                }
            }
            else if (b && b >= c) {
                if (l >= 2 && ans[l - 1] == 'b' && ans[l - 2] == 'b') {
                    if (a && a >= c) {
                        ans.append("a");
                        a--;
                    }
                    else if (c) {
                        ans.append("c");
                        c--;
                    }
                    else break;
                }
                else {
                    ans.append("b");
                    b--;
                }
            }

            else if (c) {
                if (l >= 2 && ans[l - 1] == 'c' && ans[l - 2] == 'c') {
                    if (a && a >= b) {
                        ans.append("a");
                        a--;
                    }
                    else if (b) {
                        ans.append("b");
                        b--;
                    }
                    else break;
                }
                else {
                    ans.append("c");
                    c--;
                }
            }
            else break;
            l++;
        }
        return ans;
    }
};
