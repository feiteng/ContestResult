class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ret="";
        int v[3];
        v[0]=a; v[1]=b; v[2]=c;
        int now=-1;
        int cnt=0;
        while (v[0]+v[1]+v[2]>0)
        {
            int p=-1;
            int big=-1;
            for (int i=0;i<3;i++) 
            {
                if (v[i]>big) 
                {
                    big=v[i];
                    p=i;
                }
            }
            if (p==now)
            {
                if (cnt<2)
                {
                    cnt++;
                    ret=ret+char('a'+p);
                    v[p]--;
                    continue;
                }
                else
                {
                    
                    int t=p;
                    big=-1;
                    p=-1;
                    for (int i=0;i<3;i++)
                    {
                        if (i==t) continue;
                        if (v[i]>big)
                        {
                            big=v[i];
                            p=i;
                        }
                    }
                    if (big==0) break;
                    now=p;
                    cnt=1;
                    ret=ret+char('a'+p);
                    v[p]--;
                    continue;
                }
            }
            else
            {
                now=p;
                cnt=1;
                ret=ret+char('a'+p);
                v[p]--;
                continue;
            }
            
        }
        return ret;
    }
};
