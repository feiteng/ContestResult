class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string s;
        int n=0;
        while(a>0||b>0||c>0){
            bool fa;
            bool fb;
            bool fc;
            fa=(a>0&&(n<2||s[n-1]!='a'||s[n-2]!='a'));
            fb=(b>0&&(n<2||s[n-1]!='b'||s[n-2]!='b'));

            fc=(c>0&&(n<2||s[n-1]!='c'||s[n-2]!='c'));
                      
            if (fa&&(a>=b||!fb)&&(a>=c||!fc)) {
                s.push_back('a');
                a--;
                n++;
            } else if (fb&&(b>=a || !fa)&&(b>=c||!fc)) {
                s.push_back('b');
                b--;
                n++;
            } else if (fc&&(c>=a||!fa)&&(c>=b||!fb)){
                s.push_back('c');
                c--;
                n++;
            } else break;
        }
        return s;
    }
};
