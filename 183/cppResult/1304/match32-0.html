class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        vector<int> flag(3, 0);
        string res = "";
        
        while(a || b || c) {
            if(flag[0] < 2 && a && (a >= b || flag[1] == 2) && (a >= c || flag[2] == 2)) {
                res += "a";
                --a;
                flag[0]++;
                flag[1] = 0;
                flag[2] = 0;
            }
            else if(flag[1] < 2 && b && (b >= a || flag[0] == 2) && (b >= c || flag[2] == 2)) {
                res += "b";
                --b;
                flag[1]++;
                flag[0] = 0;
                flag[2] = 0;
            }
            else if(flag[2] < 2 && c && (c >= b || flag[1] == 2) && (c >= a || flag[0] == 2)) {
                res += "c";
                --c;
                flag[2]++;
                flag[0] = 0;
                flag[1] = 0;
            } else break;
        }
        
        return res;
    }
};
