class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int, char>> q;
        if (a) q.push({a, 'a'});
        if (b) q.push({b, 'b'});
        if (c) q.push({c, 'c'});
        
        string res;
        while(!q.empty())
        {
            auto t = q.top(); q.pop();
            if (t.first == 0) break;
            else if (t.first == 1) 
                res += string(t.first, t.second);
            else if (t.first == 2) 
            {
                if (res.size() && res.back() == t.second)
                {
                    res += string(1, t.second);
                    t.first--;
                    q.push(t);
                }
                else
                    res += string(t.first, t.second);
            }
            else if (t.first >= 3)
            {
                if (!q.empty())
                {
                    auto t2 = q.top(); q.pop();
                    if (res.size() && res.back() == t.second)
                    {
                        res += string(1, t.second) + string(1, t2.second);
                        t.first--;
                        t2.first--;
                    }
                    else
                    {
                        res += string(2, t.second) + string(1, t2.second);
                        t.first -= 2;
                        t2.first -= 1;
                    }
                    q.push(t);
                    if (t2.first > 0) q.push(t2);
                }
                else
                {
                    res += string(2, t.second);
                    break;
                }
            }
        }
        return res;
    }
};
