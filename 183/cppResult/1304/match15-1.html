class Solution {
public:
    struct cmp{
        bool operator()(pair<int,int> p1, pair<int,int> p2){
            return p1.first<p2.first;
        }
    };
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int,int>,vector<pair<int,int>>,cmp> pq;
        pq.push({a,0});
        pq.push({b,1});
        pq.push({c,2});
        vector<int> cnt(3,0);
        cnt[0]=a;
        cnt[1]=b;
        cnt[2]=c;
        string res="";
        int idx=0;
        int candi=0;
        while(true){
            int n=res.size();

            if(res.size()>=2 && res[n-1]==res[n-2]){
                if(res[n-1]=='a'){
                    if(cnt[1]>0 && cnt[1]>=cnt[2]){
                        res+='b';
                        cnt[1]--;
                    }else if(cnt[2]>0){
                        res+='c';
                        cnt[2]--;
                    }

                }else if(res[n-1]=='b'){
                    if(cnt[0]>0 && cnt[0]>=cnt[2]){
                        res+='a';
                        cnt[0]--;
                    }else if(cnt[2]>0){
                        res+='c';
                        cnt[2]--;
                    }

                }else{
                    if(cnt[0]>0 && cnt[0]>=cnt[1]){
                        res+='a';
                        cnt[0]--;
                    }else if(cnt[1]>0){
                        res+='b';
                        cnt[1]--;
                    }
                }
            }else{
                int m=max(cnt[0],max(cnt[1],cnt[2]));
                if(m==0) break;

                if(m==cnt[0]){
                    res+='a';
                    cnt[0]--;
                }else if(m==cnt[1]){
                    res+='b';
                    cnt[1]--;
                }else {
                    res+='c';
                    cnt[2]--;
                }
            }
            if(cnt[0]==0 && cnt[1]==0 && cnt[2]==0) break;
            if(res.size()==n) break;
        }
       
        return res;
    }
};
