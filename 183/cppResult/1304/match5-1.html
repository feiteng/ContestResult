class Solution {
public:
	string longestDiverseString(int a, int b, int c) {
		string ans = "";
		int ca = 0, cb = 0, cc = 0;
		while (a + b + c > 0) {
			if (a >= b && a >= c) {
				if (ca != 2) {
					ca++;
					cb = 0;
					cc = 0;
					ans += 'a';
					a--;
				}
				else if (b == 0 && c == 0)
					break;
				else {
					ca = 0;
					if (b >= c) {
						cb++;
						cc = 0;
						ans += 'b';
						b--;
					}
					else {
						cc++;
						cb = 0;
						ans += 'c';
						c--;
					}

				}
			}

			// b zuida
			else if (b >= a && b >= c) {
				if (cb != 2) {
					cb++;
					ca = 0;
					cc = 0;
					ans += 'b';
					b--;
				}
				else if (a == 0 && c == 0)
					break;
				else {
					cb = 0;
					if (a >= c) {
						ca++;
						cc = 0;
						ans += 'a';
						a--;
					}
					else {
						cc++;
						ca = 0;
						ans += 'c';
						c--;
					}

				}
			}

			// c zuida
			else if (c >= a && c >= b) {
				if (cc != 2) {
					cc++;
					ca = 0;
					cb = 0;
					ans += 'c';
					c--;
				}
				else if (a == 0 && b == 0)
					break;
				else {
					cc = 0;
					if (a >= b) {
						ca++;
						cb = 0;

						ans += 'a';
						a--;
					}
					else {
						cb++;
						ca = 0;
						ans += 'b';
						b--;
					}
				}
			}
		}
		return ans;
	}
};
