class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        
        string res = "";
        char avoid = 'd';
        
        while (1) {
            
            if (a >= b && a >= c) {
                
                if (avoid == 'a') {
                    
                    if (b <= 0 && c <= 0) return res;
                    
                    if (b >= c) {
                        
                        res.push_back('b');
                        --b;
                        
                    } else {
                        
                        res.push_back('c');
                        --c;
                        
                    }
                    
                } else {
                    
                    if (a <= 0) return res;
                    
                    res.push_back('a');
                    --a;
                    
                }
                
                
            } else if (b >= a && b >= c) {
                
                if (avoid == 'b') {
                    
                    if (a <= 0 && c <= 0) return res;
                    
                    if (a >= c) {
                        
                        res.push_back('a');
                        --a;
                        
                    } else {
                        
                        res.push_back('c');
                        --c;
                        
                    }
                    
                } else {
                    
                    if (b <= 0) return res;
                    
                    res.push_back('b');
                    --b;
                    
                }
                
            } else {
                
                if (avoid == 'c') {
                    
                    if (a <= 0 && b <= 0) return res;
                    
                    if (a >= b) {
                        
                        res.push_back('a');
                        --a;
                        
                    } else {
                        
                        res.push_back('b');
                        --b;
                        
                    }
                    
                } else {
                    
                    if (c <= 0) return res;
                    
                    res.push_back('c');
                    --c;
                    
                }
                
            }
            
            if (res.length() >= 2 && (res.back() == res[res.length() - 2])) {
                avoid = res.back();
            } else avoid = 'd';
        }
        
        return res;
    }
};
