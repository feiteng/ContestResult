class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue< pair<int, char> > pq;
        if(a > 0)
            pq.push({a, 'a'});
        if(b > 0)
            pq.push({b, 'b'});
        if(c > 0)
            pq.push({c, 'c'});
        string s;
        while(!pq.empty())
        {
            auto p = pq.top();
            pq.pop();
            if(s.size() >= 2 && s[s.size() - 1] == p.second && s[s.size() - 2] == p.second)
            {
                if(!pq.empty())
                {
                    auto p2 = pq.top();
                    pq.pop();
                    s.push_back(p2.second);
                    p2.first -= 1;
                    if(p2.first > 0)
                        pq.push(p2);
                }
                else
                    break;
            }
            s.push_back(p.second);
            p.first -= 1;
            if(p.first > 0)
                pq.push(p);
        }
        return s;
    }
};
