class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ret = "";
        vector <int> sur;
        sur.push_back(a);
        sur.push_back(b);
        sur.push_back(c);
        while (true) {
            int ban = -1;
            if (ret.size() >= 2 && ret[ret.size() - 1] == ret[ret.size() - 2]) {
                ban = ret[ret.size() - 1] - 'a';
            }
            int sel = -1;
            for (int i = 0; i < 3; i++) {
                if (i == ban || sur[i] == 0) {
                    continue;
                }
                if (sel == -1 || sur[i] > sur[sel]) {
                    sel = i;
                }
            }
            if (sel == -1) {
                break;
            }
            ret += ('a' + sel);
            sur[sel]--;
            
        }
        
        return ret;
    }
};
