class Solution {
public:
    string longestDiverseString(int a, int b, int c, char ca = 'a', char cb = 'b', char cc = 'c') {
        a = min(a, (b + c + 1) * 2);
        b = min(b, (a + c + 1) * 2);
        c = min(c, (a + b + 1) * 2);
        string s;
        if(a >= b && b >= c){
            //cout << a << b << c << ca << cb << cc << endl;
            while(a){
                //cout << a << b << c << ca << cb << cc << endl;
                if(b == 0 && c == 0){
                    while(a){
                        s.push_back(ca);
                        --a;
                    }
                    return s;
                }
                //cout << a << b << c << ca << cb << cc << endl;
                while((b || c) && (a > b && a > c)){
                    s.push_back(ca);
                    s.push_back(ca);
                    a -= 2;
                    if(b >= c){
                        s.push_back(cb);
                        --b;
                    }else{
                        s.push_back(cc);
                        --c;
                    }
                    if(b > a){
                        s.push_back(cb);
                        --b;
                    }
                    if(c > a){
                        s.push_back(cc);
                        --c;
                    }
                }
                //cout << a << b << c << ca << cb << cc << endl;
                if(b == 0 && c == 0) continue;
                while((b || c) && (a == b || a == c)){
                    s.push_back(ca);
                    if(a == b){
                        s.push_back(cb);
                        --b;
                    }
                    if(a == c){
                        s.push_back(cc);
                        --c;
                    }
                    --a;
                }
                //cout << a << b << c << ca << cb << cc << endl;
            }
            return s;
        }else{
            if(a >= c && c >= b) return longestDiverseString(a, c, b, 'a', 'c', 'b');
            if(b >= a && a >= c) return longestDiverseString(b, a, c, 'b', 'a', 'c');
            if(b >= c && c >= a) return longestDiverseString(b, c, a, 'b', 'c', 'a');
            if(c >= a && a >= b) return longestDiverseString(c, a, b, 'c', 'a', 'b');
            if(c >= b && b >= a) return longestDiverseString(c, b, a, 'c', 'b', 'a');
        }
        return s;
    }
};
