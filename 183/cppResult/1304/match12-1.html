class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res = "";

        priority_queue<pair<int, char>> hq;
        if(a > 0) hq.push({a, 'a'});
        if(b > 0) hq.push({b, 'b'});
        if(c > 0) hq.push({c, 'c'});
        while(!hq.empty()){
            auto first = hq.top(); hq.pop();
            if(res.size() >= 2 && res[res.size()-1] == first.second && res[res.size()-2] == first.second){
                if(hq.empty()) return res;
                auto second = hq.top(); hq.pop();
                res.push_back(second.second);
                if(second.first - 1 > 0) hq.push({second.first - 1, second.second});
                hq.push(first);
            }
            else{
                res.push_back(first.second);
                if(first.first - 1 > 0) hq.push({first.first - 1, first.second});
            }
        }
        return res;
    }
};
