class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ans;
        int n=0;
        while (a+b+c>0)
        {
            int Max=max(a,max(b,c));
            if (Max==a)
            {
                if (n>=2 && ans[n-1]=='a' && ans[n-2]=='a')
                {
                    Max=max(b,c);

                    if (Max==0) break;
                    if (b==Max) n++,ans.push_back('b'),--b;
                    else n++,ans.push_back('c'),--c;
                }
                else n++,ans.push_back('a'),--a;
            }
            else if (Max==b)
            {
                if (n>=2 && ans[n-1]=='b' && ans[n-2]=='b')
                {
                    Max=max(a,c);

                    if (Max==0) break;
                    if (a==Max) n++,ans.push_back('a'),--a;
                    else n++,ans.push_back('c'),--c;
                }
                else n++,ans.push_back('b'),--b;
            }
            else
            {
                if (n>=2 && ans[n-1]=='c' && ans[n-2]=='c')
                {
                    Max=max(a,b);

                    if (Max==0) break;
                    if (a==Max) n++,ans.push_back('a'),--a;
                    else n++,ans.push_back('b'),--b;
                }
                else n++,ans.push_back('c'),--c;
            }
        }
        return ans;
    }
};
