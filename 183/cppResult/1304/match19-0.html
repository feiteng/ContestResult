class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int, char>> pq;
        if (a) pq.emplace(a, 'a');
        if (b) pq.emplace(b, 'b');
        if (c) pq.emplace(c, 'c');
        
        string ret;
        
        while (!pq.empty()) {
            auto t = pq.top();
            pq.pop();
            if (ret.size() < 2) {
                ret.push_back(t.second);
                t.first--;
                if (t.first > 0) pq.push(t);
            }
            else {
                int n = ret.size();
                char x = ret[n-1], y = ret[n-2];
                // cout << t.second << " " << x << " " << y << "\n";
                if (x == y && y == t.second) {
                    if (!pq.empty()) {
                        auto s = pq.top();
                        pq.pop();
                        ret.push_back(s.second);
                        s.first--;
                        if (s.first > 0) pq.push(s);
                        if (t.first > 0) pq.push(t);
                    }
                    else {
                        break;
                    }
                }
                else {
                    ret.push_back(t.second);
                    t.first--;
                    if (t.first > 0) pq.push(t);
                }
            }
        }
        
        return ret;
    }
};
