class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        int aux[4]={0,a,b,c};
        string ans;
        while(1){
            int can=0;
            for(int i=0;i<3;i++){
                char cc=(char)(i+97);
                if(aux[i+1]>aux[can]&&((ans.length()<2)||!(ans[(int)ans.length()-1]==cc&&ans[(int)ans.length()-2]==cc))){
                    can=i+1;
                }
            }
            if(!can)
                break;
            aux[can]--;
            ans.push_back((char)(can-1+97));
        }
        return ans;
    }
};
