class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int, char>> q;
        if(a) q.push(pair(a,'a'));
        if(b) q.push(pair(b,'b'));
        if(c) q.push(pair(c,'c'));
        string res;
        res.reserve(a+b+c);
        while(!q.empty()) {
            auto t=q.top();
            q.pop();
            res+=t.second;
            if(--t.first>0) {
                res+=t.second;
                --t.first;
            }
            if(q.empty()) break;
            if(t.first<1) continue;
            auto t2=q.top();
            if(t2.first<=t.first) {
                q.pop();
                res+=t2.second;
                if(--t2.first>0) q.push(t2);
            }
            q.push(t);
        }
        return res;
    }
};
