class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        priority_queue<pair<int, char>> q;
        
        if(a!=0)q.push({a, 'a'});
        if(b!=0)q.push({b, 'b'});
        if(c!=0)q.push({c, 'c'});
        
        string res;
        pair<int, char> cand = {0, 'a'};
        
        while(!q.empty())
        {
            auto cur = q.top();q.pop();
            
            int thre = 2;
            if(cur.first < cand.first) thre = 1;
            int cnt = min(thre, cur.first);
            res +=string(cnt, cur.second);
            
            if(cand.first > 0) q.push(cand);
            cur.first -= cnt;
            cand = cur;
        }
        
        return res;
    }
};
