class Solution {
 public:
  string longestDiverseString(int a, int b, int c) {
    priority_queue<pair<int, char>> pq;
    if (a > 0) {
      pq.emplace(a, 'a');
    }
    if (b > 0) {
      pq.emplace(b, 'b');
    }
    if (c > 0) {
      pq.emplace(c, 'c');
    }
    string ans;
    while (!pq.empty()) {
      auto [cnt, ch] = pq.top();
      pq.pop();
      if (!ans.empty() && ch == ans.back()) {
        if (pq.empty()) {
          return ans;
        }
        auto [cnt2, ch2] = pq.top();
        pq.pop();
        ans += ch2;
        pq.emplace(cnt, ch);
        if (--cnt2 != 0) {
          pq.emplace(cnt2, ch2);
        }
        continue;
      }
      ans += string(min(2, cnt), ch);
      if ((cnt -= min(2, cnt)) != 0) {
        pq.emplace(cnt, ch);
      }
    }
    return ans;
  }
};
