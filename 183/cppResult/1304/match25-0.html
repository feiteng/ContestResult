class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string st="";
        int cntc=0,cnta=0,cntb=0;
        while(1)
        {
            int ma=max(a,max(b,c));
            if(ma==0)
                break;
            if(ma==a&&a>0)
            {
                if(cnta<2)
                {
                    st+='a';
                    cnta++;
                    a--;
                    cntc=cntb=0;
                }
                else
                {
                    int ma2=max(b,c);
                    if(ma2==b&&b>0)
                    {
                        cnta=0;
                        st+='b';
                        b--;
                        cntb++;
                    }
                
                    else if(ma2==c&&c>0&&ma2!=b)
                    {
                        cnta=0;
                        st+='c';
                        c--;
                        cntc++;
                    }
                    else
                        break;
                }
            }
            else if(ma==b&&b>0&&ma!=a)
            {
                if(cntb<2)
                {
                    st+='b';
                    cntb++;
                    b--;
                    cnta=cntc=0;
                }
                else
                {
                    int ma2=max(a,c);
                    if(ma2==a&&a>0)
                    {
                        cntb=0;
                        st+='a';
                        a--;
                        cnta++;
                    }
                    else if(ma2==c&&c>0&&ma2!=a)
                    {
                        cntb=0;
                        st+='c';
                        c--;
                        cntc++;
                    }
                    else
                        break;
                }
            }
            else if(ma==c&&c>0)
            {
                if(cntc<2)
                {
                    st+='c';
                    cntc++;
                    c--;
                    cnta=cntb=0;
                }
                else
                {
                    int ma2=max(a,b);
                    if(ma2==a&&a>0)
                    {
                        cntc=0;
                        st+='a';
                        a--;
                        cnta++;
                    }
                    else if(ma2==b&&b>0&&ma2!=a)
                    {
                        cntc=0;
                        st+='b';
                        b--;
                        cntb++;
                    }
                    else
                        break;
                }
            }
        }
        return st;
    }
};
