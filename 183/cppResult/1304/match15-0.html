class Solution {
public:
    bool check(string s, char ch) {
        if (s.length() <= 1)    return true;
        else {
            int len = s.length();
            if (s[len - 1] != ch || s[len-2] != ch) return true;
            else return false;
        }
    }
    
    string longestDiverseString(int a, int b, int c) {
        string s = "";
        while (a || b || c) {
            int val = max(a, max(b, c));

            if (a == val && check(s, 'a')){
                s += "a";
                a -= 1;
            }
            else if (b == val && check(s, 'b')) {
                s += "b";
                b -= 1;
            }
            else if (c == val && check(s, 'c')) {
                s += "c";
                c -= 1;
            }
            else {
                if (a == val) {
                    if (b >= c && b > 0) {
                        s += "b";
                        b -= 1;
                    }
                    else if (c >= b && c > 0) {
                        s += "c";
                        c -= 1;
                    }
                    else break;
                }
                else if (b == val) {
                    if (a >= c && a > 0) {
                        s += "a";
                        a -= 1;
                    }
                    else if (c >= a && c > 0) {
                        s += "c";
                        c -= 1;
                    }
                    else break;
                }
                else {
                    if (a >= b && a >0) {
                        s += "a";
                        a -= 1;
                    }
                    else if (b >= a && b > 0) {
                        s += "b";
                        b -= 1;
                    }
                    else break;
                }
            }
        }
        return s;
    }
};
