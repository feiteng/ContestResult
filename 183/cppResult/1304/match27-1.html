class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        int n=0;
        string s;

        while(a+b+c>0){
            if(n>=2&&s[n-1]=='a'&&s[n-2]=='a'){
                if(b==0&&c==0) return s;
                if(b>c) {
                    s.push_back('b');
                    b--;
                }
                else {
                    s.push_back('c');
                    c--;
                }
                n++;
            }

            else if(n>=2&&s[n-1]=='b'&&s[n-2]=='b'){
                if(a==0&&c==0) return s;
                if(a>c) {
                    s.push_back('a');
                    a--;
                }
                else {
                    s.push_back('c');
                    c--;
                }
                n++;
            }

            else if(n>=2&&s[n-1]=='c'&&s[n-2]=='c'){
                if(b==0&&a==0) return s;
                if(b>a) {
                    s.push_back('b');
                    b--;
                }
                else {
                    s.push_back('a');
                    a--;
                }
                n++;
            }
            else if(a>=b&&a>=c){
                s.push_back('a');
                a--;
                n++;
            }
            else if(b>=a&&b>=c){
                s.push_back('b');
                b--;
                n++;
            }
            else if(c>=a&&c>=b){
                s.push_back('c');
                c--;
                n++;
            }
        }
        return s;
    }
};
