class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string res;
        priority_queue<pair<int,char>> que;
        if(a) que.push({a,'a'});
        if(b) que.push({b,'b'});
        if(c) que.push({c,'c'});
        while(!que.empty()){
            auto p = que.top();
            que.pop();
            int ca = p.first;
            char cp = p.second;
            if(que.empty()){
                for(int i=0;i<min(ca,2);i++){
                    res.push_back(cp);
                }
                break;
            }
            auto q = que.top();
            que.pop();
            
            
            int cb = q.first;
            char cq = q.second;
            if(ca>cb){
                res.push_back(cp);
                res.push_back(cp);
                res.push_back(cq);
                ca-=2;
                cb-=1;
            }
            else{
                res.push_back(cp);
                res.push_back(cq);
                ca-=1;
                cb-=1;
            }
            if(ca){
                que.push({ca,cp});
            }
            if(cb){
                que.push({cb,cq});
            }
            
        }
        return res;
    }
};
