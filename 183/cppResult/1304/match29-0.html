class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        int last = -1;
        int cnt[3] = {a, b, c};
        string r;
        while (true) {
            int best_all = -1;
            int best = -1;
            for(int i=0;i<3;++i) {
                if (i != last && cnt[i] > 0) {
                    if (best == -1 || cnt[i] > cnt[best]) {
                        best = i;
                    }
                }
                if (best_all == -1 || cnt[i] > cnt[best_all]) {
                    best_all = i;
                }
            }
            if (best == -1) break;
            
            if (best_all != best) {
                r += string(1, best + 'a');
                cnt[best] --;
            } else if (cnt[best] >= 2) {
                r += string(2, best + 'a');
                cnt[best] -= 2;
            } else {
                r += string(1, best + 'a');
                cnt[best] = 0;
            }
            last = best;
        }
        return r;
    }
};
