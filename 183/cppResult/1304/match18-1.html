class Solution {
public:
    bool isOk(string & s, char c) {
        if (s.size() < 2) return true;
        return s[s.size() - 1] != c || s[s.size() - 2] != c;
    }
    string longestDiverseString(int a, int b, int c) {
        string ans = "";
        while (a + b + c > 0) {
            bool oka = isOk(ans, 'a') && a > 0;
            bool okb = isOk(ans, 'b') && b > 0;

            bool okc = isOk(ans, 'c') && c > 0;
            if (oka && (a >= b || !okb) && (a >= c || !okc)) { ans += 'a'; a--; }
            else if (okb && (b >= c || !okc)) { ans += 'b'; b--; }
            else if (okc) { ans += 'c'; c--; }
            else break;
        }
        return ans;
    }
};
