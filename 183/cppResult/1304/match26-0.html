class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        vector<pair<int,char>> nums={{a,'a'},{b,'b'},{c,'c'}};
        sort(nums.rbegin(),nums.rend());
        string ret;
        if(nums[0].first>=(nums[1].first+nums[2].first+1)*2){
            while(nums[1].first>0||nums[2].first>0){
                ret+=nums[0].second;
                ret+=nums[0].second;
                if(nums[1].first>0){
                   ret+=nums[1].second;
                    nums[1].first--;
                }else{
                    ret+=nums[2].second;
                    nums[2].first--;
                }
            }
            ret+=nums[0].second;
            ret+=nums[0].second;
            return ret;
        }
        while(nums[0].first>nums[1].first+nums[2].first){
                ret+=nums[0].second;
                nums[0].first--;
                if(nums[0].first>0){
                    nums[0].first--;
                    ret+=nums[0].second;
                }
                if(nums[2].first>0){
                    nums[2].first--;
                    ret+=nums[2].second;                
                }
                if(nums[1].first>0){
                    nums[1].first--;
                    ret+=nums[1].second;                
                }            
        }
        while(nums[0].first<nums[1].first+nums[2].first){
                ret+=nums[0].second;
                nums[0].first--;
                if(nums[2].first>0){
                    nums[2].first--;
                    ret+=nums[2].second;                
                }
                if(nums[1].first>0){
                    nums[1].first--;
                    ret+=nums[1].second;                
                }            
        }
        while(nums[0].first!=0){
                ret+=nums[0].second;
                nums[0].first--;
                if(nums[2].first>0){
                    nums[2].first--;
                    ret+=nums[2].second;                
                }else{
                    nums[1].first--;
                    ret+=nums[1].second;                                    
                }
        }        
        return ret;
    }
};
