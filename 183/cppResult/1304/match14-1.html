class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        int l = 0;
        string ans;
        while (a || b || c) {
            if (a >= b && a >= c) {

                if (l < 2 || ans[l - 1] != 'a' || ans[l - 2] != 'a') {
                    ans += "a";
                    a--;
                } else if (b > 0 && b >= c) {
                    ans += "b";
                    b--;
                } else if (c > 0) {
                    ans += "c";
                    c--;
                }

            } else if (b >= a && b >= c) {
                if (l < 2 || ans[l - 1] != 'b' || ans[l - 2] != 'b') {
                    ans += "b";
                    b--;
                } else if (a > 0 && a >= c) {
                    ans += "a";
                    a--;
                } else if (c > 0) {
                    ans += "c";
                    c--;
                }

            } else if (c >= a && c >= b) {
                if (l < 2 || ans[l - 1] != 'c' || ans[l - 2] != 'c') {
                    ans += "c";
                    c--;
                } else if (a > 0 && a >= b) {
                    ans += "a";
                    a--;
                } else if (b > 0) {
                    ans += "b";
                    b--;
                }
            }
            if (ans.size() == l) {
                break;
            }
            l++;
            //cout << l << " " << ans << endl;
        }
        return ans;
    }
};
