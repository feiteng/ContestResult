class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        string ret;
        int preCh = -1;
        int cnt[3] = {a, b, c};
        const char ch[3] = {'a', 'b', 'c'};
        
        while (cnt[0] > 0 || cnt[1] > 0 || cnt[2] > 0) {
            int cho = -1;
            
            for (int i = 0; i < 3; ++i) {                
                if (cho == -1 || cnt[i] > cnt[cho]) {
                    cho = i;
                }
            }
            
            if (cho == preCh) {
                // use other characters

                int buffer = -1;
                
                for (int i = 0; i < 3; ++i) {
                    if (i == cho) {
                        continue;
                    }
                    
                    if (buffer == -1 || cnt[i] > cnt[buffer]) {
                        buffer = i;
                    }
                }
                
                if (cnt[buffer] == 0) {
                    break;
                }
                
                --cnt[buffer];
                ret += ch[buffer];            
            }
            
            int maxCnt = min(2, cnt[cho]);
            
            for (int i = 0; i < maxCnt; ++i) {
                ret += ch[cho];
            }
            
            cnt[cho] -= maxCnt;
            
            preCh = cho;
        }
        
        return ret;
    }
};
