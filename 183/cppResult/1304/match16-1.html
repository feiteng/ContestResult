class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        vector<char> seq;
        int last;

        if (a >= b && a >= c) {
            seq.push_back('a');
            --a;
            last = 0;
        }
        else if (b >= c && b > a){
            seq.push_back('b');
            --b;
            last = 1;
        }
        else if (c > a && c > b){
            seq.push_back('c');
            --c;
            last = 2;
        }
        while (true){
            if (a == 0 && b == 0 && c == 0) break;
            if (last == 0 && b == 0 && c == 0) break;
            if (a == 0 && last == 1 && c == 0) break;
            if (a == 0 && b == 0 && last == 2) break;

            if (last == 0){
                if (b == 0){
                    seq.push_back('c');
                    --c;
                    last = 2;
                }
                else if (c == 0 || b >= c){
                    seq.push_back('b');
                    --b;
                    last = 1;
                }
                else {
                    seq.push_back('c');
                    --c;
                    last = 2;
                }

            }
            else if (last == 1){
                if (a == 0){
                    seq.push_back('c');
                    --c;
                    last = 2;
                }
                else if (c == 0 || a >= c){
                    seq.push_back('a');
                    --a;
                    last = 0;
                }
                else {
                    seq.push_back('c');
                    --c;
                    last = 2;
                }
            }

            else if (last == 2){
                if (a == 0){
                    seq.push_back('b');
                    --b;
                    last = 1;
                }
                else if (b == 0 || a >= b){
                    seq.push_back('a');
                    --a;
                    last = 0;
                }
                else {
                    seq.push_back('b');
                    --b;
                    last = 1;
                }
            }
        }
        string answer = "";
        for (int i = 0;i < seq.size();++i){
            if (seq[i] == 'a'){
                if (a > 0){
                    --a;
                    answer += "aa";
                }
                else{
                    answer += 'a';
                }
            }
            else if (seq[i] == 'b'){
                if (b > 0){
                    --b;
                    answer += "bb";
                }
                else{
                    answer += 'b';
                }
            }
            else{
                if (c > 0){
                    --c;
                    answer += "cc";
                }
                else{
                    answer += 'c';
                }
            }
        }
        return answer;
    }
};
