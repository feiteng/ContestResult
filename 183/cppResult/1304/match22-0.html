class Solution {
public:
    string longestDiverseString(int a, int b, int c) {
        vector<vector<int>> v = {{a, 0}, {b, 1}, {c, 2}};
        string ans;
        while (1) {
            sort(v.begin(), v.end());
            if (v[2][0] > 1) {
                if (v[2][0] - v[1][0] < 2) {
                    ans.push_back('a' + v[2][1]);
                    ans.push_back('a' + v[1][1]);
                    --v[2][0];
                    --v[1][0];
                }
                else if (v[1][0] > 0) {
                    ans.push_back('a' + v[2][1]);
                    ans.push_back('a' + v[2][1]);
                    ans.push_back('a' + v[1][1]);
                    v[2][0] -= 2;
                    --v[1][0];
                }
                else {
                    ans.push_back('a' + v[2][1]);
                    ans.push_back('a' + v[2][1]);
                    break;
                }
            }
            else {
                for (int i = 2; i > -1; --i)
                    for (int j = 0; j < v[i][0]; ++j)
                        ans.push_back('a' + v[i][1]);
                break;
            }
        }
        return ans;
    }
};
