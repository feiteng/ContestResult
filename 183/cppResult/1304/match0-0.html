class Solution {
  public:
      string longestDiverseString(int a, int b, int c) {
          auto CMP_FUNC = [](pair<int,char>&a,pair<int,char>&b){
              return a.first<b.first;
          };
          priority_queue<pair<int,char>, vector<pair<int,char>>, decltype(CMP_FUNC)> QUEUEHEAP(CMP_FUNC);
          if (a>0) QUEUEHEAP.push({a,'a'});
          if (b>0) QUEUEHEAP.push({b,'b'});
          if ( c>0 ) QUEUEHEAP.push({c,'c'});
          string result;
          while( QUEUEHEAP.size()>0 ){
              int t = result.size();
               vector<pair<int,char>> TMP_VEC;
              auto CURRENT = QUEUEHEAP.top(); QUEUEHEAP.pop();
              if ( result.size()>1 && CURRENT.second == result.back() && (result[t-1]==result[t-2]) ){
                  TMP_VEC.push_back(CURRENT);
                  if ( QUEUEHEAP.size()==0 ) break;
                  CURRENT = QUEUEHEAP.top(); QUEUEHEAP.pop();
              }
              result += CURRENT.second;
              if (--CURRENT.first>0)
                  QUEUEHEAP.push(CURRENT);
              for(auto&z:TMP_VEC)
                  QUEUEHEAP.push(z);
          }
          return result;
      }
  };
