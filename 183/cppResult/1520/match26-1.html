class Solution {
public:
    int numSteps(string s) {
        int add = 0;
        int n = s.size();
        int res = 0;
        for(int i=n-1;i>0;i--){
            int item = s[i]-'0';
            if(item && add){
                // res++;
                // continue;
            }
            else if(item && add==0){
                add = 1;
                res++;
            }else if(item == 0 && add){
                res++;
            }
            res++;
        }
        if(add) res+=1;
        return res;
    }
};
