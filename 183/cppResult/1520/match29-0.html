class Solution {
public:
    int numSteps(string s) {
        int res = 0;
        while (s.size() > 1) {
            if (s.back() == '0') {
                s.pop_back();
                res++;
            } else {
                res++;
                while (s.size() > 0 && s.back() == '1') {
                    s.pop_back();
                    res++;
                }
                if (s.size() == 0) {
                    return res;
                } else {
                    s.pop_back();
                    s.push_back('1');
                }
            }
        }
        return res;
    }
};
