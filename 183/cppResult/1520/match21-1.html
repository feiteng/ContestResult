class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(s.size()>1){
            if(s.back()=='0'){
                s.pop_back();
                ans++;
                continue;
            }

            bool f = false;
            for(int j = s.size()-1;j>=0;--j){
                if(s[j] == '0'){
                    s[j] = '1';
                    f=true;
                    break;
                }
                s[j] = '0';
            }
            if(!f)s = "1"+s;
            ans++;
        }
        return ans;
    }
};
