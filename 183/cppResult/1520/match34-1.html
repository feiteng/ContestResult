class Solution {
public:
    int numSteps(string s) {
        int res = 0;
        while(s.size() != 1)
        {
            if(s[s.size() - 1] == '0')
            {
                res++;
                s.pop_back();
            }
            else
            {
                s = addStrings(s, "1");
                res++;
            }
            //cout << s << endl;
        }
        return res;
    }
    
    string addStrings(string num1, string num2) {
        if(num1.size() == 0) return num2;
        if(num2.size() == 0) return num1;
        reverse(num1.begin(), num1.end());
        reverse(num2.begin(), num2.end());
        
        int index1 = 0, index2 = 0;
        int c = 0;
        string res;
        while(index1 < num1.size() || index2 < num2.size() || c)
        {
            int number1 = (index1 < num1.size()) ? (num1[index1++] - '0') : 0;
            int number2 = (index2 < num2.size()) ? (num2[index2++] - '0') : 0;
            res.push_back((number1 + number2 + c) % 2 + '0');
            c = (number1 + number2 + c) / 2;
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
