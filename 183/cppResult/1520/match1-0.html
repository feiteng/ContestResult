class Solution {
public:
    int numSteps(string s) {
      int ans = 0;
      while(s.size() > 1) {
        int n = s.size();
        if(s[n - 1] == '0') {
          s = s.substr(0, n - 1);
        } else {
          int t = 1;
          for(int i = n - 1; i >= 0 && t > 0; i--) {
            if(s[i] == '0') {
              s[i] = '1';
              t = 0;
            } else {
              s[i] = '0';
              t = 1;
            }
          }
          if (t == 1) {
            s = '1' + s;
          }
        }
        ans += 1;
      }
      return ans;
    }
};
