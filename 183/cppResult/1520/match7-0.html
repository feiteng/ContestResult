class Solution {
public:
    int numSteps(string s) {
        int r=0;
        while(s.size()>1){
            //cout<<s<<endl;
            if(s[s.size()-1]=='0'){
                r++;
                s.pop_back();
            }
            else{
                int i=s.size()-1;
                while(s[i]=='1'){
                    s[i]='0';
                    i--;
                    if(i==-1)
                        return r+s.size()+1;
                }
                s[i]='1';
                r++;
            }
        }
        return r;
    }
};
