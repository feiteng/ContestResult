class Solution {
public:
    int numSteps(string s) {
        if (s.size() == 1) return 0;
        if (s.back() == '0') {
            s.pop_back();
            return numSteps(s) + 1;
        }
        int zero_idx = s.size() - 1;
        while(zero_idx >= 0 && s[zero_idx] == '1') {
            zero_idx--;
        }
        if (zero_idx == -1) {
            return 1 + s.size();
        }
        int zeros = s.size() - 1 - zero_idx;
        s = s.substr(0, zero_idx + 1);
        s[s.size() - 1] = '1';
        return numSteps(s) + 1 + zeros;
    }
};
