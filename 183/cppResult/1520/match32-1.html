class Solution {
public:
    int numSteps(string s) {
      int cnt = 0;
      while(s.size() && s != "1") {
        cnt ++;
        if (s.back() == '0') {
          s.pop_back();
        } else {
          int c = 1;
          s.back() = '0';
          for(int i = s.size() - 2; i >= 0 && c == 1; --i) {
            auto& digit = s[i];
            if (digit == '0') {
              digit = '1';
              c = 0;
            } else if (digit == '1') {
              digit = '0';
            }
          }
          if (c) {
            s = '1' + s;
          }
        }
      }
      return cnt;
    }
};
