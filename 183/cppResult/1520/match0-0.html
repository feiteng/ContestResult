class Solution {
public:
    int numSteps(string s) {
        if (s=="1") return 0;
        int last1=0;
        for (int i=s.length()-1;i>=0;i--)
            if (s[i]=='1') {last1=i; break;}
        if (last1==0) return s.length()-1;
        int ret=s.length();
        for (int i=last1;i>=0;i--)
            if (s[i]=='0') ret++;
        ret++;
        return ret;
        
    }
};
