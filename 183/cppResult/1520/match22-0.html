class Solution {
public:
    int numSteps(string s) {
        int pos = s.size() - 1;
        int res = 0;
        int seq_count = 0;
        while(pos >= 0) {
            if (s[pos] == '0' && seq_count == 0) {
                ++res;
                --pos;
            }
            else if (s[pos] == '1') {
                ++seq_count;
                --pos;
            }
            else {
                res += (seq_count + 1);
                seq_count = 1;
                --pos;
            }
        }
        if (seq_count > 1) {
            res += (seq_count + 1);
        }
        return res;
    }
};
