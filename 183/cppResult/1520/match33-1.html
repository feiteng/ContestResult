class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(s.size() > 1) {
            if (s.back() == '0') {
                s.pop_back();
                ans += 1;
            } else {
                ans += 1;
                bool found = false;
                for(int j = s.size()-1; j >= 0; j--) {
                    if (s[j] == '0') {
                        found = true;
                        s[j] = '1';
                        for(int k = j+1; k < s.size(); k++)
                            s[k] = '0';
                        break;
                    }
                } 
                if (!found) {
                    string s2 = "1";
                    for(int i = 0; i < s.size(); i++)
                        s2 += '0';
                    s = s2;
                }
            }
            // cout << s << endl;
        }
        return ans;
        
    }
};
