class Solution {
public:
    int numSteps(string s) {
        int cnt=0;
        
        while(s.size()>1) {
            if(s.back()=='0') {
                s.pop_back();
                cnt++;
            } else {
                while(s.size() && s.back()=='1') {
                    cnt++;
                    s.pop_back();
                }
                if(s.size())
                    s.pop_back();
                s.push_back('1');
                s.push_back('0');
            }
        }
        return cnt;
        
    }
};
