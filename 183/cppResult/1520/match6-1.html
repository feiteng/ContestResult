class Solution {
public:
    int numSteps(string s) {
        int st = 0;
        int carry = 0;
        for (int i = s.size() - 1; i > 0; --i) {
            int v = s[i] - '0' + carry;
            if (v == 0) {
                ++st;
            } else if (v == 1) {
                carry = 1;
                st += 2;
            } else {
                carry = 1;
                st += 1;
            }
        }
        if (carry) {
            ++st;
        }
        return st;
    }
};
