class Solution {
public:
    string plus( const string &s ) {
        int carry = 1;
        string res;
        for ( int i = s.size() -1; i >= 0; --i ) {
            int sum = carry + s[i]-'0';
            if ( sum == 2 ) {
                res = "0"+res;
                carry = 1;
            } else if ( sum == 1 ) {
                res = "1"+ res;
                carry = 0;
            } else { // sum = 0
                res = "0"+res;
                carry = 0;
            }
        }
        if ( carry ) {
            res = "1" + res;
        }
        return res;

    }
    int numSteps(string s) {
        int res = 0;
        string target("1");
        while ( s.compare(target)!= 0 ){ 
            if ( s.back() == '0' ) {
                s = s.substr(0, s.size()-1);
            } else {
                s = plus(s);
            }
            //cout << s << endl;
            
            res++;
        }
        return res;
    }
};
