class Solution {
public:
    int numSteps(string s) {
        int res = 0;
        while (s.size() != 1) {
            res++;
            if (s.back() == '0') {
                s.pop_back();
            } else {
                while (!s.empty() && s.back() == '1') {
                    s.pop_back();
                    res++;
                }
                if (s.empty()) s = "1";
                else s.back() = '1';
            }
        }
        return res;
    }
};
