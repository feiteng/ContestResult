class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while (s.size() != 1) {
            string news;
            if (s.back() == '0') {
                news = s.substr(0, s.size() - 1);
                s = news;
            } else {
                bool all1 = true;
                int last = -1;
                for (int i = s.size() - 1; i >= 0; i--) {
                    if (s[i] != '1') {
                        all1 = false;
                        last = i;
                        break;
                    }
                }
                if (all1) {
                    return ans + 1 + s.size();
                } else {
                    s[last] = '1';
                    for (int i = last + 1; i < s.size(); i++) {
                        s[i] = '0';
                    }
                }
            }
            ans++;
        }
        return ans;
    }
};
