class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(s != "1"){
            ans ++;
            if(s.back() == '0')
                s.pop_back();
            else
                add(s);
        }
        
        return ans;
    }
    
    void add(string &s){
        int n = s.length();
        int bAdd = 1;
        for(int i=n-1;i>=0 && bAdd == 1;i--){
            int temp = s[i] - '0';
            if(temp == 0){
                bAdd = 0;
                s[i] = '1';
            }
            else
                s[i] = '0';            
        }
        if(bAdd == 1)
            s = "1" + s;
    }
};
