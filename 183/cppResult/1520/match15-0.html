class Solution {
public:
    int numSteps(string s) {
        if (s == "1")
            return 0;

        int ans = 0;
        while (s != "1") {
            if (s.back() == '1') {
                s.back() = '0';
                int i = s.size() - 2;
                while (i >= 0 && s[i] == '1') {
                    s[i] = '0';
                    --i;
                }

                if (i < 0)
                    s.insert(s.begin(), '1');
                else
                    s[i] = '1';
            }
            else {
                s.pop_back();
            }

            ++ans;
        }

        return ans;
    }
};
