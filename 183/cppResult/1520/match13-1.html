class Solution {
    string add(string s) {

        for (int i = s.size()-1; i >= 0; i--) {
            if (s[i] == '1') {
                s[i] = '0';
            } else {
                s[i] = '1';
                return s;
            }
        }
        return "1"+s;
    }
    
    string div(string s) {
        return s.substr(0, s.size()-1);
    }
public:
    int numSteps(string s) {

        int cont = 0;
        while (s != "1") {
            if (s[s.size()-1] == '1') {
                s = add(s);
            } else {
                s = div(s);
            }
//            cout << s << endl;
            cont++;
        }
        return cont;
    }
};
