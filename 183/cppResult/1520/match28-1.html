class Solution {
public:
    int numSteps(string s) {
        int rtn = 0;
        while (s != "1") {
            ++rtn;
            if (s.back() == '1')
                addOne(s);
            else
                s.pop_back();
        }
        return rtn;
    }
    
    void addOne(string &s) {
        int c = 1, i = s.size()-1;
        while (i >= 0 && c) {
            if (s[i] == '1') s[i] = '0';
            else {
                s[i] = '1';
                c = 0;
            }
            --i;
        }
        
        if (c) s = "1" + s;
    }
};
