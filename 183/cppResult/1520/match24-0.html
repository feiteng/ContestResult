class Solution {
public:
    int numSteps(string s) {
        int count=0;
        while(s.size()!=1){
            int n=s.length();
            if(s[n-1]=='0'){
                s.pop_back();
                count++;
            }
            else{
                s[n-1]='0';
                int i;
                for(i=n-2;i>=0;i--){
                    if(s[i]=='0'){
                        s[i]='1';
                        break;
                    }
                    s[i]='0';
                }
                if(i<0){
                    s.insert(s.begin(),'1');
                }
                count++;
            }
        }
        return count;
    }
};
