class Solution {
public:
    int numSteps(string s) {
        int cnt = 0;
        int p = s.length()-1;
        while (p > 0) {
            if (s[p] == '1') {
                ++ cnt;
                while (p >= 0 && s[p] == '1') {
                    --p;
                    ++cnt;
                }
                if (p >= 0) {
                    s[p] = '1';
                }
            } else {
                --p;
                ++cnt;
            }
        }
        return cnt;
    }
    
};
