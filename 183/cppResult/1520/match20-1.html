class Solution {
public:
    int numSteps(string s) {
        int cnt=0;
        int N=s.size();
        int p=N-1;
        while(p>=0){
            if(p==0&&s[p]=='1') break;
            if(s[p]=='1'){
                int q=p;

                cnt++;
                while(q>=0&&s[q]=='1'){
                    s[q]='0';
                    q--;
                }
                if(q>=0){
                    s[q]='1';
                }
            }else{
                cnt++;
                p--;
            }
        }
        return cnt;
    }
};
