const ll MOD=1e9+7;

class Solution {
public:
  int numSteps(string s) {
    int res=0;
    while(s.size()>1){
      if(s.back()=='0') {
        s.pop_back();
        res++;
      }else {
        res++;
        int d=s.size()-1;
        while(d>=0 && s[d]=='1') {
          s[d--] = '0';
        }
        if (d>=0) {
          s[d]='1';
        } else {
          res += s.size();
          break;
        }
      }
    }
    return res;
  }
};
