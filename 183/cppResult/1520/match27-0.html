class Solution {
public:
    int numSteps(string s) {
        int ans=0;
        while(s.length()!=1){
            int last=s.length()-1;
            if(s[last]=='0'){
                s=s.substr(0,s.length()-1);
            } else {
                while(last>=0&&s[last]=='1'){
                    s[last]='0';
                    last--;
                }
                if(last<0){
                    s="1"+s;
                } else {
                    s[last]='1';
                }
            }
            //cout<<s<<endl;
            ans++;
        }
        return ans;
    }
};
