class Solution {
public:
    int numSteps(string s) {
        int ans = 0;
        while(!(s[0] == '1' && s.length() == 1)){
            int len = s.length();
            if(s[len - 1] == '0'){
                s.pop_back();

            } else{
                int carry = 1;
                for(int i = s.length() - 1; i >= 0; i--){
                    if(s[i] == '1'){
                        s[i] = '0';
                    } else{
                        s[i] = '1';
                        carry = 0;
                        break;

                    }
                }
                if(carry){
                    s = "1" + s;
                }
            }
            ans++;
        }
        return ans;
    }
};
