class Solution {
public:
    int numSteps(string s) {
      int ret = 0;
      int n = s.length();
      int d = 0;
      for (int i = n-1; i >= 0; --i) {
        int val = s[i] - '0' + d;
        if (i == 0 && val == 1) break;
        if (val % 2 == 0) {
          ++ret;
          d = val / 2;
          //printf("i=%d, d=%d, add=%d\n", i, d, 1);
        } else {
          ret += 2;
          d = (val+1)/2;
          //printf("i=%d, d=%d, add=%d\n", i, d, 2);
        }
        
        /*
        if (val % 2 == 0) {
          ++ret;
          d = 0;
          //printf("i=%d, d=%d, add=%d\n", i, d, 1);
        } else {
          ret += 2;
          d = 1;
          //printf("i=%d, d=%d, add=%d\n", i, d, 2);
        }
        */
      }
      return ret;
    }
};
