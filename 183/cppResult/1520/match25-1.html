class Solution {
public:
    int numSteps(string s) {
        int n=s.size();
        int cnt=0;
        int tmp=0;
        for (int i=n-1;i>0;--i){
            int cur=s[i]-'0';
            cur+=tmp;
            tmp=(cur>>1);
            cur%=2;
            if (cur==0) ++cnt;
            else{
                cnt+=2;
                tmp+=1;
            }
        }
        if (tmp==1) ++cnt;
        return cnt;
    }
};
