class Solution {
public:
    int numSteps(string s) {
        int count=0;
        while(s!="1"){
            // cout<<s<<endl;
            if(s.back()=='0'){
                s=s.substr(0,s.length()-1);
            }
            else{
                string temp;
                int carry=1,j=s.length()-1;
                
                while(carry && j!=-1){
                    if(s[j]=='1'){
                        temp+='0';
                    }
                    else{
                        temp+='1';
                        carry=0;
                        break;
                    }
                    j--;
                }
                if(j==-1){
                    temp+='1';
                    reverse(temp.begin(),temp.end());
                    s=temp;
                }
                else{
                    reverse(temp.begin(),temp.end());
                    s=s.substr(0,j)+temp;
                }
            }
            // cout<<s<<endl;
            count++;
            
        }
        return count;
    }
};
