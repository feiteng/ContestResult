class Solution {
  public:
      int numSteps(string s) {
          int COUNTER = 0;
          for(;!(s.size()==1 && s[0]=='1');){
              s = TRANS_AND_SOLVE(s);
              ++COUNTER;
          }
          return COUNTER;
          
      }
      
      string TRANS_AND_SOLVE(string str){
          int SIZE = str.size();
          if ( str[SIZE-1]=='0' ) return str.substr(0, SIZE-1);
          int j=SIZE-1;
          while(j>=0 && str[j]=='1'){
            str[j--]='0';
          }
          if (j<0) return "1"+str;
          str[j] = '1';
          return str;
      }
  };
