class Solution {
public:
    int numSteps(string s) {
        if(s.size() == 1) return 0;
        if(s.back() == '0'){
            s.pop_back();
            return 1 + numSteps(s);
        }
        int i = s.size() - 1;
        while(i >= 0 && s[i] == '1'){
            s[i] = '0';
            --i;
        }
        if(i < 0) return 1 + s.size();
        s[i] = '1';
        return 1 + numSteps(s);
    }
};
