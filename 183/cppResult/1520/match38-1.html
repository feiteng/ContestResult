string incinc(string A) {
	reverse(A.begin(),A.end());
	FORR(r,A) {
		if(r=='1') r='0';
		else {
			r++;
			break;
		}
	}
	if(A.back()=='0') A += '1';
	reverse(A.begin(),A.end());
	return A;
}


class Solution {
public:
    int numSteps(string s) {
		
		int step=0;
		while(s!="1") {
			if(s.back()=='0') {
				step++;
				s.pop_back();
			}
			else {
				step++;
				s=incinc(s);
			}
			
		}
		return step;
    }
};
