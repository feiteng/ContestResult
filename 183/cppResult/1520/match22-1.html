class Solution {
public:
    int numSteps(string s) {
        int ans=0;
        int n=s.size();
        int f=n-1;
        while(f>0)
        {
            if(s[f]=='0')
            {
                f--;
                ans++;
            }
            else
            {
                int now=f;
                while(f>=0&&s[f]=='1')f--;
                ans=ans+1+now-f;
                if(f>0)s[f]='1';
            }
        }
        return ans;
    }
};
