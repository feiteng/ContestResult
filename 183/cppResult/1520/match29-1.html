class Solution {
public:
    int numSteps(string s) {
        int res = 0;
        while (s.size() > 1) {
            if (s.back() == '2') {s.pop_back(); s.back()++;}
            else if (s.back() == '0') s.pop_back();
            else s.back()++;
            ++res;
        }
        if (s == "2") return res + 1;
        return res;
    }
};
