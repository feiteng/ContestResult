class Solution {
public:
    using bit=bitset<505>;
    bit f(const bit&a,const bit&b){
        return b.count()==0?a:f(a^b,(a&b)<<1);
    }
    int numSteps(string s) {
        int ans=0;
        bitset<505>a(s),b(1);
        while(a!=b){
            if((a&b)==1){
                a=f(a,b);
            }else{
                a>>=1;
            }
            ++ans;
        }
        return ans;
    }
};
