class Solution {
public:
    string divide_by_2(string const &s) {
        return s.substr(0, s.size() - 1);
    }
    
    string add_1(string const s) {
        string result;
        int curr = 1;
        for (int i = s.size() - 1; i >= 0; --i) {
            if (curr == 1 && s[i] == '1') {
                curr = 1;
                result.push_back('0');
            } else {
                result.push_back('0' + curr + (s[i] - '0'));
                curr = 0;
            }
            // cout << result << ";;;" << endl;
        }
        if (curr == 1) {
            result.push_back('1');
        }
        reverse(result.begin(), result.end());
        return result;
    }
    
    int numSteps(string s) {
        if (s == "1") {
            return 0;
        }
        int cnt = 0;
        while (s != "1") {
            if (s.back() == '1') {
                s = add_1(s);
            } else {
                s = divide_by_2(s);
            }
            // cout << s << endl;
            cnt++;
        } 
        return cnt;
    }
};
