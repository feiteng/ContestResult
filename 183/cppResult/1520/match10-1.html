class Solution {
public:
    int numSteps(string s) {
        int res = 0;
        while (s.size() > 1) {
          //  cout << s << endl;
            if (s.back() == '1') {
                int k = s.size();
                -- k;
                while (k >= 0 && s[k] == '1') {
                    s[k] = '0';
                    -- k;
                }
                if (k < 0) {
                    s = "1" + s;

                } else {
                    s[k] = '1';
                }
            } else {
                s.pop_back();
            }
            ++ res;
        }
        return res;
    }
};
