class Solution {
public:
    int numSteps(string s) {
        int res = 0;
        while (s != "1") {
            if (s.back() == '1') {
                addOne(s);
            }
            else {
                s.erase(s.length()-1);
            }
           // cout << s << "\n";
            ++res;
        }
        return res;
    }
private:
    void addOne(string &s) {
        int n = s.length(), c = 1;
        
        for (int i = n-1; i >= 0; --i) {
            if (s[i] == '0') {
                s[i] = '1';
                c = 0;
                break;
            }
            else {
                s[i] = '0';
            }
        }
        if (c) s = '1' + s;
    }
};
