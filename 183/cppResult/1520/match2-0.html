class Solution {
public:
    int numSteps(string S) {
        int ans = 0;
        while(true) {
            int sz = S.size();
            if(S == "1") return ans;
            if(S[sz - 1] == '1') {
                int done = false;
                for(int i = sz - 1; i >= 0; i--) {
                    if(S[i] == '0') {
                        S[i] = '1';
                        done = true;
                        break;
                    }
                    S[i] = '0';
                }
                if(not done) S.insert(S.begin(), '1');
            }
            else S.pop_back();
           // cout << S << endl;
            ans++;
        }
        //cout << endl;
        return ans;
    }
};
