class Solution {
public:
    int numSteps(string s) {
        int cnt = 0;
        while (s != "1") {
            // cout << s <<endl;
            if (s.back() == '0') {
                s.pop_back();
            } else {
                // add 1
                bool isadd = false;
                int n = s.length();

                s.back() = '0';
                
                for (int i = n - 2; i >= 0; --i) {
                    if (s[i] == '0') {
                        s[i] = '1';
                        isadd = true;
                        break;
                    } else {
                        s[i] = '0';
                    }
                }
                if (!isadd) {
                    s = '1' + s;
                }
            }
            ++cnt;
        }
        return cnt;
    }
};
