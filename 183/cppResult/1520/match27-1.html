class Solution {
public:
    int numSteps(string s) {
        int ops = 0;
        while (s.size() > 1) {

            if (s.back() == '0') {
                s.pop_back();
            } else {
                int ptr = s.size();
                --ptr;

                while (ptr >= 0 and s[ptr] == '1') {
                    s[ptr] = '0';
                    --ptr;
                }
                
                if (ptr < 0) {
                    s = '1' + s;
                } else {
                    s[ptr] = '1';
                }
            }
            ++ops;
        }
        return ops;
    }
};
