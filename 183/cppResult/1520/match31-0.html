class Solution {
public:
    string addOne(string s) {
        int indexOfFirst = s.size() - 2;
        for (; indexOfFirst >= 0; indexOfFirst--) {
            if (s[indexOfFirst] == '0') {
                break;
            }
        }
        if (indexOfFirst == -1) {
            string ans = "1";
            for (int i = 0; i < s.size(); i++) {
                ans += "0";
            }
            return ans;
        } else {
            string ans = s.substr(0, indexOfFirst);
            ans += "1";
            if (indexOfFirst + 1 < s.size()) {
                ans += s.substr(indexOfFirst + 1, s.size() - indexOfFirst - 1);
            }
            return ans;
        }
    }
    
    int numSteps(string s) {
        string cur = s;
        int steps = 0;
        while (cur != "1") {
            if (cur[cur.size() - 1] == '0') {
                cur = cur.substr(0, cur.size() - 1);
            } else {
                cur = addOne(cur);
            }
            steps++;
        }
        return steps;
    }
};
