class Solution {
public:
    int numSteps(string s) {
        int count = 0;
        // cout<<"test"<<endl;
        while (s != "1") {
            // cout<<s<<endl;
            if (s[s.size()-1] == '0') {
                s.pop_back();
                count++;
            } else {
                int k = s.size()-1;
                while (k >= 0 and s[k] == '1') {
                    s[k] = '0';
                    k--;
                }
                if (k == -1) return count + s.size()+1;
                else {
                    s[k] = '1';
                    count++;
                }
            }
        }
        return count;
    }
};
