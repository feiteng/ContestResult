class Solution {
public:
    int numSteps(string s) {
        int step = 0;
        
        while (s.length() > 1) {
            if (s[s.length() - 1] == '1') {
                int j = s.length() - 1;
                while (j >= 0 && s[j] == '1') {
                    s[j] = '0';
                    j--;
                } 
                if (j >= 0) {
                    s[j] = '1';
                } else {
                    s.insert(s.begin(), '1');
                }
            } else {
                s.pop_back();
            }
            //cout << s << "\n";
            step++;
        } 
        
        return step;
        
    }
};
