class Solution {
public:
    int numSteps(string s) {
        int cnt = 0;

        while( s!="1" ){
            s = solve(s);
            ++cnt;
        }
        return cnt;
        
    }
    
    string solve(string s){
        int n = s.size();
        if ( s[n-1]=='0' ) return s.substr(0, n-1);
        int j=n-1;
        while(j>=0 && s[j]=='1'){
            s[j]='0';
            --j;
        }
        if (j<0) return "1"+s;
        s[j] = '1';
        
        return s;
    }
};
