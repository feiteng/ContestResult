class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        heap = []
        if a:
            heap.append((-a, 'a'))
        if b:
            heap.append((-b, 'b'))
        if c:
            heap.append((-c, 'c'))
        if not heap: return ""
        # heap = [(-a, 'a'), (-b, 'b'), (-c, 'c')]
        heapq.heapify(heap)
        res = "#"
        while heap:
            cur, ch = heapq.heappop(heap)
            cur = -cur
            if ch == res[-1]:
                if not heap: return res[1:]
                cur2, ch2 = heapq.heappop(heap)
                cur2 = -cur2
                res += ch2
                if cur2 > 1:
                    heapq.heappush(heap, (-(cur2 - 1), ch2))
                heapq.heappush(heap, (-cur, ch))
            else:
                if cur <= 2:
                    res += cur * ch
                else:
                    res += ch * 2
                    heapq.heappush(heap, (-(cur - 2), ch))
                
        if 'aaa' in res or 'bbb' in res or 'ccc' in res: return ""
        return res[1:]
