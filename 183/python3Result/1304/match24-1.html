class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        heap 
= []
        if a != 0:
            heapq.heappush(heap, (-1*a, 'a'))
        if b != 0:
            heapq.heappush(heap, (-1*b, 'b'))
        if c != 0:
            heapq.heappush(heap, (-1*c, 'c'))
        
        ans = ''
        wait = collections.defaultdict(lambda: [])
        t = 0
        while len(heap) != 0 or t in wait:
            for w in wait[t]:
                heapq.heappush(heap, w)
                
            x, y = heapq.heappop(heap)
            x = -1 * x
            ans += y
            if x - 1 == 0:
                t += 1
                continue
            if len(ans) >= 2 and ans[-2] == y:
                wait[t+2].append((-1*(x-1), y))
            else:
                heapq.heappush(heap, (-1*(x-1), y))
            t += 1
        return ans
