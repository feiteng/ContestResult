class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        res = ''
        l = sorted([['a',a],['b',b],['c',c]],key=lambda x:x[1])
        while l[2][1]:
            if res
[-2:]==l[2][0]*2:
                if l[1][1]==0:
                    return res
                res+=l[1][0]
                l[1][1]-=1
            else:
                res+=l[2][0]
                l[2][1]-=1
            l.sort(key=lambda x:x[1]) 
        return res
