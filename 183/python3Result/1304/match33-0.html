class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        ans = ""
        import heapq
        L=[]
        if a!=0: heapq.heappush(L,(-a,'a'))
        if b!=0: heapq.heappush(L,(-b,'b'))
        if c!=0: heapq.heappush(L,(-c,'c'))
        while len(L)>=2:
            num1,ch1 = heapq.heappop(L)
            num2,ch2 = heapq.heappop(L)
            if len(ans)>=2 and ans[-2]==ans[-1]==ch1:
                ans+=ch2
                num2+=1
            else:
                ans+=ch1
                num1+=1
            if num1!=0:
                heapq.heappush(L,(num1,ch1))
            if num2!=0:
                heapq.heappush(L,(num2,ch2))
        num1,ch1 = heapq.heappop(L)
        if num1==0: return ans
        if len(ans)>=2 and ans[-2]==ans[-1]==ch1:
            return ans
        ans+=ch1
        num1+=1
        if num1==0: return ans
        if len(ans)>=2 and ans[-2]==ans[-1]==ch1:
            return ans
        return ans+ch1
