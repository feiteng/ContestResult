class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        cnt = {'a': a, 'b': b, 'c': c}
        chars = sorted
(cnt, key=lambda x: cnt[x], reverse=True)
        if cnt[chars[0]] > 2*(cnt[chars[1]] + cnt[chars[2]]+1):
            mids = chars[1]*cnt[chars[1]] + chars[2]*cnt[chars[2]]
            res = [chars[0]*2]
            for c in mids:
                res += [c, chars[0]*2]
            return ''.join(res)
        res = []
        while cnt[chars[0]] > 0 and cnt[chars[1]] + cnt[chars[2]] > 0:
            n0 = min(2, cnt[chars[0]])
            res.append(chars
[0]*n0)
            cnt[chars[0]] -= n0
            ub = cnt[chars[1]] + cnt[chars[2]] - (cnt[chars[0]]-1)//2
            n2 = min(2, cnt[chars[2]], ub)
            res.append(chars[2]*n2)
            cnt[chars[2]] -= n2
            ub = cnt[chars[1]] + cnt[chars[2]] - (cnt[chars[0]]-1)//2
            n1 = min(2, cnt[chars[1]], ub)
            res.append(chars[1]*n1)
            cnt[chars[1]] -= n1
        n0 = min(2, cnt[chars[0]])
        res.append(chars[0]*n0)
        return ''.join(res)
