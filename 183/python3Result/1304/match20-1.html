class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        heap 
= []
        if a:
            heapq.heappush(heap, (-a, 'a'))
        if b:
            heapq.heappush(heap, (-b, 'b'))
        if c:
            heapq.heappush(heap, (-c, 'c'))
        
        ans = []
        counter = 0
        tempStore = None
        while heap:
            num, nextChar = heapq.heappop(heap)
            
            if ans and nextChar == ans[-1]:
                counter += 1
            else:
                counter = 1
            
            ans.append(nextChar)
            num += 1

            if tempStore:
                heapq.heappush(heap, tempStore)
                tempStore = None
            
            if counter == 2:
                counter = 0
                tempStore = num, nextChar
            else:
                if num:
                    heapq.heappush(heap, (num, nextChar))
                    
            # print(heap, tempStore, ans)
                
        return "".join(ans)
