class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        ret=""
        remain=[[a,'a'],[b,'b'],[c,'c']]
        remain.sort(reverse=True)
        while remain[0][0]>0:
            add=False
            if not (len(ret)>0 and ret[-1]==remain[0][1]):
                temp=min(remain[0][0],2)
                ret+=remain[0][1]*temp
                remain[0][0]-=temp
                add=True
            elif remain[1][0]>0:
                ret+=remain[1][1]
                remain[1][0]-=1
                add=True
            else:
                break
            remain.sort(reverse=True)
        return ret
