class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        heap = []
        # [-a, "a"], [-b, "b"], [-c, "c"]
        
        cnt = {"a":a, "b":b, "c":c}
        
        if a > 0:
            heap.append([-a, "a"])
        if b > 0: 
            heap.append([-b, "b"])
        if c > 0: 
            heap.append([-c, "c"])
            
        heapq.heapify(heap)
        prev, res = None, "" 
        
        while heap:     
            curr_cnt, c = heapq.heappop(heap)
            if prev:
                heapq.heappush(heap, prev)

            res += c
            curr_cnt += 1  
            cnt[c] -= 1
            
            if curr_cnt < 0:
                prev = [curr_cnt, c]
            else:
                prev = None   
                
        i = 0
        
        while i < len(res):
            n = res[i]
            if cnt[n] > 0:
                res = res[:i] + n + res[i:]
                cnt[n] -= 1
                i+=1
            i+=1
            
        return res
