class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        a=min(a,2*(b+c+1))
        b=min(b,2*(a+c+1))
        c=min(c,2*(a+b+1))
        lis=[(a,'a'),(b,'b'),(c,'c')]
        lis=sorted(lis)
        if lis[2][0]>=(lis[0][0]+lis[1][0]+1):
            dif=lis[2][0]-(lis[0][0]+lis[1][0]+1)
            if dif<=lis[1][0]:
                s=dif*(lis[2][1]+lis[2][1]+lis[1][1])+(lis[1][0]-dif)*(lis[2][1]+lis[1][1])+lis[0][0]*(lis[2][1]+lis[0][1])+lis[2][1]
            elif dif<=lis[1][0]+lis[0][0]:
                s=lis[1][0]*(lis[2][1]+lis[2][1]+lis[1][1])+(dif-lis[1][0])*(lis[2][1]+lis[2][1]+lis[0][1])+(lis[1][0]+lis[0][0]-dif)*(lis[2][1]+lis[0][1])+lis[2][1]
            else:
                s=lis[1][0]*(lis[2][1]+lis[2][1]+lis[1][1])+lis[0][0]*(lis[2][1]+lis[2][1]+lis[0][1])+lis[2][1]+lis[2][1]
            return s
                
        elif lis[2][0]==lis[0][0]+lis[1][0]:
            s=lis[0][0]*(lis[2][1]+lis[0][1])+lis[1][0]*(lis[2][1]+lis[1][1])
            return s
        elif lis[2][0]>lis[1][0]:
            dif=lis[2][0]-(lis[1][0]+1)
            s=dif*(lis[2][1]+lis[2][1]+lis[1][1])+(lis[1][0]-dif)*(lis[2][1]+lis[1][1])+lis[2][1]
            t=''
            for i in range(lis[0][0]):
                t+=lis[0][1]+s[i]
            t+=s[lis[0][0]:]
            return t
        else:
            s=lis[2][0]*(lis[2][1]+lis[1][1])
            t=''
            for i in range(lis[0][0]):
                t+=lis[0][1]+s[i]
            t+=s[lis[0][0]:]
            return t
