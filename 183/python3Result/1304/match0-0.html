class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        value = [[a,'a'], [b,'b'], [c,'c']]
        flag = False
        answer = ""
        while(True):
            value = sorted(value, reverse=True)
            if len(answer) == 0:
                answer = answer + value[0][1]
                value[0][0]-=1
                continue
            if flag and value[0][1] == answer[-1]:
                if value[1][0] == 0:
                    break
                answer = answer + value[1][1]
                value[1][0]-=1
                flag = False
            else:
                if value[0][0] == 0:
                    break
                if value[0][1] == answer[-1]:
                    flag = True
                answer = answer + value[0][1]
                value[0][0]-=1
        return answer
