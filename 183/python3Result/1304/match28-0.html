class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        ans = ''
        nums = [[a,'a'], [b,'b'], [c,'c']]
        while True:
            nums.sort()
            if len(ans)>1 and ans[-1] == ans[-2] and ans[-1] == nums[-1][1]:
                if nums[-2][0] == 0:
                    return ans
                ans+=nums[-2][1]
                nums[-2][0]-=1
            else:
                if nums[-1][0] == 0:
                    return ans
                ans+=nums[-1][1]
                nums[-1][0]-=1
        return ans
