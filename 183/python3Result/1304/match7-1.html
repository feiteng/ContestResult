class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        s = [[a, 'a'], [b, 'b'], [c, 'c']]
        
        res = ''
        
        while True:
            s.sort(reverse=True)
            
            if s[1][0] == 0 and s[2][0] == 0:
                res += s[0][1] * min(2, s[0][0])
                break
            
            if s[1][0] >= 2:
                if s
[0][0] - s[1][0] >= 3:
                    res += s[0][1] * 2 + s[1][1] * 1
                    s = [[s[0][0]-2, s[0][1]], [s[1][0]-1, s[1][1]], s[-1]]
                else:
                    res += s
[0][1] * 2 + s[1][1] * 2
                    s = [[s[0][0]-2, s[0][1]], [s[1][0]-2, s[1][1]], s[-1]]
            elif s[1][0] == 1:
                if s[0][0] >= 2:
                    res += s[0][1] * 2 + s[1][1] * 1
                    s = [
[s[0][0]-2, s[0][1]], [s[1][0]-1, s[1][1]], s[-1]]
                else:
                    res += s[0][1] * 1 + s[1][1] * 1
                    s 
= [[s[0][0]-1, s[0][1]], [s[1][0]-1, s[1][1]], s[-1]]
            
        
        return res
