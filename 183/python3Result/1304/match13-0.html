class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        import heapq
        hold = None
        queue = []
        if a:
            heapq.heappush(queue,(-a,'a'))
        if b:
            heapq.heappush(queue,(-b,'b'))
        if c:
            heapq.heappush(queue,(-c,'c'))
        ans = []
        
        while queue:
            cnt,v = heapq.heappop(queue)
            if len(ans)>1 and ans[-1]==v and ans[-2]==v:
                hold = cnt, v
                continue
            cnt+=1
            ans.append(v)
            if cnt:
                heapq.heappush(queue,(cnt,v))
            if hold:
                heapq.heappush(queue,hold)
                hold = None
        return ''.join(ans)
