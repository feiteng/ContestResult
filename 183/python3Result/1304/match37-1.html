class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        letters = [[a, "a"],[b, "b"],[c, "c"]]
        letters.sort(key = lambda x: x[0])
        s = ""
        last = "z"
        while letters[2][0] > 0:
            toAdd = letters[2][1]
            if toAdd == last:
                # add one med
                if letters[1][0] == 0:
                    break
                s += letters[1][1]
                letters[1][0] -= 1
                last = letters[1][1]
            else:
                if letters
[2][0] == 1:
                    s += letters[2][1]
                    letters[2][0] -= 1
                else:
                    s += letters[2][1]
                    s += letters[2][1]
                    letters[2][0] -= 2
                last = letters[2][1]
            letters.sort(key = lambda x: x[0])

        return s
