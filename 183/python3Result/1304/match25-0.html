class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        ans="$"
        h=[ [a, "a"], [b, "b"], [c, "c"] ]
        h.sort()
        while h[2][0]>0:
            h.sort()
            print(h)
            if ans[-1]!=h[2][1]:
                ans+=h[2][1]*min(2,h[2][0])
                h[2][0] = max(0, h[2][0]-2)
            else:
                if h[1][0] == 0: break
                ans+=h[1][1]
                h[1][0] -= 1
            h.sort()
        print()
        return ans[1:]
