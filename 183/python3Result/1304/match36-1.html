class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        dic = [[a,"a"],[b,"b"],[c,"c"]]
        dic.sort()
        re = ""
        if dic[2][0] >= 2 * (dic[0][0] + dic[1][0] + 1):
            while 1:
                re += dic[2][1] * 2
                dic[2][0] -= 2
                if dic[0][0] > 0:
                    re += dic[0][1]
                    dic[0][0] -= 1
                elif dic[1][0] > 0:
                    re += dic[1][1]
                    dic[1][0] -= 1
                if dic[1][0] == 0:
                    re += dic[2][1] * 2
                    break
        else:
            if dic[2][0] > 2 * (dic[1][0] + 1):
                while 1:
                    if dic[2][0] > 2 and dic[2][0] > dic[1][0] + dic
[0][0]:
                        re += dic[2][1] * 2
                        dic[2][0] -= 2
                    else:
                        re += dic[2][1] * 1
                        dic[2][0] -= 1
                    if dic[0][0] > 0:
                        re += dic[0][1]
                        dic[0][0] -= 1
                    elif dic[1][0] > 0:
                        re += dic[1][1]
                        dic[1][0] -= 1
                    if dic[2][0] == 0:
                        break
            else:
                while 1:
                    if dic[2][0] > 2 and dic[2][0] > dic
[1][0]:
                        re += dic[2][1] * 2
                        dic[2][0] -= 2
                    else:
                        re += dic[2][1] * 1
                        dic[2][0] -= 1
                    if dic[0][0] > 0:
                        re += dic[0][1]
                        dic[0][0] -= 1
                    if dic[1][0] > 0:
                        re += dic[1][1]
                        dic[1][0] -= 1
                    if dic[2][0] == 0:
                        break
        return re
