class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        x = [[a, 'a'], [b, 'b'], [c, 'c']]
        
        ans = ''
        
        while True:
            x.sort(reverse=True)
            if x[0][0] == 0:
                break
            
            if not ans or ans[-1] != x[0][1]:
                n = min(2, x[0][0])
                ans += x[0][1] * n
                x[0][0] -= n
            elif x[1][0] > 0:
                if x[1][0] > x[0][0]:
                    n = min(2, x[1][0])
                    ans += x[1][1] * n
                    x[1][0] -= n
                else:
                    ans += x[1][1]
                    x[1][0] -= 1
            elif x[2][0] > 0:
                if x[2][0] > x[0][0]:
                    n = min(2, x[2][0])
                    ans += x[2][1] * n
                    x[2][0] -= n
                else:
                    ans += x[2][1]
                    x[2][0] -= 1
            else:
                break
        
        return ans
