class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        heap = []
        import heapq
        if a != 0:
            heapq.heappush(heap, (-a, 'a'))
        if b != 0:
            heapq.heappush(heap, (-b, 'b'))
        if c != 0:
            heapq.heappush(heap, (-c, 'c'))
        result = []
        steak = {'a': 0, 'b':0, 'c':0}
        
        wait_queue = [None]
        while heap or wait_queue[0]:
            #print(heap, wait_queue, result)
            if wait_queue[0] != None:
                heapq.heappush(heap, wait_queue[0])
                wait_queue[0] = None
            left, char = heapq.heappop(heap)
            if steak[char] == 2:
                wait_queue[0] = (left, char)
                if len(heap) == 0:
                    break
                left, char = heapq.heappop(heap)
            
            result.append(char)
            left += 1
            if left != 0:
                heapq.heappush(heap, (left, char))
            steak[char] += 1
            for c in steak:
                if c != char:
                    steak[c] = 0
        return ''.join(result)
