class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        s = 'dd'
        M = [a, b, c]
        K = [0, 1, 2]
        S = 'abc'
        while any(M):
            K.sort(key=lambda x:M[x], reverse=True)
            if S[K[0]] == s[-1] and S[K[0]] == s[-2]:
                # no more K[0]
                if M[K[1]] == 0:
                    break
                else:
                    s += S[K[1]]
                    M[K[1]] -= 1
            elif M[K[0]] == 1:
                    s += S[K[0]]
                    M[K[0]] -= 1
            else:
                # M[K[0]] >= 2
                s += S[K[0]] + S[K[0]]
                M[K[0]] -= 2
        return s[2:]
