class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        
        self.res = ''
        self.memo = set()
        queue 
= []
        if a:
            heapq.heappush(queue,(-1*a,'a'))
        if b:
            heapq.heappush(queue,(-1*b,'b'))
        if c:
            heapq.heappush(queue,(-1*c,'c'))
        while queue:
            val,char = heapq.heappop(queue)
            if len(self.res) >= 2 and self.res[-1] == char and self.res[-2] == char:
                # do not use char
                if not queue:
                    return self.res
                val2,char2 = heapq.heappop(queue)
                self.res += char2
                val2+=1
                if val2!= 0:
                    heapq.heappush(queue,(val2,char2))
            else:
                val+=1
                self.res += char
            if val != 0:
                heapq.heappush(queue,(val,char))
        return self.res
