class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        if a > 2*(b+c+1):
            a = 2*(b+c+1)
        if b > 2*(a+c+1):
            b = 2*(a+c+1)
        if c > 2*(b+a+1):
            c = 2*(b+a+1)
        
        res = ''
        pre = None
        while a > 0 or b > 0 or c > 0:
            if pre == None:
                mx = max(a,b,c)
            elif pre == 'a':
                mx = max(b,c)
            elif pre == 'b':
                mx = max(a,c)
            elif pre == 'c':
                mx = max(a,b)
            cur = None
            mx3 = max(a,b,c)
            if pre != 'a' and a == mx:
                if a > 1 and mx3 == mx:
                    a -= 2
                    res += 'aa'
                else:
                    a -= 1
                    res += 'a'
                pre = 'a'
            elif pre != 'b' and b == mx:
                if b > 1 and mx3 == mx:
                    b -= 2
                    res += 'bb'
                else:
                    b -= 1
                    res += 'b'
                pre = 'b'
            elif pre != 'c' and c == mx:
                if c > 1 and mx3 == mx:
                    c -= 2
                    res += 'cc'
                else:
                    c -= 1
                    res += 'c'
                pre = 'c'
            
        return res
