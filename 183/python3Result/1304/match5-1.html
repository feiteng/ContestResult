class Solution:
    def longestDiverseString(self, a: int, b: int, c: int) -> str:
        myheap = []
        if a>0:
            heapq.heappush(myheap, [-a,'a'])
        if b>0:
            heapq.heappush(myheap, [-b,'b'])
        if c>0:
            heapq.heappush(myheap, [-c,'c'])
            
        res = []
        cache = []
        
        while myheap:
            flag = False
            while myheap:
                cur = heapq.heappop(myheap)
                #print(cur)
                if not(len(res)>=2 and res[-1]==res[-2] and res[-1]==cur[1]):
                    res.append(cur[1])
                    cur[0] += 1
                    if cur[0]<0:
                        heapq.heappush(myheap, cur)
                    flag = True
                    #happy string is distroyed
                    for ele in cache:
                        heapq.heappush(myheap, ele)
                    cache = []
                else:#cannot use this char, find the happy string
                    cache.append(cur)
                #print(myheap)
            if flag is False:
                break
            for ele in cache:
                heapq.heappush(myheap, ele)
            cache = []
            
        return "".join(res)
