import heapq

class Solution:
    def longestDiverseString(self, A: int, B: int, C: int) -> str:
        pq = []
        res = []
        if A != 0:
            heapq.heappush(pq, (-A, 'a'))
        if B != 0:
            heapq.heappush(pq, (-B, 'b'))
        if C != 0:
            heapq.heappush(pq, (-C, 'c'))

        while pq:
            count, ch  = heapq.heappop(pq)
            count = -count
            if (not res) or res[-1] != ch:
                length = min(2, count)
                res.extend([ch] * length)
                if count - length > 0:
                    heapq.heappush(pq, (-(count - length), ch))
            else:
                if not pq:
                    break
                count_next, ch_next = heapq.heappop(pq)
                count_next = -count_next
                res.append(ch_next)
                if count_next - 1 > 0:
                    heapq.heappush(pq, (-(count_next - 1), ch_next))
                heapq.heappush(pq, (-count, ch))
        return "".join(res)
