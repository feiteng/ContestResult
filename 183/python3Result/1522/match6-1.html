from functools import lru_cache
class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        
        
@lru_cache(None)
        def helper(start):
            if start == len(stoneValue):
                return 0
            res = float('-inf')
            tmp = 0
            for i in range(start, min(start+3, len(stoneValue))):
                tmp += stoneValue[i]
                res = max(res, tmp - helper(i+1))
                
            return res
        
        diff = helper(0)
        
        if diff == 0:
            res = "Tie"
        elif diff > 0:
            res = "Alice"
        else:
            res = "Bob"
            
        return res
