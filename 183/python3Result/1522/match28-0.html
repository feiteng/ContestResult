'''
1 2 3 7

7
7, 0
f(-1) -> 7, 0

3 7
3, 7
10, 0
f(-2) -> 10, 0

2 3 7
2, 10
5, 7
12, 0
f(-3) -> 12, 0

1 2 3 7
1, 12 # (1, 0) + swap(f(-3))
3, 10 # (1+2, 0) + swap(f(-2))
6, 7  # (1+2+3, 0) + swap(f(-1))
-> 6, 7


'''

class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        v = stoneValue
        n = len(v)
        f = [None] * (n + 1)
        f[-1] = (0, 0)
        
        for i in range(len(v)-1, -1, -1):
            c = []
            for j in range(i, min(i+3, n)):
                # print(f'i {i}  j {j}')
                j += 1
                b, a = f[j]
                a += sum(v[i:j])
                c.append((a, b))
            f[i] = max(c, key=lambda z:z[0] - z[1])
            # print(f)
        # print(f)
        a, b = f[0]
        if a > b:
            return 'Alice'
        if b > a:
            return 'Bob'
        return 'Tie'
