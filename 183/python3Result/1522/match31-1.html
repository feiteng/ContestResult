class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        # right to left, advantage for ending at position, i.e. amount greater if ending at position
        
        stoneValue = [0,0,0] + stoneValue[::-1]
        advantage = [0 for _ in stoneValue]
        
        
for i in range(3,len(stoneValue)):
            advantage[i] = max(-advantage[i-3] + stoneValue[i] + stoneValue[i-1] + stoneValue[i-2],
                               -advantage[i-2] + stoneValue[i] + stoneValue[i-1],
                               -advantage[i-1] + stoneValue[i]) 
            
        if advantage[-1] < 0:
            return "Bob"
        if advantage[-1] > 0:
            return "Alice"
        return "Tie"
