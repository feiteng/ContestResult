class Solution:
    def turn(self, sv):
        if len(sv) == 1:
            return sv[0]
        elif len(sv) == 2:
            return max(sv[0]+sv[1], sv[0]-sv[1])
        elif len(sv) == 3:
            return max(sv[0]-self.turn(sv[1:]), sv[0]+sv[1]-sv[2], sv[0]+sv[1]+sv[2])
        else:
            return max(sv[0]-self.turn(sv[1:]), sv[0]+sv[1]-self.turn(sv[2:]), sv[0]+sv[1]+sv[2]-self.turn(sv[3:]))
        
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        sv = stoneValue
        a = [0] * n
        for i in range(n):
            if i == 0:
                a[n-i-1] = sv[n-i-1]
            elif i == 1:
                a[n-i-1] = max(sv[n-i-1]+sv[n-i], sv[n-i-1]-sv[n-i])
            elif i == 2:
                a[n-i-1] = max(sv[n-i-1]-a[n-i], sv[n-i-1]+sv[n-i]-sv[n-i+1], sv[n-i-1]+sv[n-i]+sv[n-i+1])
            else:
                a[n-i-1] = max(sv[n-i-1]-a[n-i], sv[n-i-1]+sv[n-i]-a[n-i+1], sv[n-i-1]+sv[n-i]+sv[n-i+1]-a[n-i+2])
        print(a)
        a = a[0]
        if a > 0:
            return 'Alice'
        if a == 0:
            return 'Tie'
        if a < 0:
            return 'Bob'
