class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n=len(stoneValue)
        res=[1]*n
        totals=[0]*n
        v=stoneValue[::-1]
        s=0
        for i in range(n):
            s+=v[i]
            totals[i]=s
        points=[0]*n
        points[0]=totals[0]
        if n>1:
            points[1]=max(v[0],0)+v[1]
        if n>2:
            points[2]=max(totals[2],totals[2]-points[0],v[2]+totals[1]-points[1])
        for i in range(3,n):
            points[i]=max(v[i]+totals[i-1]-points[i-1],v[i]+v[i-1]+totals[i-2]-points[i-2],v[i]+v[i-1]+v[i-2]+totals[i-3]-points[i-3])
        #print (points)
        if points[n-1]>totals[n-1]/2:
            return 'Alice'
        elif points[n-1]<totals[n-1]/2:
            return 'Bob'
        else:
            return 'Tie'
