class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        diff = 0
        state = {0:"lose",1:"tie",2:"win"}
        Awin = {}
        # Awin[0] = 0
        # Awin[1] = v[-1]
        rev = [0,0,0]+stoneValue[::-1]
        maxrew = [0,0,0]
        for i in range(3,len(rev)):
            maxrew.append(max(rev[i]-maxrew[i-1],rev[i]+rev[i-1]-maxrew[i-2],rev[i]+rev[i-1]+rev[i-2]-maxrew[i-3]))
        #print(maxrew)
        if maxrew[-1] < 0:
            return "Bob"
        elif maxrew[-1] == 0:
            return "Tie"
        else:
            return "Alice"
