class Solution:
    def stoneGameIII(self, stones: List[int]) -> str:
        n = len(stones)
        
        @functools.lru_cache(None)
        def dp(s, i):
            if i >= n:
                return 0
            res = -math.inf
            for j in range(3):
                if i+j < n:
                    take = sum(stones[i:i+j+1])
                    res = max(res, s - dp(s-take, i+j+1))
            return res
        
        s = sum(stones)
        first = dp(s, 0)
        if first > s-first:
            return 'Alice'
        if first < s-first:
            return 'Bob'
        return 'Tie'
