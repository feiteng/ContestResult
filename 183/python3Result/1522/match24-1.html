class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        revstoneValue = stoneValue[::-1]
        expecteddiff = [0] #its what (activeplayer - nonactiveplayer) is if people play optimally.
        for index in range(len(revstoneValue)):
            possible = [sum(revstoneValue[index-j:index+1])-expecteddiff[-(j+1)] for j in range(0,3) if len(expecteddiff)>j]# this might be slow to get len constantly
            if len(possible)==0:
                possible = [revstoneValue[index]]
            newdiff=max(possible)
            expecteddiff
.append(newdiff)
        
        finaldiff = expecteddiff[-1]
        #print(finaldiff)
        #print(expecteddiff)
        if finaldiff==0:
            return "Tie"
        elif finaldiff>0:
            return "Alice"
        elif finaldiff<0:
            return "Bob"
