class Solution:
    def stoneGameIII(self, stoneValue):
        N = len(stoneValue)
        if N == 1:
            if stoneValue[0] > 0:
                return "Alice"
            elif stoneValue[0] == 0:
                return "Tie"
            return "Bob"
        cumsum = [0] * (N + 1)
        for i in range(N - 1, -1, -1):
            if i == N - 1:
                cumsum[i] = stoneValue[i]
            else:
                cumsum[i] = cumsum[i+1] + stoneValue[i]
        mem = [None] * N
        mem[N-1] = stoneValue[N-1]
        mem[N-2] = stoneValue[N-2] + max(0, stoneValue[N-1])
        
        def dp(idx):
            if idx >= N:
                return 0
            if mem[idx] is not None:
                return mem[idx]
            res = stoneValue[idx] + cumsum[idx+1] - dp(idx+1)
            if idx < N - 1:
                res = max(res, stoneValue[idx] + stoneValue[idx+1] + cumsum[idx+2] - dp(idx+2))
            if idx < N - 2:
                res = max(res, stoneValue[idx] + stoneValue[idx+1] + stoneValue[idx+2] + cumsum[idx+3] - dp(idx+3))
            mem[idx] = res
            return res
        
        Alice = dp(0)
        Bob = cumsum[0] - Alice
        if Alice > Bob:
            return "Alice"
        elif Alice == Bob:
            return "Tie"
        return "Bob"
