class Solution:
    # Tie 0
    # win >0
    # lose <0
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        first = 'Alice'
        second = 'Bob'
        t = 'Tie'
        self.lookup = {}
        self.s = stoneValue
        self.s.reverse()
        ans =  self.wins(len(stoneValue)-1)
        if ans > 0:
            return first
        elif ans< 0:
            return second
        else:
            return t
    
    def wins(self, i):
        res = self.lookup.get(i,'not')
        if res == 'not':
            if i == 0:
                self.lookup[0] = self.s[0]

            elif i == 1:
                win = [self.s[1] - self.s[0], self.s[1] + self.s[0]]
                self.lookup[1] = max(win)

            elif i==2:
                win = [self.s[2] - self.wins(1), self.s[2]+ self.s[1]-self.wins(0), self.s[2]+self.s[1]+self.s[0]]
                self.lookup[2] = max(win)
            
            else:
                win = [self.s[i] - self.wins(i-1), (self.s[i] + self.s[i-1]) - self.wins(i-2) , (self.s[i] + self.s[i-1] + self.s[i-2]) - self.wins(i-3)]
                self.lookup[i] = max(win)
            return self.lookup[i]
        return res
