class Solution:
    def stoneGameIII(self, s: List[int]) -> str:
        n = len(s)
        s.extend([0,0,0])
        f = [0] * (n + 3)
        sufsum = [0] * (n + 3)
        for i in range(n, -1, -1):
            sufsum[i] = sufsum[i + 1] + s[i]
        for i in range(n, -1, -1):
            f[i] = sufsum[i] - min(f[i + 1: i + 4])
        if sufsum[0] < f[0] + f[0]:
            return 'Alice'
        if sufsum[0] > f[0] + f[0]:
            return 'Bob'
        return 'Tie'
