class Solution:
    def ans(self, value, s):
        if value==0:
            return 'Tie'
        elif value>0:
            return 'Alice'
        else:
            return 'Bob'
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        dp = [0]*len(stoneValue)
        dp[-1] = stoneValue[-1]
        s=sum(stoneValue)
        if len(stoneValue)==1:
            return self.ans(dp[0],s)
        dp[-2] = max(stoneValue[-2]-dp[-1], stoneValue[-2]+stoneValue[-1])
        if len(stoneValue)==2:
            return self.ans(dp[0],s)
        dp[-3] = max(stoneValue[-3] - dp[-2], stoneValue[-3] + stoneValue[-2] - dp[-1], stoneValue[-3] + stoneValue[-2]+stoneValue[-1])
        if len(stoneValue)==3:
            return self.ans(dp[0],s)
        
for i in range(len(stoneValue)-4, -1, -1):
            dp[i] = max(stoneValue[i]-dp[i+1], stoneValue[i]+stoneValue[i+1]-dp[i+2], stoneValue[i]+stoneValue[i+1]+stoneValue[i+2]-dp[i+3])
        return self.ans(dp[0],s)
