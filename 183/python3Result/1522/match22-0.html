class Solution:

    def stoneGameIII(self, stoneValue: List[int]) -> str:
        self.dic = {}
        for i in range(-1, -len(stoneValue)-1, -1):
            if i < -3:
                self.dic[i] = max(stoneValue[i] - self.dic[i+1], stoneValue[i] + stoneValue[i+1] - self.dic[i+2], stoneValue[i] + stoneValue[i+1] + stoneValue[i+2] - self.dic[i+3])
            elif i == -1:
                self.dic[i] = stoneValue[i]
            elif i == -2:
                self.dic[i] = max(stoneValue[i] + stoneValue[i+1], stoneValue[i] - self.dic[-1])
            elif i == -3:
                self.dic[i] = max(stoneValue[i] + stoneValue[i+1] + stoneValue[i+2], stoneValue[i] + stoneValue[i+1] - self.dic[-1], stoneValue[i] - self.dic[-2])
        if self.dic[-len(stoneValue)] > 0:
            return 'Alice'
        if self.dic[-len(stoneValue)] < 0:
            return 'Bob'
        return 'Tie'
