from functools import *
class Solution:
    def stoneGameIII(self, sv: List[int]) -> str:
        n=len(sv)
        @lru_cache(maxsize=None)
        def dp(i):
            if i>=n:
                return 0
            if i==n-1:
                return sv[-1]
            end=min(i+3,n)
            ans=float("-inf")
            acc=0
            for j in range(i,end):
                acc+=sv[j]
                ans=max(ans,acc-dp(j+1))
            return ans
        ans=dp(0)
        if ans>0: return "Alice"
        elif ans<0: return "Bob"
        else: return "Tie"
