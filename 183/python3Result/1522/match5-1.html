class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        @functools.lru_cache(None)
        def dfs(c):
            if c == n: return 0
            m = -inf
            for i in range(1, 4):
                if c + i > n: break
                m = max(m, -dfs(c + i) + sum(stoneValue[
c:c+i]))
                    
            return m
        
        r = dfs(0)
        if r > 0: return 'Alice'
        if r < 0: return 'Bob'
        return 'Tie'
