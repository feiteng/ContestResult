class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        sv = stoneValue
        n = len(stoneValue)
        @functools.lru_cache(None)
        def dp(i, p):
            
            if i >= n: return 0
            res = []

            if i < n: res.append(dp(i + 1, -p) + p * sv[i])
            if i + 1 < n: res.append(dp(i + 2, -p) + p * (sv[i] + sv[i + 1]))
            if i + 2 < n: res.append(dp(i + 3, -p) + p * (sv[i] + sv[i + 1] + sv[i + 2]))
            if p > 0:
                return max(res)
            else:
                return min(res)
        res = dp(0, 1)
        if res > 0:
            return "Alice"
        elif res < 0:
            return "Bob"
        else:
            return "Tie"
