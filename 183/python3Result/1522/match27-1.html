class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        if n == 0:
            return 'Tie'
        if n == 1:
            if stoneValue[0] < 0:
                return 'Bob'
            elif stoneValue[0] > 0:
                return 'Alice'
            else:
                return 'Tie'
        
        optime = [0] * (n + 1)

        optime[n - 1] = stoneValue[n - 1]
        optime[n - 2] = max(stoneValue[n - 2] - optime[n - 1], stoneValue[n - 2] + stoneValue
[n - 1])


        for i in range(n - 3, -1, -1):
            optime[i] = max(stoneValue[i] - optime[i + 1], 
                            stoneValue[i] + stoneValue[i + 1] - optime[i + 2],
                            stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - optime[i + 3])
        
        print(optime)
        if optime[0] > 0:
            return 'Alice'
        elif optime[0] < 0:
            return 'Bob'
        else:
            return 'Tie'
