class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        
        mem = {}
        def num_first(i):
            if i >= len(stoneValue):
                return 0
            if i in mem:
                return mem[i]
            a = stoneValue[i] - num_first(i+1)
            b = stoneValue[i] + stoneValue[i+1] - num_first(i+2) if i + 1 < len(stoneValue) else a
            c = stoneValue[i] + stoneValue[i+1] + stoneValue[i+2] - num_first(i+3) if i + 2 < len(stoneValue) else b
            res = max(a,b ,c )
            mem
[i] = res
            return res
        res = num_first(0)
        if res > 0:
            return "Alice"
        if res == 0:
            return 'Tie'
        return 'Bob'
