class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        stoneValue.extend([0] * 3)
        scores = [0] * (n + 3)
        for i in range(n - 1, -1, -1):
            m = max(stoneValue[i] - scores[i + 1],
                    stoneValue[i] + stoneValue[i + 1] - scores[i + 2],
                    stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] - scores[i + 3],)
            scores[i] = m
        if scores[0] > 0:
            return "Alice"
        elif scores[0] == 0:
            return "Tie"
        else:
            return "Bob"
