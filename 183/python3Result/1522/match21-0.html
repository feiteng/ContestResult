def get_score_diff(lis):
    n=len(lis)
    lis=lis+[0,0,0]
    newlis=lis[:]
    for i in range(n):
        newlis[n-1-i]=max(lis[n-1-i]-newlis[n-i],lis[n-1-i]+lis[n-i]-newlis[n+1-i],lis[n-1-i]+lis[n-i]+lis[n+1-i]-newlis[n+2-i])
    return newlis[0]
class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        k=get_score_diff(stoneValue)
        if k>0:
            return "Alice"
        elif k==0:
            return "Tie"
        else:
            return "Bob"
