import sys
sys.setrecursionlimit(150005)
from functools import lru_cache

class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        N = len(stoneValue)
        
        @lru_cache(None)
        def go(index):
            if index == N:
                return 0
            
            best = - (1 << 60)
            for x in range(1,4):
                x = min(N - index, x)
                best = max(best, sum(stoneValue[index:index+x]) - go(index + x))
            return best
        
        score = go(0)
        #print(N,score)
        if score == 0:
            return "Tie"
        if score > 0:
            return "Alice"
        return "Bob"
