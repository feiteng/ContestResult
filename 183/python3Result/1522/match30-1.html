class Solution:
    def stoneGameIII(self, v: List[int]) -> str:
        if len(v) == 1:
            if v[0] == 0:
                return 'Tie'
            return 'Alice' if v[0] > 0 else 'Bob'
        
        if len(v) == 2:
            if sum(v) > 0 or v[0] > v[1]:
                return 'Alice'
            if sum(v) == 0:
                return 'Tie'
            return 'Bob'
        
        s = [0] * len(v)
        s[-1] = v[-1]
        for i in range(len(v)-2, -1, -1):
            s[i] = s[i+1] + v[i]
        
        dp = [0] * len(v)
        
        dp[-1] = v[-1]
        dp[-2] = max(s[-1]-dp[-1]+v[-2], v[-1]+v[-2])
        dp[-3] = max(s[-2]-dp[-2]+v[-3], s[-1]-dp[-1]+v[-3]+v[-2], v[-1]+v[-2]+v[-3])
        for i in range(len
(v)-4, -1, -1):
            dp[i] = max(s[i+1]-dp[i+1]+v[i], s[i+2]-dp[i+2]+v[i]+v[i+1], s[i+3]-dp[i+3]+v[i]+v[i+1]+v[i+2])
            
            
        if dp[0] > s[0] - dp[0]:
            return 'Alice'
        if dp[0] == s[0] - dp[0]:
            return 'Tie'
        return 'Bob'
