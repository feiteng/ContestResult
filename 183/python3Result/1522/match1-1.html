class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        stoneValue = stoneValue[::-1]
        answer = []
        for i in range(len(stoneValue)):
            if (i == 0):
                answer.append(stoneValue[i])
                continue
            if (i == 1):
                cur_score = stoneValue[i] - stoneValue[i-1]
                cur_score = max(cur_score, stoneValue[i] + stoneValue[i-1])
                answer.append(cur_score)
                continue
            if (i == 2):
                cur_score = stoneValue[i] + stoneValue[i-1] + stoneValue[i-2]
                for j in range(1,3):
                    cur_score = max(cur_score, sum(stoneValue[i-j+1:i+1]) - answer[-j])
                answer.append(cur_score)
                continue
            cur_score = -float('inf')
            for j in range(1,4):
                cur_score = max(cur_score, sum(stoneValue[i-j+1:i+1]) - answer[-j])
            answer.append(cur_score)
        if (answer[-1] > 0):
            return "Alice"
        elif (answer[-1] < 0):
            return "Bob"
        else:
            return "Tie"
