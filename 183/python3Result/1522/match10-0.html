class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        status = [-float("inf") for i in range(len(stoneValue))]
        for i in range(len(stoneValue)-1,-1,-1):
            if i == len(stoneValue)-1:
                status[i] = stoneValue[i]
            elif i == len(stoneValue)-2:
                status[i] = max(stoneValue[i]+stoneValue[i+1],stoneValue[i]-status[i+1])
            elif i == len(stoneValue)-3:
                status[i] = max(stoneValue[i]+stoneValue[i+1]+stoneValue[i+2],stoneValue[i]+stoneValue[i+1]-status[i+2],stoneValue[i]-status[i+1])
            else:
                status[i] = max(stoneValue[i]-status[i+1],stoneValue[i]+stoneValue[i+1]-status[i+2],stoneValue[i]+stoneValue[i+1]+stoneValue[i+2]-status[i+3])
    
        # print(status)
        if status[0] > 0:
            return "Alice"
        elif status[0] < 0:
            return "Bob"
        else:
            return "Tie"
