from functools import lru_cache
class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        self.stoneValue = stoneValue
        result = self.getScore(0,len(stoneValue) - 1)
        
        if result > 0:
            return "Alice"
        elif result < 0:
            return "Bob"
        else:
            return "Tie"
        
    @lru_cache(None)
    def getScore(self,i,j):
        if i > j:
            return 0
        if i == j:
            return self.stoneValue[i]
        if i == j - 1:
            return max(self.stoneValue[i]- self.getScore(i+1,j),self.stoneValue[i] + self.stoneValue[i+1])
        if i == j - 2:
            return max(self
.stoneValue[i]- self.getScore(i+1,j),self.stoneValue[i] + self.stoneValue[i+1]- self.getScore(i + 2,j),self.stoneValue[i] + self.stoneValue[i+1] + self.stoneValue[i+2])
        return max(self
.stoneValue[i] - self.getScore(i+1,j), self.stoneValue[i] + self.stoneValue[i + 1] - self.getScore(i + 2,j),self.stoneValue[i] + self.stoneValue[i+1] + self.stoneValue[i+2] - self.getScore(i+3,j))
