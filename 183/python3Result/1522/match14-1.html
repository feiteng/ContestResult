class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        dp = [0] * n
        dp[n-1] = stoneValue[n-1]
        for i in range(n-2, -1,-1): dp[i] = dp[i+1] + stoneValue[i]
        @lru_cache
        def rec(ind):
            if ind >= n: return 0
            if ind == n-1: return stoneValue[n-1]
            ans1 = rec(ind+3)
            ans2 = rec(ind+2)
            ans3 = rec(ind+1)
            return max(dp[ind] - ans1, dp[ind] - ans2, dp[
ind] - ans3)
        alice = rec(0)
        if alice > dp[0] - alice: return "Alice"
        elif alice == dp[0] - alice: return "Tie"
        else: return "Bob"
