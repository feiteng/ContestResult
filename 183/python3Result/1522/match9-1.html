class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        total = sum(stoneValue)
        memo = {}
        n = len(stoneValue)
        s = [0]*(n+1)
        for i in range(n-1,-1,-1):
            s[i]=s[i+1]+ stoneValue[i]
        def dp(i):
            if i in memo:
                return memo[i]
            if i>=n:
                return 0
            best = float('-inf')
            for j in range(3):
                if i+j<n:
                    best = max(best,sum(stoneValue[i:i+j+1])+s[i+j+1]-dp
(i+j+1) )
            memo[i] = best
            return best
        ans = dp(0)
        if ans*2>total:
            return "Alice"
        if ans*2==total:
            return "Tie"
        return "Bob"
