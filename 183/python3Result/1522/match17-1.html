class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:

        n = len(stoneValue)

        dp = [0 for _ in range(n+1)]

        for i in range(n-1, -1, -1):
            maxd = float('-inf')
            for k in range(min(3, n-i)):
                s = sum(stoneValue[i: i+k+1])
                maxd = max(maxd, s - dp[i+k+1])
            dp
[i] = maxd

        if dp[0] > 0:
            return 'Alice'
        elif dp[0] == 0:
            return 'Tie'
        else:
            return 'Bob'
