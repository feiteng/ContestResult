class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        d  = dict()
        n = len(stoneValue)
        def dfs(pos):
            if pos == n:
                return 0
            
            if pos in d:
                return d[pos]
            res = float('-inf')
            cur = 0
            for i in range(pos, min(n, pos + 3)):
                cur += stoneValue[i]
                
                res = max (res, cur - dfs(i + 1))
            
            d[pos] = res
            return res
        
        
        tmp = (dfs(0))
        if (tmp < 0):
            return "Bob"
        elif tmp > 0:
            return "Alice"
        else:
            return "Tie"
