class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        length = len(stoneValue)
        dp = [0 for _ in range(length+2)]
        dp
[length-1] = stoneValue[-1]
        for i in range(length-2,-1,-1):
            dp[i] = max(stoneValue[i]-dp[i+1], stoneValue[i]+stoneValue[i+1]-dp[i+2], sum(stoneValue[i:i+3])-dp[i+3])
        key = dp[0]
        if key>0:
            return 'Alice'
        if key<0:
            return 'Bob'
        return 'Tie'
