class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        s = [0] * (n + 1)
        for i in range(n):
            s[i + 1] = s[i] + stoneValue[i]
        f = [0] * (n + 1)
        for i in range(n - 1, -1, -1):
            c = s[n] - s[i]
            t = c - f[i + 1]
            if i + 2 <= n:
                t = max(c - f[i + 2], t)
            if i + 3 <= n:
                t = max(c - f[i + 3], t)
            f[i] = t
        # print(f)
        if f[0] > s[n] - f[0]:
            return 'Alice'
        elif f[0] < s[n] - f[0]:
            return 'Bob'
        else:
            return 'Tie'
