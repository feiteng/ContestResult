class Solution:
    def stoneGameIII(self, sv: List[int]) -> str:
        n = len(sv)
        diff = [None]*(n+1)
        diff[n] = 0
        diff[n-1] = sv[n-1]
        INF = 10**12
        for i in range(n-2,-1,-1):
            ma = -INF
            s = sv[i]
            for j in range(3):
                k = i+j
                if k >= n: continue
                if k != i: s += sv[k]
                ma 
= max(ma, s-diff[k+1])
            diff[i] = ma

        if diff[0] == 0:
            return 'Tie'
        elif diff[0] > 0:
            return 'Alice'
        else:
            return 'Bob'
