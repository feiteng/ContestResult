class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        s=stoneValue
        last=[0]*len(s)
        last[-1]=s[-1]
        if len(s)==1:
            if s[0]>0:
                return "Alice"
            if s[0]<0:
                return "Bob"
            else:
                return "Tie"
        last[-2]=max(s[-2]-s[-1],s[-2]+s[-1])
        if len(s)==2:
            if last[-2]>0:
                return "Alice"
            if last[-2]<0:
                return "Bob"
            else:
                return "Tie"
        last[-3]=max(s[-3]-last[-2],s[-3]+s[-2]-last[-1],s[-3]+s[-2]+s[-1])
        if len(s)==3:
            if last[-3]>0:
                return "Alice"
            if last[-3]<0:
                return "Bob"
            else:
                return "Tie"
        if len(s)==1:
            if s[0]>0:
                return "Alice"
            if s[0]<0:
                return "Bob"
            else:
                return "Tie"
        
for i in range(len(s)-4,-1,-1):
            #if i<len(s)-1
            #    last[i]=s[i]-last[i+1]
            #if i<len(s)-2:
            #    last[i]=max(last[i],s[i]+s[i+1]-last[i+2])
                
            last[i]=max(s[i]-last[i+1],s[i]+s[i+1]-last[i+2],s[i]+s[i+1]+s[i+2]-last[i+3])
        print(last)
        if last[0]>0:
                return "Alice"
        if last[0]<0:
            return "Bob"
        else:
            return "Tie"
