from functools import lru_cache
class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n, S = len(stoneValue), sum(stoneValue)
        temp, cumsum = 0, [0]
        for x in reversed(stoneValue):
            temp += x
            cumsum.append(temp)
        cumsum = cumsum[::-1]
        @lru_cache(None)
        def helper(i):
            if i == n:
                ans = 0
            else:
                ans 
= stoneValue[i] + cumsum[i + 1] - helper(i + 1)
                if i + 2 <= n:
                    ans = max(ans, stoneValue[i] + stoneValue[i + 1] + (cumsum[i + 2] - helper(i + 2)))
                if i + 3 <= n:
                    ans = max(ans, stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] + (cumsum[i + 3] - helper(i + 3)))
            return ans
        alice = helper(0)
        return 'Tie' if 2 * alice == S else ('Alice' if 2 * alice > S else 'Bob')
