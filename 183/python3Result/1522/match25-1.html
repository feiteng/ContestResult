class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        n = len(stoneValue)
        stoneValue += [0, 0, 0]
        dp = [0, 0, 0]
        for i in range(n - 1, -1, -1):
            one = stoneValue[i] - dp[-1]
            two = sum(stoneValue[i: i + 2]) - dp[-2]
            three = sum(stoneValue[i: i + 3]) - dp[-3]
            dp.append(max(one, two, three))
        if dp[-1] > 0:
            return 'Alice'
        elif dp[-1] < 0:
            return 'Bob'
        return 'Tie'
