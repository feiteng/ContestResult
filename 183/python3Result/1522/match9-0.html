class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        memo = {}
        def dp(s, id, memo):
            if id == len(s):
                return 0
            if id in memo:
                return memo[id]
            res = None
            
            for end in range(id+1, id+4):
                if end > len(s):
                    break

                cur = sum(s[id:end]) - dp(s, end, memo)
                if res == None:
                    res = cur
                else:
                    res = max(res, cur)

            memo[id]=res
            return res
        score = dp(stoneValue, 0, memo)

        if score == 0:
            return 'Tie'
        if score > 0:
            return 'Alice'
        return'Bob'
