class Solution:
    def stoneGameIII(self, values: List[int]) -> str:
        memory = {}
        
        def recursive(start):
            if start == len(values):
                return 0
            elif start in memory:
                return memory[start]
            else:
                res = -float('inf')
                tmp = 0
                for i in range(start, min(start+3, len(values))):
                    res = max(res, tmp+values[i]-recursive(i+1))
                    tmp += values[i]
                memory[start] = res
                
                return memory[start]
            
            
        ans = recursive(0)
        if ans == 0:
            return "Tie"
        elif ans>0:
            return "Alice"
        else:
            return "Bob"
