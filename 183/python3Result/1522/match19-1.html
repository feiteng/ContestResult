class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        sv = list(reversed(stoneValue))
        dp = [(0,0),(sv[0],0)]
        
        for i in range(1,len(sv)) :
            r1b,r2b = sv[i] + dp[-1][1], dp[-1][0]
            # print(r1b,r2b,dp[-1][1], dp[-1][0])
            s = sv[i]
            for j in range(1,3) :
                if j >= len(dp) :
                    break
                s = s + sv[i-j]
                r1,r2 = s + dp[-(1+j)][1],dp[-(1+j)][0]
                # print(r1b,r2b,dp[-1][1], dp[-1][0],r1,r2,dp[-(1+j)][1],dp[-(1+j)][0])
                if r1b-r2b <= r1-r2 :
                    r1b,r2b 
= r1,r2
            dp.append((r1b,r2b))
            # print("")
            
        # print(dp)
        if dp[-1][0] > dp[-1][1] :
            return "Alice"
        elif dp[-1][0] == dp[-1][1] :
            return "Tie"
        else :
            return "Bob"
