class Solution:
    def stoneGameIII(self, stoneValue: List[int]) -> str:
        
        l = len(stoneValue)
        t_s = [0] * (l + 10)
        for i in range(l-1, -1, -1):
            t_s[i] = stoneValue[i] + t_s[i + 1]
        stoneValue.extend([float('-inf')] * 3)
        dp = [0] * (l + 10)
        for i in range(l - 1, -1, -1):
            dp[i] = max(stoneValue[i] + t_s[i + 1] - dp[i + 1], stoneValue[i] + stoneValue[i + 1] + t_s[i + 2] - dp[i + 2], 
                       stoneValue[i] + stoneValue[i + 1] + stoneValue[i + 2] + t_s[i + 3] - dp[
i  + 3])
        if dp[0] > t_s[0] - dp[0]:
            return "Alice"
        elif dp[0] == t_s[0] - dp[0]:
            return "Tie"
        else:
            return "Bob"
