class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort(reverse=True)
        summ = sum(nums)
        res = []
        cur_sum = 0
        for e in nums:
            cur_sum += e
            res.append(e)
            if cur_sum > summ - cur_sum:
                break
        return res
