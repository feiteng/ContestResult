class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        
        nums.sort(reverse=True)
        
        total=sum(nums)
        
        res=[]
        
        cur=0
        
        for num in nums:
            
            cur+=num
            res.append(num)
            if cur*2>total:
                return res
