class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort()
        csum = sum(nums)
        nsum = 0
        res = []
        
        for num in nums[::-1]:
            nsum+=num
            csum-=num
            res.append(num)
            if nsum>csum:
                break
        return res
