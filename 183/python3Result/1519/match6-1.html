class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        s = sum(nums)
        nums.sort(reverse=True)
        ans = []
        current = 0
        for num in nums:
            ans.append(num)
            current += num
            if current > s / 2:
                break
        return ans
