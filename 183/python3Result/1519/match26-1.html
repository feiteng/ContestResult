class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort()
        target = sum(nums) / 2
        s = 0
        ret = []
        for i in range(len(nums) - 1, -1, -1):
            s += nums[i]
            ret.append(nums[i])
            if s > target:
                return ret
