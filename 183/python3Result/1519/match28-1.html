class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort(reverse=True)
        total=sum(nums)
        ret=[]
        cumsum=0
        for val in nums:
            ret.append(val)
            cumsum+=val
            if cumsum*2>total:
                return ret
