class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort(reverse = True)
        total_sum = sum(nums)
        now_num = 0;
        for i in range(0, len(nums)):
            now_num += nums[i]
            if (now_num > total_sum - now_num):
                break
        return nums[:i+1]
