class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        sum_arr = sum(nums)
        nums.sort(reverse=True)
        result = []
        cur_sum = 0
        for num in nums:
            result.append(num)
            cur_sum += num
            if cur_sum > sum_arr - cur_sum:
                break
        return result
