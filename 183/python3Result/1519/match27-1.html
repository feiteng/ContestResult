class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        if len(nums)==1:
            return nums
        else:
            nums.sort()
            total = sum(nums)
            #print(total)
            tmp = 0
            #print(nums)
            for i in range(len(nums)-1, -1, -1):
                #print(i, tmp)
                if tmp+nums[i] > total - tmp - nums[i]:
                    #print(i, tmp, nums[i])
                    res = nums[i:len(nums)]
                    return res[::-1]
                tmp += nums[i]
