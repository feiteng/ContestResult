class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        nums.sort()
        total = sum(nums)
        tmpTotal = 0
        result = []
        for i in range(len(nums)-1,-1,-1):
            tmpTotal = tmpTotal + nums[i]
            result.append(nums[i])
            if tmpTotal * 2 > total:
                return result
