class Solution:
    def minSubsequence(self, nums: List[int]) -> List[int]:
        s = sum(nums)
        nums.sort(reverse = True)
        ans = []
        total = 0
        for n in nums:
            total +=n
            ans+=[n]
            if total > s-total:
                break
        return ans
