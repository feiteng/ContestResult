class Solution {
    public boolean hasAllCodes(String s, int k) {
        if (s.length() < (1 << k)) {
            return false;
        }
        int[] a = new int[1 << k];
        Arrays.fill(a, 0);
        int t = 0;
        for (int i = 0;i < s.length() && i < k - 1;i++) 
{
            t = t * 2 + (s.charAt(i) - '0');
        }
        for (int i = k - 1;i < s.length();i++) {
            t = t * 2 + (s.charAt(i) - '0');
            a[t] = 1;
            t -= (s.charAt(i - k + 1) - '0') * (1 << (k - 1));
        }
        for (int p: a) {
            if (p == 0) {
                return false;
            }
        }
        return true;
    }
}
