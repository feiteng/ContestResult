class Solution {
    public boolean hasAllCodes(String s, int k) {
        int p = 1 << k;
        boolean[] map = new boolean[p];
        p--;
        int t = 0;
        for (char c : s.toCharArray()){                        
            t = ((t << 1) +  (c -'0')) & p;            
            if (k == 1){
                // System.out.println(c +"," + t);
                map[t] = true;
            }else {
                k--;
            
}
        }
        for (boolean b : map){
            if (!b){
                return false;
            }
        }
        return true;
    }
}
