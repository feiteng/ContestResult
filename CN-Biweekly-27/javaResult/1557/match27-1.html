class Solution {
    public boolean hasAllCodes(String s, int k) {
        Set<Integer> set = new HashSet<>();
        for (int i = 0; i + k <= s.length(); ++i) {
            String t = s.substring(i, i + k);
            set.add(Integer.valueOf(t, 2));
        }
        return set.size() == (1 << k);
    }
}
