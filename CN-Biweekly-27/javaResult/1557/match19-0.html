class Solution {
   public boolean hasAllCodes(String s, int k) {
       if(k > s.length()){
            return false;
        }
        Set<Integer> set = new HashSet<>(1 << k);
        for (int i = 0;i< (1 << k);i++){
            set.add(i);
        }

        int cur = 0;
        for (int i = 0;i<k-1;i++){
            cur = cur * 2 + (s.charAt(i) - '0');
        }
        for (int i = k-1;i < s.length();i++){
            cur = cur * 2 + (s.charAt(i) - '0');
            set.remove(cur);
            cur = cur - ((s.charAt(i - k + 1) - '0') << (k-1));
        }

        return set.isEmpty();
    }
}
