class Solution {
    public boolean hasAllCodes(String s, int k) {
        int mask = (1 << k) - 1;
        boolean[] exist = new boolean[mask + 1];
        int bits = 0;
        for(int i = 0; i < k - 1 && i < s.length(); i++){
            bits <<= 1;
            bits &= mask;
            bits |= (s.charAt(i) - '0');
        }
        for(int i = k - 1; i < s.length(); i++){
            bits <<= 1;
            bits &= mask;
            bits |= (s.charAt(i) - '0');
            exist[bits] = true;
        }
        for(boolean x : exist){
            if(!x){
                return false;
            }
        }
        return true;
    }
}
