class Solution {
    private  int helper(String s){
        int res=0;
        for(int i=0;i<s.length();i++){
            res=res*2+(int)(s.charAt(i)-'0');
        }
        //System.out.println(res);
        return res;
    }
    public boolean hasAllCodes(String s, int k) {
        Set<Integer> set=new HashSet<>();
        int cur=0,l=0,r=0;
        int n=s.length();
        int need=(1<<k);
        if(n-k+1<need) return false;
        for(int i=0;i+k<=n;i++){
            set.add(helper(s.substring(i,i+k)));
        }
        //System.out.println(set.size()+"ok"+need);
        return set.size()==need;
    }
}
