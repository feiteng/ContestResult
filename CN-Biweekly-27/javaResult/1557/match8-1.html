class Solution {
    public boolean hasAllCodes(String s, int k) {
        boolean[] seen = new boolean[1 << k];
        int num = 0;
        for(int i = 0 ; i < s.length(); i++){
            int digit = s.charAt(i) == '0' ? 0 : 1;
            num &= ~(1 << (k - 1));
            num <<= 1;
            num |= digit;
            if(i >= k - 1){
                seen[num] = true;
            }
        
}
        for(int i = 0 ; i < (1 << k); i++){
            if(!seen[i]){
                return false;
            }
        }
        return true;
    }
}
