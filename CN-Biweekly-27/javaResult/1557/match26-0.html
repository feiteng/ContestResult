class Solution {
    public boolean hasAllCodes(String s, int k) {
        int len = s.length();
        Set<String> set = new HashSet<>();
        for (int i = 0; i + k <= len; i++) {
            set.add(s.substring(i, i + k));
        }
        return set.size() == 1 << k;
    }
}
