class Solution {
 public boolean hasAllCodes(String s, int k) {
        char[] arr = s.toCharArray();
        int len = arr.length;

        Set<String> set = new HashSet<>();
        for (int i = 0; i <= len - k; i++) {
            String subStr = s.substring(i, i + k);
            set.add(subStr);
        }

        return set.size() == (1 << k);
    }
}
