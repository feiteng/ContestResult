class Solution {
    public boolean hasAllCodes(String s, int k) {
        Set<String> set = new HashSet<>();
        int n = s.length();
        char[] chars = s.toCharArray();
        for (int l = 0, r = 0; r < n; r++) {
            int len = r - l + 1;
            if (len == k) {
                set.add(String.valueOf(chars, l, len));
                l++;
            }
        }
        int all = 1 << k;
        return set.size() == all;
    }
}
