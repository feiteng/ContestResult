class Solution {
    public boolean hasAllCodes(String s, int k) {
        if(s.length()<=k){
            return false;
        }
        int[] ans=new int[1<<k];
        int cur=0;
        for(int i=0;i<k;i++){
            cur*=2;
            cur+=(s.charAt(i)-'0');
        }
        ans[cur]=1;
        for(int i=k;i<s.length();i++){
            cur&=((1<<(k-1))-1);
            cur*=2;
            cur+=(s.charAt(i)-'0');
            // System.out.println(cur);
            ans[cur]=1;
        }
        for(int i=0;i<ans.length;i++){
            if(ans[i]==0){
                return false;
            }
        }
        return true;

    }
}
