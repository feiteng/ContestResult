class Solution {
    public boolean hasAllCodes(String s, int k) {
        if (s.length() < k) return false;

        Set<String> set = new HashSet<>();
        StringBuilder sb = new StringBuilder();

        for (int i = 0; i < k; i++) {
            sb.append(s.charAt(i));
        }

        set.add(sb.toString());
        for (int i = k; i < s.length(); i++) {
            sb.deleteCharAt(0);
            sb.append(s.charAt(i));
            
set.add(sb.toString());
        }

        if (set.size() == 1 << k) {
            return true;
        }
        return false;
    }
}
