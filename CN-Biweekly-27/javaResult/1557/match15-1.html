class Solution {
    public boolean hasAllCodes(String s, int k) {
        char[] c = s.toCharArray();
        int total = 1 << k;
        Set<String> st = new HashSet<>(1 << k);
        for (int i = 0; i <= c.length - k; i++) {
            st.add(s.substring(i, i + k));
        }
        if (st.size() == total) {
            return true;
        }
        return false;
    }

}
