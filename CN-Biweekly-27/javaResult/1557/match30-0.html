class Solution {
    public boolean hasAllCodes(String s, int k) {
        Set<Integer> set = construct(k);
        
        for (int i = 0; i <= s.length() - k; i++) {
            set.remove(helper(s.substring(i, i + k)));
        }
        
        return set.isEmpty();
    }
    
    Set<Integer> construct(int k) {
        Set<Integer> set = new HashSet<>();
        for (int i =0; i < (1 << k); i++) {
            set.add(i);
        }
        return set;
    }
    
    int helper(String s) {
        char[] arr = s.toCharArray();
        int res = 0;
        for (int i = arr.length - 1 ; i >= 0; i--) {
            int val = Integer.valueOf(s.substring(i, i+1));
            res += val * (1 << (arr.length - 1 - i));
        }
        
        return res;
    }
}
