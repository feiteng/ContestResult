class Solution {
     public boolean hasAllCodes(String s, int k) {
         if (k>s.length()) return false;
         HashSet<String> strings = new HashSet<>();
        StringBuilder builder = new StringBuilder();
        for (int i = 0; i < k; i++) {
            builder.append(s.charAt(i));
        }
        strings.add(builder.toString());
        for (int i = k; i < s.length(); i++) {
            builder.deleteCharAt(0);
            builder.append(s.charAt(i));
            strings.add(builder.toString());
        }
        return strings.size() == (1 << k);
    }

    private boolean dfs(StringBuilder builder, String target, int k) {
        if (builder.length() == k) return target.contains(builder.toString());
        builder.append('0');
    
        if (!target.contains(builder.toString())) return false;
        if (!dfs(builder, target, k)) return false;
        builder.deleteCharAt(builder.length() - 1);
        builder.append('1');
        if (!target.contains(builder.toString())) return false;
        if (!dfs(builder, target, k)) return false;
        builder.deleteCharAt(builder.length() - 1);
        return true;
    }
}
