class Solution {
    public boolean canBeEqual(int[] target, int[] arr) {
        if (target.length != arr.length) {
            return false;
        }
        target = Arrays.stream(target).sorted().toArray();
        arr = 
Arrays.stream(arr).sorted().toArray();
        for (int i = 0; i < target.length; i++) {
            if(target[i]!=arr[i]) {
                return false;
            }
        }
        return true;
    }
}
