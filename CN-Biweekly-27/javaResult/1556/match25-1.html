class Solution {
    public boolean canBeEqual(int[] target, int[] arr) {
        
int l[]=new int[1001];
        int r[]=new int[1001];
        for(int i=0;i<arr.length;i++){
            l[arr[i]]++;
            r[target[i]]++;
        }
        
        for(int i=0;i<1001;i++){
            if(l[i]!=r[i])
                return false;
        }
        return true;
    }
}
