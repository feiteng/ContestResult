class Solution {
    public boolean canBeEqual(int[] target, int[] arr) {
        int n = target.length;
        int[] ct1 = new int[1001];
        int[] ct2 = new int[1001];
        for(int i = 0; i < n; i++) {
            ct1[target[i]]++;
            ct2[arr[i]]++;
        }
        for(int i = 1; i <= 1000; i++){
            if(ct1[i] != ct2[i]) return false;
        }
        return true;
    }
}
