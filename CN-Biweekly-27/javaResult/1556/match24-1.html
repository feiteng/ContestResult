class Solution {
    public boolean canBeEqual(int[] target, int[] arr) {
        if (target.length != arr.length) {
            return false;
        }
        
int count[] = new int[1001];
        for (int t : target) {
            count[t]++;
        }
        for (int a : arr) {
            count[a]--;
        }
        for (int c : count) {
            if (c != 0) {
                return false;
            }
        }
        return true;
    }
}
