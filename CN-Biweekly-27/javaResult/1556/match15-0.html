class Solution {
    public boolean canBeEqual(int[] target, int[] arr) {
        int[] cnt1 = new int[1005];
        int[] cnt2 = new int[1005];
        for (int a : target) {
            cnt1[a] ++;
        }
        for (int a : arr) {
            cnt2[a] ++;
        }
        for (int i = 0; i <= 1000; i++) {
            if (cnt1[i] != cnt2[i]) {
                return false;
            }
        }
        return true;

    }

}
