class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
    	boolean[][] neighbor=new boolean[n][n];
    	for
(int i=0;i<prerequisites.length;i++){
			neighbor[prerequisites[i][0]][prerequisites[i][1]]=true;
		}
    	for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				for(int k=0;k<n;k++) {
					if(neighbor[j][i] && neighbor[i][k]) neighbor[j][k]=true;

				}
			}
		}
    	List<Boolean> list=new ArrayList<>();
    	for(int i=0;i<queries.length;i++){
			list.add(neighbor[queries[i][0]][queries[i][1]]);
		}
    	return list;
    }
}
