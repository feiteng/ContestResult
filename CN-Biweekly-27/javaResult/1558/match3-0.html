class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        int[][] edges = new int[n][n];
        for (int i = 0;i < n;i++) {
            for (int j = 0;j < n;j++) {
                edges[i][j] = 0;
            }
        }
        for (int[] p: prerequisites) {
            edges[p[0]][p[1]] = 1;
        }
        for (int k = 0;k < n;k++) {
            for (int i = 0;i < n;i++) {
                for (int j = 0;j < n;j++) {
                    if (edges[i][k] == 1 && edges[k][j] == 1) {
                        edges[i][j] = 1;
                    }
                }
            }
        }
        List<Boolean> ret = new ArrayList<Boolean>();
        for (int[] q: queries) {
            if (edges[q[0]][q[1]] == 1) {
                ret.add(true);
            } else {
                ret.add(false);
            }
        }
        return ret;
    }
}
