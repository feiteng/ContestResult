class Solution {
   void getPreList(int n,
List<Integer>[] pre,int cur,boolean[][] ok){
        //List<Integer> res=new ArrayList<>();
        Queue<Integer> que=new LinkedList<>();
       boolean[] vis=new boolean[n];
        for(int fa:pre[cur]){
            que.add(fa);
            vis[fa]=true;
        }
        
        while(!que.isEmpty()){
            int u=que.poll();
            ok[u][cur]=true;
            for(int f:pre[u]) if(!vis[f]){
                que.add(f);
                vis[f]=true;
            }
        }
    }
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        boolean[][] ok=new boolean[n][n];
        List<Integer>[] pre=new List[n];
        for(int i=0;i<n;i++) pre[i]=new ArrayList<>();
        for(int[] it:prerequisites){
            int a=it[0];
            int b=it[1];
            pre[b].add(a);
        }
        for(int i=0;i<n;i++){
            getPreList(n,pre,i,ok);
        }
        List<Boolean> res=new ArrayList<>();
        
for(int[] q:queries){
            res.add(ok[q[0]][q[1]]);
        }
        return res;

    }
}
