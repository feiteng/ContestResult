class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        Map<Integer,Set<Integer>> map=new HashMap();
        for(int[] i:prerequisites){
            if(!map.containsKey(i[1]))map.put(i[1],new HashSet());
            map.get(i[1]).add(i[0]);
            if(map.containsKey(i[0])){
                for(Integer j:map.get(i[0])){
                    map.get(i[1]).add(j);
                }
            }
        }
        List<Boolean> ans=new ArrayList();
        for(int i=0;i<queries.length;i++){
            if(map.containsKey(queries[i][1])&&map.get(queries[i][1]).contains(queries[i][0]))ans.add(true);
            else ans.add(false);
        }
        return ans;
    }
}
