import java.util.*;

class Solution {
    boolean[] flag;
    Map<Integer, Set<Integer>> map;

    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        flag = new boolean[n];
        
map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.put(i, new HashSet<>());
        }
        for (int[] div : prerequisites) {
            map.get(div[1]).add(div[0]);
        }
        for (int i = 0; i < n; i++) {
            dfs(i);
        }
        List<Boolean> list = new ArrayList<>();
        for (int[] query : queries) {
            list.add(
map.get(query[1]).contains(query[0]));
        }
        return list;
    }

    private Set<Integer> dfs(int i) {
        if (!flag[i]) {
            flag[i] = true;
            final Set<Integer> set = map.get(i);
            Set<Integer> additionSet = new HashSet<>();
            for (int index : set) {
                additionSet.addAll(dfs(index));
            }
            set.addAll(additionSet);
        }
        return map.get(i);
    }
}
