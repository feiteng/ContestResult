class Solution {
    HashMap<Integer,List<Integer>> map;
    boolean[] visited;
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        map=new HashMap<>();
        for(int i=0;i<n;i++){
            map.put(i,new ArrayList<>());
        }
        for(int[] p:prerequisites){
            map.get(p[0]).add(p[1]);
        }
        List<Boolean> ans=new ArrayList<>();
        for(int[] q:queries){
            visited=new boolean[n];
            ans.add(f(q[0],q[1]));
        }
        return ans;
    }
    boolean f(int a,int b){
        if(visited[a]){
            return false;
        }
        if(a==b){
            return true;
        }
        for(int next:map.get(a)){
            if(f(next,b)){
                return true;
            }
        }
        visited[a]=true;
        return false;
    }
}
