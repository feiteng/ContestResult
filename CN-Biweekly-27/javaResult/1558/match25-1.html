class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        List<List<Integer>> adjList = new ArrayList<>();
        List<Set<Integer>> allPres = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            adjList.add(new ArrayList<>());
            allPres.add(new HashSet<>());
        }
        
        int[] inDegrees = new int[n];

        
        for (int[] pre : prerequisites) {
            // to depends on from
            int from = pre[0];
            int to = pre[1];
            adjList.get(from).add(to);
            allPres.get(to).add(from);
            inDegrees[to] ++;
        }
        
        
        Queue<Integer> queue = new ArrayDeque<>();
        for (int v = 0; v < n; v++) {
            if (inDegrees[v] == 0) {
                queue.offer(v);
            
}
        }
        
        while(!queue.isEmpty()) {
            final int v = queue.poll();
            Set<Integer> pres = allPres.get(v);
            Set<Integer> newPres = new HashSet<>();
            for (int p : pres) {
                newPres.addAll(allPres.get(p));
            }
            pres.addAll(newPres);
            
            for (int to : adjList.get(v)) {
                inDegrees[to] --;
                
if (inDegrees[to] == 0) {
                    queue.offer(to);
                }
            }
        }
        
        List<Boolean> res = new ArrayList<>();
        for (int[] q : queries) {
            
if (allPres.get(q[1]).contains(q[0])) {
                res.add(true);
            } else {
                res.add(false);
            }
        }
        return res;
        
    }
}
