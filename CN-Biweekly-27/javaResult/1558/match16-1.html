	Set<Integer> getSub(int p) {
		if (memo[p] != null) {
			return memo[p];
		}
		memo[p] = new HashSet<>();
		memo[p].add(p);
		for (int s : lists[p]) {
			memo[p].addAll(getSub(s));
		}
		return memo[p];
