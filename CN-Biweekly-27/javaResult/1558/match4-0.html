class Solution {
  public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {

    List<Boolean> ret = new ArrayList<>(queries.length);

    int[][] graph = new int[n][n];

    for (int[] edge : prerequisites) {
      int pre = edge[0], target = edge[1];
      graph[pre][target] = 1;
    }

    for (int a = 0; a < n; a++) {
      for (int b = 0; b < n; b++) {
        if (graph[b][a] == 1) {
          for (int c = 0; c < n; c++) {
            if (graph[a][c] == 1) graph[b][c] = 1;
          }
        }
      }
    }

    for (int[] query : queries) {
      ret.add(graph[query[0]][query[1]] == 1);
    }

    return ret;
  }
}
