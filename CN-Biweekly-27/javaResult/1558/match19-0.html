class Solution {

    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        Map<Integer, Set<Integer>> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.put(i, new HashSet<>());
        }
        for (int[] prerequisite : prerequisites) {
            map.get(prerequisite[0]).add(prerequisite[1]);
        }
        Boolean[][] cache = new Boolean[n][n];
        List<Boolean> ans = new ArrayList<>();
        for (int[] query : queries) {
            ans.add(dfs(map, query[0], query[1], cache));
        }
        return ans;
    }

    private boolean dfs(Map<Integer, Set<Integer>> map, int start, int end, Boolean cache[][]) {
        if (cache[start][end] != null) {
            return cache[start][end];
        }
        Set<Integer> set = map.get(start);
        boolean ans = false;
        if (set.contains(end)) {
            ans = true;
        } else {
            for (Integer child : set) {
                if (dfs(map, child, end, cache)) {
                    ans = true;
                    break;
                }
            }
        }
        return cache[start][end] = ans;
    }
}
