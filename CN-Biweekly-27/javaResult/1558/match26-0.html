class Solution {
    
    List<Integer>[] graph;
    
    Map<Integer, Set<Integer>> map = new HashMap<>();
    
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        List<Boolean> res = new ArrayList<>();
        int[] used = new int[n];
        
        graph = new List[n];
        for (int i = 0; i < n; i++) {
            graph[i] = new ArrayList<>();
            map.put(i, new HashSet<>());
        }
        for (int[] e : prerequisites) {
            graph[e[0]].add(e[1]);
        }
        
        
        for (int[] e : queries) {
            int from = e[0];
            int to = e[1];
            
            res.add(helper(from, to, new HashSet<>()));
            
//             if (used[from] == 0) {
//                 helper(from, map.get(from));
//                 used[from] = 1;
//             }
            
//             if (map.get(from).contains(to)) res.add(true);
//             else res.add(false);
        }
        
        return res;
    }
    
    private boolean helper(int from, int to, Set<Integer> used) {
        if (from == to) return true;
        if (used.contains(from)) return false;
        used.add(from);
        // if (map.get(from).size() != 0) return map.get(from).contains(to);
        for (int next : graph[from]) {
            if (helper(next, to, used)) return true;
        }
        return false;
        
    }
}
