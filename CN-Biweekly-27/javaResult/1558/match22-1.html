class Solution {
    Boolean[][] res;
    
    public List<Boolean> checkIfPrerequisite(int n, int[][] pre, int[][] que) {
        List<Boolean> ans = new LinkedList<>();
        List<Set<Integer>> graph = new ArrayList<>(n);
        res = new Boolean[n][n];
        
for(int i=0;i<n;i++){
            graph.add(new HashSet<Integer>());
        }
        for(int[] p:pre){
            graph.get(p[1]).add(p[0]);
        }
        
for(int[] q:que){
            ans.add(ask(q,graph));
        }
        return ans;
    }
    
    private boolean ask(int[] q, List<Set<Integer>> graph){
        return dfs(q[1],q[0],graph);
    }
    
    private boolean dfs(int f, int child, List<Set<Integer>> graph){
        if(res[f][child]!=null) return res[f][child];
        if(graph.get(f).contains(child)){
            res[f][child] = true;
            return true;
        }
        for(int x:graph.get(f)){
            if(dfs(x,child,graph)){
                res[f][child] = true;
                return true;
            }
        }
        res[f][child] = false;
        return false;
    }
}
