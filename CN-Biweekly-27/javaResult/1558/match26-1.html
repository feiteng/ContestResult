class Solution {

  private boolean[] res;
  private Set<Integer>[] adj;
  private Map<Integer, Map<Integer, List<Integer>>> map;

  public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
    int m = queries.length;
    adj = new Set[n];
    map = new HashMap<>();
    
res = new boolean[m];
    for (int i = 0; i < n; i++) {
      adj[i] = new HashSet<>();
      map.put(i, new HashMap<>());
    }
    for (int[] prerequisite : prerequisites) {
      adj[prerequisite[0]].add(prerequisite[1]);
    }
    for (int i = 0; i < m; i++) {
      int[] query = queries[i];
      map.get(query[0]).putIfAbsent(query[1], new ArrayList<>());
      map.get(query[0]).get(query[1]).add(i);
    }
    for (int i = 0; i < n; i++) {
      boolean[] seen = new boolean[n];
      dfs(i, i, seen);
    }
    List<Boolean> resList = new ArrayList<>();
    for (boolean i : res) {
      resList.add(i);
    
}
    return resList;
  }

  private void dfs(int origin, int cur, boolean[] seen) {
    for (int id : map.get(origin).getOrDefault(cur, Collections.emptyList())) res[id] = true;
    seen[cur] = true;
    for (int next : adj[cur]) {
      if(seen[next]){
        continue;
      }
      dfs(origin, next, seen);
    }
  }
}
