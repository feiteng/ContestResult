class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        Map<Integer, Set<Integer>> preMap = new HashMap<>();
        for (int[] pre : prerequisites) {
            Set<Integer> set = preMap.get(pre[1]);
            if (set == null) {
                set = new HashSet<>();
                preMap.put(pre[1], set);
            }
            set.add(pre[0]);
            if (preMap.containsKey(pre[0])) {
                set.addAll(preMap.get(pre[0]));
            
}
        }
        List<Boolean> result = new ArrayList<>();
        for (int[] query : queries) {
            Set<Integer> set = preMap.get(query[1]);
            if (set == null) {
                result.add(false);
            } else if (set.contains(query[0])) {
                result.add(true);
            } else {
                result.add(false);
            }
        }
        return result;
    }
}
