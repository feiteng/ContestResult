class Solution {
    class Node{
        List<Node> childs=new ArrayList<>();
        int idx;
        
        public Node(int i){
            idx=i;
        }
    }
    
    Map<Integer,Boolean> map=new HashMap<>();
    
    int MAX=200;
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        Node[] nodes=new Node[n];
        
        for(int i=0;i<n;i++){
            nodes[i]=new Node(i);
        }
        for(int[] p:prerequisites){
            int before=p[0];
            int aft=p[1];
            nodes[before].childs.add(nodes[aft]);
            map.put(before*MAX+aft,true);
            map.put(aft*MAX+before,false);
        }
        List<Boolean> res=new ArrayList<>(queries.length);
        
        for(int[] p:queries){
            int before=p[0];
            int aft=p[1];
            boolean tmp=isBefore(before,aft,nodes);
            res.add(tmp);
            map.put(before*MAX+aft,tmp);
        }
        return res;
    }
    
    private boolean isBefore(int b,int a,Node[] nodes){
        //from b,BFS find a
        Queue<Node> queue=new LinkedList<>();//无环，所以无需visited
        queue.add(nodes[b]);
        Set<Node> visited=new HashSet<>();
        while(!queue.isEmpty()){
            Node cur=queue.poll();
            if(cur.idx==a){
                return true;
            }
            int tmp=cur.idx*MAX+a;
            if(map.containsKey(tmp)){
                if( map.get(tmp))
                    return true;
            }
            for(Node c:cur.childs){
                if(visited.add(c))
                    queue.add(c);
            }
        }
        return false;
    }
}
