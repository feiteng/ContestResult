class Solution {
    public List<Boolean> checkIfPrerequisite(int n, int[][] prerequisites, int[][] queries) {
        List<Integer>[] path=new ArrayList[n];
        for(int i=0;i<n;i++){
            path[i]=new ArrayList<Integer>();
        }
        for(int i=0;i<prerequisites.length;i++){
            path[prerequisites[i][0]].add(prerequisites[i][1]);
        }
        List<Integer>[] path2=new ArrayList[n];
        for(int i=0;i<n;i++){
            path2[i]=new ArrayList<Integer>();
        }
        for(int i=0;i<n;i++){
            boolean[] visited=new boolean[n];
            dfs(path,i,i,path2,visited);
        }
        List<Boolean> res=new ArrayList<>();
        for(int i=0;i<queries.length;i++){
            if(path2[queries[i][0]].contains(queries[i][1])){
                res.add(true);
            }else{
                res.add(false);
            }
        }
        return res;
    }
    private void dfs(List<Integer>[] path,int start,int start1,List<Integer>[] path2,boolean[] visited){
        visited[start1]=true;
        if(path[start1].size()==0) return;
        for(int i=0;i<path[start1].size();i++){
            if(visited[path[start1].get(i)]==false){
                
                path2[start].add(path[start1].get(i));
                dfs(path,start,path[start1].get(i),path2,visited);
            }
        }
    }
}
