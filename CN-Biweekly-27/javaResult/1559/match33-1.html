class Solution {
    public static int[] dirs = new int[]{-1, 0, 1};
    public int cherryPickup(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        
        int[][][] res = new int[n][m][m];
        res[0][0][m-1] = grid[0][0] + grid[0][m - 1];
        int[] init = new int[]{0, m - 1};
        Queue<int[]> que = new LinkedList<>();
        que.add(init);
        
        
for(int i = 0; i < n - 1; i++) {
            int k = que.size();
            for(int t = 0; t < k; t++) {
                int[] loc = que.poll();
                for(int dir : dirs) {
                    int next1 = loc[0] + dir;
                    for(int dir2 : dirs) {
                        int next2 = loc[1] + dir2;
                        if(next1 < 0 || next2 < 0 || next1 >= m || next2 >= m) continue;
                        if(res[i + 1][next1][next2] == 0) que.add(new int[]{next1, next2});
                        int val = 0;
                        if(next1 == next2) val = res[i][loc[0]][loc[1]] + grid[i + 1][next1];
                        else val = res[i][loc[0]][loc[1]] + grid[i + 1][next1] + grid[i + 1][next2];
                        res[i + 1][next1][next2] = Math.max(res[i + 1][next1][next2], val);
                    
}
                }
            }
        }
        
        int ans = 0;
        for(int j = 0; j < m; j++) {
            for(int i = 0; i < m; i++) {
                ans = Math.max(ans, res[n - 1][j][i]);
            }
        }
        return ans;
    }
}
