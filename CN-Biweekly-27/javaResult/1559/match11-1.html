class Solution {
    public int cherryPickup(int[][] grid) {
        int row = grid.length;
        int column = grid[0].length;
        
int[][][] dp = new int[row][column][column];
        for(int r = 0;r < row;r++) {
            for (int rb1 = 0; rb1 < column; rb1++) {
                for (int rb2 = 0; rb2 < column; rb2++) {
                    dp[r][rb1][rb2] = Integer.MIN_VALUE;
                }
            }
        }
        dp[0][0][column - 1] = grid[0][0] + grid[0][column - 1];
        for(int r = 1;r < row;r++) {
            for(int rb1 = 0;rb1 < column;rb1++) {
                for(int rb2 = 0;rb2 < column;rb2++) {
                    int maxBase = Integer.MIN_VALUE;
                    for(int prevRb1 = rb1 - 1;prevRb1 <= rb1 + 1;prevRb1++) {
                        for(int prevRb2 = rb2 - 1;prevRb2 <= rb2 + 1;prevRb2++) {
                            if(prevRb1 >= 0 && prevRb1 < column && prevRb2 >= 0 && prevRb2 < column && dp[r-1][prevRb1][prevRb2] != Integer.MIN_VALUE) {
                                maxBase = Math.max(maxBase, dp[r - 1][prevRb1][prevRb2]);
                            }
                        }
                    }
                    if(maxBase != Integer.MIN_VALUE) {
                        if (rb1 == rb2) {
                            dp[r][rb1][rb2] = maxBase + grid[r][rb1];
                        } else {
                            dp[r][rb1][rb2] = maxBase + grid[r][rb1] + grid[r][rb2];
                        }
                    
}
                }
            }
        }
        int res = Integer.MIN_VALUE;
        for(int rb1 = 0;rb1 < column;rb1++){
            for(int rb2 = 0;rb2 < column;rb2++) {
                res = Math.max(res, dp[row - 1][rb1][rb2]);
            }
        }
        return res;
    }
}
