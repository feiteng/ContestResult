class Solution {
    public int cherryPickup(int[][] grid) {
        int rows=grid.length,cols=grid[0].length;
        int[][][] dp=new int[rows][cols][cols];
        
dp[0][0][cols-1]=grid[0][0]+grid[0][cols-1];
        for(int a=0;a<rows;a++){
            for(int b=0;b<cols;b++){
                
                for(int d=0;d<cols;d++){
                    
if(a==0){
                        if(b==0&&d==cols-1){
                            dp[a][b][d]=grid[0][0]+grid[0][cols-1];
                        }else{
                            dp[a][b][d]=-1;
                        }
                    }else{
                        if(b==0){
                            if(d==0){
                                int max1=Math.max(dp[a-1][0][0],dp[a-1][0][1]);
                                max1=Math.max(max1,dp[a-1][1][0]);
                                max1=Math.max(max1,dp[a-1][1][1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else if(d==cols-1){
                                int max1=Math.max(dp[a-1][0][cols-1],dp[a-1][0][cols-2]);
                                max1=Math.max(max1,dp[a-1][1][cols-1]);
                                max1=Math.max(max1,dp[a-1][1][cols-2]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else{
                                int max1=Math.max(dp[a-1][0][d],dp[a-1][0][d-1]);
                                max1=Math.max(max1,dp[a-1][1][d]);
                                max1=Math.max(max1,dp[a-1][1][d-1]);
                                max1=Math.max(max1,dp[a-1][0][d+1]);
                                max1=Math.max(max1,dp[a-1][1][d+1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }
                        }else if(b==cols-1){
                            if(d==0){
                                int max1=Math.max(dp[a-1][b][0],dp[a-1][b][1]);
                                max1=Math.max(max1,dp[a-1][b-1][0]);
                                max1=Math.max(max1,dp[a-1][b-1][1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else if(d==cols-1){
                                int max1=Math.max(dp[a-1][b][cols-1],dp[a-1][b][cols-2]);
                                max1=Math.max(max1,dp[a-1][b-1][cols-1]);
                                max1=Math.max(max1,dp[a-1][b-1][cols-2]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else{
                                int max1=Math.max(dp[a-1][b][d],dp[a-1][b][d-1]);
                                max1=Math.max(max1,dp[a-1][b-1][d]);
                                max1=Math.max(max1,dp[a-1][b-1][d-1]);
                                max1=Math.max(max1,dp[a-1][b][d+1]);
                                max1=Math.max(max1,dp[a-1][b-1][d+1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }
                        }else{
                            if(d==0){
                                int max1=Math.max(dp[a-1][b][0],dp[a-1][b][1]);
                                max1=Math.max(max1,dp[a-1][b-1][0]);
                                max1=Math.max(max1,dp[a-1][b-1][1]);
                                max1=Math.max(max1,dp[a-1][b+1][0]);
                                max1=Math.max(max1,dp[a-1][b+1][1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else if(d==cols-1){
                                int max1=Math.max(dp[a-1][b][cols-1],dp[a-1][b][cols-2]);
                                max1=Math.max(max1,dp[a-1][b-1][cols-1]);
                                max1=Math.max(max1,dp[a-1][b-1][cols-2]);
                                max1=Math.max(max1,dp[a-1][b+1][cols-1]);
                                max1=Math.max(max1,dp[a-1][b+1][cols-2]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else if(d==b){
                                int max1=Math.max(dp[a-1][b][d],dp[a-1][b][d-1]);
                                max1=Math.max(max1,dp[a-1][b-1][d]);
                                max1=Math.max(max1,dp[a-1][b-1][d-1]);
                                max1=Math.max(max1,dp[a-1][b][d+1]);
                                max1=Math.max(max1,dp[a-1][b-1][d+1]);
                                max1=Math.max(max1,dp[a-1][b+1][d-1]);
                                max1=Math.max(max1,dp[a-1][b+1][d]);
                                max1=Math.max(max1,dp[a-1][b+1][d+1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                            }else{
                                int max1=Math.max(dp[a-1][b][d],dp[a-1][b][d-1]);
                                max1=Math.max(max1,dp[a-1][b-1][d]);
                                max1=Math.max(max1,dp[a-1][b-1][d-1]);
                                max1=Math.max(max1,dp[a-1][b][d+1]);
                                max1=Math.max(max1,dp[a-1][b-1][d+1]);
                                max1=Math.max(max1,dp[a-1][b+1][d-1]);
                                max1=Math.max(max1,dp[a-1][b+1][d]);
                                max1=Math.max(max1,dp[a-1][b+1][d+1]);
                                if(max1>=0){
                                    dp[a][b][d]=max1+grid[a][b]+grid[a][d];
                                }else{
                                   dp[a][b][d]=-1; 
                                }
                                //if(b>a||d<cols-a) dp[a][b][d]=-1;
                            }
                        
}
                    }
                }
                
            }
        }
        
        int res=0;
        
        for(int b=0;b<cols;b++){
            for(int d=0;d<cols;d++){
                res=Math.max(res,dp[rows-1][b][d]);
            }
        }
        
        return res;
    }
}
