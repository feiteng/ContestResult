class Solution {

    public int cherryPickup(int[][] grid) {
        int[][][] dp = new int[grid.length][grid[0].length][grid[0].length];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                for (int k = 0; k < grid[0].length; k++) {
                    dp[i][j][k] = -10000000;
                }
            }
        }
        dp[0][0][grid[0].length - 1] = grid[0][0] + grid[0][grid[0].length - 1];
        for (int i = 1; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                for (int k = 0; k < grid[0].length; k++) {
                    if (j == k) {
                        dp[i][j][k] = grid[i][j] + Math.max(get(dp, i - 1, j - 1, k - 1),
                                Math.max(get(dp, i - 1, j - 1, k),
                                        Math.max(get(dp, i - 1, j - 1, k + 1), Math.max(get(dp, i - 1, j, k),
                                                Math.max(get(dp, i - 1, j, k + 1), get(dp, i - 1, j + 1, k + 1))))));
                    } else {
                        int max = -1000000;
                        for (int n = -1; n <= 1; n++) {
                            for (int m = -1; m <= 1; m++) {
                                max = Math.max(max, get(dp, i - 1, j + n, k + m));
                            }
                        }
                        dp[i][j][k] = grid[i][j] + grid[i][k] + max;
                    }
                }
            }
        }
        int max = 0;
        for (int i = 0; i < grid[0].length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                max = Math.max(max, dp[grid.length - 1][i][j]);
            }
        }
        return max;
    }

    private int get(int[][][] dp, int i, int j, int k) {
        return j < 0 || j >= dp[0].length || k < 0 || k >= dp[0].length ? -1000000 : dp[i][j][k];
    }
}
