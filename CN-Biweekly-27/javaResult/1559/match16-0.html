class Solution {
    public int cherryPickup(int[][] grid) {
        int l = grid[0].length;
        int r = grid.length;
        int[] dirs = {-1,0,1};
        int[][][] dp = new int[r][l][l];
        for (int i = 0; i < r; i++) {
            for (int j = 0; j < l; j++) {
                for (int k = 0; k < l; k++) {
                    dp[i][j][k] = -0x3f3f3f3f;
                }
            }
        }
        dp[0][0][l - 1] = grid[0][0] + grid[0][l - 1];
        for (int i = 1; i < r; i++) {
            int jm = Math.min(i + 1, l);
            for (int j = 0; j < jm; j++) {
                int km = Math.max(l - i - 1, 0);
                for (int k = km; k < l; k ++) {
                    for (int dir : dirs) {
                        for (int dir2 : dirs){
                            int dj = dir + j;
                            int dk = dir2 + k;
                            if (dj < 0 || dj >= l || dk < 0 || dk >= l) {
                                continue;
                            }
                            if (j == k) {
                                dp[i][j][k] = Math.max(dp[i][j][k], dp[i - 1][dj][dk] + grid[i][j]);
                            } else {
                                dp[i][j][k] = Math.max(dp[i][j][k], dp[i - 1][dj][dk] + grid[i][j] + grid[i][k]);
                            }
                        }
                    }
                }
            }
        }
        int ans = -0x3f3f3f3f;
        for (int i = 0; i < l; i++) {
            for (int j = 0; j < l; j++) {
                ans = Math.max(ans, dp[r-1][i][j]);
            }
        }
        return ans;
    }

}
