class Solution {
   private int rowCount;
    private int colCount;
    private Integer[][][] memo;
    private int[][] grid;

    private int dp(int curCol1, int curCol2, int curRow) {
        if (curCol1 < 0 || curCol1 >= colCount || curCol2 < 0 || curCol2 >= colCount) {
            return 0;
        }

        if (curRow == rowCount) {
            return 0;
        }

        if (memo[curCol1][curCol2][curRow] != null) {
            return memo[curCol1][curCol2][curRow];
        }

        int ans = 0;
        if (curCol1 == curCol2) {
            ans += grid[curRow][curCol1];
        } else {
            ans += grid[curRow][curCol1] + grid[curRow][curCol2];
        }

        int max = 0;
        for (int dy1 = -1; dy1 <= 1; dy1++) {
            for (int dy2 = -1; dy2 <= 1; dy2++) {
                max = Math.max(max, dp(curCol1 + dy1, curCol2 + dy2, curRow + 1));
            }
        }

        memo[curCol1][curCol2][curRow] = ans + max;
        return memo[curCol1][curCol2][curRow];
    }

    public int cherryPickup(int[][] grid) {
        rowCount = grid.length;
        colCount = grid[0].length;
        this.grid = grid;

        memo = new Integer[colCount][colCount][rowCount];
        return dp(0, colCount - 1, 0);
    }
}
