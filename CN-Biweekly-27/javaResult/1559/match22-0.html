class Solution {
    public int max(int x,int y,int z,int m,int n, int a1,int a2,int a3,int a4)
    {
        
        return Math.max(x,Math.max(y,Math.max(z,Math.max(m,Math.max(n,Math.max(a1,Math.max(a2,Math.max(a3,a4))))))));
    }
    public int cherryPickup(int[][] grid) {
        int rows = grid.length;
        int cols = grid[0].length;
        int[][][] dp = new int[cols][cols][2];
        for(int j = 0 ; j < cols; j++){
                for (int k = 0; k < cols; k++){
                    dp[j][k][0] = -10000;           
                }
        }
        dp[0][cols - 1][0] = grid[0][0] + grid[0][cols - 1];
        for (int i = 1 ; i < rows; i++){
            for(int j = 0 ; j < cols; j++){
                for (int k = 0; k < cols; k++){
                    if (j == k) 
                    {
                        int j1 = Math.max(j-1,0);
                        int j2 = Math.min(j + 1,cols - 1);
                        int k1 = Math.max(k-1,0);
                        int k2 = Math.min(k + 1,cols - 1);
                        dp[j][k][i % 2] = max(dp[j1][k1][(i + 1) % 2] ,
                                             dp[j1][k][(i + 1) % 2],
                                             dp[j1][k2][(i + 1) % 2] ,
                                             dp[j][k1][(i + 1) % 2] ,
                                             dp[j][k][(i + 1) % 2],
                                             dp[j][k2][(i + 1) % 2] ,
                                             dp[j2][k1][(i + 1) % 2],
                                             dp[j2][k][(i + 1) % 2] ,
                                             dp[j2][k2][(i + 1) % 2] 
                                             ) + grid[i][j] ;
                    }
                    else 
                    {
                        int j1 = Math.max(j-1,0);
                        int j2 = Math.min(j + 1,cols - 1);
                        int k1 = Math.max(k-1,0);
                        int k2 = Math.min(k + 1,cols - 1);
                        dp[j][k][i % 2] =max(dp[j1][k1][(i + 1) % 2] ,
                                             dp[j1][k][(i + 1) % 2],
                                             dp[j1][k2][(i + 1) % 2] ,
                                             dp[j][k1][(i + 1) % 2] ,
                                             dp[j][k][(i + 1) % 2],
                                             dp[j][k2][(i + 1) % 2] ,
                                             dp[j2][k1][(i + 1) % 2],
                                             dp[j2][k][(i + 1) % 2] ,
                                             dp[j2][k2][(i + 1) % 2] 
                                             ) + grid[i][j] + grid[i][k];
                    }
            }
        }
    }
    int res = 0;
    for (int i = 0; i < cols ; i++)
    {
        for (int j = 0; j < cols; j++)
        {
            res = Math.max(res, dp[i][j][(rows + 1) % 2]);
        }
        
    }
        return res;
    }
}
