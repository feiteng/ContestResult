class Solution {
    int[][][] dp;
    int[] dir={-1,0,1};
    int f(int[][] grid,int row,int c1,int c2){
        if(row==grid.length-1){
            return grid[row][c1]+grid[row][c2];
        }
        if(dp[row][c1][c2]!=-1){
            return dp[row][c1][c2];
        }
        dp[row][c1][c2]=0;

        for(int i=0;i<3;i++){
            for(int j=0;j<3;j++){
                int nc1=c1+dir[i];
                int nc2=c2+dir[j];
                if(nc1==nc2||nc1<0||nc2<0||nc2>=grid[0].length||nc1>=grid[0].length){
                    continue;
                }
                dp[row][c1][c2]=Math.max(dp[row][c1][c2],
                        f(grid,row+1,nc1,nc2));
            }
        }
        dp[row][c1][c2]+=(grid[row][c1]+grid[row][c2]);
        return dp[row][c1][c2];
    }
    public int cherryPickup(int[][] grid) {
        dp=new int[grid.length][grid[0].length][grid[0].length];
        for(int[][] a:dp){
            for(int[] b:a){
                Arrays.fill(b,-1);
            }
        }
        return f(grid,0,0,grid[0].length-1);
    }
}
