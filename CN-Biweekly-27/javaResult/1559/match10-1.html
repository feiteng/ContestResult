class Solution {
    public int cherryPickup(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        int[][][] dp = new int[m][n][n];
        
        for (int i = 0;i<n;i++){
            for (int j = 0;j<n;j++){
                dp[0][i][j] = -1;
            
}
        }
        
        dp[0][0][n-1] = grid[0][0] + grid[0][n-1];
        
        for (int i = 1;i<m;i++){
            for (int j = 0;j<n;j++){
                for (int k = 0;k<n;k++){
                    int cur = dp[i-1][j][k];
                    if(j-1 >= 0)
                        cur = Math.max(cur,dp[i-1][j-1][k]);
                    if(j-1 >= 0 && k-1 >= 0)
                        cur = Math.max(cur,dp[i-1][j-1][k-1]);
                    if(j-1 >= 0 && k + 1 < n)
                        cur = Math.max(cur,dp[i-1][j-1][k+1]);
                    if(k-1 >= 0)
                        cur = Math.max(cur,dp[i-1][j][k-1]);
                    if(k + 1 < n)
                        cur = Math.max(cur,dp[i-1][j][k+1]);
                    if(j + 1 < n && k-1 >= 0)
                        cur = Math.max(cur,dp[i-1][j+1][k-1]);
                    if(j + 1 < n && k + 1 < n)
                        cur = Math.max(cur,dp[i-1][j+1][k+1]);
                    if(j + 1 < n)
                        cur = Math.max(cur,dp[i-1][j+1][k]);
                    if(cur == -1){
                        dp[i][j][k] = -1;
                        continue;
                    }
                    cur += grid[i][j];
                    if(j != k){
                        cur += grid[i][k];
                    }
                    dp[i][j][k] = cur;
                
}
            }
        }
        
        int max = 0;
        for (int i = 0;i<n;i++){
            for (int j = 0;j<n;j++){
                max = Math.max(max,dp[m-1][i][j]);
            }
        }
        
        return max;
    }
}
