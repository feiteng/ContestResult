class Solution {
    private final int[] STEPS = new int[]{-1, 0, 1};
    public int cherryPickup(int[][] grid) {
        
int m = grid.length;
        int n = grid[0].length;
        
        int[][][] dp = new int[m][n][n];
        
        for (int i = 0 ; i < n; i++) for (int j = 0; j < m; j++) Arrays.fill(dp[j][i], -10000000);
        dp[0][0][n-1] = grid[0][0] + grid[0][n-1];
        // dp[0][0][n-2] = n - 2 != 0? (grid[0][0] + grid[0][n-1]) : grid[0][0];
        // dp[0][1][n-1] = n - 1 != 1 ? (grid[0][1] + grid[0][n-1]) : grid[0][1];
        //         dp[0][1][n-2] = n - 2 != 1 ? (grid[0][1] + grid[0][n-2]) : grid[0][1];

        
        for (int i =0; i < m-1; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < n; k++) {
                    // if (i == 0 && (j > 1 || k < n-2)) continue;
                    if (dp[i][j][k] < 0) continue;
                    for (int x : STEPS) {
                        for (int y : STEPS) {
                            if (dp[i][j][k] < 0) continue;
                            if (j - x >= 0 && j - x < n && k - y >= 0 && k - y < n) {
                            dp[i+1][j-x][k-y] = Math.max(((j-x == k-y) ? grid[i+1][j-x] : grid[i+1][j-x] + grid[i+1][k-y]) 
                                + dp[i][j][k], dp[i+1][j-x][k-y]);

                            }
                        
}
                    }
                }
            }
        }
        
        int res = 0;
        for (int i = 0; i < n; i++) for (int j = 0; j < n; j++) {
            res = Math.max(res, dp[m-1][i][j]);
        }
        return res;
    }
}
