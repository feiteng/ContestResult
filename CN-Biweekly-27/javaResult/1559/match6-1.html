class Solution {
    
public int cherryPickup(int[][] grid) {
    	int m=grid.length,n=grid[0].length;
    	int[][][] dp=new int[m][n+2][n+2];
    	for(int i=0;i<m;i++){
			for(int j=0;j<n+2;j++){ 
				for(int k=0;k<n+2;k++) {
					dp[i][j][k]=Integer.MIN_VALUE;
				}
			}
		}

    	dp[0][1][n]=grid[0][0]+grid[0][n-1];
    	for(int i=1;i<m;i++){
			for(int j=1;j<=n;j++){
				for(int k=j;k<=n;k++) {
					int max=-1;
					for(int x=j-1;x<=j+1;x++) {

						for(int y=k-1;y<=k+1;y++) {
							if(x>y) continue;
							max=Math.max(max, dp[i-1][x][y]);
						}
					}
					if(max<0) continue;
					dp[i][j][k]=max+grid[i][j-1];
					if(j!=k) dp[i][j][k]+=grid[i][k-1];
				}

			}
		}
    	int max=-1;
    	for(int i=1;i<=n;i++){
    		for(int j=i;j<=n;j++){
    			max=Math.max(max, dp[m-1][i][j]);
			}
		}
    	return max;
    }
}
