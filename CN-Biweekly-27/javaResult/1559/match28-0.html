class Solution {
    public int cherryPickup(int[][] grid) {
        int n = grid.length, m = grid[0].length, max = 0, res = 0;;
        int [][][] dp = new int[n][m][m];        
        for(int i = 0; i < n; i++)
            for(int j = 0; j < m; j++)
                Arrays.fill(dp[i][j], -10000000);
        dp[0][0][m - 1] = grid[0][0] + grid[0][m - 1];
        for(int i = 1; i < n; i++)
        {
            for(int left = 0; left < m; left++)
            {
                for(int right = left + 1; right < m; right++)
                {
                    for(int i0 = -1; i0 <= 1; i0++)
                    {
                        for(int j0 = -1; j0 <= 1; j0++)
                        {
                            int  dl = left + i0;
                            int dr = right + j0;
                            if(dl < 0 || dl >= dr || dr >= m) continue;
                            dp[i][left][right] = Math.max(dp[i][left][right], dp[i - 1][dl][dr] + grid[i][left] + grid[i][right]);
                        }
                    }
                }
            }
        }
        for(int left = 0; left < m; left++)        
            for(int right = left + 1; right < m; right++)            
                max = Math.max(max, dp[n - 1][left][right]);
        return max;
    }
}
