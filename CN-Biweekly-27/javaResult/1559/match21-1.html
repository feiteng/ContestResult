class Solution {
    public int cherryPickup(int[][] grid) {
    	
    	int n=grid.length;
    	int m=grid[0].length;
    	int[][][] dp=new int[n][m][m];
    	
    	for(int i=1;i<m-1;i++)
    		for(int j=1;j<m-1;j++)
    			dp[0][i][j]=-1;
    	for(int i=1;i<m;i++)
    		dp[0][i][m-1]=-1;
    	for(int i=0;i<m-1;i++)
    		dp[0][0][i]=-1;
    	
    	dp[0][0][m-1]=grid[0][0]+grid[0][m-1];
    	
    	for(int i=1;i<n;i++)
    		for(int j=0;j<m;j++)
    			for(int k=j;k<m;k
++) {
    				int mx=dp[i-1][j][k];
    				if(j+1<m) mx=Math.max(mx, dp[i-1][j+1][k]);
    				if(k+1<m) mx=Math.max(mx, dp[i-1][j][k+1]);
    				if(j+1<m && k+1<m) mx=Math.max(mx, dp[i-1][j+1][k+1]);
    				if(j>0) mx=Math.max(mx, dp[i-1][j-1][k]);
    				if(k>0) mx=Math.max(mx, dp[i-1][j][k-1]);
    				if(j>0 && k>0) mx=Math.max(mx, dp[i-1][j-1][k-1]);
    				if(j>0 && k+1<m) mx=Math.max(mx, dp[i-1][j-1][k+1]);
    				if(j+1<m && k>0) mx=Math.max(mx, dp[i-1][j+1][k-1]);
    				if(mx==-1) {
    					dp[i][j][k]=-1;
    					continue;
    				}
    				
    				if(j==k) dp[i][j][k]=mx+grid[i][j];
    				else dp[i][j][k]=mx+grid[i][j]+grid[i][k];
    				//System.out.println(i+" "+j+" "+k+" "+dp[i][j][k]);
    			}
    	
    	int ans=0;
    	for(int i=0;i<m;i++)
    		for(int j=i;j<m;j++)
    			ans=Math.max(ans, dp[n-1][i][j]);
    	
    	return ans;
    }
}
