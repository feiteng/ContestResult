class Solution(object):
    def canBeEqual(self, target, arr):
        """
        :type target: List[int]
        :type arr: List[int]
        :rtype: bool
        """
        if len(target) != len(arr):
            return False
        t = sorted(target)
        a = sorted
(arr)
        for i in range(len(target)):
            if t[i] != a[i]:
                return False
        return True
