class Solution:
    def cherryPickup(self, grid):
        a=len(grid)
        b=len(grid[0])
        c=[[[0 for i in range(b)] for j in range(b)] for k in range(a)]
        c[0][0][b-1]=grid[0][0]+grid[0][b-1]
        for i in range(1,a):
            for j in range(b-1):
                for k in range(j+1,b):
                    if j-i>0 or b-k-i>1:
                        c[i][j][k]=0
                    
                    else:
                        if j>0 and k<b-1:
                            c[i][j][k]=max(c[i-1][j][k],c[i-1][j-1][k],c[i-1][j+1][k],c[i-1][j][k-1],c[i-1][j-1][k-1],c[i-1][j+1][k-1],c[i-1][j][k+1],c[i-1][j-1][k+1],c[i-1][j+1][k+1])+grid[i][j]+grid[i][k]
                        elif j==0 and k<b-1:
                            c[i][j][k]=max(c[i-1][j][k],c[i-1][j+1][k],c[i-1][j][k-1],c[i-1][j+1][k-1],c[i-1][j][k+1],c[i-1][j+1][k+1])+grid[i][j]+grid[i][k]
                        elif j>0 and k==b-1:
                            c[i][j][k]=max(c[i-1][j][k],c[i-1][j-1][k],c[i-1][j+1][k],c[i-1][j][k-1],c[i-1][j-1][k-1],c[i-1][j+1][k-1])+grid[i][j]+grid[i][k]
                        else:
                            c[i][j][k]=max(c[i-1][j][k],c[i-1][j+1][k],c[i-1][j][k-1],c[i-1][j+1][k-1])+grid[i][j]+grid[i][k]
                    
        ans=0
        for i in range(b):
            for j in range(b):
                ans=max(ans,c[a-1][i][j])
        return ans
