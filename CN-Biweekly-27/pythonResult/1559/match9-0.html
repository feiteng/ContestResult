def getl(A,x,y,z):
    m=len(A[0])
    if(x>=0 and x<m and y>=0 and y<m):
        #print m,x,y,z,A
        return A[z][x][y]
    return 0
class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        k=len(grid)
        m=len(grid[0])
        A=[]
        for i in range(k):
            A.append([])
            for x in range(m):
                A[i].append([])
                for y in range(m):
                    A[i][x].append(0)
        for i in range(m):
            for j in range(m):
                if(i==j):
                    A[-1][i][j]=grid[-1][j]
                else:
                    A[-1][i][j]=grid[-1][i]+grid[-1][j]
        for k1 in range(k-2,-1,-1):
            for i in range(m):
                for j in range(m):
                    A[k1][i][j]=max([
                        getl(A,i+1,j+1,k1+1),
                        getl(A,i+1,j-1,k1+1),
                        getl(A,i+1,j,k1+1),
                        getl(A,i,j+1,k1+1),
                        getl(A,i,j-1,k1+1),
                        getl(A,i,j,k1+1),
                        getl(A,i-1,j+1,k1+1),
                        getl(A,i-1,j-1,k1+1),
                        getl(A,i-1,j,k1+1)
                    ])
                    if(i==j):
                        A[k1][i][j]+=grid[k1][i]
                    else:
                        A[k1][i][j]+=grid[k1][i]+grid[k1][j]
        return A[0][0][-1]
