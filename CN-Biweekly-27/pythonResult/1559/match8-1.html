class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        w = len(grid[0])
        
        dp = [[None]*w for _ in range(w)]
        dp[0][-1] = grid[0][0]+grid[0][-1]
        print(dp)
        
        for row in grid[1:]:
            ndp = [[None]*w for _ in range(w)]
            for i in range(w):
                for j in range(w):
                    if i == j:
                        for k in range(max(0,i-1),min(w,i+2)):
                            if not dp[i][j] is None:
                                if ndp[i][j] is None:
                                    ndp[i][j] = dp[k][k] + row[i]
                                else:
                                    ndp[i][j] = max(ndp[i][j], dp[k][k] + row[i])
                            
                    else:
                        for ki in range(max(0,i-1),min(w,i+2)):
                            for kj in range(max(0,j-1),min(w,j+2)):
                                if not dp[ki][kj] is None:
                                    if ndp[i][j] is None:
                                        ndp[i][j] = dp[ki][kj] + row
[i] + row[j]
                                    else:
                                        ndp[i][j] = max(ndp[i][j],dp[ki][kj] + row[i] + row[j])
                                
            dp = ndp
                                
        res = float('-inf')
        for row in dp:
            for e in row:
                if not e is None:
                    res = max(res,e)
                
        return res
