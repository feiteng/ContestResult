class Solution(object):
    def cherryPickup(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        n = len(grid)
        m = len(grid[0])
        dp = []
        for _ in range(n):
            dp.append([[-1]*m for _ in range(m)])
        if m > 1:
            dp[0][0][m-1] = grid[0][0] + grid[0][m-1]
        else:
            dp[0][0][m-1] = grid[0][0]
        for i in range(1,n):
            for j in range(m):
                for k in range(m):
                    v = grid[i][j] + grid[i][k] if j != k else grid[i][j]
                    for jj in [-1,0,1]:
                        if j + jj < 0 or j+jj >= m:
                            continue
                        for kk in [-1,0,1]:
                            if k + kk < 0 or k + kk >=m:
                                continue
                            if dp[i-1][j+jj][k+kk] == -1:
                                continue
                            dp[i][j][k] = max(dp[i-1][j+jj][k+kk]+v ,dp[i][j][k])
        ans = 0
        for j in range(m):
            for k in range(m):
                ans = max(ans,dp[n-1][j][k])
        return ans
