class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int row = grid.size(), col = grid[0].size();
    int dp[row+1][col+1][col+1];
    memset(dp, 0, sizeof(dp));

    dp[0][0][col-1] = grid[0][0]+grid[0][col-1];
    for (int k = 1; k < row; ++k) {
        for (int i = 0; i < col; ++i) {
            for (int j = i; j < col; ++j) {
                int tmp = INT32_MIN;

                for (int m = i-1; m <= i+1; m++) {
                    for (int n = j-1; n <= j+1; n++) {
                        if (m < 0 || m > k-1 || n < col-k || n >= col || m > n) continue;
                        tmp = max(tmp, dp[k-1][m][n]);
                    }
                }
                dp[k][i][j] = tmp+grid[k][i]+grid[k][j];
                if (i == j) dp[k][i][j] -= grid[k][j];

            }
        }
    }
    int ret = 0;
    for (int i = 0; i < col; ++i) {
        for (int j = 0; j < col; ++j) {
            ret = max(ret, dp[row-1][i][j]);
        }
    }
    return ret;
    }
};
