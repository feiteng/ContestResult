class Solution {
public:
    
    const static int N = 75;
    const static int INF = 1e9;
    
    int dp[N][N][N];
    
    int cherryPickup(vector<vector<int>>& grid) {

        int n = grid.size();
        int m = grid[0].size();
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = 0; k < m; k++) {
                    dp[i][j][k] = -INF;
                }
            }
        }
        
        dp[0][0][m-1] = grid[0][0] + grid[0][m-1];
        for (int i = 0; i < n-1; i++) {
            for (int j = 0; j < m; j++) {
                for (int k = j; k < m; k++) {
                    if (dp[i][j][k] < 0) continue;
                    for (int d1 = -1; d1 <= 1; d1++) {
                        for (int d2 = -1; d2 <= 1; d2++) {
                            int y1 = j+d1, y2 = k+d2;
                            if (y1 >= 0 && y1 < m && y2 >= 0 && y2 < m) {
                                if (y1 > y2) swap(y1, y2);
                                int t = grid[i+1][y1] + grid[i+1][y2];
                                if (y1 == y2) t -= grid[i+1][y1];
                                dp[i+1][y1][y2] = max(dp[i+1][y1][y2], dp[i][j][k]+t);
                            }
                        }
                    }
                }
            }
        }
        
        int ans = 0;
        for (int i = 0; i < m; i++) {
            for (int j = i; j < m; j++) {
                ans = max(ans, dp[n-1][i][j]);
            }
        }
        return ans;
        
    }
};
