class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        const int rows = grid.size(), cols = grid[0].size();
        vector<vector<vector<int>>> dp(rows, vector<vector<int>>(cols, vector<int>(cols, -1)));

        dp[0][0][cols - 1] = grid[0][0] + grid[0][cols - 1];
        for(int r = 1; r < rows; ++r){
            for(int cl = 0; cl < cols; ++cl){
                for(int cr = cl; cr < cols; ++cr){
                    int g = cl == cr ? grid[r][cl] : grid[r][cl] + grid[r][cr];
                    int pre = -1;

                    for(int i = -1; i <= 1; ++i){
                        if(cl + i < 0 || cl + i >= cols) continue;
                        for(int j = -1; j <= 1; ++j){
                            if(cr + j < 0 || cr + j >= cols || cr < cl) continue;
                            pre = max(pre, dp[r - 1][cl + i][cr +j]);
                        }
                    }
                    if(pre >= 0)
                        dp[r][cl][cr] = g + pre;

                }
            }
        }
        int res = 0;
        for(int cl = 0; cl < cols; ++cl){
            for(int cr = cl; cr < cols; ++cr){
                res = max(res, dp[rows - 1][cl][cr]);
            }
        }
        return res;
    }
};
