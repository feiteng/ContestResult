class Solution {
public:
    int dp[80][80][80];
    bool ok[80][80][80];
    int ddd[9] = {-1, 0, 1};
    int cherryPickup(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();

        dp[0][0][n-1] = grid[0][n-1] + grid[0][0];
        ok[0][0][n-1] = true;
        for (int row = 1; row < m; row++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (i == j) {
                        dp[row][i][j] = grid[row][i];
                    } else {
                        dp[row][i][j] = grid[row][i] + grid[row][j];
                    }
                    int maxV = 0;

                    for (int k = 0; k < 3; k++) {
                        for (int kk = 0; kk < 3; kk++) {
                            if (0 <= i+ddd[k] && i+ddd[k] < n && 0 <= j+ddd[kk] && j+ddd[kk] < n && ok[row-1][i+ddd[k]][j+ddd[kk]]) {
                                maxV = max(maxV, dp[row-1][i+ddd[k]][j+ddd[kk]]);
                                ok[row][i][j] = 1;
                            }
                        }
                    }
                    if (ok[row][i][j])
                        dp[row][i][j] += maxV;
                    else
                        dp[row][i][j] = 0;

                }
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                ans = max(ans, dp[m-1][i][j]);
                // cout << dp[m-1][i][j] << " ";
            }
            // cout << endl;
        }
        return ans;
    }
};
