class Solution {
public:
    int dp[75][75][75];
    bool f[75][75][75];
    int n,m;
    bool legal(int x,int y){
        return 0<=x &&  x<m && 0<=y && y<m;
    }
    int cherryPickup(vector<vector<int>>& grid) {
        n = grid.size(),m=grid[0].size();
        memset(dp,0,sizeof(dp));
        memset(f,false,sizeof(f));
        dp[0][0][m-1] = grid[0][0] + grid[0][m-1];

        f[0][0][m-1] = true;
        for(int i=0;i<n-1;i++){
            for(int j1=0;j1<m;j1++){
                for(int j2=0;j2<m;j2++){
                    if(dp[i][j1][j2]==0 && !f[i][j1][j2])continue;
                    for(int t1=-1;t1<=1;t1++){
                        for(int t2=-1;t2<=1;t2++){
                            int y1 = j1 + t1,y2 = j2+t2;
                            if(legal(y1,y2)){
                                f[i+1][y1][y2] = true;
                                int increase = y1==y2?  grid[i+1][y1]:grid[i+1][y1]+grid[i+1][y2];

                                dp[i+1][y1][y2] = max(dp[i+1][y1][y2] , dp[i][j1][j2]+increase);
                            }
                        }
                    }
                }
            }
        }
        int ans = dp[n-1][0][0];
        for(int j1=0;j1<m;j1++){
            for(int j2=0;j2<m;j2++){
                ans = max(ans,dp[n-1][j1][j2]);
            }
        }
        return ans;
    }
};
