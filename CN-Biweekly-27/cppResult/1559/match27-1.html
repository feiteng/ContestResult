class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
    	int rows = grid.size(),cols = grid[0].size();
    	int dp[rows+1][cols+1][cols+1];
    	memset(dp,-1,sizeof(dp));
    	

    	dp[0][0][cols-1] = grid[0][0] + grid[0][cols-1];
    	
    	for(int i = 1;i < rows;i++){
    		for(int j = 0;j < cols;j++){
    			for(int k = 0;k < cols;k++){
    				if(k == j) continue;
    				for(int pre_j = j-1;pre_j <= j+1;pre_j++){
    					if(pre_j < 0 || pre_j >= cols)	continue;
    					for(int pre_k = k-1;pre_k <= k+1;pre_k++){
	    					if(pre_k < 0 || pre_k >= cols || pre_k == pre_j) continue;
	    					if(dp[i-1][pre_j][pre_k] != -1){
	    						dp[i][j][k] = max(dp[i][j][k],dp[i-1][pre_j][pre_k]+grid[i][j] + grid[i][k]);
	    						//printf("%d %d %d %d\n",i,j,k,dp[i][j][k]);
							}

						}
					}
				}
			}
		}
		
		int res = 0;
		for(int j = 0;j < cols;j++){
			for(int k = 0;k < cols;k++){
				if(dp[rows-1][j][k] != -1){
					res = max(res,dp[rows-1][j][k]);
					//printf("%d %d %d\n",j,k,dp[rows-1][j][k]);
				}
			}
		}
		
		return res;
		
    }
};
