class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> dp(m, vector<int>(m, -1));
        dp[0][m - 1] = grid[0][0] + grid[0][m - 1];
        for (int i = 1; i < n; ++i) {
            vector<vector<int>> ndp(m, vector<int>(m, -1));
            for (int j = 0; j < m; ++j)
                for (int k = 0; k < m; ++k) {
                    if (dp[j][k] == -1)
                        continue;
                    for (int dj = -1; dj <= 1; ++dj)
                        for (int dk = -1; dk <= 1; ++dk) {
                            int nj = j + dj, nk = k + dk;
                            if (nj < 0 || nj >= m || nk < 0 || nk >= m)
                                continue;
                            ndp[nj][nk] = max(ndp[nj][nk], dp[j][k] + grid[i][nj] + (nj == nk ? 0 : grid[i][nk]));
                        }
                }
            swap(dp, ndp);
        }
        int ans = 0;
        for (int i = 0; i < m; ++i)
            for (int j = 0; j < m; ++j)
                ans = max(ans, dp[i][j]);
        return ans;
    }
};
