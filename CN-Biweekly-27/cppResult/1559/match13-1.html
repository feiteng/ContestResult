class Solution {
public:
    int f[75][75][75];
    int cherryPickup(vector<vector<int>>& g) {

        int n = g.size(), m = g[0].size();
        for (int i = 0; i < n; i++) {
            for (int l = 0; l < m; l++) {
                for (int r = 0; r < m; r++) {
                    f[i][l][r] = -1;
                }
            }
        }
        f[0][0][m - 1] = g[0][0] + g[0][m - 1];
        for (int i = 0; i < n - 1; i++) {
            for (int l = 0; l < m; l++) {
                for (int r = 0 ; r < m; r++) {
                    int x = f[i][l][r];
                    if (x < 0) {
                        continue;
                    }

                    for (int ll = max(l - 1, 0); ll < min(m, l + 2); ll++) {
                        for (int rr = max(r - 1, 0); rr < min(m, r + 2); rr++) {
                            int now = g[i + 1][ll] + g[i + 1][rr];
                            if (ll == rr) {
                                now -= g[i + 1][ll];
                            }
                            f[i + 1][ll][rr] = max(f[i + 1][ll][rr], x + now);

                        }
                    }
                }
            }
        }
        int ans= 0 ;
        for (int l = 0; l < m; l++) {
            for (int r = 0; r < m; r++) {
                ans = max(ans, f[n - 1][l][r]);
            }
        }
        return ans;
    }
};
