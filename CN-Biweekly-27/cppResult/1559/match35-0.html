class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int m=grid.size(),n=grid[0].size();
        int dp[m][n][n];
        for(int k=0;k<m;k++) for(int i=0;i<n;i++) for(int j=0;j<n;j++) dp[k][i][j]=-1;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                if(i==0&&j==n-1){
                    dp[0][i][j]=grid[0][0]+grid[0][n-1];
                }
            }
        }
        for(int k=1;k<m;k++){
            for(int i=0;i<n;i++){
                for(int j=0;j<n;j++){
                    if(i>0&&j>0&&dp[k-1][i-1][j-1]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i-1][j-1]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i-1][j-1]+grid[k][i]+grid[k][j]);
                    }
                    if(i>0&&dp[k-1][i-1][j]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i-1][j]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i-1][j]+grid[k][i]+grid[k][j]);
                    }
                    if(i>0&&j<n-1&&dp[k-1][i-1][j+1]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i-1][j+1]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i-1][j+1]+grid[k][i]+grid[k][j]);
                    }
                    if(j>0&&dp[k-1][i][j-1]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j-1]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j-1]+grid[k][i]+grid[k][j]);
                    }
                    if(dp[k-1][i][j]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j]+grid[k][i]+grid[k][j]);
                    }
                    if(j<n-1&&dp[k-1][i][j+1]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j+1]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j+1]+grid[k][i]+grid[k][j]);
                    }
                    if(i<n-1&&j>0&&dp[k-1][i+1][j-1]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i+1][j-1]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i+1][j-1]+grid[k][i]+grid[k][j]);
                    }
                    if(i<n-1&&dp[k-1][i+1][j]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i+1][j]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i+1][j]+grid[k][i]+grid[k][j]);
                    }
                    if(i<n-1&&j<n-1&&dp[k-1][i+1][j+1]!=-1){
                        if(i==j) dp[k][i][j]=max(dp[k][i][j],dp[k-1][i+1][j+1]+grid[k][j]);
                        else dp[k][i][j]=max(dp[k][i][j],dp[k-1][i+1][j+1]+grid[k][i]+grid[k][j]);
                    }
                    //cout<<dp[k][i][j]<<endl;
                }
            }
        }
        int res=INT_MIN;
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                res=max(res,dp[m-1][i][j]);
            }
        }
        return res;
    }
};
