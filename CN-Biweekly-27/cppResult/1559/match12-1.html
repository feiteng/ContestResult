class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int m=grid.size(), n=grid[0].size();
        vector<vector<vector<int>>> dp(m,vector<vector<int>>(n,vector<int>(n,0))); // dp[row][green][blue];
        dp[0][0][n-1]=grid[0][0]+grid[0][n-1];
        for (int r=1; r<m; r++) {
            // cout << r << ": \n";
            for (int g=0; g<n && g<=r; g++) {
                for (int b=n-1; b>=0 && n-1-b<=r; b--) {
                    for (int gg=max(0,g-1); gg<min(n,g+2); gg++)
                        for (int bb=max(0,b-1); bb<min(n,b+2); bb++) {
                            if (g!=b)
                                dp[r][g][b]=max(dp[r][g][b],dp[r-1][gg][bb]+grid[r][g]+grid[r][b]);
                            else
                                dp[r][g][b]=max(dp[r][g][b],dp[r-1][gg][bb]+grid[r][g]);
                        }
                    // cout << g << "," << b << "_" << dp[r][g][b] << " ";

                }
                // cout << endl;
            }
            // cout << endl;
        }
        int res=0;
        // int tx, ty;
        for (int i=0; i<n; i++)
            for (int j=0; j<n; j++) {
                // if (dp[m-1][i][j]>res) {
                //     res=dp[m-1][i][j];
                //     tx=i; ty=j;
                // }
                res=max(res,dp[m-1][i][j]);
            }
        // cout << tx << "," << ty;
        return res;
    }
};
