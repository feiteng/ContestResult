int dp[75][75][75];
class Solution {
public:
    int row, col;
    int cherryPickup(vector<vector<int>>& grid) {
        memset(dp, -1, sizeof(dp));
        row = grid.size();
        col = grid[0].size();
        dp[0][0][col - 1] = grid[0][0] + grid[0][col - 1];
        for (int i = 0; i < row - 1; i++)
        {
            for (int s0 = 0; s0 < col; s0++)
            {
                for (int s1 = s0; s1 < col; s1++)
                {
                    if (dp[i][s0][s1] == -1) continue;
                    for (int a0 = -1; a0 <= 1; a0++)
                    {
                        if (s0 + a0 < 0 || s0 + a0 >= col) continue;
                        for (int a1 = -1; a1 <= 1; a1++)
                        {
                            if (s1 + a1 < 0 || s1 + a1 >= col) continue;
                            int n0 = s0 + a0;
                            int n1 = s1 + a1;
                            int val = 0;
                            if (n0 == n1) val = grid[i + 1][n0];
                            else val = grid[i + 1][n0] + grid[i + 1][n1];
                            int nval = dp[i][s0][s1] + val;
                            if (dp[i + 1][n0][n1] < nval)dp[i + 1][n0][n1] = nval;
                        }
                    }
                }
            }
        }
        int res = 0;
        for (int i = 0; i < col; i++)
        {
            for (int j = i; j < col; j++)
            {
                res = max(res, dp[row - 1][i][j]);
            }
        }
        return res;
    }
};
