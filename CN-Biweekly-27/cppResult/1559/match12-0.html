class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int row=grid.size(),col=grid[0].size();
        vector<vector<vector<int>>> dp(row,vector<vector<int>>(col,vector<int>(col,-1e8)));
        dp[0][0][col-1]=grid[0][0]+grid[0][col-1];
        for(int i=1;i<row;i++){
            for(int j=0;j<col;j++){
                for(int k=0;k<col;k++){
                    dp[i][j][k]=max(dp[i][j][k],dp[i-1][j][k]+grid[i][j]+grid[i][k]);
                    if(j>0 && k>0) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j-1][k-1]+grid[i][j]+grid[i][k]);
                    if(j>0) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j-1][k]+grid[i][j]+grid[i][k]);
                    if(k>0) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j][k-1]+grid[i][j]+grid[i][k]);
                    if(j<col-1 && k<col-1) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j+1][k+1]+grid[i][j]+grid[i][k]);
                    if(j<col-1) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j+1][k]+grid[i][j]+grid[i][k]);
                    if(k<col-1) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j][k+1]+grid[i][j]+grid[i][k]);
                    if(j<col-1 && k>0) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j+1][k-1]+grid[i][j]+grid[i][k]);
                    if(j>0 && k<col-1) dp[i][j][k]=max(dp[i][j][k],dp[i-1][j-1][k+1]+grid[i][j]+grid[i][k]);
                    if(j==k) dp[i][j][k]-=grid[i][j];
                }
            }
        }
        
        int ans=0;
        for(int i=0;i<col;i++) for(int j=0;j<col;j++){
            ans=max(ans,dp[row-1][i][j]);
        }
        return ans;
    }
};
