class Solution {
public:
    int dp[80][80][80];
    int cherryPickup(vector<vector<int>>& grid) {
        memset(dp,-1,sizeof(dp));
        int R=grid.size(), C=grid[0].size();
        dp[0][0][C-1]=grid[0][0]+grid[0][C-1];
        
        for(int ri=0;ri<R-1;ri++){
            for(int a=0;a<C;a++){
                for(int b=0;b<C;b++){
                    if(dp[ri][a][b]==-1)continue;
                    for(int ad=-1;ad<=1;ad++){
                        int na=a+ad;
                        if(na<0||na>=C)continue;
                        for(int bd=-1;bd<=1;bd++){
                            int nb=b+bd;
                            if(nb<0||nb>=C)continue;
                            int tmp=dp[ri][a][b]+grid[ri+1][na]+grid[ri+1][nb];
                            if(na==nb)tmp-=grid[ri+1][nb];
                            dp[ri+1][na][nb]=max(dp[ri+1][na][nb],tmp);
                        }
                    }

                }
            }
        }
        // for(int ri=0;ri<R;ri++){
        //     for(int a=0;a<C;a++){
        //         for(int b=0;b<C;b++){
        //             printf("%d %d %d %d\n", ri,a,b,dp[ri][a][b]);
        //         }
        //     }
        // }
        int ret = 0;
        for(int a=0;a<C;a++){
            for(int b=0;b<C;b++){
                ret = max(ret, dp[R-1][a][b]);
            }
        }
        return ret;
    }
};
