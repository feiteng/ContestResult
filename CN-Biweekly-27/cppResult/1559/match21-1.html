class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int dp[72][72][72] ;
        memset(dp, -2, sizeof(dp));

        dp[0][0][m-1] = grid[0][0] + grid[0][m-1];
        int res = 0;
        for (int i=0; i+1<n; ++i) {
            for (int a = 0; a<m; ++a) {
                for (int b =0; b<m; ++ b) {
                    int v = dp[i][a][b];

                    for (int s = -1; s<=1; ++s) {
                        for(int t = -1; t<=1; ++t) {
                            int x = a+s, y = b+t;
                            if (x<0|| x>=m || y<0|| y>=m) continue;
                            int v1 = v + grid[i+1][x];
                            if (x!=y) v1 += grid[i+1][y];
                            dp[i+1][x][y] = max(dp[i+1][x][y], v1);
                            res = max(res, v1);
                        }
                    }
                }
            }
        }
        return res;
    }
};
