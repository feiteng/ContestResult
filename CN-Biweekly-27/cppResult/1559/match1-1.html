class Solution {

public:
    int cherryPickup(vector<vector<int>>& grid) {
        int r = grid.size(), c = grid[0].size(), f[70][70][70];
        memset(f, -1, sizeof(f));
        f[0][0][c - 1] = grid[0][0] + grid[0][c - 1];
        for(int i = 1; i < r; ++i) {
            for(int j = 0; j < c; ++j) for(int k = 0; k < c; ++k) {
                if(f[i - 1][j][k] == -1) continue;
                for(int dj = -1; dj <= 1; ++dj) for(int dk = -1; dk <= 1; ++dk) {
                    int nj = j + dj, nk = k + dk;
                    if(nj < 0 || nj >= c || nk < 0 || nk >= c) continue;
                    int tmp = f[i - 1][j][k] + grid[i][nj];
                    if(nj != nk) tmp += grid[i][nk];
                    f[i][nj][nk] = max(f[i][nj][nk], tmp);

                }
            }
        }
        int ans = 0;
        for(int j = 0; j < c; ++j) for(int k = 0; k < c; ++k) ans = max(ans, f[r - 1][j][k]);
        return ans;
    }
};
