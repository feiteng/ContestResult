class Solution {
public:
    int cherryPickup(vector<vector<int>>& grid) {
        int const rows = grid.size();
        int const cols = grid.front().size();
        auto dp = vector<vector<vector<int>>>(cols, vector<vector<int>>(cols, vector<int>(rows, 0xf0000000)));
        dp[cols - 1][0][0] = dp[0][cols - 1][0] = grid[0][0] + grid[0][cols - 1];
        for (int i = 1; i < rows; i++) {
            for (int j1 = 0; j1 < cols; j1++) {
                for (int j2 = 0; j2 < cols; j2++) {
                    int gain;
                    if (j1 == j2) {
                        gain = grid[i][j1];
                    } else {
                        gain = grid[i][j1] + grid[i][j2];
                    }
                    int gain1 = 0xf0000000;
                    for (int dj1 = -1; dj1 <= 1; dj1++) {
                        for (int dj2 = -1; dj2 <= 1; dj2++) {
                            int jj1 = j1 + dj1;
                            int jj2 = j2 + dj2;
                            if (jj1 < 0 || jj2 < 0 || jj1 >= cols || jj2 >= cols) continue;
                            gain1 = max(gain1, dp[jj1][jj2][i - 1]);
                        }
                    }
                    dp[j1][j2][i] = gain + gain1;
                }
            }
        }
        int ans = 0;
        for (int j1 = 0; j1 < cols; j1++) {
            for (int j2 = 0; j2 < cols; j2++) {
                ans = max(ans, dp[j1][j2][rows - 1]);
            }
        }
        // for (int i = 0; i < rows; i++) {
        //     for (int j1 = 0; j1 < cols; j1++) {
        //         for (int j2 = 0; j2 < cols; j2++) {
        //             cout << dp[j1][j2][i] << " ";
        //         }
        //         cout << endl;
        //     }
        //     cout << endl;
        // }
        return ans;
    }
};
