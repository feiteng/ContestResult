class Solution {
public:
    
    int dp[150][71][71] ;
    
    int cherryPickup(vector<vector<int>>& grid) {
        int m = grid.size() , n = grid[0].size() ;
        memset(dp , -0x7f7f7f , sizeof(dp)) ;
        dp[0][0][n - 1] = grid[0][0] + grid[0][n - 1] ;
        for (int l = 1 ; l < m ; ++l) {
            for (int i = 0 ; i < n ; ++i) {
                for (int j = 0 ; j < n ; ++j) {
                    int t = 0 , s = -0x7f7f7f ;
                    if (i == j) {
                        t = grid[l][i];
                    } else t = grid[l][i] + grid[l][j] ;
                        for (int k1 = -1 ; k1 <= 1 ; ++k1) {
                            for (int k2 = -1 ; k2 <= 1 ; ++k2) {
                                if (i + k1 >= 0 && i + k1 < n && j + k2 >= 0 && j + k2 < n) {
                                    s = max(s , dp[l - 1][i + k1][j + k2]) ;
                                }
                            }
                        }
                        dp[l][i][j] = s + t ;
                        
                    
                }
            }
        }
        int res = 0 ;
        for (int i = 0 ; i < n ; ++i) {
            for (int j = 0 ; j < n ; ++j) {
                res = max(res , dp[m - 1][i][j]) ;
            }
        }
        return res ;
    }
};
