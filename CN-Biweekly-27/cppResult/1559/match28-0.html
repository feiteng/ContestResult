class Solution {
public:
    int dp[100][100][100];
    int cherryPickup(vector<vector<int>>& a) {
        int n=a.size(),m=a[0].size();
        memset(dp,128,sizeof dp);
        dp[1][1][m]=a[0][0]+a[0][m-1];
        for(int i=2;i<=n;i++){
            for(int j=1;j<=m;j++){
                for(int k=1;k<=m;k++){
                    if(j==k){
                        dp[i][j][k]=max(dp[i-1][j-1][k],max(dp[i-1][j-1][k+1],max(dp[i-1][j][k+1],dp[i-1][j][k])))+a[i-1][j-1];
                    }else {
                        for(int l=-1;l<=1;l++){
                            for(int r=-1;r<=1;r++){
                                if(j+l<k+r)
                                dp[i][j][k]=max(dp[i-1][j+l][k+r]+a[i-1][j-1]+a[i-1][k-1],dp[i][j][k]);
                            }
                        }
                    }
                }
            }
        }
        int ans=0;
        for(int i=1;i<=m;i++){
            for(int j=1;j<=m;j++){
                ans=max(dp[n][i][j],ans);
            }
        }
        return ans;
    }
};
