class Solution {
public:
    int dp[80][80][80];
    int cherryPickup(vector<vector<int>>& g) {
        int m=g.size();
        int n=g[0].size();

        memset(dp,-1,sizeof dp);
        dp[0][0][n-1]=g[0][0]+g[0][n-1];
        vector<int>dir={-1,0,1};
        for(int i=1;i<m;i++)
            for(int j=0;j<n;j++)
                for(int k=0;k<n;k++)
                {
                    for(int a=0;a<3;a++)
                        for(int b=0;b<3;b++)
                        {
                            int nj=j+dir[a];
                            int nk=k+dir[b];
                            if(nj>=0&&nj<n&&nk>=0&&nk<n&&dp[i-1][nj][nk]!=-1)
                                dp[i][j][k]=max(dp[i][j][k],dp[i-1][nj][nk]);            
                        }
                    if(dp[i][j][k]==-1)continue;
                    dp[i][j][k]+=(g[i][j]+g[i][k]);
                    if(j==k)
                        dp[i][j][k]-=g[i][j];
                }
        int max_=0;
        for(int i=0;i<n;i++)
            for(int j=0;j<n;j++)
                max_=max(max_,dp[m-1][i][j]);
        return max_;
    }
};
