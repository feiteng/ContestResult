class Solution
{
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>> &prerequisites, vector<vector<int>> &queries)
    {
        int m = queries.size();
        vector<bool> ret(m,false);
        // vector<int> in(n);
        vector<vector<int>> G(n,vector<int>(n,0x3f3f3f3f));
        for(auto v:prerequisites){
            // G[v[0]].push_back(v[1]);
            G[v[0]][v[1]] = 1;
            // in[v[1]]++;
        }
        // stack<int> st;
        // for(int i=0;i<n;++i){
        //     if(in[i]==0){
        //         st.push(i);
        //     }
        // }
        for(int k=0;k<n;++k){
            for(int i=0;i<n;++i){
                for(int j=0;j<n;++j){
                    G[i][j] = min(G[i][j],G[i][k]+G[k][j]);
                }
            }
        }
        for(int i=0;i<m;++i){
            if(G[queries[i][0]][queries[i][1]]!=0x3f3f3f3f){
                ret[i] = true;
            }
        }
        return ret;
    }
};
