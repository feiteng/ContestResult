class Solution {
public:
    int weight[150][150];
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        for (int i=0;i<n;i++){
            for (int j=0;j<n;j++) weight[i][j] = 0x3f3f3f3f;
        }
        for (int i=0;i<n;i++) weight[i][i]=0;
        for (const auto &it:prerequisites){
            //cout<<it[0]<<" "<<it[1]<<endl;

            weight[it[0]][it[1]]=1;
        }
        for (int k=0;k<n;k++){
            for (int i=0;i<n;i++){
                for (int j=0;j<n;j++){
                    //cout<<"i="<<i<<" "<<j<<" "<<k<<endl;
                    weight[i][j]=min(weight[i][j],weight[i][k]+weight[k][j]);
                }
            }
        }
        std::vector<bool> res;
        for (const auto &it:queries){
            //cout<<weight[it[0]][it[1]]<<endl;
            res.push_back(weight[it[0]][it[1]]>0&&weight[it[0]][it[1]]<0x3f3f3f3f);
        }
        return res;
    }
};
