class Solution {
    int mp[105][105];
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& g, vector<vector<int>>& q) {
        vector<bool> ret;
        for(auto i:g) mp[i[0]][i[1]]=1;
        for(int k=0;k<n;++k)
            for(int i=0;i<n;++i)
                for(int j=0;j<n;++j)
                    mp[i][j]=mp[i][j]|(mp[i][k]&mp[k][j]);
        for(auto i:q) {
            ret.push_back(mp[i[0]][i[1]]);
        }
        return ret;
    }
};
