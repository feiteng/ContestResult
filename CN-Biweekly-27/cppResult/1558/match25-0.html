class Solution {
public:
    int f[105][105];
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& pre, vector<vector<int>>& q) {
        vector <bool> ret;
        int m = pre.size();
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                f[i][j] = (i == j);
            }
        }
        for (int i = 0; i < m; i++) {
            int a = pre[i][0], b = pre[i][1];
            f[a][b] = 1;
        }
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    f[i][j] |= (f[i][k] && f[k][j]);
                }
            }
        }
        for (int i = 0; i < q.size(); i++) {
            int a = q[i][0], b = q[i][1];
            ret.push_back(f[a][b]);
        }
        return ret;
    }
};
