class Solution {
public:
    
    int f[101][101] , g[101][101] ;
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        
        memset(f , 0x3f3f3f3f , sizeof(f)) ;
        
        for (int i = 0 ; i < prerequisites.size() ; ++i) {
            f[prerequisites[i][0]][prerequisites[i][1]] = 1  ;
        }
        

        

        for (int i = 0 ; i < n ; ++i) {
            f[i][i] = 0 ;
        }
        
        for (int k = 0 ; k < n ; ++k) {
            for (int i = 0 ; i < n ; ++i) {
                for (int j = 0 ; j < n ; ++j) {
                    f[i][j] = min(f[i][k] + f[k][j] , f[i][j]) ;

                }
            }
        }
        
//         for (int i = 0 ; i < n ; ++i) {
//             for (int j = 0 ; j < n ; ++j) {
//                 cout << f[i][]
//             }
//         }
        
        int m = queries.size() ;
        
        vector<bool> res(m , false) ;
        
        for (int i = 0 ; i < m ; ++i) {
            if (f[queries[i][0]][queries[i][1]] < 0x3f3f3f3f) {
                res[i] = true ;
            }
        }
        
        return res ;
        
    }
};
