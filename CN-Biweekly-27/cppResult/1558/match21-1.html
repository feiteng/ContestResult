class Solution {
public:
    int ans[105][105];
    vector<int> eg[105];
    bool mark[105];
    int N;
    void Dfs(int rt)
    {
        if (mark[rt]) return;
        mark[rt] = true;
        for (int i = 0; i < eg[rt].size(); i++)
        {
            int v = eg[rt][i];
            Dfs(v);
            ans[rt][v] = 1;
            for (int j = 0; j < N; j++)
            {
                ans[rt][j] |= ans[v][j];
            }
        }
    }
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& pq, vector<vector<int>>& q) {
        N = n;
        vector<bool> res;
        memset(ans, 0, sizeof(ans));
        memset(mark, 0, sizeof(mark));
        for (int i = 0; i < 105; i++) eg[i].clear();

        for (int i = 0; i < pq.size(); i++)
        {
            eg[pq[i][0]].push_back(pq[i][1]);
        }
        for (int i = 0; i < n; i++)
        {
            Dfs(i);
        }
        for (int i = 0; i < q.size(); i++)
        {
            int v0 = q[i][0];
            int v1 = q[i][1];
            res.push_back (ans[v0][v1]);
        }
        return res;
    }
};
