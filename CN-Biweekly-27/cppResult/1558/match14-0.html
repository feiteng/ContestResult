class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<bool>> nei(n, vector<bool>(n, false));
        for (int i = 0; i < prerequisites.size(); ++i) {
            nei[prerequisites[i][0]][prerequisites[i][1]] = true;
        }
        for (int i = 0; i < n; ++i){
            for (int j = 0; j < n; j++) {
                if (nei[j][i])
                for (int k = 0; k < n; ++k) {
                    if (!nei[j][k]) {
                        nei[j][k] = nei[i][k];
                    }
                }
            }
        }
        int m = queries.size();
        vector<bool> res(m, false);
        for (int i = 0; i < m; ++i) {
            if (nei[queries[i][0]][queries[i][1]]) {
                res[i] = true;
            }
        }
        return res;
    }
};
