class Solution {
    bool trans[105][105];
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<bool> ans;

        for (auto& v: prerequisites){
            trans[v[0]][v[1]] = true;
        }
        for (int k = 0; k < n; ++k){
            for (int i = 0; i < n; ++i){
                for (int j = 0; j < n; ++j)
                    trans[i][j] = trans[i][j] || (trans[i][k] && trans[k][j]);
            }
        }
        for (auto& v: queries){
            ans.push_back(trans[v[0]][v[1]]);
        }
        return ans;
    }
};
