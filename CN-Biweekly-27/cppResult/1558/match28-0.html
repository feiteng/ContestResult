class Solution {
public:
    void search_pre(vector<vector<int>>& graph, vector<int>& flag, vector<unordered_map<int, int>>& search_result, int i){
        if(flag[i] > 0){
            return;
        }
        for(int j = 0; j < graph[i].size(); j ++){
            if(flag[graph[i][j]] == 0)
                search_pre(graph, flag, search_result, graph[i][j]);
            for(auto key: search_result[graph[i][j]]){
                search_result[i][key.first] = 1;
            }
            search_result[i][graph[i][j]] = 1;
        }
        flag[i] = 1;
    }
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<int>> graph(n);
        for(int i = 0; i < prerequisites.size(); i ++){
            graph[prerequisites[i][0]].push_back(prerequisites[i][1]);
        }
        vector<unordered_map<int, int>> pre(n);
        vector<int> flag(n, 0);
        for(int i = 0; i < n; i ++){
            search_pre(graph, flag, pre, i);
        }
        vector<bool> result(queries.size(), false);
        for(int i = 0; i < queries.size(); i ++){
            if(pre[queries[i][0]][queries[i][1]] > 0){
                result[i] = true;
            }
        }
        return result;
    }
};
