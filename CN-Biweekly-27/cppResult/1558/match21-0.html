const int MX = 108;
class Solution {
public:
    set<int> my_son[MX];
    int vis[MX];
    vector<int> graph[MX];
    void dfs(int u){
        if(vis[u]) return ;
        vis[u] = 1;
        my_son[u].insert(u);
        for(int i = 0;  i < graph[u].size(); i++){
            int next = graph[u][i];
            dfs(next);
            my_son[u].insert(my_son[next].begin(), my_son[next].end());
        }
    }
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        for(int i = 0; i < n; i++) my_son[i].clear(),graph[i].clear();
        memset(vis, 0, sizeof(vis));
        for(int i = 0; i < prerequisites.size(); i++){
            graph[prerequisites[i][0]].push_back(prerequisites[i][1]);
        }
        for(int i = 0; i < n; i++){
                dfs(i);
        }
        vector<bool> ans;
        for(int i = 0; i < queries.size(); i++){
            if(my_son[queries[i][0]].find(queries[i][1]) != my_son[queries[i][0]].end()) ans.push_back(true);
            else ans.push_back(false);
        }
        return ans;
    }
};
