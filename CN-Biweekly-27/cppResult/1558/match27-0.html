class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) 
    {
        int a[n][n];
        int inf=1e8+7;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
                a[i][j]=inf;
            a[i][i]=0;
        }
        for(auto &e: prerequisites)
            a[e[0]][e[1]]=1;
           
        for(int k=0;k<n;k++)
            for(int i=0;i<n;i++)
                for(int j=0;j<n;j++)
                    a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
        
        vector<bool> res;
        for(auto &e: queries)
        {
            if(a[e[0]][e[1]]<inf)
                res.push_back(true);
            else
                res.push_back(false);
        }
        return res;
        
    }
};
