class Solution {
public:
        map<int,vector<int>> fa;
    int edge[110][110];
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        memset(edge,0x3f3f3f3f,sizeof(edge));

        for(int i = 0; i < prerequisites.size(); i++){
            edge[prerequisites[i][0]][prerequisites[i][1]] = 1;
        }
        for(int k = 0; k < 110; k++){
            for(int i = 0; i < 110; i++){

                for(int j = 0; j < 110; j++){
                    if(edge[i][k] + edge[k][j] < edge[i][j]){
                        edge[i][j] = edge[i][k] + edge[k][j];
                    }
                }
            }
        }
        vector<bool> ans(queries.size());
        for(int i = 0; i < queries.size(); i++){
            ans[i] = edge[queries[i][0]][queries[i][1]] != 0x3f3f3f3f;
        }
        return ans;
    }
    bool dfs(int now, int aim){
        if(now == aim) return true;
        for(int i = 0; i < fa[now].size(); i++){
            if(dfs(fa[now][i],aim)) return true;
        }
        return false;
    }
};
