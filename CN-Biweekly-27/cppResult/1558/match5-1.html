class Solution {
public:
    
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        bool zg[110][110] = {0};
        bool pre[110][110] = {0};

        for (auto & z: prerequisites) {
            zg[z[0]] [z[1]]= true;
        }
        for (int k=0; k<=n; ++k) {
            for (int i=0; i<=n; ++i) {
                for (int j=0; j<=n; ++j) {
                    if (zg[i][k] && zg[k][j]) zg[i][j] = true;
                }
            }
        }
        vector<bool> zr;
        for (auto & z: queries) {
            if (zg[z[0]][z[1]]) zr.push_back(true);
            else zr.push_back(false);
        }
        return zr;
    }
};
