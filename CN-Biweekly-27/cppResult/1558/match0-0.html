class Solution {
public:
    vector<vector<bool>>ret;
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        ret=vector<vector<bool>>(n,vector<bool>(n));
        for(auto& it:prerequisites)
            ret[it[0]][it[1]]=true;
        for (int k = 0; k <n; k++)
		    for (int i = 0; i <n; i++)
			    for (int j = 0; j <n; j++)
				    if (ret[i][k]&&ret[k][j])
					    ret[i][j]=true;
        vector<bool>ans(queries.size());
        for(int i=0;i<queries.size();i++)
        {
            auto& it=queries[i];
            if(ret[it[0]][it[1]])
                ans[i]=true;
        }
        return ans;
    }
};
