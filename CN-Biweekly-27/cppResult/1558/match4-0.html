int f[103][103];

class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        memset(f,0,sizeof(f));
        for (int i = 0; i < n; i++) f[i][i] = 1;
        for (auto e:prerequisites) {
            f[e[0]][e[1]] = 1;
        }
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    if (f[i][k] && f[k][j]) f[i][j] = 1;
                }
            }
        }
        vector<bool> rst;
        for (auto q:queries) {
            if (f[q[0]][q[1]]) rst.push_back(true);
            else rst.push_back(false);
            
        }
        return rst;

    }
};
