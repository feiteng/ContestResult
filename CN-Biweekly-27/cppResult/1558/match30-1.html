class Solution {
public:    
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<int>> g(n, vector<int>(n));

        for(auto pr: prerequisites) g[pr[0]][pr[1]] = 1;
        for(int k = 0; k < n; ++k)
            for(int i = 0; i < n; ++i)
                for(int j = 0; j < n; ++j)
                    g[i][j] |= g[i][k] & g[k][j];
        vector<bool> ans;
        for(auto pr: queries) ans.push_back(g[pr[0]][pr[1]]);
        return ans;
    }
};
