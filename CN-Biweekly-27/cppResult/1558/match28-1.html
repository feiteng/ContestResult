class Solution {
public:

    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries) 
    {
        vector<vector<int>> nJump(n);
        for (int i = 0; i < prerequisites.size(); i++)
        {
            nJump[prerequisites[i][1]].push_back(prerequisites[i][0]);
        }
        
        vector<set<int>> nJumpSet(n);
        for (int i = 0; i < n; i++)
        {
            vector<int> beforeArr = nJump[i];
            for (int j = 0; j < beforeArr.size(); j++)
            {
                if (nJumpSet[i].find(beforeArr[j]) == nJumpSet[i].end())
                {
                    nJumpSet[i].insert(beforeArr[j]);
                    beforeArr.insert(beforeArr.end(), nJump[beforeArr[j]].begin(), nJump[beforeArr[j]].end());
                }
            }

        }
        
        vector<bool> ans(queries.size(), false);
        for (int i = 0; i < queries.size(); i++)
        {
            if (nJumpSet[queries[i][1]].find(queries[i][0]) != nJumpSet[queries[i][1]].end())
            {
                ans[i] = true;
            }
        }
        return ans;

    }
};
