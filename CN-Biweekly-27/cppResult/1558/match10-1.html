class Solution {
public:
    bool vis[105][105];
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& g, vector<vector<int>>& q) {
        for (int i=0; i<g.size(); i++) {
            vis[g[i][0]][g[i][1]]=true;
        }
        for (int i=0; i<n; i++) vis[i][i]=true;

        for (int k=0; k<n; k++) {
            for (int i=0; i<n; i++) {
                for (int j=0; j<n; j++) {
                    vis[i][j]|=(vis[i][k]&&vis[k][j]);
                }
            }
        }
        vector<bool> ans;
        for (int i=0; i<q.size(); i++) {
            ans.push_back(vis[q[i][0]][q[i][1]]);
        }
        return ans;
    }
};
