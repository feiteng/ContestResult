class Solution {
public:
    vector<bool> checkIfPrerequisite(int n, vector<vector<int>>& prerequisites, vector<vector<int>>& queries)
    {
        map<int,set<int>>par;//
        map<int,set<int>>next;
        vector<int>inCount(n,0);
        for(auto& pre:prerequisites){
            inCount[pre[1]]++;
            next[pre[0]].insert(pre[1]);
            par[pre[1]].insert(pre[0]);
        }
        queue<int>Q;

        for(int i=0;i<n;i++){
            if(inCount[i]==0){
                Q.push(i);
            }
        }
        while(!Q.empty()){
            int t=Q.front();
            Q.pop();
            if(next.count(t)==0){
                continue;
            }
            for(auto num:next[t]){
                inCount[num]--;
                if(par.count(t)){
                    for(auto num2:par[t]){
                    par[num].insert(num2);
                    }
                }
                if(inCount[num]==0){
                    Q.push(num);

                }
            }
        }
        vector<bool>res;
        for(auto&query:queries){
            if(par[query[1]].count(query[0])){
                res.push_back(true);
            }
            else{
                res.push_back(false);
            }
        }
        return res;
    }
};
