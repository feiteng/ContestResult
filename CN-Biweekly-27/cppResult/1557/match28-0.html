bool f[1<<20];

class Solution {
public:
    bool hasAllCodes(string s, int k) {
        int limit = 1 << k;
        int mask = limit - 1;
        fill(f, f+limit, false);
        int ls = s.size();
        int now = 0;
        for (int i = 0; i < ls; i++) {
            now <<= 1;
            now |= (s[i] - '0');
            now &= mask;
            if (i >= k-1) {
                f[now] = true;
            }
        }
        
        for (int i = 0; i < limit; i++)
            if (!f[i]) return false;
        return true;
        
        
    }
};
