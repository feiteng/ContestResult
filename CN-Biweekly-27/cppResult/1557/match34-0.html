class Solution {
public:
    int toNum(string tmp){
        int res=0;
        int now=1;
        for(int i=tmp.size()-1;i>=0;i--){
            if(tmp[i]=='1'){
                res+=now;
            }
            now<<=1;
        }
        return res;
    }
    bool hasAllCodes(string s, int k) {
        if(s.size()<k){
            return false;
        }
        vector<bool>num((1<<k),0);
        for(int i=0;i<=s.size()-k;i++){
            string tmp=s.substr(i,k);
            num[toNum(tmp)]=true;
            //cout<<toNum(tmp);
        }
        for(int i=0;i<(1<<k);i++){
            if(!num[i]){
                return false;
            }
        }
        return true;
    }
};
