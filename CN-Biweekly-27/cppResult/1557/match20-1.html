class Solution {
public:
    bool hasAllCodes(string s, int k) {
        int n = s.size();
        int num = 0;
        if (n < k) {
            return false;
        }
        bool seen[1 << k];
        memset(seen, 0, sizeof(seen));

        for (int i = 0; i < k; i++) {
            num |= (s[i] - '0') << (k - i - 1);
        }
        seen[num] = true;
        int cnt = 1;
        for (int i = k; i < n; i++) {
            num -= (s[i - k] - '0') << (k - 1);
            num <<= 1;
            num |= (s[i] - '0');
            if (!seen[num]) {
                cnt++;
                seen[num] = true;
            }
        }
        return cnt == (1 << k);
    }
};
