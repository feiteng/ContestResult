class Solution {
public:
    bool hasAllCodes(string s, int k) {
        unordered_map< string, int > Index;
        string tmp;
        tmp.clear();
        int n = s.length();
        for( int i = 0; i < k; i ++ ) tmp += s[ i ];
        Index[ tmp ] ++;
        for( int i = k; i < n; i ++ )
        {
            tmp = tmp.substr( 1 ) + s[ i ];
            Index[ tmp ] ++;
        }
        int tot = 1;
        for( int i = 1; i <= k; i ++ ) tot *= 2;
        return Index.size() == tot;
    }
};
