class Solution {
public:
    bool hasAllCodes(string s, int k) {
        unordered_set<int> m;
        
        int start = 0;
        for (int i = 0; i < k; ++i) {
            start = (start << 1) | (s[i] == '0' ? 0 : 1);
        }
        
        m.insert(start);
        
        int l = s.length();
        for (int i = k; i < l; ++i) {
            start = ((start << 1) | (s[i] == '0' ? 0 : 1)) & ((1 << k) - 1);
            
            if (m.count(start) == 0) {
                m.insert(start);
            }
        }
        
        return m.size() == (1 << k);
    }
};
