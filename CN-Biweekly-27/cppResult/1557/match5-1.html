class Solution {
public:
    bool hasAllCodes(string s, int k) {
        vector<bool> st(65537, false);
        if (s.length() < (1 << k)) {
            return false;
        }

        int cur = 0;
        for (int i = 0; i < k; ++i) {
            cur = (cur << 1) + static_cast<int>(s[i] == '1');
        }
        st[cur] = true;
        // cout << cur << endl;
        // cout << " " << endl;
        for (int i = k; i < s.length(); ++i) {
            if (s[i - k] == '1') {
                cur -= (1 << (k - 1));
            }
            // cout << "minus: " << cur << endl;

            cur = (cur << 1) + static_cast<int>(s[i] == '1');
            st[cur] = true;
            // cout << cur << endl;
        }
        for (int i = 0; i < (1 << k); ++i) {
            if (!st[i]) return false;
        }
        return true;
    }
};
