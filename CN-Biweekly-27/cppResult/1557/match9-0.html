class Solution {
public:
    bool hasAllCodes(string s, int k) 
    {
        if(s.size()<k)
            return false;
        int ls=(1<<k);
        vector<int> vt(ls,0);
        int res=0;
        for(int i=0;i<k;i++)
            res=2*res+s[i]-'0';
        vt[res]++;
        for(int i=k;i<s.size();i++)
        {
            if(s[i-k]=='1')
                res-=ls/2;
            res=res*2+s[i]-'0';
            vt[res]++;
        }
        for(int i=0;i<ls;i++)
        {
            if(!vt[i])
                return false;
        }
        return true;
    }
};
