class Solution {
public:
    bool sig[1<<20];
    bool hasAllCodes(string s, int k) {
        if(s.size()<k)return false;
        memset(sig,0,sizeof sig);
        int msk=(1<<k)-1;
        int init=0;
        for(int i=0;i<k;i++){
            init|=((int)(s[i]-'0'))<<i;
        }
        sig[init]=true;//cout<<init<<endl;
        for(int i=0;i<s.size()-k;i++){
            init=(init>>1)|(((int)(s[i+k]-'0'))<<(k-1));//cout<<init<<endl;
            sig[init]=true;
        }
        for(int i=0;i<1<<k;i++){
            if(sig[i]==false)return false;
        }
        return true;
    }
};
