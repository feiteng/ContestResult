class Solution {
private:
    int to_int(const string& s) {

        int ret = 0;
        int pow = 1;
        for (int i = s.size() - 1; i >= 0; --i) {
            ret += (s[i] - '0') * pow;
            pow *= 2;
        }
        return ret;
    }
public:
    bool hasAllCodes(const string& s, int k) {
        int pow_num = pow(2, k);
        if (s.size() < pow_num) {
            return false;
        }
        vector<int> recd(pow_num, 0);
        for (int i = 0; i + k <= s.size(); ++i) {

            int tmp = to_int(s.substr(i, k));
            recd[to_int(s.substr(i, k))] = 1;
        }
        for (int i = 0; i < recd.size(); ++i) {
            if (recd[i] == 0) {
                return false;
            }
        }
        return true;
    }
};
