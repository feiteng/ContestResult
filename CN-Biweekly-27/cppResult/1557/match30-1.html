class Solution {
public:
    bool hasAllCodes(string s, int k) {
        if (k>s.size()) return false;
        unordered_set<string> st;
        for(int i=0; i+k<=s.size(); ++i){
            string cur = s.substr(i,k);
            st.insert(cur);
        }
        for(int t=0; t<(1<<k); ++t){
            string cur = helper(t);
            while(cur.size()<k)
                cur = "0"+cur;
            if ( st.count(cur)==0 ){
                return false;
            }
        }
        return true;
    }
    
    string helper( int d ){
        string res;
        while(d>0){
            res += to_string(d%2);
            d/=2;
        }
        reverse(res.begin(), res.end());
        return res;
    }
};
