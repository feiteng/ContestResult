class Solution {
public:
    bool hasAllCodes(string s, int k) {
        int used[1 << 20];
        memset(used,0,size(used));
        int cur = 0;
        int mask = (1 << k) - 1;
        for(int i = 0; i < k && i < s.size(); i++){
            cur <<= 1;
            cur |= (s[i] - '0');
        }
        used[cur] = 1;
        // cout<<cur<<endl;
        for(int i = k; i < s.size(); i++){
            // cout<<i<<endl;
            // cout<<cur<<endl;
            cur <<= 1;
            // cout<<cur<<endl;
            cur |= (s[i] - '0');
            cur &= mask;
            used[cur] = 1;
        }
        for(int i = 0; i < (1<<k); i++){
            // cout<<i<<endl;
            if(used[i]==0)return false;
        }
        return true;
        
    }
};
