class Solution {
public:
    bool hasAllCodes(string s, int k) {
        int MAXN=(1<<20)+10;
        vector<bool> cnt(MAXN, false);
        int n=s.size();
        for(int i=0;i+k<=n;i++){
            int num=0;
            for(int j=0;j<k;j++){
                num=num*2+s[i+j]-'0';
            }
            // cout<<num<<endl;
            cnt[num]=true;
        }
        bool ans=true;
        int imax=1<<k;
        for(int i=0;i<imax;i++){
            if(!cnt[i]){
                ans=false;
            }
        }
        return ans;
    }
};
