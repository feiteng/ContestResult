class Solution {
public:
    unordered_set<string> _set;
    bool hasAllCodes(string s, int k) {
        _set.clear();
        int n = s.size();
        int c = 0;
        for(int i = k - 1; i < n; ++i){
            string t = s.substr(i - k + 1 , k);
            if(_set.find(t) == _set.end()){
                ++ c;
                _set.insert(t);
            }
        }
        return (c == (1 << k));
    }
};
