class Solution {
    map<string,int> mp;
public:
    bool hasAllCodes(string s, int k) {
        if(s.length()-k+1<(1<<k)) return 0;
        for(int i=0;i+k-1<s.length();++i) {
            mp[s.substr(i,k)]=1;
        }
        if(mp.size()==(1<<k)) return 1;
        else return 0;
    }
};
