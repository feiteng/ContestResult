class Solution {
public:
  bool hasAllCodes(string s, int k) {
    int M = 1 << k;
    vector<bool> A(M);
    int cur = 0;
    for (int i = 0; i < s.size(); i++) {
      if (i >= k) cur ^= (s[i-k]-'0')<<(k-1);
      cur <<= 1;
      cur |= s[i]-'0';
      if (i >= k - 1) A[cur] = true;
    }
    return count(A.begin(), A.end(), false) == 0;
  }
};
