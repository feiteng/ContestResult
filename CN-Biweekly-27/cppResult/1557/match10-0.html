class Solution {
public:
    bool hasAllCodes(string s, int k) {
        int has[1 << k];
        memset(has, 0, sizeof(has));
        int w = 1 << k;
        for(int i = 0, cur = 0; i < s.size(); ++i) {
            cur = (cur << 1) + s[i] - '0';
            if(i-k >= 0)
                cur -= (s[i-k] - '0')*w;
            if(i >= k-1)
                has[cur] = 1;
        }
        for(int i = 0; i < (1 << k); ++i)
            if(!has[i])
                return false;
        return true; 
    }
};
