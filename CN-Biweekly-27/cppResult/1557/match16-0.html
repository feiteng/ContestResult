class Solution {
public:
    int B[1 << 20];
    bool hasAllCodes(string s, int k) {
        if (s.size() < k) return false;
        int mask = ((1 << k) - 1) ^ (1 << (k - 1));
        memset(B, 0, sizeof(B));
        int t = 0;
        for (int i = 0; i < k; ++i) {
            t <<= 1;
            t |= s[i] - '0';
        }
        B[t] = 1;
        for (int i = k; i < s.size(); ++i) {
            t &= mask;
            t <<= 1;
            t |= s[i] - '0';
            B[t] = 1;
        }
        for (int i = 0; i < (1 << k); ++i) {
            if (B[i] == 0) return false;
        }
        return true;
    }
};
