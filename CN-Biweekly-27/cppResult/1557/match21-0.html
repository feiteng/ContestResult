class Solution {
public:
    bool hasAllCodes(string s, int k) {
		map<string,bool>mp;
		if( s.size()-k+1<(1<<k) ) return false;
		for( int i=0;i<=s.size();i++ )
		{
			if( i+k>s.size() ) break;
			string tmp=s.substr(i,k);
			mp[tmp]=true;
		}
		// for( auto v:mp )
		// {
		// 	cout<<v.first<<endl;
		// }
		if( mp.size()<(1<<k) )return false;
		return true;
    }
};
