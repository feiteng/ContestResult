class Solution {
public:
    bool hasAllCodes(string s, int k) {
        const int n = s.length();
        if(n<k) return false;
        unordered_set<int> occured;
        int num=0;
        for(int i=0;i<k;++i){
            num <<= 1;
            num += s[i]-'0';
        }
        occured.insert(num);
        for(int i=k;i<n;++i){
            num <<=1;
            num += s[i]-'0';
            num &= ((1<<k)-1);
            occured.insert(num);
        }
        return occured.size() == (1<<k);
    }
};
