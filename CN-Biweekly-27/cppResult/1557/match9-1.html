class Solution {
public:
    bool hasAllCodes(string s, int k) {
        if (s.size() < k)
            return false;
        std::vector<bool> vis(1 << k);
        int mask = (1 << k) - 1;

        int val = 0;
        for (int i = 0; i < k; ++i)
            val = (val << 1) + s[i] - '0';
        vis[val] = true;
        for (int i = k; i < s.size(); ++i)
        {

            val = ((val << 1) & mask) + s[i] - '0';
            vis[val] = true;
        }
        for (int i = 0; i < vis.size(); ++i)
        {
            if (!vis[i])
                return false;
        }
        return true;
    }
};
