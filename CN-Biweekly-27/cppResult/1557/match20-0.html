class Solution {
public:
    int vis[1<<21];
    bool hasAllCodes(string s, int k) {
        int o=0;
        if(k>s.size())return false;
        for(int i=0;i<k;i++)
        {
            o+=(s[i]-'0')<<(k-i-1);
        }
        vis[o]=1;
        int cnt=1;
        for(int i=k;i<s.size();i++)
        {
            o-=(s[i-k]-'0')<<(k-1);
            o*=2;
            o+=s[i]-'0';
            if(!vis[o])
            {
                vis[o]=1;
                cnt++;
            }
        }
        if(cnt==pow(2,k))return true;
        else return false;
    }
};
