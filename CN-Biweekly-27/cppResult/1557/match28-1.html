class Solution {
public:
    bool hasAllCodes(string s, int k) {
        int tot = 1<<k;
        vector<bool> zf(tot);
        int res = 0;
        int tm = tot - 1;
        int t = 0;
        for (int i=0;i+1<k; ++i) {
            t = t*2 + s[i] - '0';
        }
        for (int i=k-1; i<s.size(); ++i) {
            t = t*2 + s[i] -'0';
            t&= tm;
            if (zf[t]) continue;
            ++res;
            zf[t] = true;
        }
        return (res == tot);
    }
};
