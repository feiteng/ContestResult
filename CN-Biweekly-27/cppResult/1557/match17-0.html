class Solution {
public:
    int a[1<<22];
    bool hasAllCodes(string s, int k) {
        int n = s.size();
        if(n < k) return false;
        int x = 0;
        int t = 1 << (k - 1);
        for(int i = 0; i < k; ++i) {
            x *= 2;
            x += s[i] - '0';
        }
        // cout << x << endl;
        a[x]++;
        for(int i = k; i < n; ++i) {
            x -= (s[i - k] - '0') * t;
            x *= 2;
            x += s[i] - '0';
            a[x]++;
            // cout << x << endl;
        }
        for(int i = 0; i < (1 << k); ++i) {
            if(a[i] == 0) {
                return false;
            }
        }
        return true;
    }
};
