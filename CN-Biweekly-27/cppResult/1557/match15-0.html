class Solution {
public:
    bool hasAllCodes(string s, int k) {
        
        set<int> st;
        
        int n = (int)s.size();
        
        for (int i = 0; i+k-1 < n; i++) {
            int cur = 0;
            for (int j = 0; j < k; j++) {
                cur = cur*2 + s[i+j]-'0';
            }
            st.insert(cur);
        }
        return st.size() == (1<<k);
        
    }
};
