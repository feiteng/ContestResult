class Solution {
public:
    set<int>ret;
    int fun(string& str,int l,int r)
    {
        int ret=0;
        for(int i=l;i<=r;i++)
            ret=ret*2+str[i]-'0';
        return ret;
    }
    bool hasAllCodes(string s, int k) {
        int r=k-1,l=0;
        for(;r<s.size();r++)
        {        
            ret.insert(fun(s,l,r));
            l++;
        }
        int n=1<<(k);
        for(int i=0;i<n;i++)
        {
            if(ret.count(i)==0)
                return false;
        }
        return true;
    }
};
