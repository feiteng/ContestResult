class Solution {
public:
    bool hasAllCodes(string s, int k) {
        vector<int> h(1<<k,0);
        int t= 0;
        for(int i = 0 ; i < s.size() ;i ++) {
            t += s[i]-'0';
            if(i>=k) t -= (s[i-k]-'0')<<k;
            if(i>=k-1) h[t]++;
            t <<= 1;
        }
        
        for(int i = 0 ; i < (1<<k) ; i++) if(h[i]==0) return false;
        return true;
    }
};
