class Solution {
public:
    set<int>ret;
    int fun(string& str,int l,int r)
    {
        int sum=0;
        for(int i=l;i<=r;i++)
            sum=sum*2+str[i]-'0';
        return sum;
    }
    bool hasAllCodes(string s, int k) {
        int r=k-1,l=0;
        for(;r<s.size();r++)
        {        
            ret.insert(fun(s,l,r));
            l++;
        }
        
        for(int i=0;i<(1<<k);i++)
        {
            if(ret.count(i)==0)
                return false;
        }
        return true;
    }
};
