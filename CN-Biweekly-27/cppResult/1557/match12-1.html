class Solution {
public:
    bool hasAllCodes(string s, int k) {
        set<string> S;
        int len=s.length();
        int cnt=len-k+1;
        for (int i=0; i+k-1<len; i++) {
            S.insert(s.substr(i,k));
            int num=S.size();
            if (cnt+num<(1<<k)) return false;
        }
        return S.size()==(1<<k);
    }
};
