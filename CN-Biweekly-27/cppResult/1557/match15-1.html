class Solution {
public:
    bool hasAllCodes(string s, int k) {
        std::set<int> res;
        int n = s.size();

        for (int i=0;i+k<=n;i++){
            int mask = 0;
            for (int j=i;j<i+k;j++){
                mask = mask*2+s[j]-'0';
            }
            res.insert(mask);
        }
        return (res.size()==((1<<k)));
    }
};
