class Solution {
public:
    int getnum(string& s,int a,int b)
    {
        int ans=0;
        for(int i=a;i<=b-1;i++)
        {
            ans*=2;
            if(s[i]=='1')
            {
                ans+=1;
            }
        }
        return ans;
    }
    int helper(string& s,int a,int b,int curnum,int ss)
    {
        if(s[a]=='1')
        {
            curnum-=ss;
        }
        curnum*=2;
        if(s[b]=='1')
        {
            curnum+=1;
        }
        return curnum;
    }
    bool hasAllCodes(string s, int k) {
    if(k==1)
    {
        bool has1=false;
        bool has0=false;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='1')
            {
                has1=true;
            }
            else
            {
                has0=true;
            }
        }
        return has1&&has0;
    }
    int i=pow(2,k);
    if(s.size()<i+k-1)
    {
        return false;
    }
    vector<bool> vb;
    vb.resize(i);
    for(int j=0;j<i;j++)
    {
        vb[j]=false;
    }
    int curnum=getnum(s,0,k);
    int ss=i/2;
    for(int j=0;j<s.size()-k;j++)
    {
        vb[curnum]=true;
        curnum=helper(s,j,j+k,curnum,ss);
    }
    vb[curnum]=true;
    for(int j=0;j<i;j++)
    {
        if(vb[j]==false)
        {
            return false;    
        }
    }
    return true;
    }
};
