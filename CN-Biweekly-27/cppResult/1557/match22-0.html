class Solution {
public:
    bool hasAllCodes(string s, int k) {
        unordered_set<string> st;
        if(s.size() <= k) return false;
        for (int i = 0; i + k - 1 < s.size(); ++i) {
            st.insert(s.substr(i, k));
           // cout << s.substr(i, k ) << endl;
        }
        int cnt = pow(2, k);
        if (cnt != st.size()) return false;
        return true;
    }
};
