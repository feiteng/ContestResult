const int N = 1 << 20;

bool st[N];

class Solution {
public:
    bool hasAllCodes(string s, int k) {
        memset(st, 0, sizeof st);
        

        for(int i = 0, x = 0; i < s.size(); i ++)
        {
            x = x * 2 + s[i] - '0';
            if(i - k >= 0)  x -= (s[i - k] - '0') << k;
            if(i >= k - 1)  st[x] = true;
        }
        for(int i = 0; i < 1 << k; i ++)
            if(!st[i])  return false;
        return true;
    }
};
