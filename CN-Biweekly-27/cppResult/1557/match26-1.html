class Solution {

public:
    bool hasAllCodes(string s, int k) {
        if (s.size() < k) return false;
    unordered_set<string> flag;
    for (int i = 0; i <= s.size()-k; ++i) {
        string tmp = s.substr(i, k);
        flag.insert(tmp);
    }

    int i = 0, ttl = 1;
    while (i++ < k) ttl *= 2;
    return flag.size() >= ttl;
    }
};
