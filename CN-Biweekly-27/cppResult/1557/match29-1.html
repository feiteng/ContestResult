class Solution {
public:
    bool i[10000007];
    bool hasAllCodes(string s, int k) {
        if(s.size()<k)return false;
        memset(i,0,sizeof(i));

        int sum=0;
        for(int u=0;u!=k-1;u++){
            sum+=s[u]-'0';
            sum<<=1;
        }
        for(int u=k-1;u!=s.size();u++){
            sum+=s[u]-'0';
            i[sum]=true;
            sum<<=1;
            sum&=(1<<k)-1;
        }
        for(int u=0;u!=(1<<k);u++)if(!i[u])return false;
        return true;
    }
};
