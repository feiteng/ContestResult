class Solution {
    class Node implements Comparable<Node> {
        String str;
        int idx;
        public Node(String str, int idx) {
            this.str = str;
            this.idx = idx;
        }

        @Override
        public int compareTo(Node o) {
            if (str.length() != o.str.length()) {
                return str.length() - o.str.length();
            }
            return idx - o.idx;
        }
    }

    public String arrangeWords(String text) {
        String[] arr = text.split(" ");
        
Node[] all = new Node[arr.length];
        for (int i = 0; i < arr.length; i++) {
            all[i] = new Node(arr[i].toLowerCase(), i);
        }
        Arrays.sort(all);

        StringBuilder res = new StringBuilder();
        for (int i = 0; i < all.length; i++) {
            if (i == 0) {
                String str = all[i].str;
                char[] chr = str.toCharArray();
                chr[0] = Character.toUpperCase(chr[0]);
                
res.append(new String(chr));
            } else {
                res.append(all[i].str);
            }
            if (i < arr.length - 1) {
                res.append(" ");
            }
        }
        return res.toString();
    }
}
