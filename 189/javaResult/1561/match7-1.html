class Solution {
     class Key {
        String str;
        int index;

        public Key(String str, int index) {
            this.str = str;
            this.index = index;
        }
    }

    public String arrangeWords(String text) {
        String[] ts = text.split(" ");
        Map<String, Integer> pos = new HashMap<>();
        
List<Key> list = new ArrayList<>();
        for (int i = 0; i < ts.length; i++) {
            list.add(new Key(ts[i], i));

        }
        Collections.sort(list, new Comparator<Key>() {
            @Override
            public int compare(Key o1, Key o2) {
                if (o1.str.length() != o2.str.length()) {
                    return Integer.compare(o1.str.length(), o2.str.length());
                } else {
                    return Integer.compare(o1.index, o2.index);
                }
            }
        });

        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < ts.length; i++) {
            String str = list.get(i).str;
            if (i == 0) {
                str = Character.toUpperCase(str.charAt(0)) + str.substring(1, str.length());
            }else{
                str = Character.toLowerCase(str.charAt(0)) + str.substring(1, str.length());
            }
            
if (sb.length() > 0) {
                sb.append(" ");
            }
            sb.append(str);
        }
        return sb.toString();
    }

}
