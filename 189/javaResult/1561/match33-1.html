class Solution {
    public String arrangeWords(String text) {
        String[] words = text.split(" ");
        List<Token> tokens = new ArrayList<>();
        for (int i = 0; i < words.length; i++) {
            tokens.add(new Token(i, words[i]));
        
}
        tokens.sort(new Comparator<Token>() {
            @Override
            public int compare(Token o1, Token o2) {
                int c = Integer.compare(o1.getS().length(), o2.getS().length());
                return c != 0 ? c : Integer.compare(o1.getI(), o2.getI());
            }
        });
        String s = tokens.stream().map(Token::getS).collect(Collectors.joining(" "));
        return s.substring(0, 1).toUpperCase() + s.substring(1).toLowerCase();
    }
    
    class Token {
        int i;
        String s;

        public Token(int i, String s) {
            this.i = i;
            this.s = s;
        }

        public int getI() {
            return i;
        }

        public String getS() {
            return s;
        }
    }
}
