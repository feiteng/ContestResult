class Solution {
    class Pair{
        String s;
        int index;
        public Pair(String s, int index){
            this.s = s;
            this.index = index;
        }
    }
    public String arrangeWords(String text) {
        String[] split = text.split(" ");
        int n = split.length;
        Pair[] p = new Pair[n];
        for(int i=0; i<n; i++){
            p[i] = new Pair(split[i].toLowerCase(), i);
        }
        Arrays.sort(p, (a,b)->(a.s.length() == b.s.length() ? a.index-b.index : a.s.length() - b.s.length()));
        StringBuilder sb = new StringBuilder();
        for(int i=0; i<n; i++){
            if(i==0){
                char[] cur = p[i].s.toCharArray();
                cur[0] = (char)(cur[0]-'a'+'A');
                sb.append(new String(cur));
            }
            else{
                sb.append(p[i].s);
            }
            if(i!=n-1){
                sb.append(" ");
            }
        }
        return sb.toString();
    }
}
