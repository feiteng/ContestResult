import java.util.Arrays;

class Solution {
    public String arrangeWords(String text) {
        text = text.toLowerCase();
        String[] s = text.split(" ");
        Arrays.sort(s, (a, b) -> a.length() - b.length());
        String ans = String.join(" ", s);
        return (char)(ans.charAt(0) + 'A' - 'a') + ans.substring(1);
    }
}
