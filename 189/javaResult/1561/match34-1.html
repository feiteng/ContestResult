class Solution {
    public String arrangeWords(String text) {
        String[] words = text.split(" ");
        List<Word> list = new ArrayList<>();
        
        for (int i = 0; i < words.length; i++) {
            if (i == 0) {
                list.add(new Word(words[i].toLowerCase(), i));
            } else {
                list.add(new Word(words[i], i));
            }
        }
        
        Collections.sort(list, new Comparator<Word>() {
            @Override
            public int compare(
Word s1, Word s2) {
                if (s1.s.length() == s2.s.length()) {
                    return s1.pos - s2.pos;
                }
                
                return s1.s.length() - s2.s.length();
            }
        });
        
        StringBuilder sb = new StringBuilder();
        
        for (Word w : list) {
            if (sb.length() == 0) {
                char c = w.s.charAt(0);
                sb.append(Character.toUpperCase(c));
                if (w.s.length() > 1) {
                    sb.append(w.s.substring(1));
                }
            } else {
                sb.append(" ");
                sb.append(w.s);
            }
        }
        
        return 
sb.toString();
    }
    
    class Word {
        String s;
        int pos;
        
        public Word(String s, int pos) {
            this.s = s;
            this.pos = pos;
        }
    }
}
