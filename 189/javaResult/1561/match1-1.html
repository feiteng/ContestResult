class Solution {
    public String arrangeWords(String text) {
        text = text.toLowerCase();
        String[] words = text.split(" ");
        PriorityQueue<Word> pq = new PriorityQueue<>(new Comparator<Word>() {
            @Override
            public int compare(Word o1, Word o2) {
                if(o1.len == o2.len){
                    return o1.ori - o2.ori;
                }else{
                    return o1.len - o2.len;
                }
            }
        });
        for(int i  = 0 ; i < words.length; i++ ){
            pq.offer(new Word(words[i], i));
        }
        List<String> list = new ArrayList<>();
        while(!pq.isEmpty()){
            list.add(pq.poll().word);
        }
        StringBuilder sb = new StringBuilder();
        for(int i = 0;i < list.size(); i++){
            sb.append(list.get(i));
            if(i!=list.size()-1){
                sb.append(" ");
            }
        }
        String ans = sb.toString();
        return ans.substring(0, 1).toUpperCase() + ans.substring(1);
    }

}


class Word{
    int len;
    int ori;
    String word;
    public Word(String word, int ori){
        this.len = word.length();
        this.word = word;
        this.ori = ori;
    }
}
