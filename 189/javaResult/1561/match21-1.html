class Solution {
    private class Word implements Comparable<Word>{
        int id;
        String st;
        public Word(int i, String s) {
            id = i;
            st = s;
        }
        public int compareTo(Word other) {
            return st.length() == other.st.length() ? id - other.id : st.length() - other.st.length();
        
}
    }
    public String arrangeWords(String text) {
        String[] split = text.split(" ");
        int n = split.length;
        Word[] words = new Word[n];
        for (int i=0; i<n; i++) {
            words[i] = new Word(i, split[i]);
        }
        Arrays.sort(words);
        StringBuilder sb = new StringBuilder();
        boolean f = false;
        for (Word word : words) {
            if (!f) {
                sb.append(word.st.substring(0, 1).toUpperCase());
                sb.append(word.st.substring(1));
                f = true;
            }
            else {
                sb.append(" ");
                sb.append(
word.st.toLowerCase());
            }
        }
        return sb.toString();
    }
}
