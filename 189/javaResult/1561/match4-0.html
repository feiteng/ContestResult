class Solution {
    public String arrangeWords(String text) {
        String[] words = text.toLowerCase().split("\\s+");
        Arrays.sort(words, (a,b) -> (a.length() - b.length()));
        String firstWord = "" + Character.toUpperCase(words[0].charAt(0)) + words[0].substring(1);
        StringBuilder ans = new StringBuilder(firstWord);
        for (int i = 1; i < words.length; i++){
            ans.append(' ');
            ans.append(words[i]);
        }
        return ans.toString();
    }
}
