class Solution {
    public String arrangeWords(String text) {
        String[] strs = text.split(" ");
        List<Word> words = new ArrayList();
        for (int i = 0; i < strs.length; i ++) {
            words.add(new Word(strs[i], i));
        }
        
        Collections.sort(words, (a,b)-> Integer.compare(a.str.length(), b.str.length()) == 0 ? Integer.compare(a.idx, b.idx) : Integer.compare(a.str.length(), b.str.length()));
        
        StringBuilder sb = new StringBuilder();
        String w1 = words.get(0).str;
        String first = w1.substring(0, 1).toUpperCase() + w1.substring(1);
        sb.append(first);
        for (int i = 1; i < strs.length; i ++) {
            sb.append(" ");
            sb.append(words.get(i).str.toLowerCase());
        }
        return sb.toString();
    }
    
    public class Word {
        String str;
        int idx;
        public Word(String s, int i) {
            idx = i;
            str = s;
        }
    }
}
