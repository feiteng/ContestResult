class Solution {
    class C {
        String w;
        int i;
        public C(String w, int i) {
            this.w = w;
            this.i = i;
        }
    }
    
    public String arrangeWords(String text) {
        String[] w = text.split(" ");
        C[] c = new C[w.length];
        for (int i = 0; i < w.length; i++) {
            c[i] = new C(w[i], i);
        
}
        Arrays.sort(c, (a,b) -> a.w.length() ==  b.w.length() ? a.i - b.i : a.w.length() - b.w.length());

        StringBuilder sb = new StringBuilder();
        sb.append(c[0].w.substring(0, 1).toUpperCase() + c[0].w.substring(1));
        for (int i = 1; i < c.length; i++) {
            sb.append(" ");
            sb.append(c[i].w.toLowerCase());
        }
        
        return sb.toString();
    }
}
