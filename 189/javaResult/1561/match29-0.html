class Solution {
    public String arrangeWords(String text) {
        TreeMap<Integer, List<String>> map = new TreeMap<>();//length -> words
        for(String word: text.split(" ")){
            int l = word.length();
            map.putIfAbsent(l, new ArrayList<>());
            map.get(l).add(word.toLowerCase());
        }
        StringBuilder sb = new StringBuilder();
        for(int l : map.navigableKeySet()){
            for(String word: map.get(l)){
                sb.append(word);
                sb.append(" ");
            }
        }
        String result = sb.toString();
        String first = result.substring(0, 1).toUpperCase();
        result = first + result.substring(1);
        return result.trim();
    }
}
