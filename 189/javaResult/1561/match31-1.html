class Solution {

    public String arrangeWords(String text) {
        String[] split = text.toLowerCase().split(" ");
        Arrays.sort(split, (a, b) -> a.length() - b.length());
        String result = Character.toUpperCase(split[0].charAt(0)) + split[0].substring(1);
        for (int i = 1; i < split.length; i++) {
            result += " " + split[i];
        }
        return result;
    }
}
