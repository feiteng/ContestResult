class Solution {
  static class Bundle5413 {
    String w;
    int idx;
    public Bundle5413(String w, int indx) {
      this.w = w;
      this.idx = indx;
    }
  }
    public String arrangeWords(String text) {
           String[] words = text.toLowerCase().split("\\s+");
    int n = words.length;
    List<Bundle5413> wa = new ArrayList<>();
    for (int i = 0; i < n; i++) {
      wa.add(new Bundle5413(words[i], i));
    }
    Collections.sort(wa, new Comparator<Bundle5413>() {
      @Override
      public int compare(Bundle5413 o1, Bundle5413 o2) {
        if (o1.w.length() != o2.w.length()) {
          return Integer.compare(o1.w.length(), o2.w.length());
        } else {
          return Integer.compare(o1.idx, o2.idx);
        }
      }});
    StringBuilder sb = new StringBuilder();
    for (Bundle5413 b : wa) {
      if (sb.length() > 0) {
        sb.append(" ");
      }
      sb.append(b.w);
    }
    sb.setCharAt(0, Character.toUpperCase(sb.charAt(0)));
    return sb.toString();
    }
}
