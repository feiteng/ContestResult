class Solution {
    public String arrangeWords(String text) {
        text = text.toLowerCase();
        String[] split = text.split(" ");
        Arrays.sort(split, (a, b) -> (a.length() - b.length()));
        String res = String.join(" ", split);
        if (res.length() > 0) {
            String zero = res.charAt(0) + "";
            zero = zero.toUpperCase();
            res = zero + res.substring(1);
        }
        return res;

    }
}
