class Solution {
    public String arrangeWords(String text) {
        String[] s = text.split(" ");
        s[0] = 
s[0].toLowerCase();
        Arrays.sort(s, (a,b) -> a.length() - b.length());
        StringBuilder sb = new StringBuilder();
        sb.append(s[0].substring(0,1).toUpperCase() + s[0].substring(1, s[0].length()));
        for(int i = 1; i < s.length; i++){
            sb.append(" " + s[i]);
        }
        return sb.toString();
    }
}
