class Solution {
    static Comparator cpt = new Comparator(){
        @Override
        public int compare(Object o1, Object o2){
            S s1 = (S)o1, s2 = (S)o2;
            if(s1.l == s2.l) return Integer.compare(s1.i, s2.i);
            return Integer.compare(s1.l, s2.l);
        }
    };
    static int d = (int)('a' - 'A');
    public String arrangeWords(String text) {
        PriorityQueue<S> q = new PriorityQueue(cpt);
        StringBuilder stb = new StringBuilder();
        for(int i = 0; i < text.length(); i++){
            char c = text.charAt(i);
            if(c == ' '){
                q.add(new S(stb.toString(), i));
                stb = new StringBuilder();
            }
            else{
                if(c < 'a') c += d;
                stb.append(c);
            }
        }
        q.add(new S(stb.toString(), text.length()));
        stb = new StringBuilder();
        boolean first = true;
        while(!q.isEmpty()){
            S z = q.remove();
            if(first){
                String v = z.v;
                for(int i = 0; i < v.length(); i++){
                    stb.append(i == 0 ?  (char)(v.charAt(i) - d) : v.charAt(i));
                }
                first = false;
            }
            else{
                stb.append(z.v);
            }
            if(!q.isEmpty()) stb.append(' ');
        }
        return stb.toString();
    }
}

class S{
    String v;
    int i;
    int l;
    S(String v, int i){
        this.v = v;
        this.l = v.length();
        this.i = i;
    }
}
