import java.util.Arrays;

class Solution {
  public String arrangeWords(String paramString) {
    String[] arrayOfString = paramString.toLowerCase().split(" ");
    Arrays.sort(arrayOfString, (paramString1, paramString2) -> paramString1.length() - paramString2.length());
    String str = String.valueOf(Character.toUpperCase(arrayOfString[0].charAt(0))) + arrayOfString[0].substring(1);
    for (int b = 1; b < arrayOfString.length; b++)
      str = String.valueOf(str) + " " + arrayOfString[b]; 
    return str;
  }
}
