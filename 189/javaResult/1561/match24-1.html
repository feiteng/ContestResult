class Solution {
    public String arrangeWords(String text) {
        String[] t = text.split(" ");
        t[0] = 
t[0].toLowerCase();
        List<Node> l = new ArrayList<>();
        for (int i = 0; i < t.length; i++) {
            l.add(new Node(t[i], i));
        }
        Collections.sort(l, (e1, e2) -> {
            if (e1.s.length() == e2.s.length()) {
                return Integer.compare(e1.i, e2.i);
            }
            return Integer.compare(e1.s.length(), e2.s.length());
        });
        Node first = l.get(0);
        String fs = first.s;
        first.s = fs.substring(0, 1).toUpperCase() + fs.substring(1);
        List<String> ls = new ArrayList<>();
        for (Node x : l) {
            
ls.add(x.s);
        }
        return String.join(" ", ls);
    }

    private static class Node {
        String s;
        int i;
        private Node(String s, int i) {
            this.s = s;
            this.i = i;
        }
    }
}
