class Solution {
    private class Ele {
        String s;
        int idx;
        public Ele(String s, int idx) {
            this.s = s;
            this.idx = idx;
        }
    }
    public String arrangeWords(String text) {
        String[] strs = text.split(" ");
        int n = strs.length;
        Ele[] eles = new Ele[n];
        for (int i = 0; i < n; i++) {
            eles[i] = new Ele(strs[i].toLowerCase(), i);
        }
        
        Arrays.sort(eles, new Comparator<Ele>(){
            @Override
            public int compare(Ele e1, Ele e2) {
                if (e1.s.length() == e2.s.length()) {
                    return e1.idx - e2.idx;
                }
                return e1.s.length() - e2.s.length();
            }
        });
        
        // System.out.println(eles[0].s.substring(0, 1).toUpperCase());
        eles[0].s = eles[0].s.substring(0, 1).toUpperCase() + 
                    (eles[0].s.length() > 1 ? eles[0].s.substring(1) : "");
        // eles[0]
        StringBuilder sb = new StringBuilder();
        for (Ele e : eles) {
            if (sb.length() != 0) sb.append(" ");
            sb.append(e.s);
        }
        return sb.toString();
    }
}
