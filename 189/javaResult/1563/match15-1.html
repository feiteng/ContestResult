class Solution {
    
class Point
	{
		double x, y;

		Point(double x, double y)
		{
			this.x = x;
			this.y = y;
		}
	}

	private
 double getDistance(Point p1, Point p2)
	{
		return Math.sqrt(
				(p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y));
	}

	private Point getCircleCenter(Point p1, Point p
2, double r)
	{
		Point mid = new Point((p1.x + p2.x) / 2, (p1.y + p2.y) / 2);
		double angle = Math.atan2(p1.x - p2.x, p2.y - p1.y);
		double d = Math.sqrt(r * r - Math.pow(getDistance(p1, mid), 2));
		return new Point(mid.x + d * Math.cos(angle),
				mid.y + d * Math.sin(angle));
	}

	public int numPoints(int[][] points, int r)
	{

		Point[] p = new Point[poin
ts.length];
		for (int i = 0; i < points.length; i++)
		{
			p[i] = new Point((double) points[i][0], (double) points[i][1]);
		}

		int ans = 1;
		for (int i = 0; i < p.length; i++)
		{
			for (int j = i + 1; j < p.length; j++)
			{

				if (getDistance(p[i], p[j]) > 2.0 * r)
					continue;
				Point center = getCircleCenter(p[i], p[j], (double) r);
				int count = 0;
				for (int k = 0; k < p.length; k++)
				{
					if (getDistance(center, p[k]) < 1.0 * r + 0.000001)
						count++;
				}

				ans = Math.max(ans, count);
			}
		}

		return ans;
	}
}
