class Solution {
    public int numPoints(int[][] points, int r) {
    int ans = 1;
    for(int i=0; i<points.length; i++){
      for(int j=0; j<points.length; j++){
        if(i==j||disInt(points[i],points[j])>0.0001+2*r) continue;
        int cnt = 0;
        double center[] = getCenterPos(points[i],points[j],r);
        for(int k=0; k<points.length; k++){
          if(disDouble(points[k],center)<=r+0.0001)
            cnt++;
        }
        ans = Math.max(ans,cnt);
      }
    }
    return ans;
  }
    
    public double disDouble(int[] x, double[] y){
    return Math.sqrt((x[0]-y[0])*(x[0]-y[0])+(x[1]-y[1])*(x[1]-y[1]));
  }
  public double disInt(int[] x, int[] y){
    return Math.sqrt((x[0]-y[0])*(x[0]-y[0])+(x[1]-y[1])*(x[1]-y[1]));
  }

  public double[] getCenterPos(int[] a, int[] b, int r){
    double[] mid = new double[2], res = new double[2];
    mid[0] = (double) (a[0]+b[0])/2;
    mid[1] = (double) (a[1]+b[1])/2;
    double angle = Math.atan2(b[1]-a[1],b[0]-a[0]);
    double temp = disInt(a,b)/2;
    double dist = Math.sqrt(r*r-temp*temp);
    res[0] = mid[0]-dist*Math.sin(angle);
    res[1] = mid[1]+dist*Math.cos(angle);
    return res;
  }


}
