class Solution {

    class Point
    {
        double x,y;
        Point(){}
        
Point(double tx,double ty){x=tx;y=ty;}
    };
 
double dist(Point p1,Point p2)
{
	return Math.sqrt(Math.pow(p1.x-p2.x,2)+Math.pow(p1.y-p2.y,2));
}
 
Point GetCircleCenter(Point p1,
Point p2, int r)
{
	Point mid = new Point((p1.x+p2.x)/2,(p1.y+p2.y)/2);
	double angle = Math.atan2(p1.x-p2.x,p2.y-p1.y);
	double d = Math.sqrt(r * r - Math.pow(dist(p1,mid),2));
	return new Point(mid.x+d * Math.cos(angle), mid.y+d * Math.sin(angle));
}



    public int numPoints(int[][] points, int r) {
        
        int num = points.length;
        Point[] p = 
new Point[num];
        
        for (int i = 0; i < points.length; i++) {
            p[i] = new Point(points[i][0] * 1.0, points[i][1] * 1.0);
        }
        
        int ans = 1;
		for(int i=0;i<num;i++)
		{
			for(int j=i+1;j<num;j++)
			{

				if(dist(p[i],p[j]) > 2.0*r) continue;
				Point center = GetCircleCenter(p[i],p[j], r);
				int cnt = 0;
				for(int k=0;k<num;k++)
					if(dist(center,p[k]) < 1.0*r+0.00000001) cnt++;
				ans = Math.max(ans,cnt);
			}
		}
        return ans;
    }
}
