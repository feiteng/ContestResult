class Solution {
    public int numPoints(int[][] points, int r) {
        double eps = 1e-8;
        int count = 1;
        for (int i = 0; i < points.length; i++) {
            for (int j = i + 1; j < points.length; j++) {
                if (dist(points[i][0], points[i][1], points[j][0], points[j][1]) > 2 * r) continue;
                double[] center = getCircleCenter(points[i][0], points[i][1], points[j][0], points[j][1], r);
                int tempCount = 0;
                for (int k = 0; k < points.length; k++) {
                    if (dist(center[0], center[1], points[k][0], points[k][1]) < 1.0 * r + eps) {
                        tempCount++;
                    }
                }
                count = Math.max(count, tempCount);
            }
        }

        return count;
    }

    private double[] getCircleCenter(double p1X, double p1Y, double p2X, double p2Y, int r) {
        double[] mid = {(p1X + p2X) / 2, (p1Y + p2Y) / 2};
        double angle = Math.atan2(p1X - p2X, p2Y - p1Y);
        double d = Math.sqrt(r * r - Math.pow(dist(p1X, p1Y, mid[0], mid[1]), 2));
        return new double[]{mid[0] + d * Math.cos(angle), mid[1] + d * Math.sin(angle)};
    }

    private double dist(double p1X, double p1Y, double p2X, double p2Y) {
        return Math.sqrt(Math.pow(p1X - p2X, 2) + Math.pow(p1Y - p2Y, 2));
    }
}
