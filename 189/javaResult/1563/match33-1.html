class Solution {
    
    
    
    public int numPoints(int[][] points, int R) {
        int n = points.length;
        double r = (double) R;
        double eps = 1e-8;
        double[][] ps = new double[n][2];
        for (int i = 0; i < n; i++) {
            ps[i][0] = (double) points[i][0];
            ps[i][1] = (double) points[i][1];
        }
        
        
int res = 1;
        
        for (int i = 0; i < n; i++) {
			for (int j = i + 1; j < n; j++) {
				if (dist(ps[i], ps[j]) > 2.0*r) {
                    continue;
                
}
				double[] center = getCircleCenter(ps[i], ps[j], r);
				int cnt = 0;
				for (int k = 0; k < n; k++) {
                    if (dist(center, ps[k]) < 1.0 * r + eps) {
                        cnt++;
                    }
                }	
				res = Math.max(res,cnt);
			}
		}
        
        return res;
    }
    
    private double[] getCircleCenter(double[] p1, double[] p2, double r) 
{
	    double[] mid = new double[] {(p1[0] + p2[0]) / 2, (p1[1] + p2[1]) / 2};
	    double angle = Math.atan2(p1[0] - p2[0], p2[1] - p1[1]);
	    double d = Math.sqrt(r * r - Math.pow(dist(p1, mid), 2));
	    return new double[] {mid[0] + d * Math.cos(angle), mid[1] + d * Math.sin(angle)};
    }
    
    private double dist(double[] p1, double[] p2) {
        return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
    }

}
