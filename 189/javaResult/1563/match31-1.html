class Solution {
    public int numPoints(int[][] origin_points, int r) {
        if (origin_points.length <= 1) {
            return origin_points.length;
        }
        int n = origin_points.length;
        double[][] points = new double[n][2];
        for (int i = 0; i < n; i++) {
            points[i][0] = origin_points[i][0];
            points[i][1] = origin_points[i][1];
        }
        
int ans = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i == j || dis(points[i], points[j]) - 2 * r > 0) continue;
                double[] center = center(r, points[i], points[j]);
                int cnt = 0;
                
for (double[] point : points) {
                    if (dis(point, center) - r <= 0.000001) cnt++;
                }
                ans = Math.max(ans, cnt);
            }
        }
        return ans;
    }

    private double[] center(int r, double[] a, 
double[] b) {
        double[] mid = new double[2];
        double[] res = new double[2];
        mid[0] = (a[0] + b[0]) / 2;
        mid[1] = (a[1] + b[1]) / 2;
        double angle = Math.atan2(b[1] - a[1], b[0] - a[0]);
        if (angle < 0) angle += 2 * Math.acos(-1);
        
double tmp = dis(a, b) / 2;
        double d = Math.sqrt(r * r - tmp * tmp);
        res[0] = mid[0] - d * Math.sin(angle);
        res[1] = mid[1] + d * Math.cos(angle);
        return res;
    
}

    private double dis(double[] a, double[] b) {
        return Math.sqrt((a[0] - b[0]) * (a[0] - b[0]) +
                (a[1] - b[1]) * (a[1] - b[1]));
    }

    public static void main(String[] args) {
        System.out.println(new Solution().numPoints(new int[][]{{-2, 0}, {2, 0}, {0, 2}, {0, -2}}, 2));
    }
}
