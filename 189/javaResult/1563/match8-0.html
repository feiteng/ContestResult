class Solution {
    private static final double EPS = 1e-8;

    private class Point {
        double x, y;

        public Point(double nx, double ny) {
            x = nx;
            y = ny;
        }
    }

    public int numPoints(int[][] points, int r) {
        Point[] ps = new Point[points.length];
        for (int i = 0; i < points.length; i++) {
            ps[i] = new Point(points[i][0], points[i][1]);
        }
        int ans = 1;
        for (int i = 0; i < ps.length; i++) {
            for (int j = i + 1; j < ps.length; j++) {
                if (dist(ps[i], ps[j]) > 2.0 * r) {
                    continue;
                }
                Point center = getCircleCenter(ps[i], ps[j], r);
                int cnt = 0;
                for (int k = 0; k < ps.length; k++) {
                    if (dist(center, ps[k]) < 1.0 * r + EPS) {
                        cnt++;
                    }
                }
                ans = Math.max(ans, cnt);
            }
        }
        return ans;
    }

    private double dist(Point p1, Point p2) {
        return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
    }

    private Point getCircleCenter(Point p1, Point p2, double r) {
        Point center = new Point((p1.x + p2.x) / 2, (p1.y + p2.y) / 2);
        double angle = Math.atan2(p1.x - p2.x, p2.y - p1.y);
        double tmp = Math.sqrt(r * r - Math.pow(dist(p1, center), 2));
        return new Point(center.x + tmp * Math.cos(angle), center.y + tmp * Math.sin(angle));
    }
}
