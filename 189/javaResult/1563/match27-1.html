class Solution {
    public int numPoints(int[][] points, int r) {
        double[][] p = new double[points.length][2];
        int n = p.length;
        
for (int i = 0; i < n; i++) {
            p[i][0] = points[i][0];
            p[i][1] = points[i][1];
            // System.out.println("x = " + p[i][0] + " y = " + p[i][1]);
        }
        int res = 1;
        
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                // System.out.println(dist(p[i],p[j]));
                // System.out.println(dist(p[i],p[j]) > 2.0 * r);
                if (dist(p[i],p[j]) > 2.0 * r) continue
;
                double[] center = getCenter(p[i],p[j], r);
                int count = 0;
                for (int k = 0; k < n; k++) {
                    if (dist(center, p[k]) < 1.0 * r + 1e-8) count++
;
                }
                // System.out.println(count);
                res = Math.max(res, count);
            }
        }
        return res;
    }
    
    public double dist(double[] a, double[] b) {
        return Math.sqrt(Math.pow(a[0]-b[0],2) + Math.pow(a[1]-b[1],2));
    
}
    
    public double[] getCenter(double[] a, double[] b, int r) {
        double[] mid = new double[]{(a[0] + b[0])/2, (a[1] + b[1])/2};
        double angle = Math.atan2(a[0]-b[0], b[1]-a[1]);
        double d = Math.sqrt(r * r - Math.pow(dist(a,mid),2));
        return new double[]{mid[0] + d * Math.cos(angle), mid[1] + d * Math.sin(angle)};
    }
}
