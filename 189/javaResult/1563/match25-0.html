class Solution {
    double eps=1e-8;
    double r;

class Point
{
    double x,y;
    
   public Point(double tx,double ty)
    {
        x=tx;
        y=ty;
    }
}
Point[] p;
    
double dist(Point p1,Point p2)
{
    return Math.sqrt((p1.x-p2.x)*(p1.x-p2.x)+(p1.y-p2.y)*(p1.y-p2.y));
}

Point get_circle(Point p1,Point p2)
{
    Point mid = new Point((p1.x+p2.x)/2,(p1.y+p2.y)/2);
    double angle = Math.atan2(p1.x-p2.x,p2.y-p1.y);
    double d = Math.sqrt(r*r- Math.pow(dist(p1,mid),2));
    return new Point(mid.x+d*Math.cos(angle),mid.y+d*Math.sin(angle));
}

    public int numPoints(int[][] points, int r) {
         int num = 0;
   this.r = r;
        this.p = new Point[300];
    for(int[]tmp : points){
       
        p[num] = new Point(tmp[0], tmp[1]);
        num++;
    }
    int i,j;
    int ans = 1;
    for(i=0; i<num; i++)
    {
        for(j=i+1; j<num; j++)
        {
            if(dist(p[i],p[j]) > 2.0*r) continue;
            Point center = get_circle(p[i],p[j]);
            int cnt = 0;
            for(int k=0; k<num; k++)
                if(dist(center,p[k]) < 1.0*r+eps) cnt++;
            ans = Math.max(ans,cnt);
        }
    }
   return ans;
    }
}
