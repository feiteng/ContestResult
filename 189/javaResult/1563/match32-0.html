class Solution {
    public int numPoints(int[][] points, int r) {
        int n = points.length;
        double[][] p = new double[n][2]; 
        for (int i = 0; i < n; i++) {
            p[i][0] = points[i][0] * 1.0;
            p[i][1] = points[i][1] * 1.0;
        }
        int ret = 1;
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
          //      System.out.println(i + " " + j + " " +  " dist: " + dist(p[i], p[j]));
                if (dist(p[i], p[j]) > 2.0 * r) continue;
                double[] center = getCircleCenter(p[i],p[j], r);
                double[] center1 = new double[]{center[0], center[1]};
                double[] center2 = new double[]{center[2], center[3]};
            //   System.out.println(i + " " + j + " " + center[0] + " " + center[1]);
				int cnt = 0;
				for(int k = 0; k < n; k++)
					if (dist(center1,p[k]) < 1.0 * r + 1e-8) cnt++;
                ret = Math.max(ret, cnt);
                cnt = 0;
				for(int k = 0; k < n; k++)
					if (dist(center2,p[k]) < 1.0 * r + 1e-8) cnt++;
                // System.out.println("cnt: " + cnt);
                ret = Math.max(ret, cnt);
            }
        }
        return ret;
    }
    
    private double dist(double[] p1,double[] p2) {
        return Math.sqrt(Math.pow(p1[0]-p2[0], 2)+ Math.pow(p1[1]-p2[1], 2));
    }
    
    private double[] getCircleCenter(double[] t1, double[] t2, int r) {
        Point p1 = new Point(t1[0], t1[1]);
        Point p2 = new Point(t2[0], t2[1]);
        double k = 0.0,k_verticle = 0.0;
        double mid_x = 0.0,mid_y = 0.0;
        double a = 1.0;
        double b = 1.0;
        double c = 1.0;
        Point center1 = new Point(0, 0);
        Point center2 = new Point(0, 0);
        k = (p2.y - p1.y) / (p2.x - p1.x);
        if(k == 0)
        {
            center1.x = (p1.x + p2.x) / 2.0;
            center2.x = (p1.x + p2.x) / 2.0;
            center1.y = p1.y + Math.sqrt(r * r -(p1.x - p2.x) * (p1.x - p2.x) / 4.0);
            center2.y = p2.y - Math.sqrt(r * r -(p1.x - p2.x) * (p1.x - p2.x) / 4.0);
        }
        else
        {
            k_verticle = -1.0 / k;
            mid_x = (p1.x + p2.x) / 2.0;
            mid_y = (p1.y + p2.y) / 2.0;
            a = 1.0 + k_verticle * k_verticle;
            b = -2 * mid_x - k_verticle * k_verticle * (p1.x + p2.x);
            c = mid_x * mid_x + k_verticle * k_verticle * (p1.x + p2.x) * (p1.x + p2.x) / 4.0 - 
                (r * r- ((mid_x - p1.x) * (mid_x - p1.x) + (mid_y - p1.y) * (mid_y - p1.y)));

            center1.x = (-1.0 * b + Math.sqrt(b * b -4 * a * c)) / (2 * a);
            center2.x = (-1.0 * b - Math.sqrt(b * b -4 * a * c)) / (2 * a);
            center1.y = Y_Coordinates(mid_x,mid_y,k_verticle,center1.x);
            center2.y = Y_Coordinates(mid_x,mid_y,k_verticle,center2.x);
        }
        return new double[]{center1.x, center1.y, center2.x, center2.y};

    }
    double Y_Coordinates(double x,double y,double k,double x0)
{
    return k * x0 - k * x + y;
}

    private class Point {
        double x; 
        double y; 
        public Point(double x, double y) {
            this.x = x;
            this.y = y;
        }
    }

}
