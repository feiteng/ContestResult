class Solution {
    class Point{
        double x, y;
        Point(){}
        Point(double tx, double ty){
            x = tx;
            y = ty;
        }
    }
    public double dist(Point p1, Point p2){
        return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
    }
    public Point getCircleCenter(Point p1, Point p2, int r){
        Point mid = new Point((p1.x + p2.x) / 2, (p1.y + p2.y) / 2);
        double angle = Math.atan2(p1.x-p2.x,p2.y-p1.y);
	    double d = Math.sqrt(r*r-Math.pow(dist(p1,mid), 2));
	    return new Point(mid.x+d*Math.cos(angle), mid.y+d*Math.sin(angle));
    }
    public int numPoints(int[][] points, int r) {
        int n = points.length;
        int ans = 1;
        for(int i = 0; i < n; i++){
            for(int j = i + 1; j < n; j++){
                Point p1 = new Point(points[i][0], points[i][1]);
                Point p2 = new Point(points[j][0], points[j][1]);
                if(dist(p1, p2) > 2.0 * r)
                    continue;
                Point center = getCircleCenter(p1, p2, r);
                int cnt = 0;
                for(int k = 0; k < n; k++){
                    Point p3 = new Point(points[k][0], points[k][1]);
                    if(dist(center, p3) < 1.0 * r + 1e-8)
                        cnt++;
                }
                ans = Math.max(cnt, ans);
            }
        }
        return ans;
    }
}
