class Solution {
    public List<Integer> peopleIndexes(List<List<String>> f) {
        List<Set<String>> l = new ArrayList<>();
        for (List<String> t : f) {
            Set<String> set = new HashSet<>();
            for(String c : t) {
                set.add(c);
            }
            l.add(set);
        }
        
        int n = f.size();
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < n; ++i) {
            boolean exist = false;
            for (int j = 0; j < n; ++j) {
                if (i == j) {
                    continue;
                }
                
                boolean isSub = true;
                for (String c : l.get(i)) {
                    if (!l.get(j).contains(c)) {
                        isSub = false;
                    }
                }
                if (isSub) 
{
                    exist = true;
                    break;
                }
            }
            if (!exist) {
                ans.add(i);
            }
        }
        return ans;
    }
}
