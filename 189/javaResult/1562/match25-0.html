class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        int n = favoriteCompanies.size();
        List<HashSet<String>> setList = new ArrayList<>();
        List<Integer> res = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            HashSet<String> set = new HashSet<>(favoriteCompanies.get(i));
            setList.add(set);
        }
        
        for (int i = 0; i < n; i++) {
            boolean valid = true;
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    if (setList.get(j).containsAll(setList.get(i))) {
                        valid = false;
                        break;
                    }
                }
            }   
            
            if (valid) {
                res.add(i);
            }
        }
        
        return res;
    }
}
