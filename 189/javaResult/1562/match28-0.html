import java.util.ArrayList;
import java.util.HashSet;
import java.util.List;

class Solution {
  public List<Integer> peopleIndexes(List<List<String>> paramList) {
    HashSet[] arrayOfHashSet = new HashSet[paramList.size()];
    for (byte b1 = 0; b1 < arrayOfHashSet.length; b1++)
      arrayOfHashSet[b1] = new HashSet(paramList.get(b1)); 
    ArrayList<Integer> arrayList = new ArrayList();
    for (byte b2 = 0; b2 < paramList.size(); b2++) {
      boolean bool = true;
      for (byte b = 0; b < paramList.size(); b++) {
        if (b2 != b && arrayOfHashSet[b].containsAll(arrayOfHashSet[b2])) {
          bool = false;
          break;
        } 
      } 
      if (bool)
        arrayList.add(Integer.valueOf(b2)); 
    } 
    return arrayList;
  }
}
