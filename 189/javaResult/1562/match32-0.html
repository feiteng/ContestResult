class Solution {
    public List<Integer> peopleIndexes(List<List<String>> f) {
        int n = f.size();
        
        Set<String>[] sets = new HashSet[n];
        
        for (int i = 0; i < n; i++) {
            sets[i] = new HashSet<>(f.get(i));
        }
        
        List<Integer> ans = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            boolean flag = false;
            for (int j = 0; j < n; j++) {
                if (i == j) {
                    continue;
                }
                
                if (sets[j].containsAll(sets[i])) {
                    flag = true;
                    break;
                }
            }
            
            if (!flag) {
                ans.add(i);
            }
        }
        
        return ans;
    }
}
