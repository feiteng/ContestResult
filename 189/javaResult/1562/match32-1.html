class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        final int N = favoriteCompanies.size();
        
        HashMap<String, Integer> companyIDs = new HashMap<>();
        for (List<String> lst : favoriteCompanies) {
            for (String company : lst) {
                if (companyIDs.containsKey(company)) {
                    continue;
                }
                companyIDs.put(company, companyIDs.size());
            }
        }
        
        BitSet[] masks = new BitSet[N];
        for (int i = 0; i < N; ++i) {
            masks[i] = new BitSet(companyIDs.size());
            for (String company : favoriteCompanies.get(i)) {
                int id = companyIDs.get(company);
                masks[i].set(id);
                // masks[i] |= 1 << id;
            }
        
}
        
        // HashSet<Integer> maskSet = new HashSet<>();
        // for (int m : masks) {
        //     maskSet.add(m);
        // }
        // System.out.format("N = %d, maskSet.size() = %d\n", N, maskSet.size());
        
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < N; ++i) {
            boolean isSubset = false;
            for (int j = 0; j < N; ++j) {
                if (i == j) {
                    continue;
                }
                BitSet copyMask = (BitSet)masks[i].clone();
                copyMask.and(masks[j]);
                
if (copyMask.equals(masks[i])) {
                    isSubset = true;
                    break;
                }
            }
            if (!isSubset) {
                ans.add(i);
            }
        }
        return ans;
    }
}
