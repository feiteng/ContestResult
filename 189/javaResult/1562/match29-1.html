class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        for (var l : favoriteCompanies)
            Collections.sort(l)
;
        int n = favoriteCompanies.size();
        List<Integer> ans = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            List<String> li = favoriteCompanies.get(i);
            boolean isSubset = false;
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                List<String> lj = favoriteCompanies.get(j);
                if (li.size() > lj.size()) continue;
                boolean found = true;
                for (String s : li) {
                    int k = Collections.binarySearch(lj, s);
                    if (k < 0) {
                        found = false;
                        break;
                    }
                }
                if (found) 
{
                    isSubset = true;
                    break;
                }
            }
            if (!isSubset) ans.add(i);
        }
        return ans;
    }
}
