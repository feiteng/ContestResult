class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies)
	{
		List<Integer> 
result = new ArrayList<Integer>();
		for (int i = 0; i < favoriteCompanies.size(); i++)
		{
			boolean isValid = true;
			for (int j = 0; j < favoriteCompanies.size(); j++)
			{
				if (i != j)
				{
					if (isSubset(favoriteCompanies.get(i),
							favoriteCompanies.get(j)))
					{
						isValid = false;
						break;
					}
				}

			}
			if (isValid)
				result.add(i);
		}

		return result;
	}

	private boolean isSubset(List<String> l1, List<String> l2)
	{
		HashSet<String> set = new HashSet<String>();
		set.addAll(l2);
		for (String s 
: l1)
		{
			if (!set.contains(s))
				return false;
		}

		return true;
	}
}
