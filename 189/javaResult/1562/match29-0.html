class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        List<Set<String>> sets = new ArrayList<>();
        for (List<String> c : favoriteCompanies) {
            sets.add(new HashSet<>(c));
        }
        
        int N = sets.size();
        List<Integer> indices = new ArrayList<>();
        for (int i = 0; i < N; i++) {
            Set<String> a = sets.get(i);
            boolean isSubset = false;
            for (int j = 0; j < N; j++) {
                if (i == j) continue;
                
                Set<String> b = sets.get(j);
                if (b.containsAll(a)) {
                    isSubset = true;
                    break;
                }
            }
            
            if (!isSubset) indices.add(i);
        }
        
        return indices;
    }
}
