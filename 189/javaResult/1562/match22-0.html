class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        int n = favoriteCompanies.size();
        
        List<Set<String>> lists = new ArrayList<>();
        
        for(List<String> list : favoriteCompanies){
            lists.add(new HashSet<>(list));
        }
        
        List<Integer> result = new ArrayList<>();
        for(int i =0;i<n;i++){
            boolean contained = false;
            Set<String> seti = lists.get(i);
            for(int j = 0;j<n;j++){
                if(j == i){
                    continue;
                }
                Set<String> setj = lists.get(j);
                if(contained(seti, setj)){
                    contained = true;
                    break;
                }
            }
            if(!contained){
                result.add(i);
            }
        }
        return result;
    }
    
    private boolean contained(Set<String> set1, Set<String> set2){
        if(set1.size() > set2.size()){
            return false;
        }
        for(String str:set1){
            if(!set2.contains(str)){
                return false;
            }
        }
        return true;
    }
}
