class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        Map<String, BitSet> owners = new HashMap<>();
        for (int i = 0;i<favoriteCompanies.size();i++) {
            for (String comp : favoriteCompanies.get(i)) {
                if (!owners.containsKey(comp)) {
                    owners.put(comp, new BitSet());
                }
                owners.get(comp).set(i);
            }
        }
        List<Integer> result = new ArrayList<>();
        for (int i = 0;i<favoriteCompanies.size();i++) {
            BitSet bs = null;
            for (String comp : favoriteCompanies.get(i)) {
                if (bs == null) {
                    bs = (BitSet) owners.get(comp).clone();
                    bs.clear(i);
                } else {
                    bs.and(owners.get(comp));
                }
            }
            if (bs.cardinality() == 0) {
                result.add(i);
            }
        }
        return result;
    }
}
