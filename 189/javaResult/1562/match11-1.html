class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        return sln1(favoriteCompanies);
    }
    
    
private List<Integer> sln1(List<List<String>> ll){
        Map<String, Set<Integer>> map = new HashMap<>();
        for(int i =0;i< ll.size();i++){
            for(String s: ll.get(i)){
                if(!map.containsKey(s)) map.put(s, new HashSet<>());
                map.get(s).add(i);
            }
        }
        List<Integer> res = new ArrayList<>();
        for(int i =0;i< ll.size();i++){
            Set<Integer> set= new HashSet<>(map.get(ll.get(i).get(0)));
            for(int j = 1;j< ll.get(i).size();j++){
                Set<Integer> set2 = map.get(ll.get(i).get(j));
                set.retainAll(set2);
            
}
            if(set.size()==1) res.add(i);
        }
        return res;
    }
}
