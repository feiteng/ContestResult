class Solution {
    public List<Integer> peopleIndexes(List<List<String>> A) {
        List<Integer> res = new ArrayList<>();
        
        
List<Set<String>> lists = new ArrayList<>();
        for (List<String> l : A) {
            lists.add(new HashSet<>(l));
        }
        System.out.println(lists);
        int n = A.size();
        outer:
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) 
{
                if (i == j) continue;
                
                if (lists.get(j).containsAll(lists.get(i))) {
                    // System.out.println(i + ", j=" + j);
                    continue outer;
                }  
            }
            res.add(i);
        }
        return res;
    }
}
