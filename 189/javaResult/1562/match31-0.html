class Solution {
        Map<String, Integer> companyToIndex;
        List<Set<Integer>> favorCompanies;
        public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
            companyToIndex = new HashMap<>();
            favorCompanies = new ArrayList<>();
            for(List<String> companies : favoriteCompanies) {
                Set<Integer> companyIds = new HashSet<>();
                for (String company : companies) {
                    companyIds.add(companyToIndex.computeIfAbsent(company, c -> companyToIndex.size()));
                }
                favorCompanies.add(companyIds);
                //System.out.println(companyIds);
            }
            List<Integer> result = new ArrayList<>();
            for (int i = 0; i < favorCompanies.size(); i++) {
                boolean valid = true;
                for (int j = 0; j < favorCompanies.size(); j++) {
                    if (j == i) {
                        continue;
                    }
                    if (contains(favorCompanies.get(j), favorCompanies.get(i))) {
                        valid = false;
                        break;
                    }
                }
                if (valid) {
                    result.add(i);
                }
            }
            return result;
        }

        private boolean contains(Set<Integer> parent, Set<Integer> child) {
           // System.out.println(String.format("%s %s", parent, child));
            Set<Integer> ids = new HashSet<>(parent);
            ids.retainAll(child);
            return ids.size() == child.size();
        }

}
