class Solution {
    public List<Integer> peopleIndexes(List<List<String>> f) {
        
        // 500*500 * 20*1
        
        
        Map<String, Integer> map = new HashMap<>();
        int index = 0;
        
        List<List<Integer>> list = new ArrayList<>();
        for (int i = 0; i < f.size(); i++) {
            List<Integer> temp = new ArrayList<>();
            for (String s : f.get(i)) {
                if (!map.containsKey(s)) {
                    map.put(s, index++);
                }
                temp.add(map.get(s));
            }
            Collections.sort(temp);
            list.add(temp);
        }
        
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < list.size(); i++) {
            boolean add = true;
            for (int j = 0; j < list.size(); j++) {
                if (i == j) continue;
                if (isSub(list, i, j)) {
                    add = false;
                    break;
                }
            }
            if (add) {
                res.add(i);
            }
        }
        
        return res;
    }
    
    private boolean isSub(List<List<Integer>> list, int i, int j) {
        int a = 0, b = 0;
        while(a < list.get(i).size() && b < list.get(j).size()) {
            if (list.get(i).get(a) == list.get(j).get(b)) {
                a++;
                b++;
            }
            else {
                b++;
            }
        }
        
        return a == list.get(i).size();
    }
}
