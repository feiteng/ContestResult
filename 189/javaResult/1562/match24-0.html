class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        ArrayList<String> companies = new ArrayList<>();
        HashMap<String, Integer> indices = new HashMap<>();
        for (List<String> list : favoriteCompanies){
            for (String company : list){
                if (indices.containsKey(company)) continue;
                indices.put(company, companies.size());
                companies.add(company);
            }
        }
        
        ArrayList<HashSet<Integer>> companiesSets = new ArrayList<>();
        for (List<String> list : favoriteCompanies){
            HashSet<Integer> hs = new HashSet<>();
            for (String com : list) hs.add(indices.get(com));
            companiesSets.add(hs);
        }
        
        List<Integer> ans = new ArrayList<>();
        
        for (int i = 0; i < companiesSets.size(); i++){
            boolean isSubset = false;
            for (int j = 0; j < companiesSets.size(); j++){
                if (i==j) continue;
                boolean isSub = true;
                for (int k : companiesSets.get(i)){
                    if (!companiesSets.get(j).contains(k)){
                        isSub = false; break;
                    }
                }
                if (isSub == true){
                    isSubset = true;
                    break;
                }
            }
            if (!isSubset) ans.add(i);            
        }
        
        return ans;
    }
}
