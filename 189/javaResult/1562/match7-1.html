class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        List<TreeSet<String>> fs = new ArrayList<>();
    for (List<String> v : favoriteCompanies) {
      fs.add(new TreeSet<>(v));
    }
    List<Integer> output = new ArrayList<>();
    for (int i = 0; i < fs.size(); i++) {
      boolean ok = true;
      for (int j = 0; j < fs.size(); j++) {
        if (j == i) {
          continue;
        }
        
if (fs.get(j).containsAll(fs.get(i))) {
          ok = false;
          break;
        }
      }
      if (ok) {
        output.add(i);

      }
    }
    return output;
    }
}
