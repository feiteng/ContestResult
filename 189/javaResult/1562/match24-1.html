class Solution {
    public List<Integer> peopleIndexes(List<List<String>> fcs) {
        List<HashSet<String>> ss = new ArrayList<>();
        HashMap<String, Integer> map = new HashMap<>();
        for (List<String> ls : fcs) {
            HashSet<String> set = new HashSet<>();
            for (String s : ls) 
set.add(s);
            ss.add(set);
        }
        List<Integer> res = new ArrayList<>();
        for (int i = 0;i < fcs.size();i++) {
            boolean is = true;
            for (int j = 0;j < fcs.size();j++) {
                if (j == i) continue;
                boolean sub = true;
                HashSet<String> se = ss.get(j);
                for (String s : fcs.get(i)) {
                    if (!
se.contains(s)) {sub = false; break;}
                }
                if (sub) {
                    is = false;
                    break;
                }
            }
            if(is) res.add(i);
        }
        return res;
    }
}
