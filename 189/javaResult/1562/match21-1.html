class Solution {
    public List<Integer> peopleIndexes(List<List<String>> fav) {
        List<Set<String>> favSet = new ArrayList<>();
        for (List<String> list : fav) {
            favSet.add(new HashSet<>(list));
        }
        List<Integer> ret = new ArrayList<>();
        for (int i = 0; i < favSet.size(); i++) {
            Set<String> base = favSet.get(i);
            boolean good = true;
            for (int j = 0; j < favSet.size(); j++) {
                if (i != j) {
                    Set<String> other = favSet.get(j);
                    if (other.containsAll(base)) {
                        good = false;
                        break;
                    }
                
}
            }
            if (good) {
                ret.add(i);
            }
        }
        return ret;
    }
}
