class Solution {
    public List<Integer> peopleIndexes(List<List<String>> fc) {
        int n = fc.size();
        List<Integer> rtn = new ArrayList<>();
        List<Set<String>> fcs = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            Set<String> s = new HashSet<>();
            for (int j = 0; j < fc.get(i).size(); j++) {
                s.add(fc.get(i).get(j));
            }
            fcs.add(s);
        }
        for (int i = 0; i < n; i++) {
            boolean s = true;
            for (int j = 0; j < n; j++) {
                if (i == j) {
                    continue;
                }
                if (isSub(fcs, i, j)) {
                    s = false;
                    break;
                }
            }
            if (s) {
                rtn.add(i);
            }
        }
        return rtn;
    }
    
    private boolean isSub(List<Set<String>> fcs, int i, int j) {
        boolean s = true;
        Set<String> si = fcs.get(i);
        Set<String> sj = fcs.get(j);
        for (String sti : si) {
            if (!sj.contains(sti)) {
                s = false;
                break;
            }
        }
        return s;
    }
}
