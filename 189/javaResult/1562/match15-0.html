class Solution {
    public List<Integer> peopleIndexes(List<List<String>> favoriteCompanies) {
        List<Integer> res = new ArrayList<>();
        List<Set<String>> fcs = new ArrayList<>();
        for (List<String> fc : favoriteCompanies) {
            fcs.add(new HashSet<>(fc));
        }
        
        for (int i = 0; i < fcs.size(); i++) {
            Set<String> cur = fcs.get(i);
            boolean isSubset = false;
            for (int j = 0; j < fcs.size(); j++) {
                if (i == j) continue;
                if (isSubset(cur, fcs.get(j))) {
                    isSubset = true;
                    break;
                }
            }
            if (!isSubset) res.add(i);
        }
        
        return res;
    }
    
    private boolean isSubset(Set<String> l1, Set<String> l2) {
        if (l1.size() > l2.size()) return false;
        for (String s : l1) {
            if (!l2.contains(s)) return false;
        }
        
        return true;
    }
}
