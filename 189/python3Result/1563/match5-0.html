from typing import List

class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        import math
        def dist(p1, p2):
            x1, y1 = p1
            x2, y2 = p2
            return math.sqrt((x1 - x2)**2 + (y1 - y2)**2)

        def get_circle(p1, p2):
            x1, y1 = p1
            x2, y2 = p2
            midx = (x1 + x2) / 2
            midy = (y1 + y2) / 2
            angle = math.atan2(x1 - x2, y2 - y1)
            d = math.sqrt(r*r - math.pow(dist(p1, [midx, midy]), 2))
            return (midx + d*math.cos(angle), midy + d*math.sin(angle))
        
        eps = 1e-8
        ans = 1
        n = len(points)
        for i in range(n):
            for j in range(i+1, n):
                if dist(points[i], points[j]) > 2.0 * r:
                    continue
                center = get_circle(points[i], points[j])
                cnt = 0
                for k in range(n):
                    if dist(center, points[k]) <= r+eps:
                        cnt += 1
                #print(center, cnt)
                ans = max(ans, cnt)
        return ans
