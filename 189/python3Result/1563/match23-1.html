import math

class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        
        def getDist(p1, p2):
            return ((p1
[0]-p2[0])**2 + (p1[1]-p2[1])**2)**0.5
        
        def getCenter(p1, p2):
            mid = [(p1[0]+p2[0]) / 2, (p1[1]+p2[1]) / 2]
            dist = (r**2 - getDist(p1, mid)**2)**0.5
            angle = math.atan2(p1[0]-p2[0],p2[1]-p1[1])
            return [mid[0]+dist*math.cos(angle), mid[1]+dist*math.sin(angle)]
        
        n = len(points)
        res = 1
        eps = 1e-8
        for i in range(n):
            for j in range
(i+1, n):
                if getDist(points[i], points[j]) > 2 * r:
                    continue
                center = getCenter(points[i], points[j])
                cnt = 0
                for k in range(n):
                    
if getDist(center, points[k]) < r + eps:
                        cnt += 1
                res = max(res, cnt)
            
        return res
