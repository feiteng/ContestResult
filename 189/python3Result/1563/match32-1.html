import math
from typing import List



class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        r2 = r ** 2
        def dist2(point1, point2):
            return (point1[0] - point2[0]) ** 2 + (point1[1] - point2[1]) ** 2

        def inCircle(center, point):
            return dist2(center, point) <= r2

        def getCenter(point1, point2):
            if point2[0] < point1[0]:
                return getCenter(point2, point1)

            d = (r2 - dist2(point1, point2) / 4) ** 0.5
            mid = [(point1[0] + point2[0]) / 2, (point1[1] + point2[1]) / 2]

            if point2[0] == point1[0]:
                return [mid[0] + d, mid[1]]

            angle = math.atan((point2[1] - point1[1]) / (point2[0] - point1[0]))
            return [mid[0] + d * math.sin(angle), mid[1] - d * math.cos(angle)]

        n = len(points)
        if n == 1:
            return 1
        ans 
= 1
        for i in range(n - 1):
            for j in range(i + 1, n):
                if dist2(points[i], points[j]) > 4 * r2:
                    continue
                center = getCenter(points[i], points[j])
                step = 0
                for k in range(n):
                    if k == i or k == j:
                        step += 1
                    elif dist2
(center, points[k]) - 0.00000001 <= r2:
                        step += 1
                ans = max(ans, step)
        return ans
