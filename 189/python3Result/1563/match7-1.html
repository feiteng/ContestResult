import numpy as np
class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        def helper(i):
            angles = []
            for j in range(n):
                if i == j or dist[(i, j)] > 2 * r: continue
                B = np.arccos(dist[(i, j)] / (2 * r))
                dx 
= points[j][0] - points[i][0]
                dy = points[j][1] - points[i][1]
                A = np.angle(dx + 1j * dy)
                alpha, beta = A - B, A + B
                angles.append((alpha, True))
                angles.append((
beta, False))
            angles.sort(key=lambda x: (x[0], not x[1]))
            curr, max_cnt = 1, 1
            for _, b in angles:
                curr += 1 if b else -1
                max_cnt = max(max_cnt, curr)
            return max_cnt
        
        dist, n = {}, len(points)
        for i, (x1, y1) in enumerate(points):
            for j, (x2, y2) in enumerate(points[i + 1:], i + 1):
                dist[(j, i)] = dist[(i, j)] = np.sqrt((x2 - x1) **2 + (y2 - y1) ** 2)
        return max(helper(i) for i in range(n))
