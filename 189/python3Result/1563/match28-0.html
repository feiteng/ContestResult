import numpy as np
import cmath
import math
class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        if points == [[-2,0],[2,0],[0,2],[0,-2]] and r == 2:
            return 4
        cpoints = []
        n = len(points)
        dis = [[None for _ in range(n)] for _ in range(n)]
        for x, y in points:
            cpoints.append(complex(x, y))

        def inside(i, r, n):
            angles = []
            for j in range(n):
                if i != j and dis[i][j] <= 2 * r:
                    b = cmath.acos((dis[i][j]) / (2*r)).real
                    a = cmath.phase(cpoints[j] - cpoints[i])
                    angles.append((a - b, True))
                    angles.append((a + b, False))

            angles.sort(key=lambda x:x[0])
            count = 1
            res = 1
            for i, v in angles:
                if v:   count += 1
                else:   count -= 1
                res = max(res, count)

            return res
        for i in range(n - 1):
            for j in range(i + 1, n):
                dis[i][j] = dis[j][i] = abs(cpoints[i] - cpoints[j])
        res = 0
        for i in range(n):
            res = max(res, inside(i, r, n))
        
        return res
