class Solution:
    def numPoints(self, p: List[List[int]], r: int) -> int:
        def dist(x, y):
            return math.sqrt( (x[0]-y[0])**2 + (x[1] - y[1])**2)
        
        def getc(x, y):
            midx = (x[0] + y[0]) / 2.0
            midy = (x[1] + y[1]) / 2.0
            ang = math.atan2(x[0]-y[0], y[1]-x[1])
            d = math.sqrt( r*r - ((x[0]-midx)*(x[0]-midx)) - ((x[1]-midy)*(x[1]-midy)) )

            return [midx+d*math.cos(ang), midy+d*math.sin(ang)]
        
        n = len(p)
        ans 
= 1
        for i in range(n):
            for j in range(i+1, n):
                if dist(p[i], p[j]) > 2.0*r:
                    continue
                c = getc(p[i], p[j])
                cnt = 0
                for k in range(n):
                    if dist(c, p[k]) < 1.0*r + 1e-6:
                        cnt += 1
                ans = max(ans, cnt)
        return ans
