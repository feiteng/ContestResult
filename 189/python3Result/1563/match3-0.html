class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        def dist(p1, p2):
            return math.sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)
        
        def center(p1, p2):
            x1, y1 = p1
            x2, y2 = p2
            mx = (x1+x2) / 2.0
            my = (y1+y2) / 2.0
            ang = math.atan2(x1-x2, y2-y1)
            d = math.sqrt(r**2-(x1-mx)**2-(y1-my)**2)
            return [mx+d*math.cos(ang), my+d*math.sin(ang)]
            
        p = points
        n = len(p)
        
        epsilon = 1e-9
        
        ans = 1
        for i in range(n):
            for j in range(i+1, n):
                if dist(p[i], p[j]) > 2.0*r:
                    continue
                c = center(p[i], p[j])
                cnt = 0
                for k in range(n):
                    if dist(c, p[k]) - r < epsilon:
                        cnt += 1
                ans = max(ans, cnt)
        return ans
