class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        def solver(x1, y1, x2, y2):
            if x2 == x1:
                if abs(y2 - y1) > 2 * r:
                    return []
                elif abs(y2 - y1) == 2 * r:
                    return [[x1, (y1+y2)/2]]
                d = abs(y1-y2) / 2
                return [[x1- (r**2 - d**2)**0.5, (y1+y2)/2], [x1+ (r**2 - d**2)**0.5, (y1+y2)/2]]
            c1 = (x2**2 - x1**2 + y2**2 - y1**2) / (2 * (x2-x1))
            c2 = (y2 - y1) / (x2 - x1)
            A = 1.0 + c2 ** 2
            B = 2 * (x1 - c1) * c2 - 2 * y1
            C = (x1-c1) ** 2 + y1 ** 2 - r**2
            y1 = (-B + (B*B - 4 * A * C) ** 0.5) / (2*A)
            x1 = c1 - c2 * y1
            y2 = (-B - (B*B - 4 * A * C) ** 0.5) / (2*A)
            x2 = c1 - c2 * y2
            return [[x1, y1], [x2, y2]]
        
        ans = 1
        for i in range(len(points)):
            for j in range(i+1, len(points)):
                circle = solver(points[i][0], points[i][1], points[j][0], points[j][1])
                for c1, c2 in circle:
                    if isinstance(c1,complex): continue
                    cur = 0
                    for x, y in points:
                        if (c1-x)**2 + (c2-y)**2 <= r ** 2:
                            cur += 1
                    ans = max(cur, ans)
        return ans
