from math import sqrt,atan2,cos,sin
class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        def dist(p1,p2):
            return sqrt((p1[0]-p2[0])*(p1[0]-p2[0])+(p1[1]-p2[1])*(p1[1]-p2[1]))
        def center1(p1,p2):
            mid = ((p1[0]+p2[0])/2,(p1[1]+p2[1])/2)
            d = sqrt(r*r - (dist(p1,p2)/2)**2)
            ang = atan2(p2[1]-p1[1],p2[0]-p1[0])
            return (mid[0]+d*sin(ang),mid[1]-d*cos(ang))
        def center2(p1,p2):
            mid = ((p1[0]+p2[0])/2,(p1[1]+p2[1])/2)
            d = sqrt(r*r - (dist(p1,p2)/2)**2)
            ang = atan2(p2[1]-p1[1],p2[0]-p1[0])
            return (mid[0]-d*sin(ang),mid[1]+d*cos(ang))
        n =len(points)
        ans = 1
        for i in range(n):
            for j in range(i+1,n):
                if dist(points[i],points[j])>2.0*r:
                    continue
                c1 = center1(points[i],points[j])
                c2 = center2(points[i],points[j])
                # print(c1,c2)
                tmp=2
                for k in range(n):
                    if k==i or k==j:
                        continue
                    if dist(c1,points[k])<=r:
                        tmp+=1
                ans = max(ans,tmp)
                tmp=2
                for k in range(n):
                    if k==i or k==j:
                        continue
                    if dist(c2,points[k])<=r:
                        tmp+=1
                ans = max(ans,tmp)
        return ans
