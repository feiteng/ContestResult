class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        def dist(p1, p2):
            return ((p1[0] - p2[0]) ** 2 + (p1[1] - p2[1]) ** 2) ** 0.5
        
        def getcenter(p1, p2):
            mid = (p1[0] + p2[0]) / 2, (p1[1] + p2[1]) / 2
            d = (r * r - dist(p1, mid) ** 2) ** 0.5
            angle = math.atan2(p1[1] - p2[1], p1[0] - p2[0])
            #print(mid, d, angle)
            return mid[0] - d * math.sin(angle), mid[1] + d * math.cos(angle)
        
        n = len(points)
        res = 0
        eps = 1e-4
        #print(getcenter(points[0], points[1]))
        for i, p in enumerate(points):
            for j in range(i + 1, n):
                if dist(p, points[j]) > 2.0 * r:
                    continue
                cnt = 0
                center = getcenter(p, points[j])
                for k in range(n):
                    if dist(center, points[k]) < 1.0 * r + eps:
                        cnt += 1
                    #else:
                    #    print(p, points[j], center, points[k])
                res = max(res, cnt)
        return max(1, res)
