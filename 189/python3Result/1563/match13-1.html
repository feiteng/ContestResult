class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        
        def dist(p1, p2):
            return sqrt((p1
[0]-p2[0])**2 + (p1[1]-p2[1])**2);

        def getCircle(p1, p2):
            mid =[(p1[0] + p2[0]) / 2, (p1[1] + p2[1]) / 2]
            angle = atan2(p1[0]-p2[0], p2[1] - p1[1])
            d = sqrt(abs(r**2 - dist(p1, mid)**2))
            return [mid[0] + d*cos(angle), mid[1] + d*sin(angle)];
        
        ans = 1;
        for i in range(len(points)):
            for j in range(len
(points)):
                if dist(points[i], points[j]) > r**2:
                    continue
                center = getCircle(points[i], points[j])
                cnt = 0
                for k in range(len(points)):
                    
if dist(center, points[k]) < r + 1e-8:
                        cnt += 1
                ans = max(ans, cnt)
        
        return ans
