class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        def dist(p1, p2):
            return math.sqrt((p1
[0]-p2[0])**2 + (p1[1]-p2[1])**2)
        def get_center(p1, p2):
            mid = [(p1[0]+p2[0])/2, (p1[1]+p2[1])/2 ]
            angle = math.atan2(p1[0]-p2[0], p2[1]-p1[1])
            d = math.sqrt(r*r-dist(p1, mid) ** 2)
            return [mid[0]+d*math.cos(angle), mid[1]+d*math.sin(angle)]
        res = 1
        n = len(points)
        for i in range(n):
            for j in range(i+1, n):
                if (dist
(points[i], points[j]) > 2.0 * r):
                    continue
                center = get_center(points[i], points[j])
                cnt = 0
                for k in range(n):
                    if (dist
(center, points[k]) < 1.0*r + 1e-8):
                        cnt += 1
                res = max(res, cnt)
        
        return res
