class Solution:

    def numPoints(self, points: List[List[int]], r: int) -> int:
        def dist2(p1, p2):
            return (p1[0]-p2[0])**2 + (p1[1]-p2[1])**2

        def get_center1(p1, p2):
            angle = math.atan2(p2[1]-p1[1], p2[0]-p1[0])
            mid = ((p1[0]+p2[0])/2, (p1[1]+p2[1])/2)
            d = math.sqrt(r**2 - dist2(mid, p1))
            return (mid[0] - d*math.sin(angle), mid[1] + d*math.cos(angle))
        
        def get_center2(p1, p2):
            angle = math.atan2(p2[1]-p1[1], p2[0]-p1[0])
            mid = ((p1
[0]+p2[0])/2, (p1[1]+p2[1])/2)
            d = math.sqrt(r**2 - dist2(mid, p1))
            return (mid[0] + d*math.sin(angle), mid[1] - d*math.cos(angle))

        ans = 1
        for i in range(len(points)):
            for j in range(i+1, len
(points)):
                if dist2(points[i], points[j]) > (2*r)**2:
                    continue
                center = get_center1(points[i], points[j])
                # print(center)
                cnt = 0
                for k in range(len(points)):
                    if dist2(points[k], center) <= r**2+1e-6:
                        cnt += 1
                ans = max(cnt, ans)
                center = get_center2(points[i], points[j])
                # print(center)
                cnt = 0
                for k in range(len(points)):
                    
if dist2(points[k], center) <= r**2+1e-6:
                        cnt += 1
                ans = max(cnt, ans)
        return ans
