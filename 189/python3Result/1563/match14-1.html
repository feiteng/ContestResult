class Solution:
    def numPoints(self, p: List[List[int]], r: int) -> int:
        res = 1
        
        
def dis(pi, pj):
            return sqrt((pi[0] - pj[0]) ** 2 + (pi[1] - pj[1]) ** 2)
        
        def center(pi, pj):
            mid = ((pi[0] + pj[0]) / 2, (pi[1] + pj[1]) / 2)
            
            angle = atan2(pi[0]-pj[0], pj[1]-pi[1])
            
            # print(pi, pj)
            d = sqrt(r ** 2 - dis(pi, mid) ** 2)
            
            return (mid[0] + d * cos(angle), mid[1] + d * sin(angle))
        
        
        n = len(p)
        for i in range(n):
            for j in range
(i+1, n):
                if dis(p[i], p[j]) > r * 2:
                    continue
                
                c = center(p[i], p[j]) 
                cur = 0
                for k in range(n):
                    
if dis(c, p[k]) <= r + 10 ** (-7):
                        cur += 1
                
                # print(cur)
                res = max(res, cur)
        return res
