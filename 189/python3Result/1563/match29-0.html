import math
class Solution:
    def numPoints(self, points: List[List[int]], r: int) -> int:
        ans = 1
        eps = 1e-8
        def dist(p1, p2):
            return math.sqrt((p1[0] - p2[0]) * (p1[0] - p2[0]) + (p1[1] - p2[1]) * (p1[1] - p2[1]))
        
        def get_circle(p1, p2):
            x = (p1[0] + p2[0]) / 2
            y = (p1[1] + p2[1]) / 2
            angle = atan2(p1[0] - p2[0], p2[1] - p1[1])
            d = math.sqrt(r * r - math.pow(dist(p1, [x, y]), 2))
            return [x + d*cos(angle), y + d * sin(angle)]
        
        n = len(points)
        for i in range(n):
            for j in range(i + 1, n):
                if dist(points[i], points[j]) > 2.0 * r + eps:
                    continue
                center = get_circle(points[i], points[j])
                cnt = 0
                for k in range(n):
                    if dist(center, points[k]) < 1.0 * r + eps:
                        cnt += 1
                ans = max(ans, cnt)
        return ans
