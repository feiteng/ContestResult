class Solution:
    def numPoints(self, p: List[List[int]], r: int) -> int:
        if p:
            res = 1
        else:
            res = 0
        def cal_circle(x1, y1, x2, y2):
            if x2 - x1 == 0:
                if abs(y2 - y1) > 2 * r:
                    return []
                elif abs(y2 - y1) == 2 * r:
                    return [[x1, (y1+y2)/2]]
                else:
                    d = abs(y1-y2) / 2
                    return [[x1- (r**2 - d**2)**0.5, (y1+y2)/2], [x1
+ (r**2 - d**2)**0.5, (y1+y2)/2]]
            
            c1 = (x2**2 - x1**2 + y2**2 - y1**2) / (2 * (x2-x1))
            c2 = (y2 - y1) / (x2 - x1)
            
            A = 1.0 + c2 ** 2
            B = 2 * (x1 - c1) * c2 - 2 * y1
            C = (x1-c1) ** 2 + y1 ** 2 - r**2
            
            y01 = (-B + (B*B - 4 * A * C) ** 0.5) / (2*A)
            x01 = c1 - c2 * y01
            
            y02 = (-B - (B*B - 4 * A * C) ** 0.5) / (2*A)
            x02 = c1 - c2 * y02
            # print(x1, y1, x2, y2, x01, y01, x02, y02)
            return [[x01, y01], [x02, y02]]
        
        n = len(p)
        for i in range(n):
            for j in range
(i+1, n):
                circle = cal_circle(p[i][0], p[i][1], p[j][0], p[j][1])
                # print(circle)
                for c1, c2 in circle:
                    if isinstance(c1,complex): 
                        continue
                    ans = 0
                    for p1, p2 in p:
                        if (c1-p1)**2 + (c2-p2)**2 <= r ** 2:
                            ans += 1
                    # print(ans, c1, c2, p[i][0], p[i][1], p[j][0], p[j][1])
                    res = max(res, ans)
        print(res)
        return res
