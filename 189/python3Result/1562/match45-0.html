class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        FC = [set(s) for s in favoriteCompanies]
        # print(FC)
        ans = []
        for i in range(len(FC)):
            flag = True
            for j in range(len(FC)):
                if i == j: continue
                if FC[i].issubset(FC[j]):
                    flag = False
                    break
            if flag:
                ans.append(i)
        return ans
