class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        fav2 = [set(x) for x in favoriteCompanies]
        out = []
        for a in range(len(fav2)):
            valid = True
            for b in range(len(fav2)):
                if a != b:
                    if fav2[a].issubset(fav2[b]):
                        valid = False
                        break
                        
            if valid:
                out.append(a)
        return out
