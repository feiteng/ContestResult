class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        fav=[]
        for comps in favoriteCompanies:
            fav.append(set(comps))
        ret=[]
        for i in range(len(fav)):
            dist=True
            for j in range(len(fav)):
                if i==j:
                    continue
                sub=True
                for comp in fav[i]:
                    if comp not in fav[j]:
                        sub=False
                        break
                if sub:
                    dist=False
                    break
            if dist:
                ret.append(i)
        return ret
