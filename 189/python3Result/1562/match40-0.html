class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        res = []
        #print(set(["google","faceboqok"]) <= set(["leetcode","google","facebook"]))
        for i in range(len(favoriteCompanies)):
            subset = False
            for j in range(len(favoriteCompanies)):
                if i!=j and set(favoriteCompanies[i]) <= set(favoriteCompanies[j]):
                    subset = True
                    break
            if not subset:
                res.append(i)
                
        return res
