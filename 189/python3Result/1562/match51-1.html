class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        fa = [set(i) for i in favoriteCompanies]
        l = len(fa)
        res = []
        for i in range(l):
            flag = 1
            for j in range(l):
                if i == j:
                    continue
                if fa[i].issubset(fa[j]):
                    flag 
= 0
                    break
            if flag:
                res.append(i)
        return res
