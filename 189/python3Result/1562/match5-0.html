class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        sl = [set(l) for l in favoriteCompanies]
        ans = []
        for i in range(len(sl)):
            a = 0
            for j in range(len(sl)):
                if j != i:
                    if sl[i].issubset(sl[j]):
                    
                        a+=1
                        break
            if a == 0:
                ans.append(i)
        return ans
