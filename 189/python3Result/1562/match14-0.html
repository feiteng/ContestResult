class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        sets = [set(c) for c in favoriteCompanies]
        res = []
        for i, s1 in enumerate(sets):
            for j, s2 in enumerate(sets):
                if i == j:
                    continue
                if s1.issubset(s2):
                    break
            else:
                res.append(i)
        return res
