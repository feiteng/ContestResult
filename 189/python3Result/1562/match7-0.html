class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        temp = []
        for companies in favoriteCompanies:
            temp.append(set(companies))
        res = []
        for i in range(len(favoriteCompanies)):
            flag = False
            for j in range(len(favoriteCompanies)):
                if i == j:
                    continue
                sub = True
                for cur in temp[i]:
                    if cur not in temp[j]:
                        sub = False
                        break
                if sub:
                    flag = True
                    break
            if not flag:
                res.append(i)
        return res
