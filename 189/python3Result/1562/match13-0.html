class Solution:
    def peopleIndexes(self, sets: List[List[str]]) -> List[int]:
        xs = [set(x) for x in sets]
        ans = []
        for i, x in enumerate(xs):
            for j, y in enumerate(xs):
                if i == j: continue
                if y.intersection(x) == x:
                    break
            else:
                ans.append(i)
        return ans
