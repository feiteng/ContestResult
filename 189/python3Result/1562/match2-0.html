class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        fcsets = [set(fc) for fc in favoriteCompanies]
        ans = []
        for i in range(len(fcsets)):
            subset = False
            for j in range(len(fcsets)):
                if i == j: continue
                if fcsets[i] < fcsets[j]:
                    subset = True
                    break
            if not subset:
                ans.append(i)
        return ans
