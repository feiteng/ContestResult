class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        favoriteSet = [set(favor) for favor in favoriteCompanies]
        ans = []
        for i in range(len(favoriteSet)):
            subset = False
            for j in range(len(favoriteSet)):
                if i != j and favoriteSet[i].issubset(favoriteSet[j]):
                    subset = True
                    break
            
            if not subset:
                ans.append(i)
        
        return ans
