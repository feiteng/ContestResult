class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        
        favoriteCompanies = [set(lst) for lst in favoriteCompanies]
        
        n = len(favoriteCompanies)
        
        ans = []
        
        for i in range(n):
            is_subst = False
            for j in range(n):
                if i == j:
                    continue
                if favoriteCompanies[i] - favoriteCompanies[j] == set():
                    is_subst = True
                    break
            if is_subst == False:
                ans.append(i)
        
        return ans
