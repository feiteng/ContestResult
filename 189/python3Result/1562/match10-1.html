class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        favoriteCompaniesSets = [set(c) for c in favoriteCompanies]
        ans = []
        for i in range(len(favoriteCompanies)):
            is_subset = False
            for j in range(len(favoriteCompanies)):
                if i != j and  favoriteCompaniesSets[i].issubset(favoriteCompaniesSets[j]):
                    is_subset = True
                    break
            if not is_subset:
                ans.append(i)
        return ans
