class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        fav_companies = [set(c) for c in favoriteCompanies]
        results = []
        for i in range(len(fav_companies)):
            flag = True
            for j in range(len(fav_companies)):
                if i != j and fav_companies[i].issubset(fav_companies[j]):
                    flag = False
                    break
            if flag:
                results.append(i)
        return results
