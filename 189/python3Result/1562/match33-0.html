class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        sets = [set(cs) for cs in favoriteCompanies]
        
        result = []
        for i in range(len(sets)):
            hasSubset = False
            for j in range(len(sets)):
                if i == j:
                    continue
                if sets[i].issubset(sets[j]):
                    hasSubset = True
                    break
            if not hasSubset:
                result.append(i)
        return result
