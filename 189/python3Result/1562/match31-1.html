class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        res = []
        N = len(favoriteCompanies)
        for i in range(N):
            flag = True
            for j in range(N):
                if i == j: continue
                
if set(favoriteCompanies[i]) < set(favoriteCompanies[j]):
                    flag = False
                    break
                    
            if flag: res.append(i)
        return res
