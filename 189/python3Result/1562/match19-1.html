class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        ret = []
        l = len(favoriteCompanies)
        for i in range(l):
            flag=1
            for j in range(l):
                if set(favoriteCompanies[i]) < set(favoriteCompanies[j]):
                    flag=0
                    break
            if flag==1:
                ret.append(i)
        return ret
