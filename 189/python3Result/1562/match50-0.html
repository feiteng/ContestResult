class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        ss = [set(x) for x in favoriteCompanies]
        n = len(ss)
        ans = []
        for i in range(n):
            ok = 1
            for j in range(n):
                if i==j:
                    continue
                if ss[i].issubset(ss[j]):
                    ok=0
                    break
            if ok:
                ans.append(i)
        return ans
