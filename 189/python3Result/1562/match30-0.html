class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        res = []
        n = len(favoriteCompanies)
        for i in range(n):
            flag = False
            for j in range(n):
                if i != j and set(favoriteCompanies[i]) <= set(favoriteCompanies[j]):
                    flag = True
                    break
            if not flag:
                res.append(i)
        return res
