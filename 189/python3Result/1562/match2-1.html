class Solution:
    def peopleIndexes(self, f: List[List[str]]) -> List[int]:
        f = [set(x) for x in f]
        res = []
        for i in range(len(f)):
            success = True
            for j in range(len(f)):
                if i == j:
                    continue
                if f[i] <= f[j]:
                    success = False
                    break
            if success:
                res.append(i)
        return res
