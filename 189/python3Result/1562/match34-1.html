class Solution:
    def peopleIndexes(self, favoriteCompanies: List[List[str]]) -> List[int]:
        comps = [set(x) for x in favoriteCompanies]
        ans = []
        for i in range(len(comps)):
            good = True
            for j in range(len(comps)):
                if i == j:
                    continue
                if comps[i] <= comps[j]:
                    good 
= False
                    break
            if good:
                ans.append(i)
        return ans
