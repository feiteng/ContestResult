class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        vector<vector<int>> fc;
        for (int i = 0; i < favoriteCompanies.size(); i++) {
            vector<int> tmp;
            for (int j = 0; j < favoriteCompanies[i].size(); j++) {
                int val = 0;
                for (int k = 0; k < favoriteCompanies[i][j].size(); k++) {
                    val = val * 31 + favoriteCompanies[i][j][k] - 'a' + 1;
                    val %= 1000007;
                }
                tmp.push_back(val);
            }
            fc.push_back(tmp);
        }
        map<int, int> mp[600];
        for (int i = 0; i < fc.size(); i++) {
            for (int j = 0; j < fc[i].size(); j++) {
                mp[i][fc[i][j]] = 1;
            }
        }
        vector<int> ans;
        for (int i = 0; i < fc.size(); i++) {
            bool skip = false;
            for (int j = 0; j < fc.size(); j++) {
                if (i == j) continue;
                bool ok = true;
                for (int k = 0; k < fc[i].size(); k++) {
                    if (mp[j][fc[i][k]] != 1) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    skip = true;
                    break;
                }
            }
            if (!skip) {
                ans.push_back(i);
            }
        }
        return ans;
    }
};
