class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& f) {
        vector<int> ret;
        unordered_map<string, int> m;
        int ccnt = 0;
        for (int i = 0; i < f.size(); i++){
            sort(f[i].begin(), f[i].end());
            for (string s : f[i]){
                if (m.find(s) == m.end()){
                    m[s] = ccnt;
                    ccnt++;
                }
            }
        }
        map<int, unordered_set<int>> m2;

        for (int i = 0; i < f.size(); i++){
            for (int j = 0; j < f[i].size(); j++){
                m2[i].insert(m[f[i][j]]);
            }
        }

        for (int i = 0; i < f.size(); i++){
            bool isSub = false;
            
            for (int j = 0; j < f.size(); j++){
                if (i == j) 
                    continue;
                int k = 0;
                for (k = 0; k < f[i].size(); k++){
                    if (m2[j].find(m[f[i][k]]) == m2[j].end())

                        break;
                }
                if (k == f[i].size()){
                    isSub = true;
                    break;
                }
            }
            if (!isSub)
                ret.push_back(i);
        }
        return ret;
    }
};
