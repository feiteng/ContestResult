class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& fc) {
        for(auto &it: fc) {
            sort(it.begin(), it.end());
        }

        int np = fc.size();
        vector<int> ret;
        for(int i = 0; i < np; ++i) {
            bool ok = true;
            for(int j = 0; j < np; ++j) {
                if(i == j) continue;
                // check fc[i] and fc[j]
                if(includes(fc[j].begin(), fc[j].end(), fc[i].begin(), fc[i].end())) {
                    ok = false;
                    break;
                }
            }
            if(ok) {
                ret.push_back(i);
            }
        }
        return ret;
    }
};
