class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& f) {
        unordered_map<string, int> mp;
        int cur=0;
        for(auto & v:f) {
            for(auto & s:v) {
                if(mp.count(s)==0){
                    mp[s]=cur++;
                }
            }
        }
        
        vector<unordered_set<int>> a;
        int n=f.size();
        for(int i=0;i<n;i++) {
            a.push_back(unordered_set<int>()); 

            for(auto & s:f[i]) {
                a.back().insert(mp[s]);
            }
        }
        vector<int> res;
        for(int i=0;i<n;i++) {
            int tmp=1;
            for(int j=0;j<n;j++){
                if(j==i)continue;
                int ok=1;
                for(auto & s:a[i]) {
                    if(a[j].count(s)==0) {
                        ok=0;
                        break;
                    }
                }
                if(ok){
                    tmp=0;
                    break;
                }
            }
            if(tmp) res.push_back(i);
        }
        return res;
    }
};
