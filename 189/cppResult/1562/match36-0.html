class Solution {
    bool wasContained(const vector<string>& a, const vector<string>& b) {
        if (a.size() >= b.size()) return false;
        int i = 0, j = 0;
        while (i < a.size() && j < b.size()) {
            if (a[i] == b[j]) {
                ++i;
                ++j;
            } else {
                ++j;
            }
        }
        return i == a.size();
    }
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        for (auto& v : favoriteCompanies) {
            sort(v.begin(), v.end());
        }
        const int n = favoriteCompanies.size();
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            bool remove = false;
            for (int j = 0; j < n; ++j) {
                if (i == j) continue;
                if (wasContained(favoriteCompanies[i], favoriteCompanies[j])) {
                    remove = true;
                    break;
                }
            }
            if (!remove) {
                ans.push_back(i);
            }
        }
        return ans;
    }
};
