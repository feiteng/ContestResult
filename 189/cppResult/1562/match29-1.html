class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();
        vector<unordered_set<int>> fc(n, unordered_set<int>());
        
        int cnt = 0;

        unordered_map<string, int> cmap;
        for(int i = 0; i < n; ++i) {
            for(string s: favoriteCompanies[i]) {
                if(cmap.find(s) == cmap.end()) {
                    cmap[s] = cnt;
                    cnt += 1;
                }
                fc[i].insert(cmap[s]);
            }
        }
        
        vector<int> answer;

        for(int i = 0; i < n; ++i) {
            bool f = false;
            for(int j = 0; j < n && !f; ++j) {
                if(j == i) { continue; }

                bool k = true;
                for(int x: fc[i]) {
                    if(fc[j].find(x) == fc[j].end()) {
                        k = false;
                        break;
                    }
                }
                if(k) { f = true; }
            }
            if(!f) { answer.push_back(i); }
        }
        return answer;
    }
};
