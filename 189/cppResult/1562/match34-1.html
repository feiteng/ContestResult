class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();
        
        vector< unordered_set<string> > dicts(n);

        for (int i = 0; i < n; ++i)
        {
            for (int t = 0; t < favoriteCompanies[i].size(); ++t)
            {
                dicts[i].insert( favoriteCompanies[i][t] );
            }            
        }
        
        vector<int> res;
        for (int i = 0; i < n; ++i)
        {
            bool flag = true;
            for (int t = 0; t < n; ++t)
            {
                if (i == t) continue;
                
                int cnt = 0;
                for (auto item : dicts[i])
                {
                    if (dicts[t].count( item ) != 0) ++cnt;
                    else break;

                }
                
                if (cnt == dicts[i].size()) 
                {
                    flag = false;
                    break;
                }
            }
            
            if (flag == true) res.push_back(i);
        }
        
        return res;
    }
};
