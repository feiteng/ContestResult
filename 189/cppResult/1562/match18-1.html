class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int index = 0, n = static_cast<int>(favoriteCompanies.size());
        unordered_map<string, int> indices;
        vector<unordered_set<int>> codes(n);

        for (int i = 0; i < n; ++i) {
            for (auto& company : favoriteCompanies[i]) {
                if (indices.find(company) == indices.end()) {
                    indices[company] = index++;
                }
                codes[i].insert(indices[company]);
            }
        }
        vector<int> results;
        for (int i = 0; i < n; ++i) {
            bool valid = true;
            for (int j = 0; j < n; ++j) {
                if (i != j) {
                    bool notFound = false;

                    for (auto code : codes[i]) {
                        if (codes[j].find(code) == codes[j].end()) {
                            notFound = true;
                            break;
                        }
                    }
                    if (!notFound) {
                        valid = false;
                        break;
                    }
                }
            }
            if (valid) {
                results.push_back(i);
            }
        }
        return results;
    }
};
