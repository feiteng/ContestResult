const int MAXN = 105;

bool cmp(vector<string> a, vector<string> b)
{
    return a.size() < b.size();
}

class Solution {
public:
    
    
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        
        
        // sort(favoriteCompanies.begin(), favoriteCompanies.end(), cmp);
        
        for(auto& t : favoriteCompanies) sort(t.begin(), t.end());
        
        
        unordered_set<string> G[MAXN];
        int n = favoriteCompanies.size();
        
        for(int i = 0;i < n; ++i) G[i].clear();
        
        for(int i = 0;i < n; ++i)
        {
            for(string s : favoriteCompanies[i])
            {
                G[i].insert(s);
            }

        }
        
        vector<int> ans;
        
        for(int i = 0;i < n; ++i)
        {
            bool flag = true;
            for(int j = 0;j < n; ++j)
            {
                if(i == j) continue;
                bool f = true;
                
                for(string s : favoriteCompanies[i])
                {
                    if(!G[j].count(s))
                    {
                        f = false;
                        break;
                    }
                        
                }
                
                if(f)
                {
                    flag = false;
                    break;
                }
                
            }
            if(flag) ans.push_back(i);
        }
        
        return ans;
    }
};
