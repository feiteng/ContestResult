class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();
        map<string,int> m;
        int idx = 0;
        vector<bitset<505>> bs(n);
        for(int i=0;i<n;i++) {
            for(auto s:favoriteCompanies[i]) {
                if(m.find(s)==m.end()) {
                    m[s]=idx;
                    idx++;
                }
                bs[i].set(m[s]);
            }
        }
        vector<int> ans;
        bool flag;
        for(int i=0;i<n;i++) {
            flag = true;
            for(int j=0;j<n;j++) {
                if( (i!=j) && (bs[i] | bs[j]) == bs[j] ) {
                    flag = false;
                    break;
                }
            }
            if(flag) ans.push_back(i);
        }
        return ans;
    }
};
