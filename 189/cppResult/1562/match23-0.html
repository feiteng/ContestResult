class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) 
    {
        int n=favoriteCompanies.size();
        vector<int> res;
        for(int i=0;i<n;i++)
            sort(favoriteCompanies[i].begin(),favoriteCompanies[i].end());
        for(int i=0;i<n;i++)
        {
            int flag=0;
            for(int j=0;j<n;j++)
            {
                if(i==j)
                    continue;
                int ok=0;
                for(int p=0,q=0;p<favoriteCompanies[i].size();p++)
                {
                    while(q<favoriteCompanies[j].size()&&favoriteCompanies[i][p]!=favoriteCompanies[j][q])
                        q++;
                    if(q==favoriteCompanies[j].size())
                    {
                        ok=1;
                        break;
                    }
                }
                if(!ok)
                {
                    flag=1;
                    break;
                }
            }
            if(!flag)
                res.push_back(i);
        }
        return res;
    }
};
