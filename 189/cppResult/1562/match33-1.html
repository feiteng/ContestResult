class Solution {
public:
    unordered_set<string> st[150];
    vector<int> peopleIndexes(vector<vector<string>> &favoriteCompanies) {
        int perm[150] = {0};
        int fa[150] = {0};
        for (int i = 0; i < 150; i++) {
            st[i].clear();
            perm[i] = i;
        }
        int len = favoriteCompanies.size();
        sort(perm, perm + len, [&](const int &x, const int &y) {
            return favoriteCompanies[x].size() < favoriteCompanies[y].size();
        });
        for (int i = 0; i < len; i++) {
            fa[perm[i]] = i;
        }

        for (int i = 0; i < len; i++) {
            for (auto s : favoriteCompanies[perm[i]]) {
                st[perm[i]].insert(s);
            }
        }
        vector<int> ans;
        for (int i = 0; i < len; i++) {
            bool flag1 = true;
            for (int j = i + 1; j < len; j++) {
                // for (auto s : favoriteCompanies[perm[i]]) {
                //     cout << s << " ";
                // }
                // cout << " --> ";
                // for (auto s : favoriteCompanies[perm[j]]) {
                //     cout << s << " ";
                // }
                bool flag = true;

                for (auto s : favoriteCompanies[perm[i]]) {
                    if (!st[perm[j]].count(s)) {
                        flag = false;
                        break;
                    }
                }
                // cout << flag << endl;
                if (flag) {
                    flag1 = false;
                    break;
                }
            }
            if (flag1) {
                // for (auto s : favoriteCompanies[perm[i]]) {
                //     cout << s << " ";
                // }
                // cout << perm[i] << " " << fa[perm[i]] << " " << i;
                // cout << endl;
                ans.push_back(perm[i]);
            }
        }
        sort(ans.begin(), ans.end());
        return ans;
    }
};
