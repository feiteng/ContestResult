class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();
        unordered_set<string> companies[n];
        for(int i = 0; i < n; i++) {
            for(string c : favoriteCompanies[i]) {

                companies[i].insert(c);
            }
        }
        vector<int> ans;
        for(int i = 0; i < n; i++) {
            bool isSubset = false;
            for(int j = 0; j < n; j++) {
                if(i != j) {
                    bool isContained = true;

                    for(auto it : companies[i]) {
                        if(companies[j].find(it) == companies[j].end()) {
                            isContained = false;
                            break;
                        }
                    }
                    if(isContained) {
                        isSubset = true;
                        break;
                    }
                }
            }
            if(!isSubset) {
                ans.push_back(i);
            }
        }
        return ans;
    }
};
