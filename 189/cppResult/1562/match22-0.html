class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        std::vector<int> res;
        std::set<std::string> a[150];
        int n=favoriteCompanies.size();
        for (int i=0;i<n;i++){
            for (const auto &it:favoriteCompanies[i]){
                a[i].insert(it);
            }
        }
        for (int i=0;i<n;i++){
            int flaga=0;
            for (int j=0;j<n;j++){
                if (i==j) continue;
                int flag=1;
                for (const auto &it:a[i]){
                    if (!a[j].count(it)){
                        flag=0;
                        break;
                    }
                }
                if (flag){
                    flaga=1;
                    break;
                }
            }
            if (!flaga) res.push_back(i);
        }
        return res;
    }
};
