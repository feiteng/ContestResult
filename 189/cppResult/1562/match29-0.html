class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        vector<int> res;
        unordered_map<string, int> mp;
        int ind = 0, n = favoriteCompanies.size();
        vector<set<int>> c(n, set<int>());
        for (int i = 0; i < n; i++) {
            for (string& s : favoriteCompanies[i]) {
                if (mp.find(s) == mp.end()) {
                    mp[s] = ind++;
                }
                c[i].insert(mp[s]);
            }
        }
        for (int i = 0; i < n; i++) {
            bool sub = false;
            for (int j = 0; j < n; j++) {
                if (c[i].size() < c[j].size()) {
                    bool same = true;
                    for (int t : c[i]) {
                        if (c[j].find(t) == c[j].end()) {
                            same = false;
                            break;
                        }
                    }
                    if (same) {
                        sub = true;
                        break;
                    }
                }
            }
            if (!sub) res.push_back(i);
        }
        return res;
    }
};
