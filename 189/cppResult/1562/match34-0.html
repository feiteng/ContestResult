class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        map<string,int> p;int cnt=1;
        for(int i=0;i<favoriteCompanies.size();i++){
            for(int j=0;j<favoriteCompanies[i].size();j++){
                if(!p[favoriteCompanies[i][j]]) p[favoriteCompanies[i][j]]=cnt++;
            }
        }
        vector<int> s[favoriteCompanies.size()];
        for(int i=0;i<favoriteCompanies.size();i++){
            for(int j=0;j<favoriteCompanies[i].size();j++){
                s[i].push_back(p[favoriteCompanies[i][j]]);
            }
            sort(s[i].begin(),s[i].end());
        }
        
        vector<int> ans;
        for(int i=0;i<favoriteCompanies.size();i++){
            bool fg=true;
            for(int j=0;j<favoriteCompanies.size();j++){
                if(i==j) continue;
                int p=0,q=0;
                while(p<s[i].size()&&q<s[j].size()){
                    if(s[i][p]==s[j][q]){
                        p++;q++;
                    }else{
                        if(s[i][p]<s[j][q]){
                            break;
                        }
                        q++;
                    }
                }
                if(p==s[i].size()){
                    fg=false;break;
                }
            }
            if(fg) ans.push_back(i);
        }
        return ans;
    }
};
