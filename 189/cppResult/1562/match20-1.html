class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {

        unordered_map<string, int> hash; vector<vector<string>>& f = favoriteCompanies;
        vector<unordered_set<string>> fs;
        for (auto &ff : f) {
            unordered_set<string> se;
            for (auto &s : ff) {
                se.insert(s);
            }
            fs.push_back(se);
        }
        vector<int> ans;
        for (int i = 0; i < f.size(); ++i) {
            int flag1 = 1;
            for (int j = 0; j < f.size(); ++j) {
                if (i == j) continue;

                int flag = 0;   
                    for (auto &s_in_se: fs[i]) {
                        // cout << s_in_se << endl;
                        if (!fs[j].count(s_in_se)) {
                            flag = 1; break;
                        }
                    }
                if (!flag) {flag1 = 0; break;}
            }
            if (flag1) ans.push_back(i);
        }
        return ans;
    }
};
