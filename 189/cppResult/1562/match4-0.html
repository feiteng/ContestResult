class Solution {
 public:
  vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
    vector<unordered_set<string>> book;
    int n = favoriteCompanies.size();
    for (int i = 0; i < n; ++i) {
      book.emplace_back(favoriteCompanies[i].begin(), favoriteCompanies[i].end());
    }
    vector<int> ans;
    for (int i = 0; i < n; ++i) {
      bool is_ans = true;
      for (int j = 0; j < n; ++j) {
        if (i == j)
          continue;
        bool flag = true;
        for (string &str : favoriteCompanies[i]) {
          if (book[j].find(str) == book[j].end()) {
            flag = false;
            break;
          }
        }
        if (flag) {
          is_ans = false;
          break;
        }
      }
      if (is_ans) {
        ans.push_back(i);
      }
    }
    return ans;
  }
};
