class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();
        vector<unordered_set<string>> fc(n);
        for (int i = 0; i < n; ++i) {
            for (const auto& s : favoriteCompanies[i]) {
                fc[i].insert(s);
            }
        }
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            bool flag = true;
            for (int j = 0; j < n; ++j) {
                if (i == j) {
                    continue;
                }
                bool sub = true;
                for (const auto& s : fc[i]) {
                    if (fc[j].find(s) == fc[j].end()) {
                        sub = false;
                        break;
                    }
                }
                if (sub) {
                    flag = false;
                    break;
                }
            }
            if (flag) {
                ans.push_back(i);
            } 
        }
        return ans;
    }
};
