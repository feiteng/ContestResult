class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& f)
    {
        vector<int> ans;
        int n = f.size();
        unordered_set<string> s[n];
        for(int i = 0; i < f.size(); ++i)
        {
            s[i].clear();
            for(int j = 0; j < f[i].size(); ++j)
                s[i].insert(f[i][j]);

        }
        for(int i = 0; i < f.size(); ++i)
        {
            bool ok = true;
            for(int j = 0; j < f.size(); ++j)
            {
                if(i == j)
                    continue;
                bool mis = false;
                for(int  k = 0; k < f[i].size(); ++k)

                {
                    if(!s[j].count(f[i][k]))
                    {
                        mis = true;
                        break;
                    }
                }
                if(!mis)
                {
                    ok = false;
                    break;
                }
            }
            if(ok)
                ans.push_back(i);
        }
        return ans;
        
    }
};
