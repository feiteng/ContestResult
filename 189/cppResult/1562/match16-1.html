class Solution {
public:
    
    bool has(vector<string> &arr, string &x) {
        return binary_search(arr.begin(), arr.end(), x);
    }
    
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {

        int n = favoriteCompanies.size();
        unordered_set<string> s[n];
        for(int i = 0; i < n; ++i) {
            for(string &x: favoriteCompanies[i]) {
                s[i].insert(x);
            }
        }
        vector<int> ans;
        // cout << n << endl;
        for(int i  =0; i < n; ++i) {
            bool valid = true;
            // cout << i << endl;
            for(int j = 0; j < n; ++j) {
                if(i == j)
                    continue;
                bool all = true;
                for(string &x: favoriteCompanies[i]) {
                    if(!s[j].count(x))
                        all = false;

                }
                if(all) {
                    // cout << i << ' ' << j << endl;
                    valid = false;
                    break;
                }
            }
            if(valid)
                ans.push_back(i);
        }
        return ans;
    }
};
