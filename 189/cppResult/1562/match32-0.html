class Solution {
public:
	vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
		int cid = 0;
		map<string ,int > mp;
		int len = favoriteCompanies.size();
		vector < set<int > > fc(len);
		
		for (int i = 0; i < len; i++) {
			auto& t = favoriteCompanies[i];
			fc;
			for (int j = 0; j < t.size(); j++) {
				if (mp.count(t[j]) == 0) {
					mp[t[j]] = cid;
					cid++;
				}
				fc[i].insert(mp[t[j]]);
			}
		}
		vector<int> ans;
		for (int i = 0; i < len; i++) {
			int fg = 1;
			for (int j = 0; j < len; j++) {
				if (i == j) continue;
				auto& s1 = fc[i];
				auto& s2 = fc[j];
				int flag = 0;
				for (auto& n : s1) {
					if (s2.count(n) == 0) {
						flag = 1;
						break;
					}

				}
				if (flag == 0) {
					fg = 0;
					break;
				}
			}
			if (fg == 1) {
				ans.push_back(i);
			}
		}
		return ans;
		
	}
};
