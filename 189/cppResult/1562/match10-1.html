class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();

        unordered_map<string, int> dict;
        int cnt = 0;
        for(int i = 0; i < n; ++i) {
            for(auto& s : favoriteCompanies[i]) {
                if(dict.find(s) == dict.end()) {
                    dict[s] = cnt++;
                }
            }
        }
        vector<bitset<50005>> sets(n);
        for(int i = 0; i < n; ++i) {
            for(auto& s : favoriteCompanies[i]) {
                sets[i] |= bitset<50005>(1) << dict[s];
            }
        }


        vector<int> ans;
        for(int i = 0; i < n; ++i) {
            bool ok = true;
            for(int j = 0; j < n; ++j) {
                if(i == j) continue;
                if((sets[i] & sets[j]) == sets[i]) {
                    ok = false;
                    break;
                }
            }
            if(ok) {
                ans.push_back(i);
            }
        }
        return ans;
    }
};
