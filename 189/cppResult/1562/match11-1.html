class Solution {
public:
    
    bool static compareInterval(pair<unordered_set<string>, int>& s1, pair<unordered_set<string>, int>& s2) { 
        return (s1.first.size() < s2.first.size());
    }
    

    vector<int> peopleIndexes(vector<vector<string>>& c) {
        int n = c.size();
        vector<pair<unordered_set<string>, int>> vec(n);
        int x = 0;
        for(int i = 0; i < n; i++) {
            unordered_set<string> st(c[i].begin(), c[i].end());

            vec[i] = {st, x++};
        }
        sort(vec.begin(), vec.end(), compareInterval);
        vector<int> ans;
        for(int i = 0 ; i < n; i++) {
            bool flag = false;
            for(int j = i + 1; j < n; j++) {
                if(ins(vec[i].first, vec[j].first)) {
                    flag = true;
                    break;
                }
            }
            if(!flag) ans.push_back(vec[i].second);
        }
        sort(ans.begin(), ans.end());
        return ans;
    }
    
    bool ins(unordered_set<string>& s1, unordered_set<string>& s2) {
        int count = 0;
        for(string x : s1) {
            if(s2.count(x)) count++;
            else return false;
        }
        return count == s1.size();
    }
    
};
