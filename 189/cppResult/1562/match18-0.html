class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        unordered_map<string, int> mp;
        vector<unordered_set<int>> fav(favoriteCompanies.size());
        int id=0;
        for (int i=0; i<fav.size(); ++i) {
            for (auto s: favoriteCompanies[i]) {
                if (mp.find(s)==mp.end()) {
                    mp[s]=id;
                    id++;
                }
                fav[i].insert(mp[s]);
            }
        }
        vector<int> res;
        for (int i=0; i<fav.size(); ++i) {
            bool flag=true;
            for (int j=0; j<fav.size(); ++j) {
                if (i==j) continue;
                if (fav[i].size()>fav[j].size()) continue;
                bool flag2=false;
                for (int s: fav[i]) {
                    if (fav[j].count(s)==0) {
                        flag2=true;
                        break;
                    }
                }
                if (!flag2) {
                    flag=false;
                    break;
                }
            }
            if (flag) {
                res.push_back(i);
            }
        }
        return res;
    }
};
