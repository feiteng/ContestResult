class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {

        unordered_map<string, int> d;
        int num = 0;
        bitset<50000> B[100];
        for (int i = 0; i < favoriteCompanies.size(); ++i) {
            for (auto &j : favoriteCompanies[i]) {
                if (d.find(j) == d.end()) d[j] = num++;
                B[i].set(d[j]);

            }
        }
        
        vector<int> ret;
        for (int i = 0; i < favoriteCompanies.size(); ++i) {
            bool f = 0;
            for (int j = 0; j < favoriteCompanies.size(); ++j) if (i != j) {

                if ((B[i] | B[j]) == B[j]) {
                    f = 1;
                    break;
                }
            }
            
            if (!f) ret.push_back(i);
        }
        
        return ret;
    }
};
