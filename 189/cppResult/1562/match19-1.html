class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& fc) {
        unordered_map<string, int> dict;
        int numComp = 0;
        for (auto& v : fc) {
            for (string& s : v) {

                if (dict.find(s) == dict.end()) {
                    dict[s] = numComp++;
                }
            }
        }
        vector<unordered_set<int>> vs(fc.size(), unordered_set<int>());
        for (int i = 0; i < fc.size(); i++) {
            for (string& s : fc[i]) {
                vs[i].insert(dict[s]);

            }
        }
        vector<int> res;
        for (int i = 0; i < vs.size(); i++) {
            bool distinct = true;
            for (int j = 0; j < vs.size(); j++) {
                if (i == j) continue;
                bool good = false;
                for (int x : vs[i]) {

                    if (vs[j].find(x) == vs[j].end()) {
                        good = true;
                        break;
                    }
                }
                if (!good) {
                    distinct = false;
                    break;
                }
            }
            if (distinct) {
                res.push_back(i);
            }
        }
        return res;
    }
};
