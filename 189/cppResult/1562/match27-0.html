class Solution {
public:
    bool check(vector<string>& minset, vector<string>& majorset){
        if(majorset.size() <= minset.size())
            return false;

        // check occurrence
        int curSet2Index = 0;
        for (int i = 0; i < minset.size(); i++) {
            string curSet1 = minset[i];
            string curSet2;
            while (true) {
                if (curSet2Index >= majorset.size()) return false;

                curSet2 = majorset[curSet2Index];
                if      (curSet1 > curSet2) curSet2Index++;
                else if (curSet1 < curSet2) return false;
                else                        break;
            }
        }

        return true;
    }
    
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        for(int i = 0; i < favoriteCompanies.size(); i++){
            sort(favoriteCompanies[i].begin(), favoriteCompanies[i].end());
        }
        vector<int> ans;
        for(int i = 0; i < favoriteCompanies.size(); i++){
            bool canAdd = true;
            for(int j = 0; j < favoriteCompanies.size(); j++){
                if(j == i)
                    continue;
                if(check(favoriteCompanies[i], favoriteCompanies[j])){
                    canAdd = false;
                    break;
                }
            }
            if(canAdd){
                ans.push_back(i);
            }
        }
        return ans;
        
        
        
    }
};
