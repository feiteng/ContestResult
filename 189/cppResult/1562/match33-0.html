class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& fa) {
        int n = fa.size();
        unordered_set<string> lists[n];
        for(int i = 0; i < n; ++i) {
            for(auto& str : fa[i]) {
                lists[i].insert(str);
            }
        }
        vector<int> ans;
        for(int i = 0; i < n; ++i) {
            bool valid = true;
            for(int j = 0; j < n; ++j) {
                bool isSubset = true;
                if(j == i) continue;
                for(auto& str : fa[i]) {
                    if(!lists[j].count(str)) {
                        isSubset = false;
                        break;
                    }
                }
                if(isSubset) {
                    valid = false;
                    break;
                }
            }
            if(valid)
                ans.push_back(i);
        }
        return ans;
    }
};
