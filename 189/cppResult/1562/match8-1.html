class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& fc) {
        using US = unordered_set<string>;
        vector<US> v;

        for(auto& f: fc) {
            US us; 
            for(auto c: f) {
                us.insert(c);    
            }
            v.push_back(us);
        }
        vector<int> ret;
        for(int i=0; i<v.size(); i++) {
            bool isSub = false;
            for(int j=0; j<v.size(); j++) {
                if(i == j)
                    continue;
                if(v[i].size() > v[j].size())
                    continue;
                bool isThis = true;

                for(auto& cur: v[i]) {
                    if(not v[j].count(cur)) {
                        isThis = false;
                        break;
                    }
                }
                if(isThis) {
                    isSub = true;
                    break;
                }
            }
            if(not isSub)
                ret.push_back(i);
        }
        return ret;
    }
};
