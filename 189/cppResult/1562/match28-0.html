class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& a) {
        int n = a.size();
        for (int i = 0; i < n; i++) {
            sort(a[i].begin(), a[i].end());
        }
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            int flag = 1;
            for (int j = 0; j < n; j++) if (j != i) {
                int p = 1;
                for (int x = 0, y = 0; x < a[i].size(); x++) {
                    while (y < a[j].size() && a[j][y] != a[i][x]) y++;
                    if (y == a[j].size()) {
                        p = 0; break;
                    }
                }
                if (p == 1) {
                    flag = 0;
                    break;
                }
            }
            if (flag) {
                ans.push_back(i);
            }
        }
        return ans;
    }
};
