class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& favoriteCompanies) {
        int n = favoriteCompanies.size();
        vector<unordered_set<string>> hash(n);
        for (int i = 0; i < n; ++i) {
            for (auto& s : favoriteCompanies[i]) {
                hash[i].insert(s);
            }

        }
        vector<int> res;
        for (int i = 0; i < n; ++i) {
            bool found = false;
            for (int j = 0; j < n; ++j) {
                if (i == j) {
                    continue;

                }
                if (isSubSet(favoriteCompanies[i], hash[j])) {
                    found = true;
                    break;
                }
            }
            if (!found) {
                res.push_back(i);
            }
        }
        return res;
    }
    
    bool isSubSet(const vector<string>& vec, const unordered_set<string>& sets) {
        for (const auto& s : vec) {
            if (!sets.count(s)) {
                return false;
            }
        }
        return true;
    }
};
