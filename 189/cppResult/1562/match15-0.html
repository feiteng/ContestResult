class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& arr) {
        vector<unordered_set<string>> v;
        int n = arr.size();
        vector<int> ans;
        for (const auto & vs: arr) {
            v.push_back(unordered_set<string>(vs.begin(), vs.end()));
        }
        for (int i = 0; i < n; i++) {
            bool flag = false;
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                bool subflag = true;
                for (const string & s: arr[i]) {
                    if (!v[j].count(s)) {
                        subflag = false;
                        break;
                    }
                }
                if (subflag == true) {
                    flag = true;
                    break;
                }
            }
            if (!flag)
                ans.push_back(i);
        }
        return ans;
    }
};
