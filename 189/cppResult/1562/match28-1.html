class Solution {
public:
    vector<int> peopleIndexes(vector<vector<string>>& fs) {

        for(int i=0; i<fs.size(); ++i) {
            sort(fs[i].begin(), fs[i].end());
        }
        vector<int> rst;
        for(int i=0; i<fs.size(); ++i) {
            int flag=1;
            for(int j=0; j<fs.size(); ++j) {
                if(j==i) continue;

                if(includes(fs[j].begin(), fs[j].end(), fs[i].begin(), fs[i].end())) {
                    flag = 0;
                    break;
                }
            }
            if(flag) {
                rst.push_back(i);
            }
        }
        return rst;
    }
};
