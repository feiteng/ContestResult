#define eps 1e-8
class Solution {
public:
    struct Point{
        double x,y;
        Point(){}
        Point(double tx,double ty){x=tx;y=ty;}
    }p[300];
    
    double dist(Point p1,Point p2){
        return sqrt((p1.x-p2.x)*(p1.x-p2.x)+(p1.y-p2.y)*(p1.y-p2.y));
    }
    
    Point get_circle(Point p1,Point p2,int r){
        Point mid=Point((p1.x+p2.x)/2,(p1.y+p2.y)/2);
        double angle=atan2(p1.x-p2.x,p2.y-p1.y);
        double d=sqrt(r*r-dist(p1,mid)*dist(p1,mid));
        return Point(mid.x+d*cos(angle),mid.y+d*sin(angle));
    }
    
    int numPoints(vector<vector<int>>& points, int r) {
        int i = 0;
        for(auto x:points){
            p[i] = Point(x[0],x[1]);
            i++;
        }
        int n = points.size();
        int ans=1;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                if(dist(p[i],p[j])>2*r) continue;
                Point central=get_circle(p[i],p[j],r);
                int cnt=0;
                for(int k=0;k<n;k++)
                    if(dist(central,p[k])<r+eps)
                        cnt++;
                ans=max(ans,cnt);
            }
        }
        return ans;
    }
};
