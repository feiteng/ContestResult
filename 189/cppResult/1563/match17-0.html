class Solution {
public:
    
    #define eps 1e-8
    
    struct Point
	{
		double x,y;
		Point(){}
		Point(double a,double b) { x=a;y=b; }
	}p[200];

	double dist(Point p1,Point p2) { return sqrt(pow(p1.x-p2.x,2)+pow(p1.y-p2.y,2)); }
	int r1;
	Point GetCircleCenter( Point p1,Point p2 )
	{
		Point mid = Point((p1.x+p2.x)/2,(p1.y+p2.y)/2);
		double angle = atan2(p1.x-p2.x,p2.y-p1.y);
		double d = sqrt(r1*r1-pow(dist(p1,mid),2));
		return Point(mid.x+d*cos(angle),mid.y+d*sin(angle));
	}
	
	int numPoints(vector<vector<int>>& points, int r) {
		int num=points.size();
		r1=r;
		for( int i=0;i<points.size();i++ )
		{
			p[i].x=points[i][0];
			p[i].y=points[i][1];
		}
		int ans=1;
		for(int i=0;i<num;i++ )
		{
			for(int j=i+1;j<num;j++ )
			{
				if(dist(p[i],p[j]) > 2.0*r) continue;
				Point center = GetCircleCenter(p[i],p[j]);
				int cnt = 0;
				for(int k=0;k<num;k++)
					if( dist(center,p[k])<1.0*r+eps ) cnt++;
				ans = max(ans,cnt);
			}
		}
	    return ans;
    }
};
