const double eps=1e-7;
struct Point{
	double x,y;
	Point(){}
    Point(double tx,double ty){x=tx;y=ty;}
};
double dist(Point p1,Point p2){
	return sqrt((p1.x-p2.x)*(p1.x-p2.x)+(p1.y-p2.y)*(p1.y-p2.y));
}

Point get_circle(Point p1,Point p2){
	Point mid=Point((p1.x+p2.x)/2,(p1.y+p2.y)/2);
	double angle=atan2(p1.x-p2.x,p2.y-p1.y);
    double d=sqrt(1-dist(p1,mid)*dist(p1,mid));
	return Point(mid.x+d*cos(angle),mid.y+d*sin(angle));
}

class Solution {
public:
    int numPoints(vector<vector<int>>& points, int r) {
		int n=points.size();
		vector<Point> p;
		p.clear();
		for (int i=0;i<n;++i)
		{
			p.push_back(Point(1.0*points[i][0]/r,1.0*points[i][1]/r));

		}
		int ans=1;
		for (int i=0;i<n;++i)
		{
			for(int j=i+1;j<n;++j)
			{
				if (dist(p[i],p[j])>2.0) continue;
				Point o=get_circle(p[i],p[j]);
				int now=0;
				for (int k=0;k<n;++k) if (dist(p[k],o)<1.0+eps) now++;
				ans=max(ans,now);
			}
		}
		return ans;
    }
};
