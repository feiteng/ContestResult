class Solution {
    //split char
std::vector<std::string> split(const std::string& input, const char& regex) {
	std::vector<std::string> out;
	std::string i;
	std::istringstream tokenStream(input);
	while (std::getline(tokenStream, i, regex))
	{
		out.push_back(i);
	}
	// std::cout << out.size();
	return out;
}
public:
    string arrangeWords(string text) {
        if(text.empty()) return "";
        text[0] = tolower(text[0]);
        string ans;
        vector<string> str_v = split(text, ' ');
        stable_sort(str_v.begin(), str_v.end(), [](auto& a, auto& b) {
           return a.size() < b.size();
        });
        for(auto& str : str_v) {
            ans += str;
            ans += " ";
        }
        ans.pop_back();
        ans[0] = toupper(ans[0]);
        return ans;
    }
};
