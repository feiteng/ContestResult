class Solution {
public:
    string arrangeWords(string text) {
        text[0] = tolower(text[0]);
        stringstream ss(text);
        int p = 0;
        string tmp;
        vector< pair<int, pair<int,string> > > T;
        while( ss >> tmp ){
            T.push_back({tmp.size(), {p,tmp}});
            ++p;

        }
        string ans;
        sort(T.begin(),T.end());
        bool first = true;
        for(auto &x:T){
            if( first )
                x.second.second[0] = toupper(x.second.second[0]);
            else 
                ans += " ";
            first = false;
            ans+= x.second.second;
        }
        return ans;
    }
};
