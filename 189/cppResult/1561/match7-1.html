class Solution {
 public:
  string arrangeWords(string text) {
    text[0] = text[0] - 'A' + 'a';
    map<int, vector<string>> mp;
    istringstream is(text);
    string token;
    while (getline(is, token, ' ')) {
      mp[token.size()].push_back(token);
    }
    string ans;

    for (auto& it : mp) {
      for (string& s : it.second) {
        ans += s;
        ans += ' ';
      }
    }
    ans.pop_back();
    ans[0] = ans[0] - 'a' + 'A';
    return ans;
  }
};
