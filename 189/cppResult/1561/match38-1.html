class Solution {
public:
    string arrangeWords(string text) {
        auto split = [&](string& s, string delimeters = "    .,;") -> vector<string> {
            vector<string> res;
            int n = s.length();
            
            for (int i = 0; i < n; ++i) {
                auto j = s.find_first_of(delimeters, i);
                if (j == string::npos) {
                    j = n;
                }
                if (j - i > 0) {
                    res.push_back(s.substr(i, j - i));
                }
                i = j;
            }
            return res;
        };
        
        auto words = split(text);
        
        for (auto& s : words) {
            if (isupper(s[0])) {
                s[0] = s[0] + 32;
            }
        }
        
        auto cmp = [&](int l, int r) {
            if (words[l].length() == words[r].length()) {
                return l < r;
            }
            return words[l].length() < words[r].length();
        };
        int n = words.size();
        vector<int> ord(n, 0);
        iota(ord.begin(), ord.end(), 0);
        
        sort(ord.begin(), ord.end(), cmp);
        
        string res;
        

        for (auto x : ord) {
            if (!res.empty()) {
                res += ' ';
                res += words[x];
            } else {
                words[x][0] = toupper(words[x][0]);
                res += words[x];
            }
        }
        
        return res;
    }
};
