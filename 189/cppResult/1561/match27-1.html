class Solution {
public:
    static bool compare(string s1, string s2){
        int m = s1.size(), n = s2.size();
        return m < n;
    }
    string arrangeWords(string text) {
        text[0] = text[0] - 'A' + 'a';
        int n = text.size();
        vector<string> words;
        int prev = 0;
        for(int i = 1; i < n; i++){
            if(text[i] == ' '){
                words.push_back(text.substr(prev, i-prev));

                prev = i+1;
            }
        }
        words.push_back(text.substr(prev, n-prev));
        stable_sort(words.begin(), words.end(), compare);
        string res = words[0];
        for(int i = 1; i < words.size(); i++){
            res += " " + words[i];
        }
        res[0] = res[0] -'a' + 'A';
        return res;
    }
};
