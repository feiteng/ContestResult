class Solution {
public:
    string arrangeWords(string text)
    {
        vector<string> tokens;
        text.push_back(' '); // for the last word
        string word;
        for (auto &&ch : text)
        {
            if (ch == ' ')
            {
                tokens.push_back(word);
                word.clear();
            }
            else
            {
                word.push_back(tolower(ch));

            }
        }
        stable_sort(tokens.begin(), tokens.end(), [](const auto &a, const auto &b) -> bool { return a.length() < b.length(); });
        string ret;
        for (auto &&word : tokens)
        {
            ret += word;
            ret.push_back(' ');
        }
        if (ret.length() > 0)
        {
            string temp;
            temp.push_back(toupper(ret[0]));
            ret = temp + ret.substr(1, ret.length() - 2);
        }
        return ret;
    }
};
