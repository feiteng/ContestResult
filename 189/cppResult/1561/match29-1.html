class Solution {
public:
    struct node{
        string s;
        int id;
        bool operator <(const node &tmp)const{
            if(s.size()==tmp.s.size())return id<tmp.id;
            return s.size()<tmp.s.size();
        }
    };
    vector<node>str;

    string arrangeWords(string text) {
        string box;
        int id=0;
        for(int i=0;i<text.size();i++){
            if(text[i]==' '){
                str.push_back({box,id++});
                box.clear();
            }else box.push_back(tolower(text[i]));
        }
        str.push_back({box,id++});
        sort(str.begin(),str.end());
        string ans;
        for(int i=0;i<str.size();i++){
            ans += str[i].s;
            if(i==0)ans[0] = toupper(ans[0]);
            if(i!=str.size()-1)ans+=' ';
        }
        return ans;
    }
};
