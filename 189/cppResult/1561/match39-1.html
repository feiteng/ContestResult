#include <bits/stdc++.h>
// BEGIN NO SAD
#define all(x) x.begin(), x.end()
#define sz(x) (int)(x).size()
// END NO SAD
using namespace std;
// REMEMBER CLEAR GLOBAL STATE
// cerr doesn't work for output, only use cout

bool ssort(pair<string, int> a, pair<string, int> b) {
  if(sz(a.first) != sz(b.first)) return sz(a.first) < sz(b.first);
  return a.second < b.second;
}
class Solution {
public:

    string arrangeWords(string text) {
        vector<pair<string, int>> v;
        int id = 0;
        for(int i = 0; i < sz(text);) {
          int j = i+1;
          while(j < sz(text) && text[j] != ' ') j++;
          v.push_back({text.substr(i, j-i), id++});
          i = j+1;
        }
        sort(all(v), ssort);
        string ret = "";
        bool p = false;
        for(auto out: v) {
          string cand = "";
          if(p) {
            for(int i = 0; i < sz(out.first); i++) cand += tolower(out.first[i]);
          }
          else {
            cand += toupper(out.first[0]);
            for(int i = 1; i < sz(out.first); i++) cand += tolower(out.first[i]);
          }
          if(p) ret += " ";
          p = true;
          ret += cand;
        }
        return ret;
    }
};
