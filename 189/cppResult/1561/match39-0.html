bool compare(pair<string,int> &s1,pair<string,int> &s2){ 
    if(s1.first.length()!=s2.first.length()) return s1.first.length() < s2.first.length(); 
    return s1.second < s2.second;
    
}
class Solution {
public:
    string arrangeWords(string text) {    
        vector <pair<string,int> > tks;     
        stringstream ss(text);       
        string tmp;           
        int cnt=0;
        while(getline(ss, tmp, ' ')){ 
            tmp[0] = tolower(tmp[0]);
            tks.push_back(make_pair(tmp,cnt++));             
        } 
        sort(tks.begin(),tks.end(),compare);
        string res;
        tks[0].first[0] = toupper(tks[0].first[0]);
        res=tks[0].first;
        for(int i=1; i<tks.size();i++){
            res+= " " + tks[i].first;
        }
        return res;
    }
};
