class Solution {
public:
    string arrangeWords(string text) {
        stringstream ss(text);
        vector<pair<string, int>> t;

        int idx = 0;
        while (ss >> text) {
            if (isupper(text[0])) {
                text[0] += 'a' -'A';
            }
            t.push_back({text, idx++});
        }
        
        sort(t.begin(), t.end(), [](pair<string, int>& a, pair<string, int>& b) {
            if (a.first.size() != b.first.size()) return a.first.size() < b.first.size();
            return a.second < b.second;
        });
        
        string ans;
        
        for (int i = 0; i < idx-1; i++) {
            ans.append(t[i].first); 
            ans.append(" ");
        }
            
        ans.append(t.back().first);
        ans[0] += 'A'-'a';
        return ans;
        
        
        
    }
};
