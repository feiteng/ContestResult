class Solution {
public:
    string arrangeWords(string text) {
        text[0] += 'a' - 'A';
        istringstream iss(text);
        string temp;

        int idx = 0;
        map<int, vector<string>> all;
        
        while(iss >> temp) {
            all[temp.size()].push_back(temp);

        }
        
        
        string res = "";
        for(auto it: all) {
            for(string& s: it.second) {
                res += s + " ";
            }
        }
        res.pop_back();
        res[0] += 'A' - 'a';
        
        return res;
    }
};
