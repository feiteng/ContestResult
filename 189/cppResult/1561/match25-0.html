class Solution {
public:
    string arrangeWords(string text) {
        vector<pair<string, int>> vec;
        string ans = "";
        text[0] = text[0] + 'a' - 'A';
        for(char c: text) {
            if(c==' ') {
                if(!ans.empty()) {
                    vec.push_back({ans, vec.size()});
                }
                ans.clear();
            } else ans.push_back(c);
        }
        if(!ans.empty()) {
            vec.push_back({ans, vec.size()});
        }
        sort(vec.begin(), vec.end(), [](pair<string, int>& p1, pair<string, int>& p2) {
            if(p1.first.size()!=p2.first.size()) return p1.first.size() < p2.first.size();
            return p1.second < p2.second;
        });
        string res = "";
        for(auto& p: vec) {
            if(!res.empty()) res += " ";
            res+=p.first;
        }
        res[0] = res[0] + 'A' -'a';
        return res;
    }
};
