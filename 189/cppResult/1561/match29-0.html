class Solution {
public:
    struct node{
        string a;
        int size,ind;
        bool operator<(const node&rhs)const{
            if(size==rhs.size)return ind<rhs.ind;
            return size<rhs.size;
        }
    };
    vector<node>s;
    string arrangeWords(string text) {
        stringstream ss(text);
        string word;
        int cnt=0;
        while(ss>>word){
            node x;
            for(int i=0;i<word.size();i++){
                word[i]=tolower(word[i]);
            }
            x.ind=cnt++;
            x.a=word;
            x.size=word.size();
            s.push_back(x);
        }
        sort(s.begin(),s.end());
        int n=s.size();
        string ans="";
        ans+=s[0].a;
        ans[0]=toupper(ans[0]);
        for(int i=1;i<n;i++){
            ans+=' ';
            ans+=s[i].a;
        }
        return ans;
    }
};
