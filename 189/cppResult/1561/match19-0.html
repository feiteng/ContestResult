class Solution {
public:
    string arrangeWords(string text) {
        istringstream in(text);
        vector<pair<string, int>>str;
        string t = "";
        int i = 0;
        while(in>>t)
        {
            if(t[0] >= 'A')
            {
                t[0] = tolower(t[0]);
            }
            str.push_back({t, i++});
        }
        sort(str.begin(), str.end(), [](pair<string, int>& a, pair<string, int>& b)
             {
                 return a.first.size() < b.first.size() || (a.first.size() == b.first.size() && a.second < b.second);
             });
        string res;
        for(int i = 0; i < str.size(); i++)
        {
            res += str[i].first + " ";
        }
        res.pop_back();
        if(!res.empty())res[0] = toupper(res[0]);
        return res;
    }
};
