class Solution {
public:
    string arrangeWords(string text) {
        vector<pair<string,int>> vs;
        int cnt=0;
        text[0]+=32;
        for(int i=0;i<text.length();i++){
            string tmp="";
            while(i<text.length() && text[i]!=' '){
                tmp+=text[i];
                i++;
            }
            
            if(tmp!="") vs.push_back({tmp,cnt});
            cnt++;
        }
        
        sort(vs.begin(),vs.end(),[&](pair<string,int>& i,pair<string,int>& j){
           return i.first.length()<j.first.length() || (i.first.length()==j.first.length() && i.second<j.second); 
        });
        string ans=vs[0].first;
        ans[0]-=32;
        for(int i=1;i<vs.size();i++){
            ans+=' ';
            ans+=vs[i].first;
        }
        return ans;
    }
};
