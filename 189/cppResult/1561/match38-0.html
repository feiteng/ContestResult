class Solution {
public:
  string arrangeWords(string text) {
    map<int, vector<string>> m;
    
    stringstream ss;
    ss << text;
    
    string str;
    while (ss >> str) {
      m[(int) str.length()].push_back(str);
    }
    
    string res = "";
    for (auto p : m) {
      for (auto word : p.second) {
        if (res.empty()) {
          word[0] = toupper(word[0]);
          res = word;
        }
        else {
          word[0] = tolower(word[0]);
          res += " " + word;
        }
      }
    }
    return res;
  }
};
