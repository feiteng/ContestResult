class Solution {
public:
    string arrangeWords(string text) {
        vector<string> words;
        string ans = "";
        
        int i = 0, j = 0;
        while (i < text.size()) {
            if (text[i] == ' ') {
                if ('A' <= text[j] && text[j] <= 'Z') {
                    text[j] -= ('A' - 'a');
                }
                words.push_back(text.substr(j, i - j));
                j = i = i + 1;
            } else {
                i++;
            }
        }
        if (i > j) {
            words.push_back(text.substr(j, i - j));
        }
        
        vector<vector<int>> idx(2000);
        for (int i = 0; i < words.size(); i++) {
            int l = words[i].size();
            idx[l].push_back(i);
        }
        int k;
        for (int i = 1; i < idx.size(); i++) {
            for (int j = 0; j < idx[i].size(); j++) {
                k = idx[i][j];
                ans += words[k] + ' ';
            }
        }
        
        /*vector<int> idx;
        for (int i = 0; i < words.size(); i++) {
            idx.push_back(i);
        }
        
        sort(idx.begin(), idx.end(), [words] (int a, int b) {
            if (words[a].size() == words[b].size()) {
                return a < b;
            } else {
                return words[a].size() < words[b].size();
            }
        });
        
        
        for (int i = 0; i < words.size(); i++) {
            ans += words[idx[i]] + ' ';
        }*/
        
        ans[0] += 'A' - 'a';
        ans = ans.substr(0, ans.size() - 1);
        
        return ans;
    }
};
