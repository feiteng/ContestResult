class Solution {
public:
    string arrangeWords(string text) {
        stringstream ss(text);
        string word;
        map<int, vector<string>> m;
        while (ss >> word) {
            word[0] = tolower(word[0]);
            m[word.size()].push_back(word);
        }
        string result = "";
        for (auto kv : m) {
            for (auto w : kv.second) {
                result += w + " ";
            }
        }
        result.pop_back();
        result[0] = toupper(result[0]);
        return result;
    }
};
