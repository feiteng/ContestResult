class Solution {
public:
    string arrangeWords(string text) {
        text[0] -= 'A' - 'a';
        istringstream in(text);
        vector<string> zs;
        while(in) {
            string s;
            in >> s;
            if (!in) break;
            zs.push_back(s);
        }
        stable_sort(zs.begin(), zs.end(), [](const string & a, const string & b){
            return a.size()< b.size();
        });
        cout<<zs.size();
        ostringstream out;
        for (int i=0; i<zs.size(); ++i) {
            
            out<<zs[i];
            if (i!=zs.size()-1) out<<' ';
        }
        auto s = out.str();
        s[0] = s[0] +( 'A' - 'a');
        return s;
    }
};
