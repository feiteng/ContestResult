class Solution {
public:
    string arrangeWords(string text) {
        int n = text.size();
        
        if (n <= 1) {
            return text;
        }
        
        map<int, vector<string>> mp;
        
        string cur = "";
        cur.push_back(tolower(text[0]));
        
        
        for (int i = 1; i < text.size(); i ++) {
            if (text[i] == ' ') {
                mp[cur.size()].push_back(cur);
                cur = "";
            }
            else {
                cur.push_back(text[i]);
            }
        }
        
        mp[cur.size()].push_back(cur);
        
        string res = "";
        
        
        for (auto it : mp) {
            for (string s : it.second) {
                res += s;
                res.push_back(' ');
            }
        }
        
        res.pop_back();
        res[0] = toupper(res[0]);
        
        return res;
    }
};
