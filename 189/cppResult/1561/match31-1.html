class Solution {
public:
    string arrangeWords(string text) {
        string ans;
        vector<string> v;
        text[0] = tolower(text[0]);
        
        stringstream ss(text);
        string s; 
        while(ss >> s) {
            v.push_back(s);
        }
        
        stable_sort(v.begin(), v.end(), [](string s, string t) {
           return s.size() < t.size();  
        });
        

        for(int i = 0; i < v.size(); i++) {
            if(i) ans += ' ';
            ans += v[i];
        }
        
        ans[0] = toupper(ans[0]);
        return ans;
    }
};
