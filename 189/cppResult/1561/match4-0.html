class Solution {
public:
    string arrangeWords(string text) {
        text[0] = text[0] - 'A' + 'a';
        map<int, list<string>> map;
        istringstream ss(text);
        string s;
        while (getline(ss, s, ' ')) {
            map[s.size()].push_back(s);
        }
        string res = "";
        for (auto i : map) {
            for (auto j : map[i.first]) {
                res += j + " ";
            }
        }
        res[0] = res[0] - 'a' + 'A';
        res.pop_back();
        return res;
    }
};
