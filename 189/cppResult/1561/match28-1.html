class Solution {
public:
    string arrangeWords(string text) {
        vector<string> words;
        words.emplace_back();
        for (auto c: text) {
            if (c == ' ')words.emplace_back();
            else {
                if (c >= 'A' && c <= 'Z') c = c - 'A' + 'a';
                words.back().push_back(c);
            }
        }
        stable_sort(words.begin(), words.end(),

                    [](const string &word1, const string &word2) -> bool {
                        return word1.size() < word2.size();
                    });
        string result;
        for (auto &word:words) {
            result += word;
            result += ' ';
        }
        result.pop_back();
        result[0] = result[0] - 'a' + 'A';
        return result;
    }
};
