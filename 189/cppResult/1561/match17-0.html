class Solution {
public:
    string arrangeWords(string text) {
        stringstream ss(text);
        vector<pair<string, int>> v;
        string s;
        int i = 0;
        while(ss >> s) {
            for(char& c : s)
                c = tolower(c);
            v.emplace_back(s, i++);
        }
        
        sort(v.begin(), v.end(), [&](const pair<string, int>& a, pair<string, int>& b){
            if(a.first.size() == b.first.size()) return a.second < b.second;
            return a.first.size() < b.first.size();
        });
        
        string ans;
        bool fst = true;
        for(auto& x : v) {
            if(!fst) ans += " ";
            else x.first[0] = toupper(x.first[0]);
            ans += x.first;
            fst = false;
        }
        
        return ans;
    }
};
