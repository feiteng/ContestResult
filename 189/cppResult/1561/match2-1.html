class Solution {
public:
    string arrangeWords(string text) {
        text[0]=tolower(text[0]);
        map<int,vector<string>> m;
        istringstream iss(text);
        string token;
        while(getline(iss,token,' ')){
            m[token.length()].push_back(token);
        }
        string ans;

        for(auto i:m){
            for(auto j:i.second){
                ans+=j+" ";
            }
        }
        ans[0]=toupper(ans[0]);
        return ans.substr(0,ans.length()-1);
    }
};
