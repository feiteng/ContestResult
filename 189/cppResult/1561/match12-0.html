class Solution {
public:
    string arrangeWords(string text) {
        text[0] = text[0] - 'A' + 'a';
        vector<string> tmp = {""};
        map<int, vector<string>> map;
        for(char c: text) {
            if(c == ' ') tmp.push_back("");
            else tmp.back() += c;
        }
        for(auto& x: tmp) {
            map[x.size()].push_back(x);
        }
        
        
        // for(int i = 0; i < tmp.size(); i++) map[tmp[i]] = i; 
        // stable_sort(tmp.begin(), tmp.end(), [&](auto& x, auto& y){
        //     // int a = map[x].front();
        //     // int b = map[y].front();
        //     // map[x].pop(), map[y].pop();
        //     if(x.size() > y.size()) return false;
        //     if(y.size() > x.size()) return true;
        //     if(ma >= b) return true;
        //     return false;
        // });
        // for(auto x: tmp) cout<<x<<' ';
        // cout<<endl;
        string res;
        for(auto& y: map) {
            for(auto& x: y.second) res += x + " ";
        }
        // for(auto& x: map) res += x + " ";
        res.pop_back();
        res[0] = res[0] - 'a' + 'A';
        
        return res;
    }
};
