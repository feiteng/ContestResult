class Solution {
public:
    string arrangeWords(string text) {
        stringstream ss(text);
        vector<string> wds;
        vector<int> idxs;
        string tmp;
        int cnt = 0;
        ss >> tmp;
        tmp[0] = tolower(tmp[0]);
         wds.push_back(tmp);
            idxs.push_back(cnt++);
        while(ss >> tmp) {
            wds.push_back(tmp);
            idxs.push_back(cnt++);
        }
        sort(idxs.begin(), idxs.end(), [&](int a, int b) {
            return (wds[a].size() != wds[b].size()) ? (wds[a].size() < wds[b].size()) : (a < b);
        });
        tmp = wds[idxs[0]];
        tmp[0] = toupper(tmp[0]);
        for(int i = 1; i < cnt; i++) {
            tmp += " " + wds[idxs[i]];
        }
        return tmp;
        
    }
};
