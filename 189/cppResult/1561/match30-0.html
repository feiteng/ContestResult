struct P{
  string s;
  int x;
  P(string s, int x):s(s), x(x){}  
    bool operator<(const P&a)const {
        return s.size() == a.s.size()?x<a.x:s.size()<a.s.size();
    }
};
class Solution {
public:
    string arrangeWords(string s) {
        vector<P>v;
        string tt = "";
        int cur = 0;
        for(int i=0;i<s.size();i++){
            if(s[i]==' '){
                v.push_back(P(tt, cur));
                tt = "";
                cur++;
            } else {
                if(s[i]>='A'&&s[i]<='Z')s[i] = s[i]-'A'+'a';
                tt+=s[i];
            }
        }
        if(tt.size()){
            v.push_back(P(tt, cur));
        }
        sort(v.begin(),v.end());
        string ret;
        for(int i=0;i<v.size();i++){
            if(i)ret+=' ';
            ret+=v[i].s;
        }
        ret[0] = ret[0] - 'a' + 'A';
        return ret;
    }
};
