class Solution {
public:
    string arrangeWords(string text) {
        text[0] = text[0] + 'a' - 'A';
        stringstream ss;
        ss << text;
        map<int, vector<string>> mp;
        while (!ss.eof())
        {
            string t;
            ss >> t;
            mp[t.length()].push_back(t);
        }
        string ret;
        for (auto p : mp)
        {
            for (auto x : p.second) ret += x + " ";
        }
        ret.pop_back();
        ret[0] = ret[0] + 'A' - 'a';
        return ret;
    }
};
