class Solution {
public:
    string arrangeWords(string text) {
        text[0] = tolower(text[0]);
        int n = text.size();
        vector<pair<string, int>> arr;
        string buffer;
        int pos = 0;
        for(int i = 0;i < n;i++) {
            if (text[i] == ' ') {
                arr.emplace_back(buffer, pos++);
                buffer = "";
            } else {
                buffer += text[i];
            }
        }
        arr.emplace_back(buffer, pos++);
        sort(arr.begin(), arr.end(), [](pair<string, int> a, pair<string,int> b) {
            if (a.first.size() == b.first.size()) {
                return a.second < b.second;
            }
            return a.first.size() < b.first.size();
        });
        string sol;
        for(auto i : arr) {
            if (sol.empty()) {
                sol += i.first;
            } else {
                sol += ' ';
                sol += i.first;
            }
        }
        sol[0] = toupper(sol[0]);
        return sol;
    }
};
