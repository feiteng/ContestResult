class Solution {
public:
    string arrangeWords(string text) {
        vector<string> sents;
        string sent="";
        text[0] = text[0]+'a'-'A';
        for(int i=0; i< text.size(); ++i) {
            if(text[i] == ' ') {
                sents.push_back(sent);
                sent = "";
            } else {
                sent += text[i];
            }
        }
        if(sent != "") sents.push_back(sent);

        stable_sort(sents.begin(), sents.end(), [](auto a, auto b)->bool{
            return a.size() < b.size();
        });
        string rst = "";
        rst += sents[0][0]+'A'-'a';
        for(int i=1; i<sents[0].size(); ++i) {
            rst +=sents[0][i];
        }
        for(int i=1; i<sents.size(); ++i) {
            rst += " "+sents[i];
        }
        return rst;
    }
};
