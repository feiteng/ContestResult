class Solution {
public:
    struct comp {
        bool operator () (const string &a, const string &b) {
            return a.size() < b.size();
        }
    };
    string arrangeWords(string text) {
        vector<string> words;
        if (text[0] >= 'A' && text[0] <= 'Z') text[0] = text[0] - 'A' + 'a';
        for (int i = 0; i < text.size();) {
            int next = text.find_first_of(' ', i);
            if (next != string::npos) {
                words.push_back(text.substr(i, next - i));
                i = next + 1;
            } else {
                words.push_back(text.substr(i));
                break;
            }
        }
        stable_sort(words.begin(), words.end(), comp());
        string ans;
        for (int i = 0; i < words.size(); ++ i) {
            if (i) ans += " ";
            ans += words[i];
        }
        if (ans[0] >= 'a' && ans[0] <= 'z') ans[0] = ans[0] - 'a' + 'A';
        return ans;
    }
};
