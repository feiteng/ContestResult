vector<string> split(const string &s, char stop)
{
    vector<string> ret;
    string t;
    for (auto c : s)
    {
        if (c == stop)
        {
            ret.push_back(t);
            t = "";
        }
        else
            t += c;
    }
    ret.push_back(t);
    return ret;
}

class Solution {
public:
    string arrangeWords(string s) {
        auto v = split(s, ' ');
        int n = v.size();
        vector<int> a(n);
        for (int i = 0; i < n; ++ i) a[i] = i;
        sort(a.begin(), a.end(), [&](int i, int j)

             {
                 return v[i].size() < v[j].size() || v[i].size() == v[j].size() && i < j;
             });
        string ret;
        for (int i = 0; i < n; ++ i)
        {
            if (i) ret += ' ';
            ret += v[a[i]];
        }
        for (int i = 0; i < ret.size(); ++ i)
        {
            if (isupper(ret[i])) ret[i] = ret[i]-'A'+'a';
        }
        ret[0] = ret[0]-'a'+'A';
        return ret;
    }
};
