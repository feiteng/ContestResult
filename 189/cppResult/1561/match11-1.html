class Solution {
public:
    string arrangeWords(string text) {
        if(text.empty())return "";
        vector<string> vs;
        text[0]=tolower(text[0]);
        int last=0;
        for(int i=0;i<text.size();i++){
            if(text[i]==' '){
                vs.push_back(text.substr(last,i-last));
                last=i+1;
            }
        }
        vs.push_back(text.substr(last,text.size()-last));

        stable_sort(vs.begin(),vs.end(),[](const string &a,const string &b){return a.size()<b.size();});
        string res;
        for(int i=0;i<vs.size();i++){
            res+=vs[i];
            if(i!=vs.size()-1)res+=' ';
        }
        res[0]=toupper(res[0]);
        return res;
    }
};
