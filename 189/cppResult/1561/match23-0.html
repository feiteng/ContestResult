class Solution {
public:
    string arrangeWords(string text) {
        istringstream istr(text);
        vector<pair<string, int>> st;
        int idx = 0;
        while (!istr.eof()) {
            string s;
            istr >> s; // get a word
            transform(s.begin(), s.end(), s.begin(), ::tolower);
            st.push_back(make_pair(s, idx++));
        }
        sort(st.begin(), st.end(),
             [](const pair<string, int> &x, const pair<string, int> &y) {
                 if (x.first.length() != y.first.length())
                     return x.first.length() < y.first.length();
                 else {
                     return x.second < y.second;
                 }
             });
        string ans = "";
        int len = st.size();
        for (int i = 0; i < len; i++) {
            if (i == 0) {
                ans += st[i].first[0] - 32;
                ans += st[i].first.substr(1);
            } else {

                ans += " " + st[i].first;
            }
        }
        return ans;
    }
};
