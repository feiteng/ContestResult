class Solution {
public:
    string arrangeWords(string text) {
        istringstream iss(text);
        vector<string> data;
        string s;
        while(iss >> s)
        {
            data.emplace_back(s);
        }
        data[0][0] = tolower(data[0][0]);
        stable_sort(data.begin(), data.end(), [](const string & a, const string & b) -> bool {
            return a.length() < b.length(); 
        });
        data[0][0] = toupper(data[0][0]);
        text = data[0];
        for(int i = 1; i < data.size(); ++i)
        {
            text += " " + data[i];
        }
        return text;
    }
};
