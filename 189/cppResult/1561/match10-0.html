class Solution {
public:
    bool static compareInterval(pair<string, int> s1, pair<string, int> s2) { 
        if((s1.first.size() < s2.first.size())) return true;
        else if((s1.first.size() == s2.first.size())) return s1.second < s2.second;
        return false;
    }
    
    string arrangeWords(string text) {
        int n = text.size();
        vector<pair<string, int>> vec;
        string temp = "";
        text[0] += ('a' - 'A');
        int x = 0;
        for(int i = 0; i < n; i++) {
            if(text[i] == ' ') vec.push_back({temp, x++}), temp = "";
            else temp += text[i];
        }
        vec.push_back({temp, x});
        
        sort(vec.begin(), vec.end(), compareInterval);
        
        string ans = "";
        
        for(int i = 0; i < vec.size(); i++) {
            ans += vec[i].first;
            if(i < vec.size() - 1) ans += " ";
        } 
        
        ans[0] -= ('a' - 'A');
        return ans;
    }
};
