class Solution {
public:
    string arrangeWords(string text) {
        map<int, vector<string>> words;
        int ind = 0;
        while (ind < text.length()) {
            int next = text.find_first_of(" ", ind);
            if (next == -1) next = text.length();
            if (words.find(next - ind) == words.end()) {
                words[next - ind] = vector<string>();
            }
            string tmp = text.substr(ind, next - ind);
            if (ind == 0) tmp[0] += 'a' - 'A';
            words[next - ind].emplace_back(tmp);
            ind = next + 1;
        }

        string res = "";
        for (auto& p : words) {
            for (string& s : p.second) {
                res += s;
                res += " ";
            }
        }
        res.pop_back();
        res[0] += 'A' - 'a';
        return res;
    }
};
