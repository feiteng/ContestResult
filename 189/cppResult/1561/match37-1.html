class Solution {
public:
    
    struct cmp {
        bool operator()(const string &s, const string &t) {
            return s.size() < t.size();
        }
    };
    string arrangeWords(string text) {
        stringstream ss(text);
        vector<string> v;
        string s;
        while(ss >> s) {
            s[0] = tolower(s[0]);
            v.push_back(s);
        }
        stable_sort(v.begin(), v.end(), cmp());
        s = "";
        for(string t : v) {
            s += t;
            s += ' ';
        }
        s.resize(s.size() - 1);
        s[0] = toupper(s[0]);
        return s;
    }
};
