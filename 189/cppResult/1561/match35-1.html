class Solution {
public:
void merge(vector<string> &a, int l,int m,int r)
{
    string *tmp = new string[r - l + 5];
    int i = l, j = m + 1, k = 0;
    while (i<=m&&j<=r)
        tmp[k++] = a[i].size() > a[j].size() ? a[j++] : a[i++];
    while (i<=m)
        tmp[k++] = a[i++];
    while (j<=r)
        tmp[k++] = a[j++];
    for (int i = l,j=0; i <=r;)
        a[i++] = tmp[j++];
    delete[] tmp;
}
void mergesort(vector<string> &a, int l,int r)
{
    if (l>=r)
        return;
    int m = l + (r - l) / 2;
    mergesort(a, l, m);
    mergesort(a, m+1, r);
    merge(a, l, m, r);
}
    string arrangeWords(string str) {
        if (str.empty()) return "";
        vector<string > dic;
        string tmp,res;
        for (auto &x:str){
            if (x==' ') dic.push_back(tmp),tmp.clear();
            else {
                if (x<='Z') x+=32;
                tmp+=x;
            }
        }
        dic.push_back(tmp);
        for (auto x:dic) cout<<x<<' ';
        bool flag=1;
        mergesort(dic, 0, dic.size()-1);

                

        dic[0][0]-=32;
        res+=dic[0];
        for (int i=1;i<dic.size();i++){
            res+=" "+dic[i];
        }
        return res;
    }
};
