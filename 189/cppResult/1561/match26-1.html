class Solution {
public:
    string arrangeWords(string text) {
        if(!text.size())
            return text;
        text[0] = tolower(text[0]);
        stringstream splitString;
        splitString << text;
        string oneWord;
        map<int, vector<string>> wordSize;
        string result;
        while(splitString >> oneWord) {
            wordSize[oneWord.size()].push_back(oneWord);
        }

        for(auto it = wordSize.begin(); it != wordSize.end(); it++) {
            for(int j = 0; j < (it -> second).size(); j++) {
                result += (it -> second)[j];
                result += " ";
            }
        }
        result = result.substr(0, result.size() - 1);
        result[0] = toupper(result[0]);
        return result;
    }
};
