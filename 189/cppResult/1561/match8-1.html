    bool mycmp(const pair<string, int> &a, const pair<string, int> &b) {
        if (a.first.length() != b.first.length())
            return a.first.length() < b.first.length();
        else
            return a.second < b.second;
    }


class Solution {
public:
    string arrangeWords(string s) {
        vector<pair<string, int>> d;
        int n = s.length(), idx = 0;
        string cur, ans;
        for (int i = 0; i <= n; ++i) {
            if (i == n || s[i] == ' ') {
                // cout << cur << " " << idx << endl;
                d.push_back({cur, idx});
                cur.clear();
                ++idx;
                
            } else {
                cur += tolower(s[i]);
            }

        }
        
        sort(d.begin(), d.end(), mycmp);
        
        ans = d[0].first;
        ans[0] = toupper(ans[0]);
        for (int i = 1; i < d.size(); ++i) {
            ans += " ";
            ans += d[i].first;
        }
        return ans;
    }
};
