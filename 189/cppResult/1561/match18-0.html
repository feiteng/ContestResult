class Solution {
public:
    string arrangeWords(string text) {
        istringstream iss(text);
        vector<pair<string, int>> v;
        string t;
        int cnt = 0;
        while(iss>>t) {
            if (isupper(t[0])) t[0] = tolower(t[0]);
            v.push_back({t, cnt++});
        }
        
        sort(begin(v), end(v), [](const auto & a, const auto & b) {
            return a.first.size() < b.first.size() || a.first.size() == b.first.size() && a.second < b.second;
        });
        
        string res;
        for (auto & x: v) {
            res += x.first + " ";
        }
        
        res.pop_back();
        res[0] = toupper(res[0]);
        return res;
    }
};
