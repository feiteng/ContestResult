vector<string> words;
vector<int> idx;
vector<int> len;

inline bool cmp(int a, int b){
    if (len[a] != len[b]) return len[a] < len[b];
    return a < b;
}

class Solution {
public:
    string arrangeWords(string text) {
        words.clear(); idx.clear(); len.clear();
        
        string cur = "";
        text += " ";
        for (auto c: text){
            if (c == ' '){
                words.push_back(cur);
                cur = "";
            }else{
                if ('A' <= c && c <= 'Z') c = c - 'A' + 'a';
                cur += c;
            }
        }
        
        int n = words.size();
        for (int i = 0; i < n; i++){
            idx.push_back(i);
            len.push_back(words[i].length());
        }
        
        sort(idx.begin(), idx.end(), cmp);
        
        string ans = "";
        for (int i = 0; i < n; i++){
            if (i > 0) ans += " ";
            ans += words[idx[i]];
        }
        ans[0] = ans[0] - 'a' + 'A';
        
        return ans;
    }
};
