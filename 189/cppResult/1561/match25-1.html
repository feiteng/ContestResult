class Solution {
public:
    string arrangeWords(string text) {
        vector<pair<string, int>> arr;
        string tmp = "";
        if (text.back() != ' ') text += ' ';
        int n = text.length();
        text[0] = tolower(text[0]);
        for (int i = 0; i < n; ++i) {
            if (text[i] == ' ') {
                arr.push_back({tmp, arr.size()});
                tmp = "";
                continue;
            }
            tmp += text[i];
        }
        

        sort(arr.begin(), arr.end(), [](const pair<string, int> &a, const pair<string, int> &b){
            if (a.first.length() == b.first.length()) return a.second < b.second;
            return a.first.length() < b.first.length();
        });
        string ans = "";
        for (auto &str : arr) {
            if (ans.length() > 0) ans += ' ';
            ans += str.first;
        }
        ans[0] = toupper(ans[0]);
        return ans;
    }
};
