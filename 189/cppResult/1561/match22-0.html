class Solution {
public:
    string arrangeWords(string text) {
        map<int, vector<string>> m;

        std::vector<std::string> result;
        std::istringstream iss(text);
        for(std::string s; iss >> s; )
            result.push_back(s);
        for(int i = 0; i < result.size(); i++){
            result[i][0] = tolower(result[i][0]);
            m[result[i].length()].push_back(result[i]);
        }
        int counter = 0;
        string ans = "";
        for(auto cur : m){
            for(string str : cur.second){
                ans += str;
                ans += " ";
            }
                
            
        }
        ans[0] = toupper(ans[0]);
        ans.pop_back();
        return ans;
    }
};
