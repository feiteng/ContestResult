class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        longest = -1
        d = {}
        for i, c in enumerate(s):
            if c in d:
                longest = max(longest, i - d[c] - 1)
            else:
                d[c] = i
        return longest
