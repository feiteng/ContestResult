class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        tracker = {}
        max_diff = -1
        for i, ch in enumerate(s):
            if ch not in tracker:
                tracker[ch] = i
            else:
                max_diff = max(max_diff, i-tracker[ch]-1)
        return max_diff
