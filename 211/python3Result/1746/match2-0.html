class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        first_idx = {}
        res = -1
        for i, c in enumerate(s):
            if c not in first_idx:
                first_idx[c] = i
            else:
                res = max(res, i - first_idx[c] - 1)
        return res
