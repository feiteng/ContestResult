class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        res = -1
        pos = {}
        
        for i, c in enumerate(s):
            if c in pos:
                res = max(res, i - pos[c] -1)
            else:
                pos[c] = i
            
        return res
