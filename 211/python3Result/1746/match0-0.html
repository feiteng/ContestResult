class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:

        ch_to_first_appear_idx = {}
        ans = -1

        for idx, ch in enumerate(s):
            if ch not in ch_to_first_appear_idx:
                ch_to_first_appear_idx[ch] = idx
            else:
                ans = max(ans, idx - ch_to_first_appear_idx[ch] - 1)

        return ans
