class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        slist = {}
        val = 0
        for i,c in enumerate(s):
            if c in slist:
                val = max(val, i-slist[c])
            else:
                slist[c] = i
        return val-1
