class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        res = -1
        indices = {}
        for i, ch in enumerate(s):
            if ch in indices:
                res = max(res, i-indices[ch]-1)
            else:
                indices[ch] = i
        return res
