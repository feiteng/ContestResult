class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        last = {}
        res = -1
        for i, ch in enumerate(s):
            if not ch in last:
                last[ch] = i
            else:
                res = max(res, i - 1 - last[ch])
        return res
