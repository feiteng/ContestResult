class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        char_index_map = {}
        
        result = -1
        for index, char in enumerate(s):
            if not char in char_index_map:
                char_index_map[char] = index
            else:
                result = max(result, index - char_index_map[char] - 1)
                
        return result
