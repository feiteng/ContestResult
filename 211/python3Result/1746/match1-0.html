class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        save = {}
        res = -1
        for i, ch in enumerate(s):
            if ch not in save:
                save[ch] = i
            else:
                res = max(res, i - save[ch] - 1)
        return res
