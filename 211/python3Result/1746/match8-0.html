class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        dictt = {}
        to_ret = -1
        for i, c in enumerate(s) :
            if not c in dictt :
                dictt[c] = i
            else :
                to_ret = max(to_ret, i-dictt[c]-1)
        return to_ret
