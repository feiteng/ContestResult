class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        pos = {}
        res = -1
        for i,c in enumerate(s):
            if c not in pos:
                pos[c] = i
            else:
                res = max(res, i-pos[c]-1)
        return res
