class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        d = {}
        max_length = -1
        for i, ch in enumerate(s):
            if ch in d:
                max_length = max(max_length, i - d[ch] - 1)
            else:
                d[ch] = i
        return max_length
