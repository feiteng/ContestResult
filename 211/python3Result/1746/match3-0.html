class Solution:
    def maxLengthBetweenEqualCharacters(self, s: str) -> int:
        max_len = -1
        first_idx = {}
        
        for i, c in enumerate(s):
            if c not in first_idx:
                first_idx[c] = i
            else:
                max_len = max(i - first_idx[c] - 1, max_len)
        return max_len
