class UnionFind:
    def __init__(self, n):
        self.root = list(range(
n))

    def find(self, i):
        if i!=self.root[i]:
            self.root[i] = self.find(self.root[i])
        return self.root[i]

    def union(self, i, j):
        i, j = self.find(i), self.find(j)
        if i==j:
            return False
        elif i<j:
            i, j = j, i
        self.root[i] = j
        
return True

class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        UF = UnionFind(n+1)
        for d in range(threshold+1,n//2+2):
            cur = 2*d
            while cur<=n:
                UF.union(d,cur)
                cur += d
        return [UF.find(a)==UF.find(b) for a,b in queries]
