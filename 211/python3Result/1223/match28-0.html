class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        if threshold == 0: return [True]*(len(queries))
        
        parent = {i:i for i in range(1,n+1)}
        def findroot(i):
            if parent[i] != i:
                parent[i] = findroot(parent[i])
            return parent[i]
        
        def connect(i,j):
            rootj = findroot(j)
            rooti = findroot(i)
            if rooti != rootj:
                parent[rooti] = rootj
        
        for t in range(threshold+1,n//2+1):
            times = 2
            while t*times <= n:
                #print("connect",t,t*times)
                connect(t,t*times)
                #print(parent)
                times+=1
        
        res = []
        for a, b in queries:
            res.append(findroot(a)==findroot(b))
        
        #print(parent)
        return res
