class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:                
        def gcd(x, y):
            if x > y:
                x, y = y, x
            if x == 0:
                return y
            return gcd(y % x, x)

        pre = [i for i in range(n + 1)]

        def findPre(x):
            if pre[x] != x:
                pre[x] = findPre(pre[x])
            return pre[x]
        
        def unionPre(x, y):
            x = findPre(x)
            y = findPre(y)
            if x != y:
                if x > y:
                    x, y = y, x
                pre[x] = y
        
        for i in range(threshold + 1, n + 1):
            for j in range(i, n + 1, i):
                unionPre(i, j)

        ans = []
        for x, y in queries:
            x = findPre(x)
            y = findPre(y)
            ans.append(x == y)
        return ans
