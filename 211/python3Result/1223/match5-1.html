class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        parent = list(range(n + 1))
        def find(node):
            if parent
[node] != node:
                parent[node] = find(parent[node])
            return parent[node]
        
        def union(x, y):
            px, py = find(x), find(y)
            parent[px] = py
        
        def isconnected(x, y):
            return find(x) == find(y)
        
        for fac in range(threshold + 1, n):
            for i in range(2, n // fac + 2):
                if fac * i > n:
                    break
                union
(fac, fac * i)
                
        ans = []
        for i, j in queries:
            ans.append(isconnected(i, j))
        return ans
