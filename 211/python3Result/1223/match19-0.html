class Solution:
    def areConnected(self, n: int, T: int, queries: List[List[int]]) -> List[bool]:
        cc = [i for i in range(n+1)]
        
        def root(a):
            if cc[a] != a:
                cc[a] = root(cc[a])
            return cc[a]
        
        def join(a, b):
            ra = root(a)
            rb = root(b)
            if ra != rb:
                cc[ra] = rb
            return ra != rb
        
        for i in range(T+1, n//2 + 1):
            for b in range(i*2, n+1, i):
                join(i, b)
                
        return [(root(x) == root(y)) for (x, y) in queries]
