class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        parent = [i for i in range(n+1)]
        
def findpar(i):
            if(parent[i]!=i):
                parent[i] = findpar(parent[i])
            return parent[i]
        
        def union(x,y):
            par1, par2 = findpar(x), findpar(y)
            if par1!= par2:
                parent[par1] = par2
                
        for factor in range(threshold+1, n):
            for i in range(factor+1, n+1):
                if i%factor == 0:
                    union(factor, i)
        result =[]
        for num1, num2 in queries:
            
if findpar(num1) != findpar(num2):
                result.append(False)
            else:
                result.append(True)
        return result
