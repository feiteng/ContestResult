class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        import math
        p = [i for i in range(n)]
        if not threshold:
            return [True] * len(queries)
        
def find(x):
            if x != p[x]:
                p[x] = find(p[x])
            return p[x]
        
        def union(x, y):
            p_x = find(x)
            p_y = find(y)
            if p_x != p_y:
                p
[p_x] = p_y
                
        for i in range(threshold+1, n):
            for j in range(i,n+1,i):
                # print(i,j)
                union(i-1,j-1)
        res = []
        for x,y in queries:
            res.append(find(x-1)==find(y-1))
        return res
