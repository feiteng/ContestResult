        return self.parent[x]

class Solution:
    def areConnected(self, n: int, g: int, queries: List[List[int]]) -> List[bool]:
        if g == 0:
            return [True] * len(queries)
        if g >= n:
            return [False] * len(queries)
        
        dsu = DSU(n)
        
        for candidate in range(g+1, n+1):
            current = [t * candidate for t in range(1, n // candidate + 1)]
            for curr in current:
                dsu.union(candidate-1, curr-1)

        res = []
        for k in range(len(queries)):
            origin, destination = queries[k][0], queries[k][1]
            if origin > g and destination > g and dsu.find(origin-1) ==  dsu.find(destination-1):
                res.append(True)
            else:
                res.append(False)
        return res
