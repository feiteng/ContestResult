class Solution:
    def areConnected(self, n: int, t: int, Q: List[List[int]]) -> List[bool]:
        uf = dict()
        def find(i):
            uf.setdefault(i, i)
            
if uf[i] != i:
                uf[i] = find(uf[i])
            return uf[i]
        def union(i, j):
            fi, fj = find(i), find
(j)
            if fi != fj:
                uf[fi] = fj      
        for i in range(t + 1, n // 2 + 1):
            k = 2
            while i * k <= n:
                union(i, i * k)
                k += 1
        ret = [False] * len(Q)
        for i, [a, b] in enumerate(Q):
            if find(a) == find(b):
                ret[i] = True
        return ret
