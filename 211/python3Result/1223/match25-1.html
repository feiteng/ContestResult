class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        if threshold == 0:
            return [True] * len(queries)
        p = [*range(n+1)]
        rank = [0] * (n+1)
        
def find(x):
            if x != p[x]:
                p[x] = find(p[x])
            return p[x]
        def union(x, y):
            xx = find(x)
            yy = find(y)
            if xx == yy:
                return
            if rank[xx] >= rank[yy]:
                p[yy] = xx
                if rank[xx] == rank[yy]:
                    rank[xx] += 1
            else:
                p[xx] = 
yy
        for d in range(threshold+1, n+1):
            cur = d * 2
            while cur <= n:
                union(cur, d)
                cur += d
        return [find(x) == find(y) for x, y in queries]
