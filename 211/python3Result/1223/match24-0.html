class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        f = [i for i in range(n + 1)]
        
        def find(i):
            if f[i] != i:
                f[i] = find(f[i])
            return f[i]
        
        def merge(a, b):
            fa = find(a)
            fb = find(b)
            f[fa] = fb
        
        ys = collections.defaultdict(list)
        for i in range(1, n + 1):
            ys[i].append(i)
            for y in range(1, i // 2 + 2):
                if i % y == 0:
                    ys[y].append(i)
        
        for y in ys:
            if y > threshold:
                for a in ys[y][1:]:
                    merge(a, ys[y][0])
        
        res = []
        for a, b in queries:
            if find(a) == find(b):
                res.append(True)
            else:
                res.append(False)
        return res
