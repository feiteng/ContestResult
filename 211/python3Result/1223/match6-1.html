class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        anc = [i for i in range(n + 1)]
        
        def find_anc(i):
            if anc[i] == i:
                return i
            anc[i] = find_anc(anc[i])
            return anc[i]
        
        for fac in range(threshold + 1, n + 1):
            x = fac
            # print(anc)
            while x <= n:
                anc[find_anc(x)] = find_anc
(fac)
                x += fac
        
        ans = []
        for u, v in queries:
            ans.append(find_anc(u) == find_anc(v))
        return ans
