class DisjointSet:
    def __init__(self, n):
        self.parent = [i for i in range(n)]
        self.height = [1 for _ in range(n)]
        self.size = [1 for _ in range(n)]
        self.count =
 n
    
    def find(self, x):
        if x != self.parent[x]:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def union(self, x, y):
        p = self.find(x)
        q = self.find(y)
        if p == q:
            return
        father, son = p, q
        self.parent[son] = father
        self.height[father] = max(self.height[father], self.height[son] + 1)
        self.size[father] += self.size[son]
        self.count -= 1

    def get_size(self, x):
        return self.size[self.find(
x)]

class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        # 求整个图的所有边：求任意两点的最大公约数，看是否超过threshold
        # 并查集处理整个图
        # 处理query
        ds = DisjointSet(n + 1)
        for base in range(threshold + 1, n + 1):
            i = 2
            while i * base <= n:
                ds.union(base, base * i)
                i += 1
        res = []
        for a, b in queries:
            if ds.find(a) == ds.find(b):
                res.append(True)
            else:
                res.append(False)
        return res
