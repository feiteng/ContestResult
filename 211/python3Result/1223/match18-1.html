class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        p = [-1] * (n+1)
        
        
def find(u):
            if p[u] == -1: 
                return u
            p[u] = find(p[u])
            return p[u]
        
        def union(u, v):
            u = find(u)
            v = find(v)
            if u != v:
                p[u] = v
                
        for i in range(threshold+1, n+1):
            for j in range(i+i, n+1, i):
                union(i, j)
                
        return [
            find(u) == find(v)
            for u, v in queries
        ]
