class Solution:
    def gcd(self, a, b):
        while b:
            a, b = b, a%b
        return a
    
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        if threshold == 0:
            return [True for _ in range(len
(queries))]
        
        parents = list(range(n+1))
        
        def find(i):
            if parents[i] != i:
                parents[i] = find(parents[i])
            return parents[i]
            
        def union(i, j):
            p_i = find(i)
            p_j = find(j)
            if p_i != p_j:
                if p_i > p_j:
                    parents[p_j] = p_i
                else:
                    parents[p_i] = p_j        
        
        for cd in range(threshold+1, n//2 + 1):
            a = 1
            b = 2
            while a*cd < n:
                while b*cd <= n:
                    union(a*cd, b*cd)
                    b += 1
                a += 1
                b = 
a + 1

        res = []
        
        for a, b in queries:
            if find(a) == find(b):
                res.append(True)
            else:
                res.append(False)

        return res
