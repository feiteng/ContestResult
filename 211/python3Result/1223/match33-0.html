class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        if threshold == 0:
            return [True]*len(queries)
        dp = [-1]*(n+1)
        parent = [-1]*(n+1)
        for i in range(1, n+1):
            parent[i] = i
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        
        def union(x, y):
            h1, h2 = find(x), find(y)
            if h1 < h2:
                parent[h2] = h1
            elif h1 > h2:
                parent[h1] = h2
        for i in range(threshold+1, n+1):
            if dp[i] == -1:
                start = i
                while start <= n:
                    if dp[start] == -1:
                        dp[start] = i
                    union(i, start)
                    start += i
        res = []
        for query in queries:
            x, y = query
            if find(x) != find(y):
                res.append(False)
            else:
                res.append(True)
        return res
