class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        par = [i for i in range(n+1)]
        def find(u):
            if par[u] == u:
                return u
            par[u] = find(par[u])
            return par[u]
        def merge(u,v):
            u = find(u)
            v = find(v)
            if u!= v:
                par[v] = u
                
        for i in range(threshold+1,n):
            for j in range(i+i,n+1,i):
                merge(i,j)
        ans = []
        for i in queries:
            ans.append(find(i[0])==find(i[1]))
        return ans
