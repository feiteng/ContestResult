        parent = list(range(n + 1))
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        
        for i in range(2, n + 1):
            for k in range(2, int(i ** 0.5) + 1):
                if i % k == 0:
                    rooti = find(i)
                    if k > threshold:
                        rootk = find(k)
                        if rootk != rooti:
                            parent[rootk] = rooti
                    if i // k > threshold:
                        rootik = find(i // k)
                        if rootik != rooti:

        res = []
        for a, b in queries:
            if find(a) == find(b):
                res.append(True)
            else:
                res.append(False)
