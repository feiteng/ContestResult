class Solution:
    def areConnected(self, n: int, threshold: int, queries: List[List[int]]) -> List[bool]:
        if threshold == 0:
            return [True] * (len(queries))
        parents = list(range
(n + 1))
        def find(a):
            if parents[a] != a:
                parents[a] = find(parents[a])
            return parents[a]
        def union(a1, a2):
            p1 = find(a1)
            p2 = find(a2)
            if p1 != p2:
                parents[p1] = p2
        
        # for i in range(threshold, n + 1):
        #     for j in range(threshold, i):
        #         if math.gcd(i, j) > threshold:
        #             # print(i, j)
        #             union(i, j)
        
        for t in range(threshold + 1, n):
            for divisor in range(2, (math.ceil
((n + 1) / t))):
                if t * divisor <= n:
                    union(t, t * divisor)
            
        
        res = []
        for i, j in queries:
            if find(i) == find(j):
                res.append(True)
            else:
                res.append(False)
        # print(parents)
        return res
