class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        n = len(s)
        smallest = s
        stack = [s]
        visited = set(stack)
        while stack:
            cur = stack.pop()
            smallest = min(smallest, cur)
            
            # a
            arr = [int(c) for c in cur]
            for i in range(1, n, 2):
                arr[i] = (arr[i] + a) % 10
            a_next = "".join([str
(a) for a in arr])
            if a_next not in visited:
                visited.add(a_next)
                stack.append(a_next)
            
            # b
            b_next = cur[n - b:] + cur[:n-b]
            if b_next not in visited:
                visited.add(b_next)
                stack.append(b_next)
                
        return smallest
