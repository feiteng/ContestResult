class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        visited = set([s])
        queue = collections.deque([s])
        
        def add(s):
            ans = []
            for i in range(len(s)):
                if i % 2 == 1:
                    c = (int(s[i]) + a) % 10
                    ans.append(str(c))
                else:
                    ans.append(s[i])
            return ''.join(ans)    
            
        def move(s):
            return s[len(s) - b:] + s[0: len(s) - b]
        
        while queue:
            cur = queue.popleft()
            one = add(cur)            
            two = move(cur)
            if one not in visited:
                visited.add(one)
                queue.append(one)
            if two not in visited:
                visited.add(two)
                queue.append(two)      
        
        ans = list(visited)
        ans.sort()
        return ans[0]
