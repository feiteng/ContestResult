class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        
        def insa(s):
            rst = ""
            for i in range(len(s)):
                if i%2 == 1:
                    rst += str((int(s[i])+a)%10)
                else:
                    rst += s[i]
            return rst
        
        def insb(s):
            return s[-b:] + s[:-b]
        
        memo = set()
        # print(insb(s))
        def dfs(s):
            sa = insa(s)
            # print("sa", sa)
            if sa not in memo:
                memo.add(sa)
                dfs(sa)
            sb = insb(s)
            # print("sb", sb)
            if sb not in memo:
                memo.add(sb)
                dfs(sb)
        dfs(s)
        memoList = list(memo)
        memoList.sort()
        return memoList[0]
