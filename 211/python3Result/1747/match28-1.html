class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        
        n = len(s)
        
        def rotate(s1, x = b):
            return s1[-b:] + s1[:-b]
        
        def add(s1, x = a):
            s2 
= ""
            for i in range(n):
                if i&1:
                    s2 += str((int(s1[i])+a) % 10)
                else:
                    s2 += s1[i]
            return s2
        
        S = set()
        
        def go(s1 = s):
            if s1 in S:
                return
            S.add(s1)
            go(rotate(s1))
            go(add(s1))
        
        go()
        return sorted(S)[0]
