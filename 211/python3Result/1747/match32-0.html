class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        seen = set()
        q = [s]
        res = s
        seen.add(s)

        
        def rotate(s):
            
            r = s[-1 * b: ] + s[:-1 * b]
                    
            return r
        
        def add_a(s):
            res = []
            for index, i in enumerate(s):
                if index % 2 == 1:
                    res.append(str((int(i) + a) % 10))
                else:
                    res.append(i)
            return "".join(res)
        
        
        while q:
            cur = q.pop()
            if cur < res:
                res = cur
            cur_1 = add_a(cur)
            cur_2 = rotate(cur)
            if not cur_1 in seen:
                seen.add(cur_1)
                q.append(cur_1)
            if not cur_2 in seen:
                seen.add(cur_2)
                q.append(cur_2)
        return res
