class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        visited = {s, s[-b:]+s[:len(s)-b]}
        pool = {s, s[-b:]+s[:len(s)-b]}
        res = s
        while pool:
            nxt = set()
            for state in pool:
                res = min(res, state)
                s1 = ''
                for i, char in enumerate(state):
                    if i%2 != 0:
                        s1 += str((int(char)+int(a))%10)
                    else:
                        s1 += char
                if s1 not in visited:
                    visited.add(s1)
                    nxt.add(s1)
                s2 = s1[-b:]+s1[:len(s)-b]
                if s2 not in visited:
                    visited.add(s2)
                    nxt.add(s2)
            pool = nxt
        return res
