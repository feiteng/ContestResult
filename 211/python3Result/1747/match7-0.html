class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        res = set()
        
        def _dfs(s):
            if s in res:
                return
            
            res.add(s)
            
            # first choice: rotate
            ns1 = s[-b:] + s[:-b]
            _dfs(ns1)
            
            # second choice: add 
            ns2 = ""
            for i in range(len(s) // 2):
                ns2 += s[2*i]
                ns2 += str((int(s[2*i+1])+a) % 10)
            _dfs(ns2)
            
        _dfs(s)
        return min(res)
