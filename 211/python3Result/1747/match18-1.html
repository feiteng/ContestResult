class Solution:
    def findLexSmallestString(self, s1: str, a: int, b: int) -> str:
        
        res = s1
        odd = 25
        even = 1
        n = len(s1)
        if b&1:
            even = 25
        rotate = n-b
        for cnt_odd in range(odd):
            for cnt_even in range(even):
                x = []
                s = s1[::]
                for i in range
(n):
                    if i%2:
                        x1 = int(s[i])
                        for j in range(cnt_odd):
                            x1 = (x1+a)%10
                        x.append(str(x1))
                    else:
                        x1 = int(s[i])
                        for j in range(cnt_even):
                            x1 = (x1+a)%10
                        x.append(str(x1))



                new_s = "".join(x)
                for i in range(n):
                    new_s = new_s[rotate:] + new_s[:rotate]
                    res = min(res,new_s)
        return res
