class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        visited_s = set()
        min_str = s
        
        def next_step(cur_s):
            nonlocal min_str
            
            if cur_s in visited_s:
                return None
            
            visited_s.add(cur_s)
            if cur_s < min_str:
                min_str = cur_s

            
            next_s1 = ''
            for i in range(len(cur_s)):
                if i % 2 == 0:
                    next_s1 = next_s1 + cur_s[i]
                else:
                    next_s1 = next_s1 + str((int(cur_s[i]) + a) % 10)
            next_step(next_s1)
            
            next_s2 = cur_s[(len(s) - b):] + cur_s[:(len(s) - b)]
            next_step(next_s2)
        
        next_step(s)
        return min_str
