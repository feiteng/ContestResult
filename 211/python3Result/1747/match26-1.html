class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        b %= len(s)
        visited = set([s])
        q = collections.deque([s])
        while q:
            strVal = q.popleft()
            res1 = strVal[-b:] + strVal[:-b]
            res2 = ''.join(map(str, [(int(c) + a) % 10 if i&1 else int(c) for i, c in enumerate
(strVal)]))
            print(strVal, res1, res2)
            if res1 not in visited:
                q.append(res1)
                visited.add(res1)
            if res2 not in visited:
                q.append(res2)
                visited.add(res2)
        return min(visited)
