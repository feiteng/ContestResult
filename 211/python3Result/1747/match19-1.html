class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        # generate all possible?
        ans = set()
        
        def dfs(s):
            if s in ans:
                return
            
            ans.add(s)
            
            # get next cyclic shift
            news = s[len(s)-b:] + s[:-b]
            dfs(news)
            
            # then change values
            news = s
            t = True
            while t or (news not in ans):
                ans.add(news)
                news = list(news)
                for i in range(1,len(news),2):
                    news[i] = str( (int(news[i])+a) % 10 )
                news = "".join(news)
                dfs(news)
                t = False
                
        
        dfs(s)
        #print(ans)
        return min(ans)
