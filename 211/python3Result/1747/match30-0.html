class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        n = len(s)
        visited = set()
        visited.add(s)
        q = collections.deque()
        q.append(s)
        res = s
        while q:
            size = len(q)
            for _ in range(size):
                cur = q.popleft()
                if cur < res:
                    res = cur
                new_s = []
                for i in range(len(s)):
                    if i % 2 == 0:
                        new_s.append(cur[i])
                    else:
                        new_s.append(str((int(cur[i]) + a) % 10))
                n_s = ''.join(new_s)
                if n_s not in visited:
                    visited.add(n_s)
                    q.append(n_s)
                n_s = cur[n - b:] + cur[:n - b]
                if n_s not in visited:
                    visited.add(n_s)
                    q.append(n_s)
        return res
