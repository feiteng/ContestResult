class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        BFS = [s]
        been = {s:1}
        for s in BFS:
            ls = list(s)
            
for i in range(1,len(ls),2):
                ls[i] = str((int(ls[i])+a)%10)
            sa = ''.join(ls)
            if sa not in been:
                BFS.append(sa)
                been[sa]=1
            sb = s[-b:]+s[:-b]
            if sb not in been:
                BFS.append(sb)
                been[sb]=1
        return min(BFS)
