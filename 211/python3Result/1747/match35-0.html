import collections

class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        n = len(s)
        b = b % n
        
        def add(nums, k):
            result = []
            for i in range(len(nums)):
                if i % 2 == 0:
                    result.append(str(nums[i]))
                else:
                    result.append(str((int(nums[i]) + k) % 10))

            return result
    
    
        def rotate(nums, k):
            n = len(nums)
            shift = k % n
            gcd = math.gcd(shift, n)

            for j in range(gcd):
                i, temp = shift + j, nums[j]
                while i != j:
                    nums[i], temp, i = temp, nums[i], (i + shift) % n

                nums[i] = temp
                
            return nums
        
        result = s
        used = set()
        queue = collections.deque([s])
        while len(queue) > 0:
            temp = queue.popleft()
            if temp < result:
                    result = temp
            added = ''.join(add(list(temp), a))
            rotated = ''.join(rotate(list(temp), b))
            if added not in used:
                queue.append(added)
                used.add(added)
            if rotated not in used:
                queue.append(rotated)
                used.add(rotated)
                
        return result
