class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        addmap = {str(i): str((i+a)%10) for i in range(10)}
        found = set()
        self.dfs(s, a, b, addmap, found)
        
        return min(found)
        
    def dfs(self, s, a, b, addmap, found):
        # add
        ss = "".join([addmap[n] if i%2==1 else n for i, n in enumerate(s)])
        if ss not in found:
            found.add(ss)
            self.dfs(ss, a, b, addmap, found)
            
        # rotate
        ss = s[-b:]+s[:-b]
        if ss not in found:
            found.add(ss)
            self.dfs(ss, a, b, addmap, found)
