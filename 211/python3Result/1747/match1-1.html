class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        seen = set()
        ans = s
        from collections import deque
        queue = deque()
        queue.append(s)
        while queue:
            size = len
(queue)
            for i in range(size):
                arr = queue.pop()
                flag = False
                res = []
                for num in arr:
                    if flag:
                        num = str((int(num) + a) % 10)
                    res.append(num)
                    flag = not flag
                temp_res = ''.join(res)
                ans = min(ans, temp_res)
                if temp_res not in seen:
                    queue.append(temp_res)
                after = arr[b:] + arr[0:b]
                temp_res = ''.join(after)
                ans = min(ans, temp_res)
                if temp_res not in seen:
                    queue.append(temp_res)
                seen.add(arr)
        return ans
