class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        queue = collections.deque([])
        queue.append(s)
        visited = set()
        visited.add(s)
        
        res = s
        while queue:
            s = queue.popleft()
            res = min(res, s)
            s = list(s)
            temp = s[:]
            for i in range(1, len(s), 2):
                temp[i] = str((int(temp[i]) + a) % 10)
            temp1 = ''.join(temp)
            if temp1 not in visited:
                queue.append(temp1)
                visited.add(temp1)
            last = s[-b:]
            s[b:] = s[:-b]
            s[:b] = last
            temp2 = ''.join(s)
            if temp2 not in visited:
                queue.append(temp2)
                visited.add(temp2)
        
        return res
