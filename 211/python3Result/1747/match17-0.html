class Solution:
    def _a(self, s, a):
        lst = list(s)
        for i in range(1, len(s), 2):
            lst[i] = str((int(lst[i]) + a) % 10)
        x = ''.join(lst)
        return x

    def _b(self, s, b):
        x = s[-b:] + s[:len(s) - b]
        return x

    def _dfs(self, s, a, b):
        if s in self.vis:
            return
        self.vis.add(s)
        self.result = min(self.result, s)
        self._dfs(self._a(s, a), a, b)
        self._dfs(self._b(s, b), a, b)

    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        self.result = s
        self.vis = set()
        self._dfs(s, a, b)
        return self.result
