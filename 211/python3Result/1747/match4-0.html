class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        n = len(s)
        ans = set()
        queue = [s]
        while queue:
            newQueue = []
            for s in queue:
                scopy = list(s)
                for k in range(1, n + 1, 2):
                    scopy[k] = str((int(scopy[k]) + a) % 10)
                ss = ''.join(scopy)
                if ss not in ans:
                    ans.add(ss)
                    newQueue.append(ss)

                scopy = list(s)
                ss = ''.join(scopy[n - b:] + scopy[:n-b])
                if ss not in ans:
                    ans.add(ss)
                    newQueue.append(ss)
            queue = newQueue

        ans = list(ans)
        ans.sort()
        return ans[0]
