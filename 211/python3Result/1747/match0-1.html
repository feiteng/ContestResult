class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        
        
        seen = {s, s[-b:] + s[:len(s)-b]}
        pool = {s, s[-b:] + s[:len(s)-b]}
        res = s
        while len(pool) > 0:
            nextlevel 
= set()
            for state in pool:
                res = min(res, state)
                s1 = ''
                for i, char in enumerate(state):
                    if i % 2:
                        s1 += str((int(char)+int(a)) % 10)
                    else:
                        s1 += char
                if s1 not in seen:
                    seen.add(s1)
                    nextlevel.add(s1)
                s2 = s1[-b:] + s1[:len(s)-b]
                if s2 not in seen:
                    nextlevel.add(s2)
                    seen.add(s2)
            pool = nextlevel
        return res
