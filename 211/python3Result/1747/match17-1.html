class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        visited = set()
        
def add(s):
            s = list(s)
            for i in range(1, len(s), 2):
                s[i] = str((int(s[i]) + a) % 10)
            s = ''.join(s)
            return s
        def rotate(s):
            s = s[-b:] + s[:-b]
            return s
        def dfs(s):
            if s in visited: return s
            visited.add(s)
            add_res = rot_res = s
            add_s = add(s)
            # if add_s < s: 
            add_res= dfs(add_s)
            rot_s = rotate(s)
            # if rot_s < s: 
            rot_res = dfs(rot_s)
            return min(s, add_res, rot_res)
        
        return dfs(s)
