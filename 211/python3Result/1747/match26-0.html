class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        stack = [s]
        visited = set([s])
        while stack:
            cur = stack.pop()
            next1 = cur[b:] + cur[:b]
            
            update = False
            new_s = []
            for char in cur:
                if update:
                    new_s.append((int(char) + a) % 10)
                else:
                    new_s.append(int(char))
                update = not update
            next2 = "".join([str(c) for c in new_s])
            if next1 not in visited:
                stack.append(next1)
                visited.add(next1)
            if next2 not in visited:
                stack.append(next2)
                visited.add(next2)
        return sorted(visited)[0]
