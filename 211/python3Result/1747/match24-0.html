class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        def construct(single, a, b):
            ret = set()
            ret.add(single)
            q = []
            q.append(single)
            
            while len(q) > 0:
                #what can you do
                curr = q[0]
                del q[0]
                
                segs = [int(c) for c in curr]
                #i can add a to odd
                for i in range(len(segs)):
                    if i % 2 == 1:
                        segs[i] += a
                        segs[i] %= 10
                        pass
                    pass
                cand = ''.join(str(x) for x in segs)
                if cand not in ret:
                    ret.add(cand)
                    q.append(cand)
                #i can also do shift
                cand1 = curr[-b: ] + curr[0 : len(curr) - b]
                if cand1 not in ret:
                    ret.add(cand1)
                    q.append(cand1)
                    
            return ret
        
        return min(construct(s,a,b))
