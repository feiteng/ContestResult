from collections import deque


class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        sz = len(s)
        step = b % sz

        ans = None
        seen = set([s])
        q = deque([s])
        while q:
            cur = q.popleft()
            if ans is None or cur < ans:
                ans = cur
            lcur = list(cur)
            for i in range(1, sz, 2):
                lcur[i] = chr(((ord(lcur[i]) - ord('0') + a) % 10) + ord('0'))
            ns = ''
.join(lcur)
            if ns not in seen:
                seen.add(ns)
                q.append(ns)
            ns = cur[step:] + cur[:step]
            if ns not in seen:
                seen.add(ns)
                q.append(ns)

        return ans
