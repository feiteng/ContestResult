class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:

        def change(s, a):
            _s = ''
            for i in range(len(s)):
                if i & 1:
                    _s += str((int(s[i])+a) % 10)
                else:
                    _s += s[i]
            return _s
        
        q = [s, ]
        vis = set([s])
        
        while q:
            ss = q.pop()
            s1 = change(ss, a)
            s2 = ss[-b:] + ss[:-b]
            
            if s1 not in vis:
                q.append(s1)
                vis.add(s1)
                
            if s2 not in vis:
                q.append(s2)
                vis.add(s2)
        return sorted(list(vis))[0]
