class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        visited = set()
        def helper(s, a, b):
            if s in visited:
                return
            visited.add(s)
            t = s[b:] + s[:b]
            helper(t, a, b)
            x = ""
            for i in range(len(s)):
                if i%2:
                    x += str((int(s[i])+a)%10)
                else:
                    x += s[i]
            helper(x, a, b)
            return
        helper(s, a, b)
        return min(list(visited))
