def get(s,k,k1,a,b):
    x = []
    for i in range(len(s)):
        if i%2:
            x1 = int(s[i])
            for j in range(k):
                x1 = (x1+a)%10
            x.append(str(x1))
        else:
            x1 = int(s[i])
            for j in range(k1):
                x1 = (x1+a)%10
            x.append(str(x1))
        


    s = "".join(x)
    ans = s
    n = len(s)
    x = (n-b)
    for i in range(n):
        s = s[x:] + s[:x]
        ans = min(ans,s)
    return ans
            
class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        
        ans = s
        mx = 1
        if b&1:
            mx = 20
        for i in range(20):
            for j in range(mx):
                ans = min(ans,get(s,i,j,a,b))
        return ans
