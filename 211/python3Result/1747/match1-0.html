class Solution:
    
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        import queue
        seen = set()
        res = s
        q = queue.LifoQueue()
        q.put(s)
        while q.qsize() > 0:
            size = q.qsize()
            for i in range(size):
                numbers = q.get()

                # add
                is_add = False
                after_add = []
                for num in numbers:
                    if is_add:
                        num = str((int(num) + a) % 10)
                    after_add.append(num)
                    is_add = not is_add
                temp_res = ''.join(after_add)
                res = min(res, temp_res)
                if temp_res not in seen:
                    q.put(temp_res)

                # roll
                after_roll = numbers[b:] + numbers[0:b]
                temp_res = ''.join(after_roll)
                res = min(res, temp_res)
                if temp_res not in seen:
                    q.put(temp_res)

                seen.add(numbers)
        return res
