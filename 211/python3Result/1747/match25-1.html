class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        N = len(s)
        seen 
= set()
        res = "9" * N
        
        def helper(s):
            nonlocal res
            if s in seen:
                return
            seen.add(s)
            if s < res:
                res = s
            # a
            new_s = ""
            for i in range(N):
                if i % 2 == 0:
                    new_s += s[i]
                else:
                    c = (int(s[i]) + a) % 10
                    new_s += str(c)
            helper(new_s)
            
            # b
            new_s = s[N-b:]+s[:N-b]
            helper(new_s)
            
            
        helper(s)
        return res
