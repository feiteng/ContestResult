from collections import deque
class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        res = s
        N = len(s)
        q = deque([s])
        visited = {s}
        
while q:
            cur = q.popleft()
            res = min(res,cur)
            n1 = list(cur)
            for i in range(1,N,2):
                n1[i] = str((int(n1[i])+a)%10)
            n1 = ''.join
(n1)
            if n1 not in visited:
                visited.add(n1)
                q.append(n1)
            n2 = cur[b:]+cur[:b]
            if n2 not in visited:
                visited.add(n2)
                q.append(n2)
        return res
