class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        def add(s, a):
            return ''.join([str((int(e)+int(a))%10) if i%2==1 else e for i,e in enumerate(s)])
        def rotate(s, b):
            return s[b:]+s[:b]
        
        visited = set()
        ans = s
        qu = [s]
        while qu:
            x = qu.pop()
            visited.add(x)
            ans = min(ans, x)
            x1 = add(x, a)
            if x1 not in visited:
                visited.add(x1)
                qu.append(x1)
            x2 = rotate(x,b)
            if x2 not in visited:
                visited.add(x2)
                qu.append(x2)
            #print(x, x1, x2)
        return ans
