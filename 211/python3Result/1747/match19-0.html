class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        seen = set()
        
        def dfs(curStr):
            if curStr in seen:
                return 
            
            seen.add(curStr)
            str1 = curStr[len(s)-b:] + curStr[:len(s)-b]
            str2 = ""
            for i, c in enumerate(curStr):
                if i == 1 or i %2 ==1:
                    str2 += str((int(c)+a)%10)
                else:
                    str2 += c
            
            dfs(str1)
            dfs(str2)
        
        dfs(s)
        
        return sorted(seen)[0]
