class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        def move(s):
            return s[b:] + s[:b]
        def add(s):
            ans = []
            for i, c in enumerate(s):
                if i % 2 == 0: ans.append(c)
                else:
                    ans.append(str((int(c) + a) % 10))
            return ''.join(ans)
        
        seen = set()
        q = [s]
        ans 
= s
        # bfs
        while q:
            n_q = []
            for s in q:
                n_s = move(s)
                if n_s not in seen:
                    seen.add(n_s)
                    n_q.append(n_s)
                    ans = min(ans, n_s)
                n_s = add(s)
                if n_s not in seen:
                    seen.add(n_s)
                    n_q.append(n_s)
                    ans = min(ans, n_s)
            q = n_q
        return ans
