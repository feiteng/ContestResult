class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        def opa(s):
            return ''.join([c if i % 2 == 0 else str((int(c) + a) % 10) for i, c in enumerate(s)])
        
        def opb(s):
            return s[-b:] + s[:-b]
        
        ss = set([s])
        q = [s]
        while q:
            s = q.pop()
            sa 
= opa(s)
            sb = opb(s)
            if sa not in ss:
                ss.add(sa)
                q.append(sa)
            if sb not in ss:
                ss.add(sb)
                q.append(sb)
        return min(ss)
