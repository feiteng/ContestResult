from collections import defaultdict
class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        def str_ror(s,k):
            length = len(s)
            k = k%length
            return s[length-k:] + s[:length-k]
        def add(s,a):
            n = len(s)
            ss = ""
            for i in range(n):
                if i%2:
                    ss = ss + str((a+int(s[i]))%10)
                else:
                    ss = ss + s[i]
            return ss
        d=defaultdict(int)
        def dfs(s):
            # print(s)
            if d[s]:
                return
            # print(s)
            d[s]=1
            dfs(str_ror(s,b))
            dfs(add(s,a))
        dfs(s)
        # print(d)
        # print(str_ror(s,b))
        return min(d)
