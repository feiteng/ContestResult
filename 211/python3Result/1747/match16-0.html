class Solution:
    def add(self, curr, a):
        res = list(curr)
        for i in range(1, len(curr), 2):
            res[i] = str((int(res[i]) + int(a))%10)
        return ''.join(res)
    
    def rotate(self, curr, b):
        return curr[-b:] + curr[:-b]
    
    def create(self, curr, visited, a, b):
        if curr in visited:
            return
        visited.add(curr)
        if int(curr) < self.min:
            self.min = int(curr)
            self.minstr = curr
        self.create(self.add(curr, a), visited, a, b)
        self.create(self.rotate(curr, b), visited, a, b)
        
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        self.minstr = s
        self.min = int(s)
        visited = set()
        self.create(s, visited, a, b)
        
        
        return self.minstr
