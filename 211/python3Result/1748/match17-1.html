class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        arr = zip(scores, ages)
        arr = sorted(arr, key = lambda x: (x[1], x[0]))
        
        res = 0
        dp 
= [0] * len(scores)
        dp[0] = arr[0][0]
        for i in range(len(arr)):
            dp[i] = arr[i][0]
            for j in range(i):
                if arr[j][0] <= arr[i][0] or arr[j][1] == arr[i][1]:
                    dp[i] = max(dp[i], dp[j] + arr[i][0])
        #print(dp)
        return max(dp)
