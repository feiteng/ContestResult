class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        t = []
        for i,a in enumerate(ages):
            t.append((a, scores[i]))
        s = sorted(t)
        dp = [0] * len(s)
        dp[0] = s[0][1]
        for i in range(len(dp)):
            dp[i] = s[i][1]
            for j in range(i-1, -1, -1):
                if s[i][0] >= s[j][0] and s[i][1] >= s[j][1]:
                    dp[i] = max(dp[i], s[i][1]+dp[j])
        #print(dp)
        return max(dp)
