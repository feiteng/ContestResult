class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        players = sorted(list(zip(ages, scores)))
        opt = [_[1] for _ in players] # opt[n] the maximum score of choosing n and choose from [0..n]
        # print(players)
        for i in range(1, len(players)):
            for j in range(i):
                # if not conflict
                if players[j][0] < players[i][0] and players[j][1] > players[i][1]:
                    pass
                else:
                    opt[i] = max(opt[i], opt[j] + players[i][1])
        return max(opt)
