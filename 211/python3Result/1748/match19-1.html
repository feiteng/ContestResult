class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        N = len(scores)
        merge = sorted([[ages[i], scores[i]] for i in range(N)])
        dp = [0 for i in range
(N)]
        for i in range(N):
            dp[i] = merge[i][1]
            for j in range(i-1, -1, -1):
                if merge[i][1] >= merge[j][1]:
                    dp[i] = max(dp[i], merge[i][1] + dp[j])
        return max(dp)
