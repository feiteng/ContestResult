class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        n = len(scores)
        idx = sorted(range(n), key = lambda x: (ages[x], scores[x]))
        scores = [scores[i] for i in idx]
        ages = [ages[i] for i in idx]
        dp = [0] * n
        for i in range(n):
            for j in range(i):
                if scores[j] <= scores[i]:
                    dp[i] = max(dp[i], dp[j])
            dp[i] += scores[i]
        return max(dp)
