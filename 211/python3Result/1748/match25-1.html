class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        N = len(ages)
        nums = [(ages[i],scores[i]) for i in range(N)]
        nums.sort(key=lambda x:(x[0],x[1]))
        dp = [nums[i][1] for i in range
(N)]
        for i in range(1,N):
            for j in range(i):
                if nums[i][1]>=nums[j][1]:
                    dp[i] = max(dp[i],dp[j]+nums[i][1])
        return max(dp)
