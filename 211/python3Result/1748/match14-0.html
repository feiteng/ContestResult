class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        sa = [(ages[i], s) for i, s in enumerate(scores)]
        sa.sort()
        f = [s for a, s in sa]
        for i in range(len(f)):
            for j in range(i):
                if sa[j][1] <= sa[i][1] or sa[j][0] == sa[i][0]:
                    f[i] = max(f[i], f[j] + sa[i][1])
        return max(f)
