class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        
        data = [(scores[i], ages[i]) for i in range(len(scores))]
        data.sort
()
        dp = [0] * len(scores)
        # print(data)
        for i, (score, age) in enumerate(data):
            total_score = score
            for j in range(0, i):
                if data[j][1] <= age:
                    total_score = max(total_score, dp[j] + score)
            dp[i] = total_score
        # print(dp)
        return max(dp)
