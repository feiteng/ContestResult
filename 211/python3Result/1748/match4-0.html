class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        data = sorted([(x, y) for x, y in zip(scores, ages)], key=lambda x: (x[1], x[0]))
        n = len(scores)
        dp = [0] * n 
        for i in range(n):
            dp[i] = data[i][0]
            for j in range(i):
                if data[j][0] <= data[i][0]:
                    dp[i] = max(dp[i], dp[j] + data[i][0])
        return max(dp)
