class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        n = len(ages)
        age_sort = sorted(list(range(n)), key=lambda x: ages[x] * 10000000 + scores[x])
        dp = [0] * n
        dp[0] = scores[age_sort[0]]
        for i in range(1, n):
            dp[i] = scores[age_sort[i]]
            for j in range(i):
                if scores[age_sort[i]] >= scores[age_sort[j]]:
                    dp[i] = max(dp[i], dp[j] + scores[age_sort[i]])
        return max(dp)
