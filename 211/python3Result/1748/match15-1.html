class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        ages.append(-1)
        scores.append(0)
        l = sorted(zip(ages,scores),key = lambda x:x[1])
        l = sorted(l,key = lambda x:x[0])
        d = deepcopy([i
[1] for i in l])
        for i in range(1, len(l)):
            for j in range(i):
                if l[i][1] >= l[j][1] or l[i][0] == l[j][0]:
                    d[i] = max(d[i], d[j] + l[i][1])
        #print(d)
        return max(d)
