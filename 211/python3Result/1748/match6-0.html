class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        n = len(scores)
        dp = [0 for _ in range(n)]
        
        A = sorted(zip(ages, scores))

        for i in range(n):
            dp[i] = A[i][1]
            for j in range(i):
                if A[i][1] >= A[j][1]:
                    dp[i] = max(dp[i], A[i][1]+dp[j])
        
        return max(dp)
