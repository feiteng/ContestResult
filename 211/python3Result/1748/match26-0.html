class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        
        
        persons = [(a,s) for s,a in zip(scores, ages)]
        persons.sort()
        dp = [0]*len(ages)  # dp[i] 表示以第 i 个人结尾的最大 scores
        for i,(age,score) in enumerate(persons):
            dp[i] = score
            for j in range(i):
                if persons[j][0] == age or persons[j][1] <= score:
                    dp[i] = max(dp[i], dp[j] + score)
        return max(dp)
