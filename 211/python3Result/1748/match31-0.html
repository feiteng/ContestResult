class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        ele = []
        for i in range(len(scores)):
            ele.append([scores[i], ages[i]])
        ele.sort(key=lambda x: (x[1],x[0]))
        print(ele)
        dp = [ele[i][0] for i in range(len(scores))]
        dp[0] = ele[0][0]
        for i in range(1, len(scores)):
            for j in range(i):
                temp = 0
                if ele[j][1]<ele[i][1]:
                    if ele[j][0]<=ele[i][0]:
                        dp[i] = max(dp[i], dp[j]+ele[i][0])
                else:
                    temp = max(temp, dp[j])
                    dp[i] = max(dp[i], temp+ele[i][0])
        print(dp)
        return max(dp)
