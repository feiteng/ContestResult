class Solution:
    def bestTeamScore(self, s: List[int], a: List[int]) -> int:
        n = len(s)
        rk = sorted([c for c in range(n)], key=lambda x: (a[x], s[x]))
        
        p = [0] * n
        p[0] = s[rk[0]]
        for i in range(1, n):
            x = rk[i]
            for j in range(i):
                y = rk[j]
                if a[y] == a[x] or s
[y] <= s[x]:
                    p[i] = max(p[i], p[j])
            p[i] += s[x]
        
        return max(p)
