class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        a = sorted(zip(ages, scores))
        n = len(a)
        dp = [0] * n
        
for i, (age, score) in enumerate(a):
            dp[i] = score
            for j in range(i):
                if a[i][1] >= a[j][1]:
                    dp[i] = max(dp[i], dp[j] + score)

        return max(dp)
