class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        # 最长上升子序列
        temp=[[ages[i],scores[i]] for i in range(len(scores))]
        # temp=sorted(temp,key=lambda x:[x[1],x[0]])
        temp=sorted(temp)
        # dp[i] 表示前i个人的最大得分
        dp=[0]*(len(scores)+1)
        for i in range(0,len(scores)):
            dp[i] = temp[i][1]
            for j in range(0,i):
                if not( temp[i][0]>temp[j][0] and temp[i][1]<temp[j][1]):
                    dp[i]=max(dp[i],dp[j]+temp[i][1])
        return max(dp)
