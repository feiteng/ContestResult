class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        ages_scores = [(x, y) for x, y in zip(ages, scores)]
        ages_scores.sort(key = lambda x: (x[0], x[1]))
        n = len(ages_scores)
        dp = [0 for _ in range(n)]
        dp[0] = ages_scores[0][1]
        ret = 0
        for i in range(n):
            dp[i] = ages_scores[i][1]
            for j in range(i):
                if ages_scores[i][1] >= ages_scores[j][1]:
                    dp[i] = max(dp[i], dp[j] + ages_scores[i][1])
            ret = max(ret, dp[i])
        return ret
