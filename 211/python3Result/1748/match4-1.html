class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        l = sorted(zip(ages, scores))
        n 
= len(l)
        dp = [0] * n
        for i in range(n):
            dp[i] = l[i][1]
            for j in range(i):
                if l[j][1] <= l[i][1]:
                    dp[i] = max(dp[i], dp[j] + l[i][1])
        return max(dp)
