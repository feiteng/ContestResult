class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        A = []
        for i in range(len(scores)):
            A.append([ages[i], scores[i]])
        
        A.sort()
        dp = [A
[i][-1] for i in range(len(A))]
        for i in range(1, len(A)):
            for j in range(i):
                if A[i][1] >= A[j][1]:
                    dp[i] = max(dp[i], dp[j] + A[i][1])
                    
                    
        return max(dp)
