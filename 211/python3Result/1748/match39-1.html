class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        w = sorted(list(zip(scores, ages)))
        print(w)
        
        N = len(w)
        dp = [0] * N
        
        mx 
= 0
        for i in range(N):
            dp[i] = w[i][0]
            for j in range(i):
                if w[j][1] <= w[i][1]:
                    dp[i] = max(dp[i], dp[j] + w[i][0])
            mx = max(mx, dp[i])
        return mx
