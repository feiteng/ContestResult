class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        n = len(scores)
        tup = [(ages[i], scores[i]) for i in range(n)]
        tup.sort()
        
        dp = [0] * n
        for i in range(n):
            dp[i] = tup[i][1]
            for j in range(i-1, -1, -1):
                if tup[j][1] <= tup[i][1]:
                    dp[i] = max(dp[i], tup[i][1] + dp[j])
        
        return max(dp)
