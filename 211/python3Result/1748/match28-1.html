class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        # dp
        n = len(scores)
        L = []
        for i in range(n):
            L.append([ages[i], scores[i]])
        L.sort()
        print
(L)
        dp = [0] * n
        dp[0] = L[0][1]
        for i in range(1, n):
            dp[i] = L[i][1]
            for j in range(i):
                if L[j][0] == L[i][0] or L[i][1] >= L[j][1]:
                    dp[i] = max(dp[i], dp[j] + L[i][1])
        # print(dp)
        return max(dp)
