class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        ageScores = sorted(zip(ages, scores))
        maxScores = [0] * len(scores)
        for i, (age, score) in enumerate(ageScores):
            ma = score
            for j in range(i):
                if ageScores[j][1] <= score:
                    ma = max(ma, score + maxScores[j])
            maxScores[i] = ma
        return max(maxScores)
