class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        # sorted age and score
        sorted_as = sorted(zip(ages, scores))
        
        N = len(sorted_as)
        
        # dp[i] the best score to choose i as the last player
        dp = [0] * N
        
        for i in range
(N):
            dp[i] = sorted_as[i][1]
            
            for i0 in range(i):
                # A conflict exists if a younger player has a strictly higher score than an older player.
                if sorted_as[i0][0] < sorted_as[i][0] and sorted_as[i0][1] > sorted_as[i][1]:
                    continue
                else:
                    dp
[i] = max(dp[i], sorted_as[i][1] + dp[i0])
        return max(dp)
