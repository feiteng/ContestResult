class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        pairs = list(zip(ages, scores))
        pairs.sort()
        d = [score for age, score in pairs]
        for cur in range(len(d)):
            for prev in range(cur):
                if pairs[prev][1] <= pairs[cur][1]:
                    d[cur] = max(d[cur], d[prev] + pairs[cur][1])
        # print(d)
        return max(d)
