class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        mans = []
        for s,a in zip(scores,ages):
            mans.append([a,s])
            
        mans.sort()


        n = len(mans)
        dp = [0 for i in range(n)]
        
        dp[0] = mans[0][1]
        for i in range(1,n):
            dp[i] = mans[i][1]
            for j in range(i):
                if mans[j][1] <= mans[i][1]:
                    dp[i] = max(dp[i],dp[j]+ mans[i][1])
            
        return max(dp)
