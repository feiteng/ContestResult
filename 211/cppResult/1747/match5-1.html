class Solution {
public:
    string applyAdd( string temp, int a )
    {
        for(int i=1; i< temp.size(); i+=2 )
        {
            int num = temp[i]-'0';
            num = (num + a)%10;
            temp[i] = (char)( num + '0');

        }
        return temp;
    }
    string applyRotate( string temp, int b)
    {
        string ans = temp;
        for(int i=0; i< temp.size(); i++)
        {
            ans[(i+b)%temp.size()] = temp[i];
        }
        
        return ans;
    }
    string findLexSmallestString(string s, int a, int b) {
        if(s.empty())
            return s;
        
        set<string> hashmap;
        queue<string> q;
        q.push(s);
        hashmap.insert(s);
        

        while(!q.empty())
        {
            string temp = q.front();
            q.pop();
            
            string a1 = applyAdd(temp, a );
            if(hashmap.find(a1) == hashmap.end())
            {
                hashmap.insert(a1);
                q.push(a1);
            }
            
            a1 = applyRotate(temp, b);
            if(hashmap.find(a1) == hashmap.end())
            {
                hashmap.insert(a1);
                q.push(a1);
            }
        }
        
        return *(hashmap.begin());
    }
};
