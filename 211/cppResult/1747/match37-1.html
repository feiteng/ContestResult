class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int add_interval = Gcd(a, 10);
        int rot_interval = Gcd(b, s.length());
        string str = "9";
        str.back()++;
        if (rot_interval % 2) {

            for (int i = 0; i < 10; i += add_interval) {
                for (int j = 0; j < 10; j += add_interval) {
                    for (int start = 0; start < s.length(); start += rot_interval) {
                        str = min(str, Gen(s, i, j, start));
                    }
                }
            }

        } else {
            for (int i = 0; i < 10; i += add_interval) {
                for (int start = 0; start < s.length(); start += rot_interval) {
                    str = min(str, Gen(s, i, 0, start));
                }
            }
        }
        return str;
    }
private:
    int Gcd(int a, int b) {
        while (a % b) {
            int t = a;
            a = b;
            b = t % b;
        }
        return b;
    }
    string Gen(const string& s, int odd_add, int even_add, int start) {
        string r;
        for (int i = 0; i < s.length(); i++) {
            int idx = start + i;
            idx %= s.length();
            r.push_back(s[idx]);
            r.back() -= '0';
            if (idx % 2) {
                r.back() += odd_add;
            } else {
                r.back() += even_add;
            }                
            r.back() %= 10;
            r.back() += '0';
        }
        return r;
    }
};
