class Solution {
public:
    string cal1(string s, int a){
        for(int i = 1; i < s.length(); i += 2){
                int t = s[i] - '0';
                t += a;
                if(t >= 10) t -= 10;

                s[i] = '0'+ t;
            }
        return s;
    }
    string cal2(string s, int b){
        s = s.substr(s.length()-b, b) + s.substr(0, s.length()-b);
        return s;
    }
    string findLexSmallestString(string s, int a, int b) {
        string ans = s;
        queue<string> q;
        q.push(s);
       
        set<string> se; se.insert(s);
        while(!q.empty()){
            string p = q.front(); q.pop();
            ans = min(ans, p);
            string p1 = cal1(p,a);
            string p2 = cal2(p,b);
            if(!se.count(p1)){
                q.push(p1);
                se.insert(p1);
            }
            if(!se.count(p2)){
                q.push(p2);
                se.insert(p2);
            }
        }
        return ans;
    }
};
