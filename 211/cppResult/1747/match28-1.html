class Solution {
public:

    string findLexSmallestString(string inputs, int a, int b) {
        unordered_set<string> ss;
        string news=inputs;
        string ans=inputs;
        int l=inputs.size();
        for (int loop=0;loop<inputs.size();loop++){
            news=news.substr(l-b,b)+news.substr(0,l-b);
            ss.insert(news);
        }
        for (string sources:ss){
            for (int i=0;i<10;i++)
                for(int j=0;j<10;j++){
                    //even
                    string s=sources;
                    if (b%2==1)

                        for (int k=0;k<l;k+=2){
                            s[k]= (s[k]-48+a*i)%10+48;
                        }
                    for (int k=1;k<l;k+=2){
                        s[k]= (s[k]-48+a*j)%10+48;
                    }
                    if (s<ans)
                        ans=s;
                }
        }
        return ans;
    }
};
