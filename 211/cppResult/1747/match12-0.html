class Solution {
public:
  string findLexSmallestString(string s, int a, int b) {
    int n = s.size();
    queue<string> q;
    unordered_set<string> st;
    q.push(s);
    string ret = s;
    while(!q.empty()) {
      string p = q.front();
        // cout<<p<<endl;
      ret = min(ret, p);
      q.pop();
      st.insert(p);
      // add
      string tmp = p;
      for(int i=0; i<n; i++) {
        if (i & 1) {
          tmp[i] = char('0' + ((tmp[i]-'0'+a) % 10));
        }
      }
      if (st.find(tmp) == st.end()) {
        st.insert(tmp);
        q.push(tmp);
      }
      // rotate
      string tmp1 = p;
      for (int k = 0; k < n; k++) {
        tmp1[(b + k) % n] = p[k];
      }
      if (st.find(tmp1) == st.end()) {
        st.insert(tmp1);
        q.push(tmp1);
      }
    }
    return ret;
  }
};
