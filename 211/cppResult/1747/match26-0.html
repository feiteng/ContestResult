class Solution {
public:
    set<string> ans;
    void helper(string &s, int a, int b){
        string s1 = s;
        for(int i=1;i<s.size();i+=2){
            int k = s[i] - '0';
            int n = (k+a) % 10;
            s1[i] = '0'+n;
        }
        if(!ans.count(s1)){
            ans.insert(s1);
            helper(s1,a,b);
        }
        string s2 = s.substr(s.size()-b, b) + s.substr(0, s.size()-b);
        if(!ans.count(s2)){
            ans.insert(s2);
            helper(s2,a,b);
        }
    }
    string findLexSmallestString(string s, int a, int b) {
        ans = set<string>();
        helper(s,a,b);
        return *ans.begin();
    }
};
