class Solution {

    string solvea(const string& s, int a) {
        string ret = s;
        for (int i = 1; i < s.size(); i += 2) {
            int val = s[i] - '0';
            val = (val + a) % 10;
            ret[i] = val + '0';
        }
        return ret;
    }
    string solveb(const string& s, int b) {
        int n = s.size();
        string ret = s.substr(n-b) + s.substr(0, n-b);
        assert(ret.size() == n);
        // cout << s << " rotate " << b << " " << ret << endl;
        return ret;
    }

public:
    string findLexSmallestString(string s, int a, int b) {
        set<string> table;
        queue<string> q;
        q.push(s);
        while (!q.empty()) {
            string cur = q.front(); q.pop();
            string nxt = solvea(cur, a);
            if (table.find(nxt) == table.end()) {
                table.insert(nxt);
                q.push(nxt);
            }
            // cout << "solve a" << endl;
            nxt = solveb(cur, b);
            if (table.find(nxt) == table.end()) {
                table.insert(nxt);
                q.push(nxt);
            }
            // cout << "solve b" << endl;
        }
        return *table.begin();
        
    }
};
