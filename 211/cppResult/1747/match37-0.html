class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int n=s.size();
        vector<int> ss;
        for(auto c:s) ss.push_back(c-'0');
        set<vector<int>> tmp({ss});
        for(int i=0;i<n;i++) {
            rotate(ss.begin(),ss.end() - b,ss.end());
            tmp.insert(ss);
            for(int t=0;t<10;t++) {
                for(int j=n-1;j>=0;j-=2) {
                    ss[j]=(ss[j]+a)%10;
                }
                if(__gcd(b,n)%2) {
                    for(int tt=0;tt<10;tt++) {
                        for(int j=n-2;j>=0;j-=2) {
                            ss[j]=(ss[j]+a)%10;
                        }
                        tmp.insert(ss);
                    }
                }            
                tmp.insert(ss);
            }
        }
        vector<int> res=*tmp.begin();
        string ans;
        for(auto x:res) ans+='0'+x;
        return ans;
    }
};
