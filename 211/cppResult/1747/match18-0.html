class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int n=s.size();
        string res=s;
        if(b%2==0){
            for(int i=0;!i||(i*a)%10;++i){
                string t=s;
                for(int j=1;j<n;j+=2){
                    t[j]='0'+((t[j]-'0'+i*a)%10);
                }
                for(int j=0;!j||(j*b)%n;++j){
                    string tt=t.substr((j*b)%n)+t.substr(0,(j*b)%n);
                    if(tt<res)
                        res=tt;
                }
                
            }
        }
        else{
            for(int i=0;!i||(i*a)%10;++i){
                string t=s;
                for(int j=1;j<n;j+=2){
                    t[j]='0'+((t[j]-'0'+i*a)%10);
                }
                for(int k=0;!k||(k*a)%10;++k){
                    string qt=t;
                    for(int j=0;j<n;j+=2){
                        qt[j]='0'+((qt[j]-'0'+k*a)%10);
                    }
                    for(int j=0;!j||(j*b)%n;++j){
                        string tt=qt.substr((j*b)%n)+qt.substr(0,(j*b)%n);
                        // cout<<tt<<endl;
                        if(tt<res)
                            res=tt;
                    }    
                }
                
                
            }
        }
        
        return res;
    }
};
