class Solution {
public:
    string smallest=string(120,'9');
    int ct=0;
    unordered_map<string,int> visited;
    void dfs(string s,int a,int b)
    {
        if(visited[s]>0)
            return;
        visited[s]=1;
        ct++;
        if(s<smallest)
            smallest=s;
        string ans=s;
        for(int i=0;i<s.length();i++)
        {
            if(i%2==1)
            {
                int p=ans[i]-'0';
                p=(p+a)%10;
                ans[i]=char(p+'0');
            }
        }
        dfs(ans,a,b);
        ans=s;

        int n=s.length();
        for(int i=0;i<s.length();i++)
        {
            ans[i]=s[(i+b)%n];
        }
        dfs(ans,a,b);
    }
    string findLexSmallestString(string s, int a, int b) {
        dfs(s,a,b);
        return smallest;
    }
};
