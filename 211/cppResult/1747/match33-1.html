class Solution {
public:
    
    void cmp(char ans[], char tmp[], int n) {
        int flag = 0;
        for(int i=0;i<n;i++){
            if(tmp[i]>ans[i]) break;
            if(tmp[i]<ans[i]) {
                flag = 1;break;
            }
        }
        if(flag) {
            for(int i=0;i<n;i++)ans[i] = tmp[i];
        }
    }
    string findLexSmallestString(string s, int a, int b) {
        int n = (int)s.size();
        char tmp[105],ans[105];

        for(int i=0;i<n;i++)ans[i] = s[i];
        int flag = 0;
        for(int i = 0;i || !flag; i=(i+b)%n) {
            flag = 1;
            for(int j = 0; j <= 10;j ++) {
                if(b%2==1) {

                    for(int k = 0; k <= 10; k++) {
                        for(int w = i; w < n;w ++) tmp[w-i] = s[w];
                        for(int w = 0; w < i;w ++) tmp[w + n - i] = s[w];    
                        for(int w = 1; w < n; w+=2) {
                            tmp[w] = (tmp[w] - '0' + j*a) %10 + '0';
                        }        
                        for(int w = 0; w < n; w+=2) {
                            tmp[w] = (tmp[w] - '0' + k*a) %10 + '0';
                        }
                        cmp(ans, tmp, n);
                    }
                } else {
                    for(int w = i; w < n;w ++) tmp[w-i] = s[w];
                    for(int w = 0; w < i;w ++) tmp[w + n - i] = s[w];    
                    for(int w = 1; w < n; w+=2) {
                        tmp[w] = (tmp[w] - '0' + j*a) %10 + '0';
                    }
                    cmp(ans, tmp, n);
                }
                
            }
        }
        
        string res = "";
        for(int i=0;i<n;i++)res+=ans[i];
        return res;
    }
};
