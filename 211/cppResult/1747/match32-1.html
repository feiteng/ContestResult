class Solution {
public:
    unordered_map<string, bool> seen;
    string findLexSmallestString(string s, int a, int b) {
        seen[s] = true;
        string ret = s, tmp = s;
        for (int i = 1; i < s.size(); i+=2) {
            tmp[i] = (tmp[i] - '0' + a)%10 + '0';
        }
        if (!seen[tmp]) ret = min(ret, findLexSmallestString(tmp, a, b));
        tmp = s.substr(s.size() - b) + s.substr(0, s.size() - b);
        if (!seen[tmp]) ret = min(ret, findLexSmallestString(tmp, a, b));
        return ret;
    }
};
