class Solution {
public:
set<string> index;
void dfs(string s, int a, int b) {
	string s1 = "", s2 = "";
	int n = s.size();
	//if(index.find(s1) == index.end()){
	for (int i = 0; i < n; i++) {
		if(i % 2 == 0) s1 += s[i];
		else {
			int temp = s[i] - '0' + a;
			temp = temp % 10;
			s1 += temp + '0';
		}
	}
	for (int i = n-b; i < n; i++) {
		s2 += s[i];
	}
	for (int i = 0; i < n - b; i++) {
		s2 += s[i];
	}
	if (index.find(s1) == index.end()) {
		index.insert(s1);
		dfs(s1, a, b);
	}
	if (index.find(s2) == index.end()) {
		index.insert(s2);
		dfs(s2, a, b);
	}
	//index.insert(s2);
	//dfs(s1, a, b);
	//dfs(s2, a, b);
	return;
}
string findLexSmallestString(string s, int a, int b) {
	int n = s.size();
	dfs(s, a, b);
	return *index.begin();
}
};
