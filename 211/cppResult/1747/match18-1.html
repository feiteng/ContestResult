class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int n = s.size();
        

        int hcf = __gcd(a , 10);
        string str = s;
        
        if(b % 2 == 0){
            for(int i = 0 ; i <= 10 ; i += hcf){
                string s1 = s;
                for(int j = 1 ; j < s.size() ; j += 2){
                    s1[j] = (char)((s1[j] - '0' + i) % 10 + '0');
                }
                
                string s2 = s1;
                // cout << "i : " << i << "\n";
                do{
                    string st1 = s1.substr(0 , n - b);
                    string st2 = s1.substr(n-b , b);
                    
                    string st =  st2 + st1;
                    if(st < str)str = st;
                    s1 = st;
                    
                    // cout << s1 << " " << s2 << "\n";
                }while(s2 != s1);

            }
        }
        
        else{
            for(int i = 0 ; i <= 10 ; i += hcf){
                string s1 = s;
                for(int j = 1 ; j < s.size() ; j += 2){
                    s1[j] = (char)((s1[j] - '0' + i) % 10 + '0');
                }
                
                string st1_ = s1.substr(0 , n - b);
                string st2_ = s1.substr(n - b , b);

                
                

                for(int y = 0 ; y <= 10 ; y += hcf){
                    string st_ =  st2_ + st1_;
                    for(int j = 1 ; j < s.size() ; j += 2){
                        st_[j] = (char)((st_[j] - '0' + y) % 10 + '0');
                    }

                    s1 = st_;
                    
                    cout << s1 << "\n";
                    int cnt = 0;
                    string s2 = s1;
                    do{
                        cnt++;
                        string st1 = s1.substr(0 , n - b);
                        string st2 = s1.substr(n-b , b);

                        string st =  st2 + st1;
                        if(st < str)str = st;
                        s1 = st;

                    }while(s2 != s1);
                }
            }
        }
        
        return str;
    }
};
