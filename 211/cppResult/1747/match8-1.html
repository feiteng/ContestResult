class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {

        int n = s.size();
        string ans(100, '9');
        unordered_set<string> visited;
        queue<string> q;
        q.push(s);
        visited.insert(s);
        while(!q.empty()) {
            string top = q.front();
            q.pop();
            ans = min(ans, top);
            string candidate1 = top.substr(n - b) + top.substr(0, n - b);
            string candidate2 = top;
            for (int i = 1; i < n; i+=2) {
                int val = ((candidate2[i] - '0') + a) % 10;

                candidate2[i] = (val + '0');
            }
            if (visited.count(candidate1) == 0) {
                visited.insert(candidate1);
                q.push(candidate1);
            }
            if (visited.count(candidate2) == 0) {
                visited.insert(candidate2);
                q.push(candidate2);
            }
        }
        return ans;
    }
};
