class Solution {
public:
    vector<string> poo;
    string findLexSmallestString(string s, int a, int b) {
        string ans;
        ans.assign(s.size(),'9');
        int used[110];
        memset(used,0,sizeof(used));
        int i = 0;
        string l = s;
        while(used[i] == 0)
        {
            used[i] = 1;
            string c = l.substr(l.size()-b,b);
            c += l.substr(0,l.size()-b);
            cout << c << endl;
            poo.push_back(c);
            i = (i+b)%s.size();
            l = c;
        }
        for(int i = 0; i < poo.size(); i++)
        {
            if(b % 2 == 0)
            {
                for(int j = 0; j < 10; j++)
                {
                    for(int k = 1; k < s.size(); k+=2)
                    {
                        poo[i][k] = (poo[i][k]-'0' + a)%10 + '0';
                    }
                    //cout << poo[i] << endl;
                    if(ans > poo[i])
                        ans = poo[i];
                }
                
            }
            
            else
            {
                for(int j = 0; j < 10; j++)
                {
                    for(int k = 1; k < s.size(); k+=2)
                    {
                        poo[i][k] = (poo[i][k]-'0' + a)%10 + '0';
                    }
                    if(ans > poo[i])
                        ans = poo[i];
                    
                    for(int j2 = 0; j2 < 10; j2++)
                    {
                        for(int k = 0; k < s.size(); k+=2)
                        {
                            poo[i][k] = (poo[i][k]-'0' + a)%10 + '0';
                        }
                        //cout << poo[i] << endl;
                        if(ans > poo[i])
                            ans = poo[i];
                    }
                }
            }
        }
        return ans;
    }
};
