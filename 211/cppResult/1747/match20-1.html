class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        string ans=s;

        int n=s.length();
        priority_queue<string,vector<string>,greater<string>>pq;
        pq.push(s);
        set<string>se;
        
        while(!pq.empty()){
            auto cur =pq.top();
            pq.pop();
            ans=min(ans,cur);
            // cout<<cur<<endl;
            s = cur;
            for(int j=1;j<n;j+=2){
                s[j] = (s[j]-'0'+a)%10+'0';
            }
            if(!se.count(s)){
                pq.push(s);
                se.insert(s);
            }
            s=cur;
            rotate(s.begin(),s.begin()+(n-b),s.end());
            if(!se.count(s)){
                pq.push(s);
                se.insert(s);
            }
            
        }
        return ans;
    }
};
