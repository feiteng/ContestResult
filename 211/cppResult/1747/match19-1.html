class Solution {
public:
    set<string> st;
    void dfs(string s, int a, int b) {
        if (st.count(s)) return;
        st.insert(s);
        int N = s.size();
        string ns = s.substr(N-b, b) + s.substr(0, N-b);
        dfs(ns, a, b);
        for (int i=1; i<N; i+=2) {
            s[i] = ((s[i] - '0' + a)%10) + '0';
        }
        // cout << s << endl;
        dfs(s, a, b);
    }
    string findLexSmallestString(string s, int a, int b) {
        dfs(s, a, b);
        return *st.begin();
    }
};
