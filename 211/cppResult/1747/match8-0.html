class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int n = s.size();
        set<string> visited;
        queue<string> q;
        q.push(s);
        visited.insert(s);
        string res = s;
        
        while (!q.empty()) {
            string ss = q.front();
            q.pop();
            string sa = ss, sb = ss;
            res = min (res, ss);
            
            for (int i = 1; i < n; i += 2) {
                int val = sa[i] - '0';
                if (val + a > 10) {
                    sa[i] = ((val + a - 10) % 10) + '0';
                } else {
                    sa[i] = ((sa[i] - '0' + a) % 10) + '0';   
                }
            }
            for (int i = 0; i < n; i++) {
                sb[i] = ss[(i - b + n) % n];
            }
            //cout << ss <<" " << sb << " " << sa << endl;
            if (visited.find(sa) == visited.end()) {
                q.push(sa);
                visited.insert(sa);
            }
            if (visited.find(sb) == visited.end()) {
                q.push(sb);
                visited.insert(sb);
            }
        }
        return res;
        
    }
};
