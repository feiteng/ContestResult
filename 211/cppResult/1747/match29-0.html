class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        string ans = s;
        int n = s.size();
        for (int shift = 0; ; shift += b) {
            int sh = shift % n;
            string t(s.begin(), s.end());
            reverse(t.begin(), t.begin() + sh);
            reverse(t.begin() + sh, t.end());
            reverse(t.begin(), t.end());
            if (shift % n == 0) {
                if (shift != 0)
                    break;
            }
            for (int i = 0; i <= 10; i++) {
                for (int j = 1; j < n; j += 2) {
                    t[j] += a;
                    if (t[j] > '9')
                        t[j] -= 10;
                }
                if (b & 1)
                    for (int k = 0; k <= 10; k++) {
                        for (int j = 0; j < n; j += 2) {
                            t[j] += a;
                            if (t[j] > '9')
                              t[j] -= 10;
                        }
                        ans = min(ans, t);
                    }
                ans = min(ans, t);
            }
        }
        return ans;
    }
};
