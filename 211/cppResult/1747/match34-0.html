class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        string res = s;
        int n = s.length();
        vector<string> odd, even;
        for (int i = 0; i <= 10; i++) {
            string tmp = s;
            for (int j = 1; j < n; j += 2) {
                tmp[j] = (tmp[j] - '0' + i*a) % 10 + '0';
            }
            odd.push_back(tmp);
            res = min(res, tmp);
        }
        
        if (b & 1) {
            for (int i = 1; i <= 10; i++) {
                string tmp = s;
                for (int j = 0; j < n; j += 2) {
                    tmp[j] = (tmp[j] - '0' + i*a) % 10 + '0';
                }
                even.push_back(tmp);
                res = min(res, tmp);
            }
        } else {
            even.push_back(s);
        }
        
        for (string s1 : odd) {
            for (string s2 : even) {
                string tmp;
                for (int i = 0; i < n; i++) {
                    if (i&1) tmp += s1[i];
                    else tmp += s2[i];
                }
                for (int ii = 0; ii <= n; ii++) {
                    int k = ii*b % n;
                    string now = tmp;
                    for (int i = 0; i < k; i++) now[i] = tmp[n-k+i];
                    for (int i = k; i < n; i++) now[i] = tmp[i-k];
                    
                    res = min(res, now);
                }
            }
        }
        
        return res;
    }
};
