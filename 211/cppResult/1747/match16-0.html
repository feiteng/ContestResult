class Solution {
public:
    string add(string s, int a, int offset=1){
        for (int i = offset; i < s.size();i+=2){
            s[i] = '0' + (s[i] - '0' + a) % 10;
        }
        return s;
    }
    string turn(string& s, int b){
        string res = s;
        for (int i = 0; i < s.size();i++){
            res[(i + b) % s.size()] = s[i];
        }
        return res;
    }
    string findLexSmallestString(string s, int a, int b) {
        multiset<string> mset;
        vector<string> pool;
        string temp = s;
        pool.push_back(temp);
        for (int i = 1; i < 10;i++){
            temp = add(temp, a);
            pool.push_back(temp);
        }
        if(b%2==1){
            for (int k = 0; k < 10;k++){
                temp = pool[k];
                for (int i = 1; i < 10;i++){
                    temp = add(temp, a, 0);
                    pool.push_back(temp);
                }
            }
        }
        vector<bool> vis;
        vis.resize(s.size());
        int cur = 0;
        while(!vis[cur]){
            vis[cur] = true;
            for (int i = 0; i < pool.size();i++){
                temp = turn(pool[i], cur);
                mset.insert(temp);
            }
            cur = (cur + b) % s.size();
        }
        string res = *(mset.begin());
        return res;
    }
};
