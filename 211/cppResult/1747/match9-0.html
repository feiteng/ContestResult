class Solution {
public:
    set<string> se;
    void rec(string s, int a, int b){
        if(se.find(s)!=se.end())
            return ;
        se.insert(s);
        // cout<<s<<" "<<res<<"\n";
        string str= s;
        for(int i=1;i<s.length();i+=2){
            // cout<<((s[i]-'0')+a)%9<<" ";
            s[i]= ((s[i]-'0')+a)%10+'0';
        }
        // cout<<s<<"\n";
        rec(s, a, b);
        int n= s.length();
        // cout<<str<<" ";
        s =str.substr(n-b,b)+str.substr(0,n-b);
        // cout<<s<<"\n";
        rec(s, a, b);
        
    }
    string findLexSmallestString(string s, int a, int b) {
        // str=s;
        rec(s, a, b);
        // for (auto it=se.begin(); it != se.end(); ++it) 
        //     cout << ' ' << *it; 
        return *(se.begin());
    }
};
