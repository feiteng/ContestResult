class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        string res=s;
        int n=s.length();
        for(int i=0;i<n;i++)
        {
            string str=s;
            //cout<<"fuck "<<str<<endl;
            if(str<res) res=str;//反转完成
            string minn=str;
            if(b&1)
            {
                for(int k=0;k<10;k++)
                {
                    for(int j=0;j<n;j+=2)
                        str[j]=(str[j]-'0'+a)%10+'0';
                    //cout<<"---- "<<str<<endl;
                    if(str<minn) minn=str;
                }
            }
            //cout<<"hehe "<<minn<<endl;
            for(int k=0;k<10;k++)
            {
                for(int j=1;j<n;j+=2)
                    minn[j]=(minn[j]-'0'+a)%10+'0';
                if(minn<res) res=minn;
            }
            //cout<<"aaa "<<minn<<" "<<res<<endl;
            s=s.substr(n-b)+s.substr(0,n-b);
        }
        return res;
    }
};
