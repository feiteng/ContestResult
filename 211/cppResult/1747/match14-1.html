class Solution {
public:

   string firstOperation(string s, int a) {
      string t(s);
      int n = s.size();
      for (int i = 1; i < n; i += 2) {
         t[i] = (((t[i] - '0') + a) % 10) + '0';
      }
      return t;
   }

   string secondOperation(string s, int b) {
      int n = s.size();
      string t(n, '0');

      for (int i = 0; i < n; i++) {
         t[i] = s[(i - b + n) % n];
      }
      return t;
   }

   string findLexSmallestString(string s, int a, int b) {
      set<string> set;
      set.insert(s);
      queue<string> q;
      q.push(s);
      while (!q.empty()) {

         string ss = q.front();
         q.pop();

         string firstString = firstOperation(ss, a);
         if (!set.count(firstString)) {
            set.insert(firstString);
            q.push(firstString);
         }

         string secondString = secondOperation(ss, b);
         if (!set.count(secondString)) {
            set.insert(secondString);
            q.push(secondString);
         }
      }
      return *set.begin();
   }
};
