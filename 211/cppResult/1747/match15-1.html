class Solution {
public:
    string add(string s, int m1, int m2, int a, int b) {
        string res = s;
        
        for(int i = 1; i < s.length(); i += 2) {
            int num = s[i] - '0';
            num = (num + m1 * a) % 10;
            res[i] = ('0' + num);
        }
        
        if(!(b & 1)) return res;
        
        for(int i = 0; i < s.length(); i += 2) {
            int num = s[i] - '0';
            num = (num + m2 * a) % 10;
            res[i] = ('0' + num);
        }
        
        return res;
    }
    
    string cycle(string s, int shift) {

        string res = s;
        
        for(int i = 0; i < s.length(); i++) {
            res[(i + shift) % s.length()] = s[i];
        }
        
        return res;
    }
    
    string findLexSmallestString(string s, int a, int b) {
        string best = s;
        for(int i = 0; i < 10; i++) {
            for(int j = 0; j < 10; j++) {
                string w = add(s, i, j, a, b);

                for(int k = 0; k < 100; k++) {
                    string cand = cycle(w, k * b);
                    best = min(best, cand);
                }
            }
        }
        return best;
    }
};
