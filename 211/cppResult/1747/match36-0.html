class Solution {
public:
    void rotate(string& s, int k) {
        int len = s.length();
        k %= len;
        s = s.substr(len - k) + s.substr(0, len - k);
    }
    string findLexSmallestString(string s, int a, int b) {
        vector<string> vec;
        for(int i = 0; i < 10; i++) {
            string str = s;
            for(int j = 1; j < str.length(); j+=2) {
                str[j] = char(((str[j] - '0') + a * i) % 10 + '0');
            }
            vec.push_back(str);
        }
        if(b % 2 == 1) {
            vector<string> new_v;
            for(int k = 0; k < vec.size(); k++) {
                for(int i = 0; i < 10; i++) {
                    string str = vec[k];
                    for(int j = 0; j < str.length(); j+=2) {
                        str[j] = char(((str[j] - '0' ) + a * i) % 10 + '0');
                    }
                    new_v.push_back(str);
                }
            }
            vec = new_v;
        }
        string res = vec[0];
        for(int i = 0; i < vec.size(); i++) {
            string temp = vec[i];
            for(int j = 0; j < 100; j++) {
                if(temp < vec[i])
                    vec[i] = temp;
                rotate(temp, b);
            }
            res = min(res, vec[i]);
        }
        return res;
    }
};
