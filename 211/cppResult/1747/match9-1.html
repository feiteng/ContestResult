class Solution {
public:
    void dfs(string s, int a, int b, set<string>& vis) {
      if (vis.find(s) != vis.end()) return;
      vis.insert(s);
      string _s = s;
      for (int i = 1; i < s.size(); i+=2) {
        s[i] = (s[i] - '0' + a) % 10 + '0';
      }
      dfs(s, a, b, vis);
      s = _s;
      s = s.substr(s.size() - b) + s.substr(0, s.size()-b);
      dfs(s, a, b, vis);
    }
    string findLexSmallestString(string s, int a, int b) {
      set<string> vis;
      dfs(s, a, b, vis);
      return *vis.begin();
    }
};
