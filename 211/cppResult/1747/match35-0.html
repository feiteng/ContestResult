class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        vector<string> vec;
        unordered_set<string> vis;
        dfs(s, vis, vec, a, b);
        sort(vec.begin(), vec.end());
        return vec[0];
    }
    
private:
    void dfs(string s, unordered_set<string> &vis, vector<string> &vec, int a, int b) {
        if(vis.count(s)) return ;
        vis.insert(s);
        vec.push_back(s);
        string t = add(s, a);
        dfs(t, vis, vec, a, b);
        t = rotate(s, b);
        dfs(t, vis, vec, a, b);
    }
    
    string add(string s, int a) {
        for(int i = 1; i < s.size(); i += 2) {
            s[i] -= '0';
            s[i] += a;
            s[i] %= 10;
            s[i] += '0';
        }
        return s;
    }
    
    string rotate(string s, int b) {
        int n = s.size();
        return s.substr(n - b) + s.substr(0, n - b);
    }
};
