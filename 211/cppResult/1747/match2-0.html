class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
       set<string> S;
       queue<string> Q;
       Q.push(s);
        int n = s.size();
        S.insert(s);
        while (!Q.empty()) {
            string s = Q.front(); Q.pop();
           // S.insert(s);
            string t = s;
            for (int i = 1; i < n; i++) 
                if (i % 2 == 1) {
                    t[i] = t[i] + a;
                    if (t[i] > '9') t[i] -= 10;
                }
            if (S.count(t) == 0) {
                S.insert(t);
                Q.push(t);
            }
             t = s.substr(b) + s.substr(0,b);
            if (S.count(t) == 0) {
                S.insert(t);
                Q.push(t);
            }
        }
        return *S.begin();
    }
};
