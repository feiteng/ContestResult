class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        set<string> used;
        queue<string> que;
        int len = s.length();
        used.insert(s);
        que.push(s);
        while(!que.empty())
        {
            auto cur = que.front();
            que.pop();
            string nb = cur.substr(len-b, b) + cur.substr(0, len - b);
            string na = cur;
            for(int i=1;i<len;i+=2)
            {
                na[i] = (((cur[i] - '0') + a )%10) + '0';
            }
            if(!used.count(na)) {
                used.insert(na);
                que.push(na);
            }
            if(!used.count(nb))
            {
                used.insert(nb);
                que.push(nb);
            }
        }
        auto it = used.begin();
        return *it;
        
    }
};
