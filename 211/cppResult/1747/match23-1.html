class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int len = s.length();
        string rss = s;
        string tmps = s;
        while(1)
        {
            //cout << "hh";

            for(int i=0; i<10; i++)
            {
                //cout << "jj";
                for(int j=1; j<len; j=j+2)
                    tmps[j] = '0'+((tmps[j]-'0')+a)%10;
                if(tmps.compare(rss)<0)
                    rss = tmps;
                if(b%2==1)

                    for(int k=0; k<10; k++)
                    {
                        for(int j=0; j<len; j=j+2)
                            tmps[j] = '0'+((tmps[j]-'0')+a)%10;
                        if(tmps.compare(rss)<0)
                            rss = tmps;
                    }
                //cout << "kk";
            }
            tmps = tmps.substr(b) + tmps.substr(0, b);
            //cout << tmps << ":" << rss << endl;
            if(tmps.compare(s)==0)
                break;
        }
        return rss;
    }
};
