class Solution {
    
public:
    string findLexSmallestString(string s, int a, int b) {
        const int L = size(s);
        if (L == 0) return "";
        string ans = s;
        unordered_set<string> seen({s});
        auto opa = [&](string s) {
            for (int i = 1; i < L; i += 2) {
                s[i] += a;
                if (s[i] > '9') s[i] -= 10;
            }
            return s;
        };
        auto opb = [&](const string &s) {
            return s.substr(b) + s.substr(0, b);
        };
        queue<string> q({s});
        while (!empty(q)) {
            auto front = q.front(); q.pop();
            ans = min(ans, front);
            string cand = opa(front);
            if (!seen.count(cand)) {
                seen.insert(cand);
                q.push(cand);
            }
            cand = opb(front);
            if (!seen.count(cand)) {
                seen.insert(cand);
                q.push(cand);
            }
        }
        return ans;
    }
};
