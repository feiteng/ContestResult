class Solution {
public:
    set<string> ss;
    void f(string s,int a,int b){
        if(ss.count(s))return;
        ss.insert(s);
        int n=s.length();
        string zz=s.substr(n-b)+s.substr(0,n-b);
        f(zz,a,b);
        string z;
        for(int i=0;i<n;++i){
            if(i%2==0){
                z.push_back(s[i]);
                continue;
            }
            int x=s[i]-'0';
            x+=a;
            x%=10;
            z.push_back((char)(x+'0'));
        }
        f(z,a,b);
    }
    string findLexSmallestString(string s, int a, int b) {
       
        f(s,a,b);
        return (*ss.begin());
    }
};
