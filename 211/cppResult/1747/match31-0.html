class Solution {
public:
    set<string> st;
    string findLexSmallestString(string s, int a, int b) {
        dfs(s,a,b);
        return *st.begin();
    }
    
    void dfs(string s,int a, int b){
        if ( st.count(s)>0 )
            return;
        st.insert(s);
        // 先加
        for(int i=1; i<s.size(); i+=2){
            s[i] = ( ( s[i]-'0'+a ) % 10 + '0' );
        }
        dfs(s, a, b);
        // 再移
        dfs(helper(s, b),a,b);
    }
    
    string helper(string a, int d){
        int n = a.size();
        reverses(a, 0, n-d-1);
        reverses(a, n-d, n-1);
        reverses(a, 0, n-1);
        return a;
    }
    
    void reverses(string &a,int l, int r){
        while (l < r)
        {
            int temp = a[l];
            a[l] = a[r];
            a[r] = temp;
            l++;
            r--;
        }
    }
    
};
