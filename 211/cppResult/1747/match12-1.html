class Solution {
public:
	string findLexSmallestString(string s, int a, int b) {
		int n = s.length(), i, j, k, x;
		string s2, best = s;

		unordered_set<string> ss;
		queue<string> q;
		q.push(s);
		ss.insert(s);

		while (!q.empty()) {
			s = q.front();
			q.pop();

			if (s < best) best = s;

			s2 = s;
			for (i = 1; i < n; i += 2) {
				x = '0' + (s[i] - '0' + a) % 10;
				s2[i] = x;
			}
			if (ss.find(s2) == ss.end()) {
				ss.insert(s2);
				q.push(s2);
			}


			for (i = 0; i < n; i++) {
				s2[(i + b) % n] = s[i];
			}
			if (ss.find(s2) == ss.end()) {
				ss.insert(s2);
				q.push(s2);
			}
		}

		return best;
	}
};
