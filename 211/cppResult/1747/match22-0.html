class Solution {
public:
    set<string> vst;
    // unordered_map<string
    
    string rotate(string s, int b) {
        string t = s;
        int len = s.size();

        for (int i = 0; i < len; i++) {
            t[i] = s[(i + b) % len];
        }
        
        return t;
    }
    
    string accum(string s, int a) {
        string t = s;
        int len = s.size();
        
        for (int i = 1; i < len; i += 2) {
            t[i] = ((s[i] - '0' + a) % 10) + '0';
        }

        return t;
    }
    
    void dfs(string s, int a, int b) {
        if (vst.find(s) != vst.end()) {
            return;
        }
        vst.insert(s);
        string s1 = rotate(s, b);
        dfs(s1, a, b);
        string s2 = accum(s, a);
        dfs(s2, a, b);
    }
    
    
    string findLexSmallestString(string s, int a, int b) {
        vst.clear();
        dfs(s, a, b);
        return *vst.begin();
    }
};
