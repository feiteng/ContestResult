class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int gcd10 = gcd(10, a);
        int m = 10 / gcd10;
        int n = gcd(s.size(), b);
        string res = s;
        // for (int i = 0; ; i += n) {
        //     string t = shift(s, i);
        //     // cout << "a "<< t << endl;
        //     if (i > 0 && t == s) break;
        //     for (int j = 0; ; j++) {
        //         string r = add(t, a * j);
        //         if (j > 0 && r == t) break;
        //         // cout << "b " << r << endl;
        //         if (res > r) res = r;
        //     }
        // }
        unordered_set<string> seen;
        dfs(a, b, s, seen, res);
        return res;
    }
    
private:
    void dfs(int a, int b, string& s, unordered_set<string>& seen, string& res) {
        if (!seen.insert(s).second) return;
        if (res > s) res = s;
        string t = shift(s, b);
        dfs(a, b, t, seen, res);
        t = add(s, a);
        dfs(a, b, t, seen, res);
    }
    

    string shift(string& s, int n) {
        string t = s;
        int l = s.size();
        for (int i = 0; i < l; i++) {
            t[(i + n) % l] = s[i];
        }
        return t;
    }
    
    string add(string& s, int n) {
        string t = s;
        int l = s.size();
        for (int i = 1; i < l; i += 2) {
            t[i] = (s[i] - '0' + n) % 10 + '0';
        }
        return t;
    }
};
