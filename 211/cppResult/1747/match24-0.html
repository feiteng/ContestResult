class Solution {
public:
    set<string> memo;
    string findLexSmallestString(string s, int a, int b) {
        // b 是偶数  还是 奇数
        bfs(s, a, b);
        return *memo.begin();
    }
    
    
    void bfs(string s, int a, int b) {
        if (memo.count(s)) {
            return;
        }
        memo.insert(s);
        string s1 = s;
        for (int i = 1; i < s.size(); i+=2) {
            s1[i] = (s1[i] - '0' + a) % 10 + '0';
        }
        bfs(s1, a, b);
        
        reverse(s.begin(), s.begin() + b);
        reverse(s.begin() + b, s.end());
        reverse(s.begin(), s.end());
        
        bfs(s, a, b);
        return;
    }
};
