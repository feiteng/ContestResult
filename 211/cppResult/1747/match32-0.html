int n;

string func(string s,int b)
{
    while(b--)
    {
        char c = s[n-1];
        s.erase(s.begin()+n-1);
        s.insert(s.begin(),c);
    }
    return s;
}

string solve(string s,int b)
{
    string ans = "";
    rep(i,0,n-1)
        ans += "9";
    rep(i,1,100)
    {
        s = func(s,b);
        ans = min(ans,s);
    }
    return ans;
}

class Solution {
public:
    string findLexSmallestString(string s, int a, int b) 
    {
        n = s.size();
        string ans = "";
        rep(i,0,n-1)
            ans += "9";
        for(int j=0;j<11;j++)
        {
            rep(i,0,n-1)
            {
               if(i%2==0)
                 continue;
                int f = s[i] - '0';
                f = (f+a)%10;
                s[i] = char(f+'0');
            }
            //deb1(s);
            if(b%2==0)
                ans = min(ans,solve(s,b));
            else
            {
                for(int jj=0;jj<11;jj++)
                {
                    rep(ii,0,n-1)
                    {
                       if(ii%2==1)
                         continue;
                        int f = s[ii] - '0';
                        f = (f+a)%10;
                        s[ii] = char(f+'0');
                    }
                    ans = min(ans,solve(s,b));
                }
            }
        }
        return ans;
    }
};
