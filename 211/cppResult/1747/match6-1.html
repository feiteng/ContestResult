class Solution {

    string aa(string s, int a){
        for(int i = 1; i < s.size(); i += 2){
            s[i] += a;
            if(s[i] > '9') s[i] -= 10;
        }
        return s;
    }
    string ab(string s, int b){
        string t = s;
        for(int i = 0; i < s.size(); i++){
            t[i] = s[(i + b) % s.size()];
        }
        return t;
    }
public:
    string findLexSmallestString(string s, int a, int b) {
        set<string> st, t;
        t.insert(s);
        while(!t.empty()){

            string x = *t.begin();
            t.erase(x);
            string sa = aa(x, a);
            if(!st.count(sa)){
                st.insert(sa);
                t.insert(sa);
            }
            string sb = ab(x, b);
            if(!st.count(sb)){
                st.insert(sb);
                t.insert(sb);
            }
        }
        return *st.begin();
    }
};
