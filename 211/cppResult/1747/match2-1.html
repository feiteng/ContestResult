class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int n = s.size();
        set<string> ss;
        ss.insert(s);
        queue<string> q;
        q.push(s);
        while (!q.empty()) {
            string s = q.front();
            q.pop();
            string as = s;
            for (int i = 1; i < n; i += 2) {
                char na = (s[i] - '0' + a) % 10 + '0';

                as[i] = na;
            }
            if (ss.count(as) == 0){
                ss.insert(as);
                q.push(as);
            }
                
            s = s.substr(n - b, b) + s.substr(0,n - b);
            if (ss.count(s) == 0){
                ss.insert(s);
                q.push(s);
            }
                
        }
        // for(auto as:ss)
        //     cout<<as<<endl;
        return *ss.begin();
    }
};
