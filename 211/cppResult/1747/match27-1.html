class Solution {
public:
    string rotate(string r, int b) {
        int n = r.size();
        return r.substr(n - b) + r.substr(0, n - b);
    }
    string findLexSmallestString(string s, int a, int b) {   
        string ans = s;
        int n = s.size();

        if (b % 2 == 1) {
            for (int _ = 0; _ < 10; _++) {
                            for (int i = 0; i < n; i+=2) {
                                s[i] = ((s[i] - 48) + a) % 10 + 48;
                            }

                for (int _ = 0; _ < 10; _++) {
                        for (int i = 1; i < n; i+=2) {
                           s[i] = ((s[i] - 48) + a) % 10 + 48;
                        }
                    string r = s;
                          for (int i = 0; i < n; i++) {
                            r  = rotate(r, b);
                            if (r < ans) {
                                ans = r;
                            }
        }
                }

            }
        }
           else {
               for (int _ = 0; _ < 10; _++) {
                        for (int i = n - 1; i >= 0; i-=2) {
                           s[i] = ((s[i] - 48) + a) % 10 + 48;
                        }
                    string r = s;
                          for (int i = 0; i < n; i++) {
                            r  = rotate(r, b);
                            if (r < ans) {
                                ans = r;
                            }
        }
                }
        }

        return ans;
    }
};
