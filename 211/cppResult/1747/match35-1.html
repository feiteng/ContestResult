class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        set<string> st;
        impl(s, a, b, st);
        return *st.begin();
    }
    
private:
    void impl(string& s, int a, int b, set<string>& st)
    {
        if (st.count(s))
            return;
        
        st.insert(s);
        
        auto s1 = s;
        rotate(s1.begin(), s1.end() - b, s1.end());
        impl(s1, a, b, st);
        
        auto s2 = s;
        for (int i = 1; i < s2.size(); i += 2)
            s2[i] = ((s2[i] - '0') + a) % 10 + '0';
        impl(s2, a, b, st);
    }
};
