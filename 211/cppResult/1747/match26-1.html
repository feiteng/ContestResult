class Solution {
public:
    unordered_set<string>vis;
    string ans;
    void f(string &s, int &a, int &b) {
        if(ans > s)
            ans = s;
        string str = s;
        for(int i = 1; i < s.size(); i+=2) {

            str[i] = ((str[i]-'0' + a)%10) + '0';
        }
        if(vis.find(str) == vis.end()) {
            vis.insert(str);
            f(str, a, b);
        }
        str = s.substr(s.size()-b) + s.substr(0, s.size() - b)   ;
        if(vis.find(str) == vis.end()) {
            vis.insert(str);
            f(str, a, b);
        }
    }
    string findLexSmallestString(string s, int a, int b) {
        ans = s;
        f(s, a, b);
        return ans;
    }
};
