class Solution {

public:
    string findLexSmallestString(string s, int a, int b) {
        set<string> used;
        
        queue<string> que;
        que.push(s);
        used.insert(s);
        
        while(!que.empty()){
            string cur = que.front();
            que.pop();
            string temp = addA(cur, a);
            if(used.find(temp) == used.end()){
                used.insert(temp);            
                que.push(temp);
            }
            
            temp = addB(cur, b);
            if(used.find(temp) == used.end()){
                used.insert(temp);            
                que.push(temp);
            }
            
        }
        
        return *used.begin();
    }
    

    string addA(string s, int a){
        for(int i=1;i<s.length();i+=2){
            s[i] += a;
            if(s[i] > '9')
                s[i] -= 10;
        }
        return s;
    }
    
    string addB(string s, int b){
        string str = s.substr(s.length()-b);
        str += s.substr(0, s.length()-b);        
        return str;
    }
};
