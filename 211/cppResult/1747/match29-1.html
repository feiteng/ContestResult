class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        string res = s;
        int len = s.length();
        bool flag = b%2;
        for(int i=0; i<len; i++) {
            s = s.substr(len-b, b) + s.substr(0, len-b);
            string t = s;

            for(int k=0; k<10; k++) {
                for(int j=len-1; j>=0; j-=2) {
                    t[j] += a;
                    if(t[j]>'9') t[j] -= 10;
                }
                if(t<res) res = t;
                if(!flag) continue;

                for(int kk=0; kk<10; kk++) {
                    for(int jj=len-2; jj>=0; jj-=2) {
                        t[jj] += a;
                        if(t[jj]>'9') t[jj] -= 10;
                    }
                    if(t<res) res = t;
                }
            }
        }
        return res;
    }
};
