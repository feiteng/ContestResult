class Solution {
public:
    set<string> st;
    string findLexSmallestString(string s, int a, int b) {
        dfs(s, a, b);
        return *st.begin();
    }
    void dfs(string s, int a, int b){
        if(st.count(s))
            return;
        st.insert(s);
        auto t = s;
        for(int i = 1; i < s.size(); i+=2)
            t[i] = '0'+((s[i]-'0'+a)%10);
        dfs(t, a, b);
        dfs(s.substr(s.size()-b,b)+s.substr(0, s.size()-b), a, b);
    }
};
