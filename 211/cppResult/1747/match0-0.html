int n;
map<string,int>mp;

auto solve(string rs,int b){
    vector<bool>Vis(n+1);
    int nw = b;
    vector<string>res;
    fo(i,0,n+n-1) s[i]=rs[i%n];
    while(!Vis[nw]){
        string Tp;
        fo(i,0,n-1){
            Tp+=s[(i+nw)];
        }
        if(mp.count(Tp)) {
            break;
        }
        res.push_back(Tp);
        Vis[nw]=1;
        nw+=b;
        nw%=n;
    }
    return res;
}

class Solution {
public:
    string findLexSmallestString(string rs, int a, int b) {
        mp.clear();
        n = rs.size();
        vector<string>res;
        res = solve(rs,b);
        for(auto v:res) mp[v]=1;

        fo(i,0,res.size()-1){
            string Tp=res[i];
            auto xx = solve(res[i],b);
            for(auto v:xx){
                if(!mp.count(v)) {
                    res.push_back(v);
                    mp[v]=1;
                }
            }

            fo(j,1,100000){//加上这么多次
                for(int k = 1;k<n;k+=2){
                    Tp[k]=trans(res[i][k],j*a);
                }
                if(mp.count(Tp)) break;
                res.push_back(Tp);
                mp[Tp]=1;
            }
        }

        string mi = res[0];
        for(auto v:res) mi = min(mi,v);
        return mi;
    }
};
