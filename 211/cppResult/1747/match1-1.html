class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        queue<string> q;
        set<string> vis;
        q.push(s);
        vis.insert(s);
        while(!q.empty()){
            auto s = q.front(); q.pop();
            auto u = add(s, a), v = move(s, b);

            if(!vis.count(u)){
                vis.insert(u);
                q.push(u);
            }
            if(!vis.count(v)){
                vis.insert(v);
                q.push(v);
            }
        }
        return *vis.begin();
    }
    string add(const string& s, int a){
        string t = s;
        for(int i = 1; i < t.size(); i += 2){

            t[i] = '0' + (t[i] - '0' + a) % 10;
        }
        return t;
    }
    string move(const string& s, int b){
        int n = s.size();
        return s.substr(n-b) + s.substr(0, n-b);
    }
};
