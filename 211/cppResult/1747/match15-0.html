class Solution {
public:
    string convert(string s, int a, int b, bool initr, int rotate, int prev, int post) {
        if (initr) s = s.substr(s.size() - b) + s.substr(0, s.size() - b);
        for (int i = 1 - s.size() % 2; i < s.size(); i += 2) {
            s[i] = (s[i] - '0' + a * prev) % 10 + '0';
        }
        rotate = rotate * b % s.size();
        s = s.substr(s.size() - rotate) + s.substr(0, s.size() - rotate);
        for (int i = 1 - s.size() % 2; i < s.size(); i += 2) {
            s[i] = (s[i] - '0' + a * post) % 10 + '0';
        }
        return s;
    }
    string findLexSmallestString(string s, int a, int b) {
        string ans = s;
        for (int prev = 0; prev <= 9; ++ prev) {
            for (int rotate = 0; rotate < s.size(); ++ rotate) {
                for (int post = 0; post <= 9; ++ post) {
                    ans = min(ans, convert(s, a, b, true, rotate, prev, post));
                    ans = min(ans, convert(s, a, b, false, rotate, prev, post));
                }
            }
        }
        return ans;
    }
};
