class Solution {
public:
    string findLexSmallestString(string s, int a, int b) {
        int n = s.length() ;
        string res = s ;
        if (b % 2 == 1) {
            for (int i = 0 ; i <= 9 ; ++i) {
                for (int j = 0 ; j <= 9 ; ++j) {
                    string temp = s ;
                    
                    for (int k = 0 ; k < n ; ++k) {
                        if (k % 2 == 1) {
                            temp[k] = ((temp[k] - '0' + i * a) % 10 + '0') ;
                        } else {
                            temp[k] = ((temp[k] - '0' + j * a) % 10 + '0') ;
                        }
                    }
                    string ends = temp ;
                    do {
                        //cout << i << " " << j << " " << temp << endl ;
                        if (temp < res) {
                            res = temp ;
                        }

                        string t = temp ;
                        for (int k = 0 ; k + b < temp.length() ; ++k) {
                            t[k + b] = temp[k] ;
                        }
                        for (int k = temp.length() - b , j = 0 ; k < temp.length() ; ++k , ++j) {
                            t[j] = temp[k] ;
                        }
                        temp = t ;
                    } while (temp != ends) ;

                }
            }
        } else {
            for (int i = 0 ; i <= 9 ; ++i) {
                    string temp = s ;
                    
                    for (int k = 0 ; k < n ; ++k) {
                        if (k % 2 == 1) {
                            int x = (temp[k] - '0') + i * a ;
                            x = (x + 10) % 10 ;
                            temp[k] = '0' + x ;
                        }
                    }
                string ends = temp ;
                    do {
                        //cout << i << " " << " " << temp << endl ;
                        if (temp < res) {
                            res = temp ;
                        }

                        string t = temp ;
                        for (int k = 0 ; k + b < temp.length() ; ++k) {
                            t[k + b] = temp[k] ;
                        }
                        for (int k = temp.length() - b , j = 0 ; k < temp.length() ; ++k , ++j) {
                            t[j] = temp[k] ;
                        }
                        temp = t ;
                    } while (temp != ends) ;
            }
        }
        return res ;
    }
};
