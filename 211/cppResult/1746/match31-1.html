class Solution {
public:
    int maxLengthBetweenEqualCharacters(string s) {
        int len = s.length();
        int ans = -1;
        for (int left = 0; left < len; left++) {
            for (int right = left; right < len; right++) {
                if (s[left] == s[right]) ans = max(ans, right - left - 1);
            }
        }
        return ans;
    }
};
