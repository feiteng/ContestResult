class Solution {
public:
    int maxLengthBetweenEqualCharacters(string s) {
        int res = -1;
        vector<int> a(26, -1);
        for (int i = 0; i < s.length(); ++i) {
            int c = s[i] - 'a';
            if (a[c] != -1) {
                res = max(res, i - a[c] - 1);
            } else {
                a[c] = i;
            }
        }
        return res;
    }
};
