using LL=long long;


class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n=ages.size();
        vector<pair<int, int>> a(scores.size());
        for(int i=0; i<n; i++){
            a[i].first = ages[i];
            a[i].second = scores[i];
        }
        
        sort(a.begin(), a.end());

        LL ans = 0;
        vector<LL> dp(n+10, 0);
        for(int i=1; i<=n; i++){
            dp[i]=a[i-1].second;
            for(int j=0; j<i; j++){
               if(j&&a[j-1].second<=a[i-1].second)
                   dp[i]=max(dp[i], dp[j]+a[i-1].second);
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
