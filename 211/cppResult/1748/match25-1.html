class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        vector< pair<int, int> > X;
        for (int i=0; i<scores.size(); i++)
            X.push_back({scores[i], ages[i]});
        sort(X.begin(), X.end());
        int dp[1003] = {0}, ans = 0;

        for (int i=0; i<X.size(); i++)
        {
            dp[i] = X[i].first;
            for (int j=0; j<i; j++)
            {
                if (X[j].second <= X[i].second)
                    dp[i] = max(dp[i], dp[j] + X[i].first);
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
