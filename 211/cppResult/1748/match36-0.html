class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int res = 0;
        vector<pair<int, int>> v;
        for(int i = 0; i < scores.size(); ++i) v.push_back(make_pair(ages[i], scores[i]));
        sort(v.begin(), v.end());
        vector<int> dp(ages.size(), 0);
        for(int i = 0; i < dp.size(); ++i) {
            dp[i] = v[i].second;
            for(int j = 0; j < i; ++j) {
                if(v[i].first == v[j].first || v[i].second >= v[j].second) dp[i] = max(dp[i], v[i].second + dp[j]);
            }
            res = max(dp[i], res);
        }
        return res;
    }
};
