class Solution {
public:
    int bestTeamScore(vector<int> scores, vector<int> ages) {
    vector<pair<int,int>> v;
    for(int i=0; i<scores.size(); i++)
        v.push_back({ages[i],scores[i]});
    sort(v.begin(),v.end());
    vector<int> dp(scores.size()+5,0);
    int res = 0;
    for(int i=1; i<=v.size(); i++){
        dp[i] = v[i-1].second;
        for(int k=1; k<i; k++){
            if(v[i-1].second >= v[k-1].second)
                dp[i] = max(dp[i],dp[k]+v[i-1].second);
        }
        res = max(res,dp[i]);
    }
    return res;
}
};
