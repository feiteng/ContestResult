const int maxn = 1e4 + 10;

struct node
{
    int nodes, b;
    bool operator<(const node t) const
    {
        if (b == t.b)
        {
            return nodes < t.nodes;
        }
        return b < t.b;
    }
} nodes[maxn];

int dp[maxn];
class Solution
{
public:
    int bestTeamScore(vector<int> &scr, vector<int> &ages)
    {
        int n = scr.size();
        fo(i, 0, n - 1) nodes[i + 1].nodes = scr[i];
        fo(i, 0, n - 1) nodes[i + 1].b = ages[i];
        sort(nodes + 1, nodes + 1 + n);
        me0(dp);
        fo(i, 1, n)
        {
            fo(j, 0, i - 1)
            {
                if (nodes[i].b == nodes[j].b)
                {
                    dp[i] = max(dp[i], dp[j] + nodes[i].nodes);
                }
                else if (nodes[i].nodes >= nodes[j].nodes)
                {
                    dp[i] = max(dp[i], dp[j] + nodes[i].nodes);
                }
            }
        }
        int res = 0;
        fo(i, 1, n)
             res = max(res, dp[i]);
        return res;
    }
};
