class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n = scores.size();
        vector<pair<int,int> > a; a.resize(n);
        for (int i = 0; i < n; i ++)
            a[i] = pair(scores[i], ages[i]);
        sort(a.begin(), a.end());
        vector<int> f; f.resize(n);
        int ret = 0;
        for (int i = 0; i < n; i ++)
        {
            f[i] = a[i].first;
            for (int j = 0; j < i; j ++)
                if (a[j].second <= a[i].second)
                    f[i] = max(f[i], f[j]+a[i].first);
            ret = max(ret, f[i]);
        }
        return ret;
    }
};
