class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        vector<pair<int, int>> pairs; // score, age
        
        for(int i = 0; i < scores.size(); i++)
        {
            pairs.push_back({scores[i], ages[i]});
        }
        
        sort(pairs.begin(), pairs.end());
        
        vector<int> dp(scores.size());
        
        int ans = 0;
        
        for(int i = 0; i < scores.size(); i++)
        {
            dp[i] = pairs[i].first;
            for(int j = i - 1; j >= 0; j--)
            {
                if(pairs[j].second <= pairs[i].second)
                {
                    dp[i] = max(dp[i], pairs[i].first + dp[j]);
                }
                // else if(pairs[j].second > pairs[i].second && pairs[j].first <= pairs[i].first)
                //     dp[i] = max(dp[i], pairs[i].first + dp[j]);
                    
            }
            
            ans = max(ans, dp[i]);
        }
        
        return ans;
    }
};
