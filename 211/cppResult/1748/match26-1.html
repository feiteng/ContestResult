class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int ans = 0;
        vector<pair<int, int>> member;
        for (int i = 0; i < scores.size(); ++ i)
            member.push_back(make_pair(ages[i], scores[i]));
        sort(member.begin(), member.end());
        vector<int> F(ages.size(), 0);

        for (int i = 0; i < ages.size(); ++ i) {
            F[i] = member[i].second;
            for (int j = 0; j < i; ++ j) {
                if (member[j].second <= member[i].second)
                    F[i] = max(F[i], F[j] + member[i].second);
            }
            ans = max(ans, F[i]);
        }
        return ans;
    }
};
