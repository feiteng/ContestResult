class Solution {
    #define ss second
    #define ff first
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        vector<pair<int , int>> vec;
        for(int i = 0 ; i < scores.size() ; i++){
            vec.push_back({ages[i] , scores[i]});
        }
        
        sort(vec.begin() , vec.end());
        
        int max1 = 0;
        
        int lis[scores.size()];

        for(int i = 0 ; i < scores.size() ; i++){
            lis[i] = vec[i].second;
            for(int j = i-1 ; j >= 0 ; j--){
                if(vec[j].ss <= vec[i].ss && lis[j] + vec[i].ss > lis[i]){
                    lis[i] = lis[j] + vec[i].ss;
                }
            }
            max1 = max(max1 , lis[i]);
        }
        return max1;
    }
};
