class Solution {
public:
    using PII = pair<int,int> ;
    int f22[1010] ;
    
    int bestTeamScore(vector<int>& a , vector<int>& b) {
        int n = a.size() ;
        vector<PII> ve ;
        for(int i = 0 ; i < n ; ++i)    ve.push_back({b[i],a[i]}) ;
        
        sort(ve.begin() , ve.end() ) ;
        
        for(int i = 0 ; i < n ; ++i)
        {
            f22[i] = ve[i].second ;
            for(int j = 0 ; j < i ; ++j)
            {
                if(ve[i].first > ve[j].first){
                    
                    
                    if(ve[i].second >= ve[j].second)
                        f22[i] = max(f22[i] , f22[j]+ve[i].second) ;
                    
                }
                else if(ve[i].first == ve[j].first){
                    f22[i] = max(f22[i] , f22[j]+ve[i].second) ;
                }
            }
        }
        
        int ans = 0 ;
        for(int i = 0 ; i < n ; ++i)
            ans = max(ans , f22[i]) ;
        return ans ;
    }
};
