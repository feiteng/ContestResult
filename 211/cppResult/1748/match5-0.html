class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n = scores.size();
        vector<pair<int, int>> a(n);
        for (int i = 0; i < n; ++i)
        {
            a[i].first = ages[i];
            a[i].second = scores[i];
        }
        sort(a.begin(), a.end());
        vector<int> dp(n);
        dp[0] = a[0].second;
        for (int i = 0; i < n; ++i)
        {
            dp[i] = a[i].second;
            for (int j = 0; j < i; ++j)
            {
                if (a[j].second <= a[i].second)
                {
                    dp[i] = max(dp[i], dp[j] + a[i].second);
                }
            }
        }
        return *max_element(dp.begin(), dp.end());
    }
};
