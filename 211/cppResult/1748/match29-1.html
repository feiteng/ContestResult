class Solution {
public:
    int bestTeamScore(vector<int>& S, vector<int>& A) {
        int n = A.size();
        vector<pair<int, int>> tmp(n);
        
        for (int i = 0; i < n; i++) {
            tmp[i] = {A[i], S[i]};
        }
        
        sort(tmp.begin(), tmp.end());
        for (int i = 0; i < n; i++) {
            S[i] = tmp[i].second;

        }
        
        vector<int> dp(n);
        for (int i = 0; i < n; i++) {
            dp[i] = S[i];
            for (int j = i - 1; j >= 0; j--) {
                if (S[j] <= S[i]) {
                    dp[i] = max(dp[i], S[i] + dp[j]);
                }
            }
        }
        
        return *max_element(dp.begin(), dp.end()); 
    }
};
