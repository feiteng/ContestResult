class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        vector<pair<int, int>> converted_list;
        for (int i = 0; i < scores.size(); ++i) {
            converted_list.push_back(make_pair(ages[i], scores[i]));
        }
        sort(converted_list.begin(), converted_list.end());
        int n = scores.size();
        vector<int> dp(n, 0);
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            dp[i] = converted_list[i].second;
            for (int j = 0; j < i; ++j) {
                if (converted_list[j].second <= converted_list[i].second) {
                    dp[i] = max(dp[i], dp[j] + converted_list[i].second);
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
