class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        vector<pair<int, int>> as;
        for(int i = 0; i < scores.size(); ++i) {
            as.push_back({ages[i], scores[i]});
        }
        sort(as.begin(), as.end());
        
        vector<int> res(scores.size(), 0);
        int ret = 0;
        for(int i = 0; i < as.size(); ++i) {
            int t = as[i].second;
            for(int j = i - 1; j >= 0; --j) {
                if(as[i].second >= as[j].second) {
                    t = max(t, res[j] + as[i].second);
                }
            }
            res[i] = t;
            ret = max(res[i], ret);
        }
        
        return ret;
    }
};
