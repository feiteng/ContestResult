const int maxn = 1e4+10;


struct node{
    int a,b;
    bool operator<(const node t)const{
        if(b==t.b) {
            return a<t.a;
        }
        return b<t.b;
    }
}A[maxn];

int dp[maxn];
class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n = scores.size();
        fo(i,0,n-1) A[i+1].a=scores[i];
        fo(i,0,n-1) A[i+1].b=ages[i];
        sort(A+1,A+1+n);
        me0(dp);
        fo(i,1,n){
            fo(j,0,i-1){
                if(A[i].b==A[j].b) {
                    dp[i] = max(dp[i],dp[j]+A[i].a);
                }
                else if(A[i].a>=A[j].a){
                     dp[i]=max(dp[i],dp[j]+A[i].a);
                }
            }
        }
        int ans = 0;
        fo(i,1,n) ans = max(ans,dp[i]);
        return ans;
    }
};
