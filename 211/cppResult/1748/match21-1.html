class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        
        int n = ages.size();
        vector< pair<int, int> > vec;
        for (int i = 0; i < n; ++i)
        {
            vec.push_back( {ages[i], scores[i]} );
        }
        sort(vec.begin(), vec.end());
        
        int res = 0;
        vector<int> dp(n, 0);

        for (int i = 0; i < n; ++i)
        {
            dp[i] = vec[i].second;
            
            for (int t = i - 1; t >= 0; --t)
            {
                if (vec[t].second <= vec[i].second) dp[i] = max(dp[i], dp[t] + vec[i].second);
            }
            
            res = max(res, dp[i]);
        }
        
        return res;
    }
};
