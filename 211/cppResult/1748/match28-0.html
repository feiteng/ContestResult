class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n = ages.size();
        vector<vector<int>> a(n,vector<int>(2));//unrodered_map<int,int> m;
        for(int i = 0;i<n;i++){a[i][0] = ages[i];a[i][1] = scores[i];}
        sort(a.begin(),a.end());
        int dp[n]; dp[0] = a[0][1];int ans = dp[0];
        for(int i = 1;i<n;i++){
            dp[i] = a[i][1];
            for(int j = 0;j<i;j++){
                if(a[i][0]==a[j][0]||a[i][1]>=a[j][1]) dp[i] = max(dp[i],a[i][1]+dp[j]);
            }
            ans = max(ans,dp[i]);
        }
        return ans;
    }
};
