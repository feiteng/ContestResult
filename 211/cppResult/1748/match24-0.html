class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        vector<pair<int,int>> nums;
        for (int i=0;i<scores.size();i++)
            nums.push_back({ages[i],scores[i]});
        sort(nums.begin(),nums.end());
        int res=0;
        vector<int> best(nums.size(),0);
        for (int i=0;i<scores.size();i++)
        {
            best[i]=nums[i].second;
            for (int j=0;j<i;j++)
            {
                if (nums[i].first==nums[j].first || nums[i].second>=nums[j].second)
                    best[i]=max(best[i],best[j]+nums[i].second);
            }
            res=max(res,best[i]);
        }
        return res;
    }
};
