class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n = scores.size();
        vector<pair<int, int>> people(n);
        for(int i = 0; i < n; i++){
            people[i].first = ages[i];
            people[i].second = scores[i];
        }
        
        sort(people.begin(), people.end());
        
        vector<int> dp(n);
        dp[0] = people[0].second;
        for(int i = 1; i < n; i++){
            dp[i] = people[i].second;
            for(int j = 0; j < i; j++){
                if(people[i].second >= people[j].second){
                    dp[i] = max(dp[i], people[i].second + dp[j]);
                }
            }
        }
        return *max_element(dp.begin(), dp.end());
        
    }
};
