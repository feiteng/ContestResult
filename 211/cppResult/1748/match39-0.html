class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
      int n=ages.size();
      vector<pair<int,int>> A(n);
      for(int i=0;i<n;++i){
        A[i].first=ages[i];
        A[i].second=scores[i];
      }
      sort(A.begin(),A.end());
      vector<int> dp(n);
      int res=0;
      for(int i=0;i<n;++i){
        dp[i]=A[i].second;
        for(int j=i-1;j>=0;--j){
          if(A[j].second<=A[i].second){
            dp[i]=max(dp[j]+A[i].second,dp[i]);
          }
        }
        res=max(res,dp[i]);
      }
      return res;
    }
};
