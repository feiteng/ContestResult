class Solution {
public:
    using PII = pair<int,int> ;
    int f[1010] ;
    
    int bestTeamScore(vector<int>& a , vector<int>& b) {
        int n = a.size() ;
        vector<PII> vec ;
        for(int i = 0 ; i < n ; ++i)    vec.push_back({b[i],a[i]}) ;
        
        sort(vec.begin() , vec.end() ) ;
        
        for(int i = 0 ; i < n ; ++i)
        {
            f[i] = vec[i].second ;
            for(int j = 0 ; j < i ; ++j)
            {
                if(vec[i].first > vec[j].first){
                    if(vec[i].second >= vec[j].second)
                        f[i] = max(f[i] , f[j]+vec[i].second) ;
                }else if(vec[i].first == vec[j].first){
                    f[i] = max(f[i] , f[j]+vec[i].second) ;
                }
            }
        }
        
        int ans = 0 ;
        for(int i = 0 ; i < n ; ++i)
            ans = max(ans , f[i]) ;
        return ans ;
    }
};
