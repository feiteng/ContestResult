class Solution {
public:
    int bestTeamScore(vector<int>& scores, vector<int>& ages) {
        int n=scores.size();
        vector<pair<int,int>> can;
        for(int i=0;i<n;i++){
            can.push_back({ages[i], scores[i]});
        }
        sort(can.begin(),can.end());
        vector<int> dp(n,0);
        int ret=0;

        for(int i=0;i<n;i++){
            dp[i]=can[i].second;
            for(int last=0;last<i;last++){
                
                if(can[last].first==can[i].first||can[last].first<can[i].first&&can[last].second<=can[i].second)
                    dp[i]=max(dp[i],dp[last]+can[i].second);

            }
            ret=max(ret,dp[i]);
        }
        return ret;
    }
};
