class Solution {
public:
    
    int find(int arr[],int i){
	if(arr[i]!=i)
		arr[i]=find(arr,arr[i]);
	return arr[i];
    }

    void unionn(int a,int b,int arr[]){
        int pa=find(arr,a);
        int pb=find(arr,b);
        arr[pa]=pb;
        return ;
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        
        int arr[n+1];
        for(int i=1;i<=n;i++)arr[i] = i;
        for(int i=1;i<=n;i++){
            vector<int> v1;
            for(int j=1;j*j<=i;j++){
                if(i%j==0){
                    v1.push_back(j);
                    v1.push_back(i/j);
                }
            }
            
            for(int j=0;j<v1.size();j++){
                if(v1[j]>threshold){
                    unionn(i,v1[j],arr);
                }
            }
        }
        
        
        vector<bool> res;
        int m = queries.size();
        for(int i=0;i<m;i++){
            if(find(arr,queries[i][0])==find(arr,queries[i][1]))  //>threshold and ok(queries[i][0],queries[i][1],n,threshold))
                res.push_back(true);
            else
                res.push_back(false);
        }
        return res;
    }
};
