const int maxn = 1e6+10;

int n,m,a[maxn],q,t;
char s[maxn];

class Solution {
public:

    int fa[maxn];
    int find(int x){
        return x==fa[x]?x:fa[x]=find(fa[x]);
    }

    pii edge[maxn];

    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        fo(i,1,n){
            fa[i]=i;
        }
        int tot = 0;
        fo(i,threshold+1,n){
            for(int j = i+i;j<=n;j+=i){
                edge[++tot]={i,j};
            }
        }

        fo(i,1,tot){
            int fx = find(edge[i].first),fy = find(edge[i].second);
            if(fx==fy) continue;
            fa[fy]=fx;
        }

        vector<bool>ans;
        for(auto to:queries){
            int u = to[0],v=to[1];
            ans.push_back(find(u)==find(v));
        }
        return ans;

    }
};
