class UF{
    vector<int> par;
    vector<int> sz;
    public:
        UF(int n)
        {
            for(int i = 0;i <= n;i++)
            {
                par.push_back(i);
                sz.push_back(1);
            }

        }
        int find_par(int a)
        {
            if(par[a] == a)
                return a;
            return par[a] = find_par(par[a]);
        }
        void join(int a,int b)
        {
            a = find_par(a);
            b = find_par(b);
            if(a == b)
                return;
            if(sz[b] > sz[a])
                swap(a,b);
            par[b] = a;
            sz[a] += sz[b];

        }
};
class Solution {
public:
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        vector<bool> ans;
        UF uf(n);
        for(int i = threshold+1;i < n;i++)
        {
            for(int j = i;j <= n;j+=i)
                uf.join(i,j);
        }
        for(auto x:queries)
        {
            ans.push_back(uf.find_par(x[0]) == uf.find_par(x[1]));
        }
        return ans;
    }
};
