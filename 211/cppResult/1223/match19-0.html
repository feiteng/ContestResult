const int maxn = 1e6 + 10;

int n, m, a[maxn], q, t;
char s[maxn];

class Solution
{
public:
    int fa[maxn];
    int find(int x)
    {
        return x == fa[x] ? x : fa[x] = find(fa[x]);
    }

    pii Ed[maxn];

    vector<bool> areConnected(int n, int thd, vector<vector<int>> &qrs)
    {
        fo(i, 1, n)
        {
            fa[i] = i;
        }
        int tot = 0;
        fo(i, thd + 1, n)
        {
            for (int j = i + i; j <= n; j += i)
            {
                Ed[++tot] = {i, j};
            }
        }

        fo(i, 1, tot)
        {
            int fx = find(Ed[i].first), fy = find(Ed[i].second);
            if (fx == fy)
                continue;
            fa[fy] = fx;
        }

        vector<bool> ans;
        for (auto to : qrs)
        {
            int u = to[0], v = to[1];
            ans.push_back(find(u) == find(v));
        }
        return ans;
    }
};
