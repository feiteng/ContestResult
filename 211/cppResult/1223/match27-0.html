class Solution {
public:
    int find(vector<int>& pa, int x){
        if(x != pa[x])
            pa[x] = find(pa,pa[x]);
        return pa[x];
    }
    
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        if(threshold==0) return vector<bool>(queries.size(),true);
        vector<int> pa(n+1,0);
        for(int i = 1; i <= n; i++)
            pa[i] = i;
        for(int i = threshold+1; i <= n/2; i++){
            for(int j = i*2; j <= n; j += i){
                int x = find(pa,i), y = find(pa,j);
                pa[y] = x;
                find(pa,j);
            }
        }
        vector<bool> ans;
        for(auto q:queries){
            ans.push_back(find(pa,q[0])==find(pa,q[1]));
        }
        return ans;
    }
};
