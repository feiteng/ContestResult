class Solution {
public:

    // set<int> get(int x)

    static const int N = 1e4+10 ;

    int fa[N] ;
    int find(int x)
    {
        return fa[x] = fa[x]==x?x:find(fa[x]) ;
    }
    
    vector<bool> areConnected(int n, int k, vector<vector<int>>& q) {
        int m = q.size() ;
        

        for(int i = 1 ; i <= n ; ++i)   fa[i] = i ;
        
        for(int i = k+1 ; i <= n ; ++i)
        {
            for(int j = i+i ; j <= n ; j += i)
            {
                fa[find(j)] = find(i) ;
            }
        }
        
        
        vector<bool> ans ;
        for(int i = 0 ; i < m ; ++i)
        {
            int a = q[i][0] , b = q[i][1] ;
            if(find(a) == find(b))
                ans.push_back(true) ;
            else
                ans.push_back(false) ;
           
        }
        
        return ans ;
    }
};
