class Solution {
public:
    int find(vector<int>& grp, int x) {
        if (grp[x] == x) return x;
        return grp[x] = find(grp, grp[x]);
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        vector<int> grp(n+1, 0);

        for(int i = 0; i <= n; i++)
            grp[i] = i;
        for(int th = threshold+1; th <= n/2; th++) {
            for(int j = 2; j*th <= n; j++) {
                int r1 = find(grp, th);
                int r2 = find(grp, th*j);
                if (r1 != r2)

                    grp[r1] = grp[r2] = min(r1, r2);
            }
        }
        vector<bool> ans;
        for(auto vec: queries) {
            if (find(grp, vec[0]) == find(grp, vec[1]))
                ans.push_back(true);
            else
                ans.push_back(false);
        }
        return ans;
        
    }
};
