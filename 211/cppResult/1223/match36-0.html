class Solution {
public:
    int gcd(int a,int b){
        return b == 0?a:gcd(b,a%b);
    }
    int GetFather(int a){
        int temp = a;
        while(ua[a] != a){
            a = ua[a];
        }
        while(temp != ua[temp]){
            int t = temp;
            temp = ua[temp];
            ua[t] = a;
        }
        return a;
    }
    void AddOne(int a,int b){
        int fa = GetFather(a);
        int fb = GetFather(b);
        if(fa != fb) ua[fa] = ua[fb];
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& q) {
        ua.assign(n+1,0);
        for(int i = 1;i<=n;++i){
            ua[i] = i;
        }

        for(int i = 1;i<=n;++i){
            for(int j = i;j<=n;j+=i){
                if(GetFather(i) == GetFather(j)) continue;
                if(i > threshold){
                    AddOne(i,j);
                }
            }
        }
        vector<bool> ret(q.size(),0);
        for(int i = 0;i<q.size();++i)
        {
            int a = q[i][0];
            int b = q[i][1];
            if(a > b) swap(a,b);
            ret[i] = GetFather(a) == GetFather(b);
        }
        return ret;
    }
    vector<int> ua;
};
