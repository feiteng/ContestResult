class Solution {
public:
    int fa[10001];
    

    int find(int x) {
        int tmp = x;
        while(fa[x] != x) {
            x = fa[x];
        }
        while(fa[tmp] != tmp) {
            int f = fa[tmp];
            fa[tmp] = x;
            tmp = f;
        }
        return x;
    }
    
    void merge(int u, int v) {
        int fu = find(u);
        int fv = find(v);

        fa[fu] = fv;
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        for(int i = 1; i <= n; i++) {
            fa[i] = i;
        }
        for(int i = threshold+1; i <= n; i++) {
            for(int j = i+i; j <= n; j += i) {
                merge(i, j);
            }
        }
        vector<bool> anw;
        for(const auto &q : queries) {
            anw.push_back(find(q[0]) == find(q[1]));
        }
        return anw;
    }
};
