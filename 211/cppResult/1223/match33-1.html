#define pb push_back
const int maxn=1e4+100;

class Solution {
public:
    int pre[maxn];
    int find(int x)
    {
        return x==pre[x]?pre[x]:pre[x]=find(pre[x]);
    }
    vector<bool> areConnected(int n, int thre, vector<vector<int>>& Q) {
        vector<vector<int>> E;
        int m=Q.size();
        if(thre==0) return vector<bool>(m,true); 

        for(int i=0;i<=n;i++) pre[i]=i;
        for(int i=thre+1;i<=n;i++)
        {
            for(int j=i;j<=n;j+=i)
                if(j!=i) E.pb({j,j-i});
        }
        for(auto& e:E)
        {
            int x=e[0],y=e[1];
            x=find(x),y=find(y);
            if(x!=y) pre[x]=y;
        }
        vector<bool> res;
        for(auto& q:Q)
        {
            int x=q[0],y=q[1];
            x=find(x),y=find(y);
            if(x==y) res.pb(true);
            else res.pb(false);
        }
        return res;
    }
};
