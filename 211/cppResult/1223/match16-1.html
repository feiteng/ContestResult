class Solution {
public:
    int findPar(int a,vector<int> &parent)
    {
        if(a!=parent[a])
            parent[a] = findPar(parent[a],parent);
        return parent[a];
    }
    void dsu(int a,int b,vector<int> &parent)
    {
        int par_a = findPar(a,parent);
        int par_b = findPar(b,parent);
        if(par_a!=par_b)
            parent[par_b] = par_a;

    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        vector<int> parent(n+1);
        for(int i=1;i<=n;i++)
            parent[i] = i;
        for(int i=threshold+1;i<=n;i++)
        {
            for(int k=1;k<=n&&i*k<=n;k++)
                dsu(i*k,i,parent);

        }
        vector<bool> answer;
        int m = queries.size();
        for(int i=0;i<m;i++)
        {
            if(findPar(queries[i][0],parent)==findPar(queries[i][1],parent))
                answer.push_back(true);
            else
                answer.push_back(false);
        }
        return answer;
    }
};
