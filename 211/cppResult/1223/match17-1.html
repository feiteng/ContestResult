class Solution {
private:

    int parent[10004], sizex[10004];
    int find_parent(int x) {
        if (x == parent[x])
            return x;
        return parent[x] = find_parent(parent[x]);
    }
    void unite(int a, int b) {
        a = find_parent(a);
        b = find_parent(b);
        if (a == b)
            return;
        if (sizex[a] < sizex[b])
            swap(a, b);
        parent[b] = a;
        sizex[a] += sizex[b];
    }
public:
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        threshold++;

        for (int i=1; i<=n; i++)
            parent[i] = i, sizex[i] = 1;
        for (int i=threshold; i<=n; i++) {
            for (int j=i; j<=n; j+= i)
                unite(i, j);
        }
        vector<bool> res;
        for (auto q: queries) {
            res.push_back(find_parent(q[0]) == find_parent(q[1]));
        }
        return res;
    }
};
