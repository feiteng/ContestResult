class Solution {
public:
    int par[10005];
    int find(int x) {return x==par[x]?x:par[x]=find(par[x]);}
    vector<bool> areConnected(int n, int t, vector<vector<int>>& queries) {
        for(int i=1;i<=n;++i) par[i]=i;
        for(int i=t+1;i<=n;++i) {
            for(int j=i;j<=n;j+=i) par[find(j)]=find(i);
        }
        vector<bool>ret;
        for(auto &v:queries) ret.push_back(find(v[0])==find(v[1]));
        return ret;
    }
};
