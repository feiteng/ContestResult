class Solution {
public:
    int f[10005];
    int Find(int x){
        if(f[x]==x)return x;
        return f[x]=Find(f[x]);
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        for(int i = 1;i<=n;i++)f[i]=i;
        for(int i = threshold+1;i<=n;i++){
            for(int j =i;j<=n;j+=i){
                int a=Find(j),b=Find(i);
                f[a]=b;
            }
        }
        vector<bool> res;
        for(auto it:queries){
            if(Find(it[0])==Find(it[1]))res.push_back(true);
            else res.push_back(false);
        }
        return res;
    }
};
