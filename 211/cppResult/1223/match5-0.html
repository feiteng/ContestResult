class Solution {
    int ancestor[10001];
    
    int findAncestor(int x) {
        if (ancestor[x] == x) return x;
        return ancestor[x] = findAncestor(ancestor[x]);
    }
public:
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        for (int i = 1; i <= n; i++)
            ancestor[i] = i;
        
        for (int div = threshold + 1; div <= n; div++) {
            for (int val = div; val + div <= n; val += div) {
                ancestor[findAncestor(val)] = findAncestor(val + div);
            }
        }
        
        vector<bool> res;
        for (auto &v : queries) {
            int a = v[0];
            int b = v[1];
            res.push_back(findAncestor(a) == findAncestor(b));
        }
        
        return res;
    }
};
