class Solution {
public:
    int find(vector<int> &t,int x){
        if(t[x] != x) t[x] = find(t,t[x]);
        return t[x];
    }
    vector<bool> areConnected(int n, int th, vector<vector<int>>& queries) {
        vector<int> t(n,0);
        for(int i=0;i<n;i++) t[i] = i;
        for(int j=th+1;j<=n;j++){
            for(int k=2;(k*j)<=n;k++){
                int p = j-1;
                int b = k*j-1;
                t[find(t,b)] = t[find(t,p)];
            }
        }
        vector<bool> res;
        for(auto item:queries){
            int x = item[0],y = item[1];
            if(t[find(t,x-1)]==t[find(t,y-1)]){
                res.push_back(true);
            }else res.push_back(false);
        }
        return res;
    }
};
