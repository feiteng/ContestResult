class Solution {
private:
    vector <int> root;
    
    int find_root(int x)
    {
        if (x == root[x]) return x;
        return root[x] = find_root(root[x]);
    }
    
    
public:
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        root.resize(n + 1);
        for (int i = 1; i <= n; i ++) root[i] = i;
        for (int k = threshold + 1; k <= n; k ++)
        {
            for (int i = k; i <= n; i += k)
            {
                root[find_root(i)] = find_root(k);
            }
        }
        vector <bool> rlt;
        for (int i = 0; i < queries.size(); i ++)
        {
            bool flag = false;
            int x = queries[i][0];
            int y = queries[i][1];
            if (find_root(x) == find_root(y)) flag = true;
            rlt.push_back(flag);
        }
        return rlt;
    }
};
