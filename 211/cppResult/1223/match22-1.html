class Solution {
public:
    int c[10001];
    int find(int i){
        return c[i]=(c[i]==i?i:find(c[i]));
    }
    void mer(int i,int j){
        i = find(i);j = find(j);

        c[i] = j;
    }
    vector<bool> areConnected(int n, int t, vector<vector<int>>& queries) {
        for(int i = 0;i<10001;i++) c[i] = i;
        for(int i = t+1;i<n;i++){
            for(int j = 2;i*j<=n;j++){mer(i,i*j);}
        }
        vector<bool> ans;
        for(auto q:queries){
            ans.push_back(find(q[0])==find(q[1]));
        }
        return ans;
    }
};
