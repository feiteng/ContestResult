class Solution {
public:
    int f[10010];
    int _find(int x){return x==f[x]?x:f[x]=_find(f[x]);}
    vector<bool> areConnected(int n, int t, vector<vector<int>>& q) {
        for(int i=1;i<=n;i++)f[i] = i;
        for(int i=t+1;i<=n;i++){
            for(int j=2*i;j<=n;j+=i){
                if(_find(j)!=_find(i)){
                    f[_find(j)] = _find(i);
                }
            }
        }
        //for(int i=1;i<=100;i++)printf("%d %d\n",i,f[i]);
        for(int i=1;i<=n;i++)_find(i);
        vector<bool>ans;
        for(int i=0;i<q.size();i++){
            int a = q[i][0],b = q[i][1];
            if(f[a]!=f[b])ans.push_back(false);
            else ans.push_back(true);
        }
        return ans;
    }
};
