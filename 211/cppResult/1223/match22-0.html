class Solution {
public:
    
    int par[10005];
    
    int findpar(int x) {
        return par[x] = par[x] == x ? x : findpar(par[x]);
    }
    
    void unite(int x, int y) {
        x = findpar(x), y = findpar(y);
        if (x == y) return;
        par[x] = y;
    }
    
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        vector<bool> vec;
        for (int i = 1; i <= n; i ++) par[i] = i;
        for (int z = threshold + 1; z <= n; z ++) {
            for (int i = z; i <= n; i += z)
                for (int j = i + z; j <= n; j += z)  {
                    // cout << i << " " << j << endl;
                    unite(i, j);
                }
        }
        for (auto q : queries) {
            // cout << q[0] << " " << findpar(q[0]) << " " << findpar(q[1]) << endl;
                vec.push_back(findpar(q[0]) == findpar(q[1]));
        }
        return vec;
    }
};
