class Solution {
public:
    
    
    int getP(vector<int> &par, int i){
        if(par[i] == i) return par[i];
        return par[i] = getP(par, par[i]);
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        vector<bool> ans;
        vector<int> par(n, -1);
        for(int i =0;i<n;i++) par[i] = i;
        for(int j = threshold+1;j<=n;j++){
            par[j-1] = getP(par, j-1);

            for(int q =2;q *j <=n;q++){
                int i=q*j;
                int t = getP(par, i-1);
                if(t != par[j-1]) par[t] = par[j-1];
            }
        }
        
        for(auto &q:queries){
            int a = getP(par, q[0]-1), b = getP(par, q[1]-1);
            ans.push_back(a == b);
        }
        return ans;
    }
    
    
};
