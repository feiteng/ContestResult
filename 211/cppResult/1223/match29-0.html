class Solution {
public:
    int p[10005];
    int f(int i){
        if(p[i]==i)return p[i];
        return p[i]=f(p[i]);
    }
    vector<bool> areConnected(int n, int t, vector<vector<int>>& q) {
        int i,j,k,m;
        for(i=0;i<n;i++)p[i]=i;
        for(i=t+1;i<=n;i++){
            for(j=2*i;j<=n;j+=i){
                int l=f(i);
                int r=f(j);
                if(l!=r)p[l]=r;
            }
        }
        vector<bool>ans;
        for(auto it:q){
            int l=f(it[0]);
            int r=f(it[1]);
            ans.push_back(l==r);
        }
        return ans;
    }
};
