class Solution {
public:
    int fa[10010];
    int tfind(int x){
        if (fa[x]==x) return x;
        return fa[x]=tfind(fa[x]);
    }
    vector<bool> areConnected(int n, int threshold, vector<vector<int>>& queries) {
        for (int i=1;i<=n;i++) fa[i]=i;
        for (int i=threshold+1;i<=n;i++){
            for (int j=2;i*j<=n;j++){
                int u = tfind(i*(j-1));
                int v = tfind(i*j);
                fa[u]=v;
            }
        }
        vector<bool> res;
        for (const auto &it:queries){
            int u = tfind(it[1]);
            int v = tfind(it[0]);
            if (u==v) res.push_back(true);
            else res.push_back(false);
        }
        return res;
    }
};
