class Solution {
public:
    int par[10002];
    int find(int x){
        return x==par[x]?x:par[x]=find(par[x]);
    }
    vector<bool> ans;
    
    vector<bool> areConnected(int n, int m, vector<vector<int>>& queries) {
        for(int i=0;i<=n;i++) par[i]=i;
        for(int i=m+1;i<=n;i++){
            for(int j=i*2;j<=n;j+=i){
                int fx=find(j),fy=find(j-i);
                if(fx==fy) continue;
                par[fx]=fy;
            }
        }
        for(auto x:queries){
            ans.push_back(find(x[0])==find(x[1]));
        }
        return ans;
    }
};
