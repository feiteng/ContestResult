class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int[] index = new int[26];
        Arrays.fill(index, -1);
        int ans = -1;
        for(int i=0; i<s.length(); i++) {
            int a = s.charAt(i)-'a';
            if(index[a] != -1) {
                ans = Math.max(ans, i-index[a]-1);
            } else {
                index[a] = i;
            }
        }
        return ans;
    }
}
