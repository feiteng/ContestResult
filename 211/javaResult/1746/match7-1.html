class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int[] map = new int[128];
        Arrays.fill(map, -1);
        int res = -1;
        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);
            if (map[ch] != -1){
                res = Math.max(res, i - map[ch] - 1);
            }
            else {
                map[ch] = i;
            }
        }
        return res;
    }
}
