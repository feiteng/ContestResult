class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int[] first = new int[26];
        for (int ch=0; ch<26; ch++)
            first[ch] = -1;
        int max = 0;
        for (int i=0; i<s.length(); i++) {
            int ch = s.charAt(i)-'a';
            if (first[ch] != -1) {
                max = Math.max(max, i - first[ch]);
            } else {
                first[ch] = i;
            }
        }
        return max - 1;
    }
}
