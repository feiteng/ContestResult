class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int[] first = new int[26], last = new int[26];
        int n = s.length();
        for(int i = 0; i < n; i++)
        {
            char c = s.charAt(i);
            last[c - 'a'] = i;
        }
        for(int i = n - 1; i >= 0; i--)
        {
            char c = s.charAt(i);
            first[c - 'a'] = i;
        }
        
        int len = -1;
        for(int i = 0; i < 26; i++)
        {
            len = Math.max(len, last[i] - first[i] - 1);    
        }
        return len;
    }
}
