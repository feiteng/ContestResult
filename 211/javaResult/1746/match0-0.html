class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int ans = -1;
        HashMap<Character, Integer> seen = new HashMap<>();
        for(int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            if(seen.containsKey(c))
                ans = Math.max(ans, i - seen.get(c) - 1);
            else
                seen.put(c, i);
        }
        return ans;
    }
}
