class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int res = -1;
        int l = s.length();
        for (int i = 0; i < l; i++) {
            for (int j = i + 1; j < l; j++) {
                if (s.charAt(i) == s.charAt(j)) {
                    res = Math.max(res, j - i - 1);
                }
            }
        }
        return res;
    }
}
