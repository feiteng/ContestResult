class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        int before = 0;
        int max = -1;
        
        for(int i = 0; i < 26; i++){
            char now = (char) ('a' + i);
            int loc = -1;
            for(int j = 0; j < s.length(); j++){
                char cur = s.charAt(j);
                if(cur == now){
                    if(loc == -1){
                        loc = j;
                    }
                    else{
                        max = Math.max(max, j - loc - 1);
                    }
                }
            }
        }
        
        return max;
    }
}
