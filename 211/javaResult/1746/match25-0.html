class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        Map<Character, Integer> prev = new HashMap<>();
        int out = -1;
        
        for(int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            
            if(prev.containsKey(c)) {
                out = Math.max(out, i - prev.get(c) - 1);
            } else {
                prev.put(c, i);
            }
        }
        
        return out;
    }
}
