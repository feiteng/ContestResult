class Solution {
    public int maxLengthBetweenEqualCharacters(String s) {
        HashMap<Character, Integer> map = new HashMap<>();
        int result = -1;
        for (int i = 0; i<s.length(); i++){
            char c = s.charAt(i);
            if (map.containsKey(c)){
                result = Math.max(result, i - 1 - map.get(c));
            } else {
                map.put(c, i);
            }
        }
        return result;
    }
}
