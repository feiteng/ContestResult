class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        int[] parent = new int[n+1];
        for(int i=0;i<n;i++) {
            parent[i] = i;
        }
        for(int i=threshold+1;i<=n/2;i++) {
            for(int j=2;j<=n/i;j++) {
                union(parent,i,i*j);
            
}    
        }
        
        List<Boolean> res = new LinkedList<>();
        for(int i=0;i<queries.length;i++) {
            res.add(
find(parent,queries[i][0]) == find(parent,queries[i][1]));
        }
        return res;
    }
    
    int find(int[] p,int v) {
        if (p[v] == v) {
            return v;
        } else {
            p[v] = find(p,p[v]);
            return p[v];
        }
    }
    
    void union(int[] p,int v1,int v2) 
{
        int v = find(p,v1);
        int w = find(p,v2);
        if (v != w) {
            p[v] = w;
        }
    }
}
