class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        int[] p = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            p[i] = i;
        }
        
        for (int i = 1; i <= n; i++) {
            for (int k = 1; k * k <= i; k++) {
                if (i % k == 0 && k > threshold) {
                    int pa = find(k, p);
                    int pb = find(i, p);
                    if (pa != pb) {
                        p[pa] = p[pb];
                    }
                }
                if (i % k == 0 && i / k > threshold) {
                    int pa = find(i, p);
                    int pb = find(i / k, p);
                    if (pa != pb) {
                        p[pa] = p[pb];
                    }
                }
            }
        }
        
        List<Boolean> res = new ArrayList<>();
        for (int[] q : queries) {
            int pa = find(q[0], p);
            int pb = find(q[1], p);
            res.add(pa == pb);
        }
        return res;
    }
    
    private int find(int x, int[] p) {
        if (p[x] == x) return x;
        p[x] = find(p[x], p);
        return p[x];
    }
}
