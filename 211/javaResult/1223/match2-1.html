class Solution {
    
public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        
        UnionFind uf = new UnionFind(n);
        
        for(int i = 0; i < n; i++) {
            if(i + 1 <= threshold) continue;
            for(int j = i + 1; j < n; j++) {
                if(uf.find(i) != uf.find(j) && isConnected(j + 1, i + 1, threshold)) {
                    uf.union(i, j);
                }
            
}
        }
        
        List<Boolean> res = new ArrayList<>();
        for(int[] q : queries) {
            res.add(uf.find(q[0] - 1) == uf.find(q[1] - 1));
        }
        return res;
    }
    
    
class UnionFind {
        int n;
        int[] parents;
        int[] ranks;
        public UnionFind(int n) {
            this.n = n;
            this.parents = new int[n];
            this.ranks = new int[n];
            
            for(int i = 0; i < n; i++) {
                parents[i] = i;
                
ranks[i] = 1;
            }
        }
        
        public int find(int x) {
            while(x != parents[x]) {
                parents[x] = parents[parents[x]];
                x = parents[x];
            }
            return x;
        }
        public void union(int x, int y) {
            
            int px = find(x);
            int py = find(y);
            
            if(px == py) return;
            
            if(ranks[px] > ranks[py]) {
                parents[py] = px;
            } else {
                parents[px] = py;
                if(ranks[px] == ranks[py]) ranks[py]++;
            }
            
        }
    }
    
    private boolean isConnected(int i, int j, int thres) {
        
        if(j == 0) return i > thres;
        
        if(i <= thres) return false;
        
        return isConnected(j, i % j, thres);
    }
}
