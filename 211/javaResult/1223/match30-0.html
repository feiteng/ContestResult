class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        List<Boolean> res = new LinkedList<>();
        UnionFind uf = new UnionFind(n + 1);
        if (threshold < 1) {
            for (int[] q : queries)
                res.add(true);
            return res;
        }
        
        for (int i = threshold + 1; i <= n; i++) {
            for (int j = 2; i * j <= n; j++) {
                uf.union(i, i * j);
            }
        }
        
        for (int[] q : queries) {
            if (uf.find(q[0]) == uf.find(q[1])) {
                res.add(true);
            } else {
                res.add(false);
            }
        }
        return res;
    }
}
  
class UnionFind {
    private int[] id;
    private int[] sz;
    
    public UnionFind(int n) {
        id = new int[n];
        sz = new int[n];
        for (int i = 0; i < n; i++) {
            sz[i] = 1;
            id[i] = i;
        }
    }
    
    public int find(int i) {
        while (i != id[i]) {
            id[i] = id[id[i]]; //path compression
            i = id[i];
        }
        return i;
    }
    
    public void union(int p , int q) {
        int i = find(p);
        int j = find(q);
        if (i == j) return;
        if (sz[i] < sz[j]) { // weighting
            id[i] = j; 
            sz[j] += sz[i]; 
        } else { 
            id[j] = i; 
            sz[i] += sz[j]; 
        }
        // otherwise just use id[i] = j;
    }
}
