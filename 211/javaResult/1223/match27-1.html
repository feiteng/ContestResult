class Solution {
    
class UnionFind {
        int[] parent;
        
        UnionFind(int n) {
            parent = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
            }
        }
        
        public int find(int x) {
            if (parent[x] != x) {
                parent[x] = find(parent[x]);
            }
            return parent[x];
        }
        
        public void union(int x, int y) {
            int px = find(x);
            int py = find(y);
            if (px != py) {
                parent[px] = py;
            }
        }
    }
    
    
    
public List<Boolean> areConnected(int n, int z, int[][] queries) {
        List<Boolean> res = new ArrayList<>();
        if (z < 1) {
            for (int i = 0; i < queries.length; i++) {
                res.add(true);
            }
            return res;
        }
        UnionFind uf = new UnionFind(n + 1);
        for (int i = z + 1; i <= n / 2; i++) {
            for (int j = i + i; j <= n; j += i) {
                uf.union(i, j);
            }
        }
        for (int i = 0; i < queries.length; i++) {
            int a = queries[i][0];
            int b = queries[i][1];
            
if (uf.find(a) == uf.find(b)) {
                res.add(true);
            } else {
                res.add(false);
            }
        }
        return res;
    }
    
    private int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
}
