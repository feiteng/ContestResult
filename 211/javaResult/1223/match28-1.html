class Solution {
    public List<Boolean> areConnected(int n, int t, int[][] qu) {
        
        
         List<Boolean> res = new ArrayList<>();
        int[] root = new int[n + 1];
        for(int i = 1; i <= n; i ++) {
            root[i] = i;
        }
        
       
          for (int i = t + 1; i <= n; i++){
            
for (int j = 2 * i; j <= n; j += i){         
                //if(i != j) {
                    //if(gcd(i, j, g)) {    //用gcd函数会超时
                        //System.out.println("i = " + i + ", j = " + j);
                        uf(root, i, j);
                    //}
                //}
            }
        }
        
        
        // for(int i = 1; i <= n; i ++) {
        //     //System.out.println("getRoot(root," + i + ") = " +  getRoot(root, i));
        //     //System.out.println("root[" + i + "] = " + root[i]);
        // }
        
        for(int i = 0; i < qu.length; i++) {
            if(getRoot(root, qu[i][0]) != getRoot(root, qu[i][1])) res.add(false);
            else res.add(true);
        
}
        

        
       // System.out.println(Arrays.toString(res));
        return res;
        
        
    }
    
    
    //用gcd函数会超时
//    static boolean gcd(int i, int j, int g) {
//        if(Math.min(i, j) <= g) return false;
//        for(int k = Math.min(i, j); k > g; k--) {
//            if(i % k == 0 && j %k == 0) return true;
//        }
//        return false;
//    }
    
    
    static void uf(int[] root, int p, int q) {
        while(root[p] != p) p = root[p];
        while(root[q] != q) q = root[q];
        if(root[p] != root[q]) root[q] = p
;
    }
    
    static int getRoot(int[] root, int i) {
        while(i != root[i]) {
            root[i] = root[root[i]];
            i = root[i];
        }
        return i;
    }
}
