class Solution {
    
    int[] parents, sizes;
    
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        parents = new int[n+1];
        sizes = new int[n+1];
        for (int i = 1; i <= n; i++) {
            parents[i] = i;
            sizes[i] = 1;
        }
        for (int i = 1; i <= n; i++) {
            TreeSet<Integer> set = getDivisors(i);
            for (int j : set.tailSet(threshold, false)) {
                union(i, j);
            }
        }
        List<Boolean> ans = new ArrayList<>();
        for (int[] q : queries) {
            int a = q[0], b = q[1];
            ans.add(find(a) == find(b));
        }
        return ans;
    }
    
    private int find(int x) {
        while (parents[x] != x) {
            parents[x] = parents[parents[x]];
            x = parents[x];
        }
        return x;
    }
    
    private void union(int p, int q) {
        int pRoot = find(p), qRoot = find(q);
        if (pRoot != qRoot) {
            if (sizes[pRoot] > sizes[qRoot]) {
                parents[qRoot] = pRoot;
                sizes[pRoot] += sizes[qRoot];
            } else {
                parents[pRoot] = qRoot;
                sizes[qRoot] += sizes[pRoot];
            }
        }
    }
    
    private TreeSet<Integer> getDivisors(int n) {
        TreeSet<Integer> set = new TreeSet<>();
        for (int i = 1; i*i <= n; i++) {
            if (n%i == 0) {
                set.add(i);
                set.add(n/i);
            }
        }
        return set;
    }
}
