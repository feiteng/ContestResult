class Solution {

    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        UnionFind uf = new UnionFind(n + 1);
        for (int i = threshold + 1; i <= n; i++) {
            if (uf.count() == 1) {
                break;
            }
            for (int j = 1; j < n / i; j++) {
                if (uf.count() == 1) {
                    break;
                }
                uf.union(i * j, i * (j + 1));
            
}
        }
        ArrayList<Boolean> list = new ArrayList<>();
        for (int[] query : queries) {
            list.add(uf.find(query[0]) == uf.find(query[1]));
        }
        return list;
    }

    class UnionFind {
        private int count = 0;
        private int[] parent, rank;

        public UnionFind(int n) {
            count = n;
            parent = new int[n];
            rank = new int[n];
            for (int i = 0; i < n; i++) {
                
parent[i] = i;
            }
        }

        public int find(int p) {
            while (p != parent[p]) {
                parent[p] = parent[parent[p]];
                p = parent[p];
            }
            return p;
        }

        public void union(int p, int q) {
            int rootP = find(p);
            int rootQ = find(q);
            if (rootP == rootQ)
                return;
            if (rank[rootQ] > rank[rootP]) {
                parent[rootP] = rootQ;
            } else {
                parent[rootQ] = rootP;
                if (rank[rootP] == rank[rootQ]) {
                    rank[rootP]++;
                }
            }
            count--;
        }

        public int count() {
            return count;
        }
    }
}
