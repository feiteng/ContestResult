class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        UF uf = new UF(n + 1);
        for (int i = threshold + 1; i <= n; i++) 
{
            for (int j = 2 * i; j <= n; j += i) {
                uf.connect(i, j);
            }
        }
        List<Boolean> out = new ArrayList<>();
        for (int[] q : queries) {
            out.add(uf.isConnected(q[0], q[1]));
        }
        return out;
    }

    
class UF {

        private int[] root;

        public UF(int n) {
            this.root = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
        }

        private int root(int i) {
            if (root[i] == i) {
                return i;
            } else {
                root[i] = root[root[i]];
                return root(root[i]);
            }
        }

        public void connect(int i, int j) {
            root[root(i)] = root(j);
        
}

        public boolean isConnected(int i, int j) {
            return root(i) == root(j);
        }
    }
}
