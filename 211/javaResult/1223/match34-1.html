class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        List<Boolean> res = new ArrayList<>(queries.length);
        
        if (threshold == 0) {
            for (int[] q : queries) {
                res.add(true);
            }
            return res;
        }

        Union u = new Union(n + 1);
        boolean[] used = new boolean[n + 1];
        
        for (int i = threshold + 1; i <= n; i++) {
            int v = i*2;
            while (v <= n) {
                u.union(i, v);
                v += i;
            }
        
}
        
        for (int[] q : queries) {
            res.add(u.getParent(q[0]) == u.getParent(q[1]));
        }
        
        return res;
    }
    
    static class Union {
        int[] parents;
        
        Union(int n) {
            parents = new int[n];
            for (int i = 0; i < n; i++) parents[i] = i
;
        }
        
        int getParent(int i) {
            if (parents[i] != i) parents[i] = getParent(parents[i]);
            return parents[i];
        }
        
        void union(int i1, int i2) {
            int p1 = getParent(i1), p2 = getParent(i2);
            if (p1 == p2) return;
            
            parents[p1] = p2;
        }
    }
}
