class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        DSU dset=new DSU(n+1);
        for(int i=threshold+1;i<=n;i++){
            for(int j=2*i;j<=n;j+=i){
                dset.union(i,j);
            }
        }
        List<Boolean> gg=new ArrayList<>();
        for(int x[]: queries){
            gg.add(dset.find(x[0])==dset.find(x[1]));
        }
        return gg;
    }
    
    class DSU {
        int n;
        int parent[];
        int size[];
        public DSU(int n){
            this.n=n;
            size=new int[n];
            parent=new int[n];
            for(int i=0;i<n;i++) {
                parent[i]=i;
                size[i]=1;
            }
        }
        int find(int x){
            return parent[x]= parent[x]==x?x:find(parent[x]);
        }
        void union(int x, int y){
            int p1=find(x);
            int p2=find(y);
            if(p1==p2) return;
            if(size[p1]<size[p2]){
                int temp=p1;
                p1=p2;
                p2=temp;
            }
            parent[p2]=p1;
            size[p1]+=size[p2];
        }
        
    }
    
}
