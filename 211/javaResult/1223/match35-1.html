class Solution {
    int gcd(int a,int b){
        return b==0?a:gcd(b,a%b);
    }
    
    
    static int groups = 0;
    static int[] fa;
    static int[] sz;

    static void init(int n) {
        groups = n;
        fa = new int[n];
        for (int i = 1; i < n; ++i) {
            fa[i] = i;
        }
        sz = new int[n];
        Arrays.fill(sz, 1);
    }

    static int root(int p) {
        while (p != fa[p]) {
            fa[p] = fa[fa[p]];
            p = fa[p];
        }
        
return p;
    }

    static void combine(int p, int q) {
        int i = root(p);
        int j = root(q);
        if (i == j) {
            return;
        }
        if (sz[i] < sz[j]) {
            fa[i] = j;
            sz[j] += sz[i];
        } else {
            
fa[j] = i;
            sz[i] += sz[j];
        }
        groups--;
    }
    public List<Boolean> areConnected(int n, int threshold, int[][] qs) {
        init(n);
        for(int i=threshold +1;i<=n;++i){
            int b = i;
            
for(int j = i; j <=n ;j+= i){
                combine(b-1,j-1);
            }
        }
        List<Boolean > res =new ArrayList<>();
        for(int q[]:qs){
            int x =q[0]-1;
            int y = q[1]-1;
            
if(root(x)==root(y)){
                res.add(true);
            }else{
                res.add(false);
            }
        }
        return res;
        
    }
}
