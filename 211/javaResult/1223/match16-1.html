class Solution {

    
public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
    	UF uf = new UF(n+1);
    	for(int i=threshold+1; i<=n/2; i++) {
    		for(int j=2; i*j <=n; j++) {
    			uf.union(i, i*j);
    		}
    	}

    	
    	List<Boolean> result = new ArrayList<>();
    	for(int[] q : queries) {
    		result.add(uf.connected(q[0], q[1]));
    	}
    	return result;
    }
    
    
    class UF{
    	
    	int n = 0;
    	int[] parent = null;
    	int[] size = null;
    	
    	public UF(int n) {
    		this.n = n;
    		parent = n
ew int[n];
    		size = new int[n];
    		for(int i=0; i<n; i++) {
    			parent[i] = i;
    			size[i] = 1;
    		}
    	}
    	
    	
    	private int parent(int i) {
    		if( parent[i] == i ) {
    			return i;
    		}
    		int p = parent( parent[i] );
    		parent[i] = p;
    		return p;
    	}
    	
    	private boolean connected(int i, int j) {
    		return parent(i) == parent(j);

    	}
    	
    	public void union(int i, int j) {
    		int pi = parent(i);
    		int pj = parent(j);
    		if( pi == pj ) {
    			return;
    		}

    		if( size[pi] >= size[pj] ) {
    			parent[pj] = pi;
    			size[pi] += size[pj];
    		}else {
    			parent[pi] = pj;
    			size[pj] += size[pi];
    		}
    	}
    	
    }
}
