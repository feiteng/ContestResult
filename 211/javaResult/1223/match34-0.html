class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        List<Boolean> res = new ArrayList<>();
        if (threshold == 0) {
            for (int i = 0; i < queries.length; i++) {
                res.add(true);
            }
            return res;
        }
        SetUnion su = new SetUnion(n);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j < (int)(Math.sqrt(i))+1; j++) {
                if (i % j == 0) {
                    if (j > threshold) {
                        su.union(i, j);
                    }
                    if (i / j > threshold) {
                        su.union(i, i / j);
                    }
                }
            }
        }
        
        for (int[] q : queries) {
            res.add(su.find(q[0]) == su.find(q[1]));
        }
        return res;
    }
    
    class SetUnion {
        private int[] parent;
        private int[] size;

        public SetUnion(int size) {
            this.parent = new int[size + 1];
            this.size = new int[size + 1];
            for (int i = 0; i < size + 1; ++i) {
                this.parent[i] = i;
                this.size[i] = 1;
            }
        }

        public int find(int x) {
            if (this.parent[x] != x)
                this.parent[x] = this.find(this.parent[x]);
            return this.parent[x];
        }

        public int union(int x, int y) {
            int px = this.find(x);
            int py = this.find(y);

            if (px == py)
                return px;

            if (this.size[px] > this.size[py]) {
                int temp = px;
                px = py;
                py = temp;
            }

            this.parent[px] = py;
            this.size[py] += this.size[px];
            return py;
        }
    }
    
}
