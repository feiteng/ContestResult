class Solution {
    static class UF {
        int[] root;
        int[] size;
        UF (int n) {
            root = new int[n];
            size = new int[n];
            for (int i = 0; i < n; i++) {
                root[i] = i;
            }
            Arrays.fill(size, 1);
        }
        
        int find(int node) {
            if (root[node] != node) {
                root[node] = find(root[node]);
            }
            return root[node];
        }
        
        boolean union(int x, int y) {
            int rootX = find(x);
            int rootY = find(y);
            if (rootX == rootY) {
                return false;
            }
            if (size[rootX] > size[rootY]) {
                root[rootY] = rootX;
                size[rootX] += size[rootY];
            } else {
                root[rootX] = rootY;
                size[rootY] += size[rootX];
            }
            return true;
        }
        
        
    }
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        UF uf = new UF(n + 1);
        for (int root = threshold + 1; root <= n; root++) {
            int times = 2;
            while (root * times <= n) {
                uf.union(root, times * root);
                times++;
            }
        }
        
        List<Boolean> ret = new ArrayList<>();
        for (int[] q: queries) {
            ret.add(uf.find(q[0]) == uf.find(q[1]));
        }
        
        return ret;
        
    }
}
