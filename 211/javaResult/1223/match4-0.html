class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        int[] roots = new int[n+1];
        for (int i = 1; i <= n; i++) roots[i] = i;

        for (int i = threshold + 1; i <= n/2; i++) {
            for (int j = 2; i * j <= n; j++) {
                for (int k = j - 1; k > 0; k--) {
                    int n1 = i * j;
                    int n2 = i * k;
                    connect(roots, n1, n2);
                }
            }
        }

        List<Boolean> result = new ArrayList();
        for (int[] query : queries) {
            if (isConnect(roots, query[0], query[1])) {
                result.add(true);
            } else {
                result.add(false);
            }
        }
        
        return result;
    }
    
    private int parent(int[] roots, int n) {
        while (roots[n] != n) {
            n = roots[n];
            roots[n] = roots[roots[n]];
        }
        return n;
    }
    
    private void connect(int[] roots, int n1, int n2) {
        int p1 = parent(roots, n1);
        int p2 = parent(roots, n2);
        roots[p1] = p2;
    }
    
    private boolean isConnect(int[] roots, int n1, int n2) {
        return parent(roots, n1) == parent(roots, n2);
    }
}
