class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        int[] uf =new int[n+1];
        for(int i=1;i<=n;i++){
            uf[i] = i;
        
}
        for(int i=threshold+1;i<=n/2;i++){
            for(int j=1;j*i<=n;j++){
                for(int k=j+1;k*i<=n;k++){
                    
union(uf,i*j,i*k);
                }
            }
        }
        List<Boolean> ans = new LinkedList<>();
        for(int[] q: queries){
            ans.add(root(uf,q[0])==root(uf,q[1]));
        
}
        return ans;
    }
    
    private int root(int[] list, int i){
        while(i!=list[i]){
            list[i] = list[list[i]];
            i = list[i];
        }
        return i;
    }
    
    public void union(int[] A, int p, int q){
        int i = root(A,p);
        int j = root(A,q);
        A[i] = j;
    }
}
