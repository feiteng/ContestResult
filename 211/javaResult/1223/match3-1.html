class Solution {
    
    class DisjointSet {
        int[] p;
        int[] s;
        int n;
        public DisjointSet(int n) {
            this.n = n;
            p = new int[n];
            s = new int[n];
            for (int i = 0; i < n; i++) {
                p[i] = i;
                s[i] = 1;
            }
        }
        
        public int parent(int x) {
            if (x == p[x]) return x;
            return p[x] = parent(p[x]);
        
}
        
        public void union(int x, int y) {
            int px = parent(x);
            int py = parent(y);
            if (px == py) return;
            if (s[px] < s[py]) {
                p[px] = py;
                s[py] += s[px];
            } else {
                p[py] = px;
                s[px] += s[py];
            }
        
}
    }
    public List<Boolean> areConnected(int n, int t, int[][] queries) {
        
        DisjointSet ds = new DisjointSet(n + 1);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= Math.sqrt(i) + 1; j++) {
                if (i % j == 0) {
                    if (j > t) ds.union(i, j);
                    if (i / j > t) ds.union(i, i / j);
                }
            
}
        }
        
        List<Boolean> ans = new ArrayList<>();
        for (int[] q : queries) {
            ans.add(ds.parent(q[0]) == ds.parent(q[1]));
        }
        return ans;
    }
}
