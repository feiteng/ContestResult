class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        
        List<Boolean> ans = new ArrayList<>();
        if(threshold<1){
            for(int[] q:queries){
                ans.add(true);
            }
            return ans;
        }
        
        
        UnionFind uf = new UnionFind(n);
        
        for(int i=threshold+1;i<=n;i++){
            for(int j=2;j*i<=n;j++){
                uf.union(i,j*i);
            }
            
        }
        // for(int i=threshold;i<n;i*threshold){
        //     for(int j=i+1;j<=n;j++){
        //         if(gcd(i,j)>threshold){
        //             uf.union(i,j);
        //         }
        //     }
        // }
        
        for(int[] q:queries){
            if(uf.find(q[0])==uf.find(q[1])) {
                ans.add(true);
            } else{
                ans.add(false);
            }
        }
        return ans;
    }
    
    int gcd(int a, int b)
    {
      if (b == 0)
        return a;
      return gcd(b, a % b); 
    }
}
class UnionFind{
    int[] rank;
    
    public UnionFind(int n){
        rank=new int[n+1];
        for(int i=1;i<=n;i++){
            rank[i]=i;
        }
    }
    
    boolean union(int x, int y){
        x=find(x);
        y=find(y);
        if(x==y) return false;
        rank[y]=x;
        return true;
    }
    int find(int x){
        if(x!=rank[x]){
            rank[x]=rank[rank[x]];
            x=find(rank[x]);
        }
        return x;
    }
}
