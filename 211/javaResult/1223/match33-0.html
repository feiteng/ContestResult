class Solution {
    int [] parent;
    
    public int find(int a) {
        if (parent[a] != a) {
            parent[a] = find(parent[a]);
        }
        return parent[a];
    }
    
    public void union(int a, int b) {
        int pa = find(a);
        int pb = find(b);
        
        if (pa != pb) {
            parent[pa] = pb;
        }
    }
    
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        parent = new int[n+1];
        for (int i = 0; i <= n; i++) parent[i] = i;
        
        for (int mul = threshold + 1; mul <= n / 2; mul++) {
            for (int j = mul + mul; j <= n; j += mul) {
                //System.out.println("union..." + mul + ", " + j);
                
                union(mul, j);
                //System.out.println("parent arr  " + Arrays.toString(parent));
            }
        }
        
        List<Boolean> ans = new ArrayList<>();
        for (int[] query : queries) {
            if (find(query[0]) == find(query[1])) {
                ans.add(true);
            } else {
                ans.add(false);
            }
        }
        return ans;
    }
    
}
