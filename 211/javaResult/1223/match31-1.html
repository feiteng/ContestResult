class Solution {
    int[] root;
    
public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        root = new int[n + 1];
        for (int i = 0; i <= n; i++) {
            root[i] = i;
        }
        List<Integer>[] lists = new List[n + 1];
        for (int i = 0; i <= n; i++) {
            lists[i] = new ArrayList<>();
        }
        for (int t = threshold + 1; t <= n; t++) {
            for (int i = t; i <= n; i++) {
                if (i % t == 0) {
                    lists[t].add(i);
                }
            }
        }
        for (int i = threshold + 1; i <= n; i++) {
            int size = lists[i].size();
            if (size < 2) {
                continue;
            }
            for (int j = 0; j < size - 1; j++) {
                int x = lists[i].get(j);
                int y = lists[i].get(j + 1);
                int rx = find(root, x);
                
int ry = find(root, y);
                if (rx != ry) {
                    root[rx] = ry;
                }
            }
        }
        List<Boolean> result = new ArrayList<>();
        for (int[] q : queries) {
            int x = q[0];
            int y = q[1];
            int rx = 
find(root, x);
            int ry = find(root, y);
            result.add(rx == ry);
        }
        return result;
    }
    private int find(int[] root, 
int x) {
        if (root[x] == x) {
            return x;
        }
        return root[x] = find(root, root[x]);
    }
}
