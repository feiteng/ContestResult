class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        DSU dsu = new DSU(n);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= Math.sqrt(i); j++) {
                if (i % j == 0) {
                    if (j > threshold) dsu.union(i-1, j-1);
                    if (i / j  > threshold) dsu.union(i-1, i/j-1);
                }
            }
        }
        List<Boolean> result = new ArrayList<>();
        for (int[] query: queries) {
            if (dsu.find(query[0]-1) == dsu.find(query[1]-1)) result.add(true);
            else result.add(false);
        }
        return result;
    }
}

class DSU {
    int N;
    int[] parent;
    int[] capacity;
    
    public DSU(int n) {
        this.N = n;
        this.parent = new int[n];
        this.capacity = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            capacity[i] = 1;
        }
    }
    
    public int find(int a) {
        int root = parent[a];
        while (root != parent[root]) {
            root = parent[root];
        }
        while (a != parent[a]) {
            int temp = parent[a];
            parent[a] = root;
            a = temp;
        }
        return root;
    }
    
    public void union(int a, int b) {
        int rootA = find(a);
        int rootB = find(b);
        if (rootA == rootB) return;
        if (capacity[rootA] >= capacity[rootB]) {
            capacity[rootA] += capacity[rootB];
            parent[rootB] = rootA;
        } else {
            capacity[rootB] += capacity[rootA];
            parent[rootA] = rootB;
        }
    }
}
