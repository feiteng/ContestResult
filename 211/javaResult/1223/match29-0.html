class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        UnionFind uf = new UnionFind(n);
        for (int gcd = Math.max(1, threshold + 1); gcd <= n; gcd++) {
            for (int m = 2; m * gcd <= n; m++) {
                uf.union(gcd, gcd * m);
            }
        }
        /*
        for (int i = 1; i <= n; i++) {
            for (int j = i + 1; j <= n; j++) {
                if (gcd(i, j) > threshold) {
                    uf.union(i, j);
                }
            }
        }*/
        List<Boolean> res = new ArrayList<>();
        for (int[] q : queries) {
            res.add(uf.isConnected(q[0], q[1]));
        }
        return res;
    }
    private int gcd(int a, int b) {
        while (a != b) {
            if (a > b) {
                a -= b;
            } else {
                b -= a;
            }
        }
        return a;
    }
    private static class UnionFind {
        private int[] parent;
        
        public UnionFind(int n) {
            parent = new int[n + 1];
            for (int i = 1; i <= n; i++) {
                parent[i] = i;
            }
        }
        
        public void union(int i1, int i2) {
            int rt1 = findRt(i1);
            int rt2 = findRt(i2);
            parent[rt2] = rt1;
        }
        
        public boolean isConnected(int i1, int i2) {
            return findRt(i1) == findRt(i2);
        }
        
        private int findRt(int i) {
            while (parent[i] != i) {
                int par = parent[i];
                parent[i] = parent[par];
                i = par;
            }
            return i;
        }
    }
}
