class Solution {
    class UF {
        int[] parent;
        UF(int n) {
            parent = new int[n];
            for (int i = 0; i < n; i += 1) {
                parent[i] = i;
            }
        }
        int find(int x) {
            int p = x;
            while (parent[p] != p) {
                p = parent[p];
            }
            while (x != p) {
                int temp = parent[x];
                parent[x] = p;
                x = temp;
            }
            return p;
        }
        void union(int x, int y) {
            int xx = find(x);
            int yy = find(y);
            if (xx != yy) {
                parent[xx] = yy;
            }
        }
        boolean isConnect(int x, int y) {
            return find(x) == find(y);
        }
    }
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        List<Boolean> ret = new ArrayList<>();
        if (threshold == 0) {
            for (int i = 0; i < queries.length; i += 1) {
                ret.add(true);
            }
            return ret;
        }
        UF uf = new UF(n);
        for (int i = threshold; i <= n; i += 1) {
            for (int j = i * 2; j <= n; j += 1) {
                if (gcd(i, j) > threshold) {
                    uf.union(i - 1, j - 1);
                }
            }
        }
        
        for (int[] query : queries) {
            if (uf.isConnect(query[0] -1, query[1] - 1)) {
                ret.add(true);
            }else {
                ret.add(false);
            }
        }
        return ret;
    }
    int gcd(int a, int b) {
        if (a > b) {
            return gcd(b, a);
        }
        if (a == 0) return b;
        return gcd(b % a, a);
    }
}
