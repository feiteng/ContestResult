class Solution {
    int[] components;
    int[] size;
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        components=new int[n+1];
        size=new int[n+1];
        for(int i=1;i<=n;i++)
        {    
            components[i]=i;
            size[i]=1;
        }
        for(int i=Math.max(0,threshold)+1;i<=n;i++){
            int cur=i;
            for(int j=2;j*i<=n;j++)
{
                union(i,j*i);
            }
        }
        List<Boolean> res=new ArrayList<>();
        for(int[] query:queries){
            int pa=findParent(query[0]);
            int pb=findParent(query[1]);
            if(pa==pb) res.add(true);
            else res.add(false);
        
}
        return res;
    }
    int findParent(int n){
        while(components[n]!=n){
            
n=components[components[n]];
        }
        return n;
    }
    void union(int a,int b){
        int pa=findParent(a);
        int pb=findParent(b);
        if(pa==pb) return;
        
if(size[pa]>size[pb]){
            size[pa]+=size[pb];
            components[pb]=pa;
        } else{
            size[pb]+=size[pa];
            components[pa]=pb;
        }
    }
}
