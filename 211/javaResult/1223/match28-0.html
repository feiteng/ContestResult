class Solution {
    public List<Boolean> areConnected(int n, int threshold, int[][] queries) {
        List<Boolean> res = new ArrayList<>();
        
        int[] g = new int[n+1];
        
        for (int i = 1; i <= n; i++) {
            g[i] = i;
            Set<Integer> set = divisor(i, threshold);
            for (int s : set) {
               // System.out.println(i + "," + s);
                connect(g, s, i);
            }
        }
        
        for (int[] q : queries) {
            res.add(find(g, q[0]) == find(g, q[1]));
        }
        
        
        return res;
    }
    private void connect(int[] g, int a, int b) {
        if (find(g, a) != find(g, b)) {
            g[find(g, b)] = find(g, a);
        }
    }
    private int find(int[] g, int a) {
        while (g[a] != a) {
            g[a] = g[g[a]];
            a = g[a];
        }
        return a;
    }

    
    private Set<Integer> divisor(int a, int t) {
        Set<Integer> res = new HashSet<>();
        int m = a;
        for (int i = Math.max(1, t+1); i <= m; i++) {
            if (a % i == 0) res.add(i);
        }
        return res;
        
    }
}
