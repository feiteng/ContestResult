class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        TreeSet<String> set = new TreeSet<>();
        Queue<String> queue = new LinkedList<>();
        set.add(s); queue.add(s);
        while (!queue.isEmpty()) {
            String poll = queue.poll();
            String moveStr = move(poll, b), addStr = add(poll, a);
            if (!set.contains(moveStr)) {
                set.add(moveStr);
                queue.add(moveStr);
            }
            if (!set.contains(addStr)) {
                set.add(addStr);
                queue.add(addStr);
            }
        }
        return set.first();
    }

    private String move(String s, int b) {
        return s.substring(b) + s.substring(0, b);
    }

    private String add(String s, int a) {
        StringBuilder res = new StringBuilder();
        for (int i = 0; i < s.length(); ++i) {
            if (i % 2 == 0) {
                res.append(s.charAt(i));
            } else {
                int tmp = (s.charAt(i) - '0' + a) % 10;
                res.append(tmp);
            }
        }
        return res.toString();
    }
}
