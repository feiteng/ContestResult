class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Set<String> seen = new HashSet<>();
        Queue<String> queue = new LinkedList<>();
        queue.offer(s);
        String res = s;
        while (!queue.isEmpty()) {
            String tmp = queue.poll();
            if (tmp.compareTo(res) < 0) res = tmp;
            String s1 = add(tmp, a), s2 = rotate(tmp, b);
            if (seen.add(s1)) {
                queue.offer(s1);
            }
            
if (seen.add(s2)) {
                queue.offer(s2);
            }
        }
        return res;
    }
    public String add(String s, int a) {
        int n = s.length();
        char[] array = s.toCharArray();
        for (int i = 1; i < n; i += 2) {
            array[i] = (char) ((((s.charAt(i) - '0') + a) % 10) + '0');
        
}
        return new String(array);
    }
    public String rotate(String s, int b) {
        return s.substring(b) + s.substring(0, b);
    }
}
