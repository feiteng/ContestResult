class Solution {
        public String findLexSmallestString(String s, int a, int b) {
            TreeSet<String> existed = new TreeSet<>();
            Queue<String> queue = new LinkedList<>();
            queue.add(s);
            existed.add(s);
            while (queue.isEmpty() == false) {
                String cur = queue.poll();
                String next;
                next = nextA(cur, a);
                if (existed.contains(next) == false) {
                    queue.add(next);
                    existed.add(next);
                }

                next = nextB(cur, b);
                
if (existed.contains(next) == false) {
                    queue.add(next);
                    existed.add(next);
                }
            }

            
return existed.first();
        }

        private String nextA(String s, int a) {
            StringBuilder builder = new StringBuilder();
            char[] cs = s.toCharArray();
            int len = s.length();
            for (int i=0;i<len;i++) {
                char n = cs[i];
                if (i % 2 == 1) {
                    n = (char) (((n-'0'+a) % 10) + '0');
                }
                
builder.append(n);
            }
            return builder.toString();
        }

        private String nextB(String s, int b) {
            StringBuilder builder = new StringBuilder();
            char[] cs = s.toCharArray();
            int len = s.length();
            for (int i=len-b;i<len;i++) {
                builder.append(cs[i]);
            }
            for (int i=0;i<len-b;i++) {
                builder.append(cs[i]);
            }
            return builder.toString();
        }
    }
