class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Set<String> set = new HashSet<>();
        Queue<String> queue = new LinkedList<>();
        queue.add(s);
        set.add(s);
        while(!queue.isEmpty()) {
            String curr = queue.remove();
            String next1 = change1(curr, a);
            String next2 = change2(curr, b);
            if (!set.contains(next1)) {
                set.add(next1);
                queue.add(next1);
            }
            if (!set.contains(next2)) {
                set.add(next2);
                queue.add(next2);
            }
        }
        
        List<String> list = new ArrayList<>(set);
        Collections.sort(list);
        return list.get(0);
    }
    
    private String change1(String curr, int a) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < curr.length(); i++) {
            int num = curr.charAt(i) - '0';
            if (i % 2 == 1) {
                num = (num + a)%10;
            }
            sb.append(String.valueOf(num));
        }
        return sb.toString();
    }
    
    private String change2(String curr, int b) {
        StringBuilder sb = new StringBuilder();
        String s1 = curr.substring(0, b);
        String s2 = curr.substring(b);
        sb.append(s2);
        sb.append(s1);
        return sb.toString();
    }
}
