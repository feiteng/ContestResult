class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Queue<String> q = new ArrayDeque<>();
        Set<String> visited = new HashSet<>();
        String res = s;
        q.add(s);
        visited.add(s);
        while (!q.isEmpty()) {
            String str = q.poll();
            if (str.compareTo(res) < 0) {
                res = str;
            }
            String x = op1(str, a);
            String y = op2(str, b);
            if (!visited.contains(x)) {
                q.add(x);
                visited.add(x);
            }
            if (!visited.contains(y)) {
                q.add(y);
                visited.add(y);
            
}
        }
        return res;
    }
    
    private String op1(String s, int a) {
        //System.out.println(s);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) sb.append(s.charAt(i));
            else {
                int val = s.charAt(i) - '0' + a;
                
sb.append(val % 10);
            }
        }
        //System.out.println(sb);
        return sb.toString();
    }
    
    private String op2(String s, int b) {
        b %= s.length();
        int v = s.length() - b;
        return s.substring(v) + s.substring(0, v);
    }
}
