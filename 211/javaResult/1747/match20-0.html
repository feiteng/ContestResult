class Solution {
     public String findLexSmallestString(String s, int a, int b) {
        HashSet<String> set = new HashSet<>();
        set.add(s);
        String smallest = s;
        LinkedList<String> q = new LinkedList<>();
        q.add(s);
        while(!q.isEmpty()) {
            String top = q.poll();
            if (top.compareTo(smallest) < 0) {
                smallest = top;
            }
            String str1 = do1(top, a);
            if (!set.contains(str1)) {
                set.add(str1);
                q.add(str1);
            }
            String str2 = do2(top, b);
            if (!set.contains(str2)) {
                set.add(str2);
                q.add(str2);
            }
        }
        return smallest;
    }

    private String do1(String from, int a) {
        char[] arr = from.toCharArray();
        for (int i = 1; i < arr.length; i += 2) {
            int num = (arr[i] - '0' + a) % 10;
            arr[i] = (char)('0' + num);
        }
        return new String(arr);
    }
    
    private String do2(String from, int b) {
        return from.substring(from.length() - b) + from.substring(0, from.length() - b);
    }
}
