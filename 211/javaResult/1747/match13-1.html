class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        // BFS
        TreeSet<String> set = new TreeSet<String>();
        set.add(s);
        Queue<String> queue = new LinkedList();
        
queue.add(s);
        while (!queue.isEmpty()) {
            String curr = queue.poll();
            String A = getA(curr, a);
            String B = getB(curr, b);
            //System.out.println(A +" " + B);
            if (!set.contains(A)) {
                queue.add(A);
                set.add(A);
            }
            if (!set.contains(B)) {
                queue.add(B);
                set.add(B);
            }
        }
        
return set.first();
    }
    
    private String getA(String s, int a) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (i % 2 == 1) {
                c = (char) (((c - '0' + a) % 10) + '0'); 
            }
            sb.append(c);
        
}
        return sb.toString();
    }
    
    private String getB(String s, int b) {
        StringBuilder sb = new StringBuilder();
        for (int i = b; i < s.length(); i++) {
            sb.append(s.charAt(i));
        }
        for (int i = 0; i < b; i++) {
            sb.append(s.charAt(i));
        }
        return sb.toString();
    } 
}
