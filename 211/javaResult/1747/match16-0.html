class Solution {
    
    String res;
    public String findLexSmallestString(String s, int a, int b) {
        if (s.length() == 0) {
            return "";
        }
        for (int i = 0; i < s.length(); i++) {
            res += "9";
        }
        dfs(s, new HashSet<String>(), a, b);
        return res;
    }
    
    private void dfs(String s, Set<String> seen, int a, int b) {
        if (seen.contains(s)) {
            return;
        }
        if (res.compareTo(s) > 0) {
            res = s;
        }
        seen.add(s);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 1) {
                sb.append((s.charAt(i) - '0' + a) % 10);
            } else {
                sb.append(s.charAt(i));
            }
        }
        dfs(sb.toString(), seen, a, b);
        dfs(s.substring(s.length()-b) + s.substring(0, s.length()-b), seen, a, b);
    }
}
