class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Queue<String> q = new LinkedList<>(); 
        q.add(s); 
        Set<String> set = new HashSet<>(); 
        int len = s.length();
        String ret = s; 
        while (!q.isEmpty()) {
            String str = q.poll(); 
            // System.out.println(str);
            if (str.compareTo(ret) < 0) {
                ret = str;
            }
            String pre = str.substring(0, len - b);
            String suf = str.substring(len-b);
            String tmp = suf + pre;
            // System.out.println(str + " " + tmp);
            if (!set.contains(tmp)) {
                set.add(tmp);
                q.add(tmp);
            }
            char[] cs = str.toCharArray(); 
            for (int i = 1; i < len; i += 2) {
                int val = cs[i] - '0'; 
                val += a; 
                if (val >= 10) {
                    val -= 10; 
                }
                // System.out.println(str + " " + i + " " + val);
                cs[i] = (char)(val + '0');
            }
            String x = new String(cs); 
            // System.out.println(str + " " + x);
            if (!set.contains(x)) {
                set.add(x);
                q.add(x); 
            }
        }
        return ret;
    }
}
