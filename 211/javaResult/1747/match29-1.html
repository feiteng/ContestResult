class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Queue<String> q = new LinkedList<>();
        Set<String> set = new HashSet<>();
        set.add(s);
        q.offer(s);
        String result = s;
        while (q.size() > 0) {
            String cur = q.poll();
            // System.out.println(cur);
            if (result.compareTo(cur) > 0) {
                result = cur;
            }
            String afterAdd = myAdd(cur, a);
            String afterMove = myMove(cur, b);
            if (set.add(afterAdd)) {
                q.offer(afterAdd);
            }
            
if (set.add(afterMove)) {
                q.offer(afterMove);
            }
        }
        return result;
    }
    
    public String myAdd(String s, int a) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++) {
            if (i % 2 == 0) {
                sb.append(s.charAt(i));
                continue;
            }
            int cur = (int) (s.charAt(i) - '0');
            cur = cur + a;
            cur = cur % 10;
            sb.append(cur);
        
}
        return new String(sb);
    }
    
    public String myMove(String s, int b) {
        char[] arr = s.toCharArray();
        rotate(arr, b);
        return new String(arr);
    }
    
    public void rotate(char[] nums, int k) {
        k = k % nums.length;
        if (k == 0) {
            return;
        }
        int position = nums.length - k - 1;
        reverse(nums, 0, position);
        reverse(nums, position + 1, nums.length - 1);
        reverse(nums, 0, nums.length - 1);
    }
    public void reverse(char[] nums, int left, int right) {
        while (left <= right) {
            char temp = nums[left];
            nums[left] = nums[right];
            nums[right] = temp;
            left++;
            right--;
        }
    }
}
