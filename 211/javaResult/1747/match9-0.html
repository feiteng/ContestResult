class Solution {
    public String findLexSmallestString(String s, int a, int b) {
    	Set<String> seen = new HashSet<>();
    	Queue<String> q = new LinkedList<>();
    	q.add(s);
    	seen.add(s);
    	String min = s;
    	while( !q.isEmpty() ) {
    		String current = q.remove();
    		String addedString = add(current, a);
    		//System.out.println(seen);
    		if( !seen.contains(addedString) ) {
    			q.add(addedString);
    			//System.out.println("Adding" + addedString);
    			seen.add(addedString);
    			if( addedString.compareTo(min) < 0 ) {
    				min = addedString;
    			}
    		}
    		//System.out.println(String.format("min %s", min));
    		String shiftedString = shift(current, b);
    		//System.out.println(String.format("Shifted %s min %s %d", shiftedString, min, shiftedString.compareTo(min)));
    		if( !seen.contains(shiftedString) ) {
    			q.add(shiftedString);
    			//System.out.println("Adding" + shiftedString);
    			seen.add(shiftedString);
    			if( shiftedString.compareTo(min) < 0 ) {
    				min = shiftedString;
    			}
    		}
    		//System.out.println(String.format("min %s", min));
    	}
    	return min;
    }
    
    
    private String add(String s, int a) {
    	StringBuilder sb = new StringBuilder("");
    	int l = s.length();
    	for(int i=0; i<l; ) {
    		sb.append(s.charAt(i++));
    		sb.append( (char)('0' + (s.charAt(i++) - '0' + a) % 10));
    	}
    	//System.out.println(String.format("Add %s -> %s", s, sb.toString()));
    	return sb.toString();
    }
   
    private String shift(String s, int b) {
    	StringBuilder sb = new StringBuilder("");
    	int l = s.length();
    	for(int i=0; i<l; i++) {
    		sb.append(s.charAt((i+b)%l));
    	}
    	
    	//System.out.println(String.format("Shift %s -> %s", s, sb.toString()));
    	return sb.toString();
    }
}
