class Solution {
    String ret = "";
    public String findLexSmallestString(String s, int a, int b) {
        StringBuilder sb = new StringBuilder();
        int l = s.length();
        for(int i = 0; i < l; i++)
            sb.append("9");
        ret = sb.toString();
        helper(s, a, b, new HashSet<String>());
        return ret;
    }
    
    private void helper(String s, int a, int b, Set<String> set) {
        if(set.contains(s)) return;
        if(s.compareTo(ret) < 0)
            ret = s;
        set.add(s);
        helper(add(s, a), a, b, set);
        helper(rotate(s, b), a, b, set);
    }
    
    private String rotate(String s, int b) {
        StringBuilder sb = new StringBuilder();
        int st = s.length() - b;
        for(int i = st; i < s.length(); i++) sb.append(s.charAt(i));
        for(int i = 0; i < st; i++) sb.append(s.charAt(i));
        return sb.toString();
    }
    
    private String add(String s, int a) {
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < s.length(); i++) {
            if(i % 2 == 0)
                sb.append(s.charAt(i));
            else {
                int d = s.charAt(i) - '0';
                d += a;
                d %= 10;
                sb.append((char)(d + '0'));
            }
        }
        return sb.toString();
    }
}
