import java.util.ArrayDeque;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Queue;


class Solution {
  public String findLexSmallestString(String s, int a, int b) {

    Queue<String> q= new ArrayDeque<>();
    q.add(s);
    HashSet<String> visited= new HashSet<>();
    visited.add(s);
    String ans= s;



    
while (!q.isEmpty())
    {

      String pop=q.remove();

      if (ans.compareTo(pop)>0)
      {
        ans=pop;
      }

      String rotate=rightrotate(pop,b);
      String added =add(pop,a);

      if (!visited.contains(rotate))
      {
        //System.out.println("adding "+rotate);
        visited.add(rotate);
        q.add(rotate);
      }

      
if (!visited.contains(added))
      {
        //System.out.println("adding "+added);

        visited.add(added);
        q.add(added);
      }
    }

    return ans;
  }

  String leftrotate(String str, int d)
  {
    String ans = str.substring(d) + str.substring(0, d);
    return ans;
  
}

  // function that rotates s towards right by d
  String rightrotate(String str, int d)
  {
    return leftrotate(str, str.length() - d);
  }

  String add(String str, int a)
  {
    StringBuilder sb = new StringBuilder(str);
    for (int i=1;i<sb.length();i=i+2)
    {
      sb.setCharAt(i, (char) ('0'+(((sb.charAt(i)-'0')+a)%10)));
    }
    return sb.toString();
  }


}
