class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Set<String> seen = new HashSet<>();
        Queue<String> bfs_queue = new LinkedList<>();
        
        String ans = s;
        seen.add(s);
        bfs_queue.add(s);
        
        while(!bfs_queue.isEmpty()){
            String cur = bfs_queue.poll();
            
            for(int i =0; i< 9; i++){
                String next = add(cur, a);
                if(seen.contains(next))
                    continue;
                seen.add(next);
                bfs_queue.add(next);
                if(next.compareTo(ans) < 0)
                    ans = next;
            }
            
            String rotate = rotate(cur, b);
            if(seen.contains(rotate))
                continue;
            seen.add(rotate);
            bfs_queue.add(rotate);
            if(rotate.compareTo(ans) < 0)
                ans = rotate;
        }
        
        return ans;
    }
    
    private String add(String s, int a){
        StringBuilder sb = new StringBuilder();
        for(int i=0; i< s.length(); i++){
            if(i % 2 == 0)
                sb.append(s.charAt(i));
            else{
                int digit = s.charAt(i) - '0';
                int new_digit = (digit + a)%10;
                sb.append((char)(new_digit + '0'));
            }
        }
        
        return sb.toString();
    }
    
    private String rotate(String s, int b){
        return s.substring(b) + s.substring(0, b);
    }
}
