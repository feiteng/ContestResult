class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        HashSet<String> set = new HashSet<>();
        Queue<String> q = new LinkedList<>();
        q.add(s);
        set.add(s);
        
        while(!q.isEmpty()){
            String cur = q.poll();
            
String first = first(cur, a);
            String second = second(cur, b);
            if(set.add(first)){
                q.add(first);
            }
            
            if(set.add(second)){
                q.add(second);
            }
        }
        
        // System.out.println(set);
        
        String res = s;
        for(String cur : set){
            if(res.compareTo(cur) > 0){
                res = cur;
            }
        }
        
        
return res;
    }
    
    public String first(String s, int a){
        char[] res = s.toCharArray();
        for(int i = 1; i < s.length(); i += 2){
            int cur = s.charAt(i) - '0' + a;
            res[i] = (char) ((cur % 10) + '0');
        
}
        
        return new String(res);
    }
    
    public String second(String s, int b){
        char[] res = new char[s.length()];
        for(int i = 0; i < s.length(); i++){
            res[(i + b) % s.length()] = s.charAt(i);
        }
        
        return new String(res);
    }
}
