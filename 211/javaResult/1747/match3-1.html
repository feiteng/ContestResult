class Solution {

    public String findLexSmallestString(String s, int a, int b) {
        TreeSet<String> set = new TreeSet<>();
        ArrayDeque<String> queue = new ArrayDeque<>(Set.of(s));
        while (!queue.isEmpty()) {
            String curr = queue.remove();
            if (!set.contains(curr)) {
                set.add(curr);
                String next = "", next2 = "";
                for (int i = 0; i < curr.length(); i++) {
                    if (i % 2 == 1) {
                        next += (char) ((curr.charAt(i) - '0' + a) % 10 + '0');
                    } else {
                        next += curr.charAt(i);
                    }
                    next2 += 
curr.charAt((i + b) % curr.length());
                }
                queue.add(next);
                queue.add(next2);
            }
        }
        return set.first();
    }
}
