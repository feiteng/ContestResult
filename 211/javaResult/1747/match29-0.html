class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Queue<String> queue = new ArrayDeque<>();
        Set<String> visited = new HashSet<>();
        queue.offer(s);
        visited.add(s);
        
        String rst = s;
        while (!queue.isEmpty()) {
            String curr = queue.poll();
            if (curr.compareTo(rst) < 0) {
                rst = curr;
            }
            
            String opA = performA(curr, a);
            if (visited.add(opA)) {
                queue.offer(opA);
            }
            
            String opB = performB(curr, b);
            if (visited.add(opB)) {
                queue.offer(opB);
            }
        }
        
        return rst;
    }
    
    private String performA(String s, int a) {
        int n = s.length();
        char[] chars = s.toCharArray();
        for (int i = 0; i < n; i++) {
            if (i % 2 == 1) {
                chars[i] = (char)((chars[i] - '0' + a) % 10 + '0');
            }
        }
        
        return new String(chars);
    }
    
    private String performB(String s, int b) {
        int n = s.length();
        return s.substring(n - b) + s.substring(0, n - b);
    }
}
