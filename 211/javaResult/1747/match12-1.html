class Solution {
    String ans = null;
    public String findLexSmallestString(String s, int a, int b) {
        Set<String> visit = new HashSet<>();
        helper(s, a, b, visit);
        
return ans;
    }
    
    private void helper(String s, int a, int b, Set<String> visit) {
        if (visit.contains(s)) return;
        visit.add(s);
        if (ans == null || ans.compareTo(s)>0) ans = s;
        
helper(rotate(s,b), a, b, visit);
        helper(add(s,a), a, b, visit);
    }
    
    private String rotate(String s, int b) {
        return s.substring(s.length()-b) + s.substring(0, 
s.length()-b);
    }
    private String add(String s, int a) {
        StringBuilder sb = new StringBuilder();
        for (int i=0;i<s.length();i++) {
            if (i%2 == 1) {
                int digit = (s.charAt(i)-'0'+a)%10;
                sb.append(digit);
            } else {
                
sb.append(s.charAt(i));
            }
        }
        return sb.toString();
    }
}
