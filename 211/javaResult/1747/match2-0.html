class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Set<String> visited = new HashSet<>();
        String res = s;
        Deque<String> queue = new ArrayDeque<>();
        queue.offerLast(s);
        visited.add(s);
        while (!queue.isEmpty()) {
            String cur = queue.pollFirst();
            StringBuilder sb1 = new StringBuilder();
            int len = cur.length();
            for (int i = 0; i < len; i++) {
                sb1.append(cur.charAt((i + b) % len));
            }
            String s1 = sb1.toString();
            if (visited.add(s1)) {
                visited.add(s1);
                queue.offerLast(s1);
            }
            if (s1.compareTo(res) < 0) {
                res = s1;
            }
            StringBuilder sb2 = new StringBuilder();
            for (int i = 0; i < len; i++) {
                if (i % 2 == 0) {
                    sb2.append(cur.charAt(i));
                } else {
                    sb2.append((cur.charAt(i) - '0' + a) % 10);
                }
            }
            String s2 = sb2.toString();
            if (s2.compareTo(res) < 0) {
                res = s2;
            }
            if (!visited.contains(s2)) {
                visited.add(s2);
                queue.offerLast(s2);
            }
        }
        return res;
    }
}
