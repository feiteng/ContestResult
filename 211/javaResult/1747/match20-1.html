class Solution {
    String result;

    public String findLexSmallestString(String s, int a, int b) {
        result = s;
        Set<String> vt = new HashSet<>();
        Queue<String> queue = new LinkedList<>();
        queue.offer(s);
        vt.add(s);
        b = b % 
s.length();
        while (!queue.isEmpty()) {
            String poll = queue.poll();
            if (poll.compareTo(result) < 0) {
                result = poll;
            }
            String s1 = convert(poll, a);
            if (!vt.contains(s1)) {
                vt.add(s1);
                queue.offer(s1);
            }
            String s2 = convert1(poll, b);
            if (!vt.contains(s2)) {
                vt.add(s2);
                queue.offer(s2);
            }
        }
        return result;
    }

    String convert1(String a, int b) {
        char[] chs = new char[a.length()];
        for (int i = b; i < a.length(); i++) {
            chs[i] = a.charAt(i - b);
        }

        for (int i = 0; i < b; i++) {
            chs[i] = a.charAt(a.length() - b + i);
        
}
        return new String(chs);
    }

    String convert(String s, int a) {
        StringBuilder sb = new StringBuilder(s);
        for (int i = 1; i < s.length(); i += 2) {
            int n = s.charAt(i) - '0';
            sb.setCharAt(i, (char) ((n + a) % 10 + '0'));
        }
        return sb.toString();
    }
}
