class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        if (s == null || s.isEmpty()) return s;
        
Set<String> dict = new HashSet<>();
        Queue<String> queue = new ArrayDeque<>();
        queue.offer(s);
        dict.add(s);
        String min = s;
        while (!queue.isEmpty()) {
            String cur = queue.poll();
            if (cur.compareTo(min) < 0) {
                min = cur;
            }
            String next = rotate(cur, b);
            if (dict.add(next)) {
                queue.offer(next);
            }

            next = 
add(cur, a);
            if (dict.add(next)) {
                queue.offer(next);
            }
        }

        return min;
    }

    private String rotate(String s, int b) {
        int n = s.length();
        b = b % n;
        if (b == 0) return s;
        return 
new StringBuilder().append(s, n - b, n).append(s, 0, n - b).toString();
    }

    private String add(String s, int a) {
        char[] cs = s.toCharArray();
        int n = cs.length;
        for (int i = 1; i < n; i += 2) {
            int cur = cs[i] - '0';
            int next = (cur + a) % 10;
            cs[i] = (char)('0' + next);
        }
        return String.valueOf(cs);
    }

    public static void main(String[] args) {
        Solution s = new Solution();
        System.out.println(s.findLexSmallestString("5525", 9, 2));
        System.out.println(s.findLexSmallestString("74", 5, 1));
        System.out.println(s.findLexSmallestString("0011", 4, 2));
        System.out.println(s.findLexSmallestString("43987654", 7, 3));



    }
}
