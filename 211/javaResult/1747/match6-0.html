class Solution {
    public String findLexSmallestString(String s, int a, int b) {
        Set<String> seen = new HashSet<>();
        String out = s;
        
        Queue<String> q = new LinkedList<>();
        q.add(s);
        
        while(!q.isEmpty()) {
            String next = q.poll();
            
            if(!seen.contains(next)) {
                if(next.compareTo(out) < 0) {
                    out = next;
                }
                
                q.add(add(next, a));
                q.add(rotate(next, b));
                seen.add(next);
            }
        }
        
        return out;
    }
    
    private String add(String s, int a) {
        StringBuilder sb = new StringBuilder();
        
        for(int i = 0; i < s.length(); i++) {
            if(i % 2 == 0) {
                sb.append(s.charAt(i));
            } else {
                Integer added = (Character.getNumericValue(s.charAt(i)) + a) % 10;
                sb.append(added);
            }
        }
        
        //System.out.println("add: " + s + " " + a + " " + sb.toString());
        
        return sb.toString();
    }
    
    private String rotate(String s, int b) {
        //System.out.println("rotate: " + s + " to " + s.substring(s.length() - b) + s.substring(0, s.length() - b));
        
        return s.substring(s.length() - b) + s.substring(0, s.length() - b);
    }
}
