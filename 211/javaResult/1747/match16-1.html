class Solution {
    String small = null;
    public String findLexSmallestString(String s, int a, int b) {
        small = s;
        HashSet<String> set = new HashSet<>();
        dfs(set, a, b, s);
        
return small;
    }
    void dfs(HashSet<String> set, int a, int b, String cur) {
        if (set.contains(cur)) {
            return;
        }
        if (small == null || small.compareTo(cur) > 0) {
            small = cur;
        }
        set.add(cur);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < cur.length(); i ++) {
            if (i % 2 == 1)
                sb.append((cur.charAt(i) - '0' + a) % 10);  
            else sb.append(cur.charAt(i))
;
        }
        // System.out.println(sb.toString());
        dfs(set, a, b, sb.toString());
        dfs(set, a, b, cur.substring(cur.length() - b) + cur.substring(0, cur.length() - b));
        
    }
}
