import java.util.ArrayDeque;

class Solution {

    while (!arrayDeque.isEmpty()) {
      String str = arrayDeque.remove();
      if (!treeSet.contains(str)) {
        treeSet.add(str);
        String str1 = "";
        String str2 = "";
        for (byte b = 0; b < str.length(); b++) {
          if (b % 2 == 1) {
            str1 = String.valueOf(str1) + (char)((str.charAt(b) - 48 + paramInt1) % 10 + 48);

  }
}
