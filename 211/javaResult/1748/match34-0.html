class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        Integer[] ind = new Integer[n];
        for (int i = 0; i < n; i++) ind[i] = i;
        Arrays.sort(ind, (o1, o2) -> {
            if (ages[o1] != ages[o2]) {
                return ages[o1]-ages[o2];
            } else {
                return scores[o1] - scores[o2];
            }
        });
        int[] new_scores = new int[n];
        int[] new_ages = new int[n];
        for (int i = 0; i < n; i++) {
            new_scores[i] = scores[ind[i]];
            new_ages[i] = ages[ind[i]];
        }
        
        int[] dp = new int[n];
        dp[0] = new_scores[0];
        int res = dp[0];
        for (int i = 1; i < n; i++) {
            dp[i] = new_scores[i];
            for (int j = 0; j < i; j++) {
                if (new_ages[i] == new_ages[j] || new_scores[j] <= new_scores[i]) {
                    dp[i] = Math.max(dp[i], dp[j]+new_scores[i]);
                }
            }
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}
