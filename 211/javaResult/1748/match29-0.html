class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        int[][] t = new int[n][2];
        for (int i = 0; i < n; i++) {
            t[i][0] = scores[i];
            t[i][1] = ages[i];
        }
        Arrays.sort(t, (a, b) -> a[1] == b[1] ? a[0] - b[0] : a[1] - b[1]);
        int[] dp = new int[n];
        dp[0] = t[0][0];
        int max = dp[0];
        for (int i = 1; i < n; i++) {
            dp[i] = t[i][0];
            for (int j = i - 1; j >= 0; j--) {
                if (t[j][1] == t[i][1]) {
                    dp[i] = Math.max(dp[i], dp[j] + t[i][0]);
                } else if (t[j][0] <= t[i][0]) {
                    dp[i] = Math.max(dp[i], dp[j] + t[i][0]);
                }
            }
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
