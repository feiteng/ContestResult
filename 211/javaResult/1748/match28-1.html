class Solution {

    public int bestTeamScore(int[] scores, int[] ages) {
        Integer[] index = new Integer[scores.length];
        for (int i = 0; i < scores.length; i++) {
            index[i] = i;
        
}
        Arrays.sort(index, (a, b) -> scores[a] == scores[b] ? ages[a] - ages[b] : scores[a] - scores[b]);
        int dp[] = new int[scores.length], max = 0;
        for (int i = 0; i < index.length; i++) {
            dp[i] = scores[index[i]];
            
for (int j = 0; j < i; j++) {
                if (ages[index[j]] <= ages[index[i]]) {
                    dp[i] = Math.max(dp[i], dp[j] + scores[index[i]]);
                }
            }
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
