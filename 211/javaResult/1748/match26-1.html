class Solution {
    
public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        Pair arr[] = new Pair[n];
        for(int i=0;i<n;i++){
            arr[i] = new Pair(scores[i],ages[i]);
        }
        Arrays.sort(arr,(a,b)->(a.age!=b.age)?(a.age-b.age):(a.score-b.score));
        long dp[] = new long[n];
        dp[0]=arr[0].score;
        for(int i=1;i<n;i++){
            long sum=0;
            
for(int j=i-1;j>=0;j--){
                if(arr[j].score<=arr[i].score){
                    sum = Math.max(sum,dp[j]);
                }
            }
            dp[i] = sum+arr[i].score;
        
}
        long max=0;
        for(int i=0;i<n;i++){
            max = Math.max(max,dp[i]);
        }
        return (int)(max);
    }
}

class Pair{
    int score;
    int age;
    Pair(int sc, int a){
        score=sc;
        age=a;
    }
}
