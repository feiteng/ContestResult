class Solution {
    public class Player {
        int score;
        int age;

        public Player(int score, int age) {
            this.score = score;
            this.age = age;
        }
    }

    public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        Player[] players = new Player[n];
        for (int i = 0; i < n; i++) {
            players[i] = new Player(scores[i], ages[i]);
        }
        Arrays.sort(players, (o1, o2) -> o1.age == o2.age ? o1.score - o2.score : o1.age - o2.age);
        int[] dp = new int[n];
        for (int i = 0; i < n; i++) {
            dp[i] = players[i].score;
            for (int j = 0; j < i; j++) {
                if (players[i].score >= players[j].score) {
                    dp[i] = Math.max(dp[i], players[i].score + dp[j]);
                }
            }
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}
