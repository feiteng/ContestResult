class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        int[][] tmp = new int[n][2];
        for (int i = 0; i < n; i++) {
            tmp[i][0] = scores[i];
            tmp[i][1] = ages[i];
        
} 
        Arrays.sort(tmp, (a, b) -> a[1] == b[1] ? a[0] - b[0] : a[1] - b[1]);
        int[] dp = new int[n];
        int res = 0;
        for (int i = 0; i < n; i++) {
            dp[i] = tmp[i][0];
            for (int j = 0; j < i; j++) {
                if (tmp[j][0] <= tmp[i][0]) {
                    dp[i] = Math.max(dp[i], dp[j] + tmp[i][0]);
                } 
            }
            res = Math.max(res, dp[i]);
        }
        //System.out.println(Arrays.toString(dp));
        return res;
    }
}
