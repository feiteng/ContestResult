class Solution {
    public class Comp implements Comparator<int[]>{
        public int compare(int[] a1,int[] a2){
            if(a1[1]==a2[1]){
                return a1[0]-a2[0];
            }
            return a1[1]-a2[1];
        }
    }
    
public int bestTeamScore(int[] scores, int[] ages) {
        int n=scores.length;
        int[][] array=new int[n][2];
        for(int i=0;i<n;i++){
            array[i][0]=scores[i];
            array[i][1]=ages[i];
        }
        Arrays.sort(array,new Comp());
        
int[] dp=new int[n];
        dp[0]=array[0][0];
        for(int i=1;i<n;i++){
            dp[i]=array[i][0];
            for(int j=i-1;j>=0;j--){
                if(array[i][1]==array[j][1]){
                    dp[i]=Math.max(dp[i],dp[j]+array[i][0]);
                }else if(array[i][0]>=array[j][0]){
                    dp[i]=Math.max(dp[i],dp[j]+array[i][0]);
                
}
            }
        }
        int ans=0;
        for(int v:dp){
            ans=Math.max(ans,v);
        }
        return ans;
    }
}
