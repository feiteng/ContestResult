class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int len = scores.length;
        int ps[][] = new int[len][2];
        for(int i=0;i<len;++i){
            ps[i][0] = ages[i];
            ps[i][1] = scores[i];
        }
        Arrays.sort(ps,(x,y)->{
            if(x[0]!=y[0]){
                return x[0]-y[0];
            }else {
                return x[1] - y[1];
            }
            });

        int dp[] = new int[len];
        int r = 0;
        for(int i=0;i<len;++i){
            dp[i] = ps[i][1];
            for(int j=0;j<i;++j){
                if(ps[j][1]<=ps[i][1]||ps[i][1]==ps[j][1]){
                    dp[i] = Math.max(dp[i],dp[j]+ps[i][1]);
                }
            }
            r = Math.max(r,dp[i]);
        }
        return r;
    }
}
