    public int bestTeamScore(int[] scores, int[] ages) {

//         // sort by score.

//         int l = scores.length;
//         for (int i = 0; i < l; i++) {
//             pq1.offer(new int[]{ages[i], scores[i]});
//         }
        
//         int res = 0;
//         int cur = 0;
//         while (!pq1.isEmpty()) {
//             int[] c = pq1.poll();
//             while (!pq2.isEmpty() && pq2.peek()[1] > c[1]) {
//                 cur -= pq2.poll()[1];
//             }
//             pq2.offer(c);

        int l = ages.length;
        int[][] arr = new int[l][2];
        for (int i = 0; i < l; i++) {
            arr[i][0] = ages[i];
            arr[i][1] = scores[i];
        }
        Arrays.sort(arr, (a, b) -> a[0] == b[0] ? a[1] - b[1] : a[0] - b[0]);
        
        // find the sum of not decreasing seq
        // dp[i] is the maximum sum end with i
        // dp[i] = max(dp[i], dp[j] + val[i])
        int[] dp = new int[l + 1];
        dp[1] = arr[0][1];
        for (int i = 2; i <= l; i++) {
            dp[i] = arr[i - 1][1];
            for (int j = 1; j < i; j++) {
                if (arr[i - 1][1] >= arr[j - 1][1]) {
                    dp[i] = Math.max(dp[i], dp[j] + arr[i - 1][1]);
                }
            }
        }
        
        int res = 0;
        for (int r : dp) {
            res = Math.max(res, r);
        }
        return res;
    }
}
