class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        int[][] arr = new int[n][2];
        for (int i = 0; i < n; i++) {
            arr[i][0] = scores[i];
            arr[i][1] = ages[i];
        }
        Arrays.sort(arr, (a1, a2) -> a1[1] == a2[1] ? a1[0] - a2[0] : a1[1] - a2[1]);
        // ith包括在内的最大得分
        int maxScore = arr[0][0];
        int[] dp = new int[n];
        dp[0] = arr[0][0];
        
for (int i = 1; i < n; i++) {
            dp[i] = arr[i][0]; // 至少等于自己
            for (int j = 0; j < i; j++) {
                if (arr[i][1] == arr[j][1] || arr[i][0] >= arr[j][0]) { // 年龄相同或分数更高
                    dp[i] = Math.max(dp[i], dp[j] + arr[i][0]);
                
}
            }
            maxScore = Math.max(maxScore, dp[i]);
        }
        return maxScore;
    }
}
