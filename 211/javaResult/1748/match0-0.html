class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = ages.length;
        int[][] stats = new int[n][2];
        for (int i = 0; i < n; ++i) {
            stats[i] = new int[]{scores[i], ages[i]};
        }
        Arrays.sort(stats, (a, b) -> a[1] == b[1] ? a[0] - b[0] : a[1] - b[1]);
        int[] dp = new int[n];
        int max = 0;
        for (int i = 0; i < n; ++i) {
            dp[i] = stats[i][0];
            for (int j = 0; j < i; ++j) {
                if (stats[i][1] == stats[j][1] || stats[j][0] <= stats[i][0]) {
                    dp[i] = Math.max(dp[i], dp[j] + stats[i][0]);
                }
            }
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
