class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = ages.length;
        int[][] pairs = new int[n][2];
        for (int i = 0; i < n; i++) {
            pairs[i][0] = scores[i];
            pairs[i][1] = ages[i];
        }
        Arrays.sort(pairs, (a, b) -> a[1] - b[1] == 0 ? a[0] - b[0] : a[1] - b[1]);
        // LIS
        int[] dp = new int[n + 1];
        dp[0] = 0;
        int result = 0;
        for (int i = 1; i <= n; i++) {
            int[] pair = pairs[i - 1];
            dp[i] = pair[0];
            for (int j = 1; j < i; j++) {
                int[] prev = pairs[j - 1];
                
if (prev[0] <= pair[0]) {
                    dp[i] = Math.max(dp[i], dp[j] + pair[0]);
                }
            }
            result = Math.max(result, dp[i]);
        }
        return result;
    }
}
