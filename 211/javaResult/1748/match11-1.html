class Solution {
   public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        //sort by age
        int[][] arr = new int[n][2];
        for (int i = 0; i < n; i++) {
            arr[i][0] = scores[i];
            arr[i][1] = ages[i];
        }
        Arrays.sort(arr, (a, b) -> {
            if (a[1] != b[1])
                return a[1] - b[1]; // sort by age first
            return a[0] - b[0]; //sort by score
        });
        int res = 0;
        int[] dp = new int[n];
        
for (int i = 0; i < n; i++) {
            dp[i] = arr[i][0];
            for (int j = 0; j < i; j++) {
                if (arr[j][0] <= arr[i][0]) {
                    dp[i] = Math.max(dp[i], dp[j] + arr[i][0]);
                }
            }
//            System.out.println("i=" + i + ",dp[i]=" + dp[i] );
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}
