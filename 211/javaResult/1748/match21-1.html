class Solution {
    public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length, res = 0;
        int[][] arr = new int[n][2];
        for (int i = 0; i < n; i++) {
            arr[i][0] = ages[i];
            arr[i][1] = scores[i];
        }
        Arrays.sort(arr, (e1, e2) -> e1[0] == e2[0] ? e1[1] - e2[1] : e1[0] - e2[0]);
        
        int[] dp = new int[n];
        for (int i = 0; i < n; i++) {
            dp[i] = arr[i][1];
            
for (int j = i - 1; j >= 0; j--) {
                if (arr[j][1] <= arr[i][1]) {
                    dp[i] = Math.max(dp[i], arr[i][1] + dp[j]);
                }
            }
            res = Math.max(res, dp[i]);
        }
        return res;
    }
}
