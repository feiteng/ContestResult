class Solution {
    class Member{
        int score;
        int age;
        public Member(int s, int a) {
            score = s;
            age = a;
        }
    }
    
public int bestTeamScore(int[] scores, int[] ages) {
        int n = scores.length;
        Member[] t = new Member[n];
        for (int i = 0; i < n; i++) {
            t[i] = new Member(scores[i], ages[i]);
        
}
        Arrays.sort(t, (a, b)->(a.age == b.age ? a.score - b.score : a.age - b.age));
        int[] dp = new int[n];
        dp[0] = t[0].score;
        int max = dp[0];
        for (int i = 1; i < n; i++) {
            dp[i] = t[i].score;
            for (int j = 0; j < i; j++) {
                if (!(t[i].age > t[j].age && t[i].score < t[j].score)) {
                    dp[i] = Math.max(dp[i], dp[j] + t[i].score);                
                }
            }
            max = Math.max(dp[i], max);
        }
        
        return max;
    }
}
