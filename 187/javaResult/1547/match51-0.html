class Solution {
    public String destCity(List<List<String>> paths) {
        Map<String,String> routes=new HashMap<>();
        Set<String> allPoints=new HashSet<>();
        for(List<String> path:paths){
            String start=path.get(0);
            String end=path.get(1);
            routes.put(start,end);
            allPoints.add(start);
            allPoints.add(end);
        }
        for(String p:allPoints){
            if(!routes.containsKey(p))
                return p;
        }
        return null;
    }
}
