class Solution {
    public String destCity(List<List<String>> paths) {
        HashSet<String> first = new HashSet<>();
        HashSet<String> second = new HashSet<>();
        for (List<String> path : paths) {
            first.add(path.get(0));
            second.add(path.get(1));
        }
        for (String s : second) {
            if (!first.contains(s)) {
                return s;
            }
        }
        return "";
    }
}
