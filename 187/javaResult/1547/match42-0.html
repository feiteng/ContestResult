class Solution {
    public String destCity(List<List<String>> paths) {
        Map<String, Integer> map = new HashMap<>();
        for(List<String> p:paths){
            String a = p.get(0), b = p.get(1);
            map.putIfAbsent(a, 0);
            map.putIfAbsent(b, 0);
            map.put(a, map.get(a)+1);
        }
        for(String k:map.keySet()){
            if(map.get(k)==0){
                return k;
            }
        }
        return "";
    }
}
