	    public String destCity(List<List<String>> paths) {
	        Set<String> set = new HashSet<>();
	        for(List<String> row : paths){
	        	set.add(row.get(0));
	        }
	        for(List<String> row : paths){
	        	if(!set.contains(row.get(1))){
	        		return row.get(1);
	        	}
	        }
	        return "";
	    }
