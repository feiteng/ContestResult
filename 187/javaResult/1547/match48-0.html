class Solution {
        public String destCity(List<List<String>> paths) {
        HashMap<String, String> m = new HashMap<>();
        HashSet<String> u = new HashSet<>();
        for (List<String> p : paths) {
            m.put(p.get(0), p.get(1));
            u.add(p.get(0));
            u.add(p.get(1));
        }
        for (String v : u) {
            if (!m.containsKey(v)) {
                return v;
            }
        }
        return null;        
    }
}
