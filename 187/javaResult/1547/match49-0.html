import java.util.HashSet;
import java.util.List;

class Solution {
    public String destCity(List<List<String>> paths) {

        HashSet<String> cities = new HashSet<>();
        HashSet<String> citiesWithOutpath = new HashSet<>();

        for (List<String> pair : paths) {
            String v = pair.get(0);
            String w = pair.get(1);
            cities.add(v);
            cities.add(w);
            citiesWithOutpath.add(v);
        }

        for (String city : cities) {
            if (!citiesWithOutpath.contains(city)) {
                return city;
            }
        }
        return null;

    }
}
