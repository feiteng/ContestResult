class Solution {
    public String destCity(List<List<String>> paths) {
        Set<String> start = new HashSet<>();
        Set<String> end = new HashSet<>();
        for (List<String> path : paths) {
            String from = path.get(0);
            String to = path.get(1);
            start.add(from);
            end.add(to);
        }
        for (String to : end) {
            if (!start.contains(to)) {
                return to;
            }
        }
        return null;
    }
}
