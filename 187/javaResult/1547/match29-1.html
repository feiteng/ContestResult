class Solution {
    public String destCity(List<List<String>> paths) {
        HashSet<String> set1=new HashSet<>();
        HashSet<String> set2=new HashSet<>();
        for(List<String> list:paths){
            set1.add(list.get(0));
            set1.add(list.get(1));
            set2.add(list.get(0));
        }
        for(String s:set1)
            if(!set2.contains(s))
                return s;
        return "";
    }
}
