class Solution {
    public String destCity(List<List<String>> paths) {
        
        HashMap<String,Integer> mp = new HashMap<>();
        Set<String> st =new HashSet<>();
        for(List<String> m:paths){
            String c = m.get(0);
            mp.put(c,mp.getOrDefault(c,0)+1);
            st.add(m.get(0));
            st.add(m.get(1));
        }
        for(String y:st){
            if(!mp.containsKey(y)){
                return y;
            }
        }
        return "";
        
        
        
    }
}
