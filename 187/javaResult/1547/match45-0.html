class Solution {
    public String destCity(List<List<String>> paths) {
        
        Map<String, Set<String>> graph = new HashMap<>();
        Set<String> allNodes = new HashSet<>();
        for (List<String> path : paths) {
            Set<String> connNodes = graph.getOrDefault(path.get(0), new HashSet<>());
            connNodes.add(path.get(1));
            graph.put(path.get(0), connNodes);
            
            allNodes.add(path.get(0));
            allNodes.add(path.get(1));
        } 
        
        for (String node : allNodes) {
            if (graph.getOrDefault(node, new HashSet<>()).size() ==0) {
                return node;
            }
        }
        
        return "";
        
    }
}
