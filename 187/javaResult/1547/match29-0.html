class Solution {
    public String destCity(List<List<String>> paths) {
        Set<String> m = new HashSet<>(), all = new HashSet<>();
        for (List<String> s : paths) {
            m.add(s.get(0));
            all.add(s.get(0)); all.add(s.get(1));
        }
        for (String s : all) if (!m.contains(s)) return s;
        return null;
    }
}
