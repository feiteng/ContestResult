class Solution {
    public String destCity(List<List<String>> paths) {
        HashSet<String> source = new HashSet<>();
        HashSet<String> dest = new HashSet<>();
        
        for(List<String> path: paths){
            source.add(path.get(0));
            dest.add(path.get(1));
        }
        
        for(String d: dest){
            if(!source.contains(d)){
                return d;
            }
        }
        
        return null;
    }
}
