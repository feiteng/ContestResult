class Solution {
    public String destCity(List<List<String>> paths) {
        Set<String> set1 = new HashSet<>();
        Set<String> set2 = new HashSet<>();
        for (List<String> path : paths) {
            
set1.add(path.get(0));
            set2.add(path.get(1));
        }
        for (String s : set2) {
            if (!set1.contains(s)) 
{
                return s;
            }
        }
        return "";
    }
}
