class Solution {
    public String destCity(List<List<String>> paths) {
        HashSet<String> allCities = new HashSet<>();
        HashSet<String> hasOutlinks = new HashSet<>();
        // HashSet<String> hasInlinks = new HashSet<>();
        for (List<String> lst : paths) {
            String from = lst.get(0);
            String to = lst.get(1);
            allCities.add(from);
            allCities.add(to);
            hasOutlinks.add(from);
            // hasInlinks.add(to);
        }
        for (String city : allCities) {
            if (!hasOutlinks.contains(city)) {
                return city;
            }
        }
        return null;
    }
}
