class Solution {
    public String destCity(List<List<String>> paths) {
        Set<String> not_des = new HashSet<>();
        for (List<String> list : paths) {
            not_des.add(list.get(0));
        }
        
        for (List<String> list : paths) {
            if (!not_des.contains(list.get(1))) {
                return list.get(1);
            }
        }
        
        return "";
    }
}
