class Solution {
    public String destCity(List<List<String>> paths) {
        HashSet<String> s = new HashSet<>(); 
        HashSet<String> d = new HashSet<>(); 
        for (List<String> l: paths){
            s.add(l.get(0));
            d.add(l.get(1));
        }
        for (String c: d ){
            if (!s.contains(c)) return c;
        }

        return "";
    }
}
