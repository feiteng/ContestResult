class Solution {
    public int longestSubarray(int[] nums, int limit) {
        Deque<Integer> max = new LinkedList<>();
        Deque<Integer> min = new LinkedList<>();
        
        max.add(nums[0]);
        min.add(nums[0]);
        
int beforeLoc = 0;
        int res = 1;
        for(int i = 1; i < nums.length; i++){
            int cur = nums[i];
            
            while(!max.isEmpty() && max.peekLast() < cur){
                max.removeLast();
            }
            max.addLast(cur);
            
            while(!min.isEmpty() && min.peekLast() > cur){
                min.removeLast();
            }
            min.addLast(cur);
            
            // System.out.println("max" + max);
            // System.out.println("min" + min);
            // System.out.println("----------");
            int diff = max.peek() - min.peek();
            while(diff > limit){
                int before = nums[beforeLoc++];
                if(max.peek() == before){
                    max.removeFirst();
                }
                
                if(min.peek() == before){
                    min.removeFirst();
                }
                diff = max.peek() - min.peek();
            
}
            
            res = Math.max(res, i - beforeLoc + 1);
        }
        
        return res;
    }
}
