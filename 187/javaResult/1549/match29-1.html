import java.util.*;


class Solution {
    public int longestSubarray(int[] nums, int limit) {
        SortedMap<Integer, Integer> s = new TreeMap<>();
        
        int result = 0;
        int b = 0;
        for (int i = 0; i < nums.length; i++) {
            s.put(nums[i], s.get(nums[i]) != null ? s.get(nums[i]) + 1 : 1);
            while (
s.lastKey() - s.firstKey() > limit) {
                s.put(nums[b], s.get(nums[b]) - 1);
                if (s.get(nums[b]) == 0) {
                    s.remove(nums[b]);
                }
                b++;
            }
            
            if (i - b + 1 > result) {
                result = i - b + 1;
            }
        }
        
        return result;
    }
}
