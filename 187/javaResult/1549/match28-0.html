class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int ans = 0;
        int cur = 0;
        TreeMap<Integer, Integer> map = new TreeMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(nums[i])) {
                map.put(nums[i], map.get(nums[i]) + 1);
            } else {
                map.put(nums[i], 1);
            }
            cur++;
            while (map.size() > 1 && map.lastKey() - map.firstKey() > limit) {
                int el = nums[i - cur + 1];
                int cnt = map.get(el);
                if (cnt == 1) {
                    map.remove(el);
                } else {
                    map.put(el, cnt - 1);
                }
                cur--;
            }
            ans = Math.max(ans, cur);
        }
        return ans;
    }
}
