class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int lss = 0;
        Deque<Integer> minDeque = new LinkedList();
        Deque<Integer> maxDeque = new LinkedList();
        int left = 0;
        int right = 0;
        while(right<nums.length) {
            int value = nums[right];
            while(!minDeque.isEmpty() && minDeque.peekFirst() > value) {
                minDeque.poll();
            }
            minDeque.addFirst(value);
            
            while(!maxDeque.isEmpty() && maxDeque.peekFirst() < value) {
                maxDeque.poll();
            }
            maxDeque.addFirst(value);
            
            int diff = Math.abs(minDeque.peekLast() - maxDeque.peekLast());

            while(diff > limit) {
                int lValue = nums[left];
                if(minDeque.peekLast().equals(lValue)) {
                    minDeque.pollLast();
                }
                
                if(maxDeque.peekLast().equals(lValue)) {
                    maxDeque.pollLast();
                }
                
                if(!minDeque.isEmpty())
                    diff = Math.abs(minDeque.peekLast() - maxDeque.peekLast());
                else 
                    diff=0;
                
                left++;
            }
            
            lss = Math.max(lss, right-left+1);
            
            right++;
        }
            return lss;
    }
}
