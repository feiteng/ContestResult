class Solution {
    public int longestSubarray(int[] nums, int limit) {
        PriorityQueue<Integer> pq1 = new PriorityQueue<>((a, b) -> (nums[b] - nums[a]));
        PriorityQueue<Integer> pq2 = new PriorityQueue<>((
a, b) -> (nums[a] - nums[b]));
        
        int r = 0, l = 0;
        int cnt = 0;
        while(r < nums.length) {
            pq1.offer(r);
            pq2.offer(r);
            while(nums[pq1.peek()] > nums[pq2.peek()] + limit) {
                pq1.remove((Integer)l);
                pq2.remove((Integer)l);
                l++;
            }
            cnt = Math.max(cnt, r - l + 1);
            r++;
        }
        return cnt;
    }
}
