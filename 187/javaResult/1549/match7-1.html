class Solution {
    public int longestSubarray(int[] nums, int limit) {
        TreeMap<Integer, Integer> valToFreq = new TreeMap<>();
        int slow = 0;
        int maxLen = 0;
        
        for (int fast = 0; fast < nums.length; fast++) {
            // 1. add fast
            valToFreq.put(nums[fast], valToFreq.getOrDefault(nums[fast], 0) + 1);
            
            // 2. move slow
            while (Math.abs(valToFreq.lastKey() - valToFreq.firstKey()) > limit && slow <= fast) {
                valToFreq.put(nums[slow], valToFreq.get(nums[slow]) - 1);
                if (valToFreq.get(nums[slow]) == 0) {
                    valToFreq.remove(nums[slow]);
                
}
                slow++;
            }
            
            // 3. update result
            maxLen = Math.max(maxLen, fast - slow + 1);
        }
        
        return maxLen;
    }
}
