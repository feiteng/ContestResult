class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int ans= 1;
        int beg=nums[0];
        int end = nums[0];
        TreeMap<Integer, Integer> dp = new TreeMap<>();
        dp.put(nums[0],1);
        int st=0;
        int count =1;
        for(int i=1;i<nums.length;i++){
            
            if(!dp.containsKey(nums[i])){
                dp.put(nums[i], 0);
            }
            dp.put(nums[i], dp.get(nums[i])+1);
            count++;
            
             if(nums[i]<beg){
                if(end-nums[i]>limit){
                    
                    while(end-nums[i]>limit){
                        int p=end;
                        
while(dp.containsKey(p)&&dp.get(p)>0){
                            int ck= dp.get(nums[st]);
                            ck--;
                            if(ck==0)
                                dp.remove(nums[st]);
                            else
                                dp.put(nums[st],ck);
                            st++;
                            count--;
                        }
                        
                        end = dp.lastKey();
                    }
                }
                 beg= nums[i];
                
            }
            else if(nums[i]>end){
                 if(nums[i]-beg>limit){
                    while(nums[i]-beg>limit){
                        int p=beg;
                        while(dp.containsKey(p)&&dp.get(p)>0){
                            int ck= dp.get(nums[st]);
                            ck--;
                            if(ck==0)
                                dp.remove(nums[st]);
                            else
                                dp.put(nums[st],ck);
                            st++;
                            count--;
                        }
                        beg = dp.firstKey();
                    }
                }end= nums[i];
            }
                    ans=Math.max(count, ans);
        }
        return ans;
    }
}
