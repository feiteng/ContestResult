class Solution {
    public int longestSubarray(int[] nums, int limit) {
        TreeMap<Integer, Integer> treeMap = new TreeMap();
        int maxLen = 0;
        int head = 0;
        for (int i = 0; i < nums.length; i ++) {
            int num = nums[i];
            treeMap.put(num, treeMap.getOrDefault(num, 0) + 1);
            
            while(treeMap.size() >= 2 && Math.abs(treeMap.firstKey() - treeMap.lastKey()) > limit) {
                int toremove = nums[head];
                treeMap.put(toremove, treeMap.get(toremove) - 1);
                if(treeMap.get(toremove) == 0) treeMap.remove(toremove);
                head++;
            }
            maxLen = Math.max(maxLen, i - head + 1);
        }
        return maxLen;
    }
}
