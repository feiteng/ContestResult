class Solution {
    public int longestSubarray(int[] nums, int limit) {
        // 2 deque
        ArrayDeque<Integer> qMax = new ArrayDeque<>(); 
        ArrayDeque<Integer> qMin = new ArrayDeque<>(); 
        
        int n = nums.length;
        int res = 0;
        int index = 0;
        for (int i = 0; i < n; i++) {
            int num = nums[i];
            while (!qMax.isEmpty() && nums[qMax.peekLast()] <= num) {
                qMax.pollLast();
            } 
            qMax.add(i);
            while (!qMin.isEmpty() && nums[qMin.peekLast()] >= num) {
                qMin.pollLast();
            }
            qMin.add(i);
            // check limit
            while (index <= i && nums[qMax.peekFirst()] - nums[qMin.peekFirst()] > limit) {
                index++;
                while (qMax.peekFirst() < index) qMax.pollFirst();
                while (qMin.peekFirst() < index) qMin.pollFirst();
            }
            // System.out.println("index=" + index);
            res = Math.max(res, i - index + 1);
        }
        return res;
    }
}
