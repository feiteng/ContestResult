class Solution {
    public int longestSubarray(int[] nums, int limit) {
        
TreeMap<Integer, Integer> ct = new TreeMap<>();
        int m = -1, b = 0;
        for(int i = 0; i < nums.length; i++) {
            if(!ct.containsKey(nums[i]))
                ct.put(nums[i], 0);
            ct.put(nums[i], ct.get(nums[i]) + 1);
            
while(ct.lastKey() - ct.firstKey() > limit) {
                ct.put(nums[b], ct.get(nums[b]) - 1);
                if(ct.get(nums[b]) == 0)
                    ct.remove(nums[b]);
                b++;
            }
            m = Math.max(m, i - b + 1);
        }
        return m;
    }
}
