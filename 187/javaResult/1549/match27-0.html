class Solution {
    public int longestSubarray(int[] nums, int limit) {
        PriorityQueue<Integer> que_min = new PriorityQueue<>();
        PriorityQueue<Integer> que_max = new PriorityQueue<>((o1, o2) -> (o2 - o1));
        
        if(nums.length == 0)    return 0;
        int n = nums.length;
        int l = 0, r = 0, len = 0, ans = 0;
        
        while(r < n) {
            que_min.add(nums[r]);   que_max.add(nums[r]);   len += 1;   r += 1;
            while(que_max.peek() - que_min.peek() > limit) {
                que_min.remove(nums[l]);    que_max.remove(nums[l]);
                l++;    len -= 1;
            }
            ans = Math.max(ans, len);
        }
        
        return ans;
    }
}
