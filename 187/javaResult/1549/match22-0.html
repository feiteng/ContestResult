class Solution {
    public int longestSubarray(int[] nums, int limit) {
        int best = 0;
        int bestStart = 0, bestEnd = 0;

        Deque<Integer> minQueue = new LinkedList<>();
        Deque<Integer> maxQueue = new LinkedList<>();

        int start = 0, end = 0;

        while (end < nums.length) {
            int x = nums[end];

            while (!minQueue.isEmpty() && nums[minQueue.peekLast()] >= x) {
                minQueue.removeLast();
            }
            minQueue.addLast(end);

            while (!maxQueue.isEmpty() && nums[maxQueue.peekLast()] <= x) {
                maxQueue.removeLast();
            }
            maxQueue.addLast(end);

            int minIdx = minQueue.peekFirst();
            int minVal = nums[minIdx];

            int maxIdx = maxQueue.peekFirst();
            int maxVal = nums[maxIdx];

            if (maxVal - minVal > limit) {
                start++;
                if (start > minQueue.peekFirst()) {
                    minQueue.removeFirst();
                }
                if (start > maxQueue.peekFirst()) {
                    maxQueue.removeFirst();
                }
            } else {
                if (end - start + 1 > best) {
                    best = end - start + 1;
                    bestStart = start;
                    bestEnd = end;
                }

                end++;
            }
        }

        return best;
    }
}
