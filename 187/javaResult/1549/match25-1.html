class Solution {
    public int longestSubarray(int[] nums, int limit) {
        Queue<Integer> minQueue = new PriorityQueue<>();
        Queue<Integer> maxQueue = new PriorityQueue<>((a, b) -> b - a);

        
int res = 0;
        int last = 0;
        for (int i = 0; i < nums.length; i++) {
            minQueue.offer(nums[i]);
            maxQueue.offer(nums[i]);

            while (!minQueue.isEmpty() && maxQueue.peek() - minQueue.peek() > limit) {
                minQueue.remove(nums[last]);
                maxQueue.remove(nums[last]);
                last++;
            
}

            res = Math.max(res, i - last + 1);
        }

        return res;
    }
}
