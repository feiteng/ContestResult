class Solution {
    public boolean kLengthApart(int[] nums, int k) {
         int start = -1;
        for (int i=0;i<nums.length;i++){
            if (nums[i]==0) continue;
            if (start==-1){
                start = i;
                continue;
            }
            if (i-start<=k) 
return false;
            start = i;
        }
        return true;
    }
}
