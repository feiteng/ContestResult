class Solution {
    public boolean kLengthApart(int[] nums, int k) {
        int pre = -10000000;
        
        for(int idx=0; idx<nums.length; idx++) {
            if(nums[idx] == 1) {
                //System.out.println(idx+" "+pre);
                if(idx - pre-1 < k) {
                    
                    return false;
                }
                pre = idx;
            }
        }
        return true;
    }
}
