class Solution {
    public boolean kLengthApart(int[] nums, int k) {
        int l = -nums.length - 1;
        for (int r = 0; r < nums.length; r++) {
            if (nums[r] == 1) {
                if (r - l - 1 < k) {
                    return false;
                }
                l = r;
            }
        }
        return true;
    }
}
