class Solution {
    public boolean kLengthApart(int[] nums, int k) {
        //System.out.println(Arrays.toString(nums) + " " + k);
        
        int prev = -1000000;
        
        for(int i = 0; i < nums.length; i++) {
            
            if(nums[i] == 1) {
             //System.out.println(prev + " " + i + " " + nums[i] + " " + (i - prev) + " " + k);               
                if(i - prev <= k) {
                    return false;
                }
                
                prev = i;
            }
        }
        
        return true;
    }
}
