class Solution {
   public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<Elem> queue = new PriorityQueue<>();
        
        // 1. init
        List<Integer> indices = new ArrayList<>();
        int sum = 0;
        for (int r = 0; r < mat.length; r++) {
            indices.add(0);
            sum += mat[r][0];
        }
        queue.offer(new Elem(indices, sum));
        
        Set<List<Integer>> visited = new HashSet<>();
        
        // 2. get result
        for (int i = 0; i < k-1; i++) {
            Elem cur = queue.poll();
            for (int r = 0; r < mat.length; r++) {
                if (cur.indices.get(r) + 1 < mat[r].length) {
                    List<Integer> newIndices = new ArrayList<>(cur.indices);
                    newIndices.set(r, newIndices.get(r)+1);
                    
                    if (visited.add(newIndices)) {
                        int newSum = cur.sum - mat[r][cur.indices.get(r)] + mat[r][cur.indices.get(r)+1];
                        queue.offer(new Elem(newIndices, newSum));  	
                    }

                }
            }
        }
        
        return queue.poll().sum;
    }
    
    class Elem implements Comparable<Elem> {
        List<Integer> indices;
        int sum;
        Elem(List<Integer> indices, int sum) {
            this.indices = indices;
            this.sum = sum;
        }
        
        @Override
        public int compareTo(Elem another) {
            return Integer.compare(this.sum, another.sum);
        }
    }
}
