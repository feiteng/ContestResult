class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int n = mat.length;
        int m = mat[0].length;
        
List<Integer> res = firstK(mat, 0, n-1, k);
        return res.get(k-1);
    }
    
    public List<Integer> firstK(int[][] mat, int lo, int hi, int k){
        if(lo==hi){
            List<Integer> res = new ArrayList();
            for(int i=0; i<k && i<mat[0].length; i++){
                res.add(mat[lo][i]);
            }
            return res;
        }
        
        int mid = lo + (hi-lo)/2;
        List<Integer> nums1 = firstK(mat, lo, mid, k);
        List<Integer> nums2 = firstK(mat, mid+1, hi, k);
        
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>((a,b)->(nums1.get(a[0])+nums2.get(a[1])-nums1.get(b[0])-nums2.get(b[1])));
        int n = nums1.size();
        int m = nums2.size();
        if(n==0 || m==0){
            return new ArrayList();
        }
        for(int i=0; i<n; i++){
            pq.offer(new int[]{i,0});
        }
        List<Integer> res = new ArrayList();
        int count = 0;
        while(count<Math.min(k, n*m)){
            int[] cur = pq.poll();
            res.add(nums1.get(cur[0])+nums2.get(cur[1]));
            if(++cur[1]<m){
                
pq.offer(cur);
            }
            count++;
        }
        return res;       
    }
