class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length;
        Queue<int[]> pq = new PriorityQueue<>((vec1, vec2) ->(vec1[0] - vec2[0]));
        int[] vec = new int[m + 1];
        int total = 0;
        for(int i = 0; i < m; ++i) total += mat[i][0];
        vec[0] = total;
        
        String sts = Arrays.toString(vec);
        pq.offer(vec);
        Set<String> st = new HashSet<>();
        st.add(sts);
        int count = k - 1;
        while(count > 0 && !pq.isEmpty()){
            int[] temp = pq.poll();
            int s1 = temp[0];
            for(int i = 0; i < m ; ++i){
                int id = temp[i + 1];
                if(id + 1 < mat[i].length){
                    int val1 = mat[i][id];
                    int val2 = mat[i][id + 1];
                    int s2 = s1 - val1 + val2;
                    temp[i+1]++;
                    temp[0] = s2;
                    sts = Arrays.toString(temp);
                    if(st.contains(sts)){
                        temp[i + 1]--;
                        continue;
                    }
                    st.add(sts);
                    pq.offer(temp.clone());
                    temp[i+1]--;
                }
            }
            count--;
        }
        return pq.poll()[0];
    }
}
