class Solution {
    public int kthSmallest(int[][] mat, int k) {
        // [[1,10,10]
        //  [1,4,5],
        //  [2,3,6]]
        // 0, 0, 0
        class Triple {
            List<Integer> indexes;
            int sum;
            public Triple(List<Integer> indexes, int sum) {
                this.indexes = indexes;
                this.sum = sum;
            }
        }
        if (mat.length == 0 || mat[0].length == 0) return 0;
        int m = mat.length;
        int n = mat[0].length;
        PriorityQueue<Triple> pq = new PriorityQueue<>((t1, t2) -> t1.sum - t2.sum);
        List<Integer> indexes = new ArrayList<>();
        int sum = 0;
        for (int i = 0; i < m; i++) {
            indexes.add(0);
            sum += mat[i][0];
        }
        Set<List<Integer>> visited = new HashSet<>();
        visited.add(indexes);
        pq.offer(new Triple(indexes, sum));
        while(k > 1) {
            Triple t = pq.poll();
            k--;
            List<Integer> currIndexes = t.indexes;
            int currSum = t.sum;
            for (int i = 0; i < m; i++) {
                if (currIndexes.get(i) + 1 < n) {
                    List<Integer> newIndexes = new ArrayList<>(t.indexes);
                    newIndexes.set(i, currIndexes.get(i) + 1);
                    if (visited.contains(newIndexes)) continue;
                    int newSum = currSum - mat[i][currIndexes.get(i)] + mat[i][newIndexes.get(i)];
                    pq.offer(new Triple(newIndexes, newSum));
                    visited.add(newIndexes);
                } 
            } 
        }
        return pq.peek().sum;
    }
}
