class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        
        PriorityQueue<int[]> queue = new PriorityQueue<>((a, b)->{
            return getSum(mat, a) - getSum(mat, b);
        });
        
        HashSet<String> visited = new HashSet<>();
        
        int count = 0;
        int[] indexs = new int[m];
        queue.offer(indexs);
        while(!queue.isEmpty()){
            count++;
            if (count == k){
                return getSum(mat, queue.poll());
            }
            // pick next sum, put all possible into the queue 
            int[] cur = queue.poll();
            // System.out.println(getSum(mat, cur));
            for (int i = 0; i < cur.length; i++){
                int idx = cur[i];
                if (idx < n - 1){
                    cur[i]++;
                    int[] next = new int[cur.length];
                    System.arraycopy(cur, 0, next, 0, m);
                    String key = getKey(next);
                    if (!visited.contains(key)){
                        visited.add(key);
                        queue.offer(next);
                    }
                    cur[i] = idx;
                }
            }
        }
        
        return -1;
    }
    
    private String getKey(int[] indexs){
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < indexs.length; i++){
            sb.append(indexs[i]);
        }
        return sb.toString();
    }
    
    private int getSum(int[][] mat, int[] indexs){
        int sum = 0;
        for (int i = 0; i < indexs.length; i++){
            sum += mat[i][indexs[i]];
        }
        return sum;
    }
}
