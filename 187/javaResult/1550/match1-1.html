class Solution {
    
private class Ele {
        List<Integer> idxes;
        int sum;
        public Ele(List<Integer> idxes, int sum) {
            this.idxes = idxes;
            this.sum = sum;
        }
    }
    public int kthSmallest(int[][] mat, int k) {
        
PriorityQueue<Ele> pq = new PriorityQueue<>(new Comparator<Ele>(){
            @Override
            public int compare(Ele e1, Ele e2) {
                return e1.sum - e2.sum;
            }
        });
        
        int sum = 0;
        List<Integer> list = new ArrayList<>();
        for (int i = 0; i < mat.length; i++) {
            sum += mat[i][0];
            list.add(0);
        }
        int n = mat.length, m = mat[0].length;
        pq.offer(new Ele(list, sum));
        
        Set<List<Integer>> visited = new HashSet<>();
        
visited.add(list);
        
        while (--k > 0) {
            Ele cur = pq.poll();
            for (int i = 0; i < n; i++) {
                if (cur.idxes.get(i) + 1 < m) {
                    int tmp = cur.idxes.get(i);
                    cur.idxes.set(i, tmp + 1);
                    List<Integer> cp = copy(cur.idxes);
                    if (visited.add(cp)) {
                        pq.offer(new Ele(cp, cur.sum + (mat[i][tmp + 1] - mat[i][tmp])));
                    }
                    cur.idxes.set(i, tmp);
                
}
            }
        }
        
        return pq.peek().sum;
    }
    
    private List<Integer> copy(List<Integer> idxes) {
        List<Integer> cp = new ArrayList<>();
        for (int i = 0; i < idxes.size(); i++) {
            cp.add(idxes.get(i));
        }
        
        return cp;
    }
}
