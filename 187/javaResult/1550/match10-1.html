class Solution {

	public int kthSmallest(int[][] mat, int k) {
		PriorityQueue<
int[]> queue = new PriorityQueue<>((a, b) -> a[0] - b[0]);
		int[] first = new int[mat.length + 1];
		for (int[] m : mat) {
			first[0] += m[0];
		}
		queue.add(first);
		HashSet<ArrayList<Integer>> set = new HashSet<>();
		ArrayList<Integer> list = new ArrayList<>();
		for (int i = 0; i < mat.length; i++) {
			list.add(0);
		}
		set.add(list);

		for (int i = 1; i < k; i++) {
			int[] next = queue.remove();
			for (int j = 0; j < mat.length; j++) {
				if (next[j + 1] < mat[0].length - 1) {

					int[] clone = next.clone();
					clone[j + 1]++;
					ArrayList<Integer> l = new ArrayList<>();
					for (int m = 1; m < clone.length; m++) {
						l.add(clone[m]);
					}
					if (!set.contains(l)) {
						set.add(l);
						clone[0] += mat[j][clone[j + 1]] - mat[j][clone[j + 1] - 1];
						queue.add(clone);
					}
				}
			}
		}
		return queue.remove()[0];
	}
}
