class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<state>PQ = new PriorityQueue<>();
        int[]index = new int[mat.length];
        int num = 0;
        for(int i = 0;i<mat.length;i++){
            num+=mat[i][0];
            index[i]=0;
        }
        HashSet<String>used = new HashSet<>();
        PQ.add(new state(index,num));
        int ans = 0;
        while(k!=0&&PQ.size()!=0){
            state temp = PQ.poll();
            int s = used.size();
            used.add(Arrays.toString(temp.index));
            if(used.size()==s)
                continue;
            //System.out.println(temp.val+" "+k+" "+Arrays.toString(temp.index)+" "+used.size());
            ans = temp.val;
            for(int i = 0;i<mat.length;i++){
                if(temp.index[i]==mat[0].length-1)
                    continue;
                int[]temparr = temp.index.clone();
                temparr[i]++;
                PQ.add(new state(temparr,temp.val-mat[i][temparr[i]-1]+mat[i][temparr[i]]));
            }
            k--;
        }
        //System.out.println(ans);
        return ans;
        
        
        
        
    }
    static class state implements Comparable<state>{
        int[]index;
        int val;
        public state(int[]i, int v){
            index = i;
            val = v;
        }
        public int compareTo(state s){
            return val-s.val;
        }
    }
}
