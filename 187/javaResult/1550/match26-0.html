class Solution {
    public int kthSmallest(int[][] mat, int k) {
        
        int m = mat.length;
        int n = mat[0].length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[m] - b[m]));
        Set<String> set = new HashSet<>();
        
        int[] arr = new int[m + 1];
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < m; i++) {
            arr[m] += mat[i][0];
            sb.append(0).append('|');
        }
        set.add(sb.toString());
        
        pq.offer(arr);
        
        while(k > 0 && !pq.isEmpty()) {
            
            if(k == 1) return pq.poll()[m];
            
            k--;
            int[] a = pq.poll();
            for(int i = 0; i < m; i++) {
                if(a[i] + 1 < n) {
                    int[] b = a.clone();
                    b[i] += 1;
                    sb = new StringBuilder();
                    for(int j = 0; j < m; j++) sb.append(b[j]).append('|');
                    b[m] += + mat[i][b[i]] - mat[i][b[i] - 1];
                    if(set.add(sb.toString())) pq.offer(b);
                }
            }
            
        }
        return -1;
    }
}
