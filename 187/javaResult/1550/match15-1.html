class Solution {
    int[][] mat;
    public int kthSmallest(int[][] mat, int k) {
        this.mat = mat;
        PriorityQueue<List<Integer>> queue = new PriorityQueue<>((l1, l2) -> {
            int sum1 = getSum(l1);
            int sum2 = getSum(l2);
            return sum1 - sum2;
        });
        int m = mat.length, n = mat[0].length;
        
List<Integer> mins = new ArrayList<>();
        for (int i = 0 ; i < m; i++) {
            mins.add(0);
        }
        queue.add(mins);
        int count = 0;
        Set<List<Integer>> visited = new HashSet<>();
        visited.add(mins);
        while (!queue.isEmpty()) {
            List<Integer> top = queue.poll();
            // System.out.println("Top - " + top);
            count++;
            if (count == k) {
                return getSum(top);
            }
            // Add its neighbors
            
for (int i = 0; i < m; i++) {
                int col = top.get(i);
                if (col < n-1) {
                    top.set(i, col+1);
                    List<Integer> next = new ArrayList<>(top);
                    if (!visited.contains(next)) {
                        queue.add(next);
                        visited.add(next);
                    
}
                    top.set(i, col);
                }
            }
        }
        return -1;
    }
    
    private int getSum(List<Integer> cols) {
        
int sum = 0;
        for (int i = 0; i < mat.length; i++) {
            sum += mat[i][cols.get(i)];
        }
        return sum;
    }
}
