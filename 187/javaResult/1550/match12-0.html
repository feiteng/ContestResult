class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        Set<String> set = new HashSet();
        PriorityQueue<int[]> queue = new PriorityQueue(new Comparator<int[]>(){
            public int compare(int[] a, int[] b){
                int sumA = 0, sumB = 0;
                for(int i = 0; i < m; i++){
                    sumA+=mat[i][a[i]];
                    sumB+=mat[i][b[i]];
                }
                return sumA-sumB;
            }
        });
        queue.add(new int[m]);
        while(k > 0){
            int[] cur = queue.poll();
            k--;
            if(k == 0){
                int res = 0;
                for(int i = 0;i < m; i++){
                    res+=mat[i][cur[i]];
                }
                return res;
            }
            for(int i = 0; i < m; i++){
                if(cur[i] < n-1){
                    int[] temp = new int[m];
                    for(int j = 0; j < m; j++){
                        temp[j] = cur[j];
                    }
                    temp[i] = cur[i]+1;
                    String str = helper(temp);
                    if(!set.contains(str)) {
                        set.add(str);
                        queue.add(temp);
                    }
                }
            }
        }
        return 0;
    }
    
    public String helper(int[] arr){
        StringBuilder sb = new StringBuilder();
        for(int i: arr){
            sb.append(i+" ");
        }
        return sb.toString();
    }
}
