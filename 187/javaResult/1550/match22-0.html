class Solution {
    int m, n;
    public int kthSmallest(int[][] mat, int k) {
        m = mat.length;
        n = mat[0].length;
        
        List<Integer> res = merge(mat, 0, m - 1, k);
        return res.get(k - 1);
    }
    
    private List<Integer> merge(int[][] mat, int start, int end, int k) {
        List<Integer> res = new ArrayList<>();
        if (start == end) {
            for (int i = 0; i < n; i++) res.add(mat[start][i]);
            return res;
        }
        
        int mid = start + (end - start) / 2;
        List<Integer> a = merge(mat, start, mid, k);
        List<Integer> b = merge(mat, mid + 1, end, k);
        
        return findKthSumInTwoList(a, b, k);
    }
    
    private List<Integer> findKthSumInTwoList(List<Integer> a, List<Integer> b, int k) {
        PriorityQueue<int[]> pq = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] x, int[] y) {
                return x[0] + x[1] - y[0] - y[1];
            }
        });
        
        for (int i = 0; i < a.size(); i++) {
            pq.offer(new int[]{a.get(i), b.get(0), 0});
        }
        
        List<Integer> res = new ArrayList<>();
        while (k > 0 && !pq.isEmpty()) {
            int[] curr = pq.poll();
            res.add(curr[0] + curr[1]);
            curr[2]++;
            k--;
            
            if (curr[2] < b.size()) {
                curr[1] = b.get(curr[2]);
                pq.offer(curr);
            }
        }
        
        return res;
    }
}
