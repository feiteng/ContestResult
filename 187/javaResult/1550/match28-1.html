class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n= mat[0].length;
        PriorityQueue<Tuple> queue = new PriorityQueue<>();
        int sum = 0;
        for(int i = 0; i< m; i++)sum += mat[i][0];
        queue.add(new Tuple(new int[m], sum));
        TreeSet<Tuple> set = new TreeSet<>();
        while(!queue.isEmpty()){
            Tuple t = queue.poll();
            if(k == 1)return t.sum;
            k--;
            for(int i = 0; i< m; i++){
                if(t.ind[i] != n-1){
                    int[] nxt = new int[m];
                    for(int j = 0; j< m; j++)nxt[j] = t.ind[j];
                    nxt[i]++;
                    
Tuple tmp = new Tuple(nxt, t.sum+mat[i][nxt[i]]-mat[i][nxt[i]-1]);
                    if(!set.contains(tmp)){
                        set.add(tmp);
                        queue.add(tmp);
                    }
                }
            }
        }
        return -1;
    }
    class Tuple implements Comparable<Tuple>{
        int[] ind;
        int sum;
        public Tuple(int[] x, int s){
            ind = x;sum = s;
        }
        public int compareTo(Tuple t){
            if(sum != t.sum)return Integer.compare(sum, t.sum);
            for(int i = 0; i< ind.length; i++)if(ind[i] != t.ind[i])return Integer.compare(ind[i], t.ind[i]);
            return 0;
        }
    }
}
