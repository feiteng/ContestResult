class Solution {
    public static class Node implements Comparable {
        Integer val;
        int[] pointer;

        public Node(int val, int[] pointer) {
            this.val = val;
            this.pointer = pointer;
        }

        @Override
        public int compareTo(Object o) {
            return val.compareTo(((Node) o).val);
        }

        @Override
        public boolean equals(Object o) {
            if (this == o) return true;
            if (o == null || getClass() != o.getClass()) return false;
            Node node = (Node) o;
            return Objects.equals(val, node.val) &&
                Arrays.equals(pointer, node.pointer);
        }

        @Override
        public int hashCode() {
            int result = Objects.hash(val);
            result = 31 * result + Arrays.hashCode(pointer);
            return result;
        }
    }

    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length;
        int n = mat[0].length;
        int[] l = new int[m];
        PriorityQueue<Node> q = new PriorityQueue<>();
        int sum = 0;
        for (int i = 0; i < m; i++) {
            sum += mat[i][l[i]];
        }
        Node initNode = new Node(sum, l);
        Set<Node> history = new HashSet<>();
        history.add(initNode);
        q.add(initNode);
        int cnt = 0;

        while (!q.isEmpty()) {
            Node node = q.poll();
            cnt++;
            if (cnt == k) {
                return node.val;
            }
            for (int i = 0; i < m; i++) {
                if (node.pointer[i] + 1 < n) {
                    int[] pointer = Arrays.copyOf(node.pointer, node.pointer.length);
                    pointer[i] ++;
                    Node newNode = new Node(node.val - mat[i][node.pointer[i]] + mat[i][node.pointer[i] + 1], pointer);
                    if (!history.contains(newNode)) {
                        history.add(newNode);
                        q.add(newNode);
                    }
                }
            }
        }
        return 0;
    }
}
