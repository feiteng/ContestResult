class Solution {
    int size;
    public int kthSmallest(int[][] mat, int k) {
        this.size = k;
        PriorityQueue<Cell> minHeap = new PriorityQueue<>();
        List<Integer> origin = new ArrayList<>();
        for (int i = 0; i < mat.length; i++) {
            origin.add(0);
        }
        Set<List<Integer>> set = new HashSet<>();
        set.add(origin);
        minHeap.offer(new Cell(origin, mat));
        while (k > 1 ) 
{
            Cell cur = minHeap.poll();
            for (int i = 0; i < mat.length; i++) {
                if (mat[i].length > cur.array.get(i) + 1) {
                    List<Integer> newList = new ArrayList<>(cur.array);
                    newList.set(i, cur.array.get(i) + 1);
                    if (set.add(newList)) {
                        
minHeap.offer(new Cell(newList, mat));
                    }                    
                }
            }
            
            k--;
        }
        
return minHeap.peek().sum;
        
    }
    static class Cell implements Comparable<Cell>{
        int sum;        
        List<Integer> array;
        Cell(List<Integer> array, int[][] mat) {
            this.array = array;
            int minVal = Integer.MAX_VALUE;            
            sum = 0;
            for (int i =0; i < array.size(); i++) {                
                sum += mat[i][array.get(i)];
            }
        }
        @Override
        public int compareTo(Cell another) {
            return this.sum - another.sum;
        }
    }
}
