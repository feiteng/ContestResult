class Solution {
        public int kthSmallest(int[][] mat, int k) {
        int R = mat.length;
        int C = mat[0].length;

        PriorityQueue<Integer> pq = new PriorityQueue<>((a,b) -> b - a);
        for (int c = 0; c < C; c++) {
            pq.add(mat[0][c]);
        }
        while (pq.size() > k) {
            pq.poll();
        }
        
        for (int r = 1; r < R; r++) {
            PriorityQueue<Integer> npq = new PriorityQueue<>((a,b) -> b - a);
            
for (int i : pq) {
                for (int c = 0; c < C; c++) {
                    npq.add(i + mat[r][c]);
                }
            }
            pq = npq;
            while (pq.size() > k) {
                pq.poll();
            }
        }
        
        return pq.poll();
    }
}
