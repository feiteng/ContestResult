class Solution {
    
    class Step {
        int sum;
        int[] arr;
        public Step(int _sum, int[] _arr) {
            this.sum = _sum;
            this.arr = _arr;
        }
    }
    
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        
        int cur_sum = 0;
        for (int i = 0; i < m; i++) {
            cur_sum += mat[i][0];
        }
        
        int[] cur_arr = new int[m];
        
        Set<String> visited = new HashSet<>();
        PriorityQueue<Step> queue = new PriorityQueue<>((a, b) -> Integer.compare(a.sum, b.sum));
        queue.offer(new Step(cur_sum, cur_arr));
        visited.add(Arrays.toString(cur_arr));
        
        while (k > 1) {
            Step cur = queue.poll();
            k--;
            
            cur_arr = cur.arr;
            cur_sum = cur.sum;

            for (int i = 0; i < m; i++) {
                if (cur_arr[i] < n-1) {
                    int[] nxt_arr = cur_arr.clone();
                    nxt_arr[i]++;
                    String key = Arrays.toString(nxt_arr);
                    if (!visited.contains(key)) {
                        visited.add(key);
                        int nxt_sum = cur_sum - mat[i][cur_arr[i]] + mat[i][nxt_arr[i]];
                        queue.offer(new Step(nxt_sum, nxt_arr));
                    }
                }
            }
        }
        
        return queue.poll().sum;
    }
}
