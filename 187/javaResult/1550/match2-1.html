class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length;
        Queue<int[]> q = new PriorityQueue<>((a, b)->(a[0] - b[0]));
        int sum = 0;
        for(int i = 0; i < m; i++)
        {
            sum += mat[i][0];
        }
        int[] ary = new int[m + 1];
        ary[0] = sum;

        q.offer(ary);
        String status = Arrays.toString(ary);
        Set<String> set = new HashSet<>();
        set.add(status);
        k--;
        
while(!q.isEmpty() && k > 0)
        {

            int[] cur = q.poll();
            int s = cur[0];
            for(int i = 0; i < m; i++)
            {
                int idx = cur[i + 1];
                if(idx + 1 < mat[i].length)
                {
                    int a = mat[i][idx];
                    int b = mat[i][idx + 1];
                    int s2 = s - a + b;
                    cur[i + 1]++;
                    cur[0] = s2;
                    status = Arrays.toString(cur);
                    if(set.contains(status))
                    {
                        cur[i + 1]--;
                        continue;
                    }
                    set.add(status);
                    q.offer(cur.clone());
                    cur[i + 1]--;
                }
            }
            k--;
        }
        return q.poll()[0];
    }  
}
