class Solution {
    public int kthSmallest(int[][] mat, int k) {
            int m = mat.length;
    int n = mat[0].length;
    List<Integer> sums = new ArrayList<>();
    for (int i = 0; i < n; i++) {
      sums.add(mat[0][i]);
    }
    Collections.sort(sums);
    for (int i = 1; i < m; i++) {
      List<Integer> sums2 = new ArrayList<>();
      for (int j = 0; j < n; j++) {
        for (int v : sums) {
          sums2.add(v + mat[i][j]);
        }
      }
      Collections.sort(sums2);
      sums = sums2.size() > k ? sums2.subList(0, k) : sums2;
    }
    return sums.get(k-1);
    }
}
