class Solution {
    class Entry {
        int sum;
        int[] idx;
    }
    
public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        
        Set<List<Integer>> seen = new HashSet<>();
        
        PriorityQueue<Entry> q = new PriorityQueue<>((e1, e2) -> {
            return e1.sum - e2.sum;
        });
        
        Entry e = new Entry();
        for(int i = 0; i < m; i++) {
            e.sum += mat[i][0];
        }
        e.idx = new int[m];
        q.add(e);
        seen.add(convert(e.idx));
        int ans = 0;
        
        
while(k > 0) {
            e = q.poll();
            ans = e.sum;
            for(int i = 0; i < e.idx.length; i++) {
                if(e.idx[i] == mat[i].length - 1) {
                    continue;
                }
                Entry ne = new Entry();
                ne.sum = e.sum + mat[i][e.idx[i] + 1] - mat[i][e.idx[i]];
                ne.idx = new int[m];
                for(int j = 0; j < ne.idx.length; j++) {
                    ne.idx[j] = e.idx[j];
                }
                ne.idx[i]++;
                List<Integer> list = convert(ne.idx);
                if(!seen.contains(list)) {
                    q.add(ne);
                    
seen.add(list);
                }                            
            }
            k--;
        }
        return ans;
    }
    private List<Integer> convert(int[] idx) {
        List<Integer> list = new ArrayList<>();
        for(int v : idx) {
            list.add(v);
        }
        return list;
    }
}
