class Solution {
    
    // 0,0,0  -- 0 0 1 , 
    
    public int kthSmallest(int[][] mat, int k) {
        
        Set<List<Integer>> visited = new HashSet<>();
        PriorityQueue<Element> q = new PriorityQueue();
        
        int[] startIndex = new int[mat.length];
        int currSum = 0;
        for (int i=0;i<mat.length;i++) {
            currSum+=mat[i][0];
        }
        
        Element startElement = new Element(currSum, startIndex);
        q.add(startElement);
        visited.add(arayToList(startIndex));
        while (k>1) {
            Element top = q.poll();
            k--;
            int topIndexes[] = top.indexes;
            int topSum = top.sum;
            System.out.println(topSum);
            for (int i=0;i<topIndexes.length;i++) {
                System.out.print(topIndexes[i] + " ");
            }
            System.out.println();
            for (int i=0;i<topIndexes.length;i++) {
                int currElementIndex = topIndexes[i];
                // It can be increased
                if (currElementIndex < mat[i].length-1) {
                    int [] newIndexes = Arrays.copyOf(topIndexes, topIndexes.length);
                    newIndexes[i]++;
                    int newSum  = topSum - mat[i][topIndexes[i]];
                    newSum += mat[i][newIndexes[i]];
                    Element newElement = new Element(newSum, newIndexes);
                    
                    List<Integer> newIndexList = arayToList(newIndexes);
                    if (!visited.contains(newIndexList)) {
                        visited.add(newIndexList);
                        q.add(newElement);
                    }
                }
            }
            
            
        }
        return q.poll().sum;
    }
    
    private List<Integer> arayToList(int a[]) {
        List<Integer> a1 = new ArrayList<>();
        
        for (int i=0;i<a.length;i++) {
            a1.add(a[i]);
        }
        return a1;
    }
        
    
    class Element implements Comparable<Element> {
        int[] indexes;
        
        int sum;
        
        public Element(int sum,int[] indexes) {
            this.sum = sum;
            this.indexes = indexes;
        }
        
        @Override
        public int compareTo(Element other) {
            return this.sum - other.sum;
        }
    }
}
