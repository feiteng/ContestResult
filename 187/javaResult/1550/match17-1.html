class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<Points> pq = new PriorityQueue<>(k, minHeap);
        HashSet<String> visited = new HashSet<>();
        int[] initIndex = new int[mat.length];
        int initSum = 0;
        for(int i = 0; i < mat.length; i++){
            initSum += mat[i][0];
        }
        Points p = new Points(initIndex, initSum);
        pq.offer(p);
        visited.add(coordString(p.coord));
        for(int i = 0; i < k-1; i++){
            
            Points pt = pq.poll();
            
            for(int j = 0; j < pt.coord.length; j++){
                int[] curCoord = Arrays.copyOf(pt.coord, pt.coord.length);
                if(curCoord[j] + 1 < mat[j].length) {
                    curCoord[j] ++;
                    
                    if(!visited.contains(coordString(curCoord))){
                        visited.add(coordString(curCoord));
                        
                        Points newPoint = new Points(curCoord, 
findSum(curCoord, mat));
                        
                        pq.add(newPoint);
                    }

                }
                
            }
           
            
        }
        return pq.peek().totalSum;
    }
    
    public int findSum(int[] curCoord, int[][] mat){
        int result = 0;
        for(int i = 0; i < curCoord.length; i++){
            result += mat[i][curCoord[i]];
        }
        return result;
    
}

    public String coordString(int[] coord){
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < coord.length; i++){
            sb.append(coord[i]).append(",");
        }
        return sb.toString();
    }

    private Comparator<Points> minHeap = new Comparator<Points>(){
        public int compare(Points a, Points b){
            return a.totalSum - b.totalSum;
        }
    };
    
    
public class Points{
        int[] coord;
        int totalSum;
        public Points(int[] coord, int totalSum){
            this.coord = coord;
            this.totalSum = totalSum;
        }
        
        public String toString(){
            return Arrays.toString(coord) + " totalSum: " + totalSum;
        }
    }

}
