class Solution {
    public int kthSmallest(int[][] mat, int k) {
        int m = mat.length, n = mat[0].length;
        
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> (a[m]-b[m]));
        HashSet<String> seen = new HashSet<>();
        
        int[] arr = new int[m+1];
        int sum = 0, cur = 0;
        for (int i = 0; i < m; i++) {
            sum += mat[i][0];
        }
        arr[m] = sum;
        pq.offer(arr);
        seen.add(Arrays.toString(arr));
        
        while (k > 0) {
            arr = pq.poll();
            sum = arr[m];   
            //System.out.println("processing: ");
            //System.out.println(Arrays.toString(arr));
            for (int i = 0; i < m; i++) {
                if (arr[i] < (n-1)) {
                    int[] temp = Arrays.copyOf(arr, arr.length);
                    temp[i] += 1;
                    temp[m] += mat[i][arr[i]+1] - mat[i][arr[i]];
                    if (!seen.add(Arrays.toString(temp))) continue;
                    pq.offer(temp);
                    //System.out.println("adding: ");
                    //System.out.println(Arrays.toString(temp));
                
}
            }
            k--;
        }
        
        return sum;
        
    }
}
