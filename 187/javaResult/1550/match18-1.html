class Solution {
	public int kthSmallest(int[][] mat, int k) {
		if (mat == null || mat.length == 0 || mat[0].length == 0) {
			return 0;
		}

		int n = mat[0].length;

			for (int i = 0; i < m; i++) {
				if (idx[i] == n - 1) {
					continue;
				}
				idx[i]++;
				key = formatKey(idx);
				if (visited.add(key)) {
					pq.offer(new Node(val + mat[i][idx[i]] - mat[i][idx[i] - 1], key));
				}

				idx[i]--;
			}
			k--;
		}
		return pq.poll().val;
	}

	private String formatKey(int[] array) {
		StringBuilder sb = new StringBuilder();
		for (int i : array) {
			sb.append(i + ",");
		}
		return sb.toString();
	}

	private int[] reformatKey(String key, int m) {
		int[] idx = new int[m];
		String[] keys = key.split(",");
		for (int i = 0; i < keys.length; i++) {
			idx[i] = Integer.valueOf(keys[i]);
		}
		return idx;
	}
}


class Node {
	int val;
	String key;

	public Node(int val, String key) {
		this.val = val;
		this.key = key;
	}
}
