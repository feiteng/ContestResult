class Solution {
    public int kthSmallest(int[][] mat, int k) {
        Queue<Choice> pq = new PriorityQueue<>((a, b) -> a.sum(mat) - b.sum(mat));
        Choice start = new Choice();
        start.indexes = new int[mat.length];
        pq.offer(start);
        HashSet<String> added = new HashSet<>();
        added.add(
start.toString());
        for (int i = 0; i < k - 1; i++) {
            Choice c = pq.poll();
            for (int j = 0; j < mat.length; j++) {
                if (c.indexes[j] < mat[j].length - 1) {
                    c.indexes[j]++;
                    Choice c2 = new Choice();
                    c2.indexes = Arrays.copyOf(c.indexes, mat.length);
                    c.indexes[j]--;
                    if (!added.contains(c2.toString())) {
                        pq.offer(c2);
                        
added.add(c2.toString());
                    }
                }
            }
        }
        return pq.peek().sum(mat);
    }
}
class Choice {
    int[] indexes;
    public int sum(int[][] mat) {
        int sum = 0;
        for (int i = 0; i < mat.length; i++) {
            sum += mat[i][indexes[i]];
        }
        return sum;
    }
    public String toString() {
        String res = "";
        for (int i : indexes) {
            res += (char)(i + '0');
        }
        return res;
    }
}
