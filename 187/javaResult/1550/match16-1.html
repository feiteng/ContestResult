class Solution {
    public int kthSmallest(int[][] mat, int k) {
        PriorityQueue<Record> pq = new PriorityQueue<>((a,b)->{
            return a.sum-b.sum;
        });
        HashSet<String> visited = new HashSet();
        int[] arr = new int[mat.length];
        Arrays.fill(arr, 0);
        int sum =0;
        for(int[] t : mat) sum+=t[0];
        
pq.add(new Record(arr, sum));
        int currK = 1;
        while(pq.size()!=0){
            Record record = pq.poll();
            StringBuilder sb = new StringBuilder();
            for(int v : record.idxArr) sb.append("_"+v);
            if(visited.contains(sb.toString())) continue;
            visited.add(sb.toString());
            if(currK == k) return record.sum;
            
for(int i=0; i<mat.length; i++){
                if(record.idxArr[i]==mat[i].length-1) continue;
                int[] copiedArray = Arrays.copyOf(record.idxArr, mat.length);
                copiedArray[i]++;
                pq.add(new Record(copiedArray, record.sum+mat[i][copiedArray[i]]-mat[i][copiedArray[i]-1]));
            }
            currK++;
        }
        return 0;
    }
    
    class Record{
        int[] idxArr;
        int sum;
        Record(int[] arr, int sum){
            this.idxArr=arr;
            this.sum=sum;
        }
    }
}
