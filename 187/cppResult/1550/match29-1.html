class Solution {
public:
    vector<int> process(vector<int> c, vector<int> add, int k) {
        vector<int> tmp;
        for (int i = 0; i < add.size(); i++) {
            for (int j = 0; j < c.size(); j++) {
                tmp.push_back(c[j] + add[i]);
            }
        }
        
        sort(tmp.begin(), tmp.end());
        int l = min((int)tmp.size(), k);
        tmp.resize(l);
        
        return tmp;
    }
    
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat.size();
        int m = mat[0].size();
        vector<int> tmp;
        tmp.push_back(0);
        for (int i = n - 1; i >= 0; i--) {
            tmp = process(tmp, mat[i], k);
        }
        
        return tmp[k-1];
    }
};
