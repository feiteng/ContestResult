typedef pair<int,vector<int>> iv;
class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int r=mat.size(), c=mat[0].size();
        priority_queue<iv> pq;
        int tot=0;
        for(int i=0;i<r;i++){
            tot += mat[i][0];
        }
        pq.push({-tot,vector<int>(r,0)});
        multiset<int>score;
        set<vector<int>>vis;
        vis.insert(vector<int>(r,0));
        while(!pq.empty()){
            if(score.size() == k)
                break;
            iv cur=pq.top(); pq.pop();
            int cur_score = cur.first;
            score.insert(-cur_score);
            //cout<<"cur score: "<<cur_score<<" ";
            vector<int>idx= cur.second;
            // for(const auto qq:idx){
            //     cout<<qq<<" ";
            // }cout<<endl;
            for(int i=0;i<r;i++){
                if(idx[i]+1<c){
                    idx[i]++;
                    if(vis.count(idx)) {
                        idx[i]--;
                        continue;
                    }
                    vis.insert(idx);
                    pq.push({cur_score -(mat[i][idx[i]]-mat[i][idx[i]-1]),idx});
                    idx[i]--;
                }
            }
        }
        return *score.rbegin();
    }
};
