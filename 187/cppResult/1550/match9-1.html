class Solution {
public:
    int kthSmallest(vector<vector<int>>& a, int k) {
        int n = a.size(), m = a[0].size();
        //priority_queue<int> 

        vector<int> possible = a[0];
        for(int i = 1; i < n; i++) {
            vector<int> newpossible;
            for(int j = 0; j < possible.size(); j++) {
                for(int k = 0; k < m; k++) {
                    newpossible.push_back(possible[j]+a[i][k]);
                }
            }
            sort(newpossible.begin(), newpossible.end());
            if(newpossible.size() > 200) {
                newpossible.resize(200);
            }
            possible = newpossible;
        }
        return possible[k-1];
    }
};
