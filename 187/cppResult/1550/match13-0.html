class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> data(k, INT_MAX);
        int n = mat.size(), m = mat[0].size();
        for(int i = 0; i < min(k, m); i++){
            data[i] = mat[0][i];
        }
        int len = min(k, m);
        for(int j = 1; j < n; j++){
            vector<int> tmp;
            for(int p = 0; p < len; p++){
                for(int c = 0; c < m; c++){
                    tmp.push_back(data[p]+mat[j][c]);
                }
            }
            sort(tmp.begin(), tmp.end());
            len = min(k, (int)tmp.size());
            for(int i = 0; i < len; i++){
                data[i] = tmp[i];
            }
        }
        return data[k-1];
    }
};
