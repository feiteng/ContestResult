class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> dp;
        int m = mat[0].size(), n = mat.size();
        for (int i=0;i<m;++i) dp.push_back(mat[0][i]);
        while((int)dp.size() > k) dp.pop_back();

        for (int i=1;i<n;++i) {
            vector<int> dq;
            for (int j=0;j<m;++j) {
                for (auto it : dp) {
                    dq.push_back(it + mat[i][j]);
                }
                sort(dq.begin(), dq.end());
                while((int)dq.size() > k) dq.pop_back();
            }
            dp = dq;
        }
        sort(dp.begin(), dp.end());
        return dp[k-1];
    }
};
