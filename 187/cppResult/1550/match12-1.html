class Solution {
public:

    int kthSmallest(vector<vector<int>>& mat, int k) {

        vector<int> cands = mat[0];
    for (int i = 1; i < mat.size(); ++i) {
        vector<int> tmp;
        for (int m = 0; m < cands.size() && m < k; ++m) {
            for (int n = 0; n < mat[i].size(); ++n) {
                tmp.push_back(cands[m]+mat[i][n]);
            }
        }
        sort(tmp.begin(), tmp.end());
        if (tmp.size() > k) tmp.resize(k);
        cands = tmp;
    }
    return cands[k-1];
    }
};
