class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> kSmallest;
        kSmallest.push_back(0);
        
        for (int i = 0; i < mat.size(); i++) {
            vector<int> newOne;
            for (int x : kSmallest) {
                for (int y : mat[i]) {
                    newOne.push_back(x+y);
                }
            }
            
            sort(newOne.begin(),newOne.end());
            kSmallest = newOne;
            while (kSmallest.size() > k) kSmallest.pop_back();
        }
        
        return kSmallest.back();
    }
};
