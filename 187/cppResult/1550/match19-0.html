class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        
        int n=mat.size();
        int m=mat[0].size();
        
        multiset<int> s;
        
        for(auto it:mat[0])
            s.insert(it);
        
        for(int i=1;i<n;i++)
        {
            multiset<int> s1;
            
            for(int j=0;j<m;j++)
            {
                for(auto it:s)
                {
                    s1.insert(mat[i][j]+it);
                }
            }
            
            int cnt=0;
            s.clear();
            
            auto it=s1.begin();
            while(cnt<k && it!=s1.end())
            {
                s.insert(*it);
                it++;
                cnt++;
            }
             s1.clear();   
        }
        
        return *s.rbegin();
    }
};
