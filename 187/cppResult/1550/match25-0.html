class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
    	int n = mat.size();
int m = mat[0].size();
	vector<int>  a;
	int cur = 0;
	for(int i = 0; i<m && i < k; i++) a.push_back(mat[0][i]);
	for(int i = 1; i<n; i++) {
		vector<int> atmp;
		for(auto &v: a) for(auto &b: mat[i]) atmp.push_back(v+b);
		sort(atmp.begin(), atmp.end());
		a.clear();
		for(int j = 0; j<k && j<atmp.size(); j++) a.push_back(atmp[j]);
	}
	return a[k-1];

    }
};
