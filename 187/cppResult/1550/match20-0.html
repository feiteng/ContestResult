class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int i,j,t,m=mat.size(),n=mat[0].size();
        vector<int> arr;
        vector<int> brr;
        for(i=0;i<k && i<n;i++)
            arr.push_back(mat[0][i]);
        for(t=1;t<m;t++) {
            for(i=0;i<k && i<arr.size();i++) {
                for(j=0;j<k && j<n;j++) {
                    brr.push_back(arr[i]+mat[t][j]);
                }
            }
            sort(brr.begin(),brr.end());
            arr.clear();
            for(i=0;i<k && i<brr.size();i++)
                arr.push_back(brr[i]);
            brr.clear();
        }
        return arr[k-1];
    }
};
