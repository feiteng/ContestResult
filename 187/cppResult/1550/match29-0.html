class Solution {
public:
    vector<int> kthSmallest(vector<int> &a, vector<int> &b, int &k) {
        vector<int> ans;
        for(int i = 0; i < a.size(); ++i) {
            for(int j = 0; j < b.size(); ++j) {
                ans.push_back(a[i] + b[j]);
            }
        }
        sort(ans.begin(), ans.end());
        int loop = k > ans.size()? ans.size() :k;
        vector<int> v;
        for(int i = 0; i < loop; ++i) {
            v.push_back(ans[i]);
        }
        return v;
    }
    
    int kthSmallest(vector<vector<int>>& mat, int k) {
        if(mat.size() == 1) return mat[0][k - 1];
        vector<int> t = mat[0];
        for(int i = 1; i < mat.size(); ++i) {
            t = kthSmallest(t, mat[i], k);
        }
        return t[k - 1];
    }
};
