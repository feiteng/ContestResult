class Solution {
public:
    int dp[41][200005];
    int a[585][55];
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat.size();
        int m = mat[0].size();
        int maxx = 0;
        for (int i = 0 ; i < n ; i++){
            for (int j = 0 ; j < m ; j++){
                a[i + 1][j + 1] = mat[i][j];
            }
            maxx += mat[i][m - 1];
        }
        dp[0][0] = 1;

        int sum;
        int rest = k , ans = 0;
        for (int j = 1; j <= maxx ; j++){
            for (int i = 1 ; i <= n ; i++){
                for (int k = 1 ; k <= m ; k++){
                    if (j < a[i][k]) break;
                    dp[i][j] = min(200 , dp[i][j] + dp[i - 1][j - a[i][k]]);
                }
              }
            rest -= dp[n][j];
            if (rest <= 0)
            {
                ans = j;
                break;
            }            
      }
      return ans;
