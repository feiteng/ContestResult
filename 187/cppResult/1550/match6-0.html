class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int K) {
        vector<int> dp;
        dp.push_back(0);
        for(int i = 0; i < mat.size(); i++)
        {
            vector<int> dp2;
            for(int j = 0; j < mat[i].size(); j++)
            {
                for(int k = 0; k < dp.size(); k++)
                {
                    dp2.push_back(dp[k] + mat[i][j]);
                }
            }
            dp.clear();
            sort(dp2.begin(), dp2.end());
            for(int j = 0; j < min(K + 1, (int)dp2.size()); j++)
            {
                dp.push_back(dp2[j]);
            }
        }
        return dp[K - 1];
    }
};
