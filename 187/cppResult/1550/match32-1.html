class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m = mat.size();
        int n = mat[0].size();
        //cout << m << " " << n << endl;
        int maxValue = 0;
        for (auto &v: mat)
            maxValue += v.back();
        //cout << maxValue << endl;
        vector<int> dp(1+maxValue, 0);
        dp[0] = 1;
        
        int curMax = 0;
        int cnt = 0;
        for (int i = 0; i < m; i++) {
            cnt = 0; 

            curMax += mat[i].back();
            vector<int> dp2(1+maxValue, 0);
            //cout << i << " " << curMax << endl;
            for (int j = 0; j <= curMax; j++) {
                for (int k = 0; k < mat[i].size(); k++) {
                    if (j - mat[i][k] >= 0)
                        dp2[j] += dp[j - mat[i][k]];
                    
                }
                cnt += dp2[j];
                if (cnt >= k) {
                    curMax = j;
                    break;
                }
            }
            dp = dp2;
        }
        cnt = 0;
        for (int i = 1; i <= curMax; i++) {
            cnt += dp[i];
            if (cnt >= k)
                return i;
        }
        
        return 0;
    }
};
