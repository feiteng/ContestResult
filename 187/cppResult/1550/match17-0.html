#include <iostream>
#include <sstream>
#include <string>
#include <algorithm>
#include <vector>
#include <utility>
#include <queue>
#include <map>
using namespace std;

typedef long long LL;
typedef unsigned int UINT32;

class Solution {
public:
int kthSmallest(vector<vector<int>>& mat, int k) {
    int m = mat.size();
    int n = mat[0].size();

    vector<int> res1 = mat[m-1];
    if (res1.size() > k) {
        res1.resize(k);
    }

    vector<int> res2;
    for (int i = m-2; i >= 0; --i) {
        res2.clear();

        for (int j = 0; j < n; ++j) {
            for (int l = 0; l < res1.size(); ++l) {
                res2.push_back(res1[l] + mat[i][j]);
            }
        }

        sort(res2.begin(), res2.end());
        if (res2.size() > k) {
            res2.resize(k);
        }

        res1.swap(res2);
    }

    return res1[k-1];
}
};
