class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> v = mat[0];

        vector<int>& sums = v;
        for (int i = 1; i < mat.size(); i++) {
            vector<int> sums1;
            for (int j = 0; j < sums.size(); j++) {
                for (int k = 0; k < mat[i].size(); k++) {
                    sums1.push_back(mat[i][k] + sums[j]);
                }
            }
            sort(sums1.begin(), sums1.end());
            if (sums1.size() > k) {
                sums1.erase(sums1.begin()+k, sums1.end());
            }
            sums = sums1;
        }

        return sums[k-1];
    }
};
