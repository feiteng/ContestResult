class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n=mat.size();
        int m=mat[0].size();
        vector<int>ksmallest={0},next;
        

        for(int i=0;i<n;i++){
            next.clear();
            for(int j=0;j<m;j++){
                for(int k=0;k<(int)ksmallest.size();k++){
                    next.push_back(ksmallest[k]+mat[i][j]); 
                }
            }
            sort(next.begin(),next.end());
            ksmallest.clear();
            for(int j=0;j<min(200,(int)next.size());j++){
                ksmallest.push_back(next[j]);
            }
        }
        return ksmallest[k-1];
    }
};
