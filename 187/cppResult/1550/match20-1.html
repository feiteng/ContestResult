class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int kth) {
        int n,m,i,j,k,x;
        n=mat.size();
        m=mat[0].size();

        vector<int> v,temp;
        for(i=0;i<m;i++)v.push_back(mat[0][i]);
        for(i=1;i<n;i++){
            
            for(j=0;j<v.size();j++){
                for(k=0;k<m;k++){
                    temp.push_back(v[j]+mat[i][k]);
                }
            }
            sort(temp.begin(),temp.end());
            v.clear();
            for(j=0;j<min(kth,(int)temp.size());j++){
                //cout<<temp[j]<<" ";
                v.push_back(temp[j]);
            }
            //cout<<endl;
            temp.clear();
        }
        return v[kth-1];
        
        
    }
};
