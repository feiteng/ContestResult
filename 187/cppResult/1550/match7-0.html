class Solution {
public:
    int kthSmallest(vector<vector<int>>& ar, const int k) {
        if (ar.empty()) {
            return 0;
        }
        const int m = ar.size();
        const int n = ar.back().size();
        vector<int> kr = ar[0];
        
        for(int iz = 1; iz < m; ++iz) {
            multiset<int> st;
            for(int i = 0; i < kr.size(); ++i) {
                for(int j = 0; j < n; ++j) {
                    // printf("Adding: %d\n", kr[i] + ar[iz][j]);
                    st.insert(kr[i] + ar[iz][j]);
                }
            }
            kr.resize(min(k, (int)st.size()));
            int i = 0;
            for(int x : st) {
                kr[i++] = x;
                if (i == k) {
                    break;
                }
            }
            // swap(zz, kr);
            // for(auto x : kr) cout << x << ", "; cout << endl;
        }
        
        return kr.back();
    }
};
