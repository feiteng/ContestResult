class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        vector<int> res, tmp;
        res.reserve(8000), tmp.reserve(8000);
        res.push_back(0);
        int n = mat.size(), m = mat.front().size();
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                for (int x = 0, y = res.size(); x < y; ++x) {
                    tmp.push_back(res[x] + mat[i][j]);
                }
            }
            sort(tmp.begin(), tmp.end());
            tmp.erase(tmp.begin() + min((int)tmp.size(), k), tmp.end());
            tmp.swap(res);
            tmp.clear();
        }
        return res.back();
    }
};
