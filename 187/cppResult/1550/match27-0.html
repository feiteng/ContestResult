class Solution {
    int n, m;
    vector<vector<int>> a;

public:
    int kthSmallest(vector<vector<int>>& mat, int K) {
        n = mat.size(), m = mat[0].size();
        a = mat;
        // dp[n][k] = kth smallest sum considering the first n rows
        vector<vector<int>> dp(n+1, vector<int>(K, 1000*1000));
        dp[0][0] = 0;
        for(int i = 0; i < n; i++) {
            vector<int> now;
            for(int j = 0; j < m; j++) {
                for(int k = 0; k < K; k++) {
                    now.push_back(dp[i][k]+a[i][j]);
                }
            }
            sort(now.begin(), now.end());
            for(int k = 0; k < K; k++) {
                dp[i+1][k] = now[k];
            }
            
        }
        
        return dp[n][K-1];
    }
};
