class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int m = mat.size();
        int n = mat[0].size();
        
        set<pair<int, vector<int>>> PQ;
        vector<int> dummy;
        int sum = 0;
        for(int i = 0; i < m; i++)
        {
            sum += mat[i][0];
            dummy.push_back(0);
        }
        
        PQ.insert({sum,dummy});
        for(int t = 1; t <= k; t++)
        {
            sum = PQ.begin()->first;
            vector<int> ptr = PQ.begin()->second;
            PQ.erase(PQ.begin());
            //cout << "sum = " << sum << endl;
            
            if(t == k)
            {
                return sum;
            }
            

            for(int i = 0; i < m; i++)
            {
                int f = ptr[i] + 1;
                if(f < n)
                {
                    int sneed = sum - mat[i][ptr[i]] + mat[i][f];
                    //cout << "sneed = " << sneed << endl;
                    ptr[i]++;
                    PQ.insert({sneed, ptr});
                    ptr[i]--;
                }
            }
        }
        
        return 0;
    }
};
