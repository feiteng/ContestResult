class Solution {
public:
    int kthSmallest(vector<vector<int>> &mat, int k) {

        int res[220], len = 0;
        for (int i = 0; i < mat[0].size() && len < k; ++i) {
            res[len++] = mat[0][i];
        }
        for (int i = 1; i < mat.size(); ++i) {
            priority_queue<int, vector<int>, greater<int> > q;
            for (int x = 0; x < len; ++x) {
                for (int y = 0; y < mat[i].size(); ++y) {
                    q.push(res[x] + mat[i][y]);
                }
            }
            len = 0;
            while (len < k && !q.empty()) {
                res[len++] = q.top();
                q.pop();
            }
            //for(int i=0;i<len;++i)cout<<res[i]<<' ';cout<<"len="<<len<<endl;
        }
        return res[k - 1];
    }
};
