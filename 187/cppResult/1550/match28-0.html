class Solution {
public:
    int kthSmallest(vector<vector<int>>& mat, int k) {
        int n = mat.size();
        int m = mat[0].size();
        priority_queue< pair<int, vector<int> > > PQ;
        
        set< vector<int> > done;
        
        int initsum =0;
        for(auto &x:mat)
            initsum += x[0];
        
        int ans;
        PQ.push( make_pair( -initsum , vector<int> (n,0) ));
        for(int i=0;i<k;++i){
            auto x = PQ.top(); PQ.pop();
            int curSum = -x.first;
            vector<int> &indexes = x.second;
            
            ans = curSum;
            int tsum =0, pos=0;
            for(int i:indexes)
                tsum += mat[pos++][i];
            for(int i=0;i<n;++i){
                int p = indexes[i];
                if( p + 1< m ){
                      int sum = curSum - mat[i][p] + mat[i][p+1];
                      ++indexes[i];
                      if( done.count(indexes) == 0 ){
                          done.insert(indexes);
                          PQ.push( make_pair(-sum, indexes) );
                      }
                      --indexes[i];  
                }
            }
        }
        return ans;
    }
};
