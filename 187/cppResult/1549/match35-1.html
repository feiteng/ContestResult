class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> s;
        int N = nums.size();
        int ans = 0;
        int pre = 0;
        for (int i=0; i<N; i++) {
            s.insert(nums[i]);
            auto r = s.rbegin();
            auto l = s.begin();
            while ((*r - *l) > limit) {
                auto it = s.find(nums[pre]);
                s.erase(it);
                pre++;
                l = s.begin();
                r = s.rbegin();
                //cout << "===>" << *l << " " << *r << endl;
            }
            //cout << i << " " << pre << endl;
            ans = max(ans, i-pre+1);
        }
        return ans;
    }
};
