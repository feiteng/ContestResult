class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int max_ans = 0;
        multiset<int> ms;
        int n = nums.size();
        int st = 0;
        for(int j=0; j<n; ++j){
            ms.insert(nums[j]);
            while( *prev(ms.end()) - *ms.begin() > limit ){
                ms.erase(ms.find(nums[st]));
                st++;
            } 
            max_ans = max(max_ans, j - st + 1 );
        }
        return max_ans;
        
    }
};
