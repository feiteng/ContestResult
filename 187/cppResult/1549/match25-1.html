class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int mx = 0;
        
        multiset<int> buf;
        int j = 0;
        for(int i = 0 ; i < nums.size(); ++i){
            buf.insert(nums[i]);
            
            for(;buf.size() > 0 && *prev(buf.end()) - *buf.begin() > limit; ++j){
                buf.erase(buf.find(nums[j]));
            }
            mx = max(mx, (int)buf.size());
        }
        
        return mx;
    }
};
