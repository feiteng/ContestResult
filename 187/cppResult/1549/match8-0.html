class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        
        deque<int> ql, qs;
        int res = 1;
        
        for (int i = 0, j = 0; i < nums.size(); ++i) {
            
            while (!ql.empty() && nums[ql.back()] < nums[i]) ql.pop_back();
            ql.push_back(i);
            while (!qs.empty() && nums[qs.back()] > nums[i]) qs.pop_back();
            qs.push_back(i);
            
            while (nums[ql.front()] - nums[qs.front()] > limit) {
                ++j;
                while (ql.front() < j) ql.pop_front();
                while (qs.front() < j) qs.pop_front();
            }
            
            res = max(i - j + 1, res);
        }
        
        return res;
    }
};
