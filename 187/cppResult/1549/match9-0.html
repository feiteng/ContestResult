class Solution {
public:
  int longestSubarray(vector<int>& nums, int limit) {
    int n = nums.size();
    multiset<int> a;
    int ret = 0;
    for (int i = 0, j = -1; i < n; ++i) {
      a.insert(nums[i]);
      while (!a.empty() && *a.rbegin() - *a.begin() > limit) {
        ++j;
        a.erase(a.lower_bound(nums[j]));
      }
      ret = max(ret, i - j);
    }
    return ret;
  }
};
