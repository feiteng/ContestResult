class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> s;
        int l = 0;
        s.insert(nums[0]);
        int ans = 1;
        for(int i = 1; i < nums.size(); i++){
            s.insert(nums[i]);
            while(1){
                auto it = s.end();
                it--;
                if(*it - *(s.begin()) > limit){
                    auto e = s.find(nums[l]);
                    s.erase(e);
                    l++;
                }
                else{
                    break;
                }
            }
            ans = max(ans, i-l+1);
        }
        return ans;
    }
};
