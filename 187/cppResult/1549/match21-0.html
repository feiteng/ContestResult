class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        deque<int> max_index, min_index;
        int j=0, res=0;
        for(int i=0; i<nums.size(); i++){
            
            while(!max_index.empty()&&nums[max_index.back()]<nums[i])
                max_index.pop_back();
            max_index.push_back(i);
            
            while(!min_index.empty()&&nums[min_index.back()]>nums[i])
                min_index.pop_back();
            min_index.push_back(i);
            
            while(!max_index.empty()&&!min_index.empty()&&nums[max_index.front()]-nums[min_index.front()]>limit){
                if(j==max_index.front())
                    max_index.pop_front();
                if(j==min_index.front())
                    min_index.pop_front();
                j++;
            }
            
            res=max(res, i-j+1);
        }
        return res;
    }
};
