class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> st;
        int ans = 0,l = 0, r = 0,n = nums.size();
        while(l < n) {
            while(r < n){
                st.insert(nums[r]);
                if(*st.rbegin() - *st.begin() > limit){
                    st.erase(st.find(nums[r]));
                    break;
                }
                r++;
            }
            ans = max(ans, r - l);
            st.erase(st.find(nums[l]));
            l++;
        }
        
        return ans;
    }
};
