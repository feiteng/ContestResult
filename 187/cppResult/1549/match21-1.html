class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> s;
        int ret = 0, start = 0;
        for (int i = 0; i < nums.size(); ++i) {
            s.insert(nums[i]);
            while (!s.empty() && *s.rbegin() - *s.begin() > limit) {
                s.erase(s.equal_range(nums[start]).first);
                ++start;
            }
            ret = max(ret, (int)s.size());
        }
        return ret;
    }
};
