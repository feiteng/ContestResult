class Solution {
public:
    int longestSubarray(vector<int>& a, int d) {
        int n=a.size();
        multiset<int> s;
        int res=1;
        int pre=0;
        for(int i=0;i<n;i++) {
            s.insert(a[i]);
            while(*(prev(s.end()))-*s.begin()>d) {
                s.erase(s.find(a[pre++]));
            }
            res=max(res,(int)s.size());
        }
        return res;
    }
};
