// desc, top == max
struct MonotoneQueue1 {
    std::deque<int> q, m;

    void push(const int &x) {
        q.push_back(x);
        while (!m.empty() && m.back() < x) m.pop_back();
        m.push_back(x); 
    }

    void pop() {
        int x = q.front();
        q.pop_front();
        if (x == m.front()) m.pop_front();
    }

    size_t size() {
        return q.size();
    }

    int top() {
        return m.front();
    }
};

// asc, top == min
struct MonotoneQueue2 {
    std::deque<int> q, m;

    void push(const int &x) {
        q.push_back(x);
        while (!m.empty() && m.back() > x) m.pop_back();
        m.push_back(x); 
    }

    void pop() {
        int x = q.front();
        q.pop_front();
        if (x == m.front()) m.pop_front();
    }

    size_t size() {
        return q.size();
    }

    int top() {
        return m.front();
    }
};


class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        MonotoneQueue1 q_max;
        MonotoneQueue2 q_min;
        int best = 0;
        int n = (int)nums.size();
        for (int i = 0; i < n; ++i) {
            q_max.push(nums[i]);
            q_min.push(nums[i]);
            while ((q_max.size() > 0) && q_max.top() - q_min.top() > limit) {
                q_max.pop();
                q_min.pop();
            }
            best = max(best, (int)q_max.size());
        }
        return best;
    }
};
