class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int>ms;
        int i=0,j=0,ans=0,n=nums.size();
        while(j<n){
            ms.insert(nums[j]);
            while(*ms.rbegin()-*ms.begin()>limit){
                ms.erase(ms.find(nums[i]));
                ++i;
            }
            ++j;
            ans=max(ans,j-i);
        }
        return ans;
    }
};
