class Solution {
public:
    int longestSubarray(vector<int>& nums, int k) {
        int ret = 0;
        int i = 0;
        int j = 0;

        int maxVal = 0;
        int minVal = INT_MAX;
        int n = nums.size();
        map <int, int> m;
        for(int i = 0; i < n; i++){
            m[nums[i]]++;
            
            while(abs((m.rbegin()->first) - (m.begin()->first)) > k){
                m[nums[j]]--;
                
                if(m[nums[j]] == 0) m.erase(nums[j]);
                
                j++;
            }
            ret = max(ret, i -j + 1);
            //cout << ret << endl;
        }
        return ret;
    }
};
