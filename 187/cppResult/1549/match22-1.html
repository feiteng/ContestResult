class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> s;
        int n = nums.size();
        int r = -1;
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (i != 0) {
                s.erase(s.find(nums[i - 1]));
            }
            while (r + 1 < n && (s.empty() || max(*s.rbegin(), nums[r + 1]) - min(*s.begin(), nums[r + 1]) <= limit)) {
                s.insert(nums[r + 1]);
                ++r;
            }
            // cout << i << " " << r << " " << s.size() << "\n";
            ans = max(ans, r - i + 1);
        }
        return ans;
    }
};
