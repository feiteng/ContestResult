class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
      multiset<int> memo;
      //cout << "----" << endl;
      
      // two-pointer maintain a window, for each step we 
      // need to know the max/min in the window
      int longest = 0;
      
      for (int i = 0, j = 0; i < nums.size(); i++) {
        while (j < nums.size()) {
          memo.insert(nums[j++]);
          
          if (*memo.rbegin() - *memo.begin() <= limit) {
            longest = max(longest, j - i);
            //cout << "i: " << i << ", j: " << j << ", len: " << j - i << ", longest: " << longest << endl;
          } else break;
        }
        
        memo.erase(memo.equal_range(nums[i]).first);
      }
      
      return longest;
    }
};
