class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        deque<int> qMin, qMax;
        int ans = 0;
        int j = 0;
        for (int i = 0; i < nums.size(); ++i) {
            while (qMin.size() && nums[i] < nums[qMin.back()])
                qMin.pop_back();
            while (qMax.size() && nums[i] > nums[qMax.back()])
                qMax.pop_back();
            qMin.push_back(i);
            qMax.push_back(i);
            while (nums[qMax.front()] - nums[qMin.front()] > limit) {
                j++;
                while (qMin.size() && qMin.front() < j)
                    qMin.pop_front();
                while (qMax.size() && qMax.front() < j)
                    qMax.pop_front();
            }
            ans = max(ans, i - j + 1);
        }
        return ans;
    }
};
