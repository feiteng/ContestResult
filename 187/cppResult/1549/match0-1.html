class Solution {
 public:
  int longestSubarray(vector<int>& nums, int limit) {
    multiset<int> st;
    int ans = 0;
    for (int left = 0, right = 0; right < (int)nums.size(); ++right) {
      st.insert(nums[right]);
      while (*st.rbegin() - *st.begin() > limit) {
        st.erase(st.find(nums[left++]));
      }
      ans = max(ans, right - left + 1);
    }
    return ans;
  }
};
