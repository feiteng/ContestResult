class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int n = nums.size();

        deque<int> m1, m2;
        
        int ans = 0;
        for (int i = 0, j = 0; i < n; i++) {
            while (!m1.empty() && nums[i] >= nums[m1.back()])
                m1.pop_back();
            
            m1.push_back(i);
            
            while (!m2.empty() && nums[i] <= nums[m2.back()])
                m2.pop_back();
            
            m2.push_back(i);
            
            while (!m1.empty() && !m2.empty() && nums[m1.front()] - nums[m2.front()] > limit) {
                j++;
                while (!m1.empty() && m1.front() < j)
                    m1.pop_front();
                
                while (!m2.empty() && m2.front() < j)
                    m2.pop_front();
            }
            
            ans = max(ans, i - j + 1);
        }
        
        return ans;
    }
};
