class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        const int N = size(nums);
        int left = 0;
        int ans = 1;
        map<int,int> m;
        for (int i = 0; i < N; ++i) {
            ++m[nums[i]];
            while (m.rbegin()->first - m.begin()->first > limit) {
                --m[nums[left]];
                if (m[nums[left]] == 0) m.erase(nums[left]);
                ++left;
            }
            ans = max(ans, i - left + 1);
        }
        return ans;
    }
};
