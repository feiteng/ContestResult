class Solution {
public:
  int canAdd(const multiset<int>& s, int x, int limit) {
    if (s.empty()) return true;
    
    int min_value = *s.begin();
    int max_value = *prev(s.end());
    
    assert(max_value - min_value <= limit);

    return abs(x - max_value) <= limit && abs(x - min_value) <= limit;
  }
    
    
  int longestSubarray(vector<int>& nums, int limit) {
    multiset<int> s;
    
    int res = 0, j = -1, n = nums.size();
    
    for (int i = 0; i < n; ++i) {
      while (j + 1 < n && canAdd(s, nums[j + 1], limit)) {
        s.insert(nums[j + 1]);
        ++j;
      }
      
      res = max(res, j - i + 1);
      s.erase(s.find(nums[i]));
    }
    
    return res;
  }
};
