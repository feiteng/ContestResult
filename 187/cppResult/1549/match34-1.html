class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {

        int max_len = 0, j = 0;
        map<int, int> count;
        for (int i=0;i<nums.size();i++) {
            count[nums[i]]++;
            while (!count.empty() && (((count.rbegin())->first) - ((count.begin())->first)) > limit) {
                count[nums[j]]--;
                if (count[nums[j]] == 0)
                    count.erase(nums[j]);
                j++;
            }
            max_len = max(max_len, i-j + 1);
        }
        return max_len;
    }
};
