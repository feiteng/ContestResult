class Solution {
public:

    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> w;
        int r=0, j=0, n=nums.size();
        for (int i=0; i<n; ++i) {
            w.insert(nums[i]);
            while (!w.empty() && *w.rbegin() - *w.begin() > limit) {
                w.erase(w.find(nums[j++]));
            }
            r = max(r, 1+i-j);
        }
        return r;
    }
};
