#include <set>

class Solution {
public:
    bool good(int a,int b,int l)
    {
        return (a>=b-l)&&(a<=b+l);
    }
    
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> S;
        int ans=0,i=0,j=0;
        for(int i=0;i<nums.size();i++)
        {
            S.insert(nums[i]);
            while(!good(*S.begin(),*S.rbegin(),limit))
            {
                S.erase(S.find(nums[j++]));
            }
            ans=max(ans,i-j+1);
        }
        return ans;
    }
};
