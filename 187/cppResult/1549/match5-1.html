class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int last = 0;
        int res = 0;
        multiset<int> s;
        
        for (int i = 0; i < nums.size(); ++i) {
            s.insert(nums[i]);
            while (*s.rbegin() - *s.begin() > limit) {
                s.erase(nums[last++]);
            }
            res = max<int>(res, s.size());
        }
        
        return res;
    }
};
