class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> s;
        int res = 1;
        int l = 0;
        for (int r = 0; r<nums.size(); ++r) {
            s.insert(nums[r]);
            //cout<<*s.begin()<<" "<<*s.rbegin()<<endl;
            while (*s.rbegin() - *s.begin() > limit) {
                int left = nums[l];
                s.erase(s.find(left));
                ++l;
            }
            
            res = max(res, r-l+1);
        }
        
        return res;
    }
};
