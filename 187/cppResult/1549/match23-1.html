class Solution {
public:
    int longestSubarray(vector<int>& a, int limit) {
        int n = a.size();    
        int ans = 0;
        multiset<int> s;
        for (int i = 0, j = 0; i < n; i++) {
            for (j = max(i, j); j < n; j++) {
                if (s.empty() || abs(min(a[j], *s.begin()) - max(a[j], *s.rbegin())) <= limit) {
                    s.emplace(a[j]);
                    ans = max(ans, (int)s.size());
                } else {
                    break;
                }
            }
            s.erase(a[i]);
        }
        return ans;
    }
};
