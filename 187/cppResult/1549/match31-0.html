class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        int l = 0;
        int ans = 0;
        multiset<int> set;
        for(int r = 0; r < nums.size(); r++) {
            set.insert(nums[r]);
            while(*set.rbegin() - *set.begin() > limit) {
                set.erase(set.find(nums[l++]));
            }
            ans = max(ans, r - l + 1);
        }
        
        return ans;
    }
};
