class Solution {
public:
    int longestSubarray(vector<int>& nums, int limit) {
        multiset<int> cur;
        int ans = 0;
        int start = 0;
        for (int i=0; i<nums.size(); i++) {
            cur.insert(nums[i]);
            while (start < i && *cur.rbegin() - *cur.begin() > limit) {
                cur.erase(cur.find(nums[start++]));
            }
            
            ans = max(ans, i - start + 1);
        }
        
        return ans;
    }
};
