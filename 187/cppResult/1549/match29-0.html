class Solution {
public:
    int longestSubarray(vector<int>& a, int limit) {
        multiset<int> s;
        int j = 0;
        int ans = 0;
        for(int i = 0; i < a.size(); i++) {
            s.insert(a[i]);
            while(true) {
            auto it1 = s.begin();
            auto it2 = s.end();
            --it2;
            if (*it2 - *it1 <= limit) break;
            s.erase(s.find(a[j++]));    
            }
            ans = max(ans, (int)s.size());
        }
        return ans;
    }
};
