#include <iostream>

static const int MOD = 1000000007;

template<class T>
constexpr T INF = ::numeric_limits<T>::max() / 32 * 15 + 208;

// class

class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        map<string, int> cnt;
        for (auto &&i : paths) {
            cnt[i[0]]++;
            cnt[i[1]]--;
        }
        for (auto &&j : cnt) {
            if(j.second < 0) return j.first;
        }
        return "";
    }
};
