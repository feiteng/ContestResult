class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string>st, st2;
        for(auto& v : paths){
            st2.insert(v[0]);
            st2.insert(v[1]);
            st.insert(v[0]);
        }
        for(auto x : st2){
            if(st.find(x)==st.end())return x;
        }
        return "";
    }
};
