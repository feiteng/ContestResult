class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> S;
        for(auto s:paths)
            S.insert(s[0]);
        for(auto s:paths)
           if(S.count(s[1]) == 0 )
               return s[1];
        return "";
        
    }
};
