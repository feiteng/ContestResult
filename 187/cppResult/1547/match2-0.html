class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
      set<string> a, b;
      for (auto p : paths) {
        a.insert(p[0]);
        b.insert(p[1]);
      }

      for (auto p : b) {
        if (a.find(p) == a.end()) {
          return p;
        }
      }
      return "";
    }
};
