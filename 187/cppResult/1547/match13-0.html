class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> st;
        for(auto& pa : paths) {
            st.insert(pa[0]);
        }
        for(auto& pa : paths) {
            if(st.count(pa[1]) == 0) return pa[1];
        }
        return "";
        
    }
};
