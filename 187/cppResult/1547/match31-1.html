class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        
        set<string> st1,st2;
        
        for(vector<string> v : paths)
        {
            st1.insert(v[0]);
            st2.insert(v[1]);
        }
        
        for(string x : st2)
        {
            if(st1.count(x)==0)
                return x;
        }
        
        return "";
        
    }
};
