class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> s, d;
        for(auto p: paths) {
            s.insert(p[0]);
            d.insert(p[1]);
        }
        for(auto c: d) {
            if(s.find(c) == s.end())
                return c;
        }
        return "";
    }
};
