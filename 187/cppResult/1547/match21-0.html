class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        set<string>s1;
        set<string>s2;
        for (auto x:paths){
            s2.insert(x[0]);
            s1.insert(x[1]);
        }
        for (auto x:s1){
            if (s2.find(x)==s2.end()) return x;
        }
        return "";
    }
};
