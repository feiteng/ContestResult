class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        set<string> l,r;
        for(auto& path:paths){
            l.insert(path[0]);
            r.insert(path[1]);
        }
        for(auto& dst:r){
            if(l.find(dst)==l.end()){
                return dst;
            }
        }
        return "";
    }
};
