class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> mark;
        for (const auto& p : paths)
        {
            mark.insert(p[0]);
        }
        for (const auto& p : paths)
        {
            if (mark.find(p[1]) == mark.end())
            {
                return p[1];
            }
        }
        return "";
    }
};
