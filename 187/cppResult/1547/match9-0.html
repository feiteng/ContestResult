class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> ui;
        unordered_set<string> uo;
        for(auto path: paths) {
            uo.insert(path[0]);
            ui.insert(path[1]);
        }
        for(auto city:ui) {
            if(uo.count(city) == 0) return city;
        }
        return "";
    }
};
