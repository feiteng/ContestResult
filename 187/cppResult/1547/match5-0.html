class Solution {
public:
    string destCity(vector<vector<string>>& p) {
        map < string, int > c;
        for (auto it : p) {
            c[it[0]]++;
            c[it[1]];
        }
        
        for (auto it : c) if (it.second == 0) return it.first;
        return "WAT";
    }
};
