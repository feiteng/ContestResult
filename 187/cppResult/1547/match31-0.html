class Solution {
public:
    string destCity(vector<vector<string>>& p) {
        
        set<string> cs;
        FOR(i,0,sz(p)){
            cs.ins(p[i][0]);
        }
        
        FOR(i,0,sz(p)){
            if(nex(cs,p[i][1])) return p[i][1];
        }
        
        return "";
    }
};
