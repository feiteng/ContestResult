class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        set<string> s1;
        set<string> s2;
        for(auto k : paths){
            s1.insert(k[0]);
            s2.insert(k[1]);
        }
        for(auto k : s2){
            if(s1.find(k)==s1.end()) return k;
        }
        return "";
    }
};
