class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        //
        
        set<string> M;
        for(auto lis : paths){
            
                M.insert(lis[0]);
            
        }
        //
        for(auto lis : paths){
          
                if(!M.count(lis[1])) return lis[1];
             
        }
        return " ";
    }
};
