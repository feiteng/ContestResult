class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        unordered_set<string> all;
        unordered_set<string> out;
        for (auto v: paths) {
            all.insert(v[0]);
            out.insert(v[1]);
        }
        for (string o: out) {
            if (all.find(o) == all.end()) {
                return o;
            }
        }
        return "";
    }
};
