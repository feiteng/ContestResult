ll MOD(ll a, ll m) {

ll inverse(ll a, ll m) {
    a = MOD(a, m);
    if(a <= 1)return a;
    return MOD((1 - inverse(m, a) * m) / a, m);
}

ll gcd(ll a, ll b) {
    while(b != 0) {
        a %= b;
        swap(a, b);
    }
    return abs(a);
}

ll fast(ll a, ll b, ll mod) {
    if(b < 0)a = inverse(a, mod), b = -b;
    ll ans = 1;
    while(b) {
        if(b & 1) {
            b--;
            ans = ans * a % mod;
        } else {
            a = a * a % mod;
            b /= 2;
        }
    }
    return ans % mod;
}






class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        map<string,string>mp;
        vector<string>v;
        for(auto i:paths){
            v.PB(i[0]);
            v.PB(i[1]);
            mp[i[0]] = i[1];
        }
        for(auto i:v)if(mp.count(i)==0)return i;
        return "";
    }
};
