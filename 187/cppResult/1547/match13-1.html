class Solution {
public:
    string destCity(vector<vector<string>>& paths) {
        set<string> A;
        for (auto& p : paths) {
            A.insert(p[0]);
        }
        for (auto& p : paths) {
            if (!A.count(p[1])) return p[1];
        }
        return "";
    }
};
