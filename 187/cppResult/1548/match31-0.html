class Solution {
public:
	bool kLengthApart(vector<int>& nums, int k) {
		int last = -1e9;
		for (int i = 0; i < sz(nums); i++) {
			if (!nums[i]) continue;
			if (i - last - 1 < k) return false;
			last = i;
		}
		return true;
	}
};
