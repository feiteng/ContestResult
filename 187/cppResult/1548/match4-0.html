class Solution {
public:
    bool kLengthApart(vector<int>& a, int k) {
        int last=-1e9;
        for(int i=0;i<a.size();i++){
            if(!a[i])
                continue;
            if(i-last-1<k)
                return false;
            last=i;
        }
        return true;
    }
};
