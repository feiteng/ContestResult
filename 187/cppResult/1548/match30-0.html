const ll MOD=1e9+7;
class Solution {
public:
  bool kLengthApart(vector<int>& nums, int k) {
    int lst=-1;
    for(int i=0;i<nums.size();i++){
      if(nums[i]==0) continue;
      if (lst!=-1 && i-lst<=k) return 0;
      lst=i;
    }
    return 1;
  }
};
