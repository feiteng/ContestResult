class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        pq = []
        visited = set()
        a = [0 for row in mat]
        m = len(mat)
        n = len(mat[0])
        heapq.heappush(pq, (sum([row[0] for row in mat]), a))
        # print(pq[0])
        for i in range(k - 1):
            s, d = pq[0]
            # print(s, d)
            for i in range(m):
                if d[i] < n - 1:
                    newd = [x for x in d]
                    newd[i] += 1
                    key = str(newd)
                    if key in visited:
                        continue
                    visited.add(key)
                    news = s - mat[i][d[i]] + mat[i][newd[i]]
                    heapq.heappush(pq, (news, newd))
            heapq.heappop(pq)
            
        return pq[0][0]
