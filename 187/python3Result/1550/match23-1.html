import heapq
class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        h = []
        s = set()
        n = len(mat[0])
        heapq.heappush(h,(sum(x[0] for x in mat),[0 for x in mat]))
        cur = 0
        while cur < k:
            #print(h)
            cur_sum,nums = heapq.heappop(h)
            cur += 1
            if cur == k:
                return cur_sum
            for i in range(len(nums)):
                arr = nums.copy
()
                if arr[i] + 1 < n:
                    arr[i] += 1
                    t = tuple(arr)
                    if t not in s:
                        new_sum = cur_sum + mat[i][arr[i]] - mat[i][arr[i]-1]
                        heapq.heappush(h,(new_sum,arr))
                        s.add(t)
