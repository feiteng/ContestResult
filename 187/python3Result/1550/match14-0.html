class Solution:
    def kthSmallest(self, mat, k: int) -> int:
        n, m = len(mat), len(mat[0])

        dp = mat[0]

        for i in range(1, n):
            tmpsums = []
            for j in range(m):
                for x in dp:
                    tmpsums.append(x + mat[i][j])

            tmpsums.sort()
            if len(tmpsums) >= 200:
                tmpsums = tmpsums[:200]

            dp = tmpsums
            #print(dp)

        return dp[k - 1]
