class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        f = mat[0]
        for i in range(1, m):
            row, temp = mat[i], []
            for j in range(len(f)):
                for l in range(n):
                    temp.append(f[j] + row[l])
            f = sorted(temp)[:min(k, len(temp))]
        return f[k - 1]
