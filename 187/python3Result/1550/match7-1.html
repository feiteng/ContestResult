class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        temp = mat[0]
        for r in mat[1:]:
            t 
= []
            for c in r:
                for x in temp:
                    t.append(x + c)
            temp = sorted(t)[:k]
        return temp[-1]
