import heapq
class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m,n = len(mat), len(mat[0])
        visited
=set()
        
        heap=[]
        status=[0 for i in range(m)]
        s=sum(mat[i][0] for i in range(m))
        visited.add(tuple(status))
        heapq.heappush(heap, [s, status])
        while heap and k:
            k-=1
            s, status = heapq.heappop(heap)
            if k==0:
                return s
            for i in range(m):
                if status[i]+1<n:
                    newstatus=status[:]
                    newstatus[i]+=1
                    ts=tuple(newstatus)
                    if ts not in visited:
                        visited
.add(ts)
                        newsum=s-mat[i][status[i]]+mat[i][status[i]+1]
                        heapq.heappush(heap, [newsum, newstatus])
