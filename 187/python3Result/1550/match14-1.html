import heapq
class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n 
= len(mat), len(mat[0])
        h = mat[0]
        for i in range(1, m):
            hn = []
            for j in range(n):
                for e in h:
                    hn.append(e + mat[i][j])
            heapq.heapify(hn)
            h = []
            cnt = 0
            while hn and cnt < k:
                h.append(heapq.heappop(hn))
                cnt += 1
        return h[k - 1]
