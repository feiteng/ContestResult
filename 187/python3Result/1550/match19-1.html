class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        
        if m == 1:
            return mat[0][k-1]
        
        pairs = []
        nums1 = mat[0]
        nums2 = mat[1]
        
        queue 
= []
        def push(i, j):
            if i < len(nums1) and j < len(nums2):
                heapq.heappush(queue, [nums1[i] + nums2[j], i, j])
        
        push(0, 0)
        
while queue and len(pairs) < k:
            _, i, j = heapq.heappop(queue)
            pairs.append(nums1[i] + nums2[j])
            push(i, j + 1)
            if j == 0:
                push(i + 1, 0)
        
        for i in range(2, m):
            nums1 = pairs
            pairs = []
            nums2 = mat[i]
            queue = []
    
            push(0, 0)
            while queue and len(pairs) < k:
                _, i, j = heapq.heappop(queue)
                pairs.append(nums1[i] + nums2[j])
                push(i, j + 1)
                if j == 0:
                    push(i + 1, 0)
        
        # print(pairs)
        return pairs[k-1]
