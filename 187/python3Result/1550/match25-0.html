class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        if len(mat) == 1:
            return mat[0][k-1]
        
        if len(mat[0]) == 1:
            return sum([i[0] for i in mat])
        
        re = [0]
        
        for row in mat:
            current = []
            s_row = row
            # print(row)
            if len(row) > k:
                s_row = row[:k]
            for c in s_row:
                for e in re:
                    current.append(c+e)
            current.sort()
            if len(current) > k:
                current = current[:k]
            re = current
            
            # print(re)
        return re[k-1]
