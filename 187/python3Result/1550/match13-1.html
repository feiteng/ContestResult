class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        C = sum(row[0] for row in mat)
        mat = [[x - row[0]  for x in row] for row in mat]
        M = len(mat)
        N = len(mat[0])
        m = mat[0]
[:k]
        for row in mat[1:]:
            r = []
            for mm in m:
                for c in row:
                    r.append(mm + c)
            r = sorted(r)
            m = r[:k]
        return m[k - 1] + C
