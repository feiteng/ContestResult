class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        cur = []
        for i in range(m):
            if i == 0:
                for j in range(n):
                    cur.append(mat[i][j])
                    cur.sort()
                    cur = cur[:k]
            else:
                nxt = []
                for j in range(n):
                    for c in cur:
                        nxt.append(c + mat[i][j])
                nxt.sort()
                #print(nxt)
                cur = nxt[:k]
            #print(cur)
        return cur[-1]
