class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        M, N = len(mat), len(mat[0])
        heap = [(sum(mat[i][0] for i in range(M)), [0]*M)]
        used = set(tuple([0]*M))
        
        while k:
            # print(heap)
            s, cols = heapq.heappop(heap)
            k -= 1
            if k == 0: return s
            mx = max((mat[p][cols[p]+1] - mat[p][cols[p]] if cols[p]+1 < N else 0) for p in range(M))
            for i in range(M):
                for j in range(cols[i]+1, N):
                    if mat[i][j] - mat[i][cols[i]] <= mx:
                        
                        if tuple(cols[:i] + [j] + cols[i+1:]) in used: continue
                        heapq.heappush(heap, (s - mat[i][cols[i]] + mat[i][j], cols[:i] + [j] + cols[i+1:]))
                        used.add(tuple(cols[:i] + [j] + cols[i+1:]))
                        # print("push ", s - mat[i][cols[i]] + mat[i][j])
                    else: break
        return -1
