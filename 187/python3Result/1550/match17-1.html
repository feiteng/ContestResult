class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        def solve(lo, hi):
            if hi - lo + 1 == 2:
                return self.kSmallestPairs(mat[lo], mat[hi], k)
            elif lo == hi:
                return mat[lo][:k]
            mid = (lo+hi) // 2
            up = solve(lo, mid)
            do = solve(mid+1, hi)
            ans = []
            for n1 in up:
                for n2 in do:
                    ans += n1+n2,
            return sorted(ans)[:k]
        return solve(0, len(mat)-1)[k-1]
        
    
def kSmallestPairs(self, nums1, nums2, k):
        queue = []
        def push(i, j):
            if i < len(nums1) and j < len(nums2):
                heapq.heappush(queue, [nums1[i] + nums2[j], i, j])
        push(0, 0)
        pairs = []
        while queue and len(pairs) < k:
            _, i, j = heapq.heappop(queue)
            pairs.append(nums1[i] + nums2[j])
            push(i, j + 1)
            if j == 0:
                push(i + 1, 0)
        return pairs
