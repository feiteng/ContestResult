class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        k = k - 1
        summations = sorted(mat[0])[: k + 1]

        for row in mat[1:]:
            # newSummations = set()
            newSummations = []

            for v in row:

                for summation in summations:
                    newSummations.append(summation + v)

            summations = sorted(newSummations)[: k + 1]

        # print(summations)

        return summations[k]
