class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        a = [0]
        for arr in mat:
            b = []
            for x in a:
                for y in arr:
                    b.append(x + y)
            b.sort()
            if len(b) < 200:
                a = b
            else:
                a = b[:200]
        return a[k-1]
