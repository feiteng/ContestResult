class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m = len(mat)
        def dfs(nums1, nums2):
            queue = []
            def push(i, j):
                if i < len(nums1) and j < len(nums2):
                    heapq.heappush(queue, [nums1[i] + nums2[j], i, j])
            push(0, 0)
            pairs = []
            while queue and len(pairs) < k:
                _, i, j = heapq.heappop(queue)
                pairs.append([nums1[i], nums2[j]])
                push(i, j + 1)
                if j == 0:
                    push(i + 1, 0)
                    
                    
            return [sum(p)  for p in pairs]
        
        def parition(l , r):
            
            if l >= r:
                return mat[l]
            
            mid = l + (r- l >>  1)
            
            x = parition(l , mid)
            y = parition(mid + 1, r)
            
            return dfs(x,y)
        
        
        return parition(0, m - 1)[k - 1]
