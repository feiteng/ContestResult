import heapq
from collections import deque
import copy

class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        
        m, n = len(mat), len(mat[0])

        init_state = (sum(r[0] for r in mat), [0 for _ in range(m)])

        pos = 0
        queue = [init_state]
        visited = set()
        while queue:
            cur_sum, state = heapq.heappop(queue)

            state_tuple = tuple(state)
            if state_tuple in visited: continue
            visited.add(state_tuple)
            pos += 1
                
            
            if pos == k:
                return cur_sum
    
            for i in range(m):
                next_state = state[:]
                j = next_state[i]
                if j + 1 == n: continue
                next_state[i] = j + 1
                next_sum = cur_sum + mat[i][j + 1] - mat[i][j]
                heapq.heappush(queue, (next_sum, next_state))
    
        return cur_sum
