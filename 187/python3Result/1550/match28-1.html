class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        n, m = len(mat), len(mat[0])
        heap, count, vis = [(sum(x[0] for x in mat), [0] * n)], 0, set()
        while heap:
            count += 1
            if count == k:
                return heap[0][0]
            presum, index = heapq.heappop(heap)
            for i in range(n):
                temp = copy.copy(index)
                temp[i] += 1
                if temp[i] >= m or tuple(temp) in vis:
                    
continue
                heapq.heappush(heap, (presum - mat[i][index[i]] + mat[i][temp[i]], temp))
                vis.add(tuple(temp))
        return 0
