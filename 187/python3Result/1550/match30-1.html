import heapq as hq
from itertools import product

class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        m, n = len(mat), len(mat[0])
        
        curr = mat[0]
        
        for i in range(1, m):
            curr = hq.nsmallest(k, list(sum(i) for i in product(curr, mat[i])))
        
        return curr[-1]
