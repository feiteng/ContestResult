class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        nrow, ncol = len(mat), len(mat[0])
        heap = []
        pointers = [0] * nrow
        acc = 0
        visited = set()
        for i in range(nrow):
            acc += mat[i][0]
        heapq.heappush(heap, (acc, pointers))
        visited.add(tuple(pointers))

        for _ in range(k):
            acc, pointers = heapq.heappop(heap)
            for i in range(nrow):
                new_pointers = list(pointers)
                if new_pointers[i] + 1 < ncol:
                    new_pointers[i] += 1
                    key = tuple(new_pointers)
                    if key not in visited:
                        new_acc = acc + mat[i][new_pointers[i]] - mat[i][new_pointers[i] - 1]
                        heapq.heappush(heap, (new_acc, new_pointers))
                        visited.add(key)

        return acc
