class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        s = 0
        q = 1
        ans = 1
        q_min = deque()
        q_max = deque()
        q_min.append(nums[0])
        q_max.append(nums[0])
        last_move = 1
        while q < n:
            if last_move:
                while q_max and q_max[-1] < nums[q]: q_max.pop()
                q_max.append(nums[q])
                while q_min and q_min[-1] > nums[q]: q_min.pop()
                q_min.append(nums[q])
            #print(s,q,q_min,q_max)
            if q_max[0] - q_min[0] <= limit: 
                if q - s + 1 > ans: ans = q - s + 1
                q += 1
                last_move = 1
            else:
                if nums[s] == q_min[0]: q_min.popleft()
                if nums[s] == q_max[0]: q_max.popleft()
                s += 1
                last_move = 0            
        return ans
