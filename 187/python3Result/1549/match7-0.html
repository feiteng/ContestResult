class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        n = len(nums)
        i = j = 0
        l, s = [collections.deque() for i in range(2)]
        ret = 0
        while j < n:
            nj = nums[j]
            while l and l[-1][0] < nj:
                l.pop()
            l.append((nj, j))
            while s and s[-1][0] > nj:
                s.pop()
            s.append((nj, j))
            if l[0][0] - s[0][0] <= limit:
                ret = max(ret, j - i + 1)
            else:
                if l[0][1] == i:
                    l.popleft()
                if s[0][1] == i:
                    s.popleft()
                i += 1
            j += 1
        return ret
