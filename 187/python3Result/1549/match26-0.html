class Solution:
	def longestSubarray(self, nums: List[int], limit: int) -> int:
		from collections import deque

		res = i = 0
		mnQueue, mxQueue = deque(), deque()
		for j in range(len(nums)):
			while mnQueue and nums[mnQueue[-1]] >= nums[j]: mnQueue.pop()
			mnQueue.append(j)
			
			while mxQueue and nums[mxQueue[-1]] <= nums[j]: mxQueue.pop()
			mxQueue.append(j)

			mni, mxi = mxQueue[0], mxQueue[0]
			mn, mx = nums[mni], nums[mxi]

			if nums[mxQueue[0]]-nums[mnQueue[0]] > limit:
				i += 1
				if mnQueue and i > mnQueue[0]: mnQueue.popleft()
				if mxQueue and i > mxQueue[0]: mxQueue.popleft()
			else:
				res = max(res, j-i+1)


		return res
