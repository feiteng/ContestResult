class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        if not nums:
            return 0
        fmax = [0]
        fmin = [0]
        i = 0
        j 
= 0
        m = 0
        while True:
            if nums[fmax[0]] - nums[fmin[0]] <= limit:
                m = max(m, j - i)
                j += 1
                if j == len(nums):
                    break
                k = len(fmax) - 1
                while k >= 0 and nums[j] > nums[fmax[k]]:
                    fmax.pop(k)
                    k -= 1
                fmax.append(j)
                k = len(fmin) - 1
                while k >= 0 and nums[j] < nums[fmin[k]]:

= 1
                fmin.append(j)
            else:
                if i == fmax[0]:
                    fmax.pop(0)
                if i == fmin[0]:
                    fmin.pop(0)
                i += 1
        return m + 1
