        ans = 0
        left = 0
        maxq, minq = collections.deque(), collections.deque()
        for i, v in enumerate(nums):
            while maxq and maxq[-1][1] < v:
                maxq.pop()
            maxq.append((i,v))
            while minq and minq[-1][1] > v:
                minq.pop()
            minq.append((i,v))
            while abs(maxq[0][1] - minq[0][1]) > limit:
                left += 1
                if left > maxq[0][0]: maxq.popleft()
                if left > minq[0][0]: minq.popleft()
            cand = i - left + 1
            ans = max(ans, cand)
        return ans
