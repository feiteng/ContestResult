class Solution:
    def longestSubarray(self, nums: List[int], k: int) -> int:
        dec = deque() # store max in window
        inc = deque() # store min in window
        
        res = 0
        
        leftIdx 
= 0
        
        for i, v in enumerate(nums):

            while dec and dec[-1] < v:
                dec.pop()
                
            dec.append(v)
            
            while inc and inc[-1] > v:
                inc.pop()
            
            inc.append(v)
        
            maxVal = dec[0]
            minVal = inc[0]
            
            # if maxVal - minVal <= k:
            #     res = max(res, i - leftIdx + 1)
            # else:

            
while dec[0] - inc[0] > k:
                if dec[0] == nums[leftIdx]:
                    dec.popleft()
                if inc[0] == nums[leftIdx]:
                    inc.popleft()
                leftIdx += 1
            
            res = max(res, i - leftIdx + 1)
        
        return res
