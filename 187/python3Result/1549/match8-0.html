class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        min_deque = deque()
        max_deque = deque()
        
        ans = 0
        l = r = 0
        while r < len(nums):
            num = nums[r]
            
            while min_deque and num <= nums[min_deque[-1]]:
                min_deque.pop()
            min_deque.append(r)
            
            while max_deque and num >= nums[max_deque[-1]]:
                max_deque.pop()
            max_deque.append(r)
            
            min_idx = min_deque[0]
            min_val = nums[min_idx]
            max_idx = max_deque[0]
            max_val = nums[max_idx]
            
            if max_val - min_val > limit:
                l += 1
                if l > min_deque[0]:
                    min_deque.popleft()
                if l > max_deque[0]:
                    max_deque.popleft()
            else:
                ans = max(ans, r - l + 1)
                r += 1
                
        return ans
