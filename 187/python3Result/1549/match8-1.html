class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:       
        res = 0
        
        bstart = 0
        bend = 0
        
        min_q = collections.deque()
        max_q = collections.deque()

        start 
= 0
        end = 0

        while end < len(nums):
            x = nums[end]

            while (min_q and nums[min_q[-1]] >= x):
                min_q.pop()
            
            min_q.append(end)
            
            while (max_q and nums[max_q[-1]] <= x):
                max_q.pop()
            
            max_q.append(end)
            
            min_id = min_q[0]
            min_v = nums[min_id]
            
            max_id = max_q[0]
            max_v = nums[max_id]
            
            if max_v - min_v > limit:
                start += 1
                if start > min_q[0]:
                    min_q.popleft()
                if start > max_q[0]:
                    max_q.popleft()
            else:
                res = max(res, end - start + 1)
                end += 1
        
        return res
