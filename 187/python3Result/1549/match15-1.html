class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        max_que = collections.deque()
        min_que = collections.deque()
        i = res 
= 0
        for j, v in enumerate(nums):
            while max_que and max_que[-1] < v:
                max_que.pop()
            max_que.append(v)
            
            while min_que and min_que[-1] > v:
                min_que.pop()
            min_que.append(v)
            
            while max_que[0] - min_que[0] > limit:
                if max_que and max_que[0] == nums[i]:
                    max_que.popleft()
                if min_que and min_que[0] == nums[i]:
                    min_que.popleft()
                i += 1
            res = max(res, j-i+1)
        return res
