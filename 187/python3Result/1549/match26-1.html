class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        mins = collections.deque([0])
        maxs = collections.deque([0])
        l = 0
        ans = 1
        
for i in range(1,len(nums)):
            while mins and nums[i] < nums[mins[-1]]:
                mins.pop()
            mins.append(i)
            while maxs and nums[i] > nums[maxs[-1]]:
                maxs.pop()
            maxs.append(i)
            if nums[maxs[0]] - nums[mins[0]] <= limit:
                ans = max(ans,i-l+1)
            else:
                while nums
[maxs[0]] - nums[mins[0]] > limit:
                    l += 1
                    if mins[0] < l:
                        mins.popleft()
                    if maxs[0] < l:
                        maxs.popleft()
        return ans
