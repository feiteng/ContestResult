import sortedcontainers

class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        sl = sortedcontainers.SortedList()
        sl.add(nums[0])
        i = 0
        ans = 1
        for j in range(1, len(nums)):
            sl.add(nums[j])
            while sl[-1] - sl[0] > limit:
                sl.remove(nums[i])
                i += 1
            ans = max(ans, j - i + 1)
        return ans
