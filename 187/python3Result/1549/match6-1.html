import collections
class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        ret=1
        # increasing stack to record min, descreasing stack to record max
        # [val, ind]
        inc, dec = collections.deque(), collections.deque()
        n=len(nums)
        prev=-1
        
for i, val in enumerate(nums):
            while inc and val<inc[-1][0]:
                inc.pop()
            inc.append([val, i])
            while dec and val>dec[-1][0]:
                dec.pop()
            dec.append([val, i])
            
            while dec and inc and dec[0][0]-inc[0][0]>limit:
                if dec[0][1]<inc[0][1]:
                    prev=dec[0][1]
                    dec.popleft()
                else:
                    prev=inc

                    inc.popleft()
            #print(i, prev, inc, dec)
            ret=max(ret, i-prev)
        return ret
