class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        mi = []
        mx = []
        start = 0
        i = 0
        N = len(nums)
        res = 0
        while i < N:
            n = nums[i]
            
while mi and mi[-1] > n: mi.pop(-1)
            mi.append(n)
            while mx and mx[-1] < n: mx.pop(-1)
            mx.append(n)
            while start <= i and mx and mi and mx[0] - mi[0] > limit:
                if mx[0] == nums[start]: mx.pop(0)
                if mi[0] == nums[start]: mi.pop(0)
                start += 1
            res = max(res, i - start + 1)
            i += 1
        return res
