from collections import deque
class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        
        max_q = deque()
        min_q = deque()
        ans = 0
        q = 0
        for p, num in enumerate(nums):
            while len(max_q) > 0 and num > max_q[-1]:
                max_q.pop()
            max_q.append(num)
            
            while len(min_q) > 0 and num < min_q[-1]:
                min_q.pop()
            min_q.append(num)
            
            while max_q[0] - min_q[0] > limit:
                if nums[q] == max_q[0]:
                    max_q.popleft()
                if nums[q] == min_q[0]:
                    min_q.popleft()
                q += 1
            
            ans = max(ans, p - q + 1)
            
        return ans
