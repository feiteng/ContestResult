class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        
        def check(k):
            mn = collections.deque()
            mx = collections.deque()
            for i in range(k - 1):
                while mn and mn[-1] > nums[i]:
                    mn.pop()
                mn.append(nums[i])
                while mx and mx[-1] < nums[i]:
                    mx.pop()
                mx.append(nums[i])
            
            for i in range(k - 1, len(nums)):
                while mn and mn[-1] > nums[i]:
                    mn.pop()
                mn.append(nums[i])
                while mx and mx[-1] < nums[i]:
                    mx.pop()
                mx.append(nums[i])
                
                #print(i, mx, mn)
                if mx[0] - mn[0] <= limit:
                    return True
                
                if nums[i - k + 1] == mn[0]:
                    mn.popleft()
                if nums[i - k + 1] == mx[0]:
                    mx.popleft()
            #print()
                    
            return False
        
        
        left = 0
        right = len(nums) + 1
        while left < right:
            mid = (left + right) // 2
            if check(mid):
                left = mid + 1
            else:
                right = mid
        
        return right - 1
