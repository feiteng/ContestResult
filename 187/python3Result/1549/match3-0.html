class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        rmax, rmin = collections.deque(), collections.deque()
        start, ans = 0, 0
        for i,x in enumerate(nums):
            while rmax and nums[rmax[-1]]<=x:
                rmax.pop()
            rmax.append(i)
            while rmin and nums[rmin[-1]]>=x:
                rmin.pop()
            rmin.append(i)
            while rmax and rmin and nums[rmax[0]]-nums[rmin[0]]>limit:
                if rmax[0]==start:
                    rmax.popleft()
                if rmin[0]==start:
                    rmin.popleft()
                start += 1
            ans = max(i-start+1, ans)
        return ans
