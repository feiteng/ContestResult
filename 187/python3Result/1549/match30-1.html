class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        pre, ans, ma, mi = -1, 0, deque(), deque
()
        for i, v in enumerate(nums):
            while ma and ma[-1][0] <= v:
                ma.pop()
            ma.append((v, i))
            while mi and mi[-1][0] >= v:
                mi.pop()
            mi.append((v, i))
            while pre < i and ma and mi and ma[0][0] - mi[0][0] > limit:
                pre += 1
                
if ma[0][1] == pre:
                    ma.popleft()
                if mi[0][1] == pre:
                    mi.popleft()
            ans = max(ans, i - pre)
        return ans
