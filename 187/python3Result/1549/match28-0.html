class Solution:
    def longestSubarray(self, nums: List[int], limit: int) -> int:
        if not nums: return 0
        min_stack = collections.deque([0])
        max_stack = collections.deque([0])
        start, end = 0, 0
        best = 1
        while True:
            if nums[max_stack[0]] - nums[min_stack[0]] <= limit:
                best = max(best, end - start + 1)
                end += 1
                if end >= len(nums):
                    break
                # push end
                while min_stack and nums[min_stack[-1]] > nums[end]:
                    min_stack.pop()
                min_stack.append(end)
                while max_stack and nums[max_stack[-1]] < nums[end]:
                    max_stack.pop()
                max_stack.append(end)
            else:
                # pop start
                if min_stack[0] == start:
                    min_stack.popleft()
                if max_stack[0] == start:
                    max_stack.popleft()
                start += 1
            #print(start, end, min_stack, max_stack)
        return best
