class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        
        dic = {}
        rtn = []
        
        for x in names:
            if x not in dic:
                dic[x] = 1
                rtn.append(x)
            else:
                indx = dic[x]
                newx = '%s(%s)'%(x, indx)
                indx += 1
                while newx in dic:
                    newx = '%s(%s)'%(x, indx)
                    indx += 1
                dic[x] = indx
                dic[newx] = 1
                rtn.append(newx)
        
        return rtn
