class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        a = []
        d = {}
        for name in names:
            if name not in d:
                a.append(name)
                d[name] = 1
            else:
                i = d[name]
                name2 = name + '(' + str(i) + ')'
                while name2 in d:
                    i += 1
                    name2 = name + '(' + str(i) + ')'
                a.append(name2)
                d[name] = i
                d[name2] = 1
        return a
