class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        used_names = set()
        max_index = {}
        res = []
        for name in names:
            if name in used_names:
                i = max_index.get(name, 0) + 1 
                suffix 
= "("+ str(i)+")"
                while name + suffix in used_names:
                    i += 1
                    suffix = "("+ str(i)+")"
                res.append(name + suffix)
                used_names.add(name + suffix)
                max_index[name] = i
            else:
                res.append(name)
                used_names.add(name)
        return res
