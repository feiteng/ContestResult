class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        d = {}
        res = []
        for n in names:
            if n not in d:
                res.append(n)
                d[n] = 0
            else:
                count = d[n]
                temp = n + "(" + str(count+1) + ")"
                while temp in d:
                    count += 1
                    temp = n + "(" + str(count+1) + ")"
                    
                d[temp] = 0
                d[n] += 1
                res.append(temp)
        return res
