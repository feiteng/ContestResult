class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        used = set()
        indexes = {}
        answer = []

        for name in names:
            if name in indexes:
                candidate_id = indexes[name] + 1
                candidate_name = name + "(" + str(candidate_id) + ")"

                while candidate_name in used:
                    candidate_id += 1
                    candidate_name = name + "(" + str(candidate_id) + ")"

                answer.append(candidate_name)
                used.add(candidate_name)
                indexes[name] = candidate_id
            else:
                indexes[name] = 0
                if name in used:
                    candidate_id = 1
                    candidate_name = name + "(" + str(candidate_id) + ")"

                    while candidate_name in used:
                        candidate_id += 1
                        candidate_name = name + "(" + str(candidate_id) + ")"

                    answer.append(candidate_name)
                    used.add(candidate_name)
                    indexes[name] = candidate_id
                else:
                    used.add(name)
                    answer.append(name)

        return answer
