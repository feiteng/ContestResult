class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        # dict from name to version
        version_dict = dict()
        result = []
        for name in names:
            if name not in version_dict:
                result.append(name)
                version_dict[name] = 1
            else:
                cur_v = version_dict[name]
                while f'{name}({cur_v})' in version_dict:
                    cur_v += 1
                result.append(f'{name}({cur_v})')
                version_dict[name] = cur_v + 1
                version_dict[f'{name}({cur_v})'] = 1
        return result
