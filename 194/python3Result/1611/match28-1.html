class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        d = dict()
        ret 
= []
        for i in names:
            if i not in d:
                d[i] = 1
                ret.append(i)
            else:
                j = i + '(' + str(d[i]) + ')'
                while j in d:
                    d[i] += 1
                    j = i + '(' + str(d[i]) + ')'
                d[j] = 1
                ret.append(j)
        return ret
