class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        C = {}
        ans = []

        for name in names:
            if name in C:
                i = C[name]
                tmp = name + '(' + str(i) + ')'
                while tmp in C:
                    i += 1
                    tmp = name + '(' + str(i) + ')'
                C[name] = i
                C[tmp] = 1
                ans.append(tmp)
            else:
                C[name] = 1
                ans.append(name)

        return ans
