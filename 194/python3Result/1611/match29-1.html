class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        version = dict()
        res 
= []
        for name in names:
            if name not in version:
                res.append(name)
                version[name] = 1
            else:
                i = version[name]
                while name + '(' + str(i) + ')' in version:
                    i += 1
                res.append(name + '(' + str(i) + ')')
                version[name + '(' + str(i) + ')'] = 1
                version[name] = i + 1
        return res
