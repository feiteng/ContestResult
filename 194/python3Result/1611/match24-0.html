class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        dic = {}
        ans = []
        for i in names:
            if dic.get(i,0):
                ti = dic.get(i)
                tj = i + '('+ str(ti) +')'
                while(dic.get(tj,0)):
                    ti += 1
                    tj = i + '('+ str(ti) +')'
                dic[tj] = 1
                dic[i] = ti
                ans.append(tj)
            else:
                dic[i] = 1
                ans.append(i)
        return ans
