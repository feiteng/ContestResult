class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        d={}
        ans=[]
        for n in names:
            if n not in d:
                ans.append(n)
                d[n]=1
            else:
                k=d[n]
                while n+"("+str(k)+")" in d:
                    k+=1
                ans.append(n+"("+str(k)+")")
                d[n+"("+str(k)+")"]=1
                d[n]=k
        return ans
