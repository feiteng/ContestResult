class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        res = []
        seen = {}
        for name in names:
            if name not in seen:
                res.append(name)
                seen[name] = 1
            else:
                k = seen[name]
                while f'{name}({k})' in seen:
                    k += 1
                res.append(f'{name}({k})')
                seen[name] = k + 1
                seen[f'{name}({k})'] = 1
        return res
