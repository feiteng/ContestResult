class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        count={}
        ans=[]
        for name in names:
            #print(count)
            if name not in count:
                ans.append(name)
                count[name]=1
            else:
                k=count[name]
                tmp=name+"("+str(k)+")"
                while tmp in count:
                    k+=1
                    tmp=name+"("+str(k)+")"
                ans.append(tmp)
                count[name]=k+1
                count[tmp]=1
        return ans
