class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        namespace = {}
        ans = []
        for name in names:
            if name not in namespace:
                ans.append(name)
                namespace[name] = 1
            else:
                res = "{}({})".format(name, namespace[name])
                while res in namespace:
                    namespace[name] += 1
                    res = "{}({})".format(name, namespace[name])
                ans.append(res)
                namespace[res] = 1
                namespace[name] += 1
        return ans
