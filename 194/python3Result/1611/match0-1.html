class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        d = {}
        result = []
        for x in names:
            if x not in d:
                result.append(x)
                d[x] = 1
            else:
                i = d[x]
                while f"{x}({i})" in d:
                    i += 1
                result.append(f"{x}({i})")
                d[x] = i + 1
                d[f"{x}({i})"] = 1
        return result
