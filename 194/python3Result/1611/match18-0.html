class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        res = []
        seen = {}
        for name in names:
            if name not in seen:
                seen[name] = 1
                res.append(name)
            else:
                cur = seen[name]
                temp = name+ "(" + str(cur) + ")"
                while temp in seen:
                    cur += 1
                    temp = name+ "(" + str(cur) + ")"
                seen[temp] = 1                
                seen[name] += 1
                res.append(temp)
                
        
        return res
