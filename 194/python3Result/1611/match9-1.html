class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        ans = []
        m = {}
        for name in names:
            if name not in m:
                ans.append(name)
                m[name] = 1
            else:
                i = m[name]
                s = name + '(' + str(i) + ')'
                while s in m:
                    i += 1
                    s = name + '(' + str(
i) + ')'
                ans.append(s)
                m[s] = 1
                m[name] = i+1
        return ans
