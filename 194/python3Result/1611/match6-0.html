class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        res = []
        tmp = {}
        for name in names:
            if name not in tmp:
                res.append(name)
                tmp[name] = 1
            else:
                i = tmp[name]
                while f'{name}({i})' in tmp:
                    i += 1
                res.append(f'{name}({i})')
                tmp[f'{name}({i})'] = 1
                tmp[name] = i + 1
        return res
