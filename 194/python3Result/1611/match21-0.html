class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        nameMap = {}
        res = []
        
        for n in names:
            if n not in nameMap:
                res.append(n)
                nameMap[n] = 0
            else:
                i = nameMap[n] + 1
                
                
                builtName = n + "(" + str(i) + ")"
                while builtName in nameMap:
                    i += 1
                    builtName = n + "(" + str(i) + ")"
                nameMap[n] = i
                nameMap[builtName] = 0
                res.append(builtName)
                
        return res
