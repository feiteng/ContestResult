class Solution:
    def getFolderNames(self, names: List[str]) -> List[str]:
        used = {}
        ans = []
        for name in names:
            if name not in used:
                used[name] = 1
                ans.append(name)
            else:
                new = name + '(' + str(used[name]) + ')'
                while new in used:
                    used[name] += 1
                    new = name + '(' + str(used[name]) + ')'
                used[new] = 1
                ans.append(new)
        return ans
