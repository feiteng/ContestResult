class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        cnt = {}
        ans = [-1 for _ in rains]
        availables = []
        for i, r in enumerate(rains):
            if r:
                if r in cnt:
                    pre = cnt[r]
                    idx = bisect.bisect(availables, pre)
                    if idx==len(availables):
                        return []
                    ans[availables.pop(idx)]=r
                cnt[r]=i
            else:
                availables.append(i)
        for i in availables:
            ans[i]=1
        return ans
