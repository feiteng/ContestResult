class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        lakes = dict()
        drydays = []
        ans = [-1 for _ in range(len(rains))]
        for pos,day in enumerate(rains):
            if day > 0:
                #print(pos,day,lakes)
                if day in lakes and lakes[day] > -1:
                    targetday = -1
                    #print(pos,day,drydays)
                    for i in range(len(drydays)):
                        if drydays[i] > lakes[day]:
                            targetday = drydays.pop(i)
                            break
                    if targetday > -1:
                        ans[targetday] = day
                        lakes
[day] = pos
                    else:
                        return []
                else:
                    lakes[day] = pos
            else:
                drydays.append(pos)
        for day in drydays:
            ans[day] = 1
        return ans
