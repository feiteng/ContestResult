from collections import deque
from collections import defaultdict
import heapq
class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        pos = defaultdict(deque)
        for i, v in enumerate(rains):
            if v != 0:
                pos[v].append(i)
        
        watered = set()
        rst = []
        hp = []
        for x, v in enumerate(rains):
            if v != 0:
                if v in watered:
                    return []
                rst.append(-1)
                watered.add(v)
                while pos[v] and pos[v][0]<=x:
                    pos[v].popleft()
                if pos[v]:
                    heapq.heappush(hp, (pos[v].popleft(), v))
            else:
                if not hp:
                    rst.append(1)
                else:
                    _, to_clean = heapq.heappop(hp)
                    rst.append(to_clean)
                    watered.remove(to_clean)
        return rst
