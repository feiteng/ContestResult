class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        cur = 0
        tmp = []
        ans = []
        dic = {}
        for i,r in enumerate(rains):
            if r == 0:
                ans.append(1)
                tmp.append(i)
            else:
                #print(ans)
                #print(tmp)
                ans.append(-1)
                if r in dic:
                    flag = True
                    for t in tmp:
                        if t > dic[r]:
                            tmp.remove(t)
                            ans[t] = r
                            dic[r] = i
                            flag = False
                            break
                    if flag:
                        return []
                else:
                    dic[r] = i
        return ans
