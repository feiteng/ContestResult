import bisect
class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        d_rain = {}
        res = [1] * len(rains)
        stack = []
        for i, l in enumerate(rains):
            if l:
                if l in d_rain:
                    idx 
= bisect.bisect(stack, d_rain[l])
                    if len(stack) == idx:
                        return []
                    res[stack.pop(idx)] = l
                d_rain[l] = i
                res[i] = -1
            else:
                stack.append(i)
        return res
