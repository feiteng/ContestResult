from collections import deque
class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        
        n = len(rains)
        res = [-1] * n
        
        d, avail = {}, []
        for i, r in enumerate(rains):
            if r == 0:
                avail.append(i)
                continue
                
            if r in d:
                idx = bisect.bisect_right(avail, d[r])
                if idx == len(avail):
                    return []
                
                day = avail[idx]
                avail.pop(idx)
                res[day] = r
                
            d[r] = i
        
        for day in avail:
            res[day] = 1
        return res
