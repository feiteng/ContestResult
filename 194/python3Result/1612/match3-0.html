import bisect
class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        res = [-1]*len(rains)
        slots = []
        memo = {}
        for i,r in enumerate(rains):
            if r==0:
                slots.append(i)
            else:
                if r in memo:
                    if not slots:
                        return []
                    idx = bisect.bisect_left(slots,memo[r])
                    if idx == len(slots): return []
                    res[slots.pop(idx)] = r
                memo[r] = i
        while slots:
            res[slots.pop()] = 1
        return res
