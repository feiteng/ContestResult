class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        rst = [-1] * len(rains)
        dry, full = [], {}
        for i, lack in enumerate(rains):
            if lack > 0:
                if lack in full:
                    prev = full[lack]
                    idx = bisect.bisect(dry, prev)
                    if idx >= len(dry): return []
                    rst[dry[idx]] = lack
                    del dry[idx]
                full[lack] = i
            else:
                dry.append(i)
        for i in dry:
            rst[i] = 1
        return rst
