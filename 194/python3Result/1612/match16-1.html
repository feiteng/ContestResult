class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        ans, empty, water = [], [], {}
        for id, r in enumerate(rains):
            if not r:
                empty.append(id)
                ans.append(1)
            else:
                if r in water:
                    k 
= bisect.bisect(empty, water[r])
                    if k == len(empty):
                        return []

                    ans[empty.pop(k)] = r
                water[r] = id
                ans.append(-1)
        return ans
