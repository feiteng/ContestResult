class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        res = []
        dry_index = []
        rain_index = {}
        for i, rain in enumerate(rains):
            if rain == 0:
                dry_index.append(i)
                res.append(0)
            else:
                res.append(-1)
                if rain not in rain_index:
                    rain_index[rain] = i
                else:
                    dry_flag = False
                    for j, x in enumerate(dry_index):
                        if x > rain_index[rain]:
                            res[x] = rain
                            dry_flag = True
                            dry_index.pop(j)
                            break
                    if not dry_flag: return []
                    rain_index[rain] = i
            #print(res, dry_index, i)
        rain_index_keys = list(rain_index.keys())
        for i in range(len(res)):
            if res[i] == 0:
                res[i] = rain_index_keys[0]
        return res
