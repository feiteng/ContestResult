# []

import bisect

class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        unused = []
        prev = {}

        answer = []
        for i, rain in enumerate(rains):
            if rain > 0:
                answer.append(-1)

                if rain in prev:
                    target = bisect.bisect(unused, prev[rain])

                    if target == len(unused):
                        return []

                    answer[unused[target]] = rain
                    unused.pop(target)
                
                prev[rain] = i
            else:
                answer.append(1)
                unused.append(i)

        return answer
