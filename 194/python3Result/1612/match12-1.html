class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        n = len(rains)
        full = dict()
        dry = deque()
        ret = [-1] * n
        for i in range(n - 1, -1, -1):            
            lake = rains[i]
            if lake > 0:
                if lake in full:
                    if not dry or dry[0] > full[lake]:
                        return []
                    else:
                        j = bisect_right(dry, full[lake])
                        ret[dry[j - 1]] = lake
                        del dry[j
 - 1]
                full[lake] = i
            else:
                dry.appendleft(i)
        while dry:
            ret[dry.pop()] = 1
        return ret
