import heapq
import collections
import functools
import math
from typing import List


class Solution:

    def avoidFlood(self, rains: List[int]) -> List[int]:

        is_wet = set()
        positions_for_lake = collections.defaultdict(collections.deque)
        for i, elem in enumerate(rains):
            positions_for_lake[elem].append(i)
        urgency_and_index = []
        heapq.heapify(urgency_and_index)

        ans = []
        for _i, elem in enumerate(rains):
            
if elem == 0:
                if len(urgency_and_index) == 0:
                    ans.append(1)
                else:
                    _t, next_to_clean = heapq.heappop(urgency_and_index)
                    is_wet.remove(next_to_clean)
                    ans.append(next_to_clean)
            else:
                if elem in is_wet:
                    return []
                ans.append(-1)
                is_wet.add(elem)
                positions_for_lake[elem].popleft()
                if len(positions_for_lake[elem]) > 0:
                    next_time 
= positions_for_lake[elem][0]
                    heapq.heappush(urgency_and_index, (next_time, elem))

        return ans
