import bisect
class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        res = [-1] * len(rains)
        full = {}
        dry = []
        for i, r in enumerate(rains):
            if not r:
                dry.append(i)
            elif r in full:
                e = bisect.bisect_right(dry, full[r])
                if e < 0 or e >= len(dry):
                    return []
                e = dry.pop(e)
                res[e] = r
                full[r] = i
            else:
                full[r] = i

        while dry:
            res[dry.pop()] = 1
        return res
