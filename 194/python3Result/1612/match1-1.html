from sortedcontainers import SortedList

class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        dry_days = SortedList()
        rain_dict = dict()
        ans = [-1] * len(rains)
        
        
for i, lake in enumerate(rains):
            if lake > 0:
                if lake in rain_dict:
                    rain_day = rain_dict[lake]
                    idx = dry_days.bisect_right(rain_day)
                    if idx == len(dry_days):
                        return []
                    ans[dry_days.pop(idx)] = lake
                rain_dict[lake] = i
            else:
                dry_days.add(i)
                ans[i] = 1
        
        return ans
