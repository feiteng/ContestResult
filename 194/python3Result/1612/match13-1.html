from heapq import heappush, heappop, heapify
from collections import defaultdict, deque

class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        lake_rain_days = defaultdict(deque)
        
        for day, lake_id in enumerate(rains):
            if lake_id != 0:
                lake_rain_days[lake_id].append(day+1)
            
        heap = []
        heapify(heap)
        
        ans = []
        full = set()
        for day, rain_lake in enumerate(rains):
            # print('='*100)
            # print(heap, ans, full)
            
            if rain_lake > 0:
                ans.append(-1)
                if rain_lake in full:
                    return []
                
                full
.add(rain_lake)
                while lake_rain_days[rain_lake] and lake_rain_days[rain_lake][0] <= day+1:
                    lake_rain_days[rain_lake].popleft()
                if lake_rain_days[rain_lake]:
                    heappush(heap, (lake_rain_days[rain_lake][0], rain_lake))
            else:
                if heap:
                    rain_days, lake = heappop(heap)
                    next_rain_day = lake_rain_days[lake].popleft()
                            
                    ans.append(lake)
                    if lake in full:
                        full.remove(lake)
                else:              
                    ans.append(999)
        return ans
