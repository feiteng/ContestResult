class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        lake_to_day_filled = {}
        dry_days = []
        result = [-1] * len(rains)
        
        for i, lake in enumerate(rains):
            if lake == 0:
                dry_days.append(i)
            else:
                if lake in lake_to_day_filled:
                    if len(dry_days):
                        drain_success = False
                        for j, day in enumerate(dry_days):
                            if day > lake_to_day_filled[lake]:
                                result[day] = lake
                                del lake_to_day_filled[lake]
                                dry_days.pop(j)
                                drain_success = True
                                break
                        if not drain_success:
                            return []
                    else:
                        return []
                lake_to_day_filled[lake] = i
        
        for day in dry_days:
            result[day] = 1
        return result
