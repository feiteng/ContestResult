class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        prev = None
        for num in rains:
            if prev and num > 0 and num == prev:
                return []
            prev = num
        n = len(rains)
        res = [-1] * n
        zeros 
= []
        seen = {}
        for i, num in enumerate(rains):
            if num == 0:
                zeros.append(i)
                continue
            if num in seen:
                possible = bisect.bisect_right(zeros, seen[num])
                if possible == len(zeros):
                    return []
                res[zeros[possible]] = num # dry the lake
                zeros.pop(possible)
            seen[num] = i
        for other in zeros:
            res[other] = 1
        return res
