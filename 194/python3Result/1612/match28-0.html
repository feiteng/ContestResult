class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        #维护一个已装满水的湖泊的编号
        result=[1]*len(rains)
        dic={}#代表第r个水坑，上次是在第几天才灌满的
        chou=[]#代表其第几天可抽,每次抽，应该是从距离最近的抽
        for i,r in enumerate(rains):
            if r == 0:
                chou.append(i)
            else:
                result[i]=-1
                if r not in dic:
                    dic[r]=i
                else:
                    start=0
                    flag=False
                    while start < len(chou):
                        if  chou[start] > dic[r]:
                            result[chou[start]]=r
                            chou.pop(start)
                            flag=True
                            break
                        else:
                            start+=1
                    if not flag:
                        return []
                    dic[r]=i

        return result
