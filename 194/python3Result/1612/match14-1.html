class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        pool = {}
        res = []
        chances 
= []
        for idx, lakeid in enumerate(rains):
            if lakeid == 0:
                chances.append(idx)
                res.append(1)
            elif lakeid in pool:
                if chances:
                    chance_id = bisect.bisect_right(chances, pool[lakeid])
                    if chance_id == len(chances): return []
                    res[chances[chance_id]] = lakeid
                    chances.pop(chance_id)
                    pool[lakeid] = idx
                    res.append(-1)
                else:
                    return []
            else:
                res.append(-1)
                pool[lakeid] = idx
        return res
