class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        seq = defaultdict(list)
        for i, v in enumerate(rains):
            if v!= 0: seq[v].append(i)
        #print(seq)
        cnt = {e:0 for e in rains }
        s = set()
        h = []
        ans = []
        for e in rains:
            if e == 0:
                if len(h) == 0:
                    ans.append(1)
                else:
                    _, x = heapq.heappop(h)
                    ans.append(x)
                    s.remove(x)
            elif e in s:
                return []
            else:
                ans.append(-1)
                s.add(e)
                if cnt[e] == len(seq[e])-1:
                    continue
                else:
                    cnt[e] += 1
                    next_loc = seq[e][cnt[e]]
                    heappush(h,(next_loc,e))
        return ans
