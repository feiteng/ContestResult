class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        ret = [-1]*len(rains)
        dryInd = []
        full = {}
        for i, rain in enumerate(rains):
            if rain:
                if rain in full:
                    ind = next((ind for ind, j in enumerate(dryInd) if j > full[rain]), -1)
                    if ind == -1:
                        return []
                    ret[dryInd.pop(ind)] = rain
                full[rain] = i
            else:
                dryInd.append(i)
        while dryInd:
            ret[dryInd.pop()] = 1
        return ret
