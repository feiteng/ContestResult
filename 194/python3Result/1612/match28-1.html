class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        ans = [-1] * len(rains)
        d = {} # d下雨
        l = [] # l不下雨
        for i, rain in enumerate(rains):
            if rain == 0:
                l.append(i)
                ans[i] = 1
            else:
                if d.get(rain, -1)==-1:
                    d[rain] = i
                elif len(l)==0:
                    return []
                else:
                    r, index = 0, 0
                    last = d.get(rain)
                    mark = False
                    for index, day in enumerate(l):
                        if day > last:
                            mark = True
                            break
                    if mark == False:
                        return []
                    ans[day] = rain
                    l.pop(index)
                    d[rain] = i
        return ans
