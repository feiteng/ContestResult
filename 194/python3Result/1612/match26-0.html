class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        
        res = []
        available = []
        h = {}
        
        for i, lake in enumerate(rains):
            if lake == 0:
                available.append(i)
                res.append(1) # 1 per default
            elif lake > 0:
                res.append(-1)
                if lake not in h:
                    h[lake] = i
                else: # there is a potential flood
                    if not available:
                        res = []
                        break
                    day = bisect.bisect_left(available, h[lake])
                    if day == len(available):
                        res = []
                        break
                    else: # avoid flooding
                        res[available[day]] = lake
                        h[lake] = i
                        available.pop(day)
                        
                    
                    
        return res
