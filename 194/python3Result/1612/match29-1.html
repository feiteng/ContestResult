class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        inter = []
        seen={}
        zeros=[]
        res=[]
        for i,r in enumerate(rains):
            if r ==0:
                zeros.append(i)
                res.append(1)
            elif r in seen:
                inter.append([seen[r],i,r])
                seen[r]=i
                res.append(-1)
            else:
                seen[r]=i
                res.append(-1)
        # res = [-1 if r>0 else 1 for r in rains]
        inter.sort(key=lambda i:(i[1],i[0]))
        for s,e,r in inter:
            ind = bisect.bisect(zeros,s)
            if ind == len(zeros) or zeros[ind]>e:
                return []
            res[zeros[ind]] = r
            zeros.pop(ind)
        return res
