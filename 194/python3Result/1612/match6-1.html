class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        empty = []
        seen = {}
        ans = [-1]*len(rains)
        for i, x in enumerate(rains):
            if x==0:
                empty.append(i)
            else:
                if x in seen:
                    if not empty or empty[-1]<seen[x]: return []
                    k = bisect.bisect_right(empty, seen[x])
                    ans[empty.pop(k)]=
x
                seen[x]=i
        for x in empty: ans[x]=1
        return ans
