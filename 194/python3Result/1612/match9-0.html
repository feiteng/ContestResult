class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        res = [-1] * len(rains)
        zero_day = []
        rain_set = {}
        for i,idx in enumerate(rains):
            # print(idx, zero_day, rain_set)
            if idx > 0:
                if idx in rain_set:
                    if len(zero_day) <= 0:
                        return []
                    else:
                        t_idx = 0
                        while t_idx < len(zero_day) and zero_day[t_idx] < rain_set[idx]:
                            t_idx += 1
                        if t_idx >= len(zero_day):
                            return []
                        res[zero_day[t_idx]] = idx
                        zero_day.pop(t_idx)
                        rain_set[idx] = i
                else:
                    rain_set[idx] = i
            else:
                zero_day.append(i)
        for x in zero_day:
            res[x] = 1
        return res
