class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        n = len(rains)
        full = dict()
        move = list()
        res = [-1]*n
        for i in range(n):
            if rains[i] != 0:
                if rains[i] in full.keys():
                    if not move:
                        return []
                    if full[rains[i]] > move[-1]:
                        return []
                    else:
                        bar = bisect.bisect_right(move, full[rains[i]])
                        res[move[bar]] = rains[i]
                        move = move[:bar]+move[bar+1:]
                        full[rains[i]] = i
                else:
                    full[rains[i]] = i
            else:
                move.append(i)
        while move:
            res[move.pop(0)] = 311
        return res
