class Solution:
    def avoidFlood(self, rains: List[int]) -> List[int]:
        res = []
        drain = []
        seen = {}
        
        for i, x in enumerate(rains):
            if x == 0:
                res.append(1)
                drain.append(i)
            elif x not in seen:
                res.append(-1)
                seen[x] = i
            else:
                last = seen[x]
                use = bisect.bisect(drain, last)
                if use == len(drain):
                    return []
                res.append(-1)
                res[drain.pop(use)] = x
                seen[x] = i
        
        return res
