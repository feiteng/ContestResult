class DSU:
    def __init__(self, n):
        self.d = [i for i in range(n)]
    def find(self, n):
        if self.d[n] != n:
            self.d[n] = self.find(self.d[n])
        return self.d[n]
    def join(self, n1, n2):
        s1 = self.d[n1]
        s2 = self.d[n2]
        self.d[s1] = s2
class Solution:
    def findCriticalAndPseudoCriticalEdges(self, n: int, edges: List[List[int]]) -> List[List[int]]:
        idx = sorted(range(len(edges)), key = lambda x: edges[x][2])
        edges = [edges[i] for i in idx]

        def mst(rm = -1):
            sw = 0
            dsu = DSU(n)
            for i, (n1, n2, w) in enumerate(edges):
                if i == rm:
                    continue
                if dsu.find(n1) == dsu.find(n2):
                    continue
                dsu.join(n1, n2)
                sw += w
            if len(set([dsu.find(i) for i in range(n)])) != 1:
                return float('inf')
            return sw
        def psudo(ei):
            dsu = DSU(n)
            n1, n2, sw = edges[ei]
            dsu.join(n1, n2)
            
            for i, (n1, n2, w) in enumerate(edges):
                if i == ei:
                    continue
                if dsu.find(n1) == dsu.find(n2):
                    continue
                dsu.join(n1, n2)
                sw += w
            return sw  
        
        min_w = mst()
        res = [[], []]
        for i, ei in enumerate(idx):
            if min_w < mst(i):
                res[0].append(ei)
            else:
                if psudo(i) == min_w:
                    res[1].append(ei)
        res[0].sort()
        res[1].sort()
        return res
