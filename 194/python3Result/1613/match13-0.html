class UnionFind(object):
    def __init__(self, n, recursion = False):
        self._par = list(range(n))
        self._size = [1] * n
        self._recursion = recursion

    def root(self, k):
        if self._recursion:
            if k == self._par[k]:
                return k
            self._par[k] = self.root(self._par[k])
            return self._par[k]
        else:
            root = k
            while root != self._par[root]: root = self._par[root]
            while k != root: k, self._par[k] = self._par[k], root
            return root

    def unite(self, i, j):
        i, j = self.root(i), self.root(j)
        if i == j: return False
        if self._size[i] < self._size[j]: i, j = j, i
        self._par[j] = i
        self._size[i] += self._size[j]
        return True

    def is_connected(self, i, j):
        return self.root(i) == self.root(j)

    def size(self, k):
        return self._size[self.root(k)]

class Solution:
    def findCriticalAndPseudoCriticalEdges(self, n: int, edges: List[List[int]]) -> List[List[int]]:
        E = [[] for _ in range(n)]
        m = len(edges)
        for u, v, w in edges:
            E[u].append((u, w))
            E[v].append((v, w))
        
        IUVW = [None] * m
        for i, tup in enumerate(edges):
            u, v, w = tup
            IUVW[i] = (i, u, v, w)
        IUVW.sort(key = lambda x : x[3])
        
        uf = UnionFind(n)
        
        # build MST
        total_weight = 0
        for i, u, v, w in IUVW:
            if uf.unite(u, v):
                total_weight += w
        del uf
        
        # i0 を使わないと、MSTが作れないかどうか
        critical = [False] * m
        for i0 in range(m):
            # build MST
            tmp_weight = 0
            tmp_uf = UnionFind(n)
            for i, u, v, w in IUVW:
                if i == i0:
                    continue
                if tmp_uf.unite(u, v):
                    tmp_weight += w
            if tmp_weight > total_weight or tmp_uf.size(0) != n:
                critical[i0] = True
        
        # i0 を使ってMSTが作れるかどうか
        pseudo = [False] * m
        for i0 in range(m):
            # build MST
            tmp_weight = 0
            tmp_uf = UnionFind(n)
            # i0 は使う
            u, v, w = edges[i0]
            tmp_weight += w
            tmp_uf.unite(u, v)
            for i, u, v, w in IUVW:
                if tmp_uf.unite(u, v):
                    tmp_weight += w
            if tmp_weight == total_weight:
                pseudo[i0] = True
        
        A = []
        B = []
        for i in range(m):
            if critical[i]:
                A.append(i)
            elif pseudo[i]:
                B.append(i)
        
        return [A, B]
