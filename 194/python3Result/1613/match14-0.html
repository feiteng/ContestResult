def bridges(graph: List[List[int]]):
    """
    graph: directed graph, graph[u] = [v where u -> v]
    """

    n, time = len(graph), 0
    disc, low = [0] * n, [0] * n
    visited = [False] * n
    ret = []

    def dfs(u, parent):
        nonlocal time

        visited[u] = True
        disc[u] = low[u] = time
        time += 1
        for v, idx, _ in graph[u]:
            if not visited[v]:
                dfs(v, u)
                low[u] = min(low[u], low[v])
                if low[v] > disc[u]: ret.append(idx)
            elif v != parent:
                low[u] = min(low[u], low[v])

    for i in range(n):
        if not visited[i]: dfs(i, -1)
    return ret

class UnionFindSet:
    def __init__(self, n):
        self.p = [i for i in range(n)]
        self.rk = [0] * n

    def find(self, v):
        if self.p[v] != v: self.p[v] = self.find(self.p[v])
        return self.p[v]

    def union(self, v1, v2):
        p1 = self.find(v1)
        p2 = self.find(v2)
        if p1 == p2: return True
        if self.rk[p1] < self.rk[p2]: p1, p2 = p2, p1
        self.p[p2] = p1
        self.rk[p1] += self.rk[p1] == self.rk[p2]
        return False

def MSTKruskal(n, e, k=-1):
    # vertex: 0..n-1, e: list of [v1, v2, c] edge between v1 and v2 at cost c
    e.sort(key=lambda i: i[3])
    ufs = UnionFindSet(n)
    MST = 0
    for idx, v1, v2, c in e:
        if k == idx: continue
        if not ufs.union(v1, v2): MST += c
    return MST

def calc(n, e, s):
    # vertex: 0..n-1, e: list of [v1, v2, c] edge between v1 and v2 at cost c
    e.sort(key=lambda i: i[3])
    ufs = UnionFindSet(n)
    MST = 0
    m = len(e)
    cur = 0
    while cur < m:
        new = m - 1
        for i in range(cur, m):
            if e[i][3] != e[cur][3]:
                new = i - 1
                break
            idx, v1, v2, c = e[i]
            if ufs.find(v1) != ufs.find(v2): s[e[i][0]] = True
        
        for i in range(cur, new + 1):
            idx, v1, v2, c = e[i]
            if not ufs.union(v1, v2):
                MST += c
        cur = new + 1
    return MST

class Solution:

    def findCriticalAndPseudoCriticalEdges(self, n: int, edges: List[List[int]]) -> List[List[int]]:
        g = [[] for _ in range(n)]
        m = len(edges)
        c = [False] * m
        s = [False] * m
        e = []
        for i, (u, v, w) in enumerate(edges):
            g[u].append((v, i, w))
            g[v].append((u, i, w))
            e.append((i, u, v, w))
        bg = bridges(g)
        for b in bg: c[b] = True
        
        MST = calc(n, e, s)
        for i in range(m):
            if MSTKruskal(n, e, i) != MST: c[i] = True
        
        ret = [[], []]
        for i in range(m):
            if c[i]: ret[0].append(i)
            elif s[i]: ret[1].append(i)
        return ret
