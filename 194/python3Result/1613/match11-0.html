class Solution:
    def findCriticalAndPseudoCriticalEdges(self, n: int, A: List[List[int]]) -> List[List[int]]:
        res1 = []
        res2 = []
        
        
        
        def cal_mst(g, n):
            heap = []
            vs = set()
            vs.add(0)
            w = 0
            for k, v in g[0].items():
                heappush(heap, [v, k])
            while heap:
                e, d = heappop(heap)
                if d not in vs:
                    w += e
                    vs.add(d)
                    for k, v in g[d].items():
                        if k not in vs:
                            heappush(heap, [v,k])
            if len(vs) == n: return w
            return float('inf')
            
        def cal_mst2(g, n, x, y, z):
            heap = []
            vs = set()
            vs.add(x)
            vs.add(y)
            w = z
            for k, v in g[x].items():
                if k not in vs:
                    heappush(heap, [v, k])
            for k, v in g[y].items():
                if k not in vs:
                    heappush(heap, [v, k])
            while heap:
                e, d = heappop(heap)
                if d not in vs:
                    w += e
                    vs.add(d)
                    for k, v in g[d].items():
                        if k not in vs:
                            heappush(heap, [v,k])
            if len(vs) == n: return w
            return float('inf')
        
        g = [{} for _ in range(n)]
        for x, y, z in A:
            g[x][y] = z
            g[y][x] = z
        
        # print(g)
        w = cal_mst(g, n)
        
        
        size = len(A)
        for i in range(size):
            x, y, z = A[i]
            g[x][y] = float('inf')
            g[y][x] = float('inf')
            cur_w = cal_mst(g, n)
            if cur_w > w: res1.append(i)
            else:
                cur_w = cal_mst2(g, n, x, y, z)
                if cur_w == w: res2.append(i)
            g[x][y] = z
            g[y][x] = z
            
        return [res1, res2]
