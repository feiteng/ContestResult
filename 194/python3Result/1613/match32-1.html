from typing import List

import heapq

from collections import defaultdict
class Solution:
    def findCriticalAndPseudoCriticalEdges(self, n: int, edges: List[List[int]]) -> List[List[int]]:
        m = len(edges)

        def build(edges):
            eg 
= defaultdict(list)
            for a, b, c in edges:
                eg[a].append([b, c])
                eg[b].append([a, c])
            return eg

        def mst(edge):
            ans = 0
            dis, q = {}, [[0, 0]]
            while q:
                disa, a = heapq.heappop(q)
                if a not in dis:
                    dis[a] = disa
                    ans += disa
                    for b, c in edge[a]:
                        heapq.heappush(q, [c, b])
            return ans

        mi = mst(build(edges))
        ans = []
        for i in range(m):
            ne = edges[:i] + edges[i+1:]
            re = mst(build(ne))
            if mi != re:
                ans.append(i)
        ans2 = set(range(m)) - set(ans)

        dis = [[float('inf')] * n for _ in range(n)]
        for a, b, c in edges:
            dis[a][b] = dis[b][a] = c
        for k in range(n):
            for i in range(n):
                for j in range(i):
                    dis[i][j] = dis[j][i] = min(dis[i][j], max(dis[i][k], dis[k][j]))
        for i, (a, b, c) in enumerate(edges):
            if dis[a][b] < c:
                ans2.discard(i)

        return [ans, list(ans2)]
