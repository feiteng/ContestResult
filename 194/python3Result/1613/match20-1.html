class Solution:
    
    def findCriticalAndPseudoCriticalEdges(self, n: int, edges: List[List[int]]) -> List[List[int]]:
        ez=len(edges)
        edges = [(w,x,y,i) for i, (x,y,w) in enumerate(edges)]
        def mst(edges, i):
            if i==-1:
                edges=edges[:]
                heapify(edges)
            else:
                edges=edges[:i]+edges[i+1:]
                heapify(edges)
            g=[-1]*n
            
def find(g, x):
                if g[x]==-1:return x
                g[x]=find(g, g[x])
                return g[x]
            def union(g, x, y):
                x,y=find(g,x),find(g,y)
                if x==y:
                    return False
                g[x]=y
                return True
            gcnt=n
            totalw=0
            maxw=0
            oldg=g[:]
            prevw=-1            
            discarded=set()
            while edges and (gcnt>1 or i==-1):
                #print("gcnt:%d,i:%d,edges:%s"%(gcnt,i,edges))
                w,x,y,index=edges[0]
                heappop(edges)

                if w!=prevw and i==-1:
                    oldg=g[:]
                prevw=w
                
                if i==-1:
                    if find(oldg, x)==find(oldg,y):
                        discarded.add(index)
                if union(g, x,y):
                    totalw+=w
                    gcnt-=1
                    maxw=w
            if gcnt>1:
                return 10000000, 0, discarded
            return totalw,maxw,discarded
        totalw,maxw,discarded=mst(edges,-1)
        # print("totalw:%d,maxw:%d,discarded:%s"%(totalw,maxw,discarded))
        ans=[[],[]]
        for w,x,y,i in edges:
            #print("edges:%s"%(edges))
            if i in discarded: continue 
            if w>maxw:
                continue
            ttw, tmw,_=mst(edges, i)
            #print("ttw:%d,tmw:%d,i:%d"%(ttw,tmw,i))
            if ttw>totalw:
                ans[0].append(i)
            else:
                ans[1].append(i)
        return ans
