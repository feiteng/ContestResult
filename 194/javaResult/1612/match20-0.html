class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        int[] out = new int[n];
        Arrays.fill(out, -1);
        TreeSet<Integer> dry = new TreeSet<>();
        Map<Integer, Integer> wet = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (rains[i] == 0) {
                out[i] = 1;
                dry.add(i);
            } else {
                int l = rains[i];
                if (wet.containsKey(l)) {
                    Integer j = dry.higher(wet.get(l));
                    if (j == null) {
                        return new int[0];
                    } else {
                        out[j] = l;
                        dry.remove(j);
                    }
                }
                wet.put(l, i);
            }
        }
        return out;
    }
}
