class Solution {
    public int[] avoidFlood(int[] rs) {
        int n = rs.length;
        int[] res = new int[n];
        Arrays.fill(res,-1);
        TreeSet<Integer> set = new TreeSet<>();
        Map<Integer,Integer> mp = new HashMap<>();
        for(int i=0;i<n;i++){
            if(rs[i]==0){
                set.add(i);
            }else{
               if(mp.containsKey(rs[i])){
                   int pos = mp.get(rs[i]);
                   if(set.size()==0) return new int[]{};
                   Integer idx = set.higher(pos);
                    if(idx==null) return new int[]{};
                   res[idx] = rs[i];
                   set.remove(idx);
               }
                mp.put(rs[i],i);
            }
        }
        for(int j:set){
            res[j] = 1;
        }
        return res;
    }
}
