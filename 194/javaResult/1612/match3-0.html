class Solution {
    public int[] avoidFlood(int[] rs) {
        Map<Integer, Integer> full = new HashMap<>();
        TreeSet<Integer> op = new TreeSet<>();
        int n = rs.length;
        int[] ans = new int[n];
        
        for (int i = 0; i < n; i++) {
            if (rs[i] == 0) {op.add(i);}
            
            else {
                int curLake = rs[i];
                
                if (full.containsKey(curLake)) {
                    
                    int filledDate = full.get(curLake);
                    
                    Integer ep = op.higher(filledDate);
                    if (ep == null) {return new int[0];}
                    
                    
                    
                    ans[ep] = curLake;
                    op.remove(ep);
                    
                    full.put(curLake, i);
                } else {
                    full.put(curLake, i);
                }
                ans[i] = -1;
            }
        }
        
        for (int i = 0; i < n; i++) {
            if (ans[i] == 0) {ans[i] = 1;}
        }
        return ans;
    }
}
