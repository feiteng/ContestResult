class Solution {
    public int[] avoidFlood(int[] rains) {
        int len = rains.length;
        int[] res = new int[len];
        List<Integer> zeros = new LinkedList<>();
        Map<Integer, Integer> map = new HashMap<>();
        for(int i = 0; i < len; i++) {
            int cur = rains[i];
            if(cur == 0) {
                zeros.add(i);
                continue;
            } 
 
            
            if(map.containsKey(cur)) {
                
                int firstZero = findZero(zeros, map.get(cur));
                
                if(firstZero == -1) {
                    //System.out.println(i + " " + cur + " " + zeros.size());
                    return new int[]{};
                }
                res[zeros.get(firstZero)] = cur;
                zeros.remove(firstZero);
            }
            res[i] = -1;
            map.put(cur, i);
        }
        for(int i = 0; i < len; i++) {
            if(res[i] == 0) {
                res[i] = 1;
            }
        }
        return res;
    }
    
    public int findZero(List<Integer> zeros, int start) {
        for(int i = 0; i < zeros.size(); i++) {
            int cur = zeros.get(i);
            if(cur > start) {
                return i;
            }
        }
        return -1;
    }
}
