	class Solution {
	    public int[] avoidFlood(int[] rains) {
	    	Map<Integer, Integer> fl = new HashMap<>();
	    	int n = rains.length;
	    	int[] ret = new int[n];
	    	TreeSet<Integer> space = new TreeSet<>();
	    	for(int i = 0;i < n;i++){
	    		if(rains[i] > 0){
	    			ret[i] = -1;
	    			if(fl.containsKey(rains[i])){
	    				int last = fl.get(rains[i]);
	    				Integer h = space.higher(last);
	    				if(h == null){
	    					return new int[0];
	    				}
	    				ret[h] = rains[i];
	    				space.remove(h);
	    			}
    				fl.put(rains[i], i);
	    		}else{
	    			space.add(i);
	    			ret[i] = 1;
	    		}
	    	}
	    	return ret;
	    }
	}
