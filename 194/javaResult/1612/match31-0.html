class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        int[] ret = new int[n];
        for(int i=0;i<n;i++){
            ret[i] = -1;
        }
        HashMap<Integer, Integer> lakes = new HashMap<>();
        TreeSet<Integer> index = new TreeSet<>();
        for(int i=0;i<n;i++){
            if(rains[i] > 0){
                if(lakes.containsKey(rains[i]) ){
                    if( index.isEmpty()){
                        return new int[0];                    
                    } else {
                        int idx = lakes.get(rains[i]);
                        Integer next = index.higher(idx);
                        if(next==null){
                             return new int[0];     
                        }
                        index.remove(next);
                        ret[next] = rains[i];
                        lakes.put(rains[i],i);
                    }
                } else {
                    lakes.put(rains[i],i);
                }
            } else {
                index.add(i);
            }
        }
        for(int i: index){
            ret[i] = 1;
        }
        return ret;
    }
}
