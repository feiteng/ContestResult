class Solution {
    public int[] avoidFlood(int[] rains) {
        Map<Integer, Integer> lake = new HashMap<>();
        TreeSet<Integer> dry = new TreeSet<>();
        int[] res = new int[rains.length];
        for(int i = 0; i < rains.length; i++) {
            if(rains[i] == 0) {
                dry.add(i);
            } else {
                res[i] = -1;
                if(!lake.containsKey(rains[i])) {
                    lake.put(rains[i], i);
                } else {
                    if(dry.isEmpty() || dry.ceiling(lake.get(rains[i])) == null) return new int[0];
                    int day = dry.ceiling(lake.get(rains[i]));
                    res[day] = rains[i];
                    lake.put(rains[i], i);
                    dry.remove(day);
                }                   
            
}
        }
        // System.out.println(lake);
        while(!dry.isEmpty()) {
            res[dry.pollFirst()] = 1;
        }
        return res;
    }
}
