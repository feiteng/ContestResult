class Solution {
    public int[] avoidFlood(int[] rains) {
        
TreeSet<Integer> unused = new TreeSet<>();
        Map<Integer, Integer> fillTime = new HashMap<>();
        for (int i = 0; i < rains.length; i++) {
            if (rains[i] == 0) {
                unused.add(i);
                
            } else {
                if (!fillTime.containsKey(rains[i])) {
                    fillTime.put(rains[i], i);
                
} else {
                    // We will attempt to find a suitable unused index.
                    Integer filledAt = fillTime.get(rains[i]);
                    Integer time = unused.ceiling(filledAt);
                    if (time == null) {
                        return new int[0];
                    }
                    unused.remove(time);
                    rains[time] = rains[i];
                    fillTime.put(rains[i], i);
                }
            }
            rains[i] = -1;
        
}
        for (int index : unused) {
            rains[index] = 1;
        }
        return rains;
    }
}
