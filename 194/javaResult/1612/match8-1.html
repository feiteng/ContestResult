class Solution {
    public int[] avoidFlood(int[] rains) {
        int[] res = new int[rains.length];
        TreeSet<Integer> slots = new TreeSet<>();
        Map<Integer, Integer> lakes = new HashMap<>();
        for (int i = 0; i < rains.length; i++) {
            if (rains[i] > 0) {
                Integer index = lakes.get(rains[i]);
                if (index == null) {
                    lakes.put(rains[i], i);
                } else {
                    
Integer slot = slots.higher(index);
                    if (slot == null) {
                        return new int[0];
                    } else {
                        res[slot] = rains[i];
                        lakes.put(rains[i], i);
                        
slots.remove(slot);
                    }
                }
                res[i] = -1;
            } else {
                slots.add(i);
                res[i] = 1;
            }
            //System.out.println(queue);
        }
        
        return res;
    }
}
