class Solution {
    public int[] avoidFlood(int[] rains) {
  TreeSet<Integer> empty = new TreeSet<>();
        HashMap<Integer, Integer> map = new HashMap<>();
        int[] out = new int[rains.length];
        for (int i = 0; i < rains.length; ++i) {
            int place = rains[i];
            if (place == 0) {
                empty.add(i);
                out[i] = 1;
                continue;
            }
            out[i] = -1;
            
if (map.containsKey(place)) {
                Integer first = empty.ceiling(map.get(place));
                if (first == null) return new int[0];
                empty.remove(first);
                out[first] = place;
            }
            map.put(place, i);
        }
        return out;
    }
}
