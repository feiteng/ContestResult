class Solution {
    public int[] avoidFlood(int[] rains) {
        Map<Integer, Integer> res = new HashMap<>();
        TreeSet<Integer> zeroIndex = new TreeSet<>();
        Map<Integer, Integer> filled = new HashMap<>();
        for(int i = 0; i < rains.length; i++) {
            if(rains[i] == 0) {
                zeroIndex.add(i);
            }else {
                if(filled.containsKey(rains[i])) {
                    Integer zeroPos = zeroIndex.higher(filled.get(rains[i]));
                    if(zeroPos == null) return new int[0];
                    zeroIndex.remove(zeroPos);
                    res.put(zeroPos, rains[i]);
                }
                filled.put(rains[i], i);
            }
        }
        
        int[] arr = new int[rains.length];
        for(int i = 0; i < rains.length; i++) {
            if(rains[i] != 0) {
                arr[i] = -1;
            }else {
                arr[i] = res.getOrDefault(i, 1);
            }
        }
        return arr;
    }
}
