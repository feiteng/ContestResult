class Solution {
    public int[] avoidFlood(int[] rains) {
        TreeSet<Integer> dryDate = new TreeSet<>();
        Map<Integer, Integer> lakeMap = new HashMap<>();
        int n = rains.length;
        int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            if (rains[i] == 0) {
                dryDate.add(i);
            } else {
                int lake = rains[i];
                if (lakeMap.containsKey(lake)) {
                    int index = lakeMap.get(lake);
                    Integer dryIndex = dryDate.ceiling(index);
                    if (dryIndex == null) {
                        return new int[0];
                    } else {
                        res[dryIndex] = lake;
                        dryDate.remove(dryIndex);
                    }
                }
                lakeMap.put(lake, i);
                res[i] = -1;
            }
        }
        for (int index : dryDate) {
            res[index] = 1;
        }
        return res;
    }
}
