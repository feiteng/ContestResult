class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        int[] res = new int[n];
        int[] emps = new int[n];
        int empty = 0;
        
Map<Integer, Integer> map = new HashMap();
        TreeSet<Integer> set = new TreeSet();
        int[] empties = new int[n+1];
        for(int i=0;i<n;i++){
            int num = rains[i];
            if(num>0){
                
if(map.containsKey(num)){
                    int index = map.get(num);
                    Integer higher = set.higher(index);
                    if(higher==null){
                        return new int[0];
                    }
                    set.remove(higher);
                    res[higher] = num;
                }
                res[i] = -1;
                
map.put(num, i);
            }else{
                set.add(i);
            }
        }
        for(int e:set){
            res[e] = 1;
        }
        return res;
        
    }
}
