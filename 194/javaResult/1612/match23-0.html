class Solution {
    public int[] avoidFlood(int[] rains) {
        // rains = new int[]{0,1,1};
        TreeSet<Integer> canDry = new TreeSet<>();
        Map<Integer, Integer> full = new HashMap<>();
        int[] res = new int[rains.length];
        for (int i = 0; i < rains.length; i++) {
            if (rains[i] > 0) {
                res[i] = -1;
                if (full.containsKey(rains[i])) {
                    int fullDate = full.get(rains[i]);
                    Integer dry = canDry.higher(fullDate);
                    if (dry == null) {
                        return new int[0];
                    } else {
                        canDry.remove(dry);
                        res[(int) dry] = rains[i];
                    }
                    
                } 
                full.put(rains[i], i);
            } else {
                canDry.add(i);               
            }
        }
        for (int i : canDry) {
            res[i] = 1;
        }
        return res;
    }
}
