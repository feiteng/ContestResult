class Solution {
    public int[] avoidFlood(int[] rains) {
        int N = rains.length;
        int[] ans = new int[N];
        Arrays.fill(ans, 1);
        Map<Integer, Integer> lakes = new HashMap<>();
        TreeSet<Integer> bst = new TreeSet<>();
        for (int i = 0; i < N; i++) {
            int rain = rains[i];
            if (rain == 0) {
                bst.add(i);
            } else {
                if (lakes.containsKey(rain)) {
                    int index = lakes.get(rain);
                    Integer higher = 
bst.higher(index);
                    if (higher == null) {
                        return new int[0];
                    } else {
                        ans[higher] = rain;
                        bst.remove(higher);
                    }
                } 
                ans[i] = -1;
                
lakes.put(rain, i);
            }
        }
        return ans;
    }
}
