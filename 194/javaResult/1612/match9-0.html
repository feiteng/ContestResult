class Solution {
    public int[] avoidFlood(int[] rains) {
        int[] result = new int[rains.length];
        Map<Integer, Integer> lastFilled = new HashMap<>();
        TreeSet<Integer> treeSet = new TreeSet<>();
        for (int i = 0;i<rains.length;i++) {
            if (rains[i] != 0) {
                if (!lastFilled.containsKey(rains[i])) {
                    lastFilled.put(rains[i], i);
                } else {
                    int fillDay = lastFilled.get(rains[i]);
                    Integer availDrainDay = treeSet.ceiling(fillDay);
                    if (availDrainDay == null) {
                        return new int[]{};
                    } else {
                        treeSet.remove(availDrainDay);
                        result[availDrainDay] = rains[i];
                        lastFilled.put(rains[i], i);
                    }
                }
                result[i] = -1;
            } else {
                treeSet.add(i);
                result[i] = 1;
            }
        }
        return result;
    }
}
