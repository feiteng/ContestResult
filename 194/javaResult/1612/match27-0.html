import java.util.*;

class Solution {
    public int[] avoidFlood(int[] rains) {
        Map<Integer, Integer> lakeToDay = new HashMap<>();
        TreeSet<Integer> drillDays = new TreeSet<>();
        int[] result = new int[rains.length];
        Arrays.fill(result, -1);
        for (int day = 0; day < rains.length; day++) {
            int lake = rains[day];
            if (lake == 0) {
                drillDays.add(day);
                continue;
            }
            if (!lakeToDay.containsKey(lake)) {
                lakeToDay.put(lake, day);
                continue;
            }
            Integer lastRainingDay = lakeToDay.get(lake);
            Integer drillDay = drillDays.ceiling(lastRainingDay + 1);
            if (drillDay == null) {
                return new int[0];
            }
            drillDays.remove(drillDay);
            lakeToDay.put(lake, day);
            result[drillDay] = lake;
        }
        for (int remainingDrillDay : drillDays) {
            result[remainingDrillDay] = 1;
        }
        return result;
    }
}
