class Solution {
    public int[] avoidFlood(int[] rains) {
        final int N = rains.length;
        
        
TreeSet<Integer> saves = new TreeSet<>();
        HashMap<Integer, Integer> full = new HashMap<>();
        int[] ans = new int[N];
        for (int i = 0; i < N; ++i) {
            if (rains[i] == 0) {
                saves.add(i);
            } else {
                
if (full.containsKey(rains[i])) {
                    int previous = full.get(rains[i]);
                    Integer day = saves.higher(previous);
                    if (day == null) {
                        return new int[0];
                    }
                    ans[day] = rains[i];
                    saves.remove(day);
                }
                full.put(rains[i], i);
                ans[i] = -1;
            
}
        }
        for (int d : saves) {
            ans[d] = 1;
        }
        return ans;
    }
}
