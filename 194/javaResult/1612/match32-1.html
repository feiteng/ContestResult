class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        
int[] res = new int[n];
        // queue to save all the days that can dry the lake
        List<Integer> days = new ArrayList();
        // Set to save all the full lake
        Map<Integer, Integer> map = new HashMap();
        for (int i = 0; i < n; i++) {
            int c = rains[i];
            if (c == 0) {
                days.add(i);
                
            } else 
{
                res[i] = -1;
                if (map.containsKey(c)) {
                    int d = findFirstDay(days, map.get(c), i);
                    //System.out.println(c + " " + d);
                    if (d == -1) return new int[]{};
                    res[d] = c;
                    map.put(c, i);
                
} else {
                    map.put(c, i);
                }
            }
            //System.out.println(days + " " + rains[i]);
        }
        for (int i = 0; i < n; i++) {
            if (res[i] == 0) res[i] = 1;
        }
        return res;
    }
    
    private int findFirstDay(List<Integer> days, int start, int end) {
        int n = days.size();
        if (n == 0) return -1;
        int l = 0, r = n - 1;
        while (l < r) {
            int mid = l + (r - l) / 2;
            int c = days.get(mid);
            if (c < start) {
                l = mid + 1;
            } else if (c > end) {
                r = mid - 1;
            } else {
                r = mid;
            }
        }
        int c = days.get(l);
        if (c >= start && c <= end) {
            days.remove(l);
            return c;
        }
        return -1;
    }
}
