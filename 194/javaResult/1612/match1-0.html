class Solution {
    public int[] avoidFlood(int[] rains) {
        int[] res = new int[rains.length];
        Map<Integer, Integer> fullRivers = new HashMap<>();
        TreeSet<Integer> whiteDay = new TreeSet<>();
        for (int i = 0; i < rains.length; i++) {
            if (rains[i] == 0) {
                whiteDay.add(i);
                continue;
            }
            int cur = rains[i];
            if (!fullRivers.containsKey(cur)) {
                fullRivers.put(cur, i);
                //System.out.println(fullRivers);
            } else {
                int lastRained = fullRivers.get(cur);
                Integer day = whiteDay.higher(lastRained);
                // System.out.println("Last rained day:" + lastRained + ", whiteday:" + day);
                if (day == null) {
                    return new int[]{};
                }
                res[day] = cur;
                fullRivers.put(cur, i);
                //System.out.println(fullRivers);
                whiteDay.remove(day);
            }
            res[i] = -1;
        }
        for (int i = 0; i < res.length; i++) {
            if (res[i] == 0) {
                res[i] = 1;
            }
        }
        return res;
    }
}
