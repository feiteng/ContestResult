class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        
int [] ans = new int[n];
        TreeMap<Integer, Integer> map = new TreeMap<Integer, Integer>();
        TreeSet<Integer> set = new TreeSet<>();
        for(int i = 0; i < n; i++){
            int elem = rains[i];
            if(elem == 0){
                ans[i] = -2;
                
set.add(i);
            }else{
                if(map.containsKey(elem)){
                    Integer last = map.get(elem);
                    Integer prev = set.higher(last);
                    if(prev == null || prev < last){
                        return new int[]{};
                    }
                    ans[prev] = elem;
                    set.remove(prev);
                }
                map.put(elem, i);
                ans[i] = -1;
            }
        }
        for(int i = 0; i < n; i++){
            if(ans[i] == -2){
                ans[i] = 1;
            }
        }
        return ans;
    }
}
