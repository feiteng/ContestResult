class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        int[] dry = new int[n];
        Map<Integer, Integer> filled = new HashMap<>();
        TreeSet<Integer> free = new TreeSet<>();
        
        for(int i = 0; i < n; i++) {
            if(rains[i] == 0) {
                free.add(i);
            } else {
                if(filled.containsKey(rains[i])) {
                    if(free.isEmpty()) return new int[0];
                    Integer freeDay = free.higher(filled.get(rains[i]));
                    if(freeDay == null) return new int[0];
                    dry[freeDay] = rains[i];
                    free.remove(freeDay);

                }
                filled.put(rains[i], i);
                dry[i] = -1;
            
}
        }
        
        for(Integer i : free) {
            dry[i] = 1;
        }
        
        return dry;
    }
}
