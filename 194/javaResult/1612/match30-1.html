class Solution {
    public int[] avoidFlood(int[] rains) {
        Map<Integer, Integer> fullLakes = new HashMap<>();
        TreeSet<Integer> set = new TreeSet<>();
        int[] out = new int[rains.length];
        
        for(int i=0; i<rains.length;i++){
            out[i] = -1;
            if(rains[i] == 0){
                out[i] = 1;
                
set.add(i);
            } else{
                if(fullLakes.containsKey(rains[i])){
                    Integer lastPos = set.ceiling(fullLakes.get(rains[i]));
                    if(lastPos == null) return new int[0];
                    set.remove(lastPos);
                    out[lastPos] = rains[i];
                }
                
                
fullLakes.put(rains[i], i);
            }               
        }
        
        return out;
    }
}
