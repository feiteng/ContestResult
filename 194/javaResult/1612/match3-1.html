class Solution {
    public int[] avoidFlood(int[] rains) {
        int[] res = new int[rains.length];
        
        TreeSet<Integer> toDry = new TreeSet<>();
        Map<Integer, Integer> full = new HashMap<>();
        
        
for (int i = 0; i < res.length; i++) {
            if (rains[i] == 0) {
                toDry.add(i);
            } else {
                
if (full.containsKey(rains[i])) {
                    int prev = full.get(rains[i]);
                    
                    Integer toD = toDry.ceiling(prev + 1);
                    if (toD == null) {
                        return new int[0];
                    }
                    res[toD] = rains[i];
                    toDry.remove(toD);
                }
                full.put(rains[i], i);
                
res[i] = -1;
            }
        }
        for (int i = 0; i < res.length; i++) {
            if (res[i] == 0) {
                res[i] = 1;
            }
        }
        
        return res;
    }
}
