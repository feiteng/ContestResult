class Solution {
    public int[] avoidFlood(int[] rains) {
        TreeSet<Integer> empty = new TreeSet<>();
        Map<Integer, Integer> map = new HashMap<>();
        int n = rains.length;
        
        int[] res = new int[n];
        for (int i = 0; i < n; i++) {
            if (rains[i] == 0) {
                empty.add(i);
            } else {
                if (map.containsKey(rains[i])) {
                    int at = map.get(rains[i]);
                    
                    Integer pos = empty.higher(at);

                    if (pos == null) return new int[0];
                    res[pos] = rains[i];
                    empty.remove(pos);
                }

                map.put(rains[i], i);
                res[i] = -1;
            }
        }
        
        for (int i : empty) {
            res[i] = 1;
        }
        
        return res;
    }
}
