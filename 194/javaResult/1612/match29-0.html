class Solution {
public int[] avoidFlood(int[] rains) {
        int[] res=new int[rains.length];
        int[] error=new int[0];
        Map<Integer,Integer> map=new HashMap<>();
       ArrayList<Integer> list=new ArrayList<>();
        for (int i = 0; i < rains.length; i++) {
            if (rains[i]==0){
                list.add(i);
            }else {
                res[i]=-1;
                if (map.containsKey(rains[i])){
                    int pos=map.get(rains[i]);
                    int temp=Collections.binarySearch(list,pos);
                    if (temp<0){
                        temp=-temp-1;
                        if (temp>=list.size())
                            return error;
                    }
                    int pos2=list.get(temp);
                    list.remove(temp);
                    res[pos2]=rains[i];
                    map.put(rains[i],i);
                }else {

                    map.put(rains[i],i);
                }
            }
        }
        for (Integer integer : list) {
            res[integer]=1;
        }
        return res;
    }
}
