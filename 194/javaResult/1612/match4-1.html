class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        int[] res = new int[n];
        int LIMIT = 1000 * 1000 * 1000;
        Arrays.fill(res, LIMIT);
        
TreeSet<Integer> ts = new TreeSet<>();
//        for (int i = 0; i < n; i++) {
//            if (rains[i] == 0) {
//                ts.add(i);
//            }
//        }
        Map<Integer, Integer> hm = new HashMap<>();
        for (int i = 0; i < n; i++) {
            if (rains[i] == 0) {
                ts.add(i);
            } else {
                if (!hm.containsKey(rains[i])) {
                    hm.put(rains[i], i);
                } else {
                    
int prev = hm.get(rains[i]);
                    // [prev + 1, i - 1]
                    Integer idx = ts.ceiling(prev);
                    if (idx == null) {
                        return new int[]{};
                    }
                    res[idx] = rains[i];
                    ts.remove(idx);
                    hm.put(rains[i], i);
                }
                
res[i] = -1;
            }

        }
        return res;
    }
}
