class Solution {
    public int[] avoidFlood(int[] rains) {
        int n = rains.length;
        int ans[] = new int[n];
        TreeSet<Integer> set = new TreeSet<>();
        HashMap<Integer, Integer> filledLakes = new HashMap<>();
        for(int i = 0; i < n; i++){
            if(rains[i] > 0){
                ans[i] = -1;
                if(filledLakes.containsKey(rains[i])){
                    if(set.isEmpty()){
                        return new int[0];
                    }
                    else
{
                        Integer bestDay = set.ceiling(filledLakes.get(rains[i]));
                        if(bestDay == null){
                            return new int[0];
                        }
                        ans[bestDay] = rains[i];
                        set.remove(bestDay);
                    }
                }
                filledLakes.put(rains[i], i);
            }
            else{
                
set.add(i);
            }
        }
        for(int emptyDay: set){
            ans[emptyDay] = 1;
        }
        return ans;
    }
}
