class Solution {
    public int[] avoidFlood(int[] rains) {
        int len = rains.length;
        
int[] result = new int[len];
        TreeSet<Integer> drySet = new TreeSet<>();
        Map<Integer, Integer> seenLake = new HashMap<>();
        
        for (int i = 0; i < len; i ++) {
            if (rains[i] != 0) {
                result[i] = -1;
                if (seenLake.get(rains[i]) != null) 
{
                    Integer dayToRain = seenLake.get(rains[i]);
                    Integer dryDay = drySet.ceiling(dayToRain);
                    if (dryDay == null) {
                        return new int[]{};
                    }
                    result[dryDay] = rains[i];
                    drySet.remove(dryDay);
                }
                seenLake.put(rains[i], i);
            
} else {
                drySet.add(i);
                result[i] = 1;
            }
        }
        
        return result;
    }
}
