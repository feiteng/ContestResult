class Solution {
   int[] IMPOSSIBLE = new int[0];
    //Use the hashMap to remember the lakes and the date
    //Use TreeSet to remember all dryDates
    //
    public int[] avoidFlood(int[] rains) {
        int[] ret = new int[rains.length];
        int len = rains.length;
        TreeSet<Integer> dryDates = new TreeSet<Integer>();
        Map<Integer,Integer> lakesDateMap = new HashMap<Integer,Integer>();
        
for(int i = 0;i<len;i++){

            int rain = rains[i];
//System.out.println("rain is " + rain);
            if(rain == 0){
                dryDates.add(i);
                continue;
            }
            
            //lake haven't fill yet, no need to fix.
            if(!lakesDateMap.containsKey(rain)){
                lakesDateMap.put(rain, i);
                ret[i] = -1;
                
continue;
            }
            
            //lake has filled before.... need to find a dry date to dry it.
            int lastFillDate = lakesDateMap.get(rain);
            Integer dryDate = dryDates.ceiling(lastFillDate);
            if(dryDate == null){
//System.out.println("impossible");
                return IMPOSSIBLE;
            }
            
            lakesDateMap.put(rain, i);
            dryDates.remove(dryDate);
            ret[dryDate] = rain;
            
ret[i] = -1;
            
        }
        
        for(int dryDate: dryDates){
            ret[dryDate] = 1;
        }
        //PrintUtils.print(ret);
        
        return ret;
    }
}
