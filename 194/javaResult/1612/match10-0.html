class Solution {
    public int[] avoidFlood(int[] rains) {
        int N = rains.length;
        int[] ans = new int[N];        
        
        Map<Integer, Integer> lakeIndices = new HashMap<>();
        TreeSet<Integer> dryDays = new TreeSet<>();
        
        // When we encounter a dry day, can only apply after a rain, and before second time. Cannot "store" a dry day.
    // We can bianry search to apply leftmost dry day after previous rain day.                     
        for (int i = 0; i < N; i++) {
            if (rains[i] == 0) {
                dryDays.add(i);
            }
            else {
                if (lakeIndices.containsKey(rains[i])) {
                    // We must have at least 1 dry day to get rid of the previous rain.
                    // Find leftmost index of a dry day 
                    Integer dryDay = dryDays.higher(lakeIndices.get(rains[i]));
                    
                    if (dryDay == null) {
                        return new int[] {};
                    }
                                        
                    ans[dryDay] = rains[i];
                    dryDays.remove(dryDay);
                }
                
                ans[i] = -1;
                lakeIndices.put(rains[i], i);
            }
        }
        
        // If there are any unused dry days, dry any day.
        for (int dry : dryDays) {
            ans[dry] = 1;
        }
        
        // If impossible to avoid flood, return empty array.
        return ans;
    }    
}
