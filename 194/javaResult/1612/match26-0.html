class Solution
{
    public int[] avoidFlood(int[] rains)
    {
        TreeSet<Integer> dryChance = new TreeSet<>();
        HashMap<Integer, Integer> rain = new HashMap<>();
        int[] ans = new int[rains.length];
        for (int i = 0; i < rains.length; i++)
        {
            if (rains[i] == 0)
            {
                dryChance.add(i);
            }
            else
            {
                ans[i] = -1;
                if (rain.containsKey(rains[i]))
                {
                    if (dryChance.size() == 0) return new int[0];
                    Integer res = dryChance.ceiling(rain.get(rains[i]));
                    if (res == null) return new int[0];
                    dryChance.remove(res);
                    ans[res] = rains[i];
                }
                rain.put(rains[i], i);
            }
        }
        while (!dryChance.isEmpty())
        {
            ans[dryChance.pollFirst()] = 1;
        }
        return ans;
    }
}
