class Solution {
    public int[] avoidFlood(int[] rains) {
        int[] res = new int[rains.length];
        TreeSet<Integer> free = new TreeSet<>();
        HashMap<Integer, Integer> seen = new HashMap<>();
        for (int i = 0; i < rains.length; ++i) {
            if (rains[i] == 0) free.add(i);
            else {
                res[i] = -1;
                if (seen.containsKey(rains[i])) {
                    Integer next = free.higher(seen.get(rains[i]));
                    if (next == null) return new int[0];
                    res[next] = rains[i];
                    free.remove(next);
                    seen.put(rains[i], i);
                }
                else {
                    seen.put(rains[i], i);
                }
            }
        }
        
        for (int i = 0; i < rains.length; ++i)
            if (res[i] == 0) res[i] = 1;
        return res;
    }
}
