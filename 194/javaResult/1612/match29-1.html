class Solution {
    public int[] avoidFlood(int[] rains) {
        
int[] res = new int[rains.length];
        TreeSet<Integer> dryDays = new TreeSet<>();
        Map<Integer, Integer> fullLakes = new HashMap<>();
        
        for (int i = 0; i < rains.length; ++i){       
            if (rains[i] == 0){
                dryDays.add(i);
            } else {
                res[i] = -1;
                if (fullLakes.containsKey(rains[i])){
                    if (dryDays.isEmpty() || dryDays.higher(fullLakes.get(rains[i])) == null) return new int[0];
                    int index = dryDays.higher(fullLakes.get(rains[i]));
                    dryDays.remove(index);
                    fullLakes.put(rains[i], i);
                    res[index] = rains[i];
                
} else {
                    fullLakes.put(rains[i], i);
                }
            }
        }
        for (Integer day : dryDays){
            res[day] = 1;
        }
        return res;
    }
}
