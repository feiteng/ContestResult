class Solution {
    int[] father;

    boolean bcj(int a, int b) {
        while (a != father[a])
            a = father[a];
        while (b != father[b])
            b = father[b];
        return a == b;
    }

    void bcjadd(int a, int b) { // add a to b
        while (a != father[a])
            a = father[a];
        father[a] = b;
    }

    int kru(int n, int[][] edges) {
        Arrays.sort(edges, (ints, t1) -> ints[2] - t1[2]);
        int curnum = 0;
        for (int i = 0; i < edges.length; i++) {
            if (!bcj(edges[i][0], edges[i][1])) {
                bcjadd(edges[i][0], edges[i][1]);
                curnum += edges[i][2];
            
}
        }
        return curnum;
    }
    int[][] clo(int[][] b) {
        int[][] a = new int[b.length][3];
        for (int i = 0; i < a.length; i++) {
            for (int j = 0; j < 3; j++)
                a[i][j] = b[i][j];
        
}
        return a;
    }
    public List<List<Integer>> findCriticalAndPseudoCriticalEdges(int n, int[][] edges) {
        father = new int[n];
        for (int i = 0; i < n; i++)
            father[i] = i;
        List<List<Integer>> res = new ArrayList<>();
        List<Integer> l = new ArrayList<>();
        List<Integer> r = new ArrayList<>();
        int[][] tedges = clo(edges);
        int minvalue = kru(n, tedges);
        for (int i = 0; i < edges.length; i++) {
            tedges = clo(edges);
            tedges[i][2] = Integer.MAX_VALUE / 2;
            for (int j = 0; j < n; j++)
                father[j] = j;
            if (kru(n, tedges) > minvalue) {
                l.add(i);
            } else {
                tedges = clo(edges);
                tedges[i][2] -= 1;
                for (int j = 0; j < n; j++)
                    father[j] = j;
                if (kru(n, tedges) < minvalue)
                    r.add(i)
;
            }
        }
        res.add(l);
        res.add(r);
        return res;
    }
}
