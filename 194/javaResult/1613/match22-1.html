class Solution {
    class WeightedUF {
        int[] id;
        int count;
        int[] sz;
        public WeightedUF(int n) {
            count = n;
            id = new int[n];
            for(int i = 0; i < n; i++) {
                id[i] = i;
            }
            sz = new int[n];
            
Arrays.fill(sz, 1);
        }

        public boolean isConnected(int p, int q) {
            return find(p) == find(q);
        }
        public int find(int p) {
            while(p != id[p]) {
                p = id[p];
            }
            return p;
        }

        public boolean union(int p, int q) {
            int i = find(p);
            int j = find(q);
            if(i == j) {
                return false;
            }
            if(sz[i] < sz[j]) {
                id[i] = j;
                sz[j] += sz[i];
            }
            else {
                id[j] = i;
                sz[i] += sz[j];
            }
            count--;
            return true;
        }
    }


    
public List<List<Integer>> findCriticalAndPseudoCriticalEdges(int n, int[][] edges) {
        List<List<Integer>> res = new ArrayList<>();
        for (int i = 0; i < 2; i++) {
            res.add(new ArrayList<>());
        }

        int mst = findMst(n, edges);
        for (int i = 0; i < edges.length; i++) {
            if (isCritical(n, edges, i, mst)) {
                res.get(0).add(i);
            } else if (isPseudo(n, edges, i, mst)) {
                res.get(1).add(i);
            }
        }
        return res;
    }

    boolean isCritical(int nodes, int[][] edges, int idx, int mst) {
        List<Edge> arr = new ArrayList<>();
        for (int i = 0; i < edges.length; i++) {
            if (i == idx) {
                continue;
            }
            arr.add(new Edge(i, edges[i][0], edges[i][1], edges[i][2]));
        }
        Collections.sort(arr);
        WeightedUF uf = new WeightedUF(nodes);
        int res = 0;
        for (int i = 0; i < arr.size(); i++) {
            Edge cur = arr.get(i);
            if (uf.isConnected(cur.from, cur.to)) {
                continue;
            }
            res += cur.weight;
            uf.union(cur.from, cur.to);
        }
        return uf.count != 1 || res > mst;
    }

    boolean isPseudo(int nodes, int[][] edges, int idx, int mst) {
        List<Edge> arr = new ArrayList<>();
        for (int i = 0; i < edges.length; i++) {
            if (i == idx) {
                continue;
            }
            arr.add(new Edge(i, edges[i][0], edges[i][1], edges[i][2]));
        }
        Collections.sort(arr);
        WeightedUF uf = new WeightedUF(nodes);
        int res = 0;
        res += edges[idx][2];
        uf.union(edges[idx][0], edges[idx][1]);
        for (int i = 0; i < arr.size(); i++) {
            Edge cur = arr.get(i);
            if (uf.isConnected(cur.from, cur.to)) {
                continue;
            }
            res += cur.weight;
            uf.union(cur.from, cur.to);
        }
        return uf.count == 1 && res == mst;
    }

    int findMst(int nodes, int[][] edges) {
        int n = edges.length;
        Edge[] arr = new Edge[n];
        for (int i = 0; i < n; i++) {
            arr[i] = new Edge(i, edges[i][0], edges[i][1], edges[i][2]);
        }
        Arrays.sort(arr);
        WeightedUF uf = new WeightedUF(nodes);
        int res = 0;
        for (int i = 0; i < n; i++) {
            Edge cur = arr[i];
            if (uf.isConnected(cur.from, cur.to)) {
                continue;
            }
            res += cur.weight;
            uf.union(cur.from, cur.to);
        }
        return res;
    
}

    class Edge implements Comparable<Edge> {
        int idx;
        int from;
        int to;
        int weight;
        public Edge(int idx, int from, int to, int weight) {
            this.idx = idx;
            this.from = from;
            this.to = to;
            this.weight = weight;
        }

        @Override
        public int compareTo(Edge o) {
            return Integer.compare(weight, o.weight);
        }
    }
}
