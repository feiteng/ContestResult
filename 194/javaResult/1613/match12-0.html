import java.io.*;
import java.util.*;
import java.math.*;
import java.util.stream.*;


public class Solution {
    Set<Integer> pan[];

    int n;

    int pair(int u, int w) {
        return u * 10000 + w;
    }

    public List<List<Integer>> findCriticalAndPseudoCriticalEdges(int n, int[][] edges) {
        this.n = n;
        pan = new HashSet[n];
        for (int i = 0; i < n; i++)
            pan[i] = new HashSet<>();
        for (int[] edge : edges) {
            int u = edge[0], v = edge[1], w = edge[2];
            pan[u].add(pair(v, w));
            pan[v].add(pair(u, w));
        }

        List<Integer> fu1 = new ArrayList<>();
        List<Integer> fu2 = new ArrayList<>();
        for (int i = 0; i < edges.length; i++) {
            int[] edge = edges[i];
            int u = edge[0], v = edge[1], w = edge[2];

            if (!hasPath(u, v, w, w - 1)) {
                if (hasPath(u, v, w, w))
                    fu2.add(i);
                else
                    fu1.add(i);
            }
        }
        return Arrays.asList(fu1, fu2);
    }

    private boolean _hasPath(int u, int target, int cap, boolean[] visited) {
        if (u == target)
            return true;
        visited[u] = true;
        for (int cb : pan[u]) {
            int v = cb / 10000;
            int w = cb % 10000;
            if (!visited[v] && w <= cap && _hasPath(v, target, cap, visited))
                return true;
        }

        return false;
    }

    private boolean hasPath(int u, int v, int w, int cap) {
        pan[u].remove(pair(v, w));
        pan[v].remove(pair(u, w));
        boolean fu = _hasPath(u, v, cap, new boolean[n]);
        pan[u].add(pair(v, w));
        pan[v].add(pair(u, w));
        return fu;
    }
}
