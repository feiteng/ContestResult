class Solution {
    
    class UnionFind {
        int[] roots;
        int[] ranks;
        
        UnionFind(int n) {
            roots = new int[n];
            ranks = new int[n];
            
            for (int i = 0; i < n; i++) {
                roots[i] = i;
            }
        
}
        
        int find(int i) {
            while (i != roots[i]) {
                roots[i] = roots[roots[i]];
                i = roots[i];
            }
            
            return i;
        }
        
        void union(int i, int j) {
            i = find(i);
            j = find(j);
            
            if (i == j) return;
            
            if (ranks[i] == ranks[j]) {
                roots[j] = i;
                ranks[i]++;
            } else if (ranks[i] > ranks[j]) {
                roots[j] = i;
            } else {
                roots[i] = j;
            }
        }
    }
    
    int mst(int N, UnionFind uf, PriorityQueue<int[]> Q, int sum, int count) {
        while (count < N - 1 && !Q.isEmpty()) {
            int[] e = Q.poll();
            int u = uf.find(e[0]);
            int v = uf.find(e[1]);
            int w = e[2];
            
            if (u == v) continue;
            uf.union(u, v);
            count++;
            sum += w;
        }
        
        if (count < N - 1) return -1;
        return sum;
    }
    
    int compare(int[] a, int[] b) {
        return a[2] - b[2];
    }
    
    public List<List<Integer>> findCriticalAndPseudoCriticalEdges(int n, int[][] edges) {
        PriorityQueue<int[]> Q = new PriorityQueue<>((a, b) -> compare(a, b));
        for (int[] e : edges) Q.offer(e);
        int mst = mst(n, new UnionFind(n), Q, 0, 0);
        List<Integer> criticals = new ArrayList<>();
        List<Integer> pcriticals = new ArrayList<>();
        
        Set<Integer> set = new HashSet<>();
        
        for (int i = 0; i < edges.length; i++) {
            Q = new PriorityQueue<>((a, b) -> compare(a, b));
            for (int j = 0; j < edges.length; j++) {
                if (i == j) continue;
                Q.offer(edges[j]);
            }
            
            int v = mst(n, new UnionFind(n), Q, 0, 0);
            if (v > mst || v == -1) {
                criticals.add(i);
                set.add(i);
            }
        }
        
        for (int i = 0; i < edges.length; i++) {
            int[] e = edges[i];
            UnionFind uf = new UnionFind(n);
            Q = new PriorityQueue<>((a, b) -> compare(a, b));
            for (int j = 0; j < edges.length; j++) {
                if (i == j) continue;
                Q.offer(edges[j]);
            }
            
            uf.union(e[0], e[1]);
            int v = mst(n, uf, Q, e[2], 1);
            if (v == mst && !set.contains(i)) 
{
                pcriticals.add(i);
            }
        }
        
        List<List<Integer>> ans = new ArrayList<>();
        ans.add(criticals);
        ans.add(pcriticals);
        return ans;
        
    }
}
