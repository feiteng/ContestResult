class Solution {
    private int[] id;
    
    public List<List<Integer>> findCriticalAndPseudoCriticalEdges(int n, int[][] e) {
        id = new int[n];
        for(int i=0; i<n; i++) id[i] = i;
        
        int mst = 0;
        
        int[][] edges = new int[e.length][3];
        
        for(int i=0; i<e.length; i++) {
            edges[i] = new int[]{e[i][0], e[i][1], e[i][2], i};
        }
        
        Arrays.sort(edges, (p, q)->(p[2]-q[2]));
        
        int comps = n;
        
        
        for(int[] edge: edges) {
            int u = edge[0];
            int v = edge[1];
            int w = edge[2];
            
            if (connected(u, v)) continue;
            
            union(u, v);
            
            mst+=w;
            
            comps--;
            
        }
        
        System.out.println(mst);
        
        if (comps!=1) return new ArrayList<>();
        
        List<Integer> ans1 = new ArrayList<>();
        List<Integer> ans2 = new ArrayList<>();
        
        for(int[] edge1: edges) {
            for(int i=0; i<n; i++) {
                id[i] = i;
            }
            int u1 = edge1[0];
            int v1 = edge1[1];
            int w1 = edge1[2];
            
            union(u1, v1);
            
            
            int cmst = w1;
            comps = n-1;
            
            for(int[] edge2: edges) {
                int u2 = edge2[0];
                int v2 = edge2[1];
                int w2 = edge2[2];
                
                // if (u2==u1 && v2==v1) continue;

                if (connected(u2, v2)) continue;

                union(u2, v2);
                comps--;

                cmst+=w2;

            }
            
            if (cmst!=mst || comps!=1) continue;
            
            cmst = 0;
            comps = n;
            
            
            for(int i=0; i<n; i++) {
                id[i] = i;
            }
            
            for(int[] edge2: edges) {
                int u2 = edge2[0];
                int v2 = edge2[1];
                int w2 = edge2[2];
                
                if (u2==u1 && v2==v1) continue;

                if (connected(u2, v2)) continue;

                union(u2, v2);

                cmst+=w2;
                comps--;

            }
            
//             if (edge1[3]==3) {
//                 System.out.println(comps+" --- "+cmst);
//             }
            
//             System.out.println(u1+" "+v1+" "+comps+" "+cmst);
            
            if (comps==1 && cmst==mst) {
                ans1.add(edge1[3]);
            } else {
                
                ans2.add(edge1[3]);
            }
            
            // if (comps!=1  cmst!=mst) {
            //     ans1.add(edge1[3]);
            // } else {
            //     ans2.add(edge1[3]);
            // }
            
        }
        
        List<List<Integer>> ans = new ArrayList<>();
        
        ans.add(ans2);
        ans.add(ans1);
        
        return ans;
        
    }
    
    private int root(int u) {
        while(u!=id[u]) {
            id[u] = id[id[u]];
            
u = id[u];
        }
        return u;
    }
    
    private void union(int u, int v) {
        int p = root(u);
        int q = root(v);
        id[p] = q;
    }
    
    private boolean connected(int u, int v) {
        return root(u)==root(v);
    }
}
