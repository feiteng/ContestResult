class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> hm = new HashMap<>();
        int n = names.length;
        String[] res = new String[n];
        for (int i = 0; i < n; i++) {
            String cur = names[i];
            if (hm.containsKey(cur)) {
                int idx = hm.get(cur);
                while (true) {
                    String nxt = cur + "(" + (idx + 1) + ")";
                    if (!hm.containsKey(nxt)) {
                        break;
                    }
                    idx++;
                
}

                res[i] = cur + "(" + (idx + 1) + ")";
                hm.put(cur, idx + 1);
                hm.put(cur + "(" + (idx + 1) + ")", 0);
            } else {
                res[i] = cur;
                hm.put(cur, 0);
            }
        }
        return res;
    }
}
