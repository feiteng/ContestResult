class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> map = new HashMap<>();
        String[] res = new String[names.length];
        for(int i = 0; i < names.length; i++) {
            String n = names[i];
            if(map.containsKey(n)) {
                int suffix = map.get(n) + 1;
                while(map.containsKey(n + "(" + suffix + ")")) {
                    suffix++;
                }
                res[i] = n + "(" + suffix +")";
                map.put(n, suffix);
                map.put(res[i], 0);
            }else{
                res[i] = n;
                map.put(n, 0);
            }
        }
        return res;
    }
}
