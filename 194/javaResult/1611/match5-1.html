class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> idx = new HashMap<>();
        
        int n = names.length;
        String[] res = new String[n];
        for (int i = 0; i < n; i++) {
            String s = names[i];
            
if (!idx.containsKey(s)) {
                res[i] = s;
                idx.put(s, 1);
            } else {
                int j = idx.get(s);
                do {
                    res[i] = String.format("%s(%d)", s, j);
                    j++;
                } while (idx.containsKey(res[i]));
                
                
idx.put(s, j);
                idx.put(res[i], 1);
            }
        }
        return res;
    }
}
