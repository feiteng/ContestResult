class Solution {
    public String[] getFolderNames(String[] names) {
        String[] ans = new String[names.length];
        Map<String, Integer> cache = new HashMap<>();
        for (int i = 0; i < ans.length; i++) {
            String org = names[i];
            if (!cache.containsKey(org)) {
                ans[i] = org;
                cache.put(org, 0);
            } else {
                int seq = cache.get(org) + 1;
                String key = org + "(" + seq + ")";
                while (cache.containsKey(key)) {
                    seq++;
                    key = org + "(" + seq + ")";
                }
                cache.put(org, seq);
                cache.put(key, 0);
                ans[i] = key;
            }
        }
        return ans;
    }
}
