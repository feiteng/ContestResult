class Solution {

    public String[] getFolderNames(String[] names) {
        int n = names.length;

        Map<String, Integer> next = new HashMap<>();

        
Set<String> taken = new HashSet<>();

        for (int i = 0; i < n; i++) {
            String str = names[i];
            if (!taken.contains(str)){
                taken.add(str);
                
next.put(str,1);
            } else {
                int count = next.get(str);

                String temp = str + "(" + count + ")";
                while(taken.contains(temp)){
                    count++;
                    temp = str + "(" + count + ")";
                }
                taken.add(temp);

                
names[i] = temp;
                next.put(temp, 1);
                next.put(str, count+1);

            }
        }

        return names;
    }
}
