class Solution {
    public String[] getFolderNames(String[] names) {
        int n = names.length;
        String[] ret = new String[n];
        Map<String, Integer> map = new HashMap();
        for (int i = 0; i < n; i++) {
            
if (!map.containsKey(names[i])) {
                ret[i] = names[i];
                map.put(names[i], 1);
            } else {
                int suffix = map.get(names[i]);
                while (map.containsKey(names[i] + "("+String.valueOf(suffix)+")")) suffix++;
                ret[i] = names[i] + "("+String.valueOf(suffix)+")";
                
map.put(ret[i], 1);
                map.put(names[i], ++suffix);
            }
        }
        return ret;
    }
}
