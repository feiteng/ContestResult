class Solution {
    public String[] getFolderNames(String[] names) {
        int n = names.length;
        Map<String,Integer> map = new HashMap<>();
        for(int i=0; i<n; i++){
            String cur = names[i];
            if(!map.containsKey(cur)){
                map.put(cur,1);
            }else{
                int index = map.get(cur);
                String temp = cur;
                while(map.containsKey(temp)){
                    temp = cur + "(" + index + ")";
                    index++;
                }
                map.put(temp,1);
                map.put(cur,index);
                names[i] = temp;
            }
        }
        return names;
    }
}
