class Solution {
    public String[] getFolderNames(String[] ns) {
        int n = ns.length;
        String[] res = new String[n];
        Map<String,Integer> mp = new HashMap<>();
       // System.out.println(ns[i]+" "+mp);
        for(int i=0;i<n;i++){
            //System.out.println(ns[i]+" "+mp);
            if(!mp.containsKey(ns[i])){
                mp.put(ns[i],1);
                res[i] = ns[i];
            }else{
                int v = mp.get(ns[i]);
                String neo = ns[i]+"("+String.valueOf(v)+")";
                while(mp.containsKey(neo)){
                    v++;
                    neo = ns[i]+"("+String.valueOf(v)+")";
                }
                mp.put(neo,1);
                res[i] = neo;
                mp.put(ns[i],++v);
            }
            //System.out.println(ns[i]+" "+mp);
        } 
        return res;
    }
}
