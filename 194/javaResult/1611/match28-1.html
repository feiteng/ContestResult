class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> map = new HashMap<>();
        int N= names.length;
        
String[] ans = new String[N];
        for (int i = 0; i < N; i++) {
            if (map.containsKey(names[i])) {
                int num = map.get(names[i]);
                String nn = names[i] + "(" + num + ")";
                while (map.containsKey(nn)) {
                    num++;
                    nn = names[i] + "(" + num + ")";
                }
                ans[i] = nn;
                map.put(names[i], num + 1);
                map.put(ans[i], 1);
            } else {
                ans[i] = names[i];
                map.put(ans[i], 1);
            }
        }
        return ans;
    }
}
