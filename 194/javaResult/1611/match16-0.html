class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> map = new HashMap<>();
        int n = names.length;
        String[] res = new String[n];
        for(int i = 0 ; i < n ; i++){
            String name = names[i];
            if(map.containsKey(name)){
                int next = map.get(name) + 1;
                while(map.containsKey(name + "(" + next + ")")){
                    next++;
                }
                res[i] = name + "(" + next + ")";
                map.put(name, next);
                map.put(name + "(" + next + ")", 0);
            } else {
                res[i] = name;
                map.put(name, 0);
            }
        }
        return res;
    }
}
