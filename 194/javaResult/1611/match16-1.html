class Solution {
    public String[] getFolderNames(String[] names) {
        // map
        Map<String, Integer> map = new HashMap<>();
        int n = names.length;
        String[] res = new String[n];
        for (int i = 0; i < n; i++) {
            // get number
            String name = names[i];
            if (map.containsKey(name)) {
                int ct = map.get(name);
                while (map.containsKey(name + "(" + ct + ")")) {
                    ct++;
                }
                map.put(name, ct + 1);
                String finalName = name + "(" + ct + ")";
                map.put(finalName, 1);
                res[i] = finalName;
            
} else {
                res[i] = name;
                map.put(name, 1);
            }
        }
        return res;
    }
}
