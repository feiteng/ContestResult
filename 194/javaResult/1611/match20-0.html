class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> map = new HashMap<>();
        String[] res = new String[names.length];
        for (int i = 0; i < names.length; i++) {
            if (!map.containsKey(names[i])) {
                res[i] = names[i];
                map.put(names[i], 0);
            } else {
                int smallestIndex = map.get(names[i]);
                while (true) {
                    String next = names[i] + "(" + (++smallestIndex) + ")";
                    if (!map.containsKey(next)) {
                        res[i] = next;
                        map.put(next, 0);
                        map.put(names[i], smallestIndex);
                        break;
                    }
                }
            }
        }
        return res;
    }
}
