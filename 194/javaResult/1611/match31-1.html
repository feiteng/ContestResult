class Solution {
    public String[] getFolderNames(String[] names) {
        int n = names.length;
        String[] res = new String[n];
        HashMap<String, Integer> map = new HashMap<>();
        for(int i = 0; i < n; ++i) {
            String cx = names[i];
            
if(!map.containsKey(cx)) {
                res[i] = cx;
                map.put(cx, 1);
            } else {
                int t = map.get(cx);
                while(true) {
                    String nx = cx + "(" + t + ")";
                    if(map.containsKey(nx)) ++t;
                    else break;
                }
                String nx = cx + "(" + t + ")";
                
res[i] = nx;
                map.put(cx, t + 1);
                map.put(nx, 1);
            }
        }
        return res;
    }
}
