class Solution {

	public String[] getFolderNames(String[] names) {
		HashMap<String, Integer> map = new HashMap<>();
		HashSet<String> set = new HashSet<>();
		String[] result = new String[names.length];
		for (int i = 0; i < names.length; i++) {
			if (!map.containsKey(names[i]) && !set.contains(names[i]
)) {
				result[i] = names[i];
				map.put(names[i], 1);
				set.add(names[i]);
			} else {
				int j = map.getOrDefault(names[i], 1);
				while (set.contains(names[i] + '(' + j + ')')) {
					j++;
				}
				result[i] = names[i] + '(' + j + ')';
				map.put(names[i], j + 1);

				set.add(names[i] + '(' + j + ')');
			}
		}
		return result;
	}
}
