class Solution {
    public String[] getFolderNames(String[] names) {
        Map<String, Integer> dict = new HashMap<>();
        String[] result = new String[names.length];
        int i = 0;
        for (String name : names) {
            if (dict.containsKey(name)) {
                int newIndex =  dict.get(name);
                String newName = name + "(" + newIndex + ")";
                while (dict.containsKey(newName)) {
                    newIndex++;
                    newName = name + "(" + newIndex + ")";
                }
                result[i++] = newName;
                dict.put(name, dict.get(name)+1);
                dict.put(newName, 1);
            
} else {
                dict.put(name, 1);
                result[i++] = name;
            }
        }
        return result;
    }
}
