class Solution {
    public String[] getFolderNames(String[] names) {
        Set<String> set = new HashSet<>();
        Map<String, Integer> last = new HashMap<>();
        String[] result = new String[names.length];
        for (int i = 0;i<names.length;i++) {
            String name = names[i];
            if (!set.contains(name)) {
                result[i] = name;
                set.add(name);
                last.put(name, 1);
            
} else {
                int j = last.get(name);
                while (set.contains(String.format("%s(%d)", name, j))) {
                    j++;
                }
                result[i] = String.format("%s(%d)", name, j);
                set.add(result[i]);
                last.put(name, j + 1);
                last.putIfAbsent(result[i], 1);
            }
        }
        return result;
    }
}
