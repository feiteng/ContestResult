class Solution {
    public String[] getFolderNames(String[] names) {
        
        HashSet<String> seen=new HashSet<String>();
        HashMap<String,Integer> map=new HashMap<>();
        String[] arr=new String[names.length];
        for(int i=0;i<names.length;i++){
            
            if(!seen.contains(names[i])){
                seen.add(names[i]);
                map.put(names[i],0);
                arr[i]=names[i];
            }
            
            else{
                int now=map.get(names[i]);
                
                now++;
                while(seen.contains(names[i]+"("+String.valueOf(now)+")")){
                    now++;
                }
                //System.out.println(names[i]+"("+String.valueOf(now)+")");
                arr[i]=names[i]+"("+String.valueOf(now)+")";
                seen.add(arr[i]);
                map.put(arr[i],0);
                map.put(names[i],now);
            }
            
            
            
        }
        
        return arr;
        
    }
}
