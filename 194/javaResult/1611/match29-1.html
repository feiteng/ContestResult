class Solution {
    public String[] getFolderNames(String[] names) {
    	
    	int n=names.length;
    	String[] ans=new String[n];
    	Map<String,Integer> map=new HashMap<>();
    	
    	for(int i=0;i<n;i++) {
    		String s=names[i];
    		if(map.containsKey(s)) {

    			int num=map.get(s)+1;
    			String ss=s+"("+String.valueOf(num)+")";
    			while(map.containsKey(ss)) {
    				num++;
    				ss=s+"("+String.valueOf(num)+")";
    			}
    			ans[i]=ss;
    			map.put(s, num);
    			map.put(ss, 0);
    		}
    		else {
    			ans[i]=s;

    			map.put(s, 0);
    		}
    	}
    	
    	return ans;
    
    }
}
