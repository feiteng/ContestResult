class Solution {

    public String[] getFolderNames(String[] names) {
        Map<String, Integer> map = new HashMap<>();  //key -> last index trid
        int n = names.length;
        
for(int i = 0; i<n; i++){
            String s = names[i];
            
            if(!map.containsKey(s)){
                map.put(s, 0);
            }else{
                int ni = map.get(s) +1;
                String ns = s + "(" + ni + ")";
                while(map.containsKey(ns)){
                    ni++;
                    ns = s + "(" + ni + ")";
                }
                map.put(ns, 0);
                map.put(s, ni);
                names[i] = ns;
            }
            
        }
        return names;
        
    }
}
