class Solution {
    public String[] getFolderNames(String[] names) {
        int n = names.length;
        String[] res = new String[n];
        Map<String, Integer> map = new HashMap();
        for(int i = 0; i < n; i++){
            
if(!map.containsKey(names[i])){
                res[i] = names[i];
                map.put(names[i], 1);
            } else {
                int index = map.get(names[i]);
                String new_name = names[i] + "(" + index + ")";
                while(map.containsKey(new_name)){
                    index++;
                    new_name = names[i] + "(" + index + ")";
                }
                res[i] = new_name;
                map.put(names[i], index);
                map.put(new_name, 1);
            }
        }
        return res;
    }
}
