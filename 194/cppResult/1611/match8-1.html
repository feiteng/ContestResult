class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> count;
        vector<string> res;
        for (string name : names) {
            // cout << name << ':';
            if (count.find(name) == count.end()) {
                // cout << "not found" << endl;
                count[name] = 0;
                res.push_back(name);
            } else {
                // cout << "found" << endl;
                for (int j = 1; ; j++) {

                    string new_name = name + "(" + to_string(count[name] + j) + ")";
                    if (count.find(new_name) == count.end()) {
                        count[name] += j;
                        count[new_name] = 0;
                        res.push_back(new_name);
                        break;
                    }
                }
                
                
            }
            
        }
        return res;
    }
};
