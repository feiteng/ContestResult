class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> map;
        vector<string> res;
        
        for(auto s: names) {
            if(map.find(s) == map.end()) {
                res.push_back(s);
                map[s] = 0;
            }
            else {

                int k = map[s]+1;
                string tmp = s+"("+to_string(k)+")";
                while(map.find(tmp) != map.end()) {
                    k++;
                    tmp = s+"("+to_string(k)+")";
                }
                map[s] = k;
                map[tmp] = 0;
                res.push_back(tmp);
            }
        }
        return res;
    }
};
