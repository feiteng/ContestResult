class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> res;
        unordered_map<string, int> Map;
        for(string s : names) {
            if(Map.find(s) == Map.end()) {
                res.push_back(s);
                ++Map[s];
            }
            else {
                int k = Map[s];
                string ans = s + "(" + to_string(k) + ")";
                while(Map.find(ans) != Map.end()) {
                    ++k;
                    ans = s + "(" + to_string(k) + ")";
                }
                res.push_back(ans);
                Map[s] = k + 1;
                ++Map[ans];
            }
        }
        return res;
    }
};
