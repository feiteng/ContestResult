class Solution {
public:

    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> t;
        vector<string> res;
        for(string& s: names) {
            if(t[s] == 0) {
                t[s] = 1;
                res.push_back(s);
            }
            else {
                int k = t[s];
                while(t[s+"(" + to_string(k) + ")"] != 0) {
                    k += 1;
                }
                t[s+"(" + to_string(k) + ")"] = 1;
                t[s] = k + 1;
                res.push_back(s+"(" + to_string(k) + ")");
            }
        }
        return res;
    }
};
