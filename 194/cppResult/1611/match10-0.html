class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> res;
        unordered_map<string, int> seen;
        for (auto name: names) {
            if (seen.count(name)) {
                int i = seen[name];
                while (seen.count(name + "(" + to_string(i) + ")")) {
                    i ++;
                }
                seen[name] = i+1;
                res.push_back(name + "(" + to_string(i) + ")");
                seen[name + "(" + to_string(i) + ")"] = 1;
            } else {
                res.push_back(name);
                seen[name] = 1;
            }
        }
        return res;
    }
};
