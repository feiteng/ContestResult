class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
      map<string, int> mp;
      vector<string> ans;

      for (auto name : names) {
        if (mp.find(name) != mp.end()) {
          int st = mp[name] + 1; 
          for (; ; st++) {
            string nname = name + "(" + to_string(st) + ")";
            if (mp.find(nname) == mp.end()) {
              ans.push_back(nname);
              mp[name] = st;
              mp[nname] = 0;
              break;
            }
          }
        } else {
          ans.push_back(name);
          mp[name] = 0;
        }
      }
      return ans;
    }
};
