class Solution {
    unordered_map<string, int> mp;
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> ans;
        for (auto& s: names){
            if (mp.count(s)){
                int t = mp[s];
                for (; ; ++t){
                    string ss = s + "(" + to_string(t + 1) + ")";
                    if (!mp.count(ss)) {
                        ans.push_back(ss);
                        mp[s] = t + 1;
                        mp[ans.back()] = 0;
                        break;
                    }
                }
            }else {
                ans.push_back(s);
                mp[s] = 0;
            }
        }
        return ans;
    }
};
