class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> record;
        vector<string> res;
        for (string n: names) {
            if (!record.count(n)) {
                res.push_back(n);
                record[n] = 1;
            }
            else {
                int avai = record[n];
                string tmp = n + "(" + to_string(avai) + ")";
                while (record.count(tmp)) {
                    avai ++;
                    tmp = n + "(" + to_string(avai) + ")";                    
                }
                avai ++;
                res.push_back(tmp);
                record[tmp] = 1;
                record[n] = avai;
            }
        }
        return res;
    }
};
