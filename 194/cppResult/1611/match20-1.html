class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> m;
        vector<string> res;
        for(string& s : names) {
            if(!m.count(s)) {
                m[s] = 1;
                res.push_back(s);
            } else {
                for(int i = m[s]; ; ++i) {

                    string ne = s + '(' + to_string(i) + ')';
                    if(!m.count(ne)) {
                        m[ne] = 1;
                        res.push_back(ne);
                        m[s] = i + 1;
                        break;
                    }
                }
            }
        }
        return res;
    }
};
