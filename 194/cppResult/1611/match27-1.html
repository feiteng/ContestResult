class Solution {
 public:
  vector<string> getFolderNames(vector<string>& names) {
    unordered_map<string, int> used;
    int n = names.size();
    vector<string> res(n);

    for (int i = 0; i < n; i++) {
      if (used.find(names[i]) == used.end()) {
        res[i] = names[i];
        used[names[i]] = 0;
      } else {
        int suffix = used[names[i]] + 1;
        while (true) {
          string suffix_name = names[i] + "(" + to_string(suffix) + ")";
          if (used.find(suffix_name) == used.end()) {
            res[i] = suffix_name;
            used[names[i]] = suffix;   
            used[suffix_name] = 0;
            break;
          }
          suffix++;
        }
      }
    }
    return res;
  }
};
