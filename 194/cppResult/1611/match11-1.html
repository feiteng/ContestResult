class Solution {
public:
    vector<string> getFolderNames(vector<string>& A) {
        unordered_map<string ,int> m;
        set<string> visit;
        vector<string> ans;
        for(auto& str : A) {
            if(visit.count(str)) {
                int cnt = m[str];
                string tmp = str + '(' + to_string(cnt) + ')';
                while(visit.count(tmp)) {
                    cnt++;

                    tmp = str + '(' + to_string(cnt) + ')';
                }
                visit.insert(tmp);
                m[tmp] = 1;
                ans.push_back(tmp);
                m[str] = cnt;
            }else {
                ans.push_back(str);
                visit.insert(str);
                m[str] = 1;
            }
        }
        return ans;
    }
};
