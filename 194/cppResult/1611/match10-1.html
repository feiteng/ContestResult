class Solution {
public:
  vector<string> getFolderNames(vector<string>& A) {
    unordered_map<string,int> m;
    vector<string> ans;
    for (auto& x : A) {
      if (m.count(x)) {
        int d = m[x];
        while (m.count(x + "(" + to_string(d) + ")")) {
          ++d;
        }
        ++m[x + "(" + to_string(d) + ")"];
        ans.emplace_back(x + "(" + to_string(d) + ")");
        ++m[x];
      } else {
        ans.emplace_back(x);
        ++m[x];
      }
    }
    return ans;
  }
};
