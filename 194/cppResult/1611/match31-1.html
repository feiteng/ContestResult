class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        std::unordered_map<std::string, int> m;
        std::vector<std::string> ans;
        ans.reserve(names.size());

        for(auto& val: names)
        {
            auto it = m.find(val);
            if(it == m.end())
            {
                m[val] = 1;
                ans.push_back(val);
            }
            else
            {
                int i = it->second;
                while(true)
                {
                    std::string next = val + "(" + std::to_string(i) + ")";

                    auto it1 = m.find(next);
                    if(it1 == m.end())
                    {
                        it->second++;
                        m[next] = 1;
                        ans.push_back(next);
                        break;
                    }
                    ++i;
                }
            }
        }
        return ans;
    }
};
