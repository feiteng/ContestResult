class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string,int> note;
        vector<string> ans;
        for(auto x:names){
            if(note.count(x)==0){
                ans.push_back(x);
                note[x]++;
            }else{
                int num=note[x];
                string tmp=x+"("+to_string(num)+")";
                while(note.count(tmp)){
                    tmp=x+"("+to_string(++num)+")";
                }
                
                ans.push_back(tmp);
                note[tmp]++;
                
                note[x]++;
            }
        }
        return ans;
    }
};
