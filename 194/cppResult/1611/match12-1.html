class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> res;
        set<string> saw;
        map<string,int> mmp;
        for(auto e:names){
            if(saw.count(e)==0){
                // cout<<"A\n";
                res.push_back(e);
                saw.insert(e);
            }else{
                // cout<<"B\n";
                for(int i=mmp[e]+1;i<50000+1;i++){
                    string tmp=e+"("+to_string(i)+")";
                    if(saw.count(tmp)==0){
                        res.push_back(tmp);
                        saw.insert(tmp);
                        mmp[e]++;
                        break;
                    }
                }
            }
        }
        return res;
    }
};
