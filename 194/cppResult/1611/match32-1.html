class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        map<string, int> data;
        vector<string> res;
        for (auto& s : names) {
            if (data.find(s) == data.end()) {
                ++data[s];
                res.push_back(s);
            } else {
                for (int k = data[s]; ; ++k) {

                    string n = s + '(' + to_string(k) + ')';
                    if (data.find(n) == data.end()) {
                        data[s] = k + 1;
                        ++data[n];
                        res.push_back(n);
                        break;
                    }
                }
            }
        }
        return res;
    }
};
