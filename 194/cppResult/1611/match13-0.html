class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> table;
        vector<string> ret;
        for (auto& s : names) {
            if (table.find(s) == table.end()) {
                ret.push_back(s);
                table[s] = 0;
                //cout << s  << " " << 1 << endl;
            } else {
                int t = table[s];
                string ss = s+"("+to_string(t+1) + ")";
                while (table.find(ss) != table.end()) {
                    ++t;
                    ss = s+"("+to_string(t+1) + ")";
                }
                ret.push_back(ss);
                table[ss] = 0;
               // cout << "0: " << ss << endl;
                table[s] = t+1;
                //cout << t+1 << " " << s << endl;
            }
        }
        return ret;
    }
};
