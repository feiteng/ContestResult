class Solution {
public:

    vector<string> getFolderNames(vector<string>& names) {
        vector<string> res;
        map<string, int> unames;
        for(string s: names){
            if(unames.find(s) == unames.end()){
                // Check if the suffix is a number
                unames[s] = 0;
                res.push_back(s);
            }else{
                int suf = unames[s] + 1;
                while(unames.find(s+"("+to_string(suf)+")") != unames.end()){
                    suf++;
                }
                unames[s] = suf;
                unames[s+"("+to_string(suf)+")"] = 0;
                res.push_back(s+"("+to_string(suf)+")");
            }
        }
        return res;
    }
};
