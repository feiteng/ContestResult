class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        set<string> st;
        map<string, int> number;
        vector<string> ans;
        for (string s : names) {
            if (st.find(s) == st.end()) {
                ans.push_back(s);
                st.insert(s);
                number[s] = 0;
            } else {
                while (true) {
                    number[s]++;
                    string newstr = s + "(" + to_string(number[s]) + ")";
                    if (st.find(newstr) == st.end()) {
                        st.insert(newstr);
                        ans.push_back(newstr);
                        break;
                    }
                }
            }
        }
            
        return ans;
    }
};
