class Solution {
public:
vector<string> getFolderNames(vector<string> &names)
{
    unordered_map<string, int> hash;
    vector<string> ans(names.size());
    for (int i = 0; i < names.size(); i++)
    {
        if (hash.count(names[i]) == 0)
        {
            hash[names[i]] = 0;
            ans[i] = names[i];
        }
        else
        {
            int num = hash[names[i]] + 1;
            while (true)
            {
                string nn = names[i] + "(" + to_string(num) + ")";
                if (hash.count(nn) == 0)
                {
                    ans[i] = nn;
                    hash[nn] = 0;
                    hash[names[i]] = num;
                    break;
                }
                num++;
            }
        }
    }
    return ans;
}
};
