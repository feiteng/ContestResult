class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> res;
        unordered_set<string> seen;
        unordered_map<string, int> last;
        for (auto& name : names) {
            if (!seen.count(name)) {
                seen.insert(name);
                res.push_back(name);
            } else {
                for (int i = last[name] + 1; i < 50000; ++i) {
                    string tmp = name + "(" + to_string(i) + ")";
                    if (!seen.count(tmp)) {
                        seen.insert(tmp);
                        res.push_back(tmp);
                        last[name] = i;
                        break;
                    }
                }
            }
        }
        return res;
    }
};
