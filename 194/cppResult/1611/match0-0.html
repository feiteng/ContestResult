class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> ret;
        set<string> seen;
        map<string,int> cnts;
        for (string &s: names){
            if (seen.count(s) == 0){
                seen.insert(s);
                cnts[s]=0;
                ret.push_back(s);
            } else {
                while(1){
                    cnts[s]++;
                    string s2 = s + "("+ to_string(cnts[s]) +")";
                    if (seen.count(s2) == 0){
                        seen.insert(s2);
                        ret.push_back(s2);
                        break;
                    }
                }
            }
        }
        return ret;
    }
};
