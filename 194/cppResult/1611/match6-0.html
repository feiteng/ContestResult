class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> ans;
        unordered_map<string, int> cnt;
        for (string& name : names) {
            if (cnt.find(name) == cnt.end()) {
                ans.push_back(name);
                cnt[name] = 1;
            }else {
                for (int k = cnt[name]; ; ++k) {
                    string namek = name + "(" + to_string(k) + ")";
                    if (cnt.find(namek) == cnt.end()) {
                        ans.push_back(namek);
                        cnt[namek] = 1;
                        cnt[name] = k + 1;
                        break;
                    }
                }
            }
        }
        
        return ans;
    }
};
