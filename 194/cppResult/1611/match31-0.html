class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string, int> nameToCount;
        vector<string> res;
        for(auto name: names)
        {
            auto it = nameToCount.find(name);
            if(it == nameToCount.end())
            {
                res.emplace_back(name);
                nameToCount[name] = 1;
            }
            else
            {
                int suf = it->second;
                while(true)
                {
                    auto nameWithSuf = name+'('+to_string(suf)+')';
                    if(nameToCount.find(nameWithSuf) == nameToCount.end())
                    {
                        res.emplace_back(nameWithSuf);
                        nameToCount[name] = suf+1;
                        nameToCount[nameWithSuf] = 1;
                        break;
                    }
                    suf++;
                }
            }
        }
        return res;
    }
};
