class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        vector<string> res; string tmp;
        unordered_map<string, int> cnt;
        for(int i = 0; i < names.size(); i++){
            if(cnt.count(names[i])){
                while(cnt.count(names[i]+"("+to_string(cnt[names[i]]) + ")")){
                    cnt[names[i]]++;
                }
                res.push_back(names[i]+"("+to_string(cnt[names[i]]) + ")");
                cnt[names[i]+"("+to_string(cnt[names[i]]) + ")"] = 1;
                cnt[names[i]]++;
            }else{
                cnt[names[i]] = 1;
                res.push_back(names[i]);
            }
        }
        return res;
    }
};
