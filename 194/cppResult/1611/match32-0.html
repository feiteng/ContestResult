class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        map<string,int>mp;
        vector<string>ans;
        for(auto it:names){
            if(mp[it]==0){
                mp[it]=1;
                ans.push_back(it);
            }
            else{
                string tmp=it+"(";
                for(int num=mp[it];;num++){
                    string s="";
                    int x=num;
                    while(x){
                        s+=(x%10+'0');
                        x/=10;
                    }
                    string ss="";
                    for(int i=s.length()-1;i>=0;i--){
                        ss+=s[i];
                    }
                    string now=tmp+ss+")";
                    if(mp[now]==0){
                        mp[now]=1;
                        ans.push_back(now);
                        mp[it]=num+1;
                        break;
                    }

                }
            }
        }
        return ans;
    }
};
