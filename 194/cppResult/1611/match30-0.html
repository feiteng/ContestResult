class Solution {
public:
    vector<string> getFolderNames(vector<string>& names) {
        unordered_map<string,int> mmap;
        vector<string> result;
        for(auto str:names)
        {
            if(mmap.count(str))
            {
                int k=++mmap[str];
                while(mmap.count(str+"("+to_string(k)+")"))
                {
                    ++k;   
                }
                result.push_back(str+"("+to_string(k)+")");
                mmap[str+"("+to_string(k)+")"]=0;
            }else{
                mmap[str]=0;
                result.push_back(str);
            }
        }
        return result;
    }
};
