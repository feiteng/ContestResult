class Solution {
public:
    vector<int> f;
    int getf(int x) {
        if(x != f[x])
            f[x] = getf(f[x]);
        return f[x];
    }
    set<int> used;
    
    
    int mincost(int n, vector<vector<int>>& e, int ignore) {
        f = vector<int>(n, 0);
        for (int i = 0; i < n; ++i) f[i] = i;
        int m = e.size();
        int sz = 0;
        int ret = 0;
        for (int i = 0; i < m; ++i) {
            if (i == ignore) continue;
            int fa = getf(e[i][0]);
            int fb = getf(e[i][1]);
            if (fa == fb) continue;
            ret += e[i][2];
            f[fb] = fa;
            sz++;
        }
        
        if (sz < n - 1) return 1e9;
        
        return ret;
    }
    
    int mincost2(int n, vector<vector<int>>& e, int use) {
        f = vector<int>(n, 0);
        for (int i = 0; i < n; ++i) f[i] = i;
        int m = e.size();

        int ret = 0;
        ret += e[use][2];
        f[e[use][1]] = e[use][0];
        
        for (int i = 0; i < m; ++i) {
            int fa = getf(e[i][0]);
            int fb = getf(e[i][1]);
            if (fa == fb) continue;
            ret += e[i][2];
            f[fb] = fa;
        }
        return ret;
    }
    
    vector<vector<int>> findCriticalAndPseudoCriticalEdges(int n, vector<vector<int>>& edges) {
        int m = edges.size();
        for (int i = 0; i < m; ++i) {
            edges[i].push_back(i);
        }
        sort(edges.begin(), edges.end(), [](vector<int> &a, vector<int> &b) {
            return a[2] < b[2];
        });
        
        int cost = mincost(n, edges, -1);
        used.clear();
        vector<vector<int>> ans(2, vector<int>());
        for (int i = 0; i < m; ++i) {
            if (cost < mincost(n, edges, i)) {
                ans[0].push_back(edges[i].back());
            }
            else if (cost == mincost2(n, edges, i)) {
                ans[1].push_back(edges[i].back());
            }
        }
        return ans;
    }
};
