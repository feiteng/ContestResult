class Solution {
public:
    struct edge {
        int u,v,val,idx;
        bool operator< (const edge& n) const {
            return val<n.val;
        }
    }NODE[300];
    int pre[105],n,m;
    int Find(int x) {
        if (x==pre[x]) return x;
        return pre[x]=Find(pre[x]);
    }

    int solve(int idx) {
        for (int i=0; i<n; i++) pre[i]=i;
        sort(NODE+1,NODE+m+1);
        int cnt=0;
        int res=0;
        for (int i=1; i<=m&&cnt<n-1; i++) {
            int u=NODE[i].u;
            int v=NODE[i].v;
            if (NODE[i].idx==idx) continue;
            u=Find(u),v=Find(v);
            if (u==v) continue;
            cnt++;
            pre[u]=v;
            res+=NODE[i].val;
        }
        if (cnt!=n-1) return 0x3f3f3f3f;
        return res;

    }
    bool check(int u,int v, int val, int Min) {
        for (int i=0; i<n; i++) pre[i]=i;
        sort(NODE+1,NODE+m+1);
        int cnt=1;
        int res=val;
        pre[u]=v;
        for (int i=1; i<=m&&cnt<n-1; i++) {
            int u=NODE[i].u;
            int v=NODE[i].v;
            u=Find(u),v=Find(v);
            if (u==v) continue;
            cnt++;
            pre[u]=v;
            res+=NODE[i].val;
        }
        return res==Min;
    }
    vector<vector<int>> findCriticalAndPseudoCriticalEdges(int N, vector<vector<int>>& edges) {
        m=edges.size();
        n=N;
        for (int i=0; i<edges.size(); i++) {
            NODE[i+1]={edges[i][0],edges[i][1],edges[i][2],i};
        }
        int val=solve(-1);
        vector<int> guanjian,wei;
        for (int i=0; i<edges.size(); i++) {
            if (check(edges[i][0],edges[i][1],edges[i][2],val)) {
                int cal=solve(i);
                if (cal>val) guanjian.push_back(i);
                else wei.push_back(i);
            }
        }
        vector<vector<int>>ans;
        ans.push_back(guanjian);
        ans.push_back(wei);
        return ans;
    }
};
