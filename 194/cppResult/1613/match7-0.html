class Solution {
public:
    struct E{
        int a,b,c,id;
        bool operator < (const E & e) const{
            return c < e.c;
        }
    };
    vector<E>e;
    int n;
    int fa[104];
    int find(int x){return fa[x]==x?x:fa[x]=find(fa[x]);}
    void merge(int x,int y){fa[find(x)]=find(y);}
    int fun(int w){
        for(int i = 0;i < n;i++) fa[i] = i;
        int ans = 0;
        for(int i = 0;i < e.size();i++)
            if(e[i].id == w){
                ans += e[i].c;
                merge(e[i].a,e[i].b);
            }
        for(int i = 0;i < e.size();i++){
            int x=e[i].a,y=e[i].b,z=e[i].c;
            if(find(x)==find(y)) continue;
            ans += z;
            merge(x,y);
        }
        return ans;
    }
    int gun(int w){
        for(int i = 0;i < n;i++) fa[i] = i;
        int ans = 0;
        for(int i = 0;i < e.size();i++){
            int x=e[i].a,y=e[i].b,z=e[i].c;
            if(w==e[i].id||find(x)==find(y)) continue;
            ans += z;
            merge(x,y);
        }
        return ans;
    }
    vector<vector<int>> findCriticalAndPseudoCriticalEdges(int m, vector<vector<int>>& edges) {
        n=m;
        for(int i = 0;i < edges.size();i++){
            int x = edges[i][0],y = edges[i][1],z = edges[i][2];
            e.push_back(E{x,y,z,i});
        }
        m = e.size();
        sort(e.begin(),e.end());
        vector<int>a1,a2;
        int mn = gun(-1);
        for(int i = 0;i < e.size();i++){
            int sum = gun(i);
            if(sum != mn) a1.push_back(i);
            else{
                sum = fun(i);
                if(sum == mn) a2.push_back(i);
            }
        }
        return vector<vector<int>>{a1,a2};
    }
};
