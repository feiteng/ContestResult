int const N = 111;
int const M = 222;
int n, m;
typedef long long ll;
int parents[N];
int find(int a) { return parents[a] < 0 ? a : parents[a] = find(parents[a]); }
void uni(int a, int b) {
  a = find(a), b = find(b); if (a == b) return;
  if (parents[a] < parents[b]) { parents[a] += parents[b], parents[b] = a; }
  else { parents[b] += parents[a], parents[a] = b; }
}
void init() { for (int i = 0; i <= n; ++i) parents[i] = -1; }
int e[10010][3];
struct Node {
    int a, b, c;
    friend bool operator<(Node x, Node y) {
        return x.c < y.c;
    }
} p[M];

class Solution {
public:
    int q;
    int mst(int st) {
        sort(p + st, p + q);
        int ret = 0;
        init();
        for (int i = 0; i < q; ++i) {
            int ra = find(p[i].a);
            int rb = find(p[i].b);
            if (ra != rb) {
                uni(ra, rb);
                ret += p[i].c;
            }
        }
        if (-parents[find(0)] != n) return 1e9;

        return ret;
    }
    
    vector<vector<int>> findCriticalAndPseudoCriticalEdges(int _n, vector<vector<int>>& edges) {
        n = _n, m = edges.size();
        for (int i = 0; i < m; ++i) {
            for (int j = 0; j < 3; ++j) {
                e[i][j] = edges[i][j];
            }
            p[i].a = e[i][0];
            p[i].b = e[i][1];
            p[i].c = e[i][2];
        }
        q = m;
        int z = mst(0); 
        vector<int> v1, v2;
        vector<vector<int> > ans;
        for (int i = 0; i < m; ++i) {

            q = 0;
            for (int j = 0; j < m; ++j) {
                if (i != j) {
                    p[q].a = e[j][0]; p[q].b = e[j][1]; p[q].c = e[j][2]; ++q;
                }
            }
            int t = mst(0);
            // printf("i=%d,t=%d\n",i,t);
            if (t > z) {
                v1.push_back(i); continue;
            }
            q = 0;
            for (int j = 0; j < m; ++j) {
                p[q].a = e[j][0]; p[q].b = e[j][1]; p[q].c = e[j][2]; ++q;
            }
            swap(p[0], p[i]);
            t = mst(1);
            if (t == z) {
                v2.push_back(i);
            }
        }
        ans.push_back(v1);
        ans.push_back(v2);
        return ans;
    }
};
