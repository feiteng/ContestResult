typedef pair<int,int> pii;
vector<vector<int>> edge;
bool cmp(int x,int y) {
    return edge[x][2]<edge[y][2];
}
class Solution {
public:
    int m,id[10005],f[105],table[10005],dep[105];
    pii fa[105];
    vector<pii> G[105];
    int F(int x) {
        return x==f[x]?x:f[x]=F(f[x]);
    }
    void U(int x,int y) {
        f[F(x)]=F(y);
    }
    void dfs(int u,int ff,int depth) {
        //printf("u=%d ff=%d\n",u,ff);
        dep[u]=depth;
        for (int i=0;i<(int)G[u].size();++i) {
            int v=G[u][i].first;
            if (v==ff) continue;
            fa[v]=pii(u,G[u][i].second);
            dfs(v,u,depth+1);
        }
    }
    vector<vector<int>> findCriticalAndPseudoCriticalEdges(int n, vector<vector<int>>& edges) {
        m=edges.size();
        for (int i=0;i<n;++i) f[i]=i;
        for (int i=0;i<m;++i) id[i]=i,table[i]=0;
        edge=edges;
        sort(id,id+m,cmp);
        for (int i=0;i<m;++i) {
            int u=edges[id[i]][0];
            int v=edges[id[i]][1];
            if (F(u)!=F(v)) {
                U(u,v);
                //printf("u=%d v=%d\n",u,v);
                G[u].push_back(pii(v,id[i]));
                G[v].push_back(pii(u,id[i]));
                table[id[i]]=2;
                //printf("id=%d\n",id[i]);
            }
        }
        dfs(0,-1,0);
        //for (int i=0;i<n;++i) printf("dep[%d]=%d\n",i,dep[i]);
        for (int i=0;i<m;++i) {
            if (table[id[i]]) continue;
            int u=edges[id[i]][0];
            int v=edges[id[i]][1];
            int w=edges[id[i]][2];
            if (dep[u]<dep[v]) swap(u,v);
            //if (id[i]==5) {
            //    printf("u=%d v=%d depu=%d depv=%d w=%d\n",u,v,dep[u],dep[v],w);
            //}
            int mmax=0;
            while (dep[u]>dep[v]) {
                int x=fa[u].second;
                mmax=max(mmax,edges[x][2]);
                if (edges[x][2]>=w)
                    table[x]=1;
                u=fa[u].first;
                //if (id[i]==5) {
                //    printf("u=%d mmax=%d\n",u,mmax);
                //}
            }
            //if (id[i]==5) {
            //    printf("u=%d v=%d w=%d\n",u,v,w);
            //}
            while (u!=v) {
                int x=fa[u].second;
                mmax=max(mmax,edges[x][2]);
                int y=fa[v].second;
                mmax=max(mmax,edges[y][2]);
                if (edges[x][2]>=w)
                    table[x]=1;
                if (edges[y][2]>=w)
                    table[y]=1;
                u=fa[u].first;
                v=fa[v].first;
                //if (id[i]==5) {
                //    printf("u=%d v=%d mmax=%d\n",u,v,mmax);
                //}
            }
            if (mmax>=w) table[id[i]]=1;
            //if (id[i]==5) {
            //    printf("mmax=%d u=%d v=%d w=%d\n",mmax,u,v,w);
            //}
        }
        vector<int> v1,v2;
        vector<vector<int>> ret;
        for (int i=0;i<m;++i) {
            if (table[i]==1) v2.push_back(i);
            if (table[i]==2) v1.push_back(i);
        }
        ret.push_back(v1);
        ret.push_back(v2);
        return ret;
    }
};
