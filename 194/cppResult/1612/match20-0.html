class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int i;
        map<int, int> record;
        set<int> index;
        vector<int> res(rains.size(), -1);
        
        for(i = 0; i < rains.size(); ++i) {
            if(rains[i]) {
                if(record.count(rains[i])) {
                    auto it = index.upper_bound(record[rains[i]]);
                    if(it == index.end()) return {};
                    res[*it] = rains[i];
                    index.erase(*it);
                }
                record[rains[i]] = i;
            } else index.insert(i);
        }
        
        for(auto it = index.begin(); it != index.end(); ++it) res[*it] = 1;
        
        return res;
    }
};
