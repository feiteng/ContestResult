class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        unordered_map<int,int> status;
        int n = rains.size();
        vector<int> res(n, -1);
        vector<int>em;
        for(int i = 0; i < n; i++) {
            int cur = rains[i];
            if(cur == 0) {
                em.push_back(i);
                continue;
            }
            if(status.count(cur)) {
                if(em.size() == 0 || em.back() < status[cur]) return vector<int>();
                else {
                    auto pos = lower_bound(em.begin(), em.end(), status[cur]);
                    res[*pos] = cur;
                    em.erase(pos);
                    status[cur] = i;
                }
            } else {
                status[cur] = i;
            }
        }
        
        for(int pos : em) res[pos] = 1;
        return res;
    }
};
