class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        unordered_map<int, int> lastday;
        deque<int> sunnyday;
        vector<int> ans(n, 1);
        for (int i = 0; i < n; i++) {
            if (rains[i]) {
                ans[i] = -1;
                auto &ld = lastday[rains[i]];
                if (ld > 0) {
                    int s = lower_bound(sunnyday.begin(), sunnyday.end(), ld) - sunnyday.begin();
                    if (s == sunnyday.size())
                        return {};
                    ans[sunnyday[s]] = rains[i];
                    sunnyday.erase(sunnyday.begin()+s);
                }
                ld = i + 1;
            } else {
                sunnyday.push_back(i);
            }
        }
        return ans;
    }
};
