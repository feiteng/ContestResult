class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> res(n, -1);
        set<int> v;
        unordered_map<int, int> state;
        for (int i = 0; i < n; i++) {
            if (rains[i] == 0) {
                v.insert(i);
            } else {
                if (state.find(rains[i]) != state.end()) {

                    if (v.size() == 0 || v.lower_bound(state[rains[i]]) == v.end()) {
                        return vector<int>(0);
                    }
                    int day = *v.lower_bound(state[rains[i]]);
                    v.erase(day);
                    state[rains[i]] = i;
                    res[day] = rains[i];
                } else {
                    state[rains[i]] = i;
                }
            }
        }
        for (auto d : v) {
            res[d] = 1;
        }
        return res;
    }
};
