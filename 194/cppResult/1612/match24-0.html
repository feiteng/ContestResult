class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> ret(n, -1);
        set<int> dry;
        unordered_map<int, int> lastRainedOn;
        for (int i = 0; i < n; ++i) {
            if (rains[i] == 0) {
                dry.insert(i);
            }
            else {
                if (lastRainedOn.count(rains[i]) > 0) {
                    auto it = dry.upper_bound(lastRainedOn[rains[i]]);
                    if (it == dry.end()) {
                        return {};
                    }
                    ret[*it] = rains[i];
                    dry.erase(it);
                }
                lastRainedOn[rains[i]] = i;
            }
        }
        for (auto it = dry.begin(); it != dry.end(); ++it) {
            ret[*it] = 1;
        }
        return ret;
    }
};
