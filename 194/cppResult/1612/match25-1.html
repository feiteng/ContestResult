class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> result(n, -1);
        set<int> s;
        unordered_map<int, int> m;
        for (int i = 0; i < n; ++i) {
            if (rains[i] == 0) {
                s.insert(i);
            } else {
                if (m.count(rains[i]) == 0) {
                    m[rains[i]] = i;

                } else {
                    auto it = s.upper_bound(m[rains[i]]);
                    if (it == s.end()) {
                        return {};
                    }
                    int dry = *it;
                    s.erase(it);
                    result[dry] = rains[i];
                    m[rains[i]] = i;
                }
            }
        }
        while (s.size() > 0) {
            result[*s.begin()] = 1;
            s.erase(s.begin());
        }
        return result;
    }
};
