class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        list<int> l;
        unordered_map<int,int> mp;
        vector<int> ans(rains.size(),-1);
        for(int i=0;i<rains.size();i++)
        {
            if(rains[i]==0)
            {
                l.push_back(i);
                //s.insert(i);
                continue;
            }
            if(mp.find(rains[i])==mp.end())
            {
                mp[rains[i]]=i;
                continue;
            }
            int pos=mp[rains[i]];
            if(l.empty() or l.back()<pos)
                return {};
            for(auto it=l.begin();it!=l.end();it++)
            {
                if(*it>pos)
                {
                    ans[*it]=rains[i];
                    l.erase(it);
                    break;
                }

            }
            mp[rains[i]]=i;
        }
        for(int i=0;i<rains.size();i++)
        {
            if(ans[i]==-1 && rains[i]==0)
                ans[i]=1;
        }
        return ans;
    }
};
