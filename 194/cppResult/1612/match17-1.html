class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int len = rains.size();
        /*
        unordered_map<int, vector<int>> idx;
        for (int i = 0; i < len; i++) {
            idx[rains[i]].push_back(i);
        }*/
        vector<int> ans(len, -1);
        unordered_map<int, int> full;
        set<int> zero;
        for (int i = 0; i < len; i++) {
            if (rains[i] > 0) {
                if (full.find(rains[i]) == full.end()) {
                    full[rains[i]] = i;
                }
                else {

                    auto itr = zero.upper_bound(full[rains[i]]);
                    if (itr != zero.end() && *itr < i) {
                        ans[*itr] = rains[i];
                        zero.erase(itr);
                        full[rains[i]] = i;
                    }
                    else {
                        return {};
                    }
                }
            }
            else {
                zero.insert(i);

            }
        }
        for (int i = 0; i < len; i++) {
            if (rains[i] == 0 && ans[i] == -1) {
                ans[i] = 1;
            }
        }
        return ans;
    }
};
