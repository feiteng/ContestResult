class Solution {
 public:
  vector<int> avoidFlood(vector<int>& rains) {
    int n = rains.size();
    vector<int> res(n, -1);
    set<int> days;
    unordered_map<int, int> last_day;
    for (int i = 0; i < n; i++) {
      if (rains[i] == 0)
        days.insert(i);
    }

    for (int i = 0; i < n; i++) {
      if (rains[i] > 0 && last_day.find(rains[i]) != last_day.end()) {
        int last = last_day[rains[i]];
        auto ub = days.upper_bound(last);
        if (ub == days.end() || (*ub > i)) {
          return {};
        } else {
          res[*ub] = rains[i];
          days.erase(ub);
        }
      }
      last_day[rains[i]] = i;
    }
    for (int lake : days) {
      res[lake] = 1;
    }
    return res;
  }
};
