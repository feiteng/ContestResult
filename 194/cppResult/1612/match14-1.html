class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        vector<int> ans(rains.size());
        map<int, int> full_lakes;
        set<int> days_without_rain;
        for(int i=0;i<rains.size();++i) {
            if(rains[i] > 0) {
                ans[i] = -1;
                if(full_lakes.find(rains[i]) != full_lakes.end()) {
                    int prev_day = full_lakes[rains[i]];

                    auto it = days_without_rain.lower_bound(prev_day);
                    if(it == days_without_rain.end()) {
                        return {};
                    }
                    ans[*it] = rains[i];
                    days_without_rain.erase(it);
                }
                full_lakes[rains[i]] = i;
            }
            else {
                days_without_rain.insert(i);
            }
        }
        for(auto d: days_without_rain) {
            ans[d] = 1;
        }
        return ans;
    }
};
