class Solution {
public:
  vector<int> avoidFlood(vector<int>& rains) {  
    // lake, day
    unordered_map<int, int> lakes;
    set<int> dry;
    vector<int> res;
    for (int i = 0, d = 0; i < rains.size(); ++i) {

      int r = rains[i];
      
      if (r == 0) {
        dry.insert(i);
        res.push_back(0);
      } else {
        
        if (lakes.count(r) > 0) {
          auto it = dry.upper_bound(lakes[r]);
          if (it == dry.end()) {
            return {};

          }

          res[*it] = r;
          dry.erase(it);
        }
        
        
        lakes[r] = i;
        res.push_back(-1);
        
      }
    }
    
    for (int i = 0; i < res.size(); ++i) {
      if (res[i] == 0) {
        res[i] = 1;
      }
    }
    
    return res;
  }
};
