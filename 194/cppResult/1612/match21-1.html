class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        unordered_map<int, int> full;
        int n = rains.size();
        vector<int> res(n, -1);
        set<int> A;
        for (int i = 0; i < n; ++i) {
            int cur = rains[i];
            if (cur == 0) {
                A.insert(i);
            } else if (full.count(cur)) {
                int rainedday = full[cur];

                auto p = A.upper_bound(rainedday);
                if (p == A.end())
                    return {};
                res[*p] = cur;
                A.erase(p);
                full[cur] = i;
            } else {
                full[cur] = i;
            }
        }
        for (int i : A) res[i] = 1;
        return res;
    }
};
