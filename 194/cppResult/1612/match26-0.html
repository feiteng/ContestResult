class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        vector<int> res(rains.size());
        map<int, int> full;
        set<int> dry;
        for (int i = 0; i < rains.size(); ++ i)
        {
            if (rains[i] == 0)
            {
                dry.insert(i);
            }
            else
            {
                res[i] = -1;

                if (full.count(rains[i]) > 0)
                {
                    int t = full[rains[i]];
                    //std::cout << rains[i] << " " << t << "\n";
                    auto it = dry.upper_bound(t);
                    if (it != dry.end())
                    {
                        res[*it] = rains[i];
                        dry.erase(it);
                    }
                    else
                    {
                        return vector<int>();
                    }
                }
                full[rains[i]] = i;
            }
        }
        for (int d : dry)
        {
            res[d] = 1;
        }
        return res;
    }
};
