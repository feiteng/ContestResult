class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> result(n, -1);
        set<int> pre;
        map<int, int> last;
        
        for (int i = 0; i < n; ++i) {
            if (rains[i] == 0) {
                result[i] = 1;
                pre.insert(i);
            }
            else {
                if (last.count(rains[i])) {
                    int idx = last[rains[i]];

                    auto it = pre.lower_bound(idx);
                    if (it == pre.end()) {
                        return {};
                    }
                    result[*it] = rains[i];
                    pre.erase(it);
                }
                last[rains[i]] = i;
            }
        }
        
        return result;
    }
};
