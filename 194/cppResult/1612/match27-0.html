class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> ans(n, -1);
        unordered_map<int, int> mp, mp2;
        set<int> st;
        for (int i = 0; i < n; ++i) {
            if (rains[i] == 0) {
                st.insert(i);
            }else {
                if (mp[rains[i]] > 0) {
                    if (st.empty()) return {};
                    int idx = mp[rains[i]];
                    auto p = st.lower_bound(idx);
                    if (p == st.end()) return {};
                    ans[*p] = rains[i];
                    st.erase(p);
                    mp[rains[i]] = 0;
                }
                mp[rains[i]] = i + 1;
            }
        }
        for (auto p : st) {
            ans[p] = 1;
        }
        return ans;
    }
};
