class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        unordered_map<int, int> occur;
        set<int> rem;
        vector<int> base(rains.size());
        vector<int> res(rains.size(), -1);
        for(int i = 0; i < rains.size(); ++i) {

            if(rains[i] > 0) {
                if(occur.count(rains[i])) {
                    auto it = rem.lower_bound(occur[rains[i]]);
                    if(it == rem.end() || *it > i)
                        return {};
                    res[*it] = rains[i];
                    rem.erase(it);
                }
                occur[rains[i]] = i;
            }
            else  {
                rem.insert(i);
            }
        }
        for(int i = 0; i < rains.size(); ++i)
            if(rains[i] == 0 && res[i] == -1)
                res[i] = 1;
        return res;
    }
};
