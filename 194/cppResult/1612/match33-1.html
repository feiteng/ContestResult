class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        unordered_map<int, int> lakes;
        set<int> days;
        vector<int> res(n, 1);
        for (int i = 0; i < n; i++) {
            if (rains[i] == 0) {
                days.insert(i);
            } else {
                res[i] = -1;

                if (lakes.count(rains[i]) > 0) {
                    if (days.empty()) {
                        return {};
                    } else {
                        auto it = days.lower_bound(lakes[rains[i]]);
                        if (it == days.end()) {
                            return {};
                        }
                        int day = *it;
                        res[day] = rains[i];
                        lakes.erase(rains[i]);
                        days.erase(day);
                    }
                }
                lakes[rains[i]] = i;
            }
        }
        return res;
    }
};
