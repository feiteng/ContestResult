class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n=rains.size();
        unordered_map<int,int> m;
        set<int> se;
        vector<int> v;
        vector<int> ans(n,-1);
        int z=0;
        for (int i=0;i<rains.size();++i) {
            int rain=rains[i];
            if (rain==0) {
                se.insert(i);
                continue;
            }
            if (m.count(rain)) {
                
                auto it=se.upper_bound(m[rain]);
                if (it==se.end()||*it>i) return {};
                int j=*it;
                // cout<<rain<<" "<<j<<endl;
                // if (j>i) return {};
                ans[j]=rain;
                se.erase(j);
            }
            // for (auto it=se.begin();it!=se.end();++it) cout<<*it<<endl;
            m[rain]=i;
        }
        // cout<<endl;
        for (int i=0;i<n;++i) {
            if (rains[i]==0&&ans[i]==-1) ans[i]=1;
        }
        
        // int j=0;
        // for (int i=0;i<n;++i) {
        //     if (rains[i]==0) {
        //         if (j>=v.size()) {
        //             ans[i]=1;
        //         } else {
        //             ans[i]=v[j++];
        //         }
        //     }
        // }
        return ans;
    }
};
