class Solution {
public:
  vector<int> avoidFlood(vector<int>& rains) {
    map<int, int> M;
    set<int> Q;
    vector<int> ans(rains.size(), -1);
    for (int i = 0; i < rains.size(); i++) {
      if (rains[i] == 0) Q.insert(i);
      else {
        if (M.count(rains[i])) {
          auto it = Q.upper_bound(M[rains[i]]);
          if (it == Q.end()) return {};
          ans[*it] = rains[i];
          Q.erase(it);
        }
        M[rains[i]] = i;
      }
    }
    for (auto t: Q) ans[t] = 1;
    return ans;
  }
};
