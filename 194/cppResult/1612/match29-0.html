class Solution {
public:

    vector<int> avoidFlood(vector<int>& rains) {
        map<int,int>pre;
        set<int>have;
        vector<int>ans;
        for(int i=0;i<rains.size();i++)ans.push_back(-1);
        for(int i=0;i<rains.size();i++){
            if(rains[i]>0){
                int &x=rains[i];
                if(pre.count(x)){
                    auto it=have.lower_bound(pre[x]);
                    if(it==have.end())return vector<int>();
                    ans[*it]=rains[i];
                    have.erase(it);
                }
                pre[x]=i;
            }
            else have.insert(i);
        }
        for(auto x:have)ans[x]=1;
        return ans;
    }
};
