class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int lr = rains.size();
        unordered_map<int, int> f;
        
        vector<int> ans(lr, -1);
        
        
        set<int> st;
        
        for (int i = 0; i < lr; i++) {
            if (rains[i] > 0) {
                int r = rains[i];
                
                if (f.count(r)) {
                    int first = f[r];
                    
                    auto p = st.upper_bound(first);
                    
                    if (p == st.end()) return {};
                    ans[*p] = r;
                    st.erase(p);
                }
                
                f[rains[i]] = i;
            } else {
                st.insert(i);
                ans[i] = 1;
            }
        }
        
        return ans;
    }
};
