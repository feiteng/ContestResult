class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> ans(n, -2);
        map<int, int> left;
        set<int> dry;
        for (int i = 0; i < n; ++i) {
            if (rains[i] > 0) {
                if (left.count(rains[i]) && left[rains[i]] != -1) {
                    auto ret = dry.upper_bound(left[rains[i]]);
                    if (ret == dry.end()) {
                        return vector<int>();
                    }
                    ans[*ret] = rains[i];
                    dry.erase(ret);
                    
                }
                left[rains[i]] = i;
                
                ans[i] = -1;
                continue;
            }
            dry.insert(i);
        }
        
        for (int i = 0; i < n; ++i) {
            if (ans[i] == -2) ans[i] = 1;
        }
        return ans;
        
    }
};
