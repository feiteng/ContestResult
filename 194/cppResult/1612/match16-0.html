class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        vector<int> ans;
        set<int> dry;
        unordered_map<int, int> lake;
        
        for (int i = 0; i < rains.size(); ++i) {
            if (rains[i] != 0) {           
                auto lake_it = lake.find(rains[i]);
                
                if (lake_it != lake.end()) {
                    auto dry_it = dry.upper_bound(lake_it->second);

                    if (dry_it == dry.end()) {
                        return {};
                    }
                    ans[(*dry_it)] = rains[i];
                    dry.erase(dry_it);
                }
                lake[rains[i]] = i;
                ans.push_back(-1);
            } else {
                ans.push_back(-100);
                dry.insert(i);
            }
        }
        
        for (auto &i : ans) {
            if (i == -100) {
                i = 1;
            }
        }
        
        return ans;
    }
};
