class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> ans(n, -1);
        unordered_map<int, int> seen;
        set<int> st;
        for (int i = 0; i < n; i++) {
            int lake = rains[i];
            if (!lake) st.insert(i);
            else if (lake > 0) {
                if (seen.count(lake)) {
                    //cout << i << ' ' << rains[i] << ' ';
                    auto it = st.upper_bound(seen[lake]);
                    if (it == st.end()) return {};
                    //cout << *it << endl;
                    ans[*it] = lake;
                    st.erase(it);
                    seen.erase(lake);
                }
                seen[lake] = i;
            }
        }
        
        for (int i : st) ans[i] = 1;
        return ans;
    }
};
