class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        vector<int> ans(rains.size());
        map<int,int> m;
        set<int> s;
        for(int i=0;i<rains.size();i++){
            if(rains[i]>0){

                if(m[rains[i]]!=0){
                    int pre = m[rains[i]];
                    auto it = s.upper_bound(pre);
                    if(it!=s.end()&&(*it)<i+1){
                        ans[(*it)-1] = rains[i];
                        s.erase(it);
                    }else{
                        return vector<int>();
                    }
                }
                m[rains[i]]=i+1;
                ans[i]=-1;
            }else{
                s.insert(i+1);
            }
        }
        for(int i=0;i<ans.size();i++){
            if(ans[i]==0)ans[i]=1;
        }
        return ans;
    }
};
