class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> res(n, -1);
        set<int> q;
        map<int, int> h;
        for (int i = 0; i < n; ++i) {
            if (rains[i] == 0) {
                q.insert(i);
            } else {
                if (h.find(rains[i]) != h.end()) {
                    auto it = q.lower_bound(h[rains[i]]);
                    if (it == q.end())
                        return vector<int>();
                    res[*it] = rains[i];
                    q.erase(it);
                }
                h[rains[i]] = i;
            }
        }
        for (const auto& i : q) {
            res[i] = 1;
        }
        return res;
    }
};
