class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        map<int,int> full;
        set<int> empty;
        vector<int> ans(n,-1);
        
        for(int i = 0; i < rains.size(); ++i){
            if(rains[i] == 0){
                empty.insert(i);
            }else if(rains[i] > 0){
                if(full.count(rains[i])){
                    if(empty.size() == 0) return vector<int>({});
                    auto it = empty.lower_bound(full[rains[i]]);
                    if(it == empty.end()) return vector<int>({});

                    int x = *it;
                    ans[x] = rains[i];
                    empty.erase(x);
                }
                full[rains[i]] = i;
            }
        }
        
        for(auto v : empty){
            ans[v] = 1;
        }
        
        return ans;
    }
};
