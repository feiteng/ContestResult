class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n=rains.size();
        vector<int> ans(n,1);
        set<int> se;
        map<int,int> pre;
        for(int i=0;i<n;i++)
        {
            if(rains[i]>0)
            {
                ans[i]=-1;
                if(pre.count(rains[i]))
                {
                    auto p=se.lower_bound(pre[rains[i]]);
                    if(p==se.end()) return {};

                    ans[*p]=rains[i];
                    se.erase(p);
                }
                pre[rains[i]]=i;
            }
            else
            {
                se.insert(i);
            }
        }
        return ans;
    }
};
