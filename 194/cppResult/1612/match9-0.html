class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n=rains.size();
        unordered_map<int,int> mmap;//<第几片湖泊，是否有雨水>
        set<int> mset;//第几天可以抽水
        vector<int> result;
        for(int i=0;i<n;++i)
        {
            if(rains[i]==0)
            {
                mset.insert(i);
                result.push_back(1);
            }else{
                if(mmap.count(rains[i]))
                {
                    int pre=mmap[rains[i]];
                    auto it=mset.begin();
                    while(it!=mset.end()&&(*it)<pre)
                    {
                        ++it;
                    }
                    if(it==mset.end())
                    {
                        return vector<int>();
                    }
                    int day=(*it);
                    result[day]=rains[i];
                    mset.erase(it);
                    mmap[rains[i]]=i;
                }else{
                    mmap[rains[i]]=i;
                }
                result.push_back(-1);
            }
        }
        return result;
    }
};
