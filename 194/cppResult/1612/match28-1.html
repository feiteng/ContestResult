class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> res(n, -1);
        map<int, int> vis;
        set<int> st;
        for (int i = 0; i < n; ++i) {
            if (rains[i] == 0) {
                st.insert(i);
            } else {
                if (vis.count(rains[i])) {
                    auto it = st.upper_bound(vis[rains[i]]);
                    if (it == st.end()) return vector<int>();
                    res[*it] = rains[i];
                    vis[rains[i]] = i;
                    st.erase(it);
                } else {
                    vis[rains[i]] = i;
                }
            }
        }
        for (const int &i : st) res[i] = 1;
        return res;
    }
};
