class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        vector<int> ans(rains.size());
        map<int, int> pos;
        set<int> zeros;
        for (int i = 0; i < rains.size(); i++) {
            if (rains[i] == 0) {
                zeros.insert(i);
            }
        }
        int j = 0;
        for (int i = 0; i < rains.size(); i++) {
            if (rains[i] == 0) continue;
            if (pos.find(rains[i]) != pos.end()) {

                auto it = zeros.lower_bound(pos[rains[i]]);
                if (it == zeros.end() || *it > i) return {};
                ans[*it] = rains[i];
                zeros.erase(it);
            }
            pos[rains[i]] = i;
        }
        for (int i = 0; i < rains.size(); i++) {
            if (ans[i] == 0) {
                ans[i] = 1;
            }
            if (rains[i] > 0) {
                ans[i] = -1;
            }
        }
        
        return ans;
    }
};
