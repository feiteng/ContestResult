class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        unordered_map<int,int> map;
        int n=rains.size();
        vector<int> res(n,-1);
        set<int> zero;

        for(int i=0;i<n;i++){
            int num = rains[i];
            if(num>0){
                if(map.count(num)>0){
                    if(zero.empty()) return {};
                    auto it = zero.lower_bound(map[num]);
                    if(it==zero.end()) return {};
                    res[*it] = num;
                    zero.erase(it);
                    map[num] = i;
                }
                else{
                    map[num]=i;
                }
            }
            else{
                zero.insert(i);
                res[i]=1;
            }
        }
        return res;
    }
};
