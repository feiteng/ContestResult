class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        set<int> slots;
        unordered_map<int, int> prev;
        vector<int> ans(rains.size(), -1);
        for (int i = 0; i < rains.size(); ++i) {
            if (rains[i] == 0) {
                slots.insert(i);
                continue;
            }

            if (prev.count(rains[i])) {
                int prev_rain = prev[rains[i]];
                auto slot = slots.lower_bound(prev_rain);
                if (slot == slots.end()) return {};
                ans[*slot] = rains[i];
                slots.erase(slot);
            }
            prev[rains[i]] = i;
        }
        
        for (int i = 0; i < rains.size(); ++i) {
            if (rains[i] == 0 && ans[i] == -1) ans[i] = 1;
        }
        return ans;
    }
};
