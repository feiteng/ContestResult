class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int n = rains.size();
        vector<int> ans(n, -1);
        map<int,int> lastFill;
        set<int> zeroes;
        for (int i = 0; i < n; ++i) {
            int target = rains[i];
            if (target == 0) {
                zeroes.insert(i);
            } else {
                if (lastFill.find(target) != lastFill.end()) {
                    if (zeroes.size() == 0) {
                        return {};
                    }
                    auto it = zeroes.lower_bound(lastFill[target]);
                    if (it != zeroes.end()) {
                        ans[*it] = target;
                        zeroes.erase(it);
                        lastFill[target] = i;
                    } else {
                        return {};
                    }
                } else {
                    lastFill[target] = i;
                }
            }
        }
        for (int i = 0; i < n; ++i) {
            if (ans[i] == -1 && rains[i] == 0) {
                ans[i] = 1;
            }
        }
        return ans;
    }
};
