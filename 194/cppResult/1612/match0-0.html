class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        int len = rains.size();
        vector<int> ans(len,1);
        set<int> zeros;
        unordered_map<int,int> ones;//key为有水的湖泊,value为下雨的天
        for(int i=0;i<len;++i)
        {
            //开始下雨
            if(rains[i]>0)
            {
                ans[i] = -1;
                if(ones.count(rains[i]))//之前有下过雨
                {
                    auto iter = zeros.upper_bound(ones[rains[i]]);//找到下雨之后的最近一个晴天
                    if(iter == zeros.end())//没有找到，洪水
                        return vector<int>{};
                    ans[*iter] = rains[i];
                    zeros.erase(iter);
                }
                ones[rains[i]] = i;
            }
            else
            {
                zeros.emplace(i);
            }
        }
        return ans;
    }
};
