class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        set<int> zss;
        unordered_map<int, int> mm;
        vector<int> ans(rains.size(), -1);
        for (int i = 0; i < rains.size(); ++i)
        {
            if (rains[i] == 0)
            {
                zss.insert(i);
            }
            else
            {

                auto iter = mm.find(rains[i]);
                if (iter != end(mm))
                {
                    auto it = zss.upper_bound(iter->second);
                    if (it == end(zss))
                    {
                        return {};
                    }
                    ans[*it] = iter->first;
                    zss.erase(it);
                    mm.erase(iter);
                }
                mm.emplace(make_pair(rains[i], i));
            }
        }
        for (auto it : zss)
        {
            ans[it] = 1;
        }
        return ans;
    }
};
