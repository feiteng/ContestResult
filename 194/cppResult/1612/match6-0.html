class Solution {
public:
    vector<int> avoidFlood(vector<int>& rains) {
        
        vector<int> ans(rains.size(), 1);
        
        unordered_map<int,int> full;
        set<int> drys;
            
        for (int i =0;i < rains.size();++i) {
            int r = rains[i];
            if (r == 0) {
                drys.insert(i);
            } else {
                if (full.find(r) != full.end()) {
                    auto it = drys.upper_bound(full[r]);
                    if (drys.end() == it) {
                        return vector<int>();
                    } 
                    ans[*it] = r;
                    
                    drys.erase(*it);
                } 
                full[r] = i;
            }
            
        }
        
        for (int i=0; i<rains.size();++i) {
            if (rains[i] != 0) {
                ans[i] = -1;
            }
        }
        return ans;
    }
};
