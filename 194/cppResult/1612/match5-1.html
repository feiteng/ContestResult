class Solution {
public:
  vector<int> avoidFlood(vector<int>& rains) {
    int n = rains.size();
    set<int> s;
    unordered_map<int,int> last_filled;
    vector<int> ans(n, -1);
    for (int i = 0; i < n; ++i) {
      int l = rains[i];
      if (l == 0) {
        s.emplace(i);
        continue;
      }

      if (last_filled.count(l)) {
        auto it = s.upper_bound(last_filled[l]);
        if (it == s.end()) return {};
        ans[*it] = l;
        s.erase(it);
      }
      last_filled[l] = i;
    }
    for (auto& a : s) {
      ans[a] = 1;
    }
    return ans;
  }
};
