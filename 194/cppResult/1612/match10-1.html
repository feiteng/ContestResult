class Solution {
public:
    vector<int> avoidFlood(vector<int>& a) {
        int n = a.size();
        vector<int> ret(n, -1);
        set<int> B;
        map<int, int> A;
        for (int i = 0; i < n; ++i) {
            if (a[i] > 0) {
                if (A.count(a[i])) {
                    auto it = B.lower_bound(A[a[i]]);
                    if (it == B.end()) return {};
                    ret[*it] = a[i];
                    B.erase(it);
                }
                A[a[i]] = i;
            } else {
                B.insert(i);

            }
        }
        for (int i = 0; i < n; ++i) {
            if (a[i] == 0 && ret[i] < 0) ret[i] = 1;
        }
        return ret;
    }
};
