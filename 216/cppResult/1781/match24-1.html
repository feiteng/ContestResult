class Solution {
public:
    bool arrayStringsAreEqual(vector<string>& word1, vector<string>& word2) {
        string ret1, ret2;
        for (const auto &w : word1)
            ret1 += w;
        for (const auto &w : word2)
            ret2 += w;
        return ret1 == ret2;
    }
};
