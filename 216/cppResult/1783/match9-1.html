class Solution {
public:
    void print(vector<int> a) {
        for (auto x: a) {
            cout << x << " ";
        }
        cout << endl;
    }
    

    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> oddBefore(n), evenBefore(n), oddAfter(n), evenAfter(n);
        int odd = 0, even = 0;
        for (int i=0; i<n; i++) {
            oddBefore[i] = odd;
            evenBefore[i] = even;
            if (i%2) {
                odd += nums[i];
            } else {
                even += nums[i];
            }
        }
        odd = 0, even = 0;
        for (int i=n-1; i>=0; i--) {
            oddAfter[i] = odd;
            evenAfter[i] = even;
            if (i%2) {
                odd += nums[i];
            } else {
                even += nums[i];
            }
        }
        // print(oddBefore);
        // print(evenBefore);
        // print(oddAfter);
        // print(evenAfter);
        int ans = 0;
        for (int i=0; i<n; i+=1) {
            if (oddBefore[i] + evenAfter[i] == oddAfter[i] + evenBefore[i]) {
                ans++;
            }
        }
        return ans;
    }
};
