class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int sz = nums.size();
        vector<int> evl(sz,0), odl(sz,0), evr(sz,0), odr(sz,0);
        
        for(int i = 1 ; i < sz ; ++i) {
            if(i % 2 == 0) {
                evl[i] = evl[i-1];
                odl[i] = odl[i-1] + nums[i-1];
            } else {
                odl[i] = odl[i-1];
                evl[i] = evl[i-1] + nums[i-1];
            }
        }
        
        for(int i = sz - 2 ; i >= 0 ; --i) {
            if(i % 2 == 0) {
                evr[i] = evr[i+1];
                odr[i] = odr[i+1] + nums[i+1];
            } else {
                odr[i] = odr[i+1];
                evr[i] = evr[i+1] + nums[i+1];
            }
        }
        
        int count = 0;
        for(int i = 0 ; i < sz ; ++i) {
            if(evl[i] + odr[i] == odl[i] + evr[i]) {
                //cout <<"ix: " << i << endl;
                ++count;
            }
        }
        
        return count;
    }
};
