class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int len = nums.size();
        vector<int> peven(len+1, 0);
        vector<int> podd(len+1, 0);
        vector<int> beven(len+1, 0);
        vector<int> bodd(len+1, 0);
        for(int i = 1 ; i < len ; i ++) {
            if(i & 1 == 1) {
                podd[i] = podd[i-1];
                peven[i] = peven[i-1] + nums[i-1];
            }
            else {
                podd[i] = podd[i-1] + nums[i-1];
                peven[i] = peven[i-1];
            }
        }
        for(int i = len - 2; i >= 0 ; i --) {
            if(i & 1 == 1) {
                bodd[i] = bodd[i+1];
                beven[i] = beven[i+1] + nums[i+1];
            }
            else {
                bodd[i] = bodd[i+1] + nums[i+1];
                beven[i] = beven[i+1];
            }
        }
        // for(int i = 0 ; i <= len ; i ++) {
        //     cout << peven[i] << " ";
        // }
        // cout << endl;
        // for(int i = 0 ; i <= len ; i ++) {
        //     cout << podd[i] << " ";
        // }
        // cout << endl;
        // for(int i = 0 ; i <= len ; i ++) {
        //     cout << beven[i] << " ";
        // }
        // cout << endl;
        // for(int i = 0 ; i <= len ; i ++) {
        //     cout << bodd[i] << " ";
        // }
        // cout << endl;
        int res = 0;
        for(int i = 0 ; i < len ; i ++) {
            if(peven[i] + bodd[i] == podd[i] + beven[i]) {
                res ++;
            }
        }
        return res;
    }
};
