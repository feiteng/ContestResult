class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int len = nums.size();
        vector<int> os(len + 1), es(len + 1); // s[i+1]包括n[i]的左边和
        
        for (int i = 0;i < len;++i) {
            es[i + 1] += es[i];
            os[i + 1] += os[i];

            if (i % 2 == 0) {
                es[i + 1] += nums[i];
            } else {
                os[i + 1] += nums[i];
            }
        }
        
        int ans = 0;
        for (int i = 0;i < nums.size();++i) {
            if (es[i] + os[len] - os[i+1] == os[i] + es[len] - es[i+1]) {
                ++ans;
            }
        }
        
        return ans;
    }
};
