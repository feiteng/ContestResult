class Solution {
public:
    int waysToMakeFair(vector<int>& A) {
        int n = A.size();
        vector<int> X (n+5,0), Y(n+5,0);
        for (int i = n - 1; i >= 0; i--) {
            X[i] = X[i+1];
            Y[i] = Y[i+1];
            if (i % 2==0) X[i] += A[i];
            else Y[i] += A[i];
        }
        
        int ans = 0;
        int x = 0, y = 0;
        for (int i = 0; i < n; i++) {
            if (x + Y[i+1] == y + X[i+1]) ans++;
            if (i % 2 == 0) x+= A[i];
            else y += A[i];
        }
        return ans;
    }
};
