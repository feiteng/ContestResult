class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int size = nums.size();
        vector<int> ji_sum(size+1);
        vector<int> ou_sum(size+1);
        int tmp_ji = 0;

        int tmp_ou = 0;
        for(int i=0;i<size;i++) {
            if (i%2==0) {
                tmp_ou += nums[i];
            }else{
                tmp_ji += nums[i];
            }
            ji_sum[i+1] = tmp_ji;
            ou_sum[i+1] = tmp_ou;

        }
        int res = 0;
        for(int i=0;i<size;i++) {
            int ji = ji_sum[i] + ou_sum[size] - ou_sum[i+1];
            int ou = ou_sum[i] + ji_sum[size] - ji_sum[i+1];
            if (ji==ou) {
                res++;
            }
        }
        return res;
    }
};
