class Solution {
public:
    int waysToMakeFair(vector<int>& v) 
    {
        int ans = 0;

        int n = v.size();
        int suffo[n+5];
        int suffe[n+5];
        int odd = 0;
        int even = 0;
        for(int i=n-1;i>=0;i--)
        {
            suffo[i] = odd;
            suffe[i] = even;
            if(i%2==1)
                odd += v[i];
            else
                even += v[i];
        }

        odd = 0;
        even = 0;
        for(int i=0;i<n;i++)
        {
            if(odd+suffe[i]==even+suffo[i])
                ans++;
            if(i%2==1)
                odd += v[i];
            else
                even += v[i];
        }
        return ans;
    }
};
