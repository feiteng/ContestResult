class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int>sumo(n+1,0);
        vector<int>sume(n+1,0);
        for(int i = 0;i<n;++i){
            if(i&1){
                sumo[i+1]=sumo[i]+nums[i];
                sume[i+1]=sume[i];
            }else{
                sume[i+1]=sume[i]+nums[i];
                sumo[i+1]=sumo[i];
            }
        }
        int ans = 0;
        for(int i = 1;i<=n;++i){
            int sumoo = sumo[i-1]+sume[n]-sume[i];
            int sumee = sume[i-1]+sumo[n]-sumo[i];
            if(sumee==sumoo)ans++;
        }
        return ans;
    }
};
