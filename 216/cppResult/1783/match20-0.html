class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> sfix(n+2,0);
        for(int i = n-1;i>=0;i--){
            sfix[i] = sfix[i+2]+nums[i];
        }
        vector<int> pfix(n+2,0);
        for(int i = 0;i<n;i++){
            pfix[i+2] = pfix[i]+nums[i];
        }
        int ans = 0;
        for(int i = 0;i<n;i++){
            int one = pfix[i-1+2] + sfix[i+2];
            int two = pfix[i-2+2] + sfix[i+1];
            if(one==two) ans+=1;
        }
        return ans;
    }
};
