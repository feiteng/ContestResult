class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> z1(n+1), z2(n+1);
        for (int i=0; i<n; ++i) {
            z1[i+1]=z1[i];
            if (i%2==0) z1[i+1] -= nums[i];
            else z1[i+1] += nums[i];
        }
        
        for (int i=n-1; i>=0; --i) {
            z2[i] = z2[i+1];
            if (i%2==0) z2[i] -= nums[i];
            else z2[i] += nums[i];
        }
        
        int res = 0;
        for (int i=0; i<n; ++i) {
            int a = z1[i];
            int b = z2[i+1];
            //b = -b;
            //cout<<i<<':'<<a-b<<',';
            if (a==b) {
                ++res;
            }
        }
        cout<<endl;
        return res;
    }
};
