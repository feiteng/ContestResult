class Solution {
public:
  int waysToMakeFair(vector<int>& nums) {
    
    int sz = nums.size();
    if (sz == 1) {
      return 1;
    }
    int sum[2] = {0};
    for (int i = 0; i < sz; i++) {
      sum[i % 2] += nums[i];
    }
    int rv = 0;
    
      cout << sum[0] << " " << sum[1] << endl;
    for (int i = sz - 1; i >=0; i--) {
      sum[i % 2] -= nums[i];
      if (i != sz - 1) {
        sum[i % 2] += nums[i + 1];
      }
      if (sum[0] == sum[1]) {
        rv++;
      }
      // cout << sum[0] << " " << sum[1] << endl;
    }
    return rv;
  }
};
