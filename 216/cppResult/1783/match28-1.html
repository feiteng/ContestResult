class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();

        int eve =0, odd =0;
        
        for(int i=0;i<n;++i){
            if(i&1)
                odd+=nums[i];
            else 
                eve+=nums[i];
        }
        int a = 0,b=0;
        int ans = 0;

        for(int i=0;i<n;++i){
            if(i&1){
                odd -= nums[i];
                if(a+odd == b+ eve)
                    ++ans;
                b+=nums[i];
                
            }
            else{
                eve-=nums[i];
                if(a+odd == b+ eve)
                    ++ans;
                a+=nums[i];
            
            }
        }
       return ans;
    }
};
