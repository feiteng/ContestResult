class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n), even(n), prev_odd(n), prev_even(n), next_odd(n), next_even(n);
        for (int i = 0; i < n; ++i) {
            if (i % 2 == 0) { // even
                even[i] = (i > 0 ? even[i-1]: 0) + nums[i];
                odd[i] = (i > 0 ? odd[i-1]: 0);
            } else {
                odd[i] = (i > 0 ? odd[i-1]: 0) + nums[i];
                even[i] = (i > 0 ? even[i-1]: 0);
            }
        }
        for (int i = 0; i < n; ++i) {
            if (i % 2 == 0) {
                prev_even[i] = even[i] - nums[i];
                prev_odd[i] = odd[i];
            } else {
                prev_odd[i] = odd[i] - nums[i];
                prev_even[i] = even[i];
             
            }
            next_even[i] = odd[n-1] - odd[i];
            next_odd[i] = even[n-1] - even[i];

        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (prev_even[i] + next_even[i] == prev_odd[i] + next_odd[i]) {
                ++ans;
            }
        }
        return ans;
    }
};
