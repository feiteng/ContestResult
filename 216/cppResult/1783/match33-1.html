class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int len=nums.size();
        int odd[100005]={0},even[100005]={0};
        for(int i=0;i<nums.size();i++)
        {
            if(i==0) 
            {
                even[i]=nums[i];
                odd[i]=0;
            }
            else if(i==1)
            {
                even[i]=even[i-1];
                odd[i]=nums[i];
            }
            else
            {
                if(i%2)
                {
                    even[i]=even[i-1];
                    odd[i]=odd[i-1]+nums[i];
                }
                else
                {
                    even[i]=even[i-1]+nums[i];
                    odd[i]=odd[i-1];
                }
            }
            

        }
        int count=0;
        for(int i=0;i<len;i++)
        {
            if(i==0)
            {
                if(even[len-1]-even[0]==odd[len-1]) count++;
            }
            else 
            {
                if(even[i-1]+odd[len-1]-odd[i]==odd[i-1]+even[len-1]-even[i]) count++;
            }
        }
        return count;
    }
};
