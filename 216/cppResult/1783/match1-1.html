class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> even(n+1, 0), odd(n+1, 0);
        for (int i = 0; i < n; ++i) {
            even[i+1] = even[i];
            odd[i+1] = odd[i];
            if (i % 2 == 0) even[i+1] += nums[i];
            else odd[i+1] += nums[i];
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            if (even[i] + (odd[n]-odd[i+1]) == odd[i] + (even[n]-even[i+1])) ++ans;
        }
        return ans;
    }
};
