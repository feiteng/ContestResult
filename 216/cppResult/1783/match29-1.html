class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {

        int t_even = 0, t_odd = 0;
        for (int i = 0; i < nums.size(); i++) {
            if ((i & 1) == 0) t_even += nums[i];
            else t_odd += nums[i];
        }
        int res = 0;

        int te = 0, to = 0;
        for (int i = 0; i < nums.size(); i++) {
            if ((i & 1) == 0) {
                // cout << t_even << " " << to << " == " << t_odd << " " << te << endl;
                t_even -= nums[i];
                if (t_even + to == t_odd + te) res++;
                te += nums[i];
            } else {
                // cout << t_even << " " << to << " == " << t_odd << " " << te << endl;
                t_odd -= nums[i];
                if (t_even + to == t_odd + te) res++;
                to += nums[i];
            }
        }
        return res;
    }
};
