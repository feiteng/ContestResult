class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n + 1, 0), even(n + 1, 0);
        for(int i = 1; i <= n; i ++ ){
            if(i & 1) odd[i] = odd[i - 1] + nums[i - 1], even[i] = even[i - 1];

            else even[i] = even[i - 1] + nums[i - 1], odd[i] = odd[i - 1];
        }
        int res = 0;
        for(int i = 1; i <= n; i ++ ){
            int l = odd[i - 1] + even[n] - even[i];
            int r = even[i - 1] + odd[n] - odd[i];
            if(l == r) res ++ ;
        }
        return res;
    }
};
