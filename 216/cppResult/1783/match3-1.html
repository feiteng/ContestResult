class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> pre1(n+1), pre2(n+1);
        for(int i = 0; i < n; ++i) {
            if(i & 1) {
                pre1[i+1] = pre1[i] + nums[i];
                pre2[i+1] = pre2[i];
            }
            else {
                pre2[i+1] = pre2[i] + nums[i];
                pre1[i+1] = pre1[i];
            }
        }
        int ans = 0;
        for(int i = 0; i < n ; ++i) {
            if(pre1[n] - pre1[i+1] + pre2[i] == pre2[n] - pre2[i+1] + pre1[i])
                ans ++;
        }
        return ans;
    }
};
