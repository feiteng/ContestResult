class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n + 1), even(n + 1);
        for( int i = 1 ; i <= n ; ++i ){
            if( i & 1 ){
                even[i] = even[i - 1] + nums[i - 1];            
                odd[i] = odd[i - 1];
            }else{
                odd[i] = odd[i - 1] + nums[i - 1];
                even[i] = even[i - 1];
            }
        }
        /*
        cout<<"odd"<<endl;
        for(int v:odd){
            cout<<v<<" ";
        }
        cout<<endl<<"even"<<endl;
        for(int v:even)
            cout<<v<<" ";
        cout<<endl;*/
        
        int res = 0;
        
        
        for( int i = 1 ; i <= n ; ++i ){
            //cout<<even[i - 1] + odd[n] - odd[i]<<" "<<odd[i - 1] + even[n] - even[i]<<endl;
            if( even[i - 1] + odd[n] - odd[i] == odd[i - 1] + even[n] - even[i])
                res++;
        }
        
        return res;
    }
};
