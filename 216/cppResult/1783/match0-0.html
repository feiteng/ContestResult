class Solution {
public:
  int waysToMakeFair(vector<int>& nums) {
    int n = nums.size();
    vector<int> e(n + 1), o(n + 1);
    for (int i = 0; i < n; ++i) {
      e[i + 1] = e[i];
      o[i + 1] = o[i];
      if (i % 2) o[i + 1] += nums[i]; else e[i + 1] += nums[i];
    }
    int ret = 0;
    for (int i = 0; i < n; ++i) {
      if (e[i] + o[n] - o[i + 1] == o[i] + e[n] - e[i + 1]) ++ret;
    }
    return ret;
  }
};
