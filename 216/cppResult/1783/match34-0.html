class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> pre_odd(n + 1, 0), pre_even(n + 1, 0);
        vector<int> post_odd(n + 1, 0), post_even(n + 1, 0);
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0) {
                pre_even[i] = (i > 0 ? pre_even[i - 1] : 0) + nums[i];
                pre_odd[i] = (i > 0 ? pre_odd[i - 1] : 0);
            }
            else {
                pre_odd[i] = (i > 0 ? pre_odd[i - 1] : 0) + nums[i];
                pre_even[i] = (i > 0 ? pre_even[i - 1] : 0);                
            }
        }
        for (int i = n - 1; i >= 0; --i) {
            if (i % 2 == 0) {
                post_even[i] = (i < n - 1 ? post_even[i + 1] : 0) + nums[i];
                post_odd[i] = (i < n - 1 ? post_odd[i + 1] : 0);
            }
            else {
                post_odd[i] = (i < n - 1 ? post_odd[i + 1] : 0) + nums[i];
                post_even[i] = (i < n - 1 ? post_even[i + 1] : 0);                
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (pre_odd[i] + post_even[i] == pre_even[i] + post_odd[i]) {
                ++ans;
            }
        }
        return ans;
    }
};
