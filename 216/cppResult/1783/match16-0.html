class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        vector<int> re(nums.size() + 2, 0), ro(nums.size() + 2, 0);
        vector<int> le(nums.size() + 2, 0), lo(nums.size() + 2, 0);
        for (int i = 0; i < nums.size(); i++) {
            if (i % 2 == 0) {
                le[i+1] = le[i] + nums[i];
                lo[i+1] = lo[i];
            } else {
                lo[i+1] = lo[i] + nums[i];
                le[i+1] = le[i];
            }
        }
        for (int i = nums.size() - 1; i >= 0; i--) {
            if (i % 2 == 0) {
                re[i+1] = re[i+2] + nums[i];
                ro[i+1] = ro[i+2];
            } else {
                ro[i+1] = ro[i+2] + nums[i];
                re[i+1] = re[i+2];
            }
        }
        int ans = 0;
        for (int i = 1; i <= nums.size(); i++) {
            if (le[i-1] + ro[i+1] == lo[i-1] + re[i+1]) {
                ans++;
            }
        }
        return ans;
    }
};
