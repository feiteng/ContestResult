class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n+1), even(n+1);
    
        for(int i=1; i<=n; i++){
            if((i-1)%2){
                odd[i] = odd[i-1]+nums[i-1];
                even[i] = even[i-1];
            }else{
                even[i] = even[i-1]+nums[i-1];
                odd[i] = odd[i-1];
            }
            //cout << odd[i] << " " << even[i] << endl;
        }
        int ans = 0;
        for(int i=0; i<n; i++){
            int v1 = odd[i]+(even[n]-even[i+1]); 
            int v2 = even[i]+(odd[n]-odd[i+1]);
            //cout << v1 << " " << v2 << endl;
            if(v1==v2)  ans++;
        }
        return ans;
    }
};
