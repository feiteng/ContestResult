class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n+1,0);
        vector<int> even(n+1,0);
        int even_sum = 0,odd_sum = 0;
        for (int i = 0 ; i < n; i++) {
            if (i % 2 == 0) {
                even_sum += nums[i];
            } else {
                odd_sum += nums[i];
            }
            even[i+1] = even_sum;
            odd[i+1] = odd_sum;
        }
        // for (int i = 0; i <= n; i++) {
        //     cout << even[i] << " " << odd[i] << endl;
        // }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int a = (even[i] + odd[n] - odd[i+1]);
            int b = (odd[i] + even[n] - even[i+1]);
            // cout << a << " " << b << endl;
            if (a==b) ans++;
        }
        return ans;
    }
};
