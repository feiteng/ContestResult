class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        
        vector<int> preodd(n+1), sufodd(n+1), preeven(n+1), sufeven(n+1);
        preodd[0] = 0, sufodd[n] = 0, preeven[0] = 0, sufeven[n] = 0;

        for (int i = 0; i < n; i++) {
            if (i % 2 == 0) {
                preeven[i+1] = preeven[i] + nums[i];
                preodd[i+1] = preodd[i];
            } else {
                preeven[i+1] = preeven[i];
                preodd[i+1] = preodd[i] + nums[i];
            }
        }
        
        for (int i = n-1; i >= 0; i--) {
            if (i % 2 == 0) {
                sufeven[i] = sufeven[i+1] + nums[i];
                sufodd[i] = sufodd[i+1];
            } else {
                sufeven[i] = sufeven[i+1];
                sufodd[i] = sufodd[i+1] + nums[i];
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (preodd[i] + sufeven[i+1] == preeven[i] + sufodd[i+1]) {
                ans++;
            }
        }
        return ans;
    }
};
