class Solution {
public:
    int waysToMakeFair(vector<int>& a) {
        int n = a.size();
        vector<int> pseven(n), psodd(n), sseven(n), ssodd(n);
        for(int i = 0; i < n; i++) {
            if(i%2 == 0) {
                pseven[i] = (i > 0 ? pseven[i-1] : 0) + a[i];
                psodd[i] = (i > 0 ? psodd[i-1] : 0);
            }
            else {
                psodd[i] = (i > 0 ? psodd[i-1] : 0) + a[i];
                pseven[i] = (i > 0 ? pseven[i-1] : 0);
            }
        }
        for(int i = n-1; i >= 0; i--) {

            if(i%2 == 0) {
                sseven[i] = (i+1 < n ? sseven[i+1] : 0) + a[i];
                ssodd[i] = (i+1 < n ? ssodd[i+1] : 0);
            }
            else {
                ssodd[i] = (i+1 < n ? ssodd[i+1] : 0) + a[i];
                sseven[i] = (i+1 < n ? sseven[i+1] : 0);
            }
        }
        
        // for(int x : pseven) {
        //     cout << x << " ";
        // }
        // cout << endl;
        // for(int x : psodd) {
        //     cout << x << " ";
        // }
        // cout << endl;
        // for(int x : sseven) {
        //     cout << x << " ";
        // }
        // cout << endl;
        // for(int x : ssodd) {
        //     cout << x << " ";
        // }
        // cout << endl;
        
        int ans = 0;
        for(int i = 0; i < n; i++) {
            //try to remove this index
            int even = 0, odd = 0;
            // to the left
            if(i > 0) {
                even += pseven[i-1];
                odd += psodd[i-1];
            }
            if(i < n-1) {
                even += ssodd[i+1];
                odd += sseven[i+1];
            }
            if(even == odd) {
                ans++;
            }
        }
        return ans;
    }
};
