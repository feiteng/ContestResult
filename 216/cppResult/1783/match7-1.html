class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n + 1), even(n + 1);
        for (int i = n - 1; i >= 0; --i) {
            odd[i] = odd[i + 1];
            even[i] = even[i + 1];
            if (i & 1) odd[i] += nums[i];
            else even[i] += nums[i];
        }
        int so = 0, se = 0, ans = 0;

        for (int i = 0; i < n; ++i) {
            if (so + even[i + 1] == se + odd[i + 1])
                ans++;
            if (i & 1) so += nums[i];
            else se += nums[i];
        }
        return ans;
    }
};
