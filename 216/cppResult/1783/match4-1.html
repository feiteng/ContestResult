class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> p0(n+1, 0), p1(n+1, 0);
        for (int i = 0; i < n; ++i) {
            p0[i+1] = p0[i], p1[i+1] = p1[i];
            if (i%2) p0[i+1] += nums[i];
            else p1[i+1] += nums[i];
        }
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int a = p0[i] + p1[n] - p1[i+1];
            int b = p1[i] + p0[n] - p0[i+1];
            if (a == b) ++ans;
        }
        return ans;
    }
};
