class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> oddsPreSum(n + 1, 0);
        vector<int> evenPreSum(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            if (i % 2 == 0) {
                oddsPreSum[i + 1] = oddsPreSum[i];
                evenPreSum[i + 1] = evenPreSum[i] + nums[i];
            } else {
                oddsPreSum[i + 1] = oddsPreSum[i] + nums[i];
                evenPreSum[i + 1] = evenPreSum[i];
            }
        }
        vector<int> oddsPostSum(n + 1, 0);
        vector<int> evenPostSum(n + 1, 0);
        for (int i = n - 1; i >= 0; --i) {
            if (i % 2 == 0) {
                oddsPostSum[i] = oddsPostSum[i + 1];
                evenPostSum[i] = evenPostSum[i + 1] + nums[i];
            } else {
                oddsPostSum[i] = oddsPostSum[i + 1] + nums[i];
                evenPostSum[i] = evenPostSum[i + 1];
            }
        }
        // for (int i = 0; i < n + 1; ++i) {
        //     cout << oddsPreSum[i] << ", " << evenPreSum[i] << ", " << oddsPostSum[i] << ", " << evenPostSum[i] << endl;
        // }
        int res = 0;
        for (int i = 0; i < n; ++i) {
            if (oddsPreSum[i] + evenPostSum[i + 1] == evenPreSum[i] + oddsPostSum[i + 1]) {
                ++res;
            }
        }
        return res;
    }
};
