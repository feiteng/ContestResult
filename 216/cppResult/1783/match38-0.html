class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> le(n+2,0), lo(n+2, 0), re(n+2,0), ro(n+2, 0);
        for(int i = 0; i < n; i++){
            if(i % 2 == 0){
                lo[i] = i-2 >= 0?lo[i-2] + nums[i-2]:0;
                le[i] = i-1 >= 0?le[i-1] + nums[i-1]:0;
            }else{
                lo[i] = i-1 >= 0?lo[i-1] + nums[i-1]:0;
                le[i] = i-2 >= 0?le[i-2] + nums[i-2]:0;
            }
            int j = n-1-i;
            if(j % 2 == 0){
                re[j] = j+2 < n?re[j+2] + nums[j+2]:0;
                ro[j] = j+1 < n?ro[j+1] + nums[j+1]:0;
            }else{
                ro[j] = j+2 < n?ro[j+2] + nums[j+2]:0;
                re[j] = j+1 < n?re[j+1] + nums[j+1]:0;
            }
        }
        // for(int i = 0; i <= n; i++)
        //     cout << le[i] << " ";
        // cout << endl;
        // for(int i = 0; i <= n; i++)
        //     cout << lo[i] << " ";
        // cout << endl;
        // for(int i = 0; i <= n; i++)
        //     cout << re[i] << " ";
        // cout << endl;
        // for(int i = 0; i <= n; i++)
        //     cout << ro[i] << " ";
        // cout << endl;
        int ans = 0;
        for(int i = 0; i < n; i++){
            ans += (lo[i] + ro[i] == le[i] + re[i]); 
        }
        return ans;
    }
};
