class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        
        vector<int> pref_odd(n, 0), pref_even(n, 0), suf_odd(n, 0), suf_even(n, 0);
        int sum_o = 0, sum_e = 0;
        
        for (int i = 0; i < n; i++) {
            pref_odd[i] = sum_o;
            pref_even[i] = sum_e;
            if (i & 1) {
                sum_o += nums[i];
            }
            else {
                sum_e += nums[i];
            }
        }
        
        sum_o = 0;
        sum_e = 0;
        for (int i = n - 1; i >= 0; i--) {
            suf_odd[i] = sum_o;
            suf_even[i] = sum_e;
            if (i & 1) {
                sum_o += nums[i];
            }
            else {
                sum_e += nums[i];
            }
        }
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (pref_odd[i] + suf_even[i] == pref_even[i] + suf_odd[i]) {
                res++;
            }
        }
        
        return res;
    }
};
