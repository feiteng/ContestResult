class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        int n = nums.size();
        vector<int> odd(n+1, 0), even(n+1, 0);
        for (int i = 0; i < n; i++) {
            if (i % 2) {
                // odd
                odd[i+1] = odd[i] + nums[i];
                even[i+1] = even[i];
            } else {
                // even
                odd[i+1] = odd[i];
                even[i+1] = even[i] + nums[i];
            }
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int leftodd = odd[i];
            int lefteven = even[i];
            int rightodd = odd[n] - odd[i+1];
            int righteven = even[n] - even[i+1];
            ans += (leftodd + righteven) == (lefteven + rightodd);
        }
        return ans;
    }
};
