class Solution {
public:
    int waysToMakeFair(vector<int>& nums) {
        
        int even=0,odd=0,count=0;
        for(int i=nums.size()-1;i>=0;i--)
        {
            if(i%2==0) even+=nums[i];
            else odd+=nums[i];
        }
        int leftodd=0,lefteven=0;
        int nodd,neven;
        for(int i=0;i<nums.size();i++)
        {
            if(i%2==0)
            {
                even=even-nums[i];
                if(lefteven+odd==leftodd+even)
                     count++;
                lefteven+=nums[i];
            }
            else
            {
                odd=odd-nums[i];
                if(leftodd+even==lefteven+odd)
                    count++;
                leftodd+=nums[i];
            }
            
        } 
       return count;
    }
};
