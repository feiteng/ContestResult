class Solution {
public:
    string getSmallestString(int n, int k) {
        
        string result = ""; 
  
    for(int i = 0; i < n; i++) 
        result += 'a'; 
  
    // Iteration from the last position 
    // in the array 
    for (int i = n - 1; i >= 0; i--) 
    { 
        k -= i; 
  
        // If k is a positive integer 
        if (k >= 0) 
        { 
  
            // 'z' needs to be inserted 
            if (k >= 26) 
            { 
                result[i] = 'z'; 
                k -= 26; 
            } 
  
            // Add the required character 
            else
            { 
                char c= (char)(k + 97 - 1); 
                result[i] = c; 
                k -= result[i] - 'a' + 1; 
            } 
        } 
  
        else
            break; 
  
        k += i; 
    } 
    return result; 
        
    }
};
