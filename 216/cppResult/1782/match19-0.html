class Solution {
public:
    string getSmallestString(int n, int k) {
        string res = "";
        
        for(int i = 0; i < n; ++i) {
            int num = max(1, k - (n - i - 1) * 26);
            res += 'a' + num - 1;
            k -= num;
        }
        
        return res;
    }
};
