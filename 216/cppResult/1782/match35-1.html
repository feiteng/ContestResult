class Solution {
public:
    string getSmallestString(int n, int k) {
        // add as much as a possible then smth 
        // add smallest you can add now ??? 
        // what does it mean 
        // I can add 1 if I can somehow make it up later
        string ans = "";
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= 26; j++){

                if(k - j <= 26 * (n - i)){
                    ans += char(j - 1 + 'a');
                    k -= j;
                    break;
                }
            }
        }
        return ans;
    }
};
