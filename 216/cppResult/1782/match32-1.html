class Solution {
public:
    string getSmallestString(int n, int k) {
        string result(n, 'a');
        k -= n;
        int i = n - 1;
        while (k) {
            int diff = min(k, 25);
            result[i--] += diff;
            k -= diff;
        }
        return result;
    }
};
