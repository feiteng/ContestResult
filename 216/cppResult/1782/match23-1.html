class Solution {
public:
    string getSmallestString(int n, int k) {
        string str;
        for (int i = n; i > 0; --i) {
            int b = k - (i - 1) * 26;
            int d = max(b, 1);
            str += (d - 1 + 'a');
            k -= d;
        }
        return str;
    }
};
