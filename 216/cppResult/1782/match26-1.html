class Solution {
public:
    string getSmallestString(int n, int k) {
        string s;
        for (int i = 0; i < n; i++) {
            for (int j = 1; j <= 26; j++) {
                if (j + (n-i-1)*26 >= k) {
                    s.push_back('a'+j-1);
                    k -= j;
                    break;
                }
            }
        }
        return s;
        
    }
};
