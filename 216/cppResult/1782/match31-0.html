class Solution {
public:
    string getSmallestString(int n, int k) {
        string ans(n, 'a');
        int v = k - n;
        for (int i = n-1; i >= 0; --i) {
            if (v == 0) break;
            if (v >= 25) {
                ans[i] = 'z';
                v -= 25;
            } else {
                ans[i] = 'a' + v;
                break;
            }
        }
        return ans;
    }
};
