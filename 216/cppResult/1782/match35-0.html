class Solution {
public:
    string getSmallestString(int n, int k) {
        string ans="";
        for ( int i=0; i<n; i++ ) {
            for ( int j=1; j<=26; j++ ) {
                int nk = k-j;
                if ( nk>=n-i-1&&nk<=(n-i-1)*26 ) {
                    ans+=(char)('a'+j-1);
                    k=nk;
                    break;
                }
            }   
        }
        return ans;
    }
};
