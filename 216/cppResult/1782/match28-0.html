class Solution {
public:
    string getSmallestString(int n, int k) {
        string res = "";
        for (int i = 0; i < n; i++) {
            int cur = max(k - (n - i - 1) * 26, 1);
            //cout<<cur<<endl;
            res.push_back('a' + cur - 1);
            k-=cur;
        }
        return res;
        
    }
};
