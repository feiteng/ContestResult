class Solution {
public:
    string getSmallestString(int n, int k) {
        string res = "";
        for (int i = 0; i < n; i++) {
            int s_ch = max(k - ((n - 1 - i) * 26), 1);
            res += 'a' + s_ch - 1;
            k -= s_ch;
        }
        return res;
    }
};
