class Solution {
public:
    string getSmallestString(int n, int k) {
        string res;
        /*if (n == 1) {
            res.push_back('a' + k - 1);
            return res;
        }
        for (int i = 1; i <= 26; ++i) {
            if (n - 1 <= k-i && k - i <= 26 * (n-1)) {
                res.push_back('a' + i - 1);
                return res + getSmallestString(n-1, k-i);
            }
        }*/
        for (int i = 0; i < n; ++i) {
            for (int j = 1; j <= 26; ++j) {
                if (n - i - 1 <= k - j && k - j <= 26 * (n - i - 1)) {
                    res.push_back('a' + j - 1);
                    k -= j;
                    break;
                }
            }
        }
        
        return res;
    }
};
