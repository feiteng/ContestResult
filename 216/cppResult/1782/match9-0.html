class Solution {
public:
    string getSmallestString(int n, int k) {
        string ret(n, 'a');
        k -= n;
        
        int idx = ret.size() - 1;
        while (k) {
            int v = min(25, k);
            ret[idx] += v;
            k -= v;
            --idx;
        }
        return ret;
    }
};
