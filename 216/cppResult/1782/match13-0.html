class Solution {
public:
    string getSmallestString(int n, int k) {
        
        string ans = ""; 
  
    for(int i = 0; i < n; i++) 
        ans += 'a'; 
    for (int i = n - 1; i >= 0; i--) 
    { 
        k -= i; 
        if (k >= 0) 
        { 
  
            if (k >= 26) 
            { 
                ans[i] = 'z'; 
                k -= 26; 
            } 
   
            else
            { 
                char c= (char)(k + 97 - 1); 
                ans[i] = c; 
                k -= ans[i] - 'a' + 1; 
            } 
        } 
  
        else
            break; 
  
        k += i; 
    } 
    return ans; 
        
        
        
    }
};
