class Solution {
public:
    string getSmallestString(int n, int k) {
        string s; s.resize(n);
        
        for(int i=0; i<n; i++)
        {
            int x=max(1, k-26*(n-1-i));
            s[i]='a'+(x-1);
            
            //dbg(x);
            
            k-=x;
        }
        
        return s;
    }
};
