class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [] (const vector<int>& a, const vector<int>& b) { return (a[1] - a[0] > b[1] - b[0]); } );
        
        // cout << "New test" << endl;
        // for(auto pair: tasks) {
        //     cout << pair[0] << ", " << pair[1] << endl;
        // }
        
        int curr = 0;
        int total = 0;
        for(auto t: tasks) {
            if(curr < t[1]) {
                total += (t[1] - curr);
                curr = t[1];
            }
            
            curr -= t[0];
        }
        
        return total;
    }
};
