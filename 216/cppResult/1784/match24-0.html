bool mygreater(vector<int>& a, vector<int>& b) {
    return (a[1] - a[0]) > (b[1] - b[0]);
}

class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), mygreater);
        int ans = tasks[0][1];
        int curr = ans;
        for (auto& task : tasks) {
            if (curr < task[1]) {
                ans += (task[1] - curr);
                curr = task[1];
            }
            curr -= task[0];
        }
        if (curr < 0)
            ans -= curr;
        return ans;
    }
};
