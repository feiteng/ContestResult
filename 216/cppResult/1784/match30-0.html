class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(),tasks.end(),[&](vector<int> &a,vector<int> &b)
             {
                 if(a[1]-a[0]<b[1]-b[0])
                 {
                     return true;
                 }
                 else if(a[1]-a[0]>b[1]-b[0])
                 {
                     return false;
                 }
                 else if(a[0]<b[0])
                 {
                     return true;
                 }
                 return false;
             });
        for(int i=0;i<tasks.size();i++)
        {
            cout<<tasks[i][0]<<","<<tasks[i][1]<<endl;
        }
        
        int res=0;
        for(int i=0;i<tasks.size();i++)
        {
            res=max(res+tasks[i][0],tasks[i][1]);
        }
        return res;
    }
};
