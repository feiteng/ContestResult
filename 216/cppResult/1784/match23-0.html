class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](vector<int> &a, vector<int> &b){
            if(a[1] - a[0] == b[1] -b[0])
                return a[1]> b[1];
            return (a[1] - a[0]) > (b[1] - b[0]);
            
        });
        int n = tasks.size();
        
        int ans =0;
        for(int i=0;i<n;++i){
            ans += tasks[i][0];
        }
        int temp = ans;
        for(int i=0;i<n;++i){
            if(tasks[i][1] <= temp){
                temp-=tasks[i][0];
            }
            else{
                ans+= tasks[i][1] - temp;
                temp = tasks[i][1] - tasks[i][0];
            }
        }
        return ans;
        
    }
};
