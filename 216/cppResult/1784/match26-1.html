class Solution {

    int d(vector<int> &A) {
        return A[1] - A[0];
    }
public:
    int minimumEffort(vector<vector<int>>& A) {
        int N = A.size();
        sort(begin(A), end(A), [&](vector<int> &a, vector<int> &b) { return d(a) > d(b); });

        int ans = 0, cur = 0;
        for (int i = 0; i < N; ++i) {
            if (cur < A[i][1]) {
                ans += A[i][1] - cur;
                cur = A[i][1];
            }
            cur -= A[i][0];
        }
        return ans;
    }
};
