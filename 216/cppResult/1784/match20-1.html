class Solution {
public:
    int minimumEffort(vector<vector<int>>& t) {
        sort(begin(t), end(t), [](vector<int> &t1, vector<int> &t2) { return t1[1] - t1[0] > t2[1] - t2[0]; });
        int energy = 0, res = 0;

        for (auto & task : t) {
            if (energy < task[1]) {
                res += task[1] - energy;
                energy = task[1];
            }
            energy -= task[0];
        }
        return res;
    }
};
