class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(begin(tasks), end(tasks), [](const vector<int>& a, const vector<int>& b) {
            return a[1] - a[0] > b[1] - b[0];
        });
        int ans = 0;
        int curr = 0;
        for (auto& t: tasks) {
            int cost = t[0];
            int requirement = t[1];
            int minCurr = max(cost, requirement);
            if (curr < minCurr) {
                int fill = minCurr - curr;
                curr += fill;
                ans += fill;
            }
            assert(curr >= minCurr);
            curr -= cost;
        }
        return ans;
    }
};
