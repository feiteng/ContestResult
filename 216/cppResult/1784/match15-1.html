bool compare(vector<int> &a, vector<int>&b) {
    return ((a[1]-a[0] > b[1]-b[0]) || ((a[1]-a[0] == b[1]-b[0]) && (a[1] < b[1])));
}

class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        int n = tasks.size();
        sort(tasks.begin(), tasks.end(), compare);
        int answer = 0;
        int energy = 0;
        for (int i=0; i<n; i++) {
            // cout << tasks[i][0] << " " << tasks[i][1] << endl;
            if (energy < tasks[i][1]) {
                answer += (tasks[i][1]-energy);
                energy = tasks[i][1];
            }
            energy = energy - tasks[i][0];
        }
        return answer;
    }
};
