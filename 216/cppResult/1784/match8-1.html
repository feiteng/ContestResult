class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](vector<int>& p1, vector<int>& p2){
            return p1[1] - p1[0] == p2[1] - p2[0] ? p1[1] < p2[1] : p1[1] - p1[0] < p2[1] - p2[0];
        });
        int ans = 0;
        for(auto iter : tasks){
            ans = max(iter[0] + ans, iter[1]);
        }
        return ans;
    }
};
