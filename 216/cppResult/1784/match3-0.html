class Solution {
    static bool cmp(vector<int>& a, vector<int>& b){
        return (a[1] - a[0]) > (b[1] - b[0]);
    }
    
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), cmp);
        int result = 0;
        int rest = 0;
        for(int i = 0; i < tasks.size(); i++){
            if(rest < tasks[i][1]) {
                result += tasks[i][1] - rest;
                rest = tasks[i][1];
            }
            rest -= tasks[i][0];
        }
        return result;
    }
};
