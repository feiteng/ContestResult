class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [&](const vector<int>& a, const vector<int>& b){
            return (a[1] - a[0]) > (b[1] - b[0]);
        });
        int ans = tasks[0][1];
        int rem = ans - tasks[0][0];
        int n = tasks.size();
        for(int i = 1; i < n; i ++){
            if(rem < tasks[i][1]){
                ans += tasks[i][1] - rem;
                rem = tasks[i][1];
            }
            rem -= tasks[i][0];
        }
        return ans;
    }
};
