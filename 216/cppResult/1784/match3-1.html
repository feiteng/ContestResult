class Solution {
    
    static bool compareTasks1(vector<int>& a, vector<int>& b){
        return (a[1]-a[0])>(b[1]-b[0]);
    }
    
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), compareTasks1);
        int result = tasks[0][1];
        int left = result - tasks[0][0];
        for(int i=1; i<tasks.size(); i++){
            if(left<tasks[i][1]){
                result += tasks[i][1]-left;
                left = tasks[i][1];
            }
            left = left - tasks[i][0];
        }
        return result;
    }
};
