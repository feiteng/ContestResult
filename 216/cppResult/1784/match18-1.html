class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](auto &a, auto &b){
            // if(a[1] == b[1]) return a[0] < b[0];
            return a[1] -a[0] > b[1] - b[0];
        });
        int act = 0, mn = 0, ans = 0;

        for(auto &t : tasks) {
            if(act < t[1]) {
                ans += t[1] - act;
                act = t[1];
            }
            act -= t[0];
        }
        return ans;
    }
};
