class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        // sort(tasks.begin(), tasks.end(), [](auto &v1, auto &v2){
        //     return v1[1]>v2[1] || (v1[1]==v2[1] && v1[0]>v2[0]);});
        sort(tasks.begin(), tasks.end(), [](auto &v1, auto &v2){
            return v1[1]-v1[0]>v2[1]-v2[0] || (v1[1]-v1[0]==v2[1]-v2[0] && v1[1]>v2[1]);});
        int res=0, remain=0;
        for(auto &t:tasks){
            if(remain<t[1]){
                res += t[1]-remain;
                remain = t[1];
            }
            remain -= t[0];
        }
        return res;
    }
};
