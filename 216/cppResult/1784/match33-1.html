class Solution {
public:
  int minimumEffort(vector<vector<int>>& tasks) {
    sort(tasks.begin(), tasks.end(), [](vector<int> &a, vector<int> &b) {
      if (a[1] - a[0] == b[1] - b[0]) {
        return a[1] < b[1];
      } else {
        return  a[1] - a[0] < b[1] - b[0];
      }

    });
    
    int ans = tasks[0][1];
    for (int i = 1; i < tasks.size(); ++i) {
      ans = max(ans + tasks[i][0], tasks[i][1]);
    }
    return ans;
  }
};
