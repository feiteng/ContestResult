class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(),tasks.end(),[](vector<int> &x,vector<int> &y){return x[1]-x[0]<y[1]-y[0];});
        int n=tasks.size();
        int ret=0,res=0;

        for(int i=n-1;i>=0;i--){
            if(res<tasks[i][1]){
                ret+=tasks[i][1]-res;
                res+=tasks[i][1]-res;
            }
            res-=tasks[i][0];
        }
        return ret;
    }
};
