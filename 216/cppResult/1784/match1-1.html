class Solution {
public:
    int minimumEffort(vector<vector<int>>& arr) {
        //start with min larger ???? what if actaul is really large too ???
        // start with minimum largest and do bier
        sort(begin(arr), end(arr),[](vector<int>& a, vector<int>& b){
            return a[1] - a[0] > b[1] - b[0];
        });
        int ans = 0;
        int ac = 0;
        for(int i = 0; i < arr.size(); i++){
            if(arr[i][1] > ac){
                ans += arr[i][1] - ac;
                ac += arr[i][1] - ac;
            }
            ac -= arr[i][0];
        }
        return ans;
    }
};
