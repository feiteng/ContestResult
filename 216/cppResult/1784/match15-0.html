bool cmp(vector<int> & a,vector<int> &b)
{
   
    return a[1]-a[0]>b[1]-b[0];
}
class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
       sort(tasks.begin(),tasks.end(),cmp);
        int ans=0;
        int now=0;
        int len=tasks.size();
        for(int i=0;i<len;i++)
        {
            if(now<tasks[i][1])
            {
                ans+=tasks[i][1]-now;
                now=tasks[i][1]-tasks[i][0];
            }
            else 
            {
                now-=tasks[i][0];
            }
          //  cout<<tasks[i][0]<<" "<<tasks[i][1]<<" "<<ans<<" "<<now<<endl;
        }
        return ans;
        
    }
};
