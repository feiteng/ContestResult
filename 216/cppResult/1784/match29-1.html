class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](const auto& a, const auto& b) {
            return a[1] - a[0] > b[1] - b[0];
        });
        int ans = 0;
        int cur = 0;
        for (const auto& elm : tasks) {

            if (cur < elm[1]) {
                int diff = elm[1] - cur;
                ans += diff;
                cur += diff;
            }
            cur -= elm[0];
        }
        return ans;
    }
};
