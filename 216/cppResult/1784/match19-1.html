class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(),tasks.end(),[&](const vector<int>&a,const vector<int>&b){
            return a[1]-a[0]>b[1]-b[0];
        });

        int need=0;
        int ans=0;
        for(int i=0;i<tasks.size();i++){
            if(need<tasks[i][1]){
                ans+=tasks[i][1]-need;
                need=tasks[i][1];
            }
            need-=tasks[i][0];
        }
        return ans;
    }
};
