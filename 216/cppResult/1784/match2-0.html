class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](vector<int>& v1, vector<int>& v2){
           return v1[1] - v1[0] > v2[1] - v2[0] || (v1[1] - v1[0] == v2[1] - v2[0] && v1[1] > v2[1]); 
        });
        int ans = 0;
        int shengyu = 0;
        for(int i = 0; i < tasks.size(); i++){
            if(shengyu < tasks[i][1]){
                ans += tasks[i][1] - shengyu;
                shengyu = tasks[i][1];
            }
            shengyu -= tasks[i][0];
        }
        return ans;
    }
};
