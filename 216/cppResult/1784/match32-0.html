const int maxn=1e5+5;
struct node{
    int a;
    int b;
};
node t[maxn];
bool cmp(node n1,node n2){
    // cout<<"abc"<<endl;
    // cout<<n1.a<<" "<<n1.b<<" "<<n2.a<<" "<<n2.b<<endl;
    if(max(n1.b,n1.a+n2.b)<max(n2.b,n1.b+n2.a)) return true;
    return false;
}
class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        int n=tasks.size();
        for(int i=0;i<n;i++){
            t[i].a=tasks[i][0];
            t[i].b=tasks[i][1];
        }
        sort(t,t+n,cmp);
        int ret=0;
        for(int i=n-1;i>=0;i--){
            ret=max(ret+t[i].a,t[i].b);
        }
        return ret;
    }
};
