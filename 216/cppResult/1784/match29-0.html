class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](const auto &v1, const auto &v2) {
            return v1[1] - v1[0] > v2[1] - v2[0];
        });
        
        int e = tasks[0][1];
        int c = e;
        int k = 0;
        
        for (const auto &t : tasks) {
            int s = t[0];
            int m = t[1];
            if (c < m) {
                int x = m - c;
                k += x;
                c = m;
            }
            c -= s;
        }
                        
        return e + k;
    }
};
