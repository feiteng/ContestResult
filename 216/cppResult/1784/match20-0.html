class Solution {
public:
    // [收录] 贪婪算法，先考虑运行完后剩下point最多的task
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(),[](vector<int>& a, vector<int>& b) {
            return a[1] - a[0] > b[1] - b[0] || a[1] - a[0] == b[1] - b[0] && a[1] < b[1];
        });
        
        int res(0), energy(0);
        for(auto& t: tasks) {
            if (energy < t[1]) {

                res += t[1] - energy;
                energy = t[1];
            }
            energy -= t[0];
        }
        return res;
    }
};
