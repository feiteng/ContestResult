class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(),tasks.end(),[](vector<int> &a, vector<int> &b){
            if(a[1]-a[0] == b[1]-b[0]) return a[1]>b[1];
            return a[1]-a[0]>b[1]-b[0];
        });
        int curr = 0, res = 0;

        for(int i=0; i<tasks.size(); i++){
            if(curr >= tasks[i][1]){
                curr-=tasks[i][0];
            }else{
                res+=tasks[i][1]-curr;
                curr = tasks[i][1]-tasks[i][0];
            }
        }
        return res;
    }
};
