typedef long long int ll;

bool compare(vector<int>&a,vector<int>&b)
{
    //if(a[1]==b[1]){return a[0]>a[0];}
    return (a[1]-a[0])>(b[1]-b[0]);
}
class Solution {
    
    
public:
    
    bool check(vector<vector<int>>&v,int val)
    {
        int n=v.size();
        int left=val;
        for(int i=0;i<n;i++)
        {
            if(left<max(v[i][0],v[i][1])){return false;}
            left-=v[i][0];
            
        }
        return true;
    }
    int minimumEffort(vector<vector<int>>& v) {
        
        int sum=0;
        for(int i=0;i<v.size();i++)
        {sum+=v[i][0];}
        sort(v.begin(),v.end(),compare);
        ll l=sum,r=INT_MAX-1;
        int ans=INT_MAX;
        while(l<=r)
        {
            long long int mid=(ll)(l+r)/2;
            
            if(check(v,mid))
            {
                ans=mid;
                r=mid-1;
                
            }
            else{l=mid+1;}
        }
        
        return ans;
    }
};
