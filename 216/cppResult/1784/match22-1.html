#include <algorithm>

bool cmp(vector<int>& va, vector<int>& vb) {
    return va[1]-va[0] > vb[1]-vb[0];
}
class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), cmp);
        int ans = 0;
        int cur = 0;

        for (auto & vc : tasks) {
            if (cur < vc[1]) {
                ans += vc[1]-cur;
                cur = vc[1] - vc[0];
            } else {
                cur -= vc[0];
            }
        }
        return ans;
    }
};
