class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [](const vector<int> &x, const vector<int> &y) {
            return (x[1] - x[0]) > (y[1] - y[0]);
        });
        int ans = 0;
        int now = 0;
        int n = tasks.size();
        for (int i = 0; i < n; i++) {
            if (now < tasks[i][1]) {
                ans += tasks[i][1] - now;
                now = tasks[i][1];
            }
            now -= tasks[i][0];
            //cout << ans << " " << now << endl;
        }
        return ans;
    }
};
