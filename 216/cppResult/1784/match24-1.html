bool cmp(vector<int>& a, vector<int>&b) {
    return a[1] - a[0] > b[1] - b[0];
}

class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), cmp);
        
        int ans = 0;
        int cur = 0;
        for (auto& t: tasks) {
            if (t[1] > cur) {
                ans += t[1] - cur;
                cur = t[1];
            }
            cur -= t[0];
        }
        
        return ans;
    }
};
