class Solution {
public:
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), [&](vector<int>& v1, vector<int>& v2){
            return v1[1] - v1[0] < v2[1] - v2[0];
        });
        int ans = 0;
        for(int i = 0; i < tasks.size(); ++i){
            ans = max(ans + tasks[i][0], tasks[i][1]);
        }
        return ans;
    }
};
