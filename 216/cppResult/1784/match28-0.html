bool compareTask(const vector<int>& a, const vector<int>& b){
    return a[1] - a[0] > b[1] - b[0];
}

class Solution {
public:
    
    int minimumEffort(vector<vector<int>>& tasks) {
        sort(tasks.begin(), tasks.end(), compareTask);
        
        int begin_energy = tasks[0][1], temp_energy = tasks[0][1];
        int i;
        for (i = 0; i < tasks.size(); i++){
            if (temp_energy < tasks[i][1]){
                begin_energy += tasks[i][1] - temp_energy;
                temp_energy = tasks[i][1];
            }
            temp_energy -= tasks[i][0];
        }
        return begin_energy;
    }
};
