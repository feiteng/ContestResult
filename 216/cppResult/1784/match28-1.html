bool cmp(vector<int>& a, vector<int>& b)
{
    return a[1] - a[0] > b[1] - b[0];
    /*
    if(a[1] != b[1])
        return a[1] > b[1];
    else
        return a[0] < b[0];*/
}


class Solution 
{
public:
    int minimumEffort(vector<vector<int>>& tasks)
    {
        sort(tasks.begin(), tasks.end(), cmp);
        int n = tasks.size();
        
        /*
        vector<int> maxv(n);
        maxv[0] = tasks[0][1];
        for(int i = 1; i < n; i++)
        {
            maxv[i] = max(maxv[i - 1], tasks[i][1]);
        }
        */
       // for(int i : maxv)
           // cout << i << endl;
        
        int ret = 0, val = 0;

        for(int i = 0; i < n; i++)
        {
            if(val < tasks[i][1])
            {
                ret += tasks[i][1] - val;
                val = tasks[i][1];
            }
            val -= tasks[i][0];
        }
        return ret;
    }
};
