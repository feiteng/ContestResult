class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ans = ["a"  for i in range(n)]
        i 
= n-1
        k -= n
        while(k > 0):
            if(k >= 25):
                k -= 25
                ans[i] = 'z'
            else:
                ans[i] = chr(ord('a') + k)
                k = 0
            i -= 1
        return "".join(ans)
