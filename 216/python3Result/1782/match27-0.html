class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        arr = [1 for _ in range(n)]
        k -= n
        i = n - 1
        while k > 0:
            arr[i] += min(25, k)
            k -= min(25, k)
            i -= 1
        return ''.join([chr(ord('a') - 1 + c) for c in arr])
