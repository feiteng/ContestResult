class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        answer 
= ''
        for i in range(n):
            for j in range(26):
                if 26*(n-i-1) + (j+1) >= k:
                    answer += chr(ord('a')+j)
                    k -= j+1
                    break
        return answer
