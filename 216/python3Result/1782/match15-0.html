class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        res = ''
        i = 0
        while i < n:
            # print(i, k)
            for j in range(1, 27):
                # print('-', j, (n-i-1)*26 + j, k)
                if (n-i-1)*26 + j>=k:
                    res += chr(j-1+ord('a'))
                    # print(res)
                    k -= j
                    break
            i += 1
        return res
