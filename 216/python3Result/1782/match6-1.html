class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ans = [1] * n
        k -= n
        i = n - 1
        while k > 0 and i >= 0:
            delta = min(k, 25)
            ans[i] += delta
            k -= delta
            i -= 1
        
        return ''.join([chr(x - 1 + 97) for x in ans])
