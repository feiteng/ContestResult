class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ans = [1] * n
        cur = n
        p = -1
        while cur < k:
            step = min(25, k - cur)
            ans[p] += step
            cur += step
            p -= 1
        return ''.join([chr(ord('a') - 1 + c) for c in ans])
