class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        res = ['a'] * n
        k -= 1 * n
        j = n-1
        while k > 0 and j >= 0:
            if k >= 25:
                k -= 25
                res[j] = 'z'
            else:
                res[j] = chr(ord('a') + k)
                k = 0
            j -= 1
        return ''.join(res)
