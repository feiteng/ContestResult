class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        base = ord('a')
        ans = [base for _ in range(n)]
        num = k - n
        i = n - 1
        while num > 0 and i >= 0:
            if num >= 25:
                ans[i] += 25
                num -= 25
            else:
                ans[i] += num
                num = 0
            i -= 1
        return ''.join(chr(x) for x in ans)
