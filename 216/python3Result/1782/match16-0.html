class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ans = [1]*n
        k-=n
        idx = -1
        while k:
            if k>25:
                ans[idx] += 25
                k -= 25
            else:
                ans[idx] += k
                k = 0
            idx -=1
        #print(ans)
        return ''.join([chr(ord('a')+i-1) for i in ans])
