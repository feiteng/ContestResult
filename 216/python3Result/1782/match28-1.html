class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        arr = ['a' for _ in range(n)]
        rem 
= k - n
        
        i = n-1
        while rem > 0:
            if rem >= 25:
                arr[i] = 'z'
                rem -= 25
                i -= 1
            else:
                arr[i] = chr(97 + rem)
                rem = 0
                
        return ''.join(arr)
