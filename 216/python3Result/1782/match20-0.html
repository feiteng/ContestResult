class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        # length n string w numeric value k
        s = ["a"]*n
        k -= n
        
        i = 0
        while k > 0:
            val = min(k,25)
            s[i] = chr(ord('a')+val)
            i += 1
            k -= val
            
        return "".join(reversed(s))
