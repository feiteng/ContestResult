class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        chars = ['z' for _ in range(n)]
        score 
= 26 * n
        for i in range(n):
            if score - 25 >= k:
                chars[i] = 'a'
                score -= 25
            else:
                chars[i] = chr(ord('z') - (score - k))
                break
        
        return "".join(chars)
