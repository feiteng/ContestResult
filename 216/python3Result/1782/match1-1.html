class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ans=""
        while n>0:
            if 26*(n-1)+1>=k:
                ans+="a"
                k-=1
            else:
                ans+=chr(k-26*(n-1)+96)
                k-=k-26*(n-1)
            n-=1
        return ans
