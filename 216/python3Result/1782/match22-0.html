class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        suma=0
        s=''
        for i in range(n):
            for j in range(1,27):
                if suma+j+26*(n-(i+1))>=k:
                    s+=chr(ord('a')-1+j)
                    suma+=j
                    break
        return s
