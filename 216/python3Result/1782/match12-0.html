class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        val = n
        res = ["a"] * n
        i = n - 1
        while val < k:
            if k - val >= 25:
                res[i] = "z"
            else:
                res[i] = chr(ord("a") + k - val)
            val += 25
            i -= 1
        return "".join(res)
