class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        t = ""
        index = 0
        
        while index < n:
            for x in range(1, 27):
                if x + (n - index - 1) * 26 >= k:
                    t += chr(ord('a') + x-1)
                    k -= x
                    break
                
            index += 1
        
        return t
