class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        vals = [1] * n
        k -= n
        
        for i in range(n-1, -1, -1):
            if k == 0:
                break
            sub = min(k, 25)
            k -= sub
            vals[i] += sub
        
        return "".join([chr(x+96) for x in vals])
