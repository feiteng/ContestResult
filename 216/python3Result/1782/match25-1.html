class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        res 
= ['a'] * n
        rem = k
        for i in range(n - 1, -1, -1):
            if rem >= i + 1 + 26:
                res[i] = 'z'
                rem -= 26
            else:
                res[i] = chr(ord('a') + rem - (i + 1))
                break
        return "".join(res)
