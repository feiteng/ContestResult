class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        res = [1] * n
        k -= n
        for i in range(1,n+1):
            v = min(25,k)
            res[-i] += v
            k -= v
            if not k:
                break
        return "".join([chr(i+96)for i in res])
