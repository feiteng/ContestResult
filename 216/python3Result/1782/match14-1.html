class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ans = []
        for i in range(n):
            v = k - (n - i - 1) * 26
            if v <= 1:
                ans.append("a")
                k -= 1
            else:
                ans.append(chr(ord("a") + v - 1))
                k -= v
        return "".join(ans)
