class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        out = ['a'] * n
        k -= n
        i = -1
        while k > 0:
            if k > 25:
                out[i] = 'z'
                i -= 1
                k -= 25
            else:
                out[i] = chr(k + 1 + 96)
                k -= k
        
        return ''.join(out)
