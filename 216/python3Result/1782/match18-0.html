class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        res = [1] * n
        k -= n
        for i in range(n-1, -1, -1):
            val = min(25, k)
            res[i] += val
            k -= val
            if k == 0:
                break

        return ''.join([chr(ord('a') + x - 1) for x in res])
