class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        res = ''
        while n:
            if (n - 1) * 26 >= k - 1:
                res += 'a'
                k -= 1
            else:
                res += chr(96 + k - (n - 1) * 26)
                k = (n - 1) * 26
            n -= 1
        return res
