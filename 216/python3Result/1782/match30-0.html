class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        d = {}
        for i in range(26):
            d[chr(ord('a') + i)] = i + 1
        if 26 * n == k:
            return 'z' * n
        res = ['a'] * n
        k = k - 1 * n
        for i in range(n - 1, -1, -1):
            if k >= 25:
                k -= 25
                res[i] = 'z'
            else:
                res[i] = chr(ord('a') + k)
                break
        return ''.join(res)
