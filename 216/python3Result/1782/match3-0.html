class Solution:
    def getSmallestString(self, n: int, k: int) -> str:
        ret = [1] * n
        k -= n
        i = n - 1
        while i >= 0 and k:
            t = min(k, 25)
            ret[i] += t
            k -= t
            i -= 1
        return ''.join(chr(ord('a') - 1 + i) for i in ret)
