class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda x: -(x[1] - x[0]))
        ans = 0
        cur = 0
        #print(tasks)
        for cost, mmin in tasks:
            if mmin > cur:
                ans += (mmin - cur)
                cur = mmin
            cur -= cost
            
        return ans
