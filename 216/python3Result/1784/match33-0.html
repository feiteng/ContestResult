class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda x: (x[0] - x[1], -x[1]))
        l = 0
        ret = 0
        for cost, start in tasks:
            if start > l:
                ret += start - l
                l = start
            l -= cost
        return ret
