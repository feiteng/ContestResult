class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        
        tasks = sorted(tasks, key=lambda x:(x[0] - x[1], x[0], -x[1]))
        
        out = 0
        now_v = 0
        for actual, mini in tasks:
            if now_v < mini:
                out += (mini - now_v)
                now_v = mini
            now_v -= actual
        
        return out
