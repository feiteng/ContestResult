class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda student: (student[0] - student[1], -student[1]))
        cur = 0
        tot = 0
        for actual, minimum in tasks:
            if minimum > cur:
                tot += minimum - cur
                cur = minimum
            cur -= actual
        return tot
