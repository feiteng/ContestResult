class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda x: x[0]-x[1])
        cur = ans = 0
        # print(tasks)
        for cost, m in tasks:
            # print(cost, m, cur, ans)
            if cur < m:
                ans += m - cur
                cur = m
            cur -= cost
            
        return ans
