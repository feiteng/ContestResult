class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda x: -(x[1]-x[0]))
        
        total_energy = 0
        curr_energy = 0
        for actual, minim in tasks:
            if curr_energy < minim:
                total_energy += minim - curr_energy
                curr_energy = minim
            curr_energy -= actual
                
        return total_energy
