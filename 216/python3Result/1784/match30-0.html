class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda x: x[1]-x[0], reverse=True)
        result = 0
        rem = 0
        for x in tasks:
            if rem < x[1]:
                result += x[1] - rem
                rem = x[1]
            rem -= x[0]
        return result
