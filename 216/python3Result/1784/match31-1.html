class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key = lambda x: (x[0]-x[1], -x[1], x[0]))
        print(tasks)
        currentEnergy 
= 0
        ans = 0
        for a, r in tasks:
            if currentEnergy < r:
                ans += (r-currentEnergy)
                currentEnergy = r
            currentEnergy -= a
        return ans
