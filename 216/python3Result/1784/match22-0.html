class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        '''
        task 1: [9, 12]
        
        task 2: [9, 10]
                [[1,2],[2,4],[4,8]]
                
        res  0     
        tmp  1             8     
        [[1,3],[2,4],[10,11],[10,12],[8,9]]
          2.     2.    1       2.      1
         [8,9]   [10, 11],   [1,3]   [2,4]   [10,12]
          
        0  9        19         20     22       32 
        '''
        tasks.sort(key=lambda x:(x[1]-x[0], x[1]))
        # tasks = [[0,0]] + tasks
        res = 0
        for i in range(len(tasks)):
            res = max(res + tasks[i][0], tasks[i][1])
        return  res
