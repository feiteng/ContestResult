class Solution:
    def minimumEffort(self, tasks: List[List[int]]) -> int:
        tasks.sort(key=lambda t: [t[0]-t[1], -t[0]])
        ans = curr = 0
        for t in tasks:
            if curr < t[1]:
                ans += t[1] - curr
                curr = t[1]
            curr -= t[0]
        return ans
