class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd, even 
= [0], [0]
        for i, num in enumerate(nums):
            if i % 2 == 0:
                even.append(even[-1] + num)
                odd.append(odd[-1])
            else:
                even.append(even[-1])
                odd.append(odd[-1] + num)
        odd, even = odd[1:], even[1:]
        ans 
= 0
        for i in range(len(nums)):
            oddSum = 0 if i == 0 else odd[i - 1]
            evenSum = 0 if i == 0 else even[i - 1]
            oddSum = oddSum + even[-1] - even[i]
            evenSum = evenSum + odd[-1] - odd[i]
            if oddSum == evenSum:
                ans += 1
        return ans
