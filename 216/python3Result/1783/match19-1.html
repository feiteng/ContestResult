class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        prefix_even = [0]*len(nums)
        prefix_odd = [0]*len(nums)
        suffix_even = [0]*len(nums)
        suffix_odd = [0]*len(nums)
        
        for i in range(1,len(nums)):
            if i&1:
                prefix_even[i] = prefix_even[i-1] + nums[i-1]
                prefix_odd[i] = prefix_odd[i-1]
            else:
                prefix_even[i] = prefix_even[i-1]
                prefix_odd[i] = prefix_odd[i-1] + nums
[i-1]
        
        for i in reversed(range(len(nums)-1)):
            if i&1:
                suffix_even[i] = suffix_even[i+1] + nums[i+1]
                suffix_odd[i] = suffix_odd[i+1]
            else:
                suffix_even[i] = suffix_even[i+1]
                suffix_odd[i] = suffix_odd[i+1] + nums
[i+1]
        
        res = 0
        for i in range(len(nums)):
            if prefix_even[i] + suffix_odd[i] == prefix_odd[i] + suffix_even[i]:
                res += 1
        return res
