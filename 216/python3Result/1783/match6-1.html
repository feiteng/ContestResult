class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n = len(nums)
        odd = [0] * (n+1)
        even = [0] * (n+1)
        for i in range(n):
            if i % 2 == 0:
                even[i+1] = even[i] + nums[i]
                odd[i+1] = odd[i]
            else:
                even[i+1] = even[i]
                odd[i+1] = odd[i] + nums
[i]
        
        ret = 0
        for i in range(n):
            if odd[n] - odd[i+1] + even[i] == even[n] - even[i+1] + odd[i]:
                ret += 1
        
        return ret
