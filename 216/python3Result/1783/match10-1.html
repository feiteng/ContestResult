class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n = len(nums)
        pre_odd = [0]*(n+1)
        pre_even = [0]*(n+1)
        suf_odd 
= [0]*(n+1)
        suf_even = [0]*(n+1)
        for i in range(n):
            if i%2 == 0:
                pre_even[i] = nums[i] + (pre_even[i-1] if i-1>=0 else 0)
                pre_odd[i] = pre_odd[i-1] if i-1>=0 else 0
            else:
                pre_odd[i] = nums[i] + (pre_odd[i-1] if i-1>=0 else 0)
                pre_even[i] = pre_even[i-1] if i-1>=0 else 0
        
for i in range(n-1, -1, -1):
            if i%2 == 0:
                suf_even[i] = nums[i] + (suf_even[i+1] if i+1<n else 0)
                suf_odd[i] = suf_odd[i+1] if i+1<n else 0
            else:
                suf_odd[i] = nums[i] + (suf_odd[i+1] if i+1<n else 0)
                suf_even[i] = suf_even[i+1] if i+1<n else 0
        res = 0
        for i in range(n):
            if pre_even[i-1]+suf_odd[i+1] == pre_odd[i-1]+suf_even[i+1]:
                res += 1
        return res
