class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd_left = [0]*len(nums)
        even_left = [0]*len(nums)
        
        odd_right = [0]*len(nums)
        even_right = [0]*len(nums)
        
        for i in range(1, len(nums)):
            if i%2: # odd sum
                odd_left[i] = odd_left[i-1] + nums[i-1]
                even_left[i] = even_left[i-1]
            else:
                even_left[i] = even_left[i-1] + nums[i-1]
                odd_left[i] = odd_left[i-1]
                
        for i in reversed(range(len(nums)-1)):
            if i%2: # odd sum
                odd_right[i] = odd_right[i+1] + nums[i+1]
                even_right[i] = even_right[i+1]
            else:
                even_right[i] = even_right[i+1] + nums[i+1]
                odd_right[i] = odd_right[i+1]
        
        count = 0
        for i in range(len(nums)):
            # if we remove this
            if odd_left[i] + even_right[i] == even_left[i] + odd_right[i]:
                count += 1
        
        return count
