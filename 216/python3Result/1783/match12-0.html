class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        prefEven = [0]
        prefOdd = [0]
        for i, x in enumerate(nums):
            if i % 2 == 0:
                prefEven.append(prefEven[-1] + x)
                prefOdd.append(prefOdd[-1] + 0)
            else:                
                prefEven.append(prefEven[-1] + 0)
                prefOdd.append(prefOdd[-1] + x)
        count = 0
        for i in range(len(nums)):
            evenFront = prefEven[i]
            oddFront = prefOdd[i]
            
            evenBack = prefEven[-1] - prefEven[i+1]
            oddBack = prefOdd[-1] - prefOdd[i+1]
            
            if evenFront + oddBack == oddFront + evenBack:
                count += 1
        return count
