class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        j = [0]
        o = [0]
        
        for i, num in enumerate(nums):
            if i % 2 == 0:
                o.append(o[-1] + num)
                j.append(j[-1])
            else:
                o.append(o[-1])
                j.append(j[-1] + num)
                
        n = len(nums)
        res = 0
        for i in range(n):
            oo = o[i] + (j[n] - j[i+1])
            jj = j[i] + (o[n] - o[i+1])
            if oo == jj:
                res += 1
        return res
