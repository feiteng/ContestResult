class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n = len(nums)
        even = [0] * n
        odd = [0] * n
        for i in range(1, n):
            if i % 2 == 0:
                odd[i] = odd[i - 1] + nums[i - 1]
                even[i] = even[i - 1]
            else:
                even[i] = even[i - 1] + nums[i - 1]
                odd[i] = odd[i - 1]
                
        even_r = [0] * n
        odd_r = [0] * n
        for i in range(n - 1)[::-1]:
            
if i % 2 == 0:
                even_r[i] = even_r[i + 1] + nums[i + 1]
                odd_r[i] = odd_r[i + 1]
            else:
                odd_r[i] = odd_r[i + 1] + nums[i + 1]
                even_r[i] = even_r[i + 1]


        ans = 0
        for i in range(n):
            if odd[i] + odd_r[i] == even[i] + even_r[i]:
                ans += 1
        return ans
