class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd_before = []
        even_before = []
        odd_dp, even_dp = 0, 0
        for i in range(len(nums)):
            odd_before.append(odd_dp)
            even_before.append(even_dp)
            if i % 2 == 0:
                odd_dp += nums[i]
            else:
                even_dp += nums[i]
        odd_after = []
        even_after = []
        odd_dp, even_dp = 0, 0
        for i in range(len(nums)-1, -1, -1):
            odd_after.append(odd_dp)
            even_after.append(even_dp)
            if i % 2 == 0:
                odd_dp += nums[i]
            else:
                even_dp += nums[i]
        odd_after = odd_after[::-1]
        even_after = even_after[::-1]

        ans = 0
        for i in range(len(nums)):
            if odd_before[i] + even_after[i] == even_before[i] + odd_after[i]:
                ans += 1
        return ans
