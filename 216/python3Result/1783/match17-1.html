class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n=len(nums)
        prefix_odd=[0]*(n+1)
        prefix_even=[0]*(n+1)
        for i,num in enumerate(nums):
            if i%2==0:
                prefix_even[i]=prefix_even[i-1]+num
                prefix_odd[i]=prefix_odd[i-1]
            else:
                prefix_even[i]=prefix_even
[i-1]
                prefix_odd[i]=prefix_odd[i-1]+num
        res=0
        for i,num in enumerate(nums):
            if prefix_odd[i-1]+prefix_even[n-1]-prefix_even[i]==prefix_even[i-1]+prefix_odd[n-1]-prefix_odd[i]:
                res+=1
        return res
