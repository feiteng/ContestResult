class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        pre_odd = [0]
        pre_even = [0]
        for i, num in enumerate(nums):
            if i % 2 == 0:
                pre_odd.append(pre_odd[-1])
                pre_even.append(pre_even[-1] + num)
            else:
                pre_odd.append(pre_odd[-1] + num)
                pre_even.append(pre_even[-1])
        #print(pre_odd)
        #print(pre_even)
        ans = 0
        for i in range(len(nums)):
            odd = pre_odd[i] + pre_even[-1] - pre_even[i+1]
            even = pre_even[i] + pre_odd[-1] - pre_odd[i+1]
            #print(i, odd, even)
            if odd == even:
                ans += 1
        return ans
