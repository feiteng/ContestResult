class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        even_sum = odd_sum = 0
        n = len(nums)
        for i in range(n):
            if i % 2 == 0:
                even_sum += nums[i]
            else:
                odd_sum += nums[i]
        even_front = odd_front = 0
        count = 0
        even_tail, odd_tail = odd_sum, even_sum - nums[0]
        if even_front + even_tail == odd_front + odd_tail:
            count += 1
        for i in range(1, n):
            if i % 2 == 1:
                even_front += nums[i - 1]
                even_tail -= nums[i]
            else:
                odd_front += nums[i - 1]
                odd_tail -= nums[i]
            if even_front + even_tail == odd_front + odd_tail:
                count += 1
        return count
