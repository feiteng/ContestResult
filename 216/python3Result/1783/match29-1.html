class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd_front = []
        even_front = []
        odd_back = []
        even_back 
= []
        
        odd_sum = 0
        even_sum = 0
        for i in range(len(nums)):
            odd_front.append(odd_sum)
            even_front.append(even_sum)
            if i%2 == 0:
                even_sum += nums[i]
            else:
                odd_sum += nums[i]
        
        odd_sum = 0
        even_sum = 0
        for i in range(len(nums))[::-1]:
            odd_back
.append(odd_sum)
            even_back.append(even_sum)
            if i%2 == 0:
                even_sum += nums[i]
            else:
                odd_sum += nums[i]
        
        odd_back = odd_back[::-1]
        even_back = even_back[::-1]
        
        ret = 0
        for i in range(len(nums)):
            if odd_back[i] + even_front[i] == odd_front[i] + even_back[i]:
                ret += 1
        
        return ret
