from typing import List

class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        total_odd, total_even = 0, 0
        for i,v in enumerate(nums):
            if i % 2 == 0:
                total_even += v
            else:
                total_odd += v

        curr_odd, curr_even = 0, 0
        res = 0
        for i,v in enumerate(nums):
            
if i % 2 == 0:
                total_even -= v
            else:
                total_odd -= v
            if total_odd + curr_even == total_even + curr_odd:
                res += 1
            if i % 2 == 0:
                curr_even += v
            else:
                curr_odd += v
        return res
