class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n = len(nums)
        even = sum([nums[i] for i in range(0, n, 2)])
        odd = sum([nums[i] for i in range(1, n, 2)])
        
        res = 0
        curr_even, curr_odd = 0, 0
        for i in range(n):
            if i % 2 == 0:
                if even - odd - 2 * curr_even + 2 * curr_odd == +nums[i]:
                    res += 1
                curr_even += nums[i]
            else:
                if even - odd - 2 * curr_even + 2 * curr_odd == -nums[i]:
                    res += 1
                curr_odd += nums[i]
        
        return res
