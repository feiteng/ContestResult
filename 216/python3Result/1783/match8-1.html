class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd = even = 0
        for idx, num in enumerate(nums):
            if idx & 1:
                odd += num
            else:
                even += num
        todd = teven = 0
        ans = 0
        for idx, num in enumerate(nums):
            if idx & 1:
                if todd + even - teven == teven + odd - todd - num:
                    ans += 1
                todd += num
            else:
                if todd + even - teven - num == teven + odd - todd:
                    ans += 1
                teven += num
        return ans
