class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        # presum
        # diff counts even - odd,
        diff = [0]
        for i,num in enumerate(nums):
            if i%2:
                diff.append(diff[-1]-num)
            else:
                diff.append(diff[-1]+num)
        #print(diff)
        res = 0
        for i in range(1,len(diff)):
            prev = diff[i-1]
            nxt = -(diff[-1]-diff[i])
            if prev+nxt==0:
                res+=1
        return res
