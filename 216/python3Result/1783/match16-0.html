class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n=len(nums)
        cs0=[0]*(n+1)
        cs1=[0]*(n+1)
        for i in range(n):
            if i%2:
                cs1[i+1]=cs1[i]+nums[i]
                cs0[i+1]=cs0[i]
            else:
                cs0[i+1]=cs0[i]+nums[i]
                cs1[i+1]=cs1[i]
        # print(cs0)
        # print(cs1)

        ans=0
        for i in range(n):
            if cs0[i]+cs1[n]-cs1[i+1]==cs1[i]+cs0[n]-cs0[i+1]:
                ans+=1

        return ans
