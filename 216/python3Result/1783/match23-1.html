class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd_nums = 0
        even_nums 
= 0
        ans = 0
        for i in range(len(nums)):
            if i%2 == 0:
                even_nums += nums[i]
            else:
                odd_nums += nums[i]
        temp_odd 
= 0
        temp_even = 0
        for i in range(len(nums)):
            if i%2 == 0:
                if temp_even + odd_nums - temp_odd == temp_odd + even_nums - temp_even - nums[i]:
                    ans += 1
                temp_even += nums[i]
            else:
                if temp_odd + even_nums - temp_even == temp_even + odd_nums - temp_odd - nums[i]:
                    ans += 1
                temp_odd += nums[i]
        return ans
