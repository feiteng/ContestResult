class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        a, b = [0], [0]
        for i in range(len(nums)):
            if i % 2 == 0:
                a.append(a[-1] + nums[i])
                b.append(b[-1])
            else:
                a.append(a[-1])
                b.append(b[-1] + nums[i])
        ret = 0
        for i in range(len(nums)):
            if a[i] + b[-1] - b[i + 1] == b[i] + a[-1] - a[i + 1]:
                ret += 1
        return ret
