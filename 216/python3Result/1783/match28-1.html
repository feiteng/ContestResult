class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        oddSumsl = []
        evenSumsl 
= []
        
        odd_summer = 0
        even_summer = 0
        for i in range(len(nums)):
            oddSumsl.append(odd_summer)
            evenSumsl.append(even_summer)
            if i % 2 == 0:
                even_summer += nums[i]
            else:
                odd_summer += nums[i]
        
        oddSumsr = []
        evenSumsr = []
        
        odd_summer = 0
        even_summer = 0
        
for i in range(len(nums)-1, -1, -1):
            oddSumsr.append(odd_summer)
            evenSumsr.append(even_summer)
            if i % 2 == 0:
                even_summer += nums[i]
            else:
                odd_summer += nums[i]
        oddSumsr = list(reversed(oddSumsr))
        evenSumsr = list(reversed(evenSumsr))
        
        ans 
= 0
        for i in range(len(nums)):
            if oddSumsr[i] + evenSumsl[i] == oddSumsl[i] + evenSumsr[i]:
                ans += 1
        return ans
