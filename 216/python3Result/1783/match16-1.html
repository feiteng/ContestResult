class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        N = len(nums)
        odd_pre_sum = [0] * (N + 1)
        even_pre_sum = [0] * (N + 1)
        for i in range(N):
            if i % 2 == 1:
                odd_pre_sum[i + 1] = odd_pre_sum[i] + nums[i]
                even_pre_sum[i + 1] = even_pre_sum[i]
            else:
                odd_pre_sum[i + 1] = odd_pre_sum[i]
                even_pre_sum[i + 1] = even_pre_sum[i] + nums
[i]
        
        count = 0
        for i in range(1, N + 1):
            if even_pre_sum[i - 1] + odd_pre_sum[N] - odd_pre_sum[i] == odd_pre_sum[i - 1] + even_pre_sum[N] - even_pre_sum[i]:
                count += 1
        return count
