class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        even_tot = 0
        odd_tot = 0
        for i, v in enumerate(nums):
            if i % 2 == 0:
                even_tot += v
            else:
                odd_tot += v
        even = 0
        odd = 0
        cnt = 0 
        for i, v in enumerate(nums):
            if i % 2 == 0:
                if even + odd_tot - odd == odd + even_tot - even - v:
                    cnt += 1
                even += v
            else:
                if even + odd_tot - odd - v == odd + even_tot - even:
                    cnt += 1
                odd += v
        return cnt
