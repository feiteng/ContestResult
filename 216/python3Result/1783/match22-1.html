class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd = [0] # prefix sum
        even 
= [0]
        
        for i,a in enumerate(nums):
            if i % 2 == 0:
                even.append(a + even[-1])
                odd.append(odd[-1])
            else:
                odd.append(a + odd[-1])
                even.append(even[-1])

        
        res = 0
        for i, a in enumerate(nums):
            if even[-1] - even[i+1] + odd[i] == odd[-1] - odd[i+1] + even[i]:
                res += 1
        return res
