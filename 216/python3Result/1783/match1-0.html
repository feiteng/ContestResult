class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        pfso = [0]
        pfse = [0]
        for idx, num in enumerate(nums):
            if idx % 2 == 0:
                # odd
                pfso.append(pfso[-1] + num)
                pfse.append(pfse[-1])
            else:
                # even
                pfso.append(pfso[-1])
                pfse.append(pfse[-1] + num)
        # print(pfso, pfse)
        ans = 0
        for idx in range(len(nums)):
            # if idx % 2 == 0:
            #     # odd index
            odds = pfso[idx] + (pfse[-1] - pfse[idx + 1])
            evens = pfse[idx] + (pfso[-1] - pfso[idx + 1])
            if odds == evens:
                ans += 1
            # print(pfso[idx], (pfse[-1] - pfse[idx]))
            # print(pfse[idx], (pfso[-1] - pfso[idx]))
            # print(odds, evens)
            # print("---")
        return ans
