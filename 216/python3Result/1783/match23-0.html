class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd=even=0
        for i in range(len(nums)):
            if i%2==0:
                even+=nums[i]
            else:
                odd+=nums[i]
        o=e=0
        out=0
        for i in range(len(nums)):
            if i%2==0:
                if e+odd-o==even-nums[i]-e+o:
                    out+=1
                e+=nums[i]
            else:
                if e+odd-o-nums[i]==o+even-e:
                    out+=1
                o+=nums[i]
        return out
