class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        n = len(nums)
        odd_b = [0] * n
        even_b = [0] * n
        for i in range(1, n):
            if i % 2 != 0:
                odd_b[i] += odd_b[i-1] + nums[i-1]
                even_b[i] = even_b[i-1]
            else:
                even_b[i] += even_b[i-1] + nums[i-1]
                odd_b[i] = odd_b[i-1]
        odd_a = [0] * n
        even_a = [0] * n
        for i in range(n-2,-1, -1):
            if i % 2 != 0:
                odd_a[i] = odd_a[i+1] + nums[i+1]
                even_a[i] = even_a[i+1]
            else:
                even_a[i] = even_a[i+1] + nums[i+1]
                odd_a[i] = odd_a[i+1]
        count = 0
        for i in range(n):
            if (odd_b[i] + even_a[i]) == (even_b[i] + odd_a[i]):
                count += 1
        return count
