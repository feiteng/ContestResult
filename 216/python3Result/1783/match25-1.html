class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        
        n = len(nums)
        
        odd = [0] * n
        even = [0] * n
        
        for i in range(n):
            
            if i % 2 == 1:
                odd[i] += nums[i]
            else:
                even[i] += nums[i]
            if i > 0:
                odd
[i] += odd[i-1]
                even[i] += even[i-1]
        res = 0      
        for i in range(n):
            od = 0
            ev = 0
            if i > 0:
                od = odd[i-1]
                ev = even[i-1]
            od 
+= even[-1] - even[i]
            ev += odd[-1] - odd[i]
            if od == ev:
                res += 1
        return res
