class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        esum = [0]
        osum = [0]
        for i,n in enumerate(nums):
            if i%2==0:
                esum.append(esum[-1]+n)
                osum.append(osum[-1])
            else:
                esum.append(esum[-1])
                osum.append(osum[-1]+n)
        print(f"esum {esum} osum {osum}")
        result = 0
        for i in range(1,len(nums)+1):
            pree = esum[i-1]
            preo = osum[i-1]
            poste = osum[-1]-osum[i]
            posto = esum[-1]-esum[i]
            if pree+poste == preo + posto:
                result+=1
            #print(f"pree {pree} preo {preo} poste {poste} posto {posto}")
        return result
