class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odd_sum = [0]
        even_sum = [0]
        
        for index, num in enumerate(nums):
            if index % 2 == 1:
                odd_sum.append(odd_sum[-1] + num)
                even_sum.append(even_sum[-1])
            else:
                odd_sum.append(odd_sum[-1])
                even_sum.append(even_sum[-1] + num)
                
        result = 0
        for index in range(1, len
(odd_sum)):
            removed_odd_sum = odd_sum[index - 1] + even_sum[-1] - even_sum[index]
            removed_even_sum = even_sum[index - 1] + odd_sum[-1] - odd_sum[index]
            if removed_odd_sum == removed_even_sum:
                result += 1
                
        return result
