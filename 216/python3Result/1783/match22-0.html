class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        odds, evens = [], []
        for i, x in enumerate(nums):
            if i % 2:
                odds.append(x + (odds[-1] if odds else 0))
                evens.append(evens[-1] if evens else 0)
            else:
                evens.append(x + (evens[-1] if evens else 0))
                odds.append(odds[-1] if odds else 0)
        res = 0
        for i, x in enumerate(nums):
            if odds[-1] - odds[i] + (evens[i-1] if i > 0 else 0) == evens[-1] - evens[i] + (odds[i-1] if i > 0 else 0):
                res += 1
        return res
