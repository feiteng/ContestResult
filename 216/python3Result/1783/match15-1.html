class Solution:
    def waysToMakeFair(self, nums: List[int]) -> int:
        sum0, sum1 = 0, 0
        for i, t in enumerate(nums) :
            if i % 2 == 0 :
                sum0 += t
            else :
                sum1 += t
        
        to_ret = 0
        t0, t1 = 0, 0
        for i, t in enumerate(nums) :
            
if i % 2 == 0 :
                sum0 -= t
            else :
                sum1 -= t
            
            if t0 + sum1 == t1 + sum0 :
                to_ret += 1
            if i % 2 == 0 :
                t0 += t
            else :
                t1 += t
                
        return to_ret
