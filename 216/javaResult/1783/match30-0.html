class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] leftEven = new int[n];
        int[] leftOdd = new int[n];
        int[] rightEven = new int[n];
        int[] rightOdd = new int[n];
        int even = 0;
        int odd = 0;
        for(int i = 0 ; i < n ; i++){
            if(i % 2 == 0){
                even += nums[i];
                leftEven[i] =  even;
                leftOdd[i] = odd;
            }
            else{
                odd += nums[i];
                leftOdd[i] = odd;
                leftEven[i] = even;
            }
        }
        even = 0;
        odd = 0;
        for(int i = n - 1 ; i >=0 ; i-- ){
            if(i % 2 == 0){
                even += nums[i];
                rightEven[i] =  even;
                rightOdd[i] = odd;
            }
            else{
                odd += nums[i];
                rightOdd[i] = odd;
                rightEven[i] = even;
            }
        }
        int result = 0;
        for(int i = 0 ; i < n ; i++){
            even = 0;
            odd = 0;
            if(i % 2 == 0){
                even  = leftEven[i] - nums[i];
                even += rightOdd[i];
                odd = leftOdd[i];
                odd += rightEven[i]-nums[i];
            }
            else{
                odd  = leftOdd[i] - nums[i];
                odd += rightEven[i];
                even = leftEven[i];
                even += rightOdd[i]-nums[i];
            }
            
            if(odd == even){
                result++;
            }
        }
        return result;
    }
}
