class Solution {
    public int waysToMakeFair(int[] nums) {
        int len = nums.length;
        int[] odd = new int[len + 1];
        int[] even = new int[len + 1];
        for (int i = 0; i < len; i++) {
            if (i % 2 == 0) {
                even[i + 1] = even[i] + nums[i];
                odd[i + 1] = odd[i];
            } else {
                odd[i + 1] = odd[i] + nums[i];
                even[i + 1] = even[i];
            }
        }
        int res = 0;
        for (int i = 0; i < len; i++) {
            if (i % 2 == 0 && (i == 0 ? 0 : even[i - 1]) + odd[len] - odd[i] == odd[i] + even[len] - even[i + 1]
                    || i % 2 == 1 && (i == 0 ? 0 : odd[i - 1]) + even[len] - even[i] == even[i] + odd[len] - odd[i + 1]) {
                res++;
            }
        }
        return res;
    }
}
