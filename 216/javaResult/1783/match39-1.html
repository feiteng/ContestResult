class Solution {
    public int waysToMakeFair(int[] nums) {
        int len = nums.length;
        int l = 0, r = 0;
        int[] ls = new int[len+1], rs = new int[len+1];
        for(int i=len-1;i>=0;i--){
            rs[i]=rs[i+1];
            ls[i]=ls[i+1];
            
if(i%2==0){
                rs[i]+=nums[i];
            }
            else{
                ls[i]+=nums[i];
            }
        }
        int ct = 0;
        
for(int i=0;i<len;i++){
            if(r+ls[i+1]==l+rs[i+1]){
                ct++;
            }
            if(i%2==0){
                r+=nums[i];
            }
            else{
                l+=nums[i];
            }
        }
        return ct;
    }
}
