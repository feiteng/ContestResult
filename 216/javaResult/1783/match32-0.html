class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] preEven = new int[n];
        int[] preOdd = new int[n];
        int[] postEven = new int[n];
        int[] postOdd = new int[n];
        for (int i = 1; i < n; i++) {
            if (i % 2 == 0) {
                preEven[i] = preEven[i-1];
                preOdd[i] = preOdd[i-1] + nums[i-1];
            } else {
                preEven[i] = preEven[i-1] + nums[i-1];
                preOdd[i] = preOdd[i-1];
            }
        }
        for (int i = n-2; i >= 0; i--) {
            if (i % 2 == 0) {
                postEven[i] = postEven[i+1];
                postOdd[i] = postOdd[i+1] + nums[i+1];
            } else {
                postEven[i] = postEven[i+1] + nums[i+1];
                postOdd[i] = postOdd[i+1];
            }
        }
        int result = 0;
        for (int i = 0; i < n; i++) {
            if (preEven[i] + postOdd[i] == preOdd[i] + postEven[i]) result++;
        }
        return result;
    }
}
