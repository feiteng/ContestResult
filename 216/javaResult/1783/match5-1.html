class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] odd = new int[n + 1];
        int[] even = new int[n + 1];
        for (int i = n - 1; i >= 0; i--) {
            if (i % 2 == 0) {
                odd[i] = odd[i+1] + nums[i];
                even[i] = even[i+1];
            } else {
                odd[i] = odd[i+1];
                even[i] = even[i+1] + nums[i];
            }
        }
        
        int ans = 0;
        for (int i = 0; i < n; i++) {
            int sumOdd = odd[0] - odd[i] + even[i+1];
            int sumEven = even[0] - even[i] + odd[i+1];
            if (sumOdd == sumEven) ans++;
        }
        
        return ans;
    }
}
