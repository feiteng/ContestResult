class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[][] a1 = new int[n][2];
        int[][] a2 = new int[n][2];
        for(int i = 0; i < n-1; i++){
            a1[i+1][0] = a1[i][0];
            a1[i+1][1] = a1[i][1];
            if(i%2==0){
                a1[i+1][1]+=nums[i];
            }else{
                a1[i+1][0]+=nums[i];
            }
            //System.out.println(a1[i+1][0] + " " + a1[i+1][1]);
        }
        for(int i = n-1; i > 0; i--){
            a2[i-1][0] = a2[i][0];
            a2[i-1][1] = a2[i][1];
            if(i%2 == 0){
                a2[i-1][1] += nums[i];
            }else{
                a2[i-1][0] += nums[i];
            }
            //System.out.println(a2[i-1][0] + " " + a2[i-1][1]);
        }
        int res = 0;
        for(int i = 0; i < n; i++){
            if(a1[i][0] + a2[i][1] == a1[i][1] + a2[i][0]){
                res++;
            }
        }
        
        
        
        
        
        
        return res;
        
    }
}
