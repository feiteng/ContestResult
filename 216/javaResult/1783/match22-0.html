class Solution {
    public int waysToMakeFair(int[] nums) {
        int n=nums.length;
        int[] odd=new int[n+1];
        int[] even=new int[n+1];
        for(int i=1;i<=n;i++){
            odd[i]=odd[i-1];
            even[i]=even[i-1];
            if(i%2==1) even[i]+=nums[i-1];
            else odd[i]+=nums[i-1];
        }
        // for(int i:odd) System.out.println(i);
        // for(int i:even) System.out.println(i);
        int res=0;
        for(int i=1;i<=n;i++){
            if(even[i-1]+odd[n]-odd[i]==odd[i-1]+even[n]-even[i]) {
                res++;
            }
        }
        return res;
    }
}
