class Solution {
    public int waysToMakeFair(int[] nums) {
        int n=nums.length;
        int[] odd=new int[n+1], even=new int[n+1];
        
        for(int i=0; i<n; i++){
            if(i%2==0){
                even[i+1] += even[i] + nums[i];
                odd[i+1] = odd[i];
            }else{
                even[i+1]=even[i];
                odd[i+1] += odd[i] + nums[i];
            }
        }
        // for(int o : odd) System.out.print(o+" ");
        // System.out.println();
        // for(int o : even) System.out.print(o+" ");
        // System.out.println();

        int res = 0;
        for(int i=1; i<=n; i++){
            if(odd[i-1] + even[n]-even[i] == even[i-1] + odd[n] - odd[i]){
                res ++;
            }
        }
        return res;
    }
}
