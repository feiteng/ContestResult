class Solution {
        public int waysToMakeFair(int[] nums) {
            int length = nums.length;
            int[] odd = new int[length+1];
            int[] even = new int[length+1];
            for (int i=0;i<length;i++) {
                even[i+1] = even[i] + (i%2==0 ? nums[i] : 0);
                odd[i+1] = odd[i] + (i%2==0 ? 0 : nums[i]);
            
}
            int count = 0;
            for (int i=1;i<=length;i++) {
                if (odd[i-1] + even[length] - even[i] == even[i-1] + odd[length] - odd[i]) {
                    count++;
                }
            }
            return count;
        }
    }
