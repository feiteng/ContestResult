class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] oddSum = new int[n];
        int[] evenSum = new int[n];
        for(int i = 0; i < n; i++){
            if(i == 0){
                evenSum[0] = nums[0];
            }
            else{
                if(i % 2 == 0){
                    evenSum[i] = evenSum[i-1] + nums[i];
                    oddSum[i] = oddSum[i-1];
                }
                else{
                    oddSum[i] = oddSum[i-1] + nums[i];
                    evenSum[i] = evenSum[i-1];
                }
            }
        }
        
        int res = 0;
        for(int i = 0; i < n; i++){
            //try remove index i
            int realOddSum = 0;
            int realEvenSum = 0;
            //left part
            if(i > 0){
                realOddSum += oddSum[i-1];
                realEvenSum += evenSum[i-1];
            }
            //right part
            realOddSum += evenSum[n-1] - evenSum[i];
            realEvenSum += oddSum[n-1] - oddSum[i];
            if(realOddSum == realEvenSum){
                res++;
            }
        }
        return res;
    }
}
