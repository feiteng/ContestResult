class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[][] pre = new int[n][2];
        int[][] suf = new int[n][2];
        
        for (
int i = 0, preodd = 0, preeven = 0; i < n; i++) {
            if (i % 2 == 0) {
                pre[i][0] = preeven + nums[i];
                pre[i][1] = preodd;
            }
            else {
                pre[i][0] = preeven;
                pre[i][1] = preodd + nums[i];
            }
            preeven = pre[i][0];
            preodd = pre[i][1];
        }
        
        for (int i = n - 1, preodd = 0, preeven = 0; i >= 0; i--) {
            
if (i % 2 == 0) {
                suf[i][0] = preeven + nums[i];
                suf[i][1] = preodd;
            }
            else {
                suf[i][0] = preeven;
                suf[i][1] = preodd + nums[i];
            }
            preeven = suf[i][0];
            preodd = suf[i][1];
        
}
        
        int ret = 0;
        for (int i = 0; i < n; i++) {
            if (pre[i][0] + suf[i][1] - nums[i] == pre[i][1] + suf[i][0] - nums[i]) ret += 1;
        }
        return ret;
    }
}
