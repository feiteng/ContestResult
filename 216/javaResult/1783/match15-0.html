class Solution {
    public int waysToMakeFair(int[] nums) {
        // preSumOdd[0...i]+preSumEven[i...n] = preSumEven[0...i]+preSumOdd[i...n]
        int n = nums.length;
        int[] odd = new int[n+1], even = new int[n+1];
        for (int i = 0; i < n; i++){
            if (i%2 == 0){
                even[i+1] = even[i] + nums[i];
                odd[i+1] = odd[i];
            }else{
                odd[i+1] = odd[i] + nums[i];
                even[i+1] = even[i];
            }
        }
        // System.out.println(Arrays.toString(odd));
        // System.out.println(Arrays.toString(even));
        int res = 0;
        for (int i = 0 ; i < n; i++){
            int odd0 = odd[i], odd1 = odd[n]-odd[i+1];
            int even0 = even[i], even1 = even[n]-even[i+1];
            if (odd0 + even1 == odd1 + even0){
                res++;
            }
        }
        return res;
    }
}
