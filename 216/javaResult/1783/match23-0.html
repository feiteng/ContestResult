class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] beforeOdd = new int[n];
        int[] beforeEven = new int[n];
        int[] afterOdd = new int[n];
        int[] afterEven = new int[n];
        
        for (int i = 1; i < n; ++i) {
            beforeOdd[i] = beforeOdd[i - 1] + (i % 2 == 0 ? nums[i - 1] : 0);
            beforeEven[i] = beforeEven[i - 1] + (i % 2 == 1 ? nums[i - 1] : 0);
        }
        
        for (int i = n - 2; i >= 0; --i) {
            afterOdd[i] = afterOdd[i + 1] + (i % 2 == 1 ? nums[i + 1] : 0);
            afterEven[i] = afterEven[i + 1] + (i % 2 == 0 ? nums[i + 1] : 0);
        }
        
        int count = 0;
        for (int i = 0; i < n; ++i) {
            if (beforeOdd[i] + afterOdd[i] == beforeEven[i] + afterEven[i]) {
                ++count;
            }
        }
        
        return count;
    }
}
