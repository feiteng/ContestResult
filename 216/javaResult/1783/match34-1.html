class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] even = new int[n], odd = new int[n];
        even[0] = nums[0];
        for (int i = 1; i < n; i++) {
            if (i % 2 == 0) {
                even[i] = even[i - 1] + nums[i];
            } else {
                even[i] = even[i - 1];
            }
        }
        
for (int i = 1; i < n; i++) {
             if (i % 2 == 0) {
                odd[i] = odd[i - 1];
            } else {
                odd[i] = odd[i - 1] + nums[i];
            }
        }
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            int evenSum, oddSum;
            if (i == 0) {
                evenSum = odd[n - 1] - odd[i];
                oddSum = even[n - 1] - even[i];
            } else {
                evenSum = even[i - 1] + odd[n - 1] - odd[i];
                oddSum = odd[i - 1] + even[n - 1] - even[i];
            }
            
if (evenSum == oddSum) {
                // System.out.println("Could remove index: " + i);
                cnt ++;
            }
        }
        return cnt;
    }
}
