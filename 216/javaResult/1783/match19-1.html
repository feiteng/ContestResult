class Solution {
    public int waysToMakeFair(int[] nums) {
        
long sum1 = 0;
        long sum2 = 0;
        for(int i = 0; i < nums.length; i++){
            if(i%2 == 0){
                sum2 += nums[i];
            }else{
                sum1 += nums[i];
            }
        }
        long pre1 = 0;
        
long pre2 = 0;
        int n = 0;
        for(int i = 0; i < nums.length; i++){
            if(i %2 ==0){
                sum2-= nums[i];
            }else{
                sum1-= nums[i];
            }
            if(pre1 + sum2 == pre2 + sum1){
                n++;
            }
            if(i %2 ==0){
                pre2+= nums[i];
            }else{
                pre1+= nums[i];
            }
        }
        return n;
    }
}
