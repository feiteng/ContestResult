class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] preSum = new int[n + 2];
        int[] postSum = new int[n + 2];
        
        
preSum[1] = nums[0];
        for(int i = 1; i < n; i++){
            preSum[i + 1] = preSum[i - 1] + nums[i];
        }
        
        postSum[n] = nums[n - 1];
        for(int i = n - 2; i >= 0; i--){
            postSum[i + 1] = postSum[i + 3] + nums[i];
        }
        
        int res = 0;
        for(int i = 0; i < n; i++){
            //删除nums[i];
            
if(preSum[i + 1] - nums[i] + postSum[i + 2] == postSum[i + 1] - nums[i] + preSum[i]){
                res++;
            }
        }
        return res;
    }
}
