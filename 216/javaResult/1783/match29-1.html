class Solution {
    public int waysToMakeFair(int[] nums) {
        int[] evenFromLeft = new int[nums.length + 1]; 
        int[] evenFromRight = new int[nums.length + 1]; 
        
int[] oddFromLeft = new int[nums.length + 1]; 
        int[] oddFromRight = new int[nums.length + 1]; 
        
        for (int i = 1; i < evenFromLeft.length; i++) {
            if ((i - 1) % 2 == 0) {
                evenFromLeft[i] = evenFromLeft[i - 1] + nums[i - 1];
                oddFromLeft[i] = oddFromLeft[i - 1];
            } else {
                evenFromLeft[i] = evenFromLeft[i - 1];
                oddFromLeft[i] = oddFromLeft[i - 1] + nums[i - 1];
            }
        }
        
        
for (int i = evenFromLeft.length - 2; i >= 0; i--) {
            if (i % 2 == 0) {
                evenFromRight[i] = evenFromRight[i + 1] + nums[i];
                oddFromRight[i] = oddFromRight[i + 1];
            } else {
                evenFromRight[i] = evenFromRight[i + 1];
                oddFromRight[i] = oddFromRight[i + 1] + nums[i];
            }
        }
        
        int result = 0;
        
        for (int i = 0; i < nums.length; i++) {
            if (evenFromLeft[i] + oddFromRight[i + 1] == oddFromLeft[i] + evenFromRight[i + 1]) {
                result++;
            }
        }
        
        return result;
    }
}
