class Solution {
    public int waysToMakeFair(int[] nums) {
        
        int n = nums.length;
        
        int[][] evens = new int[n][2];
        int[][] odds = new int[n][2];
        
        int leftEven = 0;
        int leftOdd = 0;
        int rightEven = 0;
        int rightOdd = 0;
        
        for (int i=0; i<n; i++) {
            if (i % 2 == 0) {
                evens[i][0] = leftEven;
                leftEven += nums[i];
                odds[i][0] = leftOdd;
            } else {
                odds[i][0] = leftOdd;
                leftOdd += nums[i];
                evens[i][0] = leftEven;
            }
            if ((n-1-i) % 2 == 0) {
                evens[n-1-i][1] = rightEven;
                rightEven += nums[n-1-i];
                odds[n-1-i][1] = rightOdd;
            } else {
                odds[n-1-i][1] = rightOdd;
                rightOdd += nums[n-1-i];
                evens[n-1-i][1] = rightEven;
            }
        }
        
        int count = 0;
        
        for (int i=0; i<n; i++) {
            if (evens[i][0] + odds[i][1] == evens[i][1] + odds[i][0]) count++;
        }
        
        return count;
    }
}
