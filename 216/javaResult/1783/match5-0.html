class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] odd = new int[n+1];
        int[] even = new int[n+1];
        
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0) {
                odd[i+1] = odd[i];
                even[i+1] = even[i] + nums[i];
            } else {
                odd[i+1] = odd[i] + nums[i];
                even[i+1] = even[i];
            }
        }
        
        int ans = 0;
        
        for (int i = 0; i < n; i++) {
            int o = odd[i];
            int e = even[i];
            o += even[n]-even[i+1];
            e += odd[n]-odd[i+1];
            if (o == e) ans++;
            //System.out.println(o + " " + e);
        }
        return ans;
        
    }
}
