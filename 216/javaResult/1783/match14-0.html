class Solution {
    public int waysToMakeFair(int[] nums) {
        
        int[] leftEvenSums = new int[nums.length + 1];
        int[] leftOddSums = new int[nums.length + 1];
        
        int[] rightEvenSums = new int[nums.length + 1];
        int[] rightOddSums = new int[nums.length + 1];
        
        leftEvenSums[0] = 0;
        leftOddSums[0] = 0;
        for (int i = 0; i < nums.length; i++)
        {
            leftEvenSums[i + 1] = leftEvenSums[i];
            leftOddSums[i + 1] = leftOddSums[i];
            if ((i % 2) == 0)
            {
                leftEvenSums[i + 1] += nums[i];
            }//if
            else
            {
                leftOddSums[i + 1] += nums[i];
            }//else
        }//for

        rightEvenSums[0] = 0;
        rightOddSums[0] = 0;
        for (int i = 0; i < nums.length; i++)
        {
            rightEvenSums[i + 1] = rightEvenSums[i];
            rightOddSums[i + 1] = rightOddSums[i];
            int index = nums.length - i - 1;
            if ((index % 2) == 0)
            {
                rightEvenSums[i + 1] += nums[index];
            }//if
            else
            {
                rightOddSums[i + 1] += nums[index];
            }//else
        }//for

        int numFair = 0;
        for (int removalIndex = 0; removalIndex < nums.length; removalIndex++)
        {
            int evensSum = leftEvenSums[removalIndex] + rightOddSums[nums.length - removalIndex - 1];
            int oddsSum = leftOddSums[removalIndex] + rightEvenSums[nums.length - removalIndex - 1];
            
            if (evensSum == oddsSum)
            {
                numFair++;
            }//if
        }//for
        
        return numFair;


    }
}
