class Solution {
    
    public int waysToMakeFair(int[] nums) {
        
        int[][] l = new int[nums.length + 1][2];
        int[][] r = new int[nums.length + 1][2];
        
        int ans = 0;
        
        for(int i=0; i<nums.length; i++) {
            l[i + 1][i % 2] = l[i][i % 2] + nums[i];
            l[i + 1][1 - (i % 2)] = l[i][ 1 - (i % 2)];
        }
        
        for(int i=nums.length-1; i>=0; i--) {
            r[i][i % 2] = r[i + 1][i % 2] + nums[i]; 
            r[i][1 - (i % 2)] = r[i + 1][1 - (i % 2)];
        }
        
        for(int i=0; i<nums.length; i++) {
            if (l[i][0] + r[i + 1][1] == l[i][1] + r[i + 1][0]) {
                ans++;
            }
        }
        
        return ans;
    }
}
