class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] oddSum = new int[n + 1];
        int[] evenSum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            if (i % 2 == 0) {
                evenSum[i + 1] = evenSum[i] + nums[i];
                oddSum[i + 1] = oddSum[i];
            } else {
                oddSum[i + 1] = oddSum[i] + nums[i];
                
evenSum[i + 1] = evenSum[i];
            }
        }

        int out = 0;
        for (int i = 0; i < n; i++) {
            int o = oddSum[i] + evenSum[n] - evenSum[i + 1];
            int e = evenSum[i] + oddSum[n] - oddSum[i + 1];
            
if (o == e) {
                out++;
            }
        }
        return out;
    }
}
