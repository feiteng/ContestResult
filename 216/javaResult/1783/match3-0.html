class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] odds = new int[n + 1];
        int[] evens = new int[n + 1];
        for (int i = 0; i < n; ++i) {
            if (i % 2 == 0) {
                evens[i + 1] = nums[i] + evens[i];
                odds[i + 1] = odds[i];
            } else {
                evens[i + 1] = evens[i];
                odds[i + 1] = nums[i] + odds[i];
            }
        }
        int cnt = 0;
        for (int i = 0; i < n; ++i) {
            int even = evens[i] + odds[n] - odds[i + 1];
            int odd = odds[i] + evens[n] - evens[i + 1];
            if (even == odd) {
                ++cnt;
            }
        }
        return cnt;
    }
}
