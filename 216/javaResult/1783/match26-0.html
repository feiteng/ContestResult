class Solution {
    public int waysToMakeFair(int[] nums) {
        int n=nums.length,ans=0;
        int pre1[]=new int[nums.length+1];
        int pre2[]=new int[nums.length+1];
        for(int i=0;i<n;i++){
            if(i%2==0){
                pre2[i+1]=pre2[i]+nums[i];
                pre1[i+1]=pre1[i];
            } else {
                pre1[i+1]=pre1[i]+nums[i];
                pre2[i+1]=pre2[i];
            }
        }
        for(int i=0;i<n;i++){
            if(pre1[i]+pre2[n]-pre2[i+1]==pre2[i]+pre1[n]-pre1[i+1]) ans++;
        }
        return ans;
    }
}
