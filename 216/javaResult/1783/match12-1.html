class Solution {
    public int waysToMakeFair(int[] nums) {
        final int N = nums.length;
        int[] oddSum = new int[N + 1];
        int[] evenSum = new int[N + 1];
        for (int i = 0; i < N; ++i) {
            if (i % 2 == 0) {
                evenSum[i + 1] = evenSum[i] + nums[i];
                oddSum[i + 1] = oddSum[i];
            } else {
                evenSum[i + 1] = evenSum[i];
                
oddSum[i + 1] = oddSum[i] + nums[i];
            }
        }
        int count = 0;
        for (int i = 0; i < N; ++i) {
            int oddPart = oddSum[i] + evenSum[N] - evenSum[i + 1];
            int evenPart = evenSum[i] + oddSum[N] - oddSum[i + 1];
            
if (oddPart == evenPart) {
                ++count;
            }
        }
        return count;
    }
}
