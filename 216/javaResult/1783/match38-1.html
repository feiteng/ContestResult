class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        if (n <= 1) return n;
        
        int[] left_oddsum = new int[n];
        left_oddsum[1] = nums[1];
        int[] left_evensum = new int[n];
        left_evensum[0] = nums[0];
        left_evensum[1] = left_evensum[0];
        
for (int i = 2; i < n; i++) {
            if (i % 2 == 0) {
                left_evensum[i] = left_evensum[i - 1] + nums[i];
                left_oddsum[i] = left_oddsum[i - 1];
            } else {
                left_oddsum[i] = left_oddsum[i - 1] + nums[i];
                left_evensum[i] = left_evensum[i - 1];
            }
        }
        
        int[] right_oddsum = new int[n];
        int[] right_evensum = new int[n];
        if (n % 2 == 0) {
            right_oddsum[n - 1] = nums[n - 1];
            right_oddsum[n - 2] = right_oddsum[n - 1];
            right_evensum[n - 2] = nums[n - 2];
        } else {
            right_evensum[n - 1] = nums[n - 1];
            right_evensum[n - 2] = right_evensum[n - 1];
            right_oddsum[n - 2] = nums[n - 2];
        }
        
for (int i = n - 3; i >= 0; i--) {
            if (i % 2 == 0) {
                right_evensum[i] = right_evensum[i + 1] + nums[i];
                right_oddsum[i] = right_oddsum[i + 1];
            } else {
                right_oddsum[i] = right_oddsum[i + 1] + nums[i];
                right_evensum[i] = right_evensum[i + 1];
            }
        }
        
        int ans = 0;
        // i == 0
        if (right_oddsum[1] == right_evensum[1]) {
            ans++;
        }
        // i == n - 1
        if (left_oddsum[n - 2] == left_evensum[n - 2]) {
            ans++;
        }
            
        // left-oddsum + right-evensum = left-evensum + right-oddsum
        
for (int i = 1; i < n - 1; i++) {
            // System.out.println("left_odd: " + left_oddsum[i - 1] + " left_even: " + left_evensum[i - 1]);
            // System.out.println("right_odd: " + right_oddsum[i + 1] + " right_even: " + right_evensum[i + 1]);
            if (left_oddsum[i - 1] + right_evensum[i + 1] == left_evensum[i - 1] + right_oddsum[i + 1]) {
                ans++;
            }
        }
        return ans;
    }
}
