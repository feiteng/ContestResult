class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        
int[][] dp = new int[n + 1][4]; 
        // dp[i][0] sum(nums[odd ..i-1])
        // dp[i][1] sum(nums[even ..i-1])
        // dp[i][2] sum(nums[odd i+1..])
        // dp[i][3] sum(nums[even i+1..])
        for (int i = 0; i < n; i++) {
            dp[i + 1][0] = dp[i][0];
            dp[i + 1][1] = dp[i][1];
            
            if (i % 2 == 0) {
                dp[i + 1][1] += nums[i];
            } else {
                dp[i + 1][0] += nums[i];
            }
        }
//         dp[n + 1][0] = dp[n][0];
//         dp[n + 1][1] = dp[n][1];
        
        
        for (int i = n - 1; i > 0; i--) {
            dp[i - 1][2] = dp[i][2];
            dp[i - 1][3] = dp[i][3];
            
            if (i % 2 == 0) {
                dp[i - 1][2] += nums[i];
            } else {
                dp[i - 1][3] += nums[i];
            }
            // System.out.println((i) + " " + dp[i][0] + " " + dp[i][1] + " " + dp[i][2] + " " + dp[i][3]);
            // System.out.println((i - 1) + " " + dp[i - 1][0] + " " + dp[i - 1][1] + " " + dp[i - 1][2] + " " + dp[i - 1][3]);
        }
        // dp[0][2] = dp[1][2];
        // dp[0][3] = dp[1][3];
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (dp[i][0] + dp[i][2] == dp[i][1] + dp[i][3]) {
                res++;
                // System.out.println(i);
            }
        }
        
        return res;
    }
}
