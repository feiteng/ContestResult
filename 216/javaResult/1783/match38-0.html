class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        if(n == 0) return 0;
        
        int[] oddSumBeforeMe = new int[n];
        int[] evenSumBeforeMe = new int[n];
        for(int i = 1; i < n; i++) {
            if(i % 2 == 0) {
                evenSumBeforeMe[i] = evenSumBeforeMe[i-1];
                oddSumBeforeMe[i] = oddSumBeforeMe[i-1] + nums[i-1];
            } else {
                evenSumBeforeMe[i] = evenSumBeforeMe[i-1] + nums[i-1];
                oddSumBeforeMe[i] = oddSumBeforeMe[i-1];
            }
        }
        
        int[] oddSumAfterMe = new int[n];
        int[] evenSumAfterMe = new int[n];
        for(int i = n-2; i >= 0; i--) {
            if(i % 2 == 0) {
                evenSumAfterMe[i] = evenSumAfterMe[i+1];
                oddSumAfterMe[i] = oddSumAfterMe[i+1] + nums[i+1];
            } else {
                evenSumAfterMe[i] = evenSumAfterMe[i+1] + nums[i+1];
                oddSumAfterMe[i] = oddSumAfterMe[i+1];
            }
        }
        
        int result = 0;
        for(int i = 0; i < n; i++) {
            // System.out.println(oddSumBeforeMe[i] + "_" + evenSumBeforeMe[i] + "_" + oddSumAfterMe[i] + "_" + evenSumAfterMe[i]);
            
            if(oddSumBeforeMe[i]+evenSumAfterMe[i] == evenSumBeforeMe[i]+oddSumAfterMe[i]) {
                result++;
            }
        }
        
        return result;
    }
}
