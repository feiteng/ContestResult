class Solution {
    public int waysToMakeFair(int[] nums) {
            int osum = 0;
    int esum = 0;
    for (int i = 0; i < nums.length; i++) {
      if (i % 2 == 0) {
        esum += nums[i];
      } else {
        osum += nums[i];
      }
    }

    int ans = 0;
    int osumpre = 0;
    int esumpre = 0;
    for (int i = 0; i < nums.length; i++) {
      // if remove i
      int e = osum - osumpre + esumpre - (i % 2 == 1 ? nums[i] : 0);
      int o = esum - esumpre + osumpre - (i % 2 == 0 ? nums[i] : 0);
      
if (o == e) {
        ans++;
      }
      // System.out.format("i:%d osum:%d esum:%d osumpre:%d esumpre:%d e:%d o:%d\n",
      //    i, osum, esum, osumpre, esumpre, e, o);

      if (i % 2 == 0) {
        esumpre += nums[i];
      } else {
        osumpre += nums[i];
      }
    }
    return ans;
    }
}
