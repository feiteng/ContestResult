class Solution {
    public int waysToMakeFair(int[] nums) {
        int n = nums.length;
        int[] oc =new int[n];
        int[] ec = new int[n];
        
        for(int i = 0 ; i < n ; i++) {
            if((i+1)%2 == 1) {
                oc[i] += (i != 0 ? oc[i-1] : 0) + nums[i];
                ec[i] += (i != 0 ? ec[i-1] : 0);
            } else {
                oc[i] += (i != 0 ? oc[i-1] : 0);
                ec[i] += (i != 0 ? ec[i-1] : 0) + nums[i];
            }
        
}
        
        // System.out.println(Arrays.toString(oc));
        // System.out.println(Arrays.toString(ec));
        
        int ans = 0;
        
        //try removing every index
        for(int i = 0 ; i < n ; i++) {
            int eCount =0, oCount = 0;
            if(i == 0) {
                eCount = oc[n-1] - nums[i];
                
oCount = ec[n-1];
            } else {
                //odd
                if((i+1)%2 == 1) {
                    oCount = oc[i-1] + ec[n-1] - ec[i];
                    eCount = ec[i-1] + oc[n-1] - oc[i];
                } else {    // even
                    oCount = oc[i-1] + ec[n-1] - ec[i];
                    eCount = ec[i-1] + oc[n-1] - oc[i];
                }
            }
            
            
if(eCount == oCount) {
                ans++;
            }
        }
        
        return ans;
    }
}
