class Solution {
    public int waysToMakeFair(int[] nums) {
        int[] oddSums = new int[nums.length + 1];
        int[] evenSums = new int[nums.length + 1];
        for (int i = 0; i < nums.length; i++) {
            if (i % 2 == 0) {
                evenSums[i + 1] = evenSums[i] + nums[i];
                oddSums[i + 1] = oddSums[i];
            } else {
                evenSums[i + 1] = evenSums[i];
                oddSums[i + 1] = oddSums[i] + nums[i];
            }
        }
        
        int count = 0;
        for (int i = 1; i <= nums.length; i++) {
            if (oddSums[i - 1] + evenSums[evenSums.length - 1] - evenSums[i] == evenSums[i - 1] + oddSums[oddSums.length - 1] - oddSums[i])
                count++;
        }
        return count;
    }
}
