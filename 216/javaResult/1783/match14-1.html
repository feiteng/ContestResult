class Solution {
    public int waysToMakeFair(int[] nums) {
        int l[][] = new int[nums.length][2];
        int r[][] = new int[nums.length][2];
        
        int o = 0, e = 0;
        
for(int i=0;i<nums.length;i++){
            l[i][0] = o;
            l[i][1] = e;
            if(i%2==1){
                o += nums[i];
            } else {
                e += nums[i];
            }
        }
        o = e = 0;
        for(int i=nums.length-1;i>=0;i--){
            r[i][0] = o;
            r[i][1] = e;
            
if(i%2==1){
                o += nums[i];
            } else {
                e += nums[i];
            }            
        }
        int c = 0;
        for(int i=0;i<nums.length;i++){
            
if(l[i][0]+r[i][1] == l[i][1]+r[i][0]){
                c++;
            }
        }
        return c;
    }
}
