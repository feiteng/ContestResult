class Solution {
    public int waysToMakeFair(int[] A) {
        int n = A.length;
        int[] even = new int[n+1], odd = new int[n+1];
        
for(int i=0;i<n;i++){
            if(i%2==0){
                even[i+1] = even[i] + A[i];
                odd[i+1] = odd[i];
            }else {
                even[i+1] = even[i];
                odd[i+1] = odd[i] + A[i];
            }
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            if(even[i]-even[0]+odd[n]-odd[i+1] == even[n]-even[i+1]+odd[i]-odd[0]) ans++;
        }
        return ans;
    }
}
