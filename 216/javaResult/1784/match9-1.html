class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a, b) -> a[1]-a[0]+b[0]-b[1]);
        int curr = 0;
        for(int i = 0; i < tasks.length; i++){
            curr = Math.max(tasks[i][1], tasks[i][0]+curr);
        }
        return curr;
    }
}
