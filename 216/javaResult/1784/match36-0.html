class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a, b) -> ((b[1] - b[0] == a[1] - a[0]) ? (b[1] - a[1]) : (b[1] - b[0] - a[1] + a[0])));
        int need = 0;
        int left = 0;
        for (int[] task : tasks) {
            if (left < task[1]) {
                need += task[1] - left;
                left = task[1];
            }
            left -= task[0];
        }
        return need;
    }
}
