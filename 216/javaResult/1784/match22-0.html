class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks , (a , b)->(b[1] - b[0] - a[1] + a[0]));
        int ans = 0;
        int cur = 0;
        for(int[] t : tasks){
            if(cur < t[1]){
                ans += t[1] - cur;
                cur = t[1];
            }
            cur -= t[0];
        }
        return ans;
    }
}
