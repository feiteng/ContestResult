class Solution {
    public int minimumEffort(int[][] tasks) {
 Arrays.sort(tasks, Comparator.comparingInt(x -> x[0] - x[1]));
        int max = 0, rm = 0;
        for (int[] task : tasks) {
            if (rm < task[1]) {
                max += task[1] - rm;
                rm = task[1];
            }
            rm -= task[0];
        }
        return max;
    }
}
