class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (t1, t2) -> (t2[1] - t2[0]) - (t1[1] - t1[0]));
        
        int ans = 0, curr = 0;
        for (int[] task : tasks) {
            if (curr < task[1]) {
                ans += task[1] - curr;
                curr = task[1];
            }
            curr -= task[0];
        }
        return ans;
    }
}
