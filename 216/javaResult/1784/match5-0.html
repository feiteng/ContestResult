class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a,b) -> (b[1] - b[0]) - (a[1] - a[0]));
        int res = tasks[0][1];
        int cur = tasks[0][1] - tasks[0][0];
        for(int i = 1; i < tasks.length; i++){
            if(cur < tasks[i][1]){
                res += tasks[i][1] - cur;
                cur = tasks[i][1] - tasks[i][0];
            }else{
                cur -= tasks[i][0];
            }
        }
        return res;
    }
}
