class Solution {
    public int minimumEffort(int[][] tasks) {
        //排序
        Arrays.sort(tasks, (t1, t2) -> t1[1] - t1[0] == t2[1] - t2[0] ? t2[1] - t1[1] : t2[1] - t2[0] - t1[1] + t1[0]);
        
        int res = 0;
        int cur = 0;
        for(int[] t : tasks){
            if(cur < t[1]){
                res += t[1] - cur;
                cur = t[1];
            }
            cur -= t[0];
        }
        return res;
    }
}
