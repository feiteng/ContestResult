class Solution {
    public int minimumEffort(int[][] arr) {
        Arrays.sort(arr, (a, b) -> b[1] - b[0] - a[1] + a[0]);
        int res = arr[0][1];
        int temp = res - arr[0][0];
        for(int i = 1; i < arr.length; i++){
            if(temp >= arr[i][1]) {
                temp -= arr[i][0];
            }
            else{
                res += (arr[i][1] - temp);
                temp = arr[i][1];
                temp -= arr[i][0];
            }
        }
        
        return res;
    }
}
