class Solution {
    public int minimumEffort(int[][] tasks) {
        // Arrays.sort(tasks, (a,b)->b[1]-a[1]==0 ? b[0]-a[0] : b[1]-a[1]);
        Arrays.sort(tasks, (a,b)-> /*(b[1]-b[0])-(a[1]-a[0])==0? b[1]-a[1]==0 ? b[0]-a[0] : b[1]-a[1] : */(b[1]-b[0])-(a[1]-a[0]));
        // for(int i=0; i<tasks.length; i++)
        //     System.out.println(Arrays.toString(tasks[i]));
        // return 0;
        
        int cur = 0, req=0;
        for(int i=0; i<tasks.length; i++)
        {
            // System.out.println("cur: "+cur);
            if(cur<tasks[i][1])
            {
                req += tasks[i][1] - cur;
                cur = tasks[i][1];
            }
            // System.out.println("cur: "+cur+" req: "+req);
            cur -= tasks[i][0];
        }
        return req;
        
//         Arrays.sort(tasks, (a,b)-> /*(b[1]-b[0])-(a[1]-a[0])==0? b[1]-a[1]==0 ? b[0]-a[0] : b[1]-a[1] : */(b[1]-b[0])-(a[1]-a[0]));
        
    }
}
