class Solution {
    /*
    Input: tasks = [[1,3],[2,4],[10,11],[10,12],[8,9]]
                      2     2      1       2      1
    Output: 32
    12 + (11 - 2) + (9 - 1) + (4 - 1) + (3 - 2)
    =12 + 9 + 8 + 3 + 1
    =33
    
    Input: tasks = [[1,2],[2,4],[4,8]]
                      1     2     4
    Output: 8
    8 + 
    */
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a,b) -> (b[1]-b[0])-(a[1]-a[0]));
        int res = 0, remain = 0, n = tasks.length;
        for (int[] t: tasks){
            // System.out.println(t[1] - remain);
            res += 
Math.max(0, t[1] - remain);
            remain = t[1] - t[0] + Math.max(0, remain - t[1]);
            // System.out.println(res + " " + remain);
        }
        // res -= tasks[n-1][1] - tasks[n-1][0];
        return res;
    }
}
