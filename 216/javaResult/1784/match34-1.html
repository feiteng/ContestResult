class Solution {
    public int minimumEffort(int[][] tasks) {
        PriorityQueue<int[]> pq=new PriorityQueue<>((x,y)->x[0]+y[1]-(y[0]+x[1]));
        for(int[] task:tasks){
            pq.offer(task);
        }
        
        
int res=0, cur=0;
        while(!pq.isEmpty()){
            int[] task=pq.poll();
            if(cur<task[1]){
                res+=task[1]-cur;
                cur=task[1];
            }
            cur-=task[0];
        }
        
        return res;
    }
}
