class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a, b) -> {
            if (a[1] - a[0] == b[1] - b[0]) return b[1] - a[1];
            return (b[1] - b[0]) - (a[1] - a[0]);
        });
        int n = tasks.length;
        int result = tasks[n-1][1];
        
for (int i = n-2; i >= 0; i--) {
            result = Math.max(tasks[i][1], result + tasks[i][0]);
        }
        return result;
    }
}
