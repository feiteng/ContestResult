class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a, b) -> (a[1] - a[0]) == (b[1] - b[0]) ? a[0] - b[0] : (a[1] - a[0]) - (b[1] - b[0]));
        // for (int[] p: tasks) System.out.println(Arrays.toString(p));
        int sum = tasks[0][1];
        int len = tasks.length;
        for (int i = 1; i < len; i++) {
            sum += tasks[i][0];
            if (sum < tasks[i][1]) {
                sum = tasks[i][1];
            }
        }
        return sum;
    }
}
