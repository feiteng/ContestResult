class Solution {
    public int minimumEffort(int[][] tasks) {
        
        Arrays.sort(tasks, (a,b) -> a[1]-a[0] != b[1]-b[0] ? (b[1]-b[0]) - (a[1]-a[0]) : b[1] - a[1]);
        
        int start = 0;
        int cur = 0;
        
        for (int[] i : tasks) {
            if (cur >= i[1]) {
                cur -= i[0];
            } else {
                start += i[1] - cur;
                
cur = i[1] - i[0];
            }
        }
        
        return start;
    }
}
