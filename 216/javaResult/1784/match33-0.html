class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                if(o1[1] - o1[0] != o2[1] - o2[0]) return (o2[1] - o2[0]) - (o1[1] - o1[0]);
                return o2[1] - o1[1];
            }
        });
        int ret = 0,cur = 0;
        for (int i = 0; i < tasks.length; i++) {
            if(cur < tasks[i][1]){
                ret += tasks[i][1] - cur;
                cur = tasks[i][1];
            }
            cur -= tasks[i][0];
        }
        return ret;
    }
}
