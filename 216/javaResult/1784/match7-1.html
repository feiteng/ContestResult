class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a, b) -> ((a[1]-a[0])-(b[1]-b[0])));
        int ans = tasks[0][1];
        for(int i = 1; i < tasks.length; i++){
            ans = Math.max(ans + tasks[i][0], tasks[i][1]);
        }
        return ans;
    }
}
