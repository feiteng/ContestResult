class Solution {
    public int minimumEffort(int[][] tasks) {
        int res = 0;
        int cur = 0;
        Arrays.sort(tasks, (a, b) -> (b[1] - b[0]) - (a[1] - a[0]));
        
for (int[] t : tasks) {
            if (cur < t[1]) {
                int diff = t[1] - cur;
                cur += diff;
                res += diff;
            }
            cur -= t[0];
        }
        return res;
    }
}
