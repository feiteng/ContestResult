class Solution {
    public int minimumEffort(int[][] tasks) {
        int res = 0, remain = 0;
        Arrays.sort(tasks, (a, b) -> (b[1] - b[0]) - (a[1] - a[0]));
        for (int[] t : tasks) {
            if (remain < t[1]) {
                res += t[1] - remain;
                remain = t[1];
            }
            remain -= t[0];
        }
        return res;
    }
}
