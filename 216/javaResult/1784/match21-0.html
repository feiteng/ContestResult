class Solution {
    public int minimumEffort(int[][] tasks) {
        //Arrays.sort(tasks, (a,b)->(b[1]-b[0]==a[1]-a[0]? b[1]-a[1] : b[1]-b[0]-a[1]+a[0]));
        Arrays.sort(tasks, (a,b)->(b[1]-b[0]-a[1]+a[0]));
        
        int e=0, balance=0;
        for(int[] t:tasks){
            if(balance<t[1]){
                e+=t[1]-balance;
                balance=t[1];
            }
            balance-=t[0];
        }
        return e;
    }
}
