class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a,b) -> (b[1]-b[0] -a[1] + a[0]));
        int sum = 0;
        int left = 0;
        for(int[] t : tasks){
            
if(left < t[1]){
                sum += (t[1] - left);
                left = t[1] - t[0];
            }else{
                left -= t[0];
            }
        }
        return sum;
    }
}
