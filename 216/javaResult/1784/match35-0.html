import java.util.Arrays;

class Solution {
	public int minimumEffort(int[][] tasks) {
		Arrays.sort(tasks, (a, b) -> b[1] - b[0] - (a[1] - a[0]));
		int sum = 0;
		int remain = 0;
		for (int[] task : tasks) {
			int a = task[0];
			int b = task[1];
			int gap = b - remain;
			if (gap > 0) {
				sum += gap;
				remain = b - a;
			} else {
				remain -= a;
			}
		}
		return sum;
	}
}
