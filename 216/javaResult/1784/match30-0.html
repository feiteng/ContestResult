class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (a, b) -> {
            if (a[1] - a[0] != b[1] - b[0]) return (a[1] - a[0]) - (b[1] - b[0]);
            return a[1] - b[1];
        });
        int res = 0;
        for (int[] t: tasks) {
            res = Math.max(t[1] - t[0], res);
            res += t[0];
            // System.out.println(t[0] + " " + t[1] + " " + res);
        }
        return res;
    }
}
