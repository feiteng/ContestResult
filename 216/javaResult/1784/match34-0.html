class Solution {
    public int minimumEffort(int[][] tasks) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> a[1] - a[0] - (b[1] - b[0]));
        
        for(int[] task : tasks) {
            pq.add(task);
        }
        
        int out = 0;
        
        while(!pq.isEmpty()) {
            int[] nxt = pq.poll();
            
            out += nxt[0];
            
            if(out < nxt[1]) {
                out = nxt[1];
            }
        }
        
        return out;
    }
}
