class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, Comparator.comparingInt(m -> m[0] - m[1]));

        int r = 0;
        int t = 0;

        for (int[] task : tasks) {
            if (t < task[1]) {
                r += task[1] - t;
                t = task[1];
            }

            t -= task[0];
        }

        return r;
    }
}
