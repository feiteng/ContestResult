class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks, (int[] a, int[] b) -> (b[1] - b[0]) - (a[1] - a[0]));
        // for (int i = 0; i < tasks.length; i++) {
        //     System.out.println(tasks[i][0] + ", " + tasks[i][1]);
        // }
        int result = tasks[0][1];
        int left = result - tasks[0][0];
        for (int i = 1; i < tasks.length; i++) {
            if (tasks[i][1] <= left) {
                left -= tasks[i][0];
            } else {
                result += tasks[i][1] - left;
                left = tasks[i][1] - tasks[i][0];
            }
        }
        
        return result;
    }
}
