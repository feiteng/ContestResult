class Solution {
    public int minimumEffort(int[][] tasks) {
        Arrays.sort(tasks,(a, b) -> (b[1]-b[0]) - (a[1]-a[0]));
        int energy = 0;
        int total = 0;
        for(int i = 0; i < tasks.length; i++){
            if(energy < tasks[i][1]){
                total += tasks[i][1] - energy;
                energy = tasks[i][1];
            }
            energy -= tasks[i][0];
        }
        return total;
    }
}
