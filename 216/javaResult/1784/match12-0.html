class Solution {
    public int minimumEffort(int[][] tasks) {
        
        Arrays.sort(tasks, (a, b) -> ((b[1] - b[0]) - (a[1] - a[0])));
        
        // for (int[] arr : tasks) System.out.println(Arrays.toString(arr));
        
        int totalEnergy = 0;
        int currEnergy = 0;
        
        for (int i = 0; i < tasks.length; i++) {
            
            if (currEnergy < tasks[i][1]) {
                totalEnergy += (tasks[i][1] - currEnergy);
                currEnergy = tasks[i][1];
            }
            
            currEnergy -= tasks[i][0];
            
        }
        
        return totalEnergy;
        
    }
}
