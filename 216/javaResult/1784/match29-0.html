class Solution {
    public int minimumEffort(int[][] tasks) {
        int init = 0,res = 0;
        Arrays.sort(tasks,(o1,o2)->o2[1]-o2[0]-(o1[1]-o1[0]));
        for(int i=0;i<tasks.length;i++) {
            // System.out.println(Arrays.toString(tasks[i]));
            if (res < tasks[i][1]) {
                init += tasks[i][1] - res;
                res = tasks[i][1];
            }
            res -= tasks[i][0];
        }
        return init;
    }
}
