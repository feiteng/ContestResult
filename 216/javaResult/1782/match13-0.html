class Solution {
    public String getSmallestString(int n, int k) {
        char[] re = new char[n];
        for(int i = 0; i < n; i++)
        {
            int upperBound = 26 * (n - 1 - i);
            int lowerBound = Math.max(1, k - upperBound);
            re[i] = (char)('a' + lowerBound - 1);
            k -= lowerBound;
        }
        return String.valueOf(re);
    }
}
