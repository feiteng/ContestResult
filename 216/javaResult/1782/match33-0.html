class Solution {
  public String getSmallestString(final int N, final int K) {
    int availables = K;
    int alphachoice = K / 26;
    StringBuilder sb = new StringBuilder(N);
    while (N > K - (25 * alphachoice))
      alphachoice--;
    availables -= alphachoice * 26;
    for (int i = 0; i + 1 + alphachoice < N; ++i, availables--)
      sb.append('a');
    sb.append((char) ('a' + availables - 1));
    for (int i = 0; i != alphachoice; i++)
      sb.append('z');
    return sb.toString();
  }
}
