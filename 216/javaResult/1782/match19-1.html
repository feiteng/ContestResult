class Solution {
    public String getSmallestString(int n, int k) {
        int extra = k-n;
        int numberofZ = (k-n) / 25;        
        char middle = (char)('a' + (k-n)%25);
        int numberofA = n - numberofZ - 1;
        
        StringBuilder sb = new StringBuilder();
        for (int i=0; i<numberofA; i++){
            sb.append('a');
        }
        sb.append(middle);
        for (int i=0; i<numberofZ; i++){
            sb.append('z');
        }
        
        return sb.toString();
    }
}
