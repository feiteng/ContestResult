class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; ++i) {
            int num = Math.min(26, k - (n - i - 1));
            sb.append((char)(num - 1 + 'a'));
            k -= num;
        }
        return sb.reverse().toString();
    }
}
