class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0;i < n;i++) {
            int cur = k - 26 * (n - i - 1);
            if (cur > 0) {
               sb.append((char)('a' + cur - 1)); 
               k -= cur;
            } else {
                sb.append('a');
                k -= 1;
            }
        }
        return sb.toString();
    }
}
