class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for (int i = 1; i <= n; i++) {
            for (int ii = 1; ii <= 26; ii ++) {
                if (k - ii <= (n - i) * 26) {
                    sb.append((char) (ii - 1 + 'a'));
                    k -= ii;
                    break;
                }
            }
        }
        return sb.toString();
    }
}
