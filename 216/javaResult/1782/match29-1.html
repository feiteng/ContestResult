class Solution {
    public String getSmallestString(int n, int k) {
        char[] arr = new char[n];
        Arrays.fill(arr, 'a');
        k -= n;
        int i = n - 1;
        while (k > 0) {
            if (k > 25) {
                arr[i--] += 25;
                k -= 25;
            }
            else {
                arr[i--] += k;
                k = 0;
            }
        }
        return new String(arr);
    }
}
