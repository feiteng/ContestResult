class Solution {
    public String getSmallestString(int n, int k) {
        char[] s = new char[n];
        for (int i = n - 1; i >= 0; i--) {
            int v = Math.min(26, k - i);
            k -= v;
            s[i] = (char) (v - 1 + 'a');
        }
        return new String(s);
    }
}
