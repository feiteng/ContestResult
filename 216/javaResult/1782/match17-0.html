class Solution {
    public String getSmallestString(int n, int k) {
StringBuilder builder = new StringBuilder();
        for (int i = 0; i < n; ++i) {
            int offset = k - (n - i - 1) * 26;
            if (offset <= 1) {
                builder.append('a');
                k -= 1;
            }
            else {
                builder.append((char)('a' + offset - 1));
                k -= offset;
            }
        }
        return builder.toString();
    }
}
