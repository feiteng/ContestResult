class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < n; ++i) {
            for (int j = 1; j <= 26; ++j) {
                if ((n - i - 1) * 26 >= k - j) {
                    sb.append((char)(j - 1 + 'a'));
                    k -= j;
                    break;
                }
            }
        }
        return sb.toString();
    }
}
