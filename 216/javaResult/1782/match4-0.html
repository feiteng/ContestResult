class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for(int i = 1; i <= n; i++){
            //确定第一个字符最小能是什么
            for(int j = 1; j <= 26; j++){
                if(j + 26 *(n - i) >= k){
                    sb.append((char)(j + 'a' - 1));
                    k -= j;
                    break;
                }
            }
        }
        return sb.toString();
    }
}
