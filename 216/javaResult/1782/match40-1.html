class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            int target = (n-i-1) * 26;
            
for (int j = 1; j <= 26; j++) {
                if (k - j <= target) {
                    sb.append((char)('a' + (j-1)));
                    k -= j;
                    break;
                }
            }
        }
        return sb.toString();
    }
}
