class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder res = new StringBuilder();
        for(int i = 0; i < n; i++){
            int left = n - i - 1;
            if(k <= left * 26){
                res.append("a");
                k -= 1;
            }
            else{
                int cur = k - left * 26;
                res.append((char)('a' - 1 + cur));
                k -= cur;
            }
        }
        return res.toString();
    }
}
