class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for(int i=0; i<n; i++) {
            int leftMax = (n-1-i)*26;
            int curVal = Math.max(1, k-leftMax);
            char cur = (char)('a'+curVal-1);
            sb.append(cur);
            k-=curVal;
        }
        return sb.toString();
    }
}
