class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        while (n-- > 0) {
            int cur = Math.max(1, k - n * 26);
            sb.append((char)('a' - 1 + cur));
            k -= cur;
        }
        return sb.toString();
    }
}
