class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder stringBuilder = new StringBuilder();
        for (int i = 0; i < n; i++) {
            int left = n - i - 1;
            //System.out.println("i = " + i + " s = " + stringBuilder + " left = " + left + " k = " + k);

            if(left * 26 >= k){
                stringBuilder.append('a');
                k--;
            }else{
                int now = k - left * 26;
                //System.out.println("now = " + now + " k = " + k);
                stringBuilder.append((char)('a' + now - 1));
                k -= now;
            }
        }
        return stringBuilder.toString();
    }
}
