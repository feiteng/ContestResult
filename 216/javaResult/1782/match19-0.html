class Solution {
    public String getSmallestString(int n, int k) {
        int m = k - n;
        int z = m / 25;
        int a = n - 1 - z;
        int c = m % 25;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < a; i++) {
            sb.append('a');
        }
        sb.append((char)(c + 'a'));
        for (int i = 0; i < z; i++) {
            sb.append('z');
        }

        return sb.toString();

    }
}
