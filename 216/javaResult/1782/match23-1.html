class Solution {
    public String getSmallestString(int n, int k) {
        char[] charr = new char[n];
        Arrays.fill(charr, 'a');
        k -= n;
        int idx = n - 1;
        while (k > 0) {
            //System.out.println(k);
            if (k > 25) {
                charr[idx] = 'z';
                k -= 25;
            } else {
                charr[idx] += k;
                k = 0;
            }
            idx--;
        }
        return new String(charr);
    }
}
