class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        while (n > 0) {
            if (n * 26 <= k) {
                sb.append('z');
                k -= 26;
            } else if ((k - 1) <= (n - 1) * 26){
                sb.append('a');
                k -=1;
            } else {
                int cur = k - (n - 1) * 26;
                sb.append((char)('a' - 1 + cur));
                k -= cur;
            }
            n--;
        }
        return sb.toString();
    }
}
