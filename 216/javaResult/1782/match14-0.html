class Solution {
    public String getSmallestString(int n, int k) {
        
        
        StringBuilder sb = new StringBuilder();
        
        while (n > 0) {
            
            // 71 and 3
            // 78
            
            int i = Math.max(1, k - (26 * (n - 1)));
            // System.out.println(i);
            
            sb.append((char)('a' + (i - 1)));
            
            k -= i;
            n -= 1;
        }
        
        return sb.toString();
        
    }
}
