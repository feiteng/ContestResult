class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            int left = (n-i-1) * 26;
            if (k-left <= 1) {
                sb.append('a');
                k -= 1;
            } else {
                sb.append((char)('a'+ k-left-1));
                k -= (k-left);
            }
        }
        return sb.toString();
    }
}
