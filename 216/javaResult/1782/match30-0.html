class Solution {
    public String getSmallestString(int n, int k) {
        char[] cc = new char[n];
        Arrays.fill(cc,'a');
        k -= n;
        int index = n - 1;
        while (k > 0) {
            if (k > 25) {
                k -= 25;
                cc[index--] = 'z';
            }
            else {
                cc[index--] += k;
                k = 0;
            }
        }
        return new String(cc);
    }
}
