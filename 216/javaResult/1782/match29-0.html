class Solution {
    public String getSmallestString(int n, int k) {
        char[] res = new char[n];
        Arrays.fill(res, 'a');
        k -= n;
        int idx = n - 1;
        while (k > 0) {
            if (k >= 25) {
                res[idx] += 25;
                k -= 25;
            } else {
                res[idx] += k;
                k = 0;
            }
            idx--;
        }
        return new String(res);
    }
}
