class Solution {
    
    // 1. value equals to k
    // 2. length is n
    
    // greedy
    // as much of small char as possible
    
    public String getSmallestString(int n, int k) {
        
        k -= n; // minus one for all char
        
        int cnt = k / 25; // cnt of char z
        int idx = k % 25;
        // System.out.println(n + " " + cnt + " " + idx);
        
        StringBuilder sb = new StringBuilder();
        
for(int i = 0; i < n - cnt - 1; i++) sb.append('a');
        sb.append((char)('a' + idx));
        for(int i = 0; i < cnt; i++) sb.append('z');
        
        return sb.toString();
    }
}
