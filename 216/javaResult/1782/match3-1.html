class Solution {
    public String getSmallestString(int n, int k) {
        StringBuilder sb = new StringBuilder();


        for (int i = 0; i < n; i++) {
            for (int j = 1; j <= 26; j++) {
                if (j + (n - i - 1) * 26 >= k) {
                    k -= j;
                    sb.append((char) (j - 1 + 'a'));
                    break;
                }
            }
        }
        //sb.append((char) (k - 1 + 'a'));
        return sb.toString();
    }
}
