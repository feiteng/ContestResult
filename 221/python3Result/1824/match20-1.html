class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        n = len(apples)
        pq = list()
        heapq.heapify
(pq)
        ans = 0
        for i in range(n):
            if apples[i] > 0:
                heapq.heappush(pq, (i + days[i], apples[i]))
            while pq and pq[0][0] <= i:
                heapq.heappop(pq)
            if pq:
                day, remaining = heapq.heappop(pq)
                ans += 1
                if remaining > 1:
                    heapq.heappush(pq, (day, remaining - 1))
        i = n
        while pq:
            while pq and pq[0][0] <= i:
                heapq.heappop(pq)
            if pq:
                day, remaining = heapq.heappop(pq)
                if remaining > 1:
                    heapq.heappush(pq, (day, remaining - 1))
                ans += 1
                i += 1
        return ans
