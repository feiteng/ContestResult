class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        res = 0
        pq = []
        for i in range(len(apples)):
            if apples[i] > 0:
                heapq.heappush(pq,[i+days[i], apples[i]])
            while pq and pq[0][0] <= i:
                heapq.heappop(pq)
            if pq:
                res += 1
                pq[0][1] -= 1
                if pq[0][1] == 0:
                    heapq.heappop(pq)
        cur = len(apples)
        while pq:
            while pq and pq[0][0] <= cur:
                heapq.heappop(pq)
            if pq:
                res += 1
                pq[0][1] -= 1
                if pq[0][1] == 0:
                    heapq.heappop(pq)
            cur += 1
        return res
