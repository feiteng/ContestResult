class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        h = []
        result = 0
        for i, apple in enumerate(apples):
            day = days[i] + i
            
if apple:
                heapq.heappush(h, (day, apple))
            while h and h[0][0] <= i:
                heapq.heappop(h)
            if h:
                day, apple = heapq.heappop(h)
                result += 1
                if apple - 1:
                    heapq.heappush(h, (day, apple - 1))
            # print(i, result, h)
        i = len(days)
        
while h:
            while h and h[0][0] <= i:
                heapq.heappop(h)
            if h:
                day, apple = heapq.heappop(h)
                result += 1
                if apple - 1:
                    heapq.heappush(h, (day, apple - 1))
            i += 1
            # print(i, result, h)
        return result
