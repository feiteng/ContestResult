class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        q = []
        ans = 0
        
        for day, ap in enumerate(apples):
            if ap:
                heapq.heappush(q, (day + days
[day], ap))
            while q and q[0][0] <= day:
                heapq.heappop(q)
            if q:
                n_day, n_ap = heapq.heappop(q)
                ans += 1
                if n_ap > 1:
                    heapq.heappush(q, (n_day, n_ap - 1))
        while q:
            day += 1
            
while q and q[0][0] <= day:
                heapq.heappop(q)
            if q:
                n_day, n_ap = heapq.heappop(q)
                ans += 1
                if n_ap > 1:
                    heapq.heappush(q, (n_day, n_ap - 1))
        return ans
