import heapq


class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        day = 0
        res = 0
        basket = []
        while basket != [] or day < len(apples):
            if day < len(apples):
                heapq.heappush(basket, [day + days[day], apples[day]])
            while basket != [] and (day >= basket[0][0] or basket[0][1] == 0):
                heapq.heappop(basket)
            if basket != []:
                rottingDay, numApples = heapq.heappop(basket)
                numApples -= 1
                res += 1
                heapq.heappush(basket, [rottingDay, numApples])
            day += 1
        return res
