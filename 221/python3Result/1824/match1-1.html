class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        
        picked = []
        count = 0
        
        for i in range(len(apples)):
            if apples[i] != 0:
                heapq.heappush(picked, [days[i] + i, apples[i]] )
            while picked and picked[0][0] <= i:
                heapq.heappop(picked)
            if picked:
                #print(i,'ate', picked[0][1])
                count += 1
                picked[0][1] -= 1
                if picked[0][1] == 0:
                    heapq.heappop(picked)
                
        i = len(apples)
        while picked:
            while picked and picked[0][0] <= i:
                heapq.heappop(picked)
            if picked:
                #print(i,'ate', picked[0][1])
                count += 1
                picked[0][1] -= 1
                if picked[0][1] == 0:
                    heapq.heappop(picked)
            i += 1
        return count
