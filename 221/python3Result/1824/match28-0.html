class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        n = len(apples)
        i = 0
        Q = []
        ans = 0
        while i<n or Q:
            if i<n:
                heapq.heappush(Q, [i+days[i], apples[i]])
            while Q and (Q[0][0] <= i or Q[0][1] == 0):
                heapq.heappop(Q)
            if Q:
                ans += 1
                Q[0][1] -= 1
                
            i += 1
        return ans
