class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        pq = []
        n = len(apples)
        res = 0
        
        i = 0
        
        while i < n or pq:
            if i < n and apples[i] > 0:
                heapq.heappush(pq, (i+days[i], apples[i]))
                
            while pq and pq[0][0] <= i:
                heapq.heappop(pq)

            if pq:
                ddl, num = heapq.heappop(pq)
                res += 1
                num -= 1
                if num > 0:
                    heapq.heappush(pq, (ddl, num))
 
            i += 1
            
        return res
