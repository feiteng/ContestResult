class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        n = len(apples)
        q = []
        res = 0
        for i, (x, y) in enumerate(zip(apples, days)):
            if x:
                heapq.heappush(q, (i+y, x))
            while q and i >= q[0][0]:
                heapq.heappop(q)
            if q:
                day, cnt = heapq.heappop(q)
                res += 1
                if cnt>1:
                    heapq.heappush(q, (day, cnt-1))
        while q:
            while q and n >= q[0][0]:
                heapq.heappop(q)
            if q:
                day, cnt = heapq.heappop(q)
                res += 1
                if cnt>1:
                    heapq.heappush(q, (day, cnt-1))
            n += 1
        return res
