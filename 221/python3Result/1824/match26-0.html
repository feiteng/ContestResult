class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        heap = []
        ans = 0
        for i, (a, d) in enumerate(zip(apples, days)):
            while heap and heap[0][0] <= i:
                heapq.heappop(heap)
            if a:
                heapq.heappush(heap, [i + d, a])
            if heap:
                ans += 1
                heap[0][1] -= 1
                if not heap[0][1]:
                    heapq.heappop(heap)
        i = len(apples)
        while heap:
            d, a = heapq.heappop(heap)
            incre = min(d - i, a)
            i += incre
            ans += incre
        return ans
