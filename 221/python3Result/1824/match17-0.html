class Solution:
    def eatenApples(self, A: List[int], D: List[int]) -> int:
        cnt = 0
        heap = []
        for i in range(len(A)):
            if A[i] > 0:
                heappush(heap, [i+D[i], A[i]])
            while heap and heap[0][0] <= i:
                heappop(heap)
            if heap:
                day, number = heappop(heap)
                number -= 1
                cnt += 1
                if number > 0:
                    heappush(heap, [day, number])
        cur = len(A)
        while heap:
            while heap and heap[0][0] <= cur:
                heappop(heap)
            if heap:
                day, number = heappop(heap)
                if day - cur >= number:
                    cur = number + cur
                    cnt += number
                else:
                    cnt += day - cur
                    cur = day
        return cnt
