"""
Apples
apple balance. Eat the ones that expire the earliest?
"""

class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        i = 0
        eaten = 0
        soonest = []
        while i < len(apples) or soonest:
            if i < len(apples):
                heapq.heappush(soonest, (i + days[i] - 1, apples[i]))
            while soonest and (soonest[0][1] == 0 or soonest[0][0] < i):
                heapq.heappop(soonest)
            if soonest:
                exp, ct = heapq.heappop(soonest)
                ct -= 1
                heapq.heappush(soonest, (exp, ct))
                eaten += 1
            i += 1
        return eaten
