class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        res = 0
        n = len(apples)
        heap = []
        for i in range(n):
            if apples[i] != 0 and days[i] != 0:
                heapq.heappush(heap, (i + days[i], apples[i]))
            while heap and heap[0][0] <= i:
                heapq.heappop(heap)
            if heap:
                res += 1
                day, apple = heapq.heappop(heap)
                if apple > 1:
                    heapq.heappush(heap, (day, apple - 1))

        i = n
        while heap:
            while heap and heap[0][0] <= i:
                heapq.heappop(heap)
            if heap:
                res += 1
                day, apple = heapq.heappop(heap)
                if apple > 1:
                    heapq.heappush(heap, (day, apple - 1))
            i += 1

        return res
