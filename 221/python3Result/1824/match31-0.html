class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        minHeap = []
        res = 0
        for i in range(len(apples)):
            if apples[i] != 0:
                heapq.heappush(minHeap, (i + days[i], apples[i]))
            while minHeap and minHeap[0][0] <= i:
                heapq.heappop(minHeap)
            if minHeap:
                rotten, num = heapq.heappop(minHeap)
                res += 1
                if num > 1:
                    heapq.heappush(minHeap, (rotten, num - 1))
        days = len(apples)
        while minHeap:
            while minHeap and minHeap[0][0] <= days:
                heapq.heappop(minHeap)
            if minHeap:
                rotten, num = heapq.heappop(minHeap)
                res += 1
                if num > 1:
                    heapq.heappush(minHeap, (rotten, num - 1))
            days += 1
        return res
