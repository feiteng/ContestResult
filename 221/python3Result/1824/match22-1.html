class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        n=len(apples)
        ans=0
        i=0
        h=[]
        while i<n or h:
            if i<n:
                heapq
.heappush(h,(i+days[i],apples[i]))
                
            while h and h[0][0]<=i:
                heapq.heappop(h)
            #print(i,h,'start')
            if h:
                a,b=heapq.heappop(h)
                b-=1
                ans+=1
                if b>0:
                    heapq.heappush(h,(a,b))
            #print(i,h)
            i+=1
        return ans
