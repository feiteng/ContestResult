class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        # greedy, use a heap
        # (minimum day, number)
        n = len(apples)
        h = []
        res = 0
        for i, c in enumerate(apples):
            # push
            if c > 0:
                heapq.heappush(h, (i + days[i] - 1, c))
            while h and h[0][0] < i:
                heapq.heappop(h)
                
            # get top apple
            if h:
                res += 1
                j, y = heapq.heappop(h)
                
if y > 1:
                    heapq.heappush(h, (j, y-1))
                    
        # now calculate how many apples we can eat after day n
        cur = n
        while h:
            j, y = heapq.heappop(h)
            # we can eat min(y, j-n)
            if j >= cur:
                res += min(y, j-cur+1)
                cur += min(y, j-cur+1) 
        return res
