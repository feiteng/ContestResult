class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        import heapq
        n = len(apples)
        h = []
        ans = 0
        day = 0
        while day < n or h:
            
while h and h[0][0] <= day:
                heapq.heappop(h)
            if day < n and apples[day] > 0:
                heapq.heappush(h,(day + days[day],apples[day]))
            if h:
                a,b = heapq.heappop(h)
                ans += 1
                b 
-= 1
                if b > 0:
                    heapq.heappush(h,(a,b))
            day += 1
        return ans
