class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        n = len(apples)
        q = []
        count = 0

        for i in range(n):
            while q and q[0][0] < i:
                heapq.heappop(q)
            if apples[i] > 0:
                heapq.heappush(q, (i + days[i] - 1, apples[i]))
            if q:
                rot_day, num = heapq.heappop(q)
                count += 1
                if num > 1 and rot_day > i:
                    heapq.heappush(q, (rot_day, num-1))
        
        i = n
        while q:
            while q and q[0][0] < i:
                heapq.heappop(q)
            if q:
                rot_day, num = heapq.heappop(q)
                count += 1
                if num > 1 and rot_day > i:
                    heapq.heappush(q, (rot_day, num-1))
            i += 1
        return count
