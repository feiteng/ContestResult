class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        tmp = []
        ans = 0
        for i in range(len(apples)):
            if apples
[i]:
                heapq.heappush(tmp, [i + days[i], apples[i]])
            while tmp:
                day, count = heapq.heappop(tmp)
                if day > i:
                    ans 
+= 1
                    count -= 1
                    if count >= 1:
                        heapq.heappush(tmp, [day, count])
                    break
        i = len(apples)
        while tmp:
            day, count = heapq.heappop(tmp)
            if day > i:
                ans += 1
                count -= 1
                if count >= 1:
                    heapq.heappush(tmp, [day, count])
                i += 1
        return ans
