class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        day = 0
        output = 0

        heap = []
        for i in range(len(apples)):
            if apples[i] != 0 and days[i] != 0:
                heapq.heappush(heap, (i + days[i], apples[i]))
            while heap:
                d, a = heapq.heappop(heap)
                if d <= i: continue
                else:
                    output += 1
                    a -= 1
                    if a > 0:
                        heapq.heappush(heap, (d, a))
                    break
        
        i = len(apples)
        while heap:
            d, a = heapq.heappop(heap)
            if d > i:
                output += 1
                a -= 1
                if a > 0:
                    heapq.heappush(heap, (d, a))
                i += 1
        
        return output
