class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        import bisect
        stack = []
        n = len(apples)
        res = 0
        for i in range(n):
            if apples[i]:
                bisect.insort(stack, (i+days[i], apples[i]))
            while stack and i >= stack[0][0]:
                stack.pop(0)
            if stack:
                a,b = stack.pop(0)
                res += 1
                if b > 1:
                    bisect.insort(stack, (a,b-1))
        i = n
        while stack:
            while stack and i >= stack[0][0]:
                stack.pop(0)
            if stack:
                a,b = stack.pop(0)
                res += 1
                if b > 1:
                    bisect.insort(stack, (a,b-1))
            i += 1
        return res
