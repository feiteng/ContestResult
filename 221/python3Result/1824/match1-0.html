class Solution:
    def eatenApples(self, apples: List[int], d: List[int]) -> int:
        res = 0
        
        days = []
        for i in range(len(apples)):
            if apples[i]:
                heapq.heappush(days, [i + d[i], apples[i]])
            while days and days[0][0] <= i:
                heapq.heappop(days)
            if days:
                res += 1
                days[0][1] -= 1
                if not days[0][1]:
                    heapq.heappop(days)
        
        N = len(apples)
        while days:
            rot, app = heapq.heappop(days)
            cnt = min(app, max(0, (rot - N)))
            N += cnt
            res += cnt
        
        return res
