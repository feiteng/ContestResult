class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        n = len(apples)
        hq = []
        ans = 0
        for i in range(n):
            heapq.heappush(hq, [i+days[i], apples[i]])
            
while hq:
                d, c = heapq.heappop(hq)
                if d > i:
                    ans += 1
                    c -= 1
                    if c:
                        heapq.heappush(hq, [d, c])
                    break
        i = n
        
while hq:
            d, c = heapq.heappop(hq)
            if d > i:
                ans += 1
                c -= 1
                if c:
                    heapq.heappush(hq, [d, c])
                i += 1
        return ans
