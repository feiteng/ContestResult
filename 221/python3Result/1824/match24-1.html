class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        q = []
        n = len(days)
        D = max(days) + n + 2
        sol = 0
        for i in range(D):
            if i < n:
                heapq
.heappush(q, (i+days[i], apples[i]))
            while q and q[0][0]<=i:
                heapq.heappop(q)
            if q:
                dd, aa = heapq.heappop(q)
                sol += 1
                if aa > 1:
                    heapq.heappush(q, (dd, aa-1))
        return sol
