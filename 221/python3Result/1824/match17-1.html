from heapq import heappush, heappop


class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        
        heap = []
        total = 0
        for i in range(len(apples)):
            if apples[i] > 0:
                heappush(heap, (i+days[i], apples[i]))
            while heap and heap[0][0] <= i:
                heappop(heap)
            if heap:
                rottDay, appleCount = heappop(heap)
                total += 1
                
if appleCount > 1:
                    heappush(heap, (rottDay, appleCount-1))
        i = len(apples)
        while heap:
            while heap and heap[0][0] <= i:
                heappop(heap)
            if heap:
                rottDay, appleCount = heappop(heap)
                total += 1
                if appleCount > 1:
                    heappush(heap, (rottDay, appleCount-1))
            i += 1
        
        return total
