class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        heap = []
        ret = 0
        for i in range(int(1e5)):
            if i < len(apples):
                heapq.heappush(heap, (i+days[i], apples[i]))
            while heap and heap[-1][0] < i+1:
                heapq.heappop(heap)
            # print(heap)
            if heap:
                cover_day, n_apple = heapq.heappop(heap)
                ret += 1
                if n_apple - 1 > 0:
                    heapq.heappush(heap, (cover_day, n_apple-1))

        
        return ret
