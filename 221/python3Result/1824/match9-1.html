class Solution:
    def eatenApples(self, apples: List[int], days: List[int]) -> int:
        ans = 0
        stack = []
        for i in range(len(apples)):
            if apples[i]:
                heapq.heappush(stack, (i + days[i], apples[i]))
            while stack and stack[0][0] <= i:
                heapq.heappop(stack)
            if stack:
                ans += 1
                day, apple = heapq.heappop(stack)
                
if apple > 1:
                    heapq.heappush(stack, (day, apple - 1))
        date = len(apples)
        while stack:
            while stack and stack[0][0] <= date:
                heapq.heappop(stack)
            if stack:
                ans += 1
                day, apple = heapq.heappop(stack)
                if apple > 1:
                    heapq.heappush(stack, (day, apple - 1))
                date += 1
        return ans
