class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums
.sort()
        q = sorted(enumerate(queries), key=lambda x:x[1][1])
        n = len(q)
        answer = [0] * n
        
        trie = {}
        for i, (x, m) in q:
            while nums and nums[0] <= m:
                p = nums.pop(0)
                z = trie
                for k in range(31, -1, -1):
                    kk = 0 if p & (1 << k) == 0 else 1
                    z.setdefault(kk, {})
                    z = z[kk]
            z = trie
            ret = 0
            for k in range(31, -1, -1):
                kk = 0 if x & (1 << k) == 0 else 1
                if 1 - kk in z:
                    z = z[1 - kk]
                    ret = ret * 2 + 1
                elif kk in z:
                    z = z[kk]
                    ret = ret * 2
                else:
                    ret = -1
                    break
            answer[i] = ret
        return answer
