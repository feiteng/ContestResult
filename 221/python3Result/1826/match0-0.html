Trie = lambda: defaultdict(Trie)


class Solution:
    def maximizeXor(self, A, queries):
        A.sort()
        for i, query in enumerate(queries):
            query.append(i)
        queries.sort(key=lambda qu: qu[1])

        def insert(x):
            cur = trie
            for i, c in enumerate(bin(x)[2:].zfill(32)):
                cur = cur[int(c)]

        def query(x):
            ans = 0
            cur = trie
            for i, c in enumerate(bin(x)[2:].zfill(32)):
                d = int(c)
                want = d ^ 1
                if want in cur:
                    cur = cur[want]
                    ans = 2 * ans + want
                elif (want ^ 1) in cur:
                    cur = cur[want ^ 1]
                    ans = 2 * ans + (want ^ 1)
                else:
                    return -1

            return ans

        trie = Trie()
        ans = [-1] * len(queries)
        i = 0
        for x, limit, qid in queries:
            while i < len(A) and A[i] <= limit:
                insert(A[i])
                i += 1

            ans[qid] = query(x) ^ x if query(x) != -1 else -1

        return ans
