class XORNode:
    
    def __init__(self):
        self.one = None
        self.zero = None

class XORTree:
    def __init__(self):
        self.root = XORNode()
    
    
    def add(self,x):
        bits = 32
        node = self.root
        while bits >= 0:
            
if x & (1 << (bits)):
                if node.one is None:
                    node.one = XORNode()
                
                node = node.one
            else:
                if node.zero is None:
                    node.zero = XORNode()
                
                node = node.zero
            bits-=1
    
    def maxXor(self,x):
        bits = 32
        node = self.root
        val = 0
        while bits>=0:
            if x & (1 << (bits)):
                if node.zero:
                    val |= (1<<bits)
                    node = node.zero
                else:
                    node = node.one
            else:
                if node.one:
                    val |= (1<<bits)
                    node = node.one
                else:
                    node = node.zero
            bits-=1
        return val
                    
                    


class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        
        que = []
        for i, l in enumerate(queries):
            que.append((l[0],l[1],i))
        
        que.sort
(key=lambda x: x[1])
        tree = XORTree()
        sol = [-1]*len(queries)
        j = 0
        for x,m,i in que:
            while j<len(nums) and nums[j]<=m:
                tree.add(nums[j])
                j+=1
            if j==0:
                sol[i] = -1
            else:
                sol[i] = tree.maxXor(x)
        
        return sol
