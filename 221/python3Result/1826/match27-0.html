class Node:
    def __init__(self):
        self.left_min=float("inf")
        self.right_min=float("inf")
        self.left=None
        self.right=None
        self.vals=[]
class Trie:
    def __init__(self):
        self.root=Node()
    def insert(self,val):
        base=1<<31
        cur=self.root
        while base!=0:
            if val&base==0:
                if not cur.left:
                    cur.left=Node()
                    cur.left_min=val
                cur.vals.append(val)
                cur=cur.left
            else:
                if not cur.right:
                    cur.right=Node()
                    cur.right_min=val
                cur.vals.append(val)
                cur=cur.right
            base>>=1
        cur.vals.append(val)
            
        
class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        trie=Trie()
        res=[]
        for n in nums:
            trie.insert(n)
        for q in queries:
            x,bound=q
            base=1<<31
            cur=trie.root
            
            while base:
                if base&x:
                    if cur.left_min<=bound:
                        cur=cur.left
                    else:
                        if cur.right_min>bound:
                            res.append(-1)
                            break
                        cur=cur.right
                else:
                    if cur.right_min<=bound:
                        cur=cur.right
                    else:
                        if cur.left_min>bound:
                            res.append(-1)
                            break
                        cur=cur.left
                base>>=1
            if base==0:
                res.append(x^cur.vals[0])
        return res
