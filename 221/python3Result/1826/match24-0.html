class Solution:
    def maximizeXor(self, nums, queries):
        t = [[0, 0]]
        
        def insert(x):
            node = 0
            for i in reversed(range(31)):
                d = (x >> i) & 1
                if t[node][d] == 0:
                    t[node][d] = len(t)
                    t.append([0, 0])
                node = t[node][d]
        
        def query(x):
            ans = 0
            node = 0
            for i in reversed(range(31)):
                d = (x >> i) & 1
                if t[node][d ^ 1]:
                    ans += 1 << i
                    node = t[node][d ^ 1]
                else:
                    node = t[node][d]
            return ans

            if len(t) > 1:
                ans[i] = query(q)
        
        return ans
