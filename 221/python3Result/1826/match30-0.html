class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        # nums.sort()
        # def bisearch(low, high, val):
        #     if high - low == 1:
        #         if nums[high] <= val:
        #             return high
        #         if nums[low] <= val:
        #             return low
        #         return low - 1
        #     mid = (low + high) // 2
        #     if nums[mid] > val:
        #         return bisearch(low, mid, val)
        #     else:
        #         return bisearch(mid, high, val)
        
        
        # 看起来像字典树
        
        z = min(nums)
        
        res = []
        
        dict_tree = {}
        for i in nums:
            b = format(i, '#034b')[2:]
            d = dict_tree
            for c in b:
                if c not in d:
                    d[c] = [{},i]
                d[c][1] = min(d[c][1], i)
                d = d[c][0]
            d['0'] = ('#', '#')
            d['1'] = ('#', '#')
        
        for q, limit in queries:
            
            if z > limit:
                res.append(-1)
                continue
            
            b = format(q, '#034b')[2:]
            d = dict_tree
            s = []
            
            for c in b:
                if c == '1':
                    if '0' not in d:
                        s.append('0')
                        d = d['1'][0]
                    else:
                        if d['0'][1] <= limit:
                            s.append('1')
                            d = d['0'][0]
                        else:
                            s.append('0')
                            d = d['1'][0]
                else:    
                    if '1' not in d:
                        s.append('0')
                        d = d['0'][0]
                    else:
                        if d['1'][1] <= limit:
                            s.append('1')
                            d = d['1'][0]
                        else:
                            s.append('0')
                            d = d['0'][0]
            res.append(int(''.join(s), 2))
        return res
