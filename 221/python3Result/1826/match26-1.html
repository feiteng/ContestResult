from bisect import bisect, bisect_left
class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        idx = sorted(range(len(queries)), key = lambda x: queries[x][1])
        i = 0
        L = 0
        res = [-1] * len(queries)
        def max_xor(i, xj):
            l = 0
            r = i - 1
            m = 0
            L = max(len(bin(xj)), len(bin(nums[i - 1]))) - 2
            for bi in range(L - 1, -1, -1):
                if l == r:
                    break
                b = 1 << bi
                b1 = 1 if (b & xj) > 0 else 0
                # print(b, b1)
                if b1 == 1:
                    k 
= bisect_left(nums, m + b, l, r + 1)
                    if k > l:
                        r = k - 1
                    else:
                        m += b
                else:
                    k = bisect_left(nums, m + b, l, r + 1)
                    if k <= r:
                        l = k
                        m += b       
                # print(l, r, m)
            return nums[l] ^ xj
        
        for j in idx:
            xj, mj = queries[j]
            while i < len(nums) and nums[i] <= mj:
                i += 1
            if i == 0:
                continue
            # print(j, xj, i, max_xor(i, xj))
            res[j] = max_xor(i, xj)
        return res
