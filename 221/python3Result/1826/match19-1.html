class Node:
    def __init__(self):
        self.a = [None, None]

class Trie:
    def __init__(self, B):
        self.root = Node()
        self.B = B

    def insert(self, x):
        m = 1<<self.B
        node = self.root
        while m:
            v = (x&m) != 0
            
if not node.a[v]:
                node.a[v] = Node()
            node = node.a[v]
            m >>= 1

    def findMaxXor(self, x):
        total = 0
        m = 1 << self.B
        node = self.root
        while m and node:
            v = (x&m) != 0
            
if node.a[1-v]:
                node = node.a[1-v]
                total += m
            elif node.a[v]:
                node = node.a[v]
            else:
                return -1
            m >>= 1
        return total


class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        n = len(nums)
        m = len(queries)
        I = sorted(range(m), key=lambda ii: queries[ii][1])
        sol = [-1]*m
        left = 0
        B = len(bin(max(max(nums), max(q[0] for q in queries)))) - 2
        T = Trie(B)
        for i in I:
            while left < n and nums[left] <= queries[i][1]:
                T.insert(nums[left])
                left += 1
            sol[i] = T.findMaxXor(queries[i][0])
        return sol
