class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:

        trie = {}

        def insert(val):
            t = trie
            for i in range(63, -1, -1):
                if (val & (1 << i)) > 0:
                    if 1 not in t:
                        t[1] = {}
                    t = t[1]
                else:
                    if 0 not in t:
                        t[0] = {}
                    t = t[0]
            t[2] = True

        def find(val):
            t = trie
            x = 0
            for i in range(63, -1, -1):
                if (val & (1 << i)) > 0:
                    if 0 in t:
                        x |= 1 << i
                        t = t[0]
                    elif 1 in t:
                        t = t[1]
                    else:
                        return -1
                else:
                    if 1 in t:
                        x |= 1 << i
                        t = t[1]
                    elif 0 in t:
                        t = t[0]
                    else:
                        return -1
            return x


        nums.sort()
        queries = [(y, i, x) for i, (x, y) in enumerate(queries)]
        queries.sort()
        n = len(queries)
        ans = [-1 for _ in range(n)]

        j = 0
        for y, i, x in queries:
            while j < len(nums) and nums[j] <= y:
                insert(nums[j])
                j += 1
            ans[i] = find(x)

        return ans
