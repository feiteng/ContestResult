class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        trie = [[-1, -1, -1]]

        def add(a):
            u = 0
            for i in range
(32, -1, -1):
                d = a >> i & 1
                if trie[u][d] == -1:
                    trie[u][d] = len(trie)
                    trie.append([-1, -1, -1])
                u = trie[u][d]
                if i == 0: trie[u][2] = a

        for a in nums: add(a)

        def dfs(u=0, i=32, s=0):
            if s > m: return -1
            v = trie[u][2]
            if v != -1: return v
            res = -1
            d = x >> i & 1
            nxt = trie[u][1-d]
            if nxt != -1:
                res = dfs(nxt, i-1, s | (1-d) << i)
            nxt = trie[u][d]
            if nxt != -1 and res == -1:
                res = dfs(nxt, i-1, s | d << i)
            return res

        ans=[]
        for x, m in queries:
            a=dfs()
            if a==-1:ans.append(-1)
            else:ans.append(a^x)
        
        return ans
