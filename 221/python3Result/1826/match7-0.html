class TrieNode:
    def __init__(self):
        self.zero = None
        self.one = None


class BitTrie:
    def __init__(self):
        self.root = TrieNode()
        
    def insert(self, n):
        current = self.root
        for w in range(31, -1, -1):
            bit = (n >> w) & 1
            if bit == 1:
                if current.one is None:
                    current.one = TrieNode()
                current = current.one
            else:
                if current.zero is None:
                    current.zero = TrieNode()
                current = current.zero
    
    def max_xor(self, x):
        if self.root.one is None and self.root.zero is None:
            return -1
        
        current = self.root
        result = 0
        for w in range(31, -1, -1):
            bit = (x >> w) & 1
            if bit == 1:
                if current.zero is not None:
                    current = current.zero
                    result += (1 << w)
                else:
                    current = current.one
            else:
                if current.one is not None:
                    current = current.one
                    result += (1 << w)
                else:
                    current = current.zero
        return result

    
class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums = sorted(nums)
        queries = sorted([(queries[i][1], (i, queries[i][0])) for i in range(len(queries))], key=lambda x: x[0])
        
        result = [None] * len(queries)
        trie = BitTrie()
        i = 0
        for q in queries:
            while i < len(nums) and nums[i] <= q[0]:
                trie.insert(nums[i])
                i += 1
            result[q[1][0]] = trie.max_xor(q[1][1])
        return result
