class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        ans = []
        for x, m in queries:
            hi = bisect.bisect_right(nums, m)
            if hi == 0:
                ans.append(-1)
                continue
            
            base, lo = 0, 0
            for i in range(31, -1, -1):
                mask = 1 << i
                p = bisect.bisect_left(nums, base + mask, lo, hi)
                if mask & x == 0:
                    if p != hi:
                        lo, base = p, base + mask
                else:
                    if p == lo: base += mask
                    else: hi = p
            ans.append(base ^ x)
        return ans
