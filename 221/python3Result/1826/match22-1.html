class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        ans=[]
        for query in queries:
            xi=query[0]
            mi=query[1]
            nowi=bisect.bisect(nums, mi, lo=0, hi=len(nums))
            if nowi==0:
                ans.append(-1)
            else:
                wei=0
                while((2**wei)<=max(xi,mi)):
                    wei=wei+1
                wei=wei-1
                tempxor=2**wei
                lowi=0
                highi=nowi
                tempxi=xi
                tempans=0
                
                while(wei>=0):
                    #print(lowi,highi,tempxor,tempans,wei)
                    nowi=bisect.bisect_left(nums, tempxor, lo=lowi, hi=highi)
                    if tempxi>=2**wei:
                        if nowi==lowi:
                            lowi=nowi
                            tempxor=tempxor+2**(wei-1)
                        else:
                            highi=nowi
                            tempxor=tempxor-2**(wei-1)
                            tempans=tempans+2**wei
                            
                        tempxi-=(2**wei)
                    else:
                        if nowi==highi:
                            highi=nowi
                            tempxor=tempxor-2**(wei-1)
                        else:
                            lowi=nowi
                            tempxor=tempxor+2**(wei-1)
                            tempans=tempans+2**wei
                    wei=wei-1
                    
                
                
                ans.append(tempans)
        return ans
