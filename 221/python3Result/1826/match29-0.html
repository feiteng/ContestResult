from functools import lru_cache
class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        nums.sort()
        @lru_cache(None)
        def div(l, u, i):
            ll, uu = l, u
            while uu - ll > 0:
                c = (ll + uu) // 2
                if (nums[c] & (1 << i)) == 0:
                    ll = c + 1
                else:
                    uu = c
            return ll
        def sub(x, m, i, l, u):
            # print('sub', (x, m, i, l, u))
            if u - l == 0:
                return -1
            if nums[l] > m:
                return -1
            if i < 0:
                return nums[l] ^ x
            ll = div(l, u, i)
            if (x & (1 << i)) != 0:
                ans = sub(x, m, i - 1, l, ll)
                if ans >= 0:
                    return ans
                return sub(x, m, i - 1, ll, u)
            else:
                ans = sub(x, m, i - 1, ll, u)
                if ans >= 0:
                    return ans
                return sub(x, m, i - 1, l, ll)
        return [sub(x, m, 32, 0, len(nums)) for x, m in queries]
