class Node:
    def __init__(self):
        self.c = [None, None]
        self.val = 0

class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        q = sorted([limit, j, i] for i, (j, limit) in enumerate(queries))
        ret = [-1] * len(q)
        nums.sort() 
        p = 0
        n = len(nums)
        root = Node()
        l = len(bin(max(nums[-1], max(i for i, j in queries)))) - 3
        for limit, j, i in q:
            while p < n and nums[p] <= limit:
                if p and nums[p] == nums[p - 1]:
                    p += 1
                    continue
                f = root
                k = 1 << l
                while k:
                    if k & nums[p]:
                        if f.c[1] is None:
                            f.c[1] = Node()
                        f = f.c[1]
                    else:
                        if f.c[0] is None:
                            f.c[0] = Node()
                        f = f.c[0]
                    k >>= 1
                f.val = nums[p]
                p += 1
            if p == 0:
                ret[i] = -1
                continue
            f = root
            k = 1 << l
            while k:
                t = 1 - int(k & j > 0)
                if f.c[t] is not None:
                    f = f.c[t]
                else:
                    f = f.c[1 - t]
                k >>= 1
            ret[i] = f.val ^ j
        return ret
