class Solution:
    def maximizeXor(self, nums: List[int], queries: List[List[int]]) -> List[int]:
        d={}
        nums.sort()
        l=len(queries)
        s=sorted(range(l),key=lambda x:queries[x][1])
        k=0
        n=len(nums)
        ans=[-1]*l
        mask=(1<<30)-1
        #print(s)
        for i in range(l):
            ma=queries[s[i]][1]
            while k<n and nums
[k]<=ma:
                cur=d
                for j in range(29,-1,-1):
                    tmp=+(nums[k]>>j&1)
                    if tmp in cur:
                        cur=cur[tmp]
                    else:
                        cur[tmp]={}
                        cur=cur[tmp]
                k+=1
            x=queries[s[i]][0]
            tmp=x^mask
            cur=d
            f=True
            ti=i
            for i in range(29,-1,-1):
                hope=tmp>>i&1
                if hope not in cur:
                    hope^=1
                    tmp^=1<<i
                if hope not in cur:
                    f=False
                    break
                cur=cur[hope]
                
            if f:
                ans[s[ti]]=tmp^x
        return ans
