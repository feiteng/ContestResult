class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        ans = []
        for j in range(n):
            pos = j
            for i in range(m):
                if grid[i][pos] == 1:
                    if pos == n - 1 or grid[i][pos + 1] == -1:
                        pos = -1
                        break
                    pos += 1
                else:
                    if pos == 0 or grid[i][pos - 1] == 1:
                        pos = -1
                        break
                    pos -= 1
            ans.append(pos)
        
        return ans
