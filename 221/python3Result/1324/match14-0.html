class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        HEIGHT, WIDTH = len(grid), len(grid[0])
        
        ans = [-1] * WIDTH
        for i in range(WIDTH):
            c = i
            for r in range(HEIGHT):
                if grid[r][c] == 1:
                    # stuck against wall or ramp
                    if c >= WIDTH - 1 or grid[r][c + 1] == -1:
                        c = -1
                        break
                    c += 1
                else:
                    # stuck against wall or ramp
                    if c <= 0 or grid[r][c - 1] == 1:
                        c = -1
                        break
                    c -= 1
                    
            ans[i] = c
        return ans
