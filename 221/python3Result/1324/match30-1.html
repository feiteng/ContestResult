class Solution:
    def findBall(self, G: List[List[int]]) -> List[int]:
        n,m = len(G),len(G[0])
        r = [i for i in range(m)]
        for i in range(n):
            for j in range
(m):
                if r[j] == -1:
                    continue
                rj = r[j]
                if G[i][rj] == 1:
                    if rj == m-1:
                        r[j] = -1
                    elif G[i][rj+1] == -1:
                        r[j] = -1
                    else:
                        r[j] = rj + 1
                else:
                    
if rj == 0:
                        r[j] = -1
                    elif G[i][rj-1] == 1:
                        r[j] = -1
                    else:
                        r[j] = rj - 1
        return r
