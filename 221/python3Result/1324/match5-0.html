class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n = len(grid)
        m = len(grid[0])
        s = [i for i in range(m)]
        for i in range(n):
            for j in range(m):
                if s[j] == -1:
                    continue
                if grid[i][s[j]] == 1:
                    if s[j] + 1 >= m or grid[i][s[j] + 1] == -1:
                        s[j] = -1
                    else:
                        s[j] += 1
                if grid[i][s[j]] == -1:
                    if s[j] - 1 < 0 or grid[i][s[j] - 1] == 1:
                        s[j] = -1
                    else:
                        s[j] -= 1
        return s
