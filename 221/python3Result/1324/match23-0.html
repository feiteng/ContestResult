class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m=len(grid)
        n=len(grid[0])
        ans=[-1]*n
        for i in range(m):
            grid[i].append(-1)
            grid[i].insert(0,1)
        n+=2
        for i in range(1,n-1):
            a=0
            b=i
            while a<m and b>=1 and b<n+2:
                if grid[a][b]==1:
                    if grid[a][b+1]==1:
                        a=a+1
                        b=b+1
                    else:
                        break
                else:
                    if grid[a][b-1]==-1:
                        a=a+1
                        b=b-1
                    else:
                        break
                # print(a,b)
            if a==m:
                ans[i-1]=b-1
        return ans
