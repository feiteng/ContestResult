class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m,n = len(grid),len(grid[0])
        ans = [-1]*n
        for j in range(n):
            curpos = j
            flag 
= True
            for i in range(m):
                if grid[i][curpos] == 1:
                    if curpos == n - 1 or grid[i][curpos+1] == -1:
                        flag = False
                        break
                    else:
                        curpos += 1
                else:
                    if curpos == 0 or grid[i][curpos-1] == 1:
                        flag = False
                        break
                    else:
                        curpos -= 1
            if flag:
                ans[j] = curpos
        return ans
