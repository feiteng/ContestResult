class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n, m = len(grid), len(grid[0])
        
        def dfs(r, c):
            if c < 0 or c >= m: return -1
            
if r == n: return c
            if grid[r][c] == 1:
                if c == m - 1 or grid[r][c + 1] == -1: return -1
                return dfs(r + 1, c + 1)
            else:
                if c == 0 or grid[r][c - 1] == 1: return -1
                return dfs(r + 1, c - 1)
        return [dfs(0, c) for c in range(m)]
