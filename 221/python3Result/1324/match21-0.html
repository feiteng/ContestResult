class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        
        n, m = len(grid), len(grid[0])
        rec = []
        
        for tar in range(m):
            j = tar
            find = True
            for i in range(n):
                
                if grid[i][j] == 1:
                    if j + 1 < m and grid[i][j+1] == 1:
                        j += 1
                    else:
                        find = False
                        break
                else:
                    if j - 1 >= 0 and grid[i][j-1] == -1:
                        j -= 1
                    else:
                        find = False
                        break
            
            if find and 0 <= j < m :
                rec.append(j)
            else:
                rec.append(-1)
        return rec
