class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        @lru_cache(None)
        def dfs(r, c):
            if r == m:
                return c if 0 <= c < n else -1
            if grid[r][c] == 1 and c < n-1 and grid[r][c+1] == 1:
                return dfs(r+1, c+1)
            if grid[r][c] == -1 and c > 0 and grid[r][c-1] == -1:
                return dfs(r+1, c-1)
            return -1
        
        m, n = len(grid), len(grid[0])
        return [dfs(0, i) for i in range(n)]
