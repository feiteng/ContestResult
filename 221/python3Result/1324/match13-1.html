class Solution:
    def findBall(self, A: List[List[int]]) -> List[int]:
        m, n = len(A), len(A[0])
        
        def out(i):
            r = 0
            c = i
            while r < m:
                t = c + A[r][c]
                
if not 0 <= t < n or A[r][c] != A[r][t]:
                    return -1
                c = t
                r += 1
            return t
        return [out(i) for i in range(n)]
