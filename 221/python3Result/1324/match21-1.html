class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        ans = []
        m, n = len(grid), len(grid[0])
        
        
for i in range(n):
            col = i
            flag = True
            
            for j in range(m):
                if grid[j][col]==1:
                    if col+1<n and grid[j][col+1]==1:
                        col += 1
                    else:
                        flag = False
                        break
                else:
                    if col-1>=0 and grid[j][col-1]==-1:
                        col -= 1
                    else:
                        flag = False
                        break
            
            if flag:
                ans.append(col)
            else:
                ans.append(-1)
    
        return ans
