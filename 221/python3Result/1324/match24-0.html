class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        
        @lru_cache(None)
        def judge(x, y):
            if x == m: return y
            if grid[x][y] == 1 and y+1<n and grid[x][y+1] == 1:
                return judge(x+1, y+1)
            if grid[x][y] == -1 and y-1>=0 and grid[x][y-1] == -1:
                return judge(x+1, y-1)
            return -1
            
        
        res = []
        m, n = len(grid), len(grid[0])
        for i in range(n):
            res.append(judge(0, i))
        return res
