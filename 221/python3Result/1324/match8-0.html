class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m,n = len(grid),len(grid[0])
        res = [-1]*n
        
        for j in range(n):
            col,valid = j,True
            for i in range(m):
                if grid[i][col] == 1:
                    if col == n-1 or col!=n-1 and grid[i][col+1] == -1:
                        valid = False
                        #res.append(-1)
                        break
                    col += 1
                else:
                    if col == 0 or col!=0 and grid[i][col-1] == 1:
                        valid = False
                        #res.append(-1)
                        break
                    col -= 1
            if valid:
                res[j] = col
        return res
