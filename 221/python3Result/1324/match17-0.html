class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n = len(grid)
        m = len(grid[0])
        ans = []
        for j in range(m):
            i = 0
            while i < n:
                if grid[i][j] == 1:
                    if j+1 >= m or grid[i][j+1] == -1:
                        break
                    i += 1
                    j += 1
                else:
                    if j-1 < 0 or grid[i][j-1] == 1:
                        break
                    i += 1
                    j -= 1
            if i == n:
                ans.append(j)
            else:
                ans.append(-1)
        return ans
