class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        rtn = []
        m = len(grid)
        n = len(grid[0])
        for j in range(n):
            tj = j
            is_end = False
            for i in range(m):
                if is_end:
                    break
                # 往右
                if grid[i][tj] == 1:
                    if tj >= n - 1 or grid[i][tj + 1] == -1:
                        is_end = True
                    else:
                        tj += 1
                # 往左
                else:
                    if tj <= 0 or grid[i][tj - 1] == 1:
                        is_end = True
                    else:
                        tj -= 1
            if not is_end:
                rtn.append(tj)
            else:
                rtn.append(-1)

        return rtn
