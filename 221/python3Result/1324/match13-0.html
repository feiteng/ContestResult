class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        def cal(k):
            i = 0
            j = k
            while i < m:
                d = grid[i][j]
                nj = j + d
                if not 0 <= nj < n:
                    return -1
                if grid[i][nj] == -grid[i][j]:
                    return -1
                i += 1
                j = nj
            return j
        return list(map(cal, range(n)))
