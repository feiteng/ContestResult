class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m = len(grid)
        n = len(grid[0])
        ans = []
        for i in range(n):
            cur = i
            for j in range(m):
                if cur+1<n and grid[j][cur] == 1 and grid[j][cur+1] == 1:
                    cur+=1
                elif cur-1>=0 and grid[j][cur] == -1 and grid[j][cur-1] == -1:
                    cur-=1
                else:
                    cur = -1
                    break
            ans.append(cur)
        return ans
