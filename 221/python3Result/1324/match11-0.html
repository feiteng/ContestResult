class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        
        num_rows = len(grid)
        num_cols = len(grid[0])
        
        answer = []
        
        for c in range(num_cols):
            curr_c = c
            for r in range(num_rows):
                # redirect to right
                if grid[r][curr_c] == 1:
                    if curr_c == num_cols-1 or grid[r][curr_c+1] == -1:
                        curr_c = -1
                        break
                    else:
                        curr_c += 1
                elif grid[r][curr_c] == -1:
                    if curr_c == 0 or grid[r][curr_c-1] == 1:
                        curr_c = -1
                        break
                    else:
                        curr_c -= 1
            
            answer.append(curr_c)
        
        return answer
