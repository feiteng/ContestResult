class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        ans = [-1] * n
        for i in range(n):
            col, h = i, 0
            while h < m:
                if grid[h][col] == 1:
                    if col == n - 1 or grid[h][col + 1] == -1:
                        break
                    col += 1
                else:
                    if col == 0 or grid[h][col - 1] == 1:
                        break
                    col -= 1
                h += 1
            if h == m:
                ans[i] = col
        return ans
