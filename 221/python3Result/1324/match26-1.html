class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        grids = [[1] + row + [-1]  for row in grid]
        m = len(grids)
        n = len(grids[0])
        #print(m ,n)
        res = []
        for start in range(1, n - 1):
            col 
= start
            for row in range(m):
                #print(row, col)
                if col == 0 or col >= n - 1:
                    break
                if grids[row][col] == 1:
                    if grids[row][col + 1] == -1:
                        col = n
                    else:
                        col += 1
                else:
                    if grids[row][col - 1] == 1:
                        col = n
                    else:
                        col -= 1
                
            if col == 0 or col >= n - 1:
                res.append(-1)
            else:
                res.append(col - 1)
        return res
