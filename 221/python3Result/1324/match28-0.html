class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m = len(grid)
        n = len(grid[0])
        
        ans = [-1] * n
        
        for i in range(n):
            col = i
            bot = True
            
            for r in range(m):
                direction = grid[r][col]
                if col + direction < 0 or col + direction >= n: 
                    bot = False
                    break
                if grid[r][col] == 1 and grid[r][col+1] == -1:
                    bot = False
                    break
                
                if grid[r][col] == -1 and grid[r][col-1] == 1:
                    bot = False
                    break
                    
                col = col + direction
            if bot: ans[i] = col
        
        return ans
