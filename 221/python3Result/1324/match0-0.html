class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        ans = []
        for j in range(n):
            an = j
            for i in range(m):
                if grid[i][an] == -1:
                    if an == 0 or grid[i][an-1] == 1:
                        an = -1
                        break
                    else:
                        an -= 1
                else:
                    if an == n-1 or grid[i][an+1] == -1:
                        an = -1
                        break
                    else:
                        an += 1
            ans.append(an)
        return ans
