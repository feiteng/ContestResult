class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m, n = len(grid), len(grid[0])
        ans = []
        for start_c in range(n):
            # c means the number of the ball
            c = start_c
            for r in range(m): # j is the number of rows
                if grid[r][c] == 1:
                    if c == n - 1 or grid[r][c + 1] == -1:
                        ans.append(-1)
                        break
                    else:
                        c += 1
                else: # -1
                    if c == 0 or grid[r][c - 1] == 1:
                        ans.append(-1)
                        break
                    else: 
                        c -= 1
            else:
                ans.append(c)
        return ans
