class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m = len(grid)
        n = len(grid[0])
        res = []
        for i in range(n):
            
for j in range(m):
                if grid[j][i] == 1:
                    if i+1 == n or grid[j][i+1] == -1:
                        res.append(-1)
                        break
                    else:
                        i += 1
                else:
                    if i == 0 or grid[j][i-1] == 1:
                        res.append(-1)
                        break
                    else:
                        i -= 1
            else:
                res.append(i)
        return res
