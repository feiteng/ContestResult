class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n = len(grid[0])
        ball_index = list(range(n))
        for i in range(len(grid)):
            for j in range(len(ball_index)):
                if ball_index[j] == -1:
                    continue
                col = ball_index[j]
                if grid[i][col] == 1:
                    if col == n - 1:
                        ball_index[j] = -1
                    elif grid[i][col+1] == -1:
                        ball_index[j] = -1
                    else:
                        ball_index[j] += 1
                if grid[i][col] == -1:
                    if col == 0:
                        ball_index[j] = -1
                    elif grid[i][col-1] == 1:
                        ball_index[j] = -1
                    else:
                        ball_index[j] -= 1
        return ball_index
