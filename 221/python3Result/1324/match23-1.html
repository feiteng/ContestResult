class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        
        n = len(grid)
        m = len(grid[0])
        res = []
        
        for i in range(m):
            j = i
            k 
= 0
            fall = True
            while k < n:
                if grid[k][j] == 1:
                    if j < m-1 and grid[k][j+1] == 1:
                        j += 1
                        k += 1
                    else:
                        break
                else:
                    if j > 0 and grid[k][j-1] == -1:
                        j -= 1
                        k += 1
                    else:
                        break
            res.append(j if k == n else -1)
        return res
