class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n, m = len(grid), len
(grid[0])
        ret = []
        for s in range(m):
            x = s
            for i in range(n):
                if grid[i][x] == 1:
                    if x == m - 1 or grid[i][x + 1] == -1:
                        x = -1
                        break
                    else:
                        x += 1
                else:
                    
if x == 0 or grid[i][x - 1] == 1:
                        x = -1
                        break
                    else:
                        x -= 1
            ret.append(x)
        return ret
