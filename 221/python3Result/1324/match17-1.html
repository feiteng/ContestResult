class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        rn=len(grid)
        cn=len(grid[0])
        ans=[-1]*cn
        for ci0 in range(cn):
            ri=0
            ci
=ci0
            while ri<rn:
                if grid[ri][ci]==1:
                    if ci==cn-1 or grid[ri][ci+1]==-1:
                        break
                    ci+=1
                    ri+=1
                else:
                    if ci==0 or grid[ri][ci-1]==1:
                        break
                    ci-=1
                    ri+=1
            if ri==rn:
                ans[ci0]=ci
        return ans
