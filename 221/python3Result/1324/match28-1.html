class Solution:
    def findBall(self, g: List[List[int]]) -> List[int]:
        n = len(g)
        m = len(g[0])
        res = [-1] * m
        for i in range(m):
            c = i
            ans = -1
            for r in range(n):
                if c == 0 and g[r][c] == -1:
                    ans = -1
                    break
                if c == m - 1 and g[r][c] == 1:
                    ans 
= -1
                    break
                if g[r][c] == 1 and g[r][c + 1] == -1:
                    ans = -1
                    break
                if g[r][c] == -1 and g[r][c - 1] == 1:
                    ans = -1
                    break
                ans = c + 1 if g[r][c] == 1 else c - 1
                c = ans
            res[i] = ans
        return res
