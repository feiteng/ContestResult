class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n = len(grid)
        m = len(grid[0])
        ans = [-1]*m
        for ball in range(m):
            j = ball
            flag 
= True
            for i in range(n):
                if grid[i][j] == 1 and j < m-1 and grid[i][j+1] == 1:
                    j += 1
                elif grid[i][j] == -1 and j > 0 and grid[i][j-1] == -1:
                    j -= 1
                else:
                    flag = False
                    break
            if flag:
                ans[ball] = j
        return ans
