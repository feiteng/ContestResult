class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        n = len(grid)
        m = len
(grid[0])
        
        res = [-1] * m
        
        for j in range(m):
            st = j
            f = True
            for i in range(n):
                if grid[i][st] == 1:
                    if st == m - 1 or grid[i][st + 1] == -1:
                        f = False
                        break
                    else:
                        st += 1
                else:
                    if st == 0 or grid[i][st - 1] == 1:
                        f = False
                        break
                    else:
                        st -= 1
            if f:
                res[j] = st
        
        return res
