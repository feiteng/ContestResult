class Solution:
    def findBall(self, grid: List[List[int]]) -> List[int]:
        m = len(grid)
        n = len(grid[0])

        def f(x, y) -> int:
            if x >= m:
                return y
            if grid[x][y] == 1:
                if y == n - 1 or grid[x][y+1] == -1:
                    return -1
                return f(x+1, y+1)
            else:
                if y == 0 or grid[x][y-1] == 1:
                    return -1
                return f(x+1, y-1)
        
        return [f(0, i) for i in range(n)]
