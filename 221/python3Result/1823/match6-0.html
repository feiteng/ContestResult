class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        m=len(s)>>1
        rcd=set(['a','e','i','o','u','A','E','I','O','U'])
        left,right=0,0
        for i in range(m):
            if s[i] in rcd:
                left+=1
        for i in range(m,len(s)):
            if s[i] in rcd:
                right+=1
        return left==right
