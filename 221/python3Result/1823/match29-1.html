class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        k 
= len(s)//2
        s1 = s[:k]
        s2 = s[k:]
        n1 = 0
        n2 = 0 
        for c in s1:
            if c in 'aeiouAEIOU':
                n1 += 1
        for c in s2:
            if c in 'aeiouAEIOU':
                n2 += 1
        return n1==n2
