class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        half_len  = len(s) // 2
        yy = set(['a','e','i','o','u','A','E','I','O','U'])
        cnt1, cnt2 = 0, 0
        for i in range(half_len):
            if s[i] in yy:
                cnt1 += 1
        for i in range(half_len, len(s)):
            if s[i] in yy:
                cnt2 += 1
        return cnt1 == cnt2
