class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        dic1 = 0
        dic2 = 0
        n = len(s)//2
        for i in range(n):
            if s[i] in ['a','e','i','o','u','A','E','I','O','U']:
                dic1 += 1
        
        for i in range(n,2*n):
            if s[i] in ['a','e','i','o','u','A','E','I','O','U']:
                dic2 += 1
        
        return dic1 == dic2
