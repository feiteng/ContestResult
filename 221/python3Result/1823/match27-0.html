class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        yuanyin = {'a','e','i','o','u','A','E','I','O','U'}
        left=0
        right=0
        n = len(s)
        for i in range(0,n//2):
            if(s[i] in yuanyin ): left+=1;
        for i in range(n//2,n):
            if(s[i] in yuanyin): right+=1;
        return left==right
