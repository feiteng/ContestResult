class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        dic = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
        t1,t2=0,0
        for i in range(len(s)//2):
            if s[i] in dic:
                t1+=1
        for i in range(len(s)//2,len(s)):
            if s[i] in dic:
                t2+=1
        return t1==t2
