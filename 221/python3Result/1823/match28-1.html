class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vowels = 'aeiouAEIOU'
        l 
= 0
        r = 0
        n = len(s)
        for c in s[:n // 2]:
            if c in vowels:
                l += 1
        for c in s[n // 2:]:
            if c in vowels:
                r += 1
        return l == r
