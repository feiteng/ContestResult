class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vowels = 'aeiouAEIOU'
        cnt = 0
        n = len(s)
        
        for c in s[:n//2]:
            if c in vowels:
                cnt += 1
        
        for c in s[n//2:]:
            if c in vowels:
                cnt -= 1
        
        return (cnt == 0)
