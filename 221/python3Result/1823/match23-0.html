class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        d = ['a','e','i','o','u','A','E','I','O','U']
        n = len(s)
        a = s[:n//2]
        b = s[n//2:]
        c1 = 0
        c2 = 0
        for i in range(n//2):
            if a[i] in d:
                c1 += 1
            if b[i] in d:
                c2 += 1
        return c1 == c2
