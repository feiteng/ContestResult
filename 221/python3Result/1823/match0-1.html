class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        n = len(s)
        vowels = 'aoeiuAOEIU'
        left, right = 0, 0
        for i in range(n):
            if s[i] in vowels:
                if i < n // 2:
                    left += 1
                else:      
                    right += 1
        return left == right
