class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        s = s.lower()
        
        s1, s2 = s[:(len(s)//2)], s[(len(s)//2):]
        # print(s1, s2)
        return s1.count('a')+s1.count('e')+s1.count('i')+s1.count('o')+s1.count('u') == s2.count('a')+s2.count('e')+s2.count('i')+s2.count('o')+s2.count('u')
