class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vow = 'aeiouAEIOU'
        a, b = 0, 0
        n=len(s)
        for i in range(n//2):
            if s[i] in vow:
                a += 1
        for i in range(n//2):
            if s[i+(n//2)] in vow:
                b += 1
        return a == b
