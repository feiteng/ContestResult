class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vs = ('a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U')
        l = len(s)
        a, b = s[:l//2], s[l//2:]
        ac, bc = 0, 0
        for c in a:
            if c in vs:
                ac += 1
        for c in b:
            if c in vs:
                bc += 1
        return ac == bc
