class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        s1 = s.lower()[:len(s)//2]
        s2 = s.lower()[len(s)//2:]
        c1 = len([t for t in s1 if t in ['a', 'e', 'i', 'o', 'u']])
        c2 = len([t for t in s2 if t in ['a', 'e', 'i', 'o', 'u']])
        return c1 == c2
