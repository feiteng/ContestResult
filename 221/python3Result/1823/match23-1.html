class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        dic = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}
        l = len(s)
        s1 = s[:l//2]
        s2 = s[l//2:]
        count1 = count2 = 0
        for i in range(len(s1)):
            
if s1[i] in dic:
                count1 += 1
            if s2[i] in dic:
                count2 += 1
        return count1==count2
