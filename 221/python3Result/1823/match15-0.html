class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        a = collections.Counter(s[:len(s)//2])
        b = collections.Counter(s[len(s)//2:])
        x = sum(a[k] for k in ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'])
        y = sum(b[k] for k in ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'])
        return x == y
