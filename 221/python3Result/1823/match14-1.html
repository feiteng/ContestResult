class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vowels = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}
        l = 0
        r = 0
        n = len(s) // 2
        for i in range(n):
            if s[i] in vowels:
                l += 1
            if s[i + n] in vowels:
                r += 1
        
        return l == r
