class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        vowels = ['a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U']
        fc = sc = 0
        n = len(s)
        for i in range(n//2):
            if s[i] in vowels: fc += 1
            if s[n-1-i] in vowels: sc += 1
                
        return fc == sc
