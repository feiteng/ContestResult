class Solution:
    def halvesAreAlike(self, s: str) -> bool:
        a = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'}
        n = len(s)
        s1 = s[: n // 2]
        s2 = s[n // 2 : ]
        x1 = 0
        x2 = 0
        for x in s1:
            if x in a:
                x1 += 1
        for x in s2:
            if x in a:
                x2 += 1
        return x1 == x2
