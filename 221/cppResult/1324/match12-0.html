class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
                
        vector<int> cur_pos;
        for (int i = 0; i < n; i++) {
            cur_pos.push_back(i);
        }
        
        
        for (int e = 0; e < m; e++) {
            for (int ball = 0; ball < n; ball++) {
                if (cur_pos[ball] == -1)
                    continue;
                
                int cur = cur_pos[ball];
                
                if (grid[e][cur] == 1) {
                    if (cur == n - 1 || grid[e][cur + 1] == -1) {
                        cur_pos[ball] = -1;
                    } else {
                        cur_pos[ball] = cur + 1;
                    }
                } else {
                    if (cur == 0 || grid[e][cur - 1] == 1) {
                        cur_pos[ball] = -1;
                    } else {
                        cur_pos[ball] = cur - 1;
                    }
                }
            }
        }
        
        return cur_pos;
    }
};
