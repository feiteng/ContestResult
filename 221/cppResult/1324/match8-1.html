class Solution {
public:
   vector<int> findBall(vector<vector<int>>& grid) {
      int n = grid.size(); // rows
      int m = grid[0].size(); // columns
      vector<int> ans(m, -1);

      for (int j = 0; j < m; j++) { // figure out ans[j]
         int currentColumn = j;

         for (int i = 0; i < n; i++) {
            if (grid[i][currentColumn] == -1) { // go left
               if (currentColumn == 0 || grid[i][currentColumn - 1] == 1) {
                  currentColumn = -1;
                  break;
               }
               currentColumn--;
            } else { // go right
               if (currentColumn == m - 1 || grid[i][currentColumn + 1] == -1) {
                  currentColumn = -1;
                  break;
               }
               currentColumn++;
            }
         }
         ans[j] = currentColumn;
      }

      return ans;
   }
};
