class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<int> ans(m);
        for(int u = 0; u < m ;u++) {
            int y = u;
            bool stuck = false;
            for(int i = 0; i < n; i++) {
                if(grid[i][y] == 1 && y+1 < m && grid[i][y+1] == 1) {
                    y++;    
                } else if(grid[i][y] == -1 && y-1 >= 0 && grid[i][y-1] == -1) {
                    y--;
                } else {
                    stuck = true;
                    break;
                }
            }
            if(stuck) {
                ans[u] = -1;
            } else {
                ans[u] = y;
            }
        }
        return ans;
    }
};
