class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> ret(n, -1);
        
        for (int i = 0; i < n; i++) {
            int col = i;
            for (int j = 0; j < m; j++) {
                if (grid[j][col] == 1) {
                    if (col < n - 1 && grid[j][col+1] == 1) {
                        col++;
                    } else {
                        col = -1;
                        break;
                    }
                } else {
                    if (col > 0 && grid[j][col-1] == -1) {
                        col--;
                    } else {
                        col = -1;
                        break;
                    }
                }
                //cout << i << j << col << endl;
            }
            ret[i] = col;
        }
        return ret;
    }
};
