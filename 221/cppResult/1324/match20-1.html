class Solution {
    int m, n;
public:

    vector<int> findBall(vector<vector<int>>& grid) {
        m = grid.size(), n = grid[0].size();
        vector<int> res(n, -1);
        for(int x=0;x<n;x++) {
            res[x] = dfs(grid, 0, x);
            // cout<<endl;
        }
        return res;
    }

    int dfs(vector<vector<int>>& grid, int x, int y) {
        // cout<<x<<" "<<y<<"->";
        if(x==m) return y;
        if(grid[x][y]==-1) {
            if(y==0 || grid[x][y-1]==1) return -1;
            return dfs(grid, x+1, y-1);
        } else {
            if(y==n-1 || grid[x][y+1]==-1) return -1;
            return dfs(grid, x+1, y+1);
        }
    }
};
