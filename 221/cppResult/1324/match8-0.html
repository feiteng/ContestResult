class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector<int> ans(m,-1);
        for(int i = 0; i < m; i++){
            int cx = 0;
            int cy = i;
            bool flag = true;
            for(int j = 0; j < n; j++){
                if(grid[j][cy]==1){
                    if(cy+1==m||grid[j][cy+1]==-1){
                        flag = false;
                        break;
                    }
                    cy++;
                }else{
                    if(cy==0||grid[j][cy-1]==1){
                        flag = false;
                        break;
                    }
                    cy--;
                }
            }
            if(flag)ans[i] = cy;
        }
        return ans;
    }
};
