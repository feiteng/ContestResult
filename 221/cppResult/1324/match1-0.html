class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> result(n);
        for (int i = 0; i < n; ++i) {
            int row = 0, col = i;
            while (true) {
                if (grid[row][col] == 1) {
                    if (col == n-1 || grid[row][col+1] == -1) {
                        result[i] = -1;
                        break;
                    } else {
                        row++;
                        col++;
                    }
                } else {
                    if (col == 0 || grid[row][col-1] == 1) {
                        result[i] = -1;
                        break;
                    } else {
                        row++;
                        col--;
                    }
                }
                if (row == m) {
                    result[i] = col;
                    break;
                }
            }
        }
        return result;
    }
};
