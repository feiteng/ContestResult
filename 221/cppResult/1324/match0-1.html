class Solution {
public:
    int m, n;
    vector<int> findBall(vector<vector<int>> &grid)
    {
        m = grid.size(), n = grid[0].size();
        vector<int> res(n, 0);
        int x, y,xx,yy;
        int cur;
        for (int i = 0; i < n; ++i)
        {
            x = 0;
            y = i;
            while(x<m&&y>=0){
                yy = y + grid[x][y];
                if (is_valid(x, yy))
                {
                    if (grid[x][y]==grid[x][yy]){
                        x++;
                        y = yy;
                    }
                    else{
                        y = -1;
                    }
                }
                else{
                    y = -1;
                }
            }
            res[i] = y;
        }
        return res;
    }
    bool is_valid(int x,int y){
        return 0 <= x && x < m && 0 <= y && y < n;
    }
};
