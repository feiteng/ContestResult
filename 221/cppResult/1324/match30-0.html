class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        
        
        vector<int> cur(n, 0);
        for(int i = 1; i < n; ++i) {
            cur[i] = i;
        }
        vector<int> cur2(n, INT_MAX);
        
        for(int i = m - 1; i >= 0; --i) {
            
            /*for(auto& c : cur) {
                std::cout << c << ", ";
            }
            std::cout << endl;*/
            
            for(int j = 0; j < n; ++j) {
                if (j == 0) {
                    if (grid[i][j] == 1) {
                        if (j < n - 1 && grid[i][j + 1] == -1) {
                            cur2[j] = -1;
                        } else if (j == n - 1) {
                            cur2[j] = -1;
                        } else {
                            cur2[j] = cur[j + 1];
                        }
                    } else {
                        cur2[j] = -1;
                    }
                } else if (j == n - 1) {
                    if (grid[i][j] == -1) {
                        if (j > 0 && grid[i][j - 1] == 1) {
                            cur2[j] = -1;
                        } else if (j == 0) {
                            cur2[j] = -1;
                        } else {
                            cur2[j] = cur[j - 1];
                        }
                    } else {
                        cur2[j] = -1;
                    }
                } else {
                    if (grid[i][j] == 1) {
                        if (grid[i][j + 1] == -1) {
                            cur2[j] = -1;
                        } else {
                            cur2[j] = cur[j + 1];
                        }
                    } else {
                        if (grid[i][j - 1] == 1) {
                            cur2[j] = -1;
                        } else {
                            cur2[j] = cur[j - 1];
                        }
                    }
                }
            }
            
            std::swap(cur, cur2);
        }        
        
        return cur;
    }
};
