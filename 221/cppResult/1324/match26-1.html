class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid)
    {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> answer(n);
        for (int i = 0; i < n; i++) {

            int c = i;
            for (int j = 0; j < m; j++) {
                int nc = c + grid[j][c];
                if (nc < 0 || nc >= n || nc + grid[j][nc] == c) { c = -1; break; }
                c = nc;
            }
            answer[i] = c;
        }
        return answer;
    }
};
