class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int rows = grid.size();
        int cols = grid[0].size();
        vector<int> result(cols);
        for (int col = 0; col < cols; ++col) {
            result[col] = col;
            for (int row = 0; row < rows; ++row) {
                int& curCol = result[col];
                if ((grid[row][curCol] == 1 && (curCol == cols - 1 || grid[row][curCol + 1] == -1)) ||
                    (grid[row][curCol] == -1 && (curCol == 0 || grid[row][curCol - 1] == 1))) {
                    curCol = -1;
                    break;
                }
                curCol += grid[row][curCol];
            }
        }
        return result;
    }
};
