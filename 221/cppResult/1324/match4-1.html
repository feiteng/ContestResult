class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> ans(n); 
        fill(ans.begin(), ans.end(), 0);
        
        for (int l = 0; l < n; l++) {
            int x = l;
            for (int i = 0; i < m; i++) {
                if (grid[i][x] == 1) {
                    if (x == n-1 ||
                       grid[i][x+1] == -1) {
                        x = -1;
                        break;
                    }
                    x++;
                } else if (grid[i][x] == -1) {

                    if (x == 0 ||
                       grid[i][x-1] == 1) {
                        x = -1;
                        break;
                    }
                    x--;
                }
            }
            ans[l] = x;
        }
        return ans;
    }
};
