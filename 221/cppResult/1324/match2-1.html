class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        vector<int> ans;
        for(int t=0;t<grid[0].size();t++) {
            int cur = t;
            int kazhu=0;
            for(int i=0;i<grid.size();i++) {
                // pdkazhu
                if (grid[i][cur]==1) {
                    if(cur+1>=grid[i].size() || grid[i][cur+1]==-1) {
                        kazhu=1;
                        break;
                    } else{
                        cur++;
                    }
                } else {
                    if (cur-1<0 || grid[i][cur-1]==1) {
                        kazhu=1;
                        break;
                    } else {
                        cur--;
                    }
                }
            }
            if (kazhu)  {
                ans.push_back(-1);
            }else {
                ans.push_back(cur);
            }
        }
        return ans;
     }
};
