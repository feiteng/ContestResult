class Solution {
public:
    vector<int> findBall(vector<vector<int>>& g) {
        int n = g.size();
        int m = g[0].size();
        vector<int> ret(m, -1);
        for(int i = 0; i < m; i++) {
            int x = i;

            bool flag = false;
            for(int r = 0; r < n; r++) {
                if (g[r][x] == 1) {
                    if (x + 1 == m || g[r][x + 1] == -1) {
                        flag = true;
                        break;
                    }
                    x++;
                } else {
                    if (x == 0 || g[r][x-1] == 1) {
                        flag = true;
                        break;
                    }
                    x--;
                }
            }
            if (!flag) ret[i] = x;
        }
        return ret;
    }
};
