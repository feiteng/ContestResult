class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid.front().size();
        
        vector<int> pos(n);
        for(int i=0; i<n; ++i) {
            pos[i] = i;

        }
        
        for(int i=0; i<m; ++i) {
            for(int j=0; j<n; ++j) {
                //cout << pos[j] << " ";
                if(pos[j] == -1) {
                    continue;
                }

                if(grid[i][pos[j]] == 1) {
                    if(pos[j] == n-1 || grid[i][pos[j] + 1] == -1) {
                        pos[j] = -1;
                    } else {
                        ++pos[j];
                    }
                } else {
                    if(pos[j] == 0 || grid[i][pos[j] - 1] == 1) {
                        pos[j] = -1;
                    } else {
                        --pos[j];
                    }
                }
                
            }
            //cout << endl;
        }
        
        return pos;
    }
};
