class Solution {
public:
    vector<int> findBall(vector<vector<int>>& g) {
        int n = g[0].size();
        int m = g.size();
        vector<int> ans;
        for(int i=0;i<n;i++){
            int p = i;
            for(int j=0;j<m;j++){
                if(g[j][p]==1){
                    if(p==n-1||g[j][p+1]==-1){
                        p = -1;
                        break;
                    }else{
                        p++;
                    }
                }else{
                    if(p==0||g[j][p-1]==1){
                        p = -1;
                        break;
                    }else{
                        p--;
                    }
                }
            }
            ans.push_back(p);
        }
        return ans;
    }
};
