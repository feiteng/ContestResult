class Solution {
public:
    vector<int> findBall(vector<vector<int>>& g) {
        int n=g.size(),m=g[0].size();
        vector<int>res(m);
        for(int j=0;j<m;j++){
            int k=j;

            bool flag=true;
            for(int i=0;i<n;i++){
                if(g[i][k]==1){
                    if(k==m-1||g[i][k+1]==-1){
                        res[j]=-1;
                        flag=false;
                        break;
                    }
                    else k++;
                }

                else {
                    if(k==0||g[i][k-1]==1){
                        res[j]=-1;
                        flag=false;
                        break;
                    }
                    else k--;
                }
            }
            if(flag)res[j]=k;
            
        }
        return res;
    }
};
