class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int n=grid.size(),m=grid[0].size();
        vector<int>ret(m);
        for(int i=0;i<m;i++){
            int y=i;

            int ok=1;
            for(int j=0;j<n;j++){
                if(grid[j][y]==1){
                    // if(i==1) cout<<j<<" "<<y<<" "<<grid[j][y+1]<<endl;
                    if(y<m-1&&grid[j][y+1]==1){
                        y++;
                    }else{
                        ok=0;
                        break;
                    }
                }else{
                    if(y>0&&grid[j][y-1]==-1){
                        y--;
                    }else{
                        ok=0;
                        break;
                    }
                }
                // if(i==0) cout<<j<<" "<<y<<endl;
            }
            if(!ok) ret[i]=-1;
            else ret[i]=y;
        }
        return ret;
    }
};
