class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        vector<int>res;
        int n = grid.size() , m = grid[0].size();
        vector<vector<int>>a(n+3,vector<int>(m+3,1));
        for(int i = 1 ; i <= n ; ++i)
            for(int j = 1 ; j <= m ; ++j) a[i][j] = grid[i-1][j-1];
        for(int i = 0 ; i <= n+1 ; ++i) a[i][m+1] = -1;
        for(int k = 1 ; k <= m ; ++k){
            int x = 1, y = k;
            bool flag = true;
            while(x <= n && flag){
                if(y+1 <= m && a[x][y] == 1 && a[x][y+1] == 1) y++,x++;
                else if(y - 1 >= 1 && a[x][y] == -1 && a[x][y-1] == -1) y--,x++;
                else flag = false;
            }
            if(flag) res.push_back(y-1);
            else res.push_back(-1);
        }
        return res;
    }
};
