class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> ans;
        for (int i = 0; i < n; ++i)
        {
            int x = i;
            
            for (int j = 0; j < m; ++j)
            {
                if (grid[j][x] == 1 && (x + 1 < n && grid[j][x + 1] == 1))
                {
                    ++x;
                }
                else if (grid[j][x] == -1 && (x - 1 >= 0 && grid[j][x - 1] == -1))
                {
                    --x;
                }
                else
                {
                    x = -1;
                    break;
                }
            }
            ans.push_back(x);
        }
        return ans;
    }
};
