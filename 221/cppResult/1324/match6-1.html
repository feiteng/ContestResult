class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        vector <int> ans;
        for (int i = 0; i < m; ++i) {
            int col = i;
            for (int row = 0; row < n; ++row) {
                if (grid[row][col] == 1) {
                    if (col == m-1 or grid[row][col+1] == -1) {
                        col = -1;
                        break;
                    }
                    ++col;
                }
                else {
                    if (col == 0 or grid[row][col-1] == 1) {
                        col = -1;
                        break;
                    }
                    --col;
                }
            }
            ans.push_back(col);
        }
        return ans;
    }
};
