class Solution {
public:
    vector<int> findBall(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        vector<int> res(m, -1);
        for (int j = 0; j < m; j++) {
            int x = j;
            for (int i = 0; i < n; i++) {
                int ch = mat[i][x];
                if (ch == 1) {
                    if (x == m - 1 || mat[i][x + 1] == -1) {
                        x = -1;
                        break;
                    }
                    x = x + 1;
                } else {
                    if (x == 0 || mat[i][x - 1] == 1) {
                        x = -1;
                        break;
                    }
                    x = x -1;
                }
            }
            
            res[j] = x;
        }
        return res;
    }
};
