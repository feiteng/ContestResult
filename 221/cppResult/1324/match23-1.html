class Solution {
   public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> ans(n);
        for (int i = 0; i < n; i++) {

            int pos = i;
            for (int j = 0; j < m; j++) {
                if (grid[j][pos] == 1) {
                    // '\'
                    if (pos == n - 1 || grid[j][pos + 1] == -1) {
                        ans[i] = -1;
                        break;
                    }
                    pos++;
                } else {
                    // '/'
                    if (pos == 0 || grid[j][pos - 1] == 1) {
                        ans[i] = -1;
                        break;
                    }
                    pos--;
                }
                                ans[i] = pos;

            }
        }
        return ans;
    }
};
