class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        vector<int>ret;
        for (int i = 0 ; i < grid[0].size() ; i++) {
            int f = 0;
            int pos = i;
            for (int j = 0 ; j < grid.size() ; j++) {
                if (grid[j][pos] == 1) {
                    if ((pos + 1 == grid[0].size() || grid[j][pos + 1] == -1)) {
                        f = 1;
                        break;
                    }
                    else {
                        pos++;
                    }
                }
                else {
                    if ((pos - 1 == -1 || grid[j][pos - 1] == 1)) {
                        f = 1;
                        break;
                    }
                    else {
                        pos--;
                    }
                }
            }
            if (f) {
                ret.push_back(-1);
            }
            else {
                ret.push_back(pos);
            }
        }
        return ret;
    }
};
