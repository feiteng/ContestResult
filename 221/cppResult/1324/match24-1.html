class Solution {
public:
    vector<int> findBall(vector<vector<int>>& a) {
        int n = a.size(), m = a[0].size();
        vector<int> ans(m);
        for (int j = 0; j < m; j++) {
            // printf("%d:\n", j);

            ans[j] = j;

            int& cur = ans[j];
            for (int i = 0; i < n; i++) {
                if (a[i][cur] == 1) {
                    if (cur == m - 1 || a[i][cur + 1] == -1) {
                        cur = -1;
                        break;
                    }
                    cur++;
                } 
                else { 
                    if (cur == 0 || a[i][cur - 1] == 1) {
                        cur = -1;
                        break;
                    }
                    cur--;
                }
                // printf("%d|", cur);
            }
            // printf("\n");
        }
        
        return ans;
    }
};
