class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> balls(n, 0);
        for (int i = 0; i < n; i++) 
            balls[i] = i;
        
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (balls[j] == -1)
                    continue;
                int loc = balls[j];
                if (grid[i][loc] == 1) {
                    if (loc == n - 1 || grid[i][loc + 1] == -1)
                        balls[j] = -1;
                    else
                        balls[j] = loc + 1;
                }
                else {
                    if (loc == 0 || grid[i][loc - 1] == 1)
                        balls[j] = -1;
                    else
                        balls[j] = loc - 1;
                }
            }
        }
        
        return balls;
        
    }
};
