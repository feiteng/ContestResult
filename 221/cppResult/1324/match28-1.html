class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        vector<int> ans;
        for (int i = 0; i < m; ++i) {
            int row = 0, col = i;
            while (row < n) {
                if (grid[row][col] == 1) {
                    if (col + 1 >= m || grid[row][col+1] != 1) {
                        break;
                    }
                    ++row;
                    ++col;
                } else {
                    if (col - 1 < 0 || grid[row][col-1] != -1) {
                        break;
                    }
                    ++row;
                    --col;
                }
            }
            if (row == n) {
                ans.emplace_back(col);
            } else {
                ans.emplace_back(-1);
            }
        }
        return ans;
    }
};
