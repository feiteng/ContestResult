class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m=grid.size(),n=grid[0].size();
        vector<int> res(n,-1);
        for(int i=0;i<n;i++){
            int y=i,flag=0;
            for(int j=0;j<m;j++){
                if(grid[j][y]==1){
                    if(y==n-1 || grid[j][y+1]==-1){
                        flag=1;
                        break;
                    }
                    y++;
                }
                else{
                    if(!y || grid[j][y-1]==1){
                        flag=1;
                        break;
                    }
                    y--;
                }
            }
            if(!flag) res[i]=y;
        }
        return res;
    }
};
