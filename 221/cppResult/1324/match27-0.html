class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m=grid.size();
        int n=grid[0].size();
        vector<int> rec;
        for(int i=0;i<n;i++)
        {
            int t=i;
            int flag=1;
            for(int j=0;j<m;j++)
            {
                if(grid[j][t]==1)
                {
                    if(t==n-1||grid[j][t+1]==-1)
                    {
                        flag=-1;
                        break;
                    }
                    else
                    {
                        t=t+1;
                    }
                }
                else if(grid[j][t]==-1)
                {
                    if(t==0||grid[j][t-1]==1)
                    {
                        flag=-1;
                        break;
                    }
                    else
                    {
                        t=t-1;
                    }
                }
            }
            if(flag>0)
            {
                rec.push_back(t);
            }
            else
            {
                rec.push_back(-1);
            }
            
        }
        return rec;
    }
};
