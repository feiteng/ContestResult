class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> ans(n, 0);
        
        for(int i=0 ; i<n ; i++) {
            int x = i, y = 0;
            // cout << "\n" << i << "\n";
            while( y < m ) {
                // cout << x << " " << y << " " << grid[y][x] << " | ";
                if( grid[y][x] == 1 ) {

                    if( x+1 == n || grid[y][x+1] == -1 ) {
                        ans[i] = -1;
                        break;
                    } else {
                        ++x;
                        ++y;
                    }
                } else {
                    if( x == 0 || grid[y][x-1] == 1 ) {
                        ans[i] = -1;
                        break;
                    } else {

                        --x;
                        ++y;
                    }
                }
            }
            if( ans[i] != -1 ) ans[i] = x;
        }
        return ans;
    }
};
