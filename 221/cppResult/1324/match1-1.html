class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> ans(n);
        for(int i = 0; i < n; i ++) {
            int x = 0, y = i;
            while(1) {
                if(grid[x][y] == 1) {
                    if(y == n - 1 || grid[x][y + 1] == -1) {
                        ans[i] = -1;
                        break;
                    }else {
                        x ++;
                        y ++;
                    }
                }else {
                    if(y == 0 || grid[x][y - 1] == 1) {
                        ans[i] = -1;
                        break;
                    }else {
                        x ++;
                        y --;
                    }
                }
                if(x == m) {
                    ans[i] = y;
                    break;
                }
            }
        }
        return ans;
    }
};
