class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> ret(n, -1);
        for(int i = 0; i < n; i++) {
            int col = i;
            bool isbottom = true;
            for(int j = 0; j < m; j++) {
                if(grid[j][col] == 1 && col < n - 1 && grid[j][col+1] == 1) {
                    col++;
                } else if (grid[j][col] == -1 && col > 0 && grid[j][col-1] == -1) {
                    col--;
                } else {
                    isbottom = false;
                    break;
                }
            }
            if(isbottom) ret[i] = col;
        }
        
        
        return ret;
    }
};
