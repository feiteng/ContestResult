class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        vector<int> ans;
        int n = grid.size(), m = grid[0].size();
        vector<vector<int>> a(n + 2, vector<int>(m + 2, 1));
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                a[i][j] = grid[i - 1][j - 1];
            }
        }

        for (int i = 0; i <= n + 1; ++i) a[i][m + 1] = -1;
        for (int k = 1; k <= m; ++k) {
            int x = 1, y = k;
            bool ok = true;
            while (x <= n and ok) {
                if (y + 1 <= m and a[x][y] == 1 and a[x][y + 1] == 1) y++, x++;
                else if (y - 1 >= 1 and a[x][y] == -1 and a[x][y - 1] == -1)y--, x++;
                else ok = false;
            }
            if (ok) ans.push_back(y - 1);
            else ans.push_back(-1);
        }
        return ans;
    }
};
