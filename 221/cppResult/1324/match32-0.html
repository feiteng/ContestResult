class Solution {
public:
    vector<int> findBall(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int>vec;
        for(int i=0;i<n;i++)
        {
            int posx = 0; 
            int posy = i;
            int ans=-1;
            while(posx!=m)
            {
                if(grid[posx][posy]==1)
                {
                    if(posy==n-1)
                    {
                        ans=-1;
                        break;
                    }
                    else if(grid[posx][posy+1]==-1)
                    {
                        ans=-1;
                        break;
                    }
                    else 
                    {
                        posx++;
                        posy++;
                    }
                }
                else
                {
                    if(posy==0)
                    {
                        ans=-1;
                        break;
                    }
                    else if(grid[posx][posy-1]==1)
                    {
                        ans=-1;
                        break;
                    }
                    else 
                    {
                        posx++;
                        posy--;
                    }
                }
                // if(i==0)
                //     cout<<posx<<" "<<posy<<endl;
            }
            if(posx==m)
                ans = posy;
            vec.push_back(ans);
        }
        return vec;
    }
};
