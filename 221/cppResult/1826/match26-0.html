class Solution {
public:
    struct node{
        node* s[2];
        node(){s[0]=s[1]=nullptr;}
    }root;
    void insert(int x){
        node* cur = &root;
        for(int i=29;i>=0;i--){
            int t = bool(x&(1<<i));
            if(cur->s[t]==nullptr)
                cur->s[t] = new node;
            cur = cur->s[t];
        }
    }
    int query(int x){
        node* cur = &root;
        int ans = 0;
        for(int i=29;i>=0;i--){
            int t = bool(x&(1<<i));
            if(cur->s[t^1]!=nullptr){
                cur = cur->s[t^1];
                ans |= 1<<i;
            }else{
                cur = cur->s[t];
            }
        }
        return ans;
    }
    vector<int> maximizeXor(vector<int>& a, vector<vector<int>>& b) {
        int n = a.size();
        int q = b.size();
        for(int i=0;i<q;i++)
            b[i].push_back(i);
        sort(b.begin(),b.end(),[](const vector<int>& a,const vector<int>& b)->bool{
            return a[1]<b[1];
        });
        sort(a.begin(),a.end());
        int ai = 0;
        for(int i=0;i<q;i++){
            while(ai<n&&a[ai]<=b[i][1])
                insert(a[ai++]);
            if(ai==0)
                b[i].push_back(-1);
            else
                b[i].push_back(query(b[i][0]));
        }
        std::sort(b.begin(),b.end(),[](const vector<int>& a,const vector<int>& b)->bool{
            return a[2]<b[2];
        });
        vector<int> ans;
        for(int i=0;i<q;i++){
            ans.push_back(b[i][3]);
        }
        return ans;
    }
};
