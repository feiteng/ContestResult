class Trie
{
public:
    vector<vector<int>> t;
    int p = 0;
    Trie(int N = 0) : t(N, vector<int>(2, 0)) {}
    void insert(int x)
    {
      int root = 0;
      for (int i = 31; i >= 0; i--)
      {
        int bit = (x >> i) & 1;
        if (!t[root][bit])
        {
          t[root][bit] = ++p;
        }
        root = t[root][bit];
      }
    }

    int maxXor(int x)
    {
      int res = 0;
      int root = 0;
      for (int i = 31; i >= 0; i--)
      {
        int bit = (x >> i) & 1;
        if (t[root][!bit])
        {
          root = t[root][!bit];
          res += (1 << i);
        }
        else
        {
          root = t[root][bit];
        }
      }
      return res;
    }
};
 
 
struct node
{
    int count;
    node *bit[2];
    node()
    {
        count = 0;
        bit[0] = bit[1] = NULL;
    }
};

void insert(int s, node *root)
{
    node *cur = root;
    int x;
    for (int i = 31; i >= 0; i--)
    {
        if (s & (1 << i))
            x = 1;
        else
            x = 0;
        if (cur->bit[x] == NULL)
        {
            cur->bit[x] = new node();
        }
        cur = cur->bit[x];
        cur->count++;
    }
}

int search(int s, node *root)
{
    node *cur = root;
    int y = 0, x;
    for (int i = 31; i >= 0; i--)
    {
        if (s & (1 << i))
            x = 1;
        else
            x = 0;
        if (cur->bit[1 - x] != NULL and cur->bit[1 - x]->count)
        {
            cur = cur->bit[1 - x];
            y += 1 << i;
        }
        else
        {
            cur = cur->bit[x];
        }
    }
    return y;
}

class Solution {
public:
    static bool cmp(const vector<int>& a, const vector<int>& b) {
        return a[1] < b[1];
    }

    vector<int> maximizeXor(vector<int>& a, vector<vector<int>>& q) {
        int n = a.size();
        sort(a.begin(),a.end());
        for(int i=0;i<q.size();i++) {
            q[i].push_back(i);
        }
        sort(q.begin(),q.end(),cmp);
        int j = 0;
        vector<int> ans(q.size(),-1);
        // Trie trie(n*32);
        node *root = new node();
        for(int i=0;i<q.size();i++) {
            while(j < a.size() and a[j] <= q[i][1]) {
                // trie.insert(a[j]);
                insert(a[j], root);
                j++;
            }
            // cout<<q[i][2]<<" "<<j<<endl;
            if(j == 0 or a[0]>q[i][1]) continue;
            // else ans[q[i][2]] = trie.maxXor(q[i][0]);
            else ans[q[i][2]] = search(q[i][0],root);
        }
        return ans;
    }
};
