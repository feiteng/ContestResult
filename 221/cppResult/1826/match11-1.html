const int maxn=1e5+5;
int trie[32*maxn][2];

int val[32*maxn];
int sz;
void ins(int x){
    int cnt=0;
    for(int i=31;i>=0;i--){
        int u=(x>>i)&1;
        if(!trie[cnt][u]){
            trie[sz][0]=trie[sz][1]=0;
            val[sz]=0;
            trie[cnt][u]=sz++;
        }
        cnt=trie[cnt][u];
    }
    val[cnt]=x;
}
void init(){
    sz=1;
    trie[0][0]=trie[0][1]=0;

}
int query(int x){
    int cnt=0;
    int ret=0;
    for(int i=31;i>=0;i--){
        int u=(x>>i)&1;
        if(trie[cnt][u^1]){
            ret+=(1<<i);
            cnt=trie[cnt][u^1];
        }
        else cnt=trie[cnt][u];
    }
    return ret;
}
struct node{
    int val,limit,id;
};
bool cmp(node& n1,node& n2){
    return n1.limit<n2.limit;
}
node q[maxn];
class Solution {
public:
    vector<int> maximizeXor(vector<int>& nums, vector<vector<int>>& qe) {
        int n=nums.size(),m=qe.size();
        sort(nums.begin(),nums.end());
        for(int i=0;i<m;i++){
            q[i].val=qe[i][0];
            q[i].limit=qe[i][1];
            q[i].id=i;
        }
        sort(q,q+m,cmp);

        vector<int>ret(m);
        int idx=0;
        init();
        for(int i=0;i<m;i++){
            while(idx<n&&nums[idx]<=q[i].limit){
                ins(nums[idx]);
                idx++;
            }
            if(idx==0) ret[q[i].id]=-1;
            else ret[q[i].id]=query(q[i].val);
        }
        return ret;
    }
};
