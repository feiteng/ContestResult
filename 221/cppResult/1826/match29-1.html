class Solution {
    //int maxn = 100010;
    int ch[32*100010][2];
    int val[32*100010];
    int nodecnt;
    void init(){
        nodecnt = 1;
        ch[0][1] = ch[0][0] = 0;

    }
    
    void tire(int x){
        int cnt=0;
        for(int i=31;i>=0;i--){
            int v=(x>>i)&1;
            if(!ch[cnt][v]){
                ch[nodecnt][0]=ch[nodecnt][1]=0;
                val[nodecnt]=0;
                ch[cnt][v]=nodecnt++;
            }
            cnt=ch[cnt][v];
        }
        val[cnt]=x;
    }
    
    int find(int x){
        int cnt = 0;
        for (int i = 31; i>=0; --i){
            int v = (x>>i)&1;
            if(ch[cnt][v^1])cnt = ch[cnt][v^1];
            else cnt=ch[cnt][v];
        }
        return val[cnt];
    }
public:
    vector<int> maximizeXor(vector<int>& nums, vector<vector<int>>& queries) {
        vector<int> q(queries.size());
        iota(q.begin(), q.end(), 0);
        sort (q.begin(), q.end(), [&] (int i , int j) { return queries[i][1] < queries[j][1];});
        init();
        int n =nums.size();
        sort(nums.begin(), nums.end());
        vector<int>ans (queries.size());
        int cur = 0;
        for(int qid: q){
            auto que = queries[qid];
            int limit = que[1];
            while (cur < n && nums[cur] <= limit){
                tire(nums[cur]);
                cur++;
            }
            ans[qid] = cur==0 ? -1 :  que[0]^ find(que[0]);
        }
        return ans;
    }
};
