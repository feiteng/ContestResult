class Solution {
    int sz = 0;
    int tree[30*100005][2];

public:
    void insert(int& x){
        int p = 0;
        for(int i = 29; i >= 0; --i){
            int q = (x >> i) & 1;
            if(!tree[p][q]){
                tree[p][q] = ++sz;

            }
            p = tree[p][q];
        }
    }
    int query(int& x){
        int sum = 0, p = 0;
        for(int i = 29; i >= 0; --i){
            int q = (x >> i) & 1;
            if(tree[p][!q]){
                sum = sum + (1 << i);
                p = tree[p][!q];
            }else p = tree[p][q];
        }
        return sum;
    }
    vector<int> maximizeXor(vector<int>& ns, vector<vector<int>>& q) {
        memset(tree, 0, sizeof tree);
        sort(ns.begin(), ns.end());
        int n = ns.size(), m = q.size();
        for(int i = 0; i < m; ++i){
            auto &v = q[i];
            v.push_back(i);
        }
        sort(q.begin(), q.end(), [](const vector<int> &a, const vector<int> &b){
            return a[1] < b[1];
        });
        
        vector<int>ans(m);
        int idx = 0;
        for(int i = 0; i < m; ++i){
            while(idx < n && q[i][1] >= ns[idx]){
                insert(ns[idx]);
                idx++;
            }
            int res = -1;
            if(idx >  0) res = query(q[i][0]);
            ans[q[i][2]] = res;
        }
        return ans;
    }
};
