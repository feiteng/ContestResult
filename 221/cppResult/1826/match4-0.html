class Solution {
private:
    vector<int>v[2];
    int cnt;
    void add(int x){
        int cur = 0;
        for(int i=30;i>=0;i--){
            int tt = (x&(1<<i))?1:0;
            if(v[tt][cur]==-1){
                v[tt][cur] = cnt++;
                v[0].push_back(-1);
                v[1].push_back(-1);
            }
            cur = v[tt][cur];
        }
    }
    int query(int x){
        int ret = 0;
        int cur = 0;
        for(int i=30;i>=0;i--){
            int tt = (x&(1<<i))?1:0;
            if(v[1-tt][cur]!=-1){
                ret|=(1<<i);
                cur = v[1-tt][cur];
            } else {
                cur=v[tt][cur];
            }
        }
        return ret;
    }
public:
    vector<int> maximizeXor(vector<int>& a, vector<vector<int>>& q) {
        v[0].clear();
        v[1].clear();
        v[0].push_back(-1);
        v[1].push_back(-1);
        cnt=1;
        int m = q.size();
        int n = a.size();
        sort(a.begin(),a.end());
        vector<int>idx = vector<int>(m);
        for(int i=0;i<m;i++)idx[i] = i;
        sort(idx.begin(),idx.end(), [&q](int x,int y){
            return q[x][1]<q[y][1];
        });
        vector<int>ret = vector<int>(m);
        int r = 0;
        for(int i=0;i<m;i++){
            while(r<n&&a[r]<=q[idx[i]][1]){
                add(a[r]);
                r++;
            }
            if(r==0)ret[idx[i]]=-1;
            else ret[idx[i]] = query(q[idx[i]][0]);
        }
        return ret;
    }
};
