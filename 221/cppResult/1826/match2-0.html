class Solution {
public:
    struct node{
        int a,b;
    }e[100007*20];
    int tot=1;

    void build(int x)
    {
        int p=1;
        for(int i=30;i>=0;i--) {
            int tmp=x&(1<<i);
            if(tmp)
                if(!e[p].a)
                    e[p].a=++tot,p=tot;
                else p=e[p].a;
            else
                if(!e[p].b) e[p].b=++tot,p=tot;
                else p=e[p].b;
        }
    }

    int query(int x)
    {
        int p=1;
        int ans=0;
        for(int i=30;i>=0;i--) {
            int tmp=x&(1<<i);
            if(!tmp) {
                if(e[p].a) p=e[p].a,ans+=(1<<i);
                else p=e[p].b;
            }
            else {
                if(e[p].b)	p=e[p].b;
                else p=e[p].a,ans+=(1<<i);
            }
        }
        return ans^x;
    }
    
    static bool mycomp(vector<int> &v1, vector<int> &v2){
        return v1[1] < v2[1];
    }
    
    vector<int> maximizeXor(vector<int>& nums, vector<vector<int>>& queries) {
        for(int i=0; i<queries.size(); i++){
            queries[i].push_back(i);
        }
        sort(nums.begin(), nums.end());
        sort(queries.begin(), queries.end(), mycomp);
        
        vector<int> res(queries.size(), 0);
        int tn = 0;
        for(int tq = 0; tq < queries.size(); tq++){
            while(tn < nums.size() && nums[tn] <= queries[tq][1]){
                build(nums[tn]);
                tn++;
            }
            if(tn == 0)
                res[queries[tq][2]] = -1;
            else
                res[queries[tq][2]] = query(queries[tq][0]);
        }
        
        return res;
        
    }
};
