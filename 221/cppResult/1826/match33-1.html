const int N = 4000010;
int a[N], son[N][2], idx, minn[N]; // idx -> min

void insert(int x){
    int p=0;
    for(int i=30;i>=0;i--){
        int num=x>>i&1;
        if(!son[p][num]) son[p][num]=++idx;
        p=son[p][num];
        minn[p]=min(minn[p], x);
    }
}

int query(int x, int maxn){

    int p=0, ans=0;
    for(int i=30;i>=0;i--){
        int num=x>>i&1;
        if(son[p][1-num]&&maxn>=minn[son[p][1-num]]){
            ans+=1<<i;
            p=son[p][1-num];
        }else if(son[p][num]&&maxn>=minn[son[p][num]]){
            p=son[p][num];
        }else return -1;
    }
    // if(ans==0) return -1;
    return ans;
}

class Solution {
public:
    vector<int> maximizeXor(vector<int>& nums, vector<vector<int>>& q) {
        for(int i=0;i<N;i++) minn[i]=INT_MAX;
        for(int i=0;i<nums.size();i++){
            insert(nums[i]);
        }
        vector<int> ans; ans.clear();
        for(int i=0;i<q.size();i++){
            ans.push_back(query(q[i][0], q[i][1]));        
        }
        return ans;    
    }
};
