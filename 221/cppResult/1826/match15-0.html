class Solution {
public:
    int trie[100005 * 32][2], idx;
    void insert(int x)
    {
        int p = 1;
        for(int i = 30; i >= 0; i--){
            int u = x >> i & 1;
            if (!trie[p][u]) {
                idx++;
                trie[idx][0] = trie[idx][1] = 0;
                trie[p][u] = idx;
            }
            p = trie[p][u];
        }
    }
    int search(int x)
    {
        int p = 1, ans = 0;
        for (int i = 30; i >= 0; i--){
            int u = x >> i & 1;
            if (trie[p][u ^ 1]) {
                p = trie[p][u ^ 1];
                ans += (1 << i);
            }
            else p = trie[p][u];
        }
        return ans;
    }
    
    vector<int> maximizeXor(vector<int>& a, vector<vector<int>>& q) {
        idx = 1;
        trie[1][0] = trie[1][1] = 0;
        vector <pair <pair <int, int>, int> > qs;
        for (int i = 0; i < q.size(); i++) {
            qs.push_back(make_pair(make_pair(q[i][1], q[i][0]), i));
        }
        vector <int> ans;
        ans.resize(q.size());
        sort(qs.begin(), qs.end());
        sort(a.begin(), a.end());
        int apos = 0;
        for (int i = 0; i < qs.size(); i++) {
            int pos = qs[i].second;
            int mx = qs[i].first.first;
            int now = qs[i].first.second;
            while (apos < a.size() && a[apos] <= mx) {
                insert(a[apos]);
                apos++;
            }
            if (apos == 0) {
                ans[pos] = -1;
            } else {
                ans[pos] = search(now);
            }
        }
        return ans;
    }
};
