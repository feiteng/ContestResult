class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int ans = 0;
        priority_queue< pair<int, int> > que;
        int n = apples.size();
        for(int i=0;i<n;++i)
        {
            que.push(make_pair(-i-days[i]+1, apples[i]));
            while(!que.empty() && (- que.top().first < i || !que.top().second))
                que.pop();
            if(!que.empty())
            {
                ++ans;
                auto tmp = que.top();
                que.pop();
                --tmp.second;
                if(tmp.second)
                    que.push(tmp);
            }
        }
        int i = n;
        while(!que.empty())
        {
            if(-que.top().first >= i)
            {
                int tmp = min(-que.top().first - i + 1, que.top().second);
                ans += tmp;
                i += max(tmp, 1);
            }
            que.pop();
        }
        return ans;
    }
};
