class Solution {
public:
    int eatenApples(vector<int>& a, vector<int>& d) {
        int n = a.size();
        typedef pair<int, int> pii;
        multiset<pii> se;
        int r = 0;
        for (int i = 0; i < n; i++) {
            if(a[i] > 0) {
                se.emplace(i + d[i], a[i]);
            }
            while(!se.empty() && se.begin()->first <= i) {
                se.erase(se.begin());
            }
            if(!se.empty()) {
                ++r;
                //--se.begin()->second;
                auto t = *se.begin();
                se.erase(se.begin());
                if(--t.second > 0)
                    se.insert(t);
            }
        }
        for (int i = n; !se.empty(); i++) {
            while(!se.empty() && se.begin()->first <= i) {
                se.erase(se.begin());
            }
            if(!se.empty()) {
                ++r;
                //--se.begin()->second;
                auto t = *se.begin();
                se.erase(se.begin());
                if(--t.second > 0)
                    se.insert(t);
            }
        }
        return r;
    }
};
