class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int ans = 0;
        int n = apples.size();
        map<int, int> C;
        for(int i=0;i<n;++i) {
            if(apples[i] != 0) {
                C[i + days[i]] += apples[i];
            }
            while(!C.empty() && C.begin()->first <= i) {
                C.erase(C.begin());
            }
            
            if(!C.empty()) {
                ans++;
                C.begin()->second--;
                if(C.begin()->second == 0) {
                    C.erase(C.begin());
                }
            }
        }
        int cur = n;
        while(!C.empty()) {
            while(!C.empty() && C.begin()->first <= cur) {
                C.erase(C.begin());
            }
            if(!C.empty()) {
                auto b = *C.begin();
                C.erase(C.begin());
                cur = min(b.first, cur + b.second);
            }
        }
        return ans + (cur - n);
    }
};
