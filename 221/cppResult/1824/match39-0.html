class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
	int D = 0;
	int ans = 0;
	priority_queue<pair<int, int> > Q;
	REP (i, apples.size()) {
	    D = i;
	    if (apples[i]) {
		Q.emplace(-(i + days[i]), apples[i]);
	    }

	    while (!Q.empty() && -Q.top().first <= D) {
		Q.pop();
	    }
	    if (!Q.empty()) {
		auto p = Q.top();
		Q.pop();
		p.second--;
		ans++;
		if (p.second) Q.push(p);
	    }
	}
        
	while (!Q.empty()) {
	    D++;
	    while (!Q.empty() && -Q.top().first <= D) {
		Q.pop();
	    }
	    if (!Q.empty()) {
		auto p = Q.top();
		Q.pop();
		p.second--;
		ans++;
		if (p.second) Q.push(p);
	    }
	}

	return ans;
    }
};
