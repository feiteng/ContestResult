class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n = apples.size();
        int res = 0;
        priority_queue<pair<int, int>> pq;
        int d = 0;
        while (pq.size() > 0 || d < n) {

            if (d < n) {
                pq.push(make_pair(-d - days[d] + 1, apples[d]));
            }
            while (pq.size() > 0 && -pq.top().first < d) {
                pq.pop();
            }
            if (pq.size() > 0) {
                auto p = pq.top();
                pq.pop();
                p.second--;
                res++;
                if (p.second > 0) {
                    pq.push(p);
                }
            }
            d++;
        }
        return res;
    }
};
