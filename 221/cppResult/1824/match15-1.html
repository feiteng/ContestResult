class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int num = 0;
        int N = apples.size();
        map<int, int> rot;
        int res = 0;
        
        for(int i = 0; i < N; i++){
            rot[i + days[i]] += apples[i];

            while(!rot.empty() && (*(rot.begin())).first <= i){
                rot.erase(rot.begin());
            }
            if(!rot.empty()){
                // Eat an apple
                res++;
                (*(rot.begin())).second--;
                if((*(rot.begin())).second == 0){
                    rot.erase(rot.begin());
                }
            }
        }
        
        int curD = N;
        while(!rot.empty()){
            while(!rot.empty() && (*(rot.begin())).first <= curD){
                rot.erase(rot.begin());
            }
            if(rot.empty()) break;
            // Eat apples
            int num_app = min((*(rot.begin())).second, (*(rot.begin())).first - curD);
            res += num_app;
            curD += num_app;
            rot.erase(rot.begin());
        }
        
        
        return res;
    }
};
