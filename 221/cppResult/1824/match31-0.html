class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> m;
        
        int n = apples.size();
        
        int ans = 0;
        for (int i = 0; i < n || !m.empty(); i++) {
            if (i < n && days[i] != 0) {
                m[i + days[i]] += apples[i];
            }
            
            while (!m.empty() && (m.begin()->first <= i || m.begin()->second == 0)) {
                m.erase(m.begin());
            }
            
            if (!m.empty()) {
                ans++;
                m.begin()->second--;
            }
            
        }
        
        return ans;
    }
};
