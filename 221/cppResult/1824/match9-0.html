class Apple{
public:
    int day;
    int num;
    Apple(int a,int b){
        day=a;
        num=b;
    }
    bool operator <(const Apple x)const{
        return day>x.day;
    }
};

class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int ans=0;
        int a=0,b=0;
        int n=apples.size();
        priority_queue<Apple>q;
        for(int i=0;i<n;i++){
            while(!q.empty()&&q.top().day<=i)
                q.pop();
            a=apples[i];
            b=i+days[i];
            if(a!=0)
                q.push(Apple(b,a));
            if(!q.empty()){
                ans++;
                Apple temp=q.top();
                q.pop();
                temp.num--;
                if(temp.num>0)
                    q.push(temp);
            }
        }
        int i=n;
        while(!q.empty()){
            while(!q.empty()&&q.top().day<=i)
                q.pop();
            if(!q.empty()){
                ans++;
                Apple temp=q.top();
                q.pop();
                temp.num--;
                if(temp.num>0)
                    q.push(temp);
            }
            i++;
        }
        return ans;
    }
};
