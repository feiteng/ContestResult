class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n = days.size();
        
        int ret = 0;
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq;
        for (int i = 0;i < n; i++)
        {
            if (apples[i] != 0)
                pq.push(make_pair(i+days[i], apples[i]));
            
            while (!pq.empty())
            {
                auto top = pq.top();
                pq.pop();
                
                if (i < top.first && top.second > 0)
                {
                    ret++;
                    top.second -= 1;
                    
                    if (top.second > 0)
                        pq.push(top);
                    break;
                }
            }
        }
        
        int now_day = n;
        while (!pq.empty())
        {
            auto top = pq.top();
            pq.pop();
            if (top.second == 0 || top.first <= now_day)
                continue;
            
            int tmp = min(top.first - now_day, top.second);
            ret += tmp;
            now_day += tmp;
        }
        
        return ret;
    }
};
