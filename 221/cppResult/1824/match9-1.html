struct Node {
    int num, bad;

    Node(int num, int bad): num(num), bad(bad) {}
    bool operator < (const Node &n) const {
        return bad > n.bad;
    }
};

class Solution {
public:
    priority_queue<Node> que;
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int ans = 0, n = apples.size();
        for(int i = 0; i < n; i ++) {
            que.push(Node(apples[i], i + days[i]));
            while(!que.empty() && que.top().bad <= i) que.pop();
            //if(apples[i] == 0) continue;
            //que.push(Node(apples[i], i + days[i]));

            if(!que.empty() && que.top().num > 0) {
                Node node = que.top();
                que.pop();
                ans ++;
                node.num --;
                if(node.num > 0) que.push(node);
            }
        }
        int day = n;
        while(!que.empty()) {
            while(!que.empty() && que.top().bad <= day) que.pop();
            if(!que.empty() && que.top().num > 0) {
                Node node = que.top();
                que.pop();
                ans ++;
                node.num --;
                if(node.num > 0) que.push(node);
            }
            day ++;
        }
        return ans;
    }
};
