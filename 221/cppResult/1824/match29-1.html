class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days)
    {
        int n = apples.size();
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        int ret = 0;
        for(int i = 0; i < n; i++)
        {
            if(apples[i] > 0)
                pq.push({i + days[i], apples[i]});
            while(!pq.empty() && pq.top().first <= i)
            {
                pq.pop();
            }
            if(!pq.empty())
            {
                ret++;
                auto t = pq.top();
                pq.pop();
                t.second--;
                if(t.second >0)
                    pq.push(t);
            }
        }
        
       // cout << ret << " " << pq.size() << endl;
        
        int i = n;
        while(!pq.empty())
        {
            auto [day, cnt] = pq.top();
            //cout << day << " " << i << " "<< cnt << endl;
            pq.pop();
            if(day <= i)
                continue;
            if(day - i > cnt)
            {
                i = i + cnt;
                ret += cnt;
            }
            else
            {
                
                ret += day - i;
                i = day;
            }
        }
        //cout << endl;
        return ret;
    }
};
