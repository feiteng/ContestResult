class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        
        typedef pair<int, int> S;  // <rut, cnt>
        priority_queue<S, vector<S>, greater<S>> pq;
        
        int n = days.size(), eat = 0;

        for(int day = 0; day < n or pq.size(); day++) {
        
            if(day < n and apples[day] > 0) {
                pq.push({day + days[day], apples[day]});
            }
            
            while(pq.size() and pq.top().first <= day) {
                pq.pop();
            }
            
            if(pq.size()) {
                eat++;
                auto p = pq.top();
                pq.pop();
                if(p.second > 1) {
                    pq.push({p.first, p.second -1});
                }
            }

        }
        return eat;
    }
};
