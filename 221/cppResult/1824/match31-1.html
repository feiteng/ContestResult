class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> mp;
        int n=apples.size();
        int res=0;
        for (int i=0; i<n; ++i) {
            if (days[i]>0) {
                mp[days[i]+i]+=apples[i];
            }
            while (!mp.empty()) {
                auto v=mp.begin();
                if (v->first>i) {
                    res++;
                    mp[v->first]--;
                    if (mp[v->first]==0) {
                        mp.erase(v->first);
                    }
                    break;
                }
                mp.erase(v->first);
            }
        }
        int i=n;
        while (!mp.empty()) {
            auto v=mp.begin();
            if (v->first>i) {
                res++;
                i++;
                mp[v->first]--;
                if (mp[v->first]==0) {
                    mp.erase(v->first);
                }
                continue;
            } else {
                mp.erase(v->first);
            }
        }
        return res;
    }
};
