class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {

        const int n = apples.size();
        int ans = 0;
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> q;
        
        for (int i = 0; i < n; i++) {
            if (apples[i] > 0)
                q.push(make_pair(i + days[i], apples[i]));
            
            while (!q.empty()) {
                auto u = q.top();
                
                q.pop();
                if (u.first <= i || u.second == 0)
                    continue;
                
                ans++;
                u.second--;
                q.push(u);
                break;

            }
        }
        
        int t = n;
        while (!q.empty()) {
            auto u = q.top();
            q.pop();
            if (u.first <= t)
                continue;
            
            int x = min(u.first - t, u.second);
            t += x;
            ans += x;
        }
        return ans;
    }
};
