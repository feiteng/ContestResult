struct cmp{
    bool operator()(const pair<int,int>&a, const pair<int,int>&b){
        // first: num, second: end
        return a.second > b.second;
    }
};

class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<pair<int,int>, vector<pair<int,int>>, cmp> pq;
        pair<int,int> now;
        int ans=0;
        for(int i=0;i<days.size();i++){
            if(days[i] > 0) pq.push(pair<int,int>(apples[i], days[i]+i-1));
            while(pq.size()>0 && pq.top().second<i){
                pq.pop();
            }
            if(pq.size()==0) continue;
            now = pq.top();
            pq.pop();
            ans++;
            now.first--;
            if(now.first > 0) pq.push(now);
        }
        for(int i=days.size();pq.size()>0;i++){
            while(pq.size()>0 && pq.top().second<i){
                pq.pop();
            }
            if(pq.size()==0) break;
            now = pq.top();
            pq.pop();
            ans++;
            now.first--;
            if(now.first > 0) pq.push(now);
        }
        return ans;
    }
};
