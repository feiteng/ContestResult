class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> s;
        int res = 0;
        int cur = 0;
        for (int i = 0; i < apples.size(); ++i) {
            if (apples[i] != 0) {
                s[i + days[i]] += apples[i];
            }
            while (cur <= i) {
                // cout << "cur: " << cur << endl;
                // for (auto [k, v] : s) {
                //     cout << k << ", " << v << "; ";
                // }
                // cout << endl;
                if (s.empty()) ++cur;
                while (!s.empty() && s.begin()->first <= cur) {
                    s.erase(s.begin());
                }
                if (!s.empty() && s.begin()->first > cur) {
                    int temp = min(min(s.begin()->first - cur, s.begin()->second), i - cur + 1);
                    res += temp;
                    s.begin()->second -= temp;
                    cur += temp;
                    if (s.begin()->second == 0) s.erase(s.begin());
                    // cout << "cur: " << cur << ", temp: " << temp  << endl;
                }
            }
            // cout << res << endl;
        }
        
        while (!s.empty()) {
            while (!s.empty() && s.begin()->first <= cur) {
                s.erase(s.begin());
            }
            if (!s.empty() && s.begin()->first > cur) {
                int temp = min(s.begin()->first - cur, s.begin()->second);
                res += temp;
                s.begin()->second -= temp;
                cur += temp;
                if (s.begin()->second == 0) s.erase(s.begin());
                // cout << "cur: " << cur << ", temp: " << temp  << endl;
            }
        }
        
        return res;
    }
};
