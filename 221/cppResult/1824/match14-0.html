class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int curr_day = 0;
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> que;
        int res = 0;
        for (; curr_day < days.size(); curr_day++) {
            int apps = apples[curr_day];
            int rot = days[curr_day];
            que.push({ curr_day + rot - 1, apps });
            while (!que.empty() && que.top().first < curr_day) que.pop();
            if (!que.empty()) {
                res++;
                auto top = que.top();
                que.pop();
                top.second--;
                if (top.second != 0) que.push(top);
            }
        }
        while (!que.empty()) {
            while (!que.empty() && que.top().first < curr_day) que.pop();
            if (!que.empty()) {
                res++;
                auto top = que.top();
                que.pop();
                top.second--;
                if (top.second != 0) que.push(top);
            }
            curr_day++;
        }
        return res;
    }
};
