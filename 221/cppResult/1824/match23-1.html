class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n=apples.size();
        priority_queue<p,vector<p>,greater<p>> pq;
        int res=0;
        F(i,n) {
            if (i<n&&days[i]) pq.emplace(i+days[i],apples[i]);
            while (!pq.empty()&&pq.top().first<=i) pq.pop();
            if (!pq.empty()) {
                p cur=pq.top();pq.pop();

                res++;
                cur.second--;
                if (cur.second) pq.push(cur);
            }
        }
        int i=n;
        while (!pq.empty()) {
            while (!pq.empty()&&pq.top().first<=i) pq.pop();
            if (!pq.empty()) {
                p cur=pq.top();pq.pop();
                res++;
                cur.second--;
                if (cur.second) pq.push(cur);
            }
            i++;
        }
        return res;
    }
};
