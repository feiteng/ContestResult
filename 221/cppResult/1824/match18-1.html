class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n = days.size(), ans = 0;
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq;
        for (int i = 0; i < n; ++i) {
            // add apples
            if (apples[i] > 0)
                pq.push({i+days[i], apples[i]});
            
            // try to eat
            while (!pq.empty() && (pq.top().second == 0 || pq.top().first <= i)) pq.pop();
            if (!pq.empty()) {
                ++ans;
                pair<int,int> x = pq.top();
                pq.pop();
                if (x.second-1 > 0)
                    pq.push({x.first, x.second-1});
            }
        }
        
        int day = n;
        while (!pq.empty()) {
            while (!pq.empty() && (pq.top().second == 0 || pq.top().first <= day)) pq.pop();
            if (!pq.empty()) {
                ++ans;
                pair<int,int> x = pq.top();
                pq.pop();
                if (x.second-1 > 0)
                    pq.push({x.first, x.second-1});
            }
            ++day;
        }
        return ans;
    }
};
