class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> table;

        int ret = 0;
        for (int i = 0; i < apples.size(); i++) {
            if (apples[i] != 0) {
                table[i+days[i]-1] += apples[i];
            }
            while (!table.empty() && table.begin()->first < i) table.erase(table.begin());
            if (!table.empty()) {
                ret++;
                table.begin()->second--;
                if (table.begin()->second == 0) table.erase(table.begin());

            }
        }
        int i = apples.size();
        while (!table.empty()) {
            while (!table.empty() && table.begin()->first < i) table.erase(table.begin());
            if (!table.empty()) {
                int tmp = table.begin()->first - i + 1;
                tmp = min(tmp, table.begin()->second);
                ret += tmp;
                i += tmp;
                table.erase(table.begin());
            }
            
        }
        return ret;
        
        
    }
};
