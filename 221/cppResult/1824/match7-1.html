class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> mp;
        int cnt = 0;
        int res = 0;
        for(int i=0; i<apples.size(); i++){
            if(apples[i] > 0){
                mp[cnt + days[i]] += apples[i];
            }
            while(!mp.empty() && mp.begin()->first <= cnt){
                mp.erase(mp.begin());
            }
            if(!mp.empty()){
                auto it = mp.begin();
                it->second = it->second - 1;
                res ++;
                if(it->second == 0)
                    mp.erase(it);
            }
            cnt++;
        }
        while(!mp.empty()){
            while(!mp.empty() && mp.begin()->first <= cnt){
                mp.erase(mp.begin());
            }
            if(!mp.empty()){
                auto it = mp.begin();
                it->second = it->second - 1;
                res ++;
                if(it->second == 0)
                    mp.erase(it);
            }
            cnt++;
        }
        return res;
    }
};
