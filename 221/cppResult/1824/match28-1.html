        int n = days.size(), res = 0;
        /*

            cout << ll << " " << rr << endl;

        set<pair<int, int>> st;
        for (int i = 0; i < n || st.size(); ++i) {
            if (i < n && apples[i]) {
                st.insert({i+days[i]-1, apples[i]});
            }

            while (st.size() && st.begin()->first < i) st.erase(st.begin());
            if (st.size()) {
                auto p = *st.begin();
                st.erase(st.begin());
                ++res;
                if (--p.second) st.insert(p);
            }
        }
        return res;
    }
};
