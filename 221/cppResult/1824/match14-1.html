class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n = apples.size();
        priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> q;
        int i=0, res=0;
        while(i<n || q.size()){
            while(q.size() && q.top().first<i){
                q.pop();
            }
            if(i<n && apples[i])
                q.push({i+days[i]-1, apples[i]});

            if(q.size()){
                res++;
                auto p = q.top(); q.pop();
                p.second--;
                if(p.second)
                    q.push(p);
            }
            i++;
        }
        return res;
    }
};
