class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        const int n = apples.size();
        map<int, int> cnts;
        int result = 0;
        for (int i = 0; i < n || !cnts.empty(); ++i) {
            if (i < n && apples[i] > 0) {
                cnts[i + days[i]] += apples[i];
            }
            if (!cnts.empty() && cnts.begin()->first <= i) {
                cnts.erase(cnts.begin());
            }
            if (!cnts.empty()) {
                ++result;
                auto itr = cnts.begin();
                if (--itr->second == 0) {
                    cnts.erase(itr);
                }
            }
        }
        return result;
    }
};
