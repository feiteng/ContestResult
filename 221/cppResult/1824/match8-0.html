using PII = pair<int, int>;

class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<PII, vector<PII>, greater<>> pq;
        int n = apples.size();
        int ans = 0;
        for (int i = 0; i < n || !pq.empty(); ++i) {
            if (i < n) {
                int apple = apples[i];
                int lastDay = i + days[i] - 1;

                if (apple > 0) {
                    pq.emplace(lastDay, apple);
                }
            }

            while (!pq.empty() && pq.top().first < i) {
                pq.pop();
            }

            if (!pq.empty()) {
                auto [topLastDay, topApples] = pq.top(); pq.pop();
                ++ans;
                if (topLastDay > i && topApples > 1) {
                    pq.emplace(topLastDay, topApples - 1);
                }
            }
        }
        return ans;
    }
};
