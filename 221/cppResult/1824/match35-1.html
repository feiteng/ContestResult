class Solution
{
public:

    int eatenApples(vector<int>& apples, vector<int>& days)
    {
        int ans = 0;
        int n = sz(apples);
        // vi cnt(4e4 + 10);

        // int mx = 1;
        multiset<pii> st;

        for (int i = 0; i < n; i++)
        {
            if (apples[i] != 0)
            {
                int cur_ed = i + days[i] - 1;
                int c = apples[i];
                st.insert({cur_ed, c});
            }

            debug(i, st);

            while (!st.empty())
            {
                auto h = *st.begin();
                st.erase(st.begin());

                int ed = h.first;
                int cnt = h.second;
                if (ed < i) continue;

                ans++, cnt--;
                if (cnt >= 1) st.insert({ed, cnt});
                break;
            }
        }

        for (int i = n; i < 40010; i++)
        {
            if (sz(st) == 0) break;

            while (!st.empty())
            {
                auto h = *st.begin();
                st.erase(st.begin());

                int ed = h.first;

                int cnt = h.second;
                if (ed < i) continue;

                ans++, cnt--;
                if (cnt >= 1) st.insert({ed, cnt});
                break;
            }
        }

        return ans;
    }
};
