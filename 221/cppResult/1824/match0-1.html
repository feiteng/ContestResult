class Solution {
public:
    struct P{
        int cnt;
        int day;
        bool operator<(const P& item) const
        {
            return day > item.day;
        }
    };
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<P> q;
        int ans = 0, n = apples.size(), i = 0;
        q.push(P{apples[0], days[0]});
        while(i<n || !q.empty()) {
            if(!q.empty()){
                P p = q.top();
                q.pop();
                if (p.cnt >= 1){
                    ans++;
                } 
                if (p.cnt - 1 > 0) {
                    q.push(P{p.cnt-1, p.day});
                }
            }
            i++;
            if(i < n) {
                q.push(P{apples[i], i + days[i]});
            }
            while(!q.empty() && q.top().day <= i) {
                q.pop();
            }
        }
        return ans;
    }
};
