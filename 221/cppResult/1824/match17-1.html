class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> m;
        int res = 0;
        int l = (*max_element(days.begin(), days.end())) + apples.size();
        // cout << l << endl;
        for (int i = 0; i < l; i++) {
            if (i < apples.size()) m[i + days[i]] += apples[i];
            else if (m.empty()) break;
            auto it = m.begin();
            // if (it != m.end()) cout << it->first << " " << it->second << endl;
            while (it != m.end()) {
                if (it->first <= i) it = m.erase(it);
                else break;

            }
            if (!m.empty()) {
                auto it = m.begin();
                if (--it->second == 0) {
                    m.erase(it);
                }
                res++;
            }
            // cout << i << " " << res << endl;
        }
        
        return res;
