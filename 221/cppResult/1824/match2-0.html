class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> rot;
        int ans = 0;
        for (int i = 0; i < 40100; ++i) {
            if (i < apples.size() && apples[i]) {
                rot[i+days[i]] += apples[i];
            }
            while (!rot.empty() && i >= rot.begin()->first) {
                rot.erase(rot.begin());
            }
            if (!rot.empty()) {
                rot.begin()->second--;
                if (rot.begin()->second == 0) rot.erase(rot.begin());
                ++ans;
            }
        }
        return ans;
    }
};
