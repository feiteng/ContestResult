class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int,int> m;
        int cnt = 0;
        int i = 0;
        while(i < apples.size() || !m.empty())
        {
            if(i < apples.size() && apples[i] != 0)
                m[i+days[i]] += apples[i];
            while(m.begin() != m.end() && m.begin()->first <= i)
            {
                m.erase(m.begin());
            }
            if(m.begin() != m.end())
            {
                auto it = m.begin();
                it->second --;
                cnt ++;
                if(it->second == 0)
                {
                    m.erase(it);
                }
            }
            ++i;
        }
        return cnt;
    }
};
