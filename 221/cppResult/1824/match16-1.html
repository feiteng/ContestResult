class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> ma;
        int res = 0, i;

        for(i=0;i<apples.size();i++) {
            ma[i+days[i]] += apples[i];
            while(!ma.empty() && (ma.begin()->second==0 || ma.begin()->first<=i)) {
                ma.erase(ma.begin());
            }
            if(!ma.empty()) {
                res++;
                ma.begin()->second--;
            }
        }
        // cout<<res<<endl;
        while(!ma.empty()) {
            while(!ma.empty() && (ma.begin()->second==0 || ma.begin()->first<=i)) {
                ma.erase(ma.begin());
            }
            if(!ma.empty()) {
                res++; ma.begin()->second--;
            }
            i++;
        }
        return res;
    }
};
