class Solution {
public:
    int eatenApples(vector<int>& app, vector<int>& days) {
        int n=app.size();
        map<int,int> mp;
        int res=0;
        for(int i=0;i<n;++i){
            if(app[i]!=0){
                mp[i+days[i]]+=app[i];
            }

            while(!mp.empty()&&i>=(*mp.begin()).first) mp.erase(mp.begin());
            if(!mp.empty()){
                ++res;
                auto cur=mp.begin();
                --((*cur).second);
                if(((*cur).second)==0) mp.erase(cur);
            }
        }
        int i=n;
        while(!mp.empty()){
            while(!mp.empty()&&i>=(*mp.begin()).first) mp.erase(mp.begin());
            if(!mp.empty()){
                ++res;
                auto cur=mp.begin();
                --((*cur).second);
                if(((*cur).second)==0) mp.erase(cur);
            }
            ++i;
        }
        return res;
    }
};
