class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {

        int n = apples.size();
        map<int, int> info;
        int ans = 0;
        for (int i = 0; i <= 40000; i++) {
            if (i < n && apples[i] > 0) {
                info.insert({days[i] + i, apples[i]});
            }
            for (auto itr = info.begin(); itr != info.end(); ) {
                if (itr->first <= i) {
                    itr = info.erase(itr);
                }
                else {
                    break;
                }

            }
            if (!info.empty()) {
                ++ans;
                auto itr = info.begin();
                if (--itr->second == 0) {
                    info.erase(itr);
                }
            }
        }
        return ans;
    }
};
