class Solution {
public:
    int eatenApples(vector<int>& a, vector<int>& days) {
        map <int, int> s;
        int n = a.size();
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (days[i] != 0 && a[i] != 0) {
                s[i + days[i] - 1] += a[i];
            }
            while (s.size() > 0 && (s.begin()->second == 0 || s.begin()->first < i)) {
                s.erase(s.begin());
            }
            if (s.size() != 0) {
                s.begin()->second--;
                ans++;
            }
        }
        int d = n;
        while (s.size() > 0) {
            while (s.size() > 0 && (s.begin()->second == 0 || s.begin()->first < d)) {
                s.erase(s.begin());
            }
            if (s.size() != 0) {
                s.begin()->second--;
                ans++;
                d++;
            } else {
                break;
            }
        }
        return ans;
    }
};
