class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> curApples; // day, numbers
        
        int today = 0;
        int ans = 0;
        
        int n = static_cast<int>(apples.size());
        for (int i = 0; i < n || !curApples.empty(); ++i) {
            ++today;
            
            if (i < n && apples[i] > 0 && days[i] > 0) {
                curApples[today + days[i]] += apples[i];
            }
            
            if (!curApples.empty() && curApples.begin()->first == today) {
                curApples.erase(today);    
            }
            
            if (!curApples.empty()) {
                auto mostRecentToBad = curApples.begin();
                if (--mostRecentToBad->second == 0) {
                    curApples.erase(mostRecentToBad);
                }
                
                ++ans;
            }
        }
        
        return ans;
    }
};
