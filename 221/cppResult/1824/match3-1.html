class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        map<int, int> a;

        int results = 0;
        for (int i = 0; i < 100000; ++i) {
            if (i < apples.size() && apples[i] > 0) {
                a[i + days[i]] += apples[i];
            }
            while (!a.empty() && (a.begin()->first <= i || a.begin()->second == 0)) {
                a.erase(a.begin());
            }
            if (!a.empty()) {
                ++results;
                --a.begin()->second;
            }
        }
        return results;
    }
};
