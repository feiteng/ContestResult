typedef pair<int, int> P;
 
struct cmp
{
    bool operator()(const P p1, const P p2)
    {
        return p1.second > p2.second; //second的小值优先
    }
};

class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n = apples.size();
        priority_queue<P, vector<P>, cmp> q;
        int ans = 0;
        int i = 0;
        for (; i < n; i++) {
            if (apples[i] != 0) {
                q.push({apples[i], i + days[i]});
            }
            while (!q.empty() && q.top().second <= i) {
                q.pop();
            }
            if (!q.empty()) {
                auto pa = q.top();
                q.pop();
                ans++;
                pa.first--;
                if (pa.first != 0) {
                    q.push(pa);
                }
            }
        }
        for (; !q.empty(); i++) {
            while (!q.empty() && q.top().second <= i) {
                q.pop();
            }
            if (!q.empty()) {
                auto pa = q.top();
                q.pop();
                ans++;
                pa.first--;
                if (pa.first != 0) {
                    q.push(pa);
                }
            }
        }
        return ans;
    }
};
