class Solution {
public:
  int eatenApples(vector<int>& apples, vector<int>& days) {
    
    int n = apples.size();
    map<int, int> storedApples;
    
    int day = 0;

    
    int ans = 0;
    for (;day < n; day++) {
      
      storedApples[day+days[day]] += apples[day];
        
      while ( !storedApples.empty() && (storedApples.begin()->first <= day || storedApples.begin()->second == 0)) {
        storedApples.erase(storedApples.begin());
      }
      
      if (!storedApples.empty()) {
        storedApples.begin()->second--;
        ans++;
      }
    }
    
    while (!storedApples.empty()) {
      while ( storedApples.size() > 0 && (storedApples.begin()->first <= day || storedApples.begin()->second == 0)) {
        storedApples.erase(storedApples.begin());
      }
      
      if (!storedApples.empty()) {
        storedApples.begin()->second--;
        ans++;
      }
      
      day++;
    }
    
    return ans;
  
  
  }
};
