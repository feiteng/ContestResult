class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<pair<int,int>,vector<pair<int,int>>, greater<pair<int,int>>> p;
        int cnt = 0;
        for(int i =0;i<apples.size();i++){
            if(apples[i]!=0){
                p.push(make_pair(days[i]+i,apples[i]));
            }
            while(!p.empty()&&p.top().first<=i){
                p.pop();
            }
            if(!p.empty()){
                cnt++;
                pair<int,int> it = p.top();
                p.pop();
                it.second--;
                if(it.second){
                    p.push(it);
                }
            }
        }
        int i = apples.size();
        while(!p.empty()){
            while(!p.empty()&&p.top().first<=i){
                p.pop();
            }
            if(!p.empty()){
                cnt++;
                pair<int,int> it = p.top();
                p.pop();
                it.second--;
                if(it.second){
                    p.push(it);
                }
            }
            i++;
        }
        return cnt;
    }
};
