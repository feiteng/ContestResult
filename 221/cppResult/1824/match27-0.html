class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<int, vector<int>, greater<int>> q;
        unordered_map<int, int> count;
        int n = apples.size();
        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (apples[i] && days[i]) {
                q.emplace(i + days[i]);
                count[i + days[i]] += apples[i];
            }
            while (!q.empty() && q.top() <= i)
                q.pop();
            if (!q.empty()) {
                ans++;
                if (--count[q.top()] == 0)
                    q.pop();
            }
        }
        int day = n;
        while (!q.empty()) {
            while (!q.empty() && q.top() <= day)
                q.pop();
            if (!q.empty()) {
                int eaten = min(count[q.top()], q.top() - day);
                ans += eaten;
                day += eaten;
                q.pop();
            }
        }
        return ans;
    }
};
