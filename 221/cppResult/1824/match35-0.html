class Solution {
public:
    typedef pair<int,int> pii;
    int eatenApples(vector<int>& apples, vector<int>& days) {
        multiset<pii> s;
        int n = apples.size();
        int ans = 0;
        for (int i=0;i<n+20000;i++) {
            if (i < n && apples[i]) {
                s.insert(pii(i + days[i], apples[i]));
            }
            while(!s.empty()) {
                auto tmp = *s.begin();
                s.erase(s.find(tmp));
                if (tmp.first <= i) continue;
                ans++;
                tmp.second--;
                if (tmp.second > 0) {
                    s.insert(tmp);
                }
                break;
            }
        }
        return ans;
    }
};
