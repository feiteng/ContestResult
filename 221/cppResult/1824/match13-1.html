class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue <pair<int,int>,vector<pair<int,int>>,greater<pair<int,int>>> good;
        int ans =0;
        for(int i = 0; i < apples.size();++i){
            good.push({i+days[i],apples[i]});
            while(!good.empty() && (good.top().first <= i || good.top().second == 0)) good.pop();
            if(!good.empty() && i < good.top().first){
                pair<int,int> temp = good.top();
                good.pop();
                temp.second -=1;
                good.push(temp);

                ans++;
            }
        }
        int i = apples.size();
        while(!good.empty()){
            while(!good.empty() && (good.top().first <= i || good.top().second == 0)) good.pop();
            if(!good.empty() && i < good.top().first) {
                pair<int,int> temp = good.top();
                good.pop();
                temp.second -=1;
                good.push(temp);
                ans++;
            }
            ++i;
        }
        return ans;
    }
};
