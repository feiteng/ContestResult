class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        
        int n=apples.size();
        map<int,int> rots;
        int ans=0;
        for(int i=0;;i++) {
            
            if (i >= n && rots.empty())
                break;
            
            if (i < n && apples[i] > 0)
                rots[i+days[i]]+=apples[i];
            
            while(!rots.empty() && rots.begin()->first <= i) {
                rots.erase(rots.begin());
            }
            
            if (!rots.empty()) {
                ans++;
                auto it = rots.begin();
                it->second--;
                if (it->second == 0) {
                    rots.erase(rots.begin());                    
                }
            }
        }
        return ans;
        
    }
};
