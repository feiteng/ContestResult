struct Item {
    int apple;
    int rotDay;
    bool operator<(const Item& rhs) const {
        return rotDay > rhs.rotDay;
    }
};


class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<Item> PQ;
        int ans = 0;
        int MAX_DAY = 4e4 + 1;
        for (int i = 0; i < MAX_DAY; ++i) {
            if (i < apples.size()) {
                int apple = apples[i];
                int rotDay = i + days[i];
                if (apple > 0) PQ.push({ apple, rotDay });   
            }
            while (!PQ.empty() && PQ.top().rotDay <= i) {
                PQ.pop();
            }
            if (PQ.empty() == false) {
                auto item = PQ.top();
                PQ.pop();
                item.apple -= 1;
                ++ans;
                if (item.apple > 0) PQ.push(item);
            }
        }
        return ans;
    }
};
