class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int ans = 0;
        int n = apples.size();
        map<int, int> cur;
        for (int i = 0; i < 41000; ++i) {
            if (i < n and days[i] != 0) {
                cur[i + days[i]] += apples[i];
            }
            while (!cur.empty()) {
                auto& x = *cur.begin();
                if (x.first <= i or x.second == 0) {
                    cur.erase(cur.begin());
                } else {
                    break;
                } 
            }
            if (cur.empty()) {
                continue;
            }
            auto& x = *cur.begin();
            --x.second;
            ++ans;
        }
        return ans;
    }
};
