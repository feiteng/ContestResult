class Solution {
public:
    int eatenApples(vector<int>& cnt, vector<int>& days) {
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
        int n = days.size();
        //1.结苹果的天数
        int ans = 0;
        for(int i = 0; i < n; ++i){

            pq.push({days[i]+i, cnt[i]}); //最后期限
            while(!pq.empty() && pq.top().first <= i){
                pq.pop();
            }
            if(!pq.empty()){
                auto p = pq.top();
                pq.pop();
                p.second -= 1;
                ++ans;
                if(p.second > 0) pq.push(p);
            }
        }
        //2.剩余的天数
        int d = n;
        while(!pq.empty()){
            auto p = pq.top();
            pq.pop();
            if(p.first > d){//能吃
                p.second -= 1;
                ++ans;
                if(p.second > 0) pq.push(p);
                ++d;
            }
        }
        return ans;
    }
};
