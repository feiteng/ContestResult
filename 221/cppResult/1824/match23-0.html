typedef pair<int,int> pii;
class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        int n = apples.size();
        int ans = 0;
        priority_queue<pii, vector<pii>, greater<pii>> pq;
        for(int i = 0; i < n; ++i){
            // push new apple <rot day, number>
            if(apples[i] > 0)
                pq.emplace(i + days[i], apples[i]);
            while(pq.size() && pq.top().first <= i) pq.pop();
            // eat top
            if(pq.size()) {
                ans ++;
                pii t = pq.top();
                pq.pop();
                if(t.second > 1)
                pq.emplace(t.first, t.second - 1);
            }
        }
        int i = n;
        while(pq.size()) {
            while(pq.size() && pq.top().first <= i) pq.pop();
            if(pq.size()) {
                ans ++;
                pii t = pq.top();
                pq.pop();
                if(t.second > 1)
                pq.emplace(t.first, t.second - 1);
            }
            ++i;
        }
        
        return ans;
    }
};
