class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        vector<int> p(20004);
        int n = apples.size(), ans = 0;
        priority_queue<pair<int,int>> pq;
        for(int i = 0; i < n; i++) {
          if(days[i])
            pq.push({-1*(i+1 + days[i]), apples[i]});
          while(pq.size() && (-1*pq.top().first <= i+1 || pq.top().second < 1) )
            pq.pop();
          if(pq.size()) {
            auto x = pq.top();
            pq.pop();
            x.second -= 1;
            if(x.second > 0)
              pq.push(x);
            ans++;
            // cout << "day=" << i << "\n";
          }
            // p[i] += apples[i];
            // p[i + days[i]] -= apples[i];
        }
      int day = n;
        while(pq.size()) {
            day++;
            while(pq.size() && (-1*pq.top().first <= day || pq.top().second < 1) )
              pq.pop();
            if(pq.size()) {
              auto x = pq.top();
              pq.pop();
              x.second -= 1;
              if(x.second > 0)
                pq.push(x);
              ans++;
            }
        }
        // cout <<"\n";
        return ans;
    }
};
