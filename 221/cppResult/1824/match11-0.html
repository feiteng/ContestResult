class Solution {
public:
    int eatenApples(vector<int>& apples, vector<int>& days) {
        priority_queue<pair<int,int> > pq;
        int n=apples.size();
        int ans=0;
        for(int i=0;i<n;++i)
        {
            pq.push({-(i+days[i]),apples[i]});
            while(pq.size()&&-pq.top().first<=i) pq.pop();
            if(pq.size())
            {
                ans++;
                auto tmp=pq.top();
                pq.pop();
                tmp.second--;
                if(tmp.second!=0) pq.push(tmp);
            }
        }
        int id=n;
        while(pq.size())
        {
            while(pq.size()&&-pq.top().first<=id) pq.pop();
            if(pq.size())
            {
                ans++;
                auto tmp=pq.top();
                pq.pop();
                tmp.second--;
                if(tmp.second!=0) pq.push(tmp);
            }
            id++;
        }
        return ans;
    }
};
