class Solution {
public:
    bool halvesAreAlike(string s) {
        int c1 = 0, c2 = 0;
        
        for (int i = 0; i < s.size() / 2; ++i) {
            char ch = tolower(s[i]);
            if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
                ++c1;
            }
        }
        for (int i = s.size() / 2; i < s.size(); ++i) {
            char ch = tolower(s[i]);
            if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u') {
                ++c2;
            }
        }
        
        return c1 == c2;
    }
};
