class Solution {
public:
    bool halvesAreAlike(string s) {
	int mid = s.size() / 2;
	int nums = 0;
	for (int i = 0; i < mid; i++)
	{
		if (s.at(i)=='a'|| s.at(i) == 'e' || s.at(i) == 'i' || s.at(i) == 'o' || s.at(i) == 'u' || 
			s.at(i) == 'A' || s.at(i) == 'E' || s.at(i) == 'I' || s.at(i) == 'O' || s.at(i) == 'U')
		{
			++nums;
		}
	}
	for (int i = mid; i < s.size(); i++)
	{
		if (s.at(i) == 'a' || s.at(i) == 'e' || s.at(i) == 'i' || s.at(i) == 'o' || s.at(i) == 'u' ||
			s.at(i) == 'A' || s.at(i) == 'E' || s.at(i) == 'I' || s.at(i) == 'O' || s.at(i) == 'U')
		{
			--nums;
		}
	}
	return nums==0;
    }
};
