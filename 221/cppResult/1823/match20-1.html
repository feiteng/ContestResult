class Solution {
public:
    bool halvesAreAlike(string s) {
        int n = s.size();
        int n1=0, n2=0,i;
        set<char> sc({'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'});
        for (i = 0; i < n / 2; ++i)
        {
            if(sc.count(s[i]))
                n1++;
        }
        for (i = n/2; i < n ; ++i)
        {
            if(sc.count(s[i]))
                n2++;
        }
        return n1 == n2;
    }
};
