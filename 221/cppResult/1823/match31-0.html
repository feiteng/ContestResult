class Solution {
public:
    bool halvesAreAlike(string s) {
        int n=s.size()/2;
        int a=0;
        for (int i=0; i<n; i++){
            if (s[i]=='a' or s[i]=='e' or s[i]=='i' or s[i]=='o' or s[i]=='u' or s[i]=='A' or s[i]=='E' or s[i]=='I' or s[i]=='O' or s[i]=='U') {a++;}
        }
        for (int i=n; i<2*n; i++){
            if (s[i]=='a' or s[i]=='e' or s[i]=='i' or s[i]=='o' or s[i]=='u' or s[i]=='A' or s[i]=='E' or s[i]=='I' or s[i]=='O' or s[i]=='U') {a--;}
        }
        return a==0;
    }
};
