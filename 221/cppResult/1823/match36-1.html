vector<char> val = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};

char isVowel(char c)
{
    for(char cc: val)
    {
        if(cc==c)
            return 1;
    }
    return 0;
}


class Solution {
public:
    bool halvesAreAlike(string s) 
    {
        int n = s.size();
        int c1 = 0 , c2 = 0;
        for(int i=0;i<n/2;i++)
        {
            if(isVowel(s[i])==1)
                c1++;
        }
        for(int i=n/2;i<n;i++)
        {
            if(isVowel(s[i])==1)
                c2++;
        }
        return c1==c2;
    }
};
