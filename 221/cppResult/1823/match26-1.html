class Solution {
public:
    bool halvesAreAlike(string s) {
        int t = 0;
        for (int i = 0; i < s.size(); ++i) {
            if (tolower(s[i])=='a' or tolower(s[i])=='e' or tolower(s[i])=='i' or tolower(s[i])=='o' or tolower(s[i])=='u') {
                if (i<s.size()/2) ++t;
                else --t;
            }
        }
        return t==0;
    }
};
