class Solution {
public:
    bool halvesAreAlike(string s) {
        int mid=s.size()/2;
        int pre=0;
        for(int i=0;i<mid;i++) {
            if(s[i]=='A'||s[i]=='a'||s[i]=='e'||s[i]=='E'||s[i]=='i'||s[i]=='I'||s[i]=='o'||s[i]=='O'||s[i]=='u'||s[i]=='U') {
                pre++;
            }
        }
        int hou=0;

        for(int i=mid;i<s.size();i++) {
            if(s[i]=='A'||s[i]=='a'||s[i]=='e'||s[i]=='E'||s[i]=='i'||s[i]=='I'||s[i]=='o'||s[i]=='O'||s[i]=='u'||s[i]=='U') {
                hou++;
            }
        }
        return pre==hou;
    }
};
