public class Solution {
    public int[] findBall(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int[][] ans = new int[n + 1][m];
        for (int i = 0; i < m; i++) {
            ans[n][i] = i;
        }
        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    if (j == m - 1 || grid[i][j + 1] == -1) {
                        ans[i][j] = -1;
                    } else {
                        ans[i][j] = ans[i + 1][j + 1];
                    }
                } else {
                    if (j == 0 || grid[i][j - 1] == 1) {
                        ans[i][j] = -1;
                    } else {
                        ans[i][j] = ans[i + 1][j - 1];
                    }
                }
            }
        }
        for(int i = 0; i <= n; i++){
        //    System.out.println(Arrays.toString(ans[i]));
        }
        return ans[0];
    }
}
