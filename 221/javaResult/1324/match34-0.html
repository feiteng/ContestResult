class Solution {
public int[] findBall(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        if (n == 1) return new int[]{-1};
        int[] ans = new int[n];
        for (int col = 0; col < n; col++) {
            int y = col;
            for (int row = 0; row < m; row++) {
                if (grid[row][y] == 1) {
                    if (y == n - 1 || grid[row][y + 1] == -1) {
                        ans[col] = -1;
                        break;
                    } else {
                        y += 1;
                    }
                } else if (grid[row][y] == -1) {
                    if (y == 0 || grid[row][y - 1] == 1) {
                        ans[col] = -1;
                        break;
                    } else {
                        y -= 1;
                    }
                }
            }
            if (ans[col] != -1)
                ans[col] = y;
        }
        return ans;
    }
}
