class Solution {
    int m, n;
    public int[] findBall(int[][] grid) {
        this.m = grid.length;
        this.n = grid[0].length;
        int[] res = new int[n];
        for (int j = 0; j < n; j++) {
            res[j] = dfs(grid, 0, j);
        }
        return res;
    }
    
    private int dfs(int[][] grid, int row, int col) {
        if (row == m) {
            return col;
        }
        
        if (grid[row][col] == 1) {
            if (col + 1 >= n || grid[row][col + 1] == -1) {
                return -1;
            } else {
                return dfs(grid, row + 1, col + 1);
            }
        } else if (grid[row][col] == -1) {
            if (col - 1 < 0 || grid[row][col - 1] == 1) {
                return -1;
            } else {
                return dfs(grid, row + 1, col - 1);
            }
        }
        return -1;
    }
}
