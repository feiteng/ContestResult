class Solution {
    public int[] findBall(int[][] g) {
        int m = g.length, n = g[0].length;
        int[] ans = new int[n];
        for(int i = 0; i < n; i++){
            int x = i, y= 0;
            while(y<m){
                //System.out.println(x+" "+y);
                if(g[y][x]==1 && x+1<n && g[y][x+1]==1){
                    x++;
                    y++;
                }else if(g[y][x]==-1 && x-1>=0 && g[y][x-1]==-1){
                    x--;
                    y++;
                }else {
                    break;
                }
            }
            if(y==m){
                ans[i] = x;
            }else {
                ans[i] = -1;
            }
        }
        return ans;
    }
}
