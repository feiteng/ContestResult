class Solution {
    public int[] findBall(int[][] grid) {
        
        int[] res = new int[grid[0].length];
        for(int i=0; i<grid[0].length; i++)
        {
            res[i]=getFallPos(grid, i);
        }
        return res;
    }
    
    private int getFallPos(int[][] grid, int stCol)
    {   
        for(int i=0; i<grid.length; i++)
        {
            if(grid[i][stCol]==1)
            {
                if(stCol==grid[0].length-1 || grid[i][++stCol]==-1)
                    return -1;
            }
            else
            {
                if(stCol==0 || grid[i][--stCol]==1)
                    return -1;
            }
        }
        return stCol;
    }
}
