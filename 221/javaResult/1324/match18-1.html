class Solution {
    int m;
    int n;
    public int[] findBall(int[][] grid) {
        m = grid.length;
        n = grid[0].length;
        
int[] ans = new int[n];
        for(int j = 0; j < n; j++) {
            ans[j] = dfs(grid, 0, j);
        }
        return ans;
    }
    
    int dfs(int[][] grid, int i, int j) {
        if(i == m) return j
;
        if(grid[i][j] == 1) {
            // right
            if(j == n - 1 || grid[i][j+1] == -1) {
                return -1;
            }
            return dfs(grid, i+1, j+1);
        } else {
            // left
            if(j == 0 || grid[i][j-1] == 1) {
                return -1;
            }
            return dfs(grid, i+1, j-1);
        }
    }
}
