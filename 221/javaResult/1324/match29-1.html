class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
int[] res = new int[n];
        
        for(int i = 0; i < n; i++) {
            res[i] = helper(grid, 0, i);
        }
        return res;
    }
    
    private int helper(int[][] grid, int row, int col) {
        if(row == grid.length) return col;
        int cur = grid[row][col];
        
if(cur == 1) {
            if(col == grid[0].length - 1 || grid[row][col + 1] == -1) {
                return -1;
            }else {
                return helper(grid, row + 1, col + 1);
            }
        }else if(cur == -1){
            if(col == 0 || grid[row][col - 1] == 1) {
                return -1;
            }else {
                return helper(grid, row + 1, col - 1);
            }
        }
        return -1;
    }
}
