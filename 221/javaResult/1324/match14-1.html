class Solution {
    public int[] findBall(int[][] grid) {
        
int n = grid[0].length;
        int[] result = new int[n];
        for(int j=0; j<n; j++) {
            int col = j;
            for(int i=0; i<grid.length; i++) {
                if(grid[i][col]==1) {
                    if(col == n-1 ||grid[i][col+1] == -1) {
                        //stuck
                        result[j] = -1;
                        break;
                    } else {
                        col++;
                    }
                } else {
                    if(col == 0 || grid[i][col-1] == 1) {
                        result[j] = -1;
                        break;
                    } else {
                        col--;
                    }
                }
            }
            if(result[j]!=-1) result[j] = col;
        }
        return result;
    }
}
