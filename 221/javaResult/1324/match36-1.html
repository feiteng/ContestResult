class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[] pos = new int[n];
        for(int i = 0; i < n; i++) pos[i] = i
;
        for(int[] r : grid) {
            for(int i = 0; i < n; i++) {
                if(pos[i] == -1) continue;
                int p = pos[i];
                if(r[p] == 1 && p + 1 < n && r[p+1] == 1) {
                    pos[i] = p + 1;
                } else 
if(r[p] == -1 && p - 1 >= 0 && r[p-1] == -1) {
                    pos[i] = p - 1;
                } else {
                    pos[i] = -1;
                
}
            }
        }
        return pos;
    }
}
