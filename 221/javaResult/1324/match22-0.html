class Solution {
   public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] ans = new int[n];

        for (int x = 0; x < n; x++) {
            int curCol = x;
            for (int i = 0; i < m; i++) {
                if (curCol == 0 && grid[i][curCol] == -1) {
                    ans[x] = -1;
                    break;
                }
                if (curCol == n - 1 && grid[i][curCol] == 1) {
                    ans[x] = -1;
                    break;
                }
                if (grid[i][curCol] == 1 && grid[i][curCol + 1] == -1) {
                    ans[x] = -1;
                    break;
                }
                if (grid[i][curCol] == -1 && grid[i][curCol - 1] == 1) {
                    ans[x] = -1;
                    break;
                }
                curCol += grid[i][curCol];
            }
            if (ans[x] != -1) {
                ans[x] = curCol;
            }
        }
        return ans;
    }
}
