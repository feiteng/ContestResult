/**
how to identify stuck cases?
- stuck with the wall
    - [-1] on the left wall, [1] on the right wall
- stuck in a V shape
    - [1][-1] is a V shape

simulate ball movement through the matrix


**/
class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        
int[] result = new int[n];
        
        for (int i = 0; i < result.length; i++) {
            result[i] = findBall(grid, m, n, i);    
        }        
        
        return result;
    }
    
    // find the output position of ball i
    private int findBall(int[][] grid, int m, 
int n, int c) {
        for (int r = 0; r < m; r++) {
            if (grid[r][c] == 1) {
                // is a '\'
                if (c == n - 1 || grid[r][c+1] == -1)
                    return -1;
                c++;
            } else {
                // is a '/'
                if (c == 0 || grid[r][c-1] == 1)
                    return -1;
                c--;
            }
        }
        return c;
    }
}
