class Solution {
  public int[] findBall(int[][] grid) {
    int m = grid.length;
    int n = grid[0].length;
    int[] ans = new int[n];
    for (int c = 0; c < n; c++) {
      int pos = c;
      int r = 0;
      for (; r < m; r++) {
        if (grid[r][pos] == 1) {
          if (pos == n - 1 || grid[r][pos + 1] == -1) break;
          pos++;
        } else {
          if (pos == 0 || grid[r][pos - 1] == 1) break;
          pos--;
        }
      }
      if (r < m) ans[c] = -1;
      else ans[c] = pos;
    }
    return ans;
  }
}
