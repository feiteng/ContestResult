class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] ret = new int[n];
        for (int i = 0; i < n; i++) {
            ret[i] = helper(grid, 0, i);
        }
        return ret;
    }
    private int helper(int[][] grid, int row, int col) {
        int m = grid.length;
        int n = grid[0].length;
        if (row == m) {
            return col;
        }
        
if (grid[row][col] == 1) {
            if (col == n - 1 || grid[row][col + 1] == -1) {
                return -1;
            } else {
                return helper(grid, row + 1, col + 1);
            }
        } else 
{
            if (col == 0 || grid[row][col - 1] == 1) {
                return -1;
            } else {
                return helper(grid, row + 1, col - 1);
            }
        }
    }
}
