class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] res = new int[grid[0].length];
        for (int k = 0; k < grid[0].length; k++) {
            int c = k;
            for (int i = 0; i < grid.length; i++) {
                if (grid[i][c] == 1) {
                    if (c + 1 == n || grid[i][c + 1] == -1) {
                        res[k] = -1;
                        break;
                    } else {
                        c++;
                    }
                } else {
                    if (c == 0 || grid[i][c - 1] == 1) {
                        res[k] = -1;
                        break;
                    } else {
                        c--;
                    }
                }
            }
            if (res[k] != -1) {
                res[k] = c;
            }
        }
        return res;
    }
}
