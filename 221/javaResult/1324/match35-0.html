class Solution {
    // int[] arr;
    
    public int[] findBall(int[][] g) {
        int n=g.length;
        int m=g[0].length;
        // arr=new int[n][m];
        int[] res=new int[m];
        for(int i=0;i<m;i++){
            res[i]=dfs(g,0,i);
        }
        return res;
        
    }
    int dfs(int[][] g,int x,int y){
        // if(arr[x][y]!=0) return arr[x][y]
        int a=g[x][y];
        if(y==0&&a==-1) return -1;
        if(y==g[0].length-1&&a==1) return -1;
        if(g[0].length==1) return -1;
        if(a==1){
            if(g[x][y+1]==-1){
                return -1;
            }else{
                if(x+1==g.length){
                    return y+1;
                }
                return dfs(g,x+1,y+1);
            }
        }else{
            if(g[x][y-1]==1){
                return -1;
            }else{
                if(x+1==g.length){
                    return y-1;
                }
                return dfs(g,x+1,y-1);
            }
        }
    }
}
