class Solution {
    public int[] findBall(int[][] grid) {
        int n = grid[0].length;
        
int[] res = new int[n];
        for(int i = 0; i < n; i++) {
            res[i] = dfs(grid, 0, i, 1);
        }
        return res;
    }
    private int dfs(int[][] grid, 
int i, int j, int d) {
        if(j < 0 || j >= grid[0].length) {
            return -1;
        }
        if(i >= grid.length) {
            return j;
        }
        if(grid[i][j] == 1) {
            if(d == 0) {
                return dfs(grid, i + 1, j, 1);
            } else if(d == 1) {
                return dfs(grid, i, j + 1, 0);
            } else {
                return -1;
            }
        } else {
            if(d == 0) {
                return -1;
            } else if(d == 1) {
                return dfs(grid, i, j - 1, 2);
            } else {
                return dfs(grid, i + 1, j, 1);
            }
        }
    }
}
