class Solution {
    int rows, cols;
    public int[] findBall(int[][] grid) {
        /*
        
        if \, check right one. if / or wall, return -1
        else check right and low
        
        if /, check left one. if \ or wall, return -1
        else, check left and low
        */
        rows = grid.length;
        cols = grid[0].length;
        int[] answ = new int[cols];
        for (int i = 0; i < cols; i++) {
            answ[i] = check(grid, 0, i);
        
}
        return answ;
    }
    
    int check(int[][] grid, int row, int col) {
        if (row == rows) return col;
        if (grid[row][col] == 1) {
            // \
            if (col == cols-1 || grid[row][col+1] == -1) return -1;
            return check(grid, row+1, col+1);
        } else {
            if (col == 0 || grid[row][col-1] == 1) return -1;
            return check(grid, row+1, col-1);
        }
    }
}
