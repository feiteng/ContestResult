class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            int c = i, r = 0;
            while (r < m) 
{
                if (grid[r][c] == 1) {
                    if (c == n - 1 || grid[r][c + 1] == -1) break;
                    r++;
                    c++;
                }
                else {
                    if (c == 0 || grid[r][c - 1] == 1) break;
                    r++;
                    c--;
                }
            }
            
if (r == m) ans[i] = c;
            else ans[i] = -1;
        }
        return ans;
    }
}
