class Solution {
    public int[] findBall(int[][] grid) {
        int m=grid.length,n=grid[0].length;
        int[] ans=new int[n];
        for(int i=0;i<n;i++){
            int loc=i;
            for(int j=0;j<m;j++){
                if(grid[j][loc]==1){
                    if(loc==n-1||grid[j][loc+1]==-1){
                        ans[i]=-1;
                        break;
                    }else{
                        loc+=1;
                    }
                }else{
                    if(loc==0||grid[j][loc-1]==1){
                        ans[i]=-1;
                        break;
                    }else{
                        loc-=1;
                    }
                }
            }
            if(ans[i]!=-1)
                ans[i]=loc;
        }
        return ans;
    }
}
