class Solution {
    
    int m;
    int n;
    
    public int[] findBall(int[][] grid) {
        m = grid.length;
        n = grid[0].length;
        int[] res = new int[n];
        
        for (int j = 0; j < n; j++)
            res[j] = helper(0, j, grid);
        
        return res;
    }
    
    public int helper(int i, int j, int[][] grid) {
        if (i == m) return j;
        
        if (grid[i][j] == 1) {
            if (j == n - 1 || grid[i][j + 1] != 1) return -1;
            return helper(i + 1, j + 1, grid);
        } else {
            if (j == 0 || grid[i][j - 1] != -1) return -1;
            return helper(i + 1, j - 1, grid);
        }
    }
}
