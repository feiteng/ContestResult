class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] o = new int[n];
        for (int i = 0; i < n; i++) {
            boolean valid = true;
            int c = i;
            for (int r = 0; r < m; r++) {
                if (grid[r][c] == 1) {
                    if (c >= n - 1 || grid[r][c + 1] == -1) {
                        valid = false;
                        break;
                    } else {
                        c += 1;
                    }
                } else {
                    if (c <= 0 || grid[r][c - 1] == 1) {
                        valid = false;
                        break;
                    } else {
                        c -= 1;
                    }
                }
            }
            if (valid) {
                o[i] = c;
            } else {
                o[i] = -1;
            }
        }
        return o;
    }
}
