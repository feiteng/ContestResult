class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        int[] result = new int[n];
        for (int i = 0; i < n; i++) {
            int pos = i;
            for (int j = 0; j < m; j++) {
                if (grid[j][pos] == 1) {
                    if (pos == n - 1 || grid[j][pos + 1] == -1) {
                        result[i] = -1;
                        break;
                    } else {
                        pos++;
                    }
                } else {
                    if (pos == 0 || grid[j][pos - 1] == 1) {
                        result[i] = -1;
                        break;
                    } else {
                        pos--;
                    }
                }
            }
            if (result[i] != -1) {
                result[i] = pos;
            }
        }
        return result;
    }
}
