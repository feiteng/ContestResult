class Solution {
    public int[] findBall(int[][] grid) {
        int[] res = new int[grid[0].length];
        
        for(int i=0;i<res.length;i++){
            res[i]= helper(grid, i);
        }
        return res;
    }
    
    
    private int helper(int[][] grid, int start){
        
int res = start;
       // System.out.println("start: "+res);
        for(int i=0;i<grid.length;i++){
          //  System.out.println("row: "+i+"  res: " +res);
            if(grid[i][res]>0){
                if(res+1>=grid[i].length || grid[i][res+1]<0){
                    return -1;
                }
                res++;
            }else {
                if(res-1<0 || grid[i][res-1]>0){
                    return -1;
                }
                res--;   
            }
        }
        return res;
    }
}
