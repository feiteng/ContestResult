class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid[0].length;
        int[] rst = new int[m];
        for (int j = 0; j < m; j++) {
            rst[j] = fall(grid, 0, j);
        }
        return rst;
    }

    private int fall(int[][] grid, int i, int j) {
        int n = grid.length;
        int m = grid[0].length;
        if (i == n) {
            return j;
        }
        if (grid[i][j] == 1) {
            if (j + 1 == m || grid[i][j + 1] == -1) {
                return -1;
            }
            return fall(grid, i + 1, j + 1);
        }
        else {
            if (j == 0 || grid[i][j - 1] == 1) {
                return -1;
            }
            return fall(grid, i + 1, j - 1);
        }
    }
}
