class Solution {
    public int[] findBall(int[][] grid) {
        int n = grid[0].length;
        int m = grid.length;
        int[] ans = new int[n];
        
        for(int i=0;i<n;i++){
            int cur = i;
            boolean f = true;
            for(int j=0;j<m;j++){
                if(grid[j][cur]==1){
                    if(cur+1>=n||grid[j][cur+1]==-1){
                        f = false;
                        break;
                    }else{
                        cur+=1;
                    }
                }else{
                    if(cur-1<0||grid[j][cur-1]==1){
                        f = false;
                        break;
                    }else{
                        cur-=1;
                    }
                }
                // System.out.println(cur);
            }
            if(!f){
                ans[i]=-1;
            }else{
                ans[i]=cur;
            }
        }
        
        return ans;
    }
}
