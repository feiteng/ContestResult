class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        int[] res = new int[n];
        for (int i = 0; i < n; i ++) {
            res[i] = helper(grid, 0, i, m, n);
        }
        
        return res;
    }
    
    private int helper(int[][] grid, int r, int c, int m, int n) {
        if (r == m) {
            return c;
        }
        
        int cur = grid[r][c];
        
        
if (cur == 1) {
            if (c + 1 >= n || grid[r][c + 1] == -1) {
                return -1;
            } else {
                return helper(grid, r + 1, c + 1, m, n);
            }
        } else {
            if (c - 1 < 0 || grid[r][c - 1] == 1) {
                return -1;
            } else {
                return helper(grid, r + 1, c - 1, m, n);
            }
        }
    }
}
