class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        
        int[] ans = new int[n];
        for (int i = 0; i < n; i++) {
            int row = 0;
            int col = i;
            while (row < m) {
                if (grid[row][col] == 1) {
                    if (col >= n - 1) {
                        ans[i] = -1;
                        break;
                    } else {
                        if (grid[row][col + 1] == -1) {
                            ans[i] = -1;
                            break;
                        } else {
                            row++;
                            col++;
                        }
                    }
                } else {
                    if (col <= 0) {
                        ans[i] = -1;
                        break;
                    } else {
                        if (grid[row][col - 1] == 1) {
                            ans[i] = -1;
                            break;
                        } else {
                            row++;
                            col--;
                        }
                    }
                }
            }
            if (row == m) {
                ans[i] = col;
            }
        }
        
        return ans;
    }
}
