class Solution {
    public int[] findBall(int[][] grid) {
        int m = grid[0].length;
        int[] res = new int[m];
        for (int i = 0; i < m; i++) {
            res[i] = dfs(grid, 0, i);
        }
        return res;
    }
    
    private int dfs(int[][] grid, int i, int j) {
        if (j < 0 || j == grid[0].length) {
            return -1;
        }
        if (i == grid.length) {
            return j;
        }
        if (grid[i][j] == 1) {
            int next = j + 1;
            if (next == grid[0].length) {
                return -1;
            }
            if (grid[i][next] == -1) {
                return -1;
            }
            return dfs(grid, i + 1, next);
        }
        else {
            int next = j - 1;
            if (next < 0) {
                return -1;
            }
            if (grid[i][next] == 1) {
                return -1;
            }
            return dfs(grid, i + 1, next);
        }
    }
}
