class Solution {
    public int[] findBall(int[][] grid) {
        int n = grid.length;
        
int m = grid[0].length;
        
        int[] res = new int[m];
        
        for(int i = 0; i < m; i++){
            res[i] = dfs(grid, 0, i);
        }
        
        return res;
    }
    
    private int dfs(int[][] grid, int x, int y){
        
if(x >= grid.length){
            return y;
        }
        
        if(grid[x][y] == 1){
            if(y == grid[0].length - 1 || grid[x][y + 1] == -1){
                return -1;
            }
            
            return dfs(grid, x + 1, y + 1);
        }
        else{
            if(y == 0 || grid[x][y - 1] == 1){
                return -1;
            }
            
            return dfs(grid, x + 1, y - 1);
        }
    }
}
