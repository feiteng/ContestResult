class Solution {
    public int[] findBall(int[][] grid) {
        int[] res = new int[grid[0].length];
        for (int i = 0; i<grid[0].length; i++) {
            res[i] = hit(grid, 0, i);
        }
        return res;
    }
    int hit(int[][] grid, int row, int col) {

        int rl = grid.length;
        int cl = grid[0].length;
        int j = col;
        for (int i = row; i<rl; i++) {
            if (grid[i][j] == 1) {
                if (j+1 == cl || grid[i][j+1] == -1) {
                    return -1;
                }
                j++;
            } else {
                if (j-1 == -1 || grid[i][j-1] == 1) {
                    return -1;
                }
                j--;
            }
        }
        return j;
    }
}
