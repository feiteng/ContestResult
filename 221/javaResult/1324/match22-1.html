class Solution {
    public int[] findBall(int[][] grid) {
    int m = grid.length;
    int n = grid[0].length;
    int[] res = new int[n];

    for (int j = 0; j < n; ++j) {
      int pos = j;
      int row = 0;

      while (row < m) 
{
        if (pos > 0 && grid[row][pos] == -1 && grid[row][pos - 1] == 1) {
          res[j] = -1;
          break;
        }
        if (pos < n - 1 && grid[row][pos] == 1 && grid[row][pos + 1] == -1) {
          res[j] = -1;
          break;
        }
        if (pos == n - 1 && grid[row][pos] == 1) {
          res[j] = -1;
          break;
        }
        if (pos == 0 && grid[row][pos] == -1) {
          res[j] = -1;
          break;
        }

        if (grid[row][pos] == 1) {
          ++pos;
        } else {
          --pos;
        }

        ++row;
      }

      
if (res[j] != -1) {
        res[j] = pos;
      }
    }

    return res;
  }
}
