class Solution {
    public int[] findBall(int[][] grid) {
        int[] ans = new int[grid[0].length];
        for(int j = 0; j < grid[0].length; j++){
            ans[j] = dfs(0, j, grid);
        }
        return ans;
    }
    public int dfs(int i, int j, int[][] grid){
        if (i == grid.length){
            return j;
        }
        if (grid[i][j] == 1){
            if (j + 1 == grid[0].length || grid[i][j + 1] == -1){
                return -1;
            }
            return dfs(i + 1, j + 1, grid);
        } else{
            if (j == 0 || grid[i][j - 1] == 1){
                return -1;
            }
            return dfs(i + 1, j - 1, grid);
        }
    }
}
