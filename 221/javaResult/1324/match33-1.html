class Solution {
  public int[] findBall(int[][] grid) {
    
final int R = grid.length, C = grid[0].length;
    int[] res = new int[C];
    for (int cell = 0; cell != C; cell++) {
      int c = cell;
      for (int r = 0; r != R; r++) {
        if (grid[r][c] == 1) { // goto right
          c++;
          if (c == C) { // stuck
            res[cell] = -1;
            break;
          } else {
            if (grid[r][c] == -1) {
              res[cell] = -1;
              break;
            } // otherwise, continue with the next r
          
}
        } else { // goto left
          c--;
          if (c == -1) { // stuck
            res[cell] = -1;
            break;
          } else {
            if (grid[r][c] == 1) {
              res[cell] = -1;
              break;
            } // otherwise, continue with the next r
          
}
        }
      }
      if (res[cell] != -1)
        res[cell] = c;
    }
    return res;
  }
}
