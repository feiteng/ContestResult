class Solution {
    public int[] findBall(int[][] grid) {
        int m=grid.length, n=grid[0].length;
        int[] res = new int[grid[0].length];
        for(int i=0;i<n;++i){
            int j=0, k=i;
            for(;j<m;++j)
{
                if(grid[j][k]==1){
                    if(k+1==n || grid[j][k+1]==-1)break;
                    ++k;
                }else{
                    if(k==0 || grid[j][k-1]==1)break;
                    --k;
                }
            }
            
if(j==m)res[i]=k;
            else res[i]=-1;
        }
        return res;
    }
}
