class Solution {
    public boolean halvesAreAlike(String s) {
    Set<Character> valid = new HashSet<>();
    valid.add('A');
    valid.add('E');
    valid.add('I');
    valid.add('O');
    valid.add('U');
      s = 
s.toUpperCase();
    return find(valid, s.substring(0, s.length() / 2)) == find(valid, s.substring(s.length() / 2));
  }

  int find(Set<Character> valid, String str) {
    int res = 0;
    for (int i = 0; i < str.length(); i++) {
      if (valid.contains(str.charAt(i))) {
        res++;
      }
    }
    return res;
  }
}
