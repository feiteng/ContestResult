class Solution {
    public boolean halvesAreAlike(String s) {
    Set<Character> set = new HashSet<>(Arrays.asList('a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'));
    
    int len = s.length() / 2;
    int first = 0;
    int second = 0;
    
    for (int i = 0; i < len; ++i) {
      if (set.contains(s.charAt(i))) {
        ++first;
      }
    }
    for (int i = len; i < s.length(); ++i) {
      if (set.contains(s.charAt(i))) {
        ++second;
      }
    }
    
    return first == second;
  }
}
