class Solution {
 
    public boolean halvesAreAlike(String s) {
        int n = s.length();
        int l = n / 2;
        int lc = 0, rc = 0;
        for (int i = 0; i < l; i++) {
            char c = s.charAt(i);
            if (c == 'a' || c == 'e' || c == 'i'
                    || c == 'o'
                    || c == 'u'
                    || c == 'A'
                    || c == 'E'
                    || c == 'I'
                    || c == 'O'
                    || c == 'U'
                    ) { lc++; }
        }
        for (int i = l; i < n; i++) {
            char c = s.charAt(i);
            if (c == 'a' || c == 'e' || c == 'i'
                    || c == 'o'
                    || c == 'u'
                    || c == 'A'
                    || c == 'E'
                    || c == 'I'
                    || c == 'O'
                    || c == 'U'
                    ) { rc++; }
        }
        return lc==rc;
    }
}
