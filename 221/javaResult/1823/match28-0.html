class Solution {
    public boolean halvesAreAlike(String s) {
        char[] chars = new char[]{'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        int len = s.length();
        int c = 0;
        for (int i = 0; i < len / 2; i++) {
            for (char aChar : chars) {
                if (s.charAt(i) == aChar) {
                    c++;
                    break;
                }
            }
        }
        for (int i = len / 2; i < len; i++) {
            for (char aChar : chars) {
                if (s.charAt(i) == aChar) {
                    c--;
                    break;
                }
            }
        }
        return c == 0;
    }
}
