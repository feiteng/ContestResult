class Solution {
    public boolean halvesAreAlike(String s) {
        int n = s.length();
        return helper(s.substring(0, n/2)) == helper(s.substring(n/2));
    }
    
    static int helper(String s) {
        int result = 0;
        for(char ch : s.toCharArray()) {
            if ("aeiouAEIOU".indexOf(ch) >= 0) 
{
                ++result;
            }
        }
        return result;
    }
}
