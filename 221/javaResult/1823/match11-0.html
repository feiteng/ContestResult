class Solution {
    public boolean halvesAreAlike(String s) {
        int n = s.length();
        char[] A = new char[]{'a','e','i','o','u','A','E','I','O','U'};
        int a =0,b=0;
        for(int i=0;i*2<n;i++){
            for(char c: A){
                if(s.charAt(i)==c){
                    a++;
                    break;
                }   
            }
        }
        for(int i=n/2;i<n;i++){
            for(char c: A){
                if(s.charAt(i)==c){
                    b++;
                    break;
                }   
            }
        }
        return a==b;
    }
}
