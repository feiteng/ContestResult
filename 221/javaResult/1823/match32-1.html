class Solution {
  public boolean halvesAreAlike(String s) {
    final int L = s.length();
    int count = 0;
    String setStr = "aeiouAEIOU";
    Set<Character> set = new HashSet<>();
    for (int i = 0; i < setStr.length(); i++) {
      set.add(setStr.charAt(i));
    }
    
for (int i = 0; i < L; i++) {
      if (set.contains(s.charAt(i))) {
        if (i < L / 2) {
          count++;
        } else {
          count--;
        }
      }
    }
    return count == 0;
  }
}
