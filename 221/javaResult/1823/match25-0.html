class Solution {
    public boolean halvesAreAlike(String s) {
        Set<Character> v = new HashSet<>();
        //'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'
        v.add('a');
        v.add('e');
        v.add('i');
        v.add('o');
        v.add('u');
        v.add('A');
        v.add('E');
        v.add('I');
        v.add('O');
        v.add('U');
        int cnt1 = 0;
        int cnt2 = 0;
        int i = 0;
        for (; i < s.length() / 2; i++) {
            if (v.contains(s.charAt(i))) {
                cnt1++;
            }
        }
        for (; i < s.length(); i++) {
            if (v.contains(s.charAt(i))) {
                cnt2++;
            }
        }
        return cnt1 == cnt2;
    }
}
