class Solution {
    public boolean halvesAreAlike(String s) {
        int mid = s.length() / 2;
        int vowels = 0;
        for (int i=0; i<mid; i++) {
            if (isVowel(s.charAt(i)))
                vowels++;
        }
        for (int i=mid; i<s.length(); i++) 
{
            if (isVowel(s.charAt(i)))
                vowels--;
        }
        return vowels == 0;
    }
    
    private static boolean isVowel(char ch) {
        return ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
            ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U';
    }
}
