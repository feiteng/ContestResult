class Solution {
    public boolean halvesAreAlike(String s) {
        Set<Character> v = new HashSet<Character>();
        v.add('a');
        v.add('e');
        v.add('i');
        v.add('o');
        v.add('u');
        v.add('A');
        v.add('E');
        v.add('I');
        v.add('O');
        v.add('U');
        int cnt = 0;
        for(int i = 0; i < s.length(); ++i)
        {
            char c = s.charAt(i);
            if(!v.contains(c)) continue;
            if(i >= s.length() / 2)
            {
                --cnt;
            }
            else
            {
                ++cnt;
            }
        }
        
        return cnt == 0;
    }
}
