class Solution {
    public boolean halvesAreAlike(String s) {
        int len = s.length()/2;
        int c1 = 0;
        int c2 = 0;
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        set.add('A');
        set.add('E');
        set.add('I');
        set.add('O');
        set.add('U');
        for(int i = 0; i < len; i++){
            if(set.contains(s.charAt(i))){
                c1++;
            }
            
if(set.contains(s.charAt(i+len))){
                c2++;
            }
        }
        return c1 == c2;
    }
}
