class Solution {
    public boolean halvesAreAlike(String s) {
        char[] vs = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        Set<Character> set = new HashSet<>();
        
        for (char v: vs) {
            set.add(v);
        }
        int c = 0;
        int c1 = 0;
        for (int i = 0;i<s.length()/2; i++) {
            if (set.contains(s.charAt(i))) {
                c++;
            }
            if (set.contains(s.charAt(s.length()/2+i))) {
                c1++;
            }
        }
        return c == c1;
    }
}
