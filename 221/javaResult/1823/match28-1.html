class Solution {
     public boolean halvesAreAlike(String s) {
        char[] chs = {'a','e','i','o','u','A','E','I','O','U'};
        int cnt = 0;
        for (int i = 0; i < s.length()/2; i++) 
{
            for (int j = 0; j < chs.length; j++) {
                if (s.charAt(i) == chs[j]) {
                    cnt++;
                    break;
                }
            }
        }
        for (int i = s.length()/2; i < s.length(); i++) 
{
            for (int j = 0; j < chs.length; j++) {
                if (s.charAt(i) == chs[j]) {
                    cnt--;
                    break;
                }
            }
        }
        return cnt == 0;
    }
}
