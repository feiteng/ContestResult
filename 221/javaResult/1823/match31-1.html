class Solution {
    public boolean halvesAreAlike(String s) {
        int n = s.length();
        char[] chars = s.toCharArray();
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        set.add('A');
        set.add('E');
        set.add('I');
        set.add('O');
        set.add('U');
        
int l = 0, r = 0;
        for(int i = 0; i < n / 2; i++) {
            if(set.contains(chars[i])) {
                l++;
            }
        }
        for(int i = n / 2; i < n; i++) {
            if(set.contains(chars[i])) {
                r++;
            }
        }
        return l == r;
    }
}
