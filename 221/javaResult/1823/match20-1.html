class Solution {
    public boolean halvesAreAlike(String s) {
        int len = s.length();
        return cnt(s.substring(0,len/2)) == cnt(s.substring(len/2));
    }
    
    
int cnt(String s) {
        int res = 0;
        for(char c: s.toCharArray()) {
            if(c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u'|| c == 'A'|| c == 'E'|| c == 'I'|| c == 'O'|| c == 'U') 
{
                res++;
            }
        }
        return res;
    }
}
