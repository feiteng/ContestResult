class Solution {
    public boolean halvesAreAlike(String s) {
        HashSet<Character> chars = new HashSet<>();
        chars.add('a');
        chars.add('e');
        chars.add('i');
        chars.add('o');
        chars.add('u');
        chars.add('A');
        chars.add('E');
        chars.add('I');
        chars.add('O');
        chars.add('U');
        int count = 0;
        for (int i = 0; i < s.length(); i++) {
            if (chars.contains(s.charAt(i))) {
                if (i < s.length() / 2) {
                    count++;
                }
                else {
                    count--;
                }
            }
        }
        return count == 0;
    }
}
