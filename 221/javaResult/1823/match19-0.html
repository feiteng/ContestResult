class Solution {
    public boolean halvesAreAlike(String s) {
        HashSet<Character> vowels = new HashSet<>();
        vowels.add('a');
        vowels.add('e');
        vowels.add('i');
        vowels.add('o');
        vowels.add('u');
        vowels.add('A');
        vowels.add('E');
        vowels.add('I');
        vowels.add('O');
        vowels.add('U');
        
        int length = s.length() / 2;
        String left = s.substring(0, length);
        String right = s.substring(length);
        int count = 0;
        for (int i = 0; i < length; i++) {
            if (vowels.contains(left.charAt(i))) {
                count++;
            }
            if (vowels.contains(right.charAt(i))) {
                count--;
            }
        }
        
        return count == 0;
    }
}
