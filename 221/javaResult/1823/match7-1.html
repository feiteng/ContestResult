class Solution {
    public boolean halvesAreAlike(String s) {
        char[] y={'a','e','i','o','u','A','E','I','O','U'};
        HashSet<Character> set=new HashSet<>();
        for(char c:y){
            set.add(c);
        }
        int leftNum=0;
        int i;
        for(i=0;i<
s.length()/2;i++){
            if(set.contains(s.charAt(i))){
                leftNum++;
            }
        }
        for (;i<
s.length();i++){
            if(set.contains(s.charAt(i))){
                leftNum--;
            }
        }
        return leftNum==0;
    }
}
