class Solution {
    public boolean halvesAreAlike(String s) {
        HashSet<Character> hs = new HashSet<>();
        hs.add('a');
        hs.add('e');
        hs.add('i');
        hs.add('o');
        hs.add('u');
        hs.add('A');
        hs.add('E');
        hs.add('I');
        hs.add('O');
        hs.add('U');
        int count1 = 0;
        int count2 = 0;
        for(int i = 0; i < s.length()/2; i++){
            if(hs.contains(s.charAt(i))){
                count1++;
            }
        }
        for(int i = s.length()/2; i < s.length(); i++){
            if(hs.contains(s.charAt(i))){
                count2++;
            }
        }
        return count1 == count2;
    }
}
