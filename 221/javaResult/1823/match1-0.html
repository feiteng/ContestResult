class Solution {
    public boolean halvesAreAlike(String s) {
        int left = 0;
        int right = 0;
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' || c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U' ) {
                if (i < s.length() / 2) {
                    left ++;
                } else {
right++;
                }
            }
        }
        return left==right;
    }
}
