class Solution {
    public boolean halvesAreAlike(String s) {
        int size = s.length();
        char[] arr = {'a', 'e', 'i', 'o', 'u', 'A', 'E', 'I', 'O', 'U'};
        
int count = 0;
        for (int i = 0; i < (size >> 1); i++) {
            for (int j = 0; j < arr.length; j++) {
                if (arr[j] == s.charAt(i)) {
                    count++;
                    break;
                }
            }
        }
        for (int i = (size >> 1); i < size; i++) {
            for (int j = 0; j < arr.length; j++) {
                if (arr[j] == s.charAt(i)) {
                    count--;
                    break;
                }
            }
        }
        return count == 0;
    }
}
