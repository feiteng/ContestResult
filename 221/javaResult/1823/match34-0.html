class Solution {
    public boolean halvesAreAlike(String s) {
        Character[] temp = new Character[]{'a','e','i','o', 'u', 'A', 'E', 'I', 'O', 'U'};
        List<Character> vol = Arrays.asList(temp);
        int n = s.length();
        int a = 0;
        int b = 0;
        for (int i = 0; i < n / 2; i++) {
            if (vol.contains(s.charAt(i))) a++;
            if (vol.contains(s.charAt(i + n / 2))) b++;
        }

        return a == b;
    }
}
