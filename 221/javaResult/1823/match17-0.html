class Solution {
    public boolean halvesAreAlike(String s) {
        int len = s.length();
        int mid = s.length() / 2;
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('e');
        set.add('i');
        set.add('o');
        set.add('u');
        set.add('A');
        set.add('E');
        set.add('I');
        set.add('O');
        set.add('U');

        int x = 0, y = 0;
        for (int i = 0; i < mid; i++) {
            if (set.contains(s.charAt(i))) {
                x++;
            }
        }
        for (int i = mid; i < len; i++) {
            if (set.contains(s.charAt(i))) {
                y++;
            }
        }
        return x == y;
    }
}
