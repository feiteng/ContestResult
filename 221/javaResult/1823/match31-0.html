class Solution {
    public boolean halvesAreAlike(String s) {
        int n = s.length();
        char[] cs = s.toCharArray();
        String ak = "aeiouAEIOU";
        int l = 0, r = 0;
        for (int i = 0; i < n / 2; i++) {
            if (ak.indexOf(cs[i]) != -1){
                l++;
            }
        }
        for (int i = n / 2; i < n; i++) {
            if (ak.indexOf(cs[i]) != -1){
                r++;
            }
        }
        return r == l;
    }
}
