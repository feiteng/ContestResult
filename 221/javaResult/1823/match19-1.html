class Solution {
    public boolean halvesAreAlike(String s) {
        Set<Character> set = new HashSet<>();
        set.add('a');
        set.add('A');
        set.add('e');
        set.add('E');
        set.add('i');
        set.add('I');
        set.add('o');
        set.add('O');
        set.add('u');
        set.add('U');
        
        int n = s.length();
        int count = 0;
        
for (int i = 0; i < n / 2; i++) {
            if (set.contains(s.charAt(i))) {
                count++;
            }
        }
        
        for (int i = n / 2; i < n; i++) {
            
if (set.contains(s.charAt(i))) {
                count--;
            }        
        }
        
        return count == 0;
    }
}
