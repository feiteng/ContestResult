class Solution {
    public boolean halvesAreAlike(String s) {
        char[] temp = {'a','e','i','o','u','A','E','I','O','U'};
        Set<Character> set = new HashSet<>();
        for (char c : temp){
            set.add(c);
        }
        int count = 0;
        for(int i = 0; i < s.length() / 2; i++){
            if (set.contains(s.charAt(i))){
                count++;
            }
        }
        for(int i = s.length() / 2; i < s.length(); i++){
            if (set.contains(s.charAt(i))){
                count--;
            }
        }
        return count == 0;
    }
}
