class Solution {
    public int eatenApples(int[] apples, int[] days) {
        Queue<int[]> queue = new PriorityQueue<int[]>((a, b)->Integer.compare(a[1], b[1]));
        int ans = 0;
        int i = 0;
        for(; i < apples.length; i++) {
            if(apples[i] != 0) {
                queue.offer(new int[]{apples[i], i + days[i]});
            }
            
            while(!queue.isEmpty() && queue.peek()[1] <= i) queue.poll();
            
            if(!queue.isEmpty()) {
                int[] item = queue.poll();
                ans++;
                if(item[0] > 1) queue.offer(new int[]{item[0]-1, item[1]});
            }
        }
        
        while(!queue.isEmpty()) {
            while(!queue.isEmpty() && queue.peek()[1] <= i) queue.poll();
            
            if(!queue.isEmpty()) {
                int[] item = queue.poll();
                ans++;
                if(item[0] > 1) queue.offer(new int[]{item[0]-1, item[1]});
            }
           
            i++;
        }
        
        return ans;
    }
}
