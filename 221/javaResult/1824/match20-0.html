class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int n = days.length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> (a[1] - b[1]));
        int cnt = 0;
        int i = 0;
        for (; i < n; i++) {
            int day = i + 1;
            while (!pq.isEmpty() && pq.peek()[1] < day){
                pq.poll();
            }
            if (apples[i] > 0){
                pq.add(new int[]{apples[i], day + days[i] - 1});
            }
            if (!pq.isEmpty()){
                int[] arr = pq.poll();
                cnt++;
                arr[0]--;
                if (arr[0] > 0) pq.add(arr);
            }
        }
        while(!pq.isEmpty()){
            int day = i + 1;
            while (!pq.isEmpty() && pq.peek()[1] < day){
                pq.poll();
            }
            if (!pq.isEmpty()){
                int[] arr = pq.poll();
                cnt++;
                arr[0]--;
                if (arr[0] > 0) pq.add(arr);
            }
            i++;
        }
        return cnt;
    }
}
