class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int n = apples.length;
        PriorityQueue<Item> pq = new PriorityQueue<>(n, (o1, o2) -> {
            return o1.end - o2.end;
        });
        int res = 0;
        
        for (int i = 0; i < n; i++) {
            if (apples[i] > 0) {
                pq.add(new Item(apples[i], i + days[i]));
            }
            while (!pq.isEmpty() && pq.peek().end <= i) {
                pq.poll();
            }
            
            if (!pq.isEmpty()) {
                Item it = pq.poll();
                res++;
                if (it.n > 1) {
                    pq.add(new Item(it.n - 1, it.end));
                }
            }
        }
        
        int day = n;
        while (!pq.isEmpty()) {
            Item it = pq.poll();
            
            if (it.end <= day) {
                continue;
            }
            
            int tmp = Math.min(it.end - day, it.n);
            day += tmp;
            res += tmp;
        }
        
        return res;
    }
    
    class Item {
        int n;
        int end;
        
        public Item(int n, int end) {
            this.n = n;
            this.end = end;
        }
    }
}
