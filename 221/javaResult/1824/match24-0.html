    public int eatenApples(int[] apples, int[] days) {
        // p: <When expire, appleCount>
        int res = 0;
        Queue<int[]> q = new PriorityQueue<>((int[] p1, int[] p2)->Integer.compare(p1[0], p2[0]));
        // Always eat apple that expires first 
        for(int i = 0; i < apples.length; i++) {
            if(apples[i] != 0) {
                q.add(new int[]{i+days[i]-1, apples[i]});
            }
            
            while(!q.isEmpty() && q.peek()[0] < i) {
                q.poll();
            }
            
            // q.isEmpty || q.peek()[0] >= i
            if(!q.isEmpty()) {
                int[] entry = q.poll();
                entry[1]--;
                if(entry[1] > 0) q.add(entry);
                res++;
            }
        }
        
        int i = apples.length;
        while(!q.isEmpty()) {
            while(!q.isEmpty() && q.peek()[0] < i) {
                q.poll();
            }
            
            // q.isEmpty || q.peek()[0] >= i
            if(!q.isEmpty()) {
                int[] entry = q.poll();
                entry[1]--;
                if(entry[1] > 0) q.add(entry);
                res++;
            }
            i++;
        }
        
        return res;
    }
