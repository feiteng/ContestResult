class Solution {
    public static int eatenApples(int[] apples, int[] days) {

        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);

        int count = 0;
        int day = 0;

        while(!pq.isEmpty() || day < apples.length) {
            if (day < apples.length && apples[day] != 0) {
                pq.add(new int[]{apples[day], days[day] + day});
            }

            while (!pq.isEmpty() && (pq.peek()[1] <= day || pq.peek()[0] == 0)) pq.poll(); // remove old and empty

            if (!pq.isEmpty()) {
                int[] backet = pq.poll();
                count++;
                backet[0]--;
                if (backet[0] > 0) pq.add(backet);
            }
            day++;
        }
        return count;
    }

    //  1 2 3 4 5 6 7 8 9 
    //  {3, 3}, 
}
