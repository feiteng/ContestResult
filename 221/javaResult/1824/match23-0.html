class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<int[]>(11, (a, b) -> (a[1] == b[1]) ? b[0] - a[0] : a[1] - b[1]);
        int n = apples.length;
        int eat = 0, day = 0;
        for(int i = 0; i < n; i++) {
            pq.add(new int[]{apples[i], days[i] + day - 1});
            while(!pq.isEmpty() && pq.peek()[1] < day) 
                pq.poll();
            if(!pq.isEmpty()) {
                int[] pp = pq.poll();
                eat++;
                pp[0]--;
                if(pp[0] > 0)
                    pq.add(pp);
            }
            day++;
        }
        while(!pq.isEmpty()) {
            while(!pq.isEmpty() && pq.peek()[1] < day) 
                pq.poll();
            if(!pq.isEmpty()) {
                int[] pp = pq.poll();
                eat++;
                pp[0]--;
                if(pp[0] > 0)
                    pq.add(pp);
            }
            day++;
        }
        return eat;
    }
}
