class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int n = apples.length;
        // a[0] number a[1] days
        int res = 0;
        PriorityQueue<int[]> queue = new PriorityQueue<int[]>((a, b) -> a[1] - b[1]);
        for (int i = 0; i < apples.length; i++) {
            if (apples[i] != 0) queue.offer(new int[]{apples[i], i + days[i]});
            while (!queue.isEmpty() && queue.peek()[1] <= i) queue.poll();
            if (!queue.isEmpty()) {
                int temp = queue.peek()[0];
                if (temp == 1) {
                    queue.poll();
                } else {
                    queue.peek()[0] = temp - 1;
                }
                res++;
            }
        }
        
        int i = n;
        while (!queue.isEmpty()) {
            while (!queue.isEmpty() && queue.peek()[1] <= i) queue.poll();
            if (!queue.isEmpty()) {
                int temp = queue.peek()[0];
                if (temp == 1) {
                    queue.poll();
                } else {
                    queue.peek()[0] = temp - 1;
                }
                res++;
            }
            i++;
        }
        return res;
    }
}
