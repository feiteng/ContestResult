class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Apple> pq = new PriorityQueue<Apple>();
        int count = 0;
        for(int j =0;j<40001;j++){
            if (j < apples.length && apples[j] != 0){
                pq.add(new Apple(j+days[j]-1, apples[j]));
            }
            while(!pq.isEmpty() && pq.peek().lastday < j){
                pq.remove();
            }
            if (!pq.isEmpty()){
                count++;
                Apple a = pq.remove();
                a.count -=1;
                if (a.count != 0){
                    pq.add(a);
                }
            }
        }
        return count;
    }
    static class Apple implements Comparable<Apple>{
        private int lastday;
        private int count;
        public Apple(int lastday, int count){
            this.count = count;
            this.lastday = lastday;
        }
        public int compareTo(Apple other){
            return Integer.compare(lastday, other.lastday);
        }
    }
}
