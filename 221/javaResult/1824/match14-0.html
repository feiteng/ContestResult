class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        
        int day = 0;
        int count = 0;
        while (day < apples.length || !pq.isEmpty()) {
            if (day < apples.length) {
                pq.offer(new int[]{apples[day], days[day] + day});
            }
            
            while (!pq.isEmpty() && pq.peek()[1] <= day) pq.poll();
            
            if (!pq.isEmpty()) {
                pq.peek()[0]--;
                count++;
                if (pq.peek()[0] == 0) pq.poll();
            }
            
            day++;
        }
        
        return count;
        
    }
}
