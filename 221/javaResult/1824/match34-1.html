class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int[] cnts = new int[50000];
        int eats = 0;
        int available = 0;
        int reduce = 0;
        for (int i = 0; i < apples.length; ++i) {
            if (days[i] > 0) {
                cnts[i + days[i]] += apples[i];
                available += apples[i];
            }
            if (cnts[i] > 0) {
                if (reduce > cnts[i]) {
                    reduce -= cnts[i];
                } else {
                    available -= cnts[i] - reduce;
                    reduce = 0;
                }
            }
            if (available > 0) {
                ++eats;
                ++reduce;
                --available;
            }
        }
        System.out.println(Arrays.toString(cnts));
        for (int i = apples.length; available > 0; ++i) {
            if (cnts[i] > 0) {
                if (reduce > cnts[i]) {
                    reduce -= cnts[i];
                } else {
                    available -= cnts[i] - reduce;
                    reduce = 0;
                }
            }
            if (available > 0) {
                ++eats;
                ++reduce;
                --available;
                System.out.println(i + ":" + available);
            }
        }
        return eats;
    }
}
