class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int o = 0;
        int n = apples.length;
        TreeMap<Integer, Integer> count = new TreeMap<>();
        
for (int i = 0; i < n || (!count.isEmpty()); i++) {
            if (i < n && apples[i] > 0) {
                int exp = i + days[i];
                count.put(exp, apples[i] + count.getOrDefault(exp, 0));
            }
            while ((!count.isEmpty()) && count.firstKey() <= i) {
                count.remove(count.firstKey());
            }
            
if (!count.isEmpty()) {
                int k = count.firstKey();
                int v = count.get(k) - 1;
                o++;
                if (v > 0) {
                    count.put(k, v);
                } else {
                    count.remove(k);
                }
            }
        }
        return o;
    }
}
