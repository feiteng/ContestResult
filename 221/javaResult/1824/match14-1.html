class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> expire = new PriorityQueue<>(11, (a, b) -> a[0] - b[0]);
        int total = 0;
        int curr = 0;
        while(curr<days.length || !expire.isEmpty()){
            if(curr<days.length && apples[curr]>0){
                expire.offer(new int[]{days[curr] + curr, apples[curr]});
            }
            while(!expire.isEmpty() && expire.peek()[0]<=curr){
                expire.poll();
            }
            if(!expire.isEmpty()){
                int[] currpack = expire.peek();
                if(currpack[1] == 1){
                    expire.poll();
                }else{
                    currpack[1]--;
                
}
                total++;
            }
            curr++;
        }
        return total;
    }
}
