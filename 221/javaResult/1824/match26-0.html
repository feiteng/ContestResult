class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Integer> pq = new PriorityQueue<>((i, j) -> (i+days[i] - (j+days[j])));        
        int res = 0;
        int i = 0;
        while (!pq.isEmpty() || i<apples.length) {
            if (i<apples.length)
                pq.add(i);
            while (!pq.isEmpty()) {
                if (pq.peek() + days[pq.peek()] - 1 >= i)
                    break;
                pq.poll();
            }
            
            if (!pq.isEmpty()) {    
                int index = pq.poll();
                apples[index]--;
                res++;
                if (apples[index] >0) {
                    pq.add(index);
                }
            }
            i++;
        }
        
        return res;
    }
}
