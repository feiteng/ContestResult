class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> heap = new PriorityQueue<int[]>((a,b)->(a[1]-b[1]));
        int i = 0;
        int len = days.length;
        int res = 0;
        while(i<len || !heap.isEmpty()){
            while(!heap.isEmpty() && heap.peek()[1]<=i)
                heap.poll();
            if(i<len && apples[i]!=0)
                heap.add(new int[]{apples[i],i+days[i]});
            if(!heap.isEmpty()){
                int[] cur = heap.poll();
                res++;
                if(cur[0]>1)
                    heap.add(new int[]{cur[0]-1,cur[1]});
            }
            i++;
        }
        return res;
    }
}
