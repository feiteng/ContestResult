class Solution {
    // seems like greedy
    // use heap to store apples
    public int eatenApples(int[] apples, int[] days) {
        // 0-last day, 1-cnt
        PriorityQueue<int[]> heap = new PriorityQueue<>((a, b) -> a[0] - b[0]);
        
        int res = 0;
        for(int i = 0; i < apples.length; i++) 
{
            if(apples[i] > 0) heap.add(new int[]{i + days[i] - 1, apples[i]}); // days[i] including today
            
            // eject rotten
            while (!heap.isEmpty() && heap.peek()[0] < i) heap.poll();
            
            // try to eat
            if(!heap.isEmpty()) {
                res++;
                heap.peek()[1]--;
                if(heap.peek()[1] == 0) heap.poll(); 
            }
        }
        
        int curDay = apples.length; // n-day
        
while(!heap.isEmpty()) {
            while(!heap.isEmpty() && heap.peek()[0] < curDay) heap.poll();
            
            if(!heap.isEmpty()) {
                int cur[] = heap.poll();
                
                int cnt = Math.min(cur[1], cur[0] - curDay + 1);
                res += cnt;
                curDay += cnt;
            }
        }
        return res;
    }
}
