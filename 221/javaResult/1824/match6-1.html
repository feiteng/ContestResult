class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int res = 0;
        TreeMap<Integer, Integer> tm = new TreeMap<>(); // <date, num>
        
        int i = 0;
        for (; i < apples.length; i++) {
            int date = i + days[i]; // at this date, no apple available
            tm.put(date, tm.getOrDefault(date, 0) + apples[i]);
            
            
while (!tm.isEmpty() && tm.firstKey() <= i) {
                tm.pollFirstEntry();
            }
            if (!tm.isEmpty()) {
                res++;
                int headDate = tm.firstKey();
                int headCount = tm.get(headDate);
                if (headCount == 1) {
                    tm.pollFirstEntry();
                } else
{
                    tm.put(headDate, headCount-1);
                }
            }
        }
        
        while (!tm.isEmpty()) {
            while (!tm.isEmpty() && tm.firstKey() <= i) {
                tm.pollFirstEntry();
            }
            if (!tm.isEmpty()) {
                res++;
                int headDate = tm.firstKey();
                int headCount = tm.get(headDate);
                if (headCount == 1) {
                    tm.pollFirstEntry();
                } else
{
                    tm.put(headDate, headCount-1);
                }
            }
            i++;
        }
        return res;
    }
}
