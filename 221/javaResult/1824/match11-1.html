class Solution {
    public int eatenApples(int[] ap, int[] days) {
        PriorityQueue<Node> heap = new PriorityQueue();
        int res = 0;
        for(int i=0;i<ap.length;++i){
            if(ap[i]>0){
                heap.offer(new Node(days[i]+i, ap[i]));
            }
            while(!heap.isEmpty() && heap.peek().r<=i)heap.poll();
            if(!heap.isEmpty()){
                Node nd = heap.peek();
                nd.c--;
                ++res;
                if(nd.c==0)heap.poll();
            }
        }
        int d = ap.length;
        while(!heap.isEmpty()){
            if(heap.peek().r<=d)heap.poll();
            else
{
                Node nd = heap.peek();
                nd.c--;
                ++res;
                if(nd.c==0)heap.poll();
                ++d;
            }
        }
        return res;
    }
}
class Node implements Comparable<Node>{
    int r;
    int c;
    public Node(int a, int b){
        r=a;
        c=b;
    }
    @Override
    public int compareTo(Node n){
        return this.r - n.r;
    }
}
