class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Node> q = new PriorityQueue<>();
        int ans = 0;
        for(int i = 0;i < apples.length; ++i){
            if(apples[i] != 0){
                q.add(new Node(apples[i], i+days[i]));
            }
            while(!q.isEmpty() && q.peek().expire <= i) q.poll();
            if(!q.isEmpty()){
                ans += 1;
                Node now = q.poll();
                if(--now.num > 0) q.add(now);
            }
        }
        int x = apples.length;
        while(!q.isEmpty()){
            while(!q.isEmpty() && q.peek().expire <= x) q.poll();
            if(!q.isEmpty()){
                ans += 1;
                Node now = q.poll();
                if(--now.num > 0) q.add(now);
            }
            x += 1;
        }
        return ans;
    }
}
class Node implements Comparable{
    int num;
    int expire;
    public Node(int num, int expire){
        this.num = num;
        this.expire = expire;
    }

    @Override
    public int compareTo(Object o) {
        Node O = (Node) o;
        return this.expire - O.expire;
    }
}
