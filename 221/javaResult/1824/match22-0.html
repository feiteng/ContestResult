class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int i=1;
        for(int j=0;j<days.length;j++){
            if(days[j]!=0){
                days[j]+=i;
            }
            i++;
        }
        int n=apples.length;
        int res=0;
        PriorityQueue<int[]> queue=new PriorityQueue<>((a,b)->a[1]-b[1]);
        i=1;
        for(int j=0;j<n;j++){
            while(!queue.isEmpty()&&queue.peek()[1]<=i){
                queue.poll();
            }
            if(apples[j]!=0){
                queue.offer(new int[]{apples[j],days[j]});
            }
            if(!queue.isEmpty()){
                int[] temp=queue.poll();
                res++;
                if(temp[0]>1){
                    queue.offer(new int[]{temp[0]-1,temp[1]});
                }
            }
            i++;
            
        }
        while(!queue.isEmpty()){
            while(!queue.isEmpty()&&queue.peek()[1]<=i){
                queue.poll();
            }
            if(!queue.isEmpty()){
                int[] temp=queue.poll();
                res++;
                if(temp[0]>1){
                    queue.offer(new int[]{temp[0]-1,temp[1]});
                }
            }
            i++;
        }
        return res;
    }
}
