class Solution {
    public int eatenApples(int[] apples, int[] days) {
        
        int eaten = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a1, a2)->a1[1]-a2[1]);
        
        int d = 0;
        for(d = 0; d < days.length; d++){
            pq.offer(new int[]{apples[d], d + days[d]});
            
            while(!pq.isEmpty() && pq.peek()[1] <= d)
                pq.poll();
            
            if(!pq.isEmpty()){
                int[] apple = pq.poll();
                eaten++;
                if(apple[0]-- > 1){
                    pq.offer(apple);
                }
            }
        }
        
        while(!pq.isEmpty()){
            while(!pq.isEmpty() && pq.peek()[1] <= d)
                pq.poll();
            
            if(!pq.isEmpty()){
                int[] apple = pq.poll();
                eaten++;
                if(apple[0]-- > 1){
                    pq.offer(apple);
                }
            }
            
            d++;
        }
        
        return eaten;
        
    }
}
