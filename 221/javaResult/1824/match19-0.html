class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int ans=0;
        Queue<int[]> q=new PriorityQueue<>((a,b)->a[0]-b[0]);
        int i=0;
        while(!q.isEmpty()||i<apples.length){
            if(i<apples.length&&!(days[i]==0&&apples[i]==0)){
                q.offer(new int[]{i+days[i], apples[i]});
            }
            while(!q.isEmpty()&&q.peek()[0]<=i){
                q.poll();
            }
            if(!q.isEmpty()){
                q.peek()[1]--;
                if(q.peek()[1]==0){
                    q.poll();
                }
                ans++;
            } 
            i++;
        }
        return ans;
    }
}
