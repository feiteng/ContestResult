class Solution {
    class Apples {
        int num;
        int rotDay;
        public Apples(int _n, int _r) {
            num = _n;
            rotDay = _r;
        }
    }
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Apples> queue = new PriorityQueue<>((a, b) -> a.rotDay - b.rotDay);
        int i = 0;
        int eaten = 0;
        while (i < apples.length || !queue.isEmpty()) {
            if (i < apples.length && apples[i] > 0) {
                queue.add(new Apples(apples[i], i + days[i]));
            }
            while (!queue.isEmpty() && queue.peek().rotDay <= i) {
                queue.poll();
            }
            if (!queue.isEmpty()) {
                Apples a = queue.poll();
                a.num--;
                eaten++;
                if (a.num > 0) {
                    queue.add(a);
                }
            }
            i++;
        }
        
        return eaten;
    }
}
