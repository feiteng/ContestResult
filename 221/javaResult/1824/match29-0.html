class Solution {
    public int eatenApples(int[] apples, int[] days) {
        TreeMap<Integer, Integer> expToApples = new TreeMap<>();
        
        int ret = 0;
        
        for (int d = 0; d < days.length || !expToApples.isEmpty(); ++d) {
            // pick up apples
            if (d < days.length && apples[d] > 0) {
                int expDate = d + days[d];
                expToApples.put(expDate, expToApples.getOrDefault(expDate, 0) + apples[d]);
            }
            
            // delete the rot
            expToApples.remove(d);
            
            // eat one apple if exists
            if (!expToApples.isEmpty()) {
                ++ret;
                int firstExp = expToApples.firstKey();
                int applesRemaining = expToApples.get(firstExp) - 1;
                if (applesRemaining == 0) {
                    expToApples.remove(firstExp);
                } else {
                    expToApples.put(firstExp, applesRemaining);
                }                
            }
        }
        
        return ret;
    }
}
