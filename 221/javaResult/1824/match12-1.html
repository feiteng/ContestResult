class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> exp = new PriorityQueue<int[]>((a,b)->{
            return a[0]-b[0];
        });
        
        int eaten =0;
        
        for(int i = 0; i < apples.length; i++){
            int expDate = i+days[i];
            
exp.add(new int[] {expDate,apples[i]});
            
            while(!exp.isEmpty()){
                int[] toEat = exp.poll();
                if(toEat[0] > i){
                    eaten++;
                    toEat[1]--;
                    if(toEat[1]>0){
                        exp.add(toEat);
                    }
                    break;
                
}
            }
        }
        int day = apples.length;
        while(!exp.isEmpty()){
            int[] toEat = exp.poll();
            if(toEat[0] > day){
                eaten++;
                day++;
                toEat[1]--;
                    if(toEat[1]>0){
                        exp.add(toEat);
                    }
            }
        }
        return eaten;
    }
}
