class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((o1, o2) -> o1[1] - o2[1]);
        int ans = 0;
        int i = 0;
        for (; i < apples.length; i++) {
            if(apples[i] > 0)
                pq.add(new int[]{apples[i], i + days[i] - 1});
            while(!pq.isEmpty() && pq.peek()[1] < i) pq.poll();
            if(pq.isEmpty()) continue;
            int[] top = pq.poll();
            //System.out.println(i + " " + top[0] + " " + top[1]);
            if(top[1] >= i) {
                ans++;
                top[0]--;
                if (top[0] > 0 && top[1] > i) {
                    pq.add(top);
                }
            }
        }
        
        while (!pq.isEmpty()) {
            while(!pq.isEmpty() && pq.peek()[1] < i) pq.poll();
            if(pq.isEmpty()) continue;
            int[] top = pq.poll();
            // System.out.println(i + " " + top[0] + " " + top[1]);
            if (top[1] >= i) {
                ans++;
                top[0]--;
                if (top[0] > 0 && top[1] > i) {
                    pq.add(top);
                }
            }
            i++;
        }
        return ans;
    }
}
