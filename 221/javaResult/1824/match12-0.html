class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> heap = new PriorityQueue<>((a, b) -> (a[1] - b[1]));
        int max = 0;
        int n = apples.length;
        for (int i = 0; i < n; i++) {
            heap.add(new int[]{apples[i], i + days[i]});
            while (!heap.isEmpty()) {
                int[] cur = heap.poll();
                if (i < cur[1]) {
                    if (cur[0] > 0) {
                        cur[0]--;
                        max++;
                        heap.add(cur);
                        break;
                    }
                }
            }
        }
        
        int day = n;
        while (!heap.isEmpty()) {
            int[] cur = heap.poll();
            if (day < cur[1]) {
                if (cur[0] > 0) {
                    cur[0]--;
                    max++;
                    heap.add(cur);
                    day++;
                }
            }
        }
        return max;
    }
}
