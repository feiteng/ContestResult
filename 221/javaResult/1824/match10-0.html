class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> {
            return a[0] - b[0];
        });
        int eat = 0;
        for(int i = 0; i < 50000; i++) {
            if(i < apples.length) {
                pq.add(new int[]{i + days[i] - 1, apples[i]});
            }
            while(pq.size() > 0 && (pq.peek()[0] < i || pq.peek()[1] <= 0)) {
                pq.poll();
            }
            if(pq.size() > 0) {
                eat++;
                pq.peek()[1]--;
            }
        }
        return eat;
    }
}
