class Solution {
    public static class Node {
        int count;
        int expire;
        public Node(int count, int expire) {
            this.count = count;
            this.expire = expire;
        }
    }
    
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Node> list = new PriorityQueue<>((a, b) -> a.expire - b.expire);
        
        int count = 0;
        int i=0;
        for(;i<apples.length;i++) {
            if(apples[i]>0) {
                list.add(new Node(apples[i], days[i]+i-1));
            }
            
            //remove expired ones
            while(!list.isEmpty() && list.peek().expire < i) {
                list.poll();
            }
            
            if (!list.isEmpty()) {
                Node n = list.poll();
                count++;
                if (n.count>1) {
                    list.add(new Node(n.count-1, n.expire));
                }
            }
        }
        
        while(!list.isEmpty()) {
            //remove expired ones
            while(!list.isEmpty() && list.peek().expire < i) {
                list.poll();
            }
            
            if (!list.isEmpty()) {
                Node n = list.poll();
                count++;
                if (n.count>1) {
                    list.add(new Node(n.count-1, n.expire));
                }
            }
            i++;
        }
        
        return count;
    }
}
