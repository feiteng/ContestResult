class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int res = 0;
        int len = apples.length;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        int i = 0;
        for (; i < len; i++) 
{
            if (apples[i] > 0) {
                pq.offer(new int[]{apples[i], i + days[i]});
            }
            while (!pq.isEmpty() && pq.peek()[1] <= i) {
                pq.poll();
            }
            if (!pq.isEmpty()) {
                int[] cur = pq.poll();
                res++;
                cur[0]--;
                if (cur[0] > 0) {
                    pq.offer(cur);
                }
            }
        }
        while (!pq.isEmpty()) {
            while (!pq.isEmpty() && pq.peek()[1] <= i) {
                pq.poll();
            }
            if (!pq.isEmpty()) {
                int[] cur = pq.poll();
                res++;
                cur[0]--;
                if (cur[0] > 0) 
{
                    pq.offer(cur);
                }
            }
            i++;
        }
        return res;
    }
}
