class Solution {
    class Day {
        int rot;
        int num;
        
        public Day(int rot, int num) {
            this.rot = rot;
            this.num = num;
        }
    }
    
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Day> minHeap = new PriorityQueue<>((a, b) -> a.rot - b.rot);
        int n = apples.length;
        int res = 0;
        int cur = 0;
        
        while (! minHeap.isEmpty() || cur < n) {
            if (cur < n) {
                int num = apples[cur];
                int day = cur + days[cur];
                
                
minHeap.offer(new Day(day, num));
            }
            
            while (! minHeap.isEmpty() && minHeap.peek().rot <= cur) {
                minHeap.poll();
            }
            
            if (! minHeap.isEmpty()) {
                Day d = minHeap.poll();
                d.num --;
                res ++;
                if (d.num != 0) {
                    minHeap.offer(d);
                
}
            }
            
            cur ++;
        }
        
        return res;
    }
}
