class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int n=apples.length,ans=0;        
        PriorityQueue<int[]>pq=new PriorityQueue<>((a,b)->a[1]-b[1]);   
        int i=0;
        int []pr;
        while(pq.size()>0||i<n)
{
            if(i<n&&apples[i]>0)
                pq.offer(new int[]{apples[i],i+days[i]});
            while(pq.size()>0&&pq.peek()[1]<=i)
                pq.poll();
            if(pq.size()>0){
                pr=pq.poll();
                pr[0]--;
                ans++;
                if(pr[0]>0)
                    pq.offer(pr)
;
            }
            i++;
        }
        return ans;
    }
}
