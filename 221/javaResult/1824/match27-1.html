class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->(a[1]-b[1]));
        int appcount = 0;
        for(int i=0; i<apples.length; i++)
{
            if(apples[i] != 0)
                pq.add(new int[]{apples[i], i+days[i]-1});
                
                while(!pq.isEmpty() && pq.peek()[1] < i)
                    pq.poll();
                if(pq.isEmpty())continue;
                int[] temp = pq.poll();
                appcount++;
                temp[0]--;
                if(temp[0] != 0)
                    pq.add(temp);
        }
        int d = apples.length;
        
while(!pq.isEmpty()){
            while(!pq.isEmpty() && pq.peek()[1] < d)
                    pq.poll();
            
            if(!pq.isEmpty()){
                appcount++;
                d++;
                int[] temp = pq.poll();
                temp[0]--;
                if(temp[0] != 0)
                    pq.add(temp)
;
            }
            
        }
        return appcount;
    }
}
