class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int res = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->a[0]-b[0]);
        int i=0;
        for(;i<apples.length;i++){
            if(apples[i]>0){
                pq.add(new int[]{days[i]+i, apples[i]});
            }
            
            while(!pq.isEmpty() && pq.peek()[0]<=i){
                pq.remove();
            }
            
            if(!pq.isEmpty()){
                res++;
                int[] a = pq.remove();
                a[1] = a[1]-1;
                if(a[1]>0){
                    pq.add(a);
                }
            }
        }

        while(!pq.isEmpty()){
            while(!pq.isEmpty() && pq.peek()[0]<=i){
                pq.remove();
            }
            if(!pq.isEmpty()){
                res++;
                int[] a = pq.remove();
                a[1] = a[1]-1;
                if(a[1]>0){
                    pq.add(a);
                }
                i++;
            }
        }
        return res;
    }
}
