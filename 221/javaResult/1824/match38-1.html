class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq=new PriorityQueue<>((a,b)->(a[1]-b[1]));
        int res=0;
        for(int i=0;i<apples.length;i++)
{
            while(!pq.isEmpty()&&pq.peek()[1]<i) pq.poll();
            if(apples[i]>0)
                pq.add(new int[]{i,i+days[i]-1});
            if(!pq.isEmpty()){
                res++;
                int[] temp=pq.peek();
                apples[temp[0]]--;
                if(apples[temp[0]]==0) pq.poll();
            }
        }
        int cur=days.length;
        while(!pq.isEmpty()){
            while(!pq.isEmpty()&&(pq.peek()[1]<cur||apples[pq.peek()[0]]==0)) pq.poll();
            if(!pq.isEmpty()){
                res++;
                int[] temp=pq.peek();
                apples[temp[0]]--;
                if(apples[temp[0]]==0) pq.poll()
;
            }
            cur++;
        }
        return res;
    }
}
