class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<Apple> pq = new PriorityQueue<>();
        int maxEat = 0;
        for(int i = 0 ; i < apples.length; ++i) {
            if(i + days[i] > i) {
                pq.offer(new Apple(apples[i], i + days[i]));
            }
            
            while(pq.size() > 0 && pq.peek().bestBefore <= i) {
                pq.poll();
            }
            
            if(pq.size() > 0) {
                Apple top = pq.peek();
                ++maxEat;
                top.count -=1;
                if(top.count == 0) {
                    pq.poll();
                }
            }
        }
        
        int day = apples.length;
        while(pq.size() > 0) {
            while(pq.size() > 0 && pq.peek().bestBefore <= day) {
                pq.poll();
            }
            
            if(pq.size() > 0) {
                Apple top = pq.peek();
                ++maxEat;
                top.count -=1;
                if(top.count == 0) {
                    pq.poll();
                }
            }
            
            ++day;
        }
        
        
return maxEat;
    }
    
    private class Apple implements Comparable<Apple> {
        public int count;
        public int bestBefore;
        
        public Apple(int count, int bestBefore) {
            this.count = count;
            this.bestBefore = bestBefore;
        }
        
        @Override
        public int compareTo(Apple other) {
            return this.bestBefore - other.bestBefore;
        }
        
    }
}
