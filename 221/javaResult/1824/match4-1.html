class Solution {
   public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> queue = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        
int n = apples.length;
        int res = 0;
        int i = 0;
        for (; i < n; i++) {
            if (apples[i] > 0 && days[i] > 0) {
                queue.offer(new int[]{i + days[i], apples[i]});
            }

            while (!queue.isEmpty() && queue.peek()[0] <= i) {
                queue.poll();
            }

            if (!queue.isEmpty()) {
                int[] poll = queue.poll();
                poll[1]--;
                
if (poll[1] > 0) {
                    queue.offer(poll);
                }
                res++;
            }
        }

        while (!queue.isEmpty()) {
            while (!queue.isEmpty() && queue.peek()[0] <= i) {
                queue.poll();
            }

            if (!queue.isEmpty()) {
                int[] poll = queue.poll();
                poll[1]--;
                
if (poll[1] > 0) {
                    queue.offer(poll);
                }
                res++;
            }
            i++;
        }

        return res;
    }
}
