class Solution {
	public int eatenApples(int[] apples, int[] days) {
        int res = 0;
        
        // count, index
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> a[1] - b[1]);
        int i = 0;
        for (; i < days.length; i++) {			
        	while (pq.size() > 0 && pq.peek()[1] < i) {
        		pq.poll();
        	}
        	if (apples[i] > 0) {
        		pq.add(new int[] { apples[i], i + days[i] - 1 });
        	}
        	if (pq.
size() > 0) {
        		int[] a = pq.poll();
        		a[0]--;
        		res++;
        		if (a[0] > 0) {
        			pq.add(a);
        		}
        	}
		}
        while (pq.size() > 0) {
        	while (pq.size() > 0 && pq.peek()[1] < i) {
        		pq.poll();
        	}

        	if (pq.size() > 0) {
        		int[] a = pq.poll();
        		a[0]--;
        		res++;
        		if (a[0] > 0) {
        			pq.add(a);
        		}
        	}
        	i++;
        }
        
        return res;
    }
}
