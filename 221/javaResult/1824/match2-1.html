class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b)->Integer.compare(a[1],b[1]));
        int res = 0, day = 0;
        while (day < apples.length || !pq.isEmpty()) 
{
            if (day < apples.length && apples[day] > 0) {
                pq.offer(new int[] { apples[day], day+days[day]-1 });
            }
            while (!pq.isEmpty() && pq.peek()[1] < day) pq.poll();
            if (!pq.isEmpty()) {
                int[] cur = pq.poll();
                res++;
                cur[0]--;
                if (cur[0] > 0) pq.offer(cur)
;
            }
            day++;
        }
        return res;
    }
}
