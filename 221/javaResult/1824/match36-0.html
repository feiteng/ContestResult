class Solution {
    public int eatenApples(int[] apples, int[] days) {
        int n = apples.length;
        int res = 0;
        PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> a[1] - b[1]);
        int i = 0;
        while (i < n || !pq.isEmpty()) {
            if (i < n && apples[i] > 0) {
                pq.add(new int[]{apples[i], i + days[i]});
            }
            while (!pq.isEmpty() && pq.peek()[1] <= i) {
                pq.poll();
            }
            if (!pq.isEmpty()) {
                res++;
                if (pq.peek()[0] == 1) {
                    pq.poll();
                } else {
                    pq.peek()[0]--;
                }
            }
            i++;
        }
        return res;
    }
}
