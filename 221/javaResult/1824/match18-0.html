class Solution {
    public int eatenApples(int[] apples, int[] days) {
        PriorityQueue<int[]> pq = new PriorityQueue<>((a1,a2) -> Integer.compare(a1[0], a2[0]));
        int index = 0;
        int count = 0;
        while(index < apples.length) {
            if(apples[index] > 0) pq.offer(new int[] {index + days[index], apples[index]});
            while(!pq.isEmpty() && pq.peek()[0] <= index) pq.poll();
            if(!pq.isEmpty()) {
                int[] cur = pq.poll();
                if(cur[1] > 0) {
                    count++;
                    cur[1]--;
                    if(cur[1] > 0) pq.offer(cur);
                }
            }
            index++;
        }
        while(!pq.isEmpty()) {
            while(!pq.isEmpty() && pq.peek()[0] <= index) pq.poll();
             if(!pq.isEmpty()) {
                int[] cur = pq.poll();
                int eat = Math.min(cur[0]-index, cur[1]);
                index += eat;
                count+=eat;
            }
        }
        return count;
    }
}
