class Solution {
    public static int eatenApples(int[] apples, int[] days) {
        if (apples == null || apples.length == 0) {
            return 0;
        }
        int n = apples.length;
        //a[0]表示过期时间,a[1]表示剩余苹果数量
        PriorityQueue<int[]> heap = new PriorityQueue<>(new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[0] - o2[0];
            }
        });
        int res = 0;//吃掉的苹果
        for (int i = 0; i < n; i++) {
            if (apples[i] > 0) {
                int[] cur = {i + days[i], apples[i]};
                heap.add(cur);
            }
            while (!heap.isEmpty() && heap.peek()[0] <= i) {
                heap.poll();
            }
            if (!heap.isEmpty()) {
                int[] cur = heap.peek();
                cur[1]--;
                res++;
                if (cur[1] == 0) {
                    heap.poll();
                }
            }
        }

        int i = n;
        while (!heap.isEmpty()) {
            while (!heap.isEmpty() && heap.peek()[0] <= i) {
                heap.poll();
            }
            if (!heap.isEmpty()) {
                int[] cur = heap.peek();
                cur[1]--;
                res++;
                if (cur[1] == 0) {
                    heap.poll();
                }
            }
            i++;
        }

        return res;
    }
}
