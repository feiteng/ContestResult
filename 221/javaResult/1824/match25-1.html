class Solution {
    
    public int eatenApples(int[] apples, int[] days) {
        
        int n = apples.length;
        
        // well there are 4 * 10^4 days to think about
        PriorityQueue<Element> pq = new PriorityQueue<>((a,b) -> {
            return a.exp - b.exp;
        });
        
        int eaten = 0;
        for (int i = 0; i < n; i++) {
            
            if (apples[i] == 0) {
                // do nothing
            } else {
                
pq.offer(new Element(apples[i], i + days[i]));
            }
            
            while (!pq.isEmpty() && pq.peek().exp == i) {
                pq.poll();
            }
            
            if (!pq.isEmpty() && pq.peek().exp > i) {
                Element e = pq.poll();
                eaten += 1;
                int newQuantity = e.quantity - 1;
                
if (newQuantity != 0 && e.exp != i+1) {
                    pq.offer(new Element(newQuantity, e.exp));
                }
            }
            
        }
        
        int t = n;
        while (!pq.isEmpty()) {
            while (!pq.isEmpty() && pq.peek().exp == t) {
                pq.poll();
            }
            if (!pq.isEmpty() && pq.peek().exp > t) {
                Element e = pq.poll();
                eaten += 1;
                int newQuantity = e.quantity - 1;
                if (newQuantity != 0 && e.exp != t+1) {
                    pq.offer(new Element(newQuantity, e.exp));
                }
            }
            t += 1;
        
}
        
        return eaten;
    }
    
    class Element {
        
        int quantity;
        int exp;
        
        public Element (int q, int exp) {
            this.quantity = q;
            this.exp = exp;
        }
        
    }
}
