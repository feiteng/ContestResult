    public int eatenApples(int[] apples, int[] days) {
        int res = 0;
        PriorityQueue<int[]> minHeap = new PriorityQueue<>((i1, i2) -> Integer.compare(i1[0], i2[0]));
        for (int i = 0; i < days.length || !minHeap.isEmpty(); i++) 
{
            if (i < days.length && apples[i] > 0) {
                minHeap.offer(new int[] {i + days[i], apples[i]});
            }
            while (!minHeap.isEmpty() && i >= minHeap.peek()[0]) {
                minHeap.poll();
            }
            if (!minHeap.isEmpty()) {
                res++;
                int[] cur = minHeap.poll();
                
if (--cur[1] > 0) {
                    minHeap.offer(cur);
                }
            }
        }
        return res;
    }
}
