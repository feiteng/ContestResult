class Solution {
    Node root = new Node(0);
    private void add(int x){
        Node now = root;
        for(int i = 30;i >= 0; --i){
            if((x&(1<<i)) > 0) {
                if (now.rson == null) now.rson = new Node(1);
                now = now.rson;
            } else {
                if(now.lson == null) now.lson = new Node(0);
                now = now.lson;
            }
        }
    }
    private int xor(int x){
        Node now = root;
        int ans = 0;
        for(int i = 30;i >= 0; --i){
            if((x&(1<<i)) > 0){
                if(now.lson != null) {
                    ans |= 1 << i;
                    now = now.lson;
                } else {
                    now = now.rson;
                }
            } else {
                if(now.rson != null){
                    ans |= 1<<i;
                    now = now.rson;
                } else {
                    now = now.lson;
                }
            }
        }
        return ans;
    }
    public int[] maximizeXor(int[] nums, int[][] queries) {
        Integer[] a = new Integer[queries.length];
        for(int i = 0;i < a.length; ++i) a[i] = i;
        Arrays.sort(a, new Comparator<Integer>() {
            @Override
            public int compare(Integer o1, Integer o2) {
                return queries[o1][1] - queries[o2][1];
            }
        });
        Arrays.sort(nums);
        int[] ans = new int[a.length];
        int x = 0;
        for(int i = 0;i < a.length; ++i){
            while(x < nums.length && nums[x] <= queries[a[i]][1]){
                add(nums[x++]);
            }
            if(x == 0) ans[a[i]] = -1;
            else ans[a[i]] = xor(queries[a[i]][0]);
        }
        return ans;
    }
}
class Node{
    int x;
    Node lson;
    Node rson;
    public Node(int x){
        this.x = x;
        lson = rson = null;
    }
}
