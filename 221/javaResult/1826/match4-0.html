class Solution {
    class tree{
        tree[] next;
        int end;
        public tree(){
            end = 0;
            next = new tree[2];
        }
    }
    tree root;

    public int[] maximizeXor(int[] nums, int[][] queries) {
        root = new tree();
        int n = queries.length, m = nums.length;
        int[] ans = new int[n];
        int[][] qq = new int[n][3];
        for(int i = 0; i < n; i++) {
            qq[i][0] = queries[i][0];
            qq[i][1] = queries[i][1];
            qq[i][2] = i;
        }
        Arrays.sort(nums);
        Arrays.sort(qq, (o1, o2) -> o1[1] - o2[1]);

        int j = 0;
        for(int i = 0; i < n; i++) {
            while(j < m && nums[j] <= qq[i][1]) {
                add(nums[j++]);
            }
            ans[qq[i][2]] = find(qq[i][0]);
        }

        return ans;
    }

    private int find(int num){
        tree p = root;
        for(int i = 31; i >= 0; i--) {
            int x = (num >> i) & 1;
            if(p.next[1 - x] != null) p = p.next[1 - x];
            else if(p.next[x] != null) p = p.next[x];
            else return -1;
        }
        return num ^ p.end;
    }

    private void add(int num) {
        tree p = root;
        for(int i = 31; i >= 0; i--) {
            int x = (num >> i) & 1;
            if(p.next[x] == null) p.next[x] = new tree();
            p = p.next[x];
        }
        p.end = num;
    }
}
