class Solution {
    public int[] maximizeXor(int[] nums, int[][] queries) {
        int n = queries.length, p=0;
        Arrays.sort(nums);
        
int[][] qr = new int[n][3];
        for(int i=0;i<n;++i){
            qr[i][0]=queries[i][0];
            qr[i][1]=queries[i][1];
            qr[i][2]=i;
        }
        Arrays.sort(qr, (a,b)->(a[1]-b[1]));
        int[] res = new int[n];
        Trie tr = new Trie();
        for(int i=0;i<n;++i){
            while(p<nums.length && nums[p]<=qr[i][1])add(tr, nums[p++]);
            if(p==0){
                res[qr[i][2]] = -1;
                continue;
            }
            int k = 1<<30, v = 0;
            Trie tem = tr;
            while(k>0){
                if((qr[i][0]&k)==0){
                    if(tem.next[1]!=null){
                        v += k;
                        tem = tem.next[1];
                    }else tem = tem.next[0]
;
                }else{
                    if(tem.next[0]!=null){
                        v += k;
                        tem = tem.next[0];
                    }else tem = tem.next[1];
                }
                k = k>>1;
            }
            
res[qr[i][2]] = v;
        }
        return res;
    }
    private void add(Trie tr, int n){
        int k = 1<<30;
        Trie tem = tr;
        while(k>0)
{
            if((n&k)==0){
                if(tem.next[0]==null)tem.next[0]=new Trie();
                tem = tem.next[0];
            }else{
                if(tem.next[1]==null)tem.next[1]=new Trie();
                tem = tem.next[1];
            }
            k = k>>1;
        }
    }
}
class Trie{
    Trie[] next;
    public Trie(){
        next = new Trie[2];
    }
}
