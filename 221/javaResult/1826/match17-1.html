class Solution {
    class Node {
        Node[] child = new Node[2];
        int value = -1;
    }
    void addNode(Node root, int num) {
        Node cur = root;
        for (int i = 30; i>= 0; i--) {
            if (contain(num, i)) 
{
                if (cur.child[1] == null) {
                    cur.child[1] = new Node();
                }
                cur = cur.child[1];
            } else {
                if (cur.child[0] == null) {
                    cur.child[0] = new Node();
                }
                cur = cur.child[0];
            }
        }
        cur.value = num;
    }
    
    boolean contain(int num, int i) {
        return (num & (1 << i)) > 0;
    }
    
    public int[] maximizeXor(int[] nums, int[][] queries) {
        Arrays.sort(nums);
        int[][] nq = new int[queries.length][3];
        for (int i = 0; i < queries.length; i++) {
            for (int j = 0; j < 2; j++) {
                nq[i][j] = queries[i][j];
            }
            nq[i][2] = i;
        }
        Arrays.sort(nq, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                return o1[1] - o2[1];
            }
        });
        
        Node root = new Node();

        int[] res = new int[nq.length];
        int j = 0;
        for (int i = 0; i < nq.length; i++) {
            while (j < nums.length && nq[i][1] >= nums[j]) {
                addNode(root, nums[j]);
                j++;
            }
            
            if (root.child[0] == null && root.child[1] == null) {
                res[nq[i][2]] = -1;
            } else {
                Node cur = root;
                for (int k = 30; k>= 0;k--) {
                    if (contain(nq[i][0], k)) 
{
                        if (cur.child[0] != null) {
                            cur = cur.child[0];
                        } else {
                            cur = cur.child[1];
                        }
                    } else {
                        if (cur.child[1] != null) {
                            cur = cur.child[1];
                        } else {
                            cur = cur.child[0];
                        }
                    }
                }
                
res[nq[i][2]] = cur.value ^ nq[i][0];
            }
        }
        return res;
    }
}
