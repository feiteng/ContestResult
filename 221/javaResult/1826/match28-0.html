class Solution {
    class TrieNode {
    TrieNode left;
    TrieNode right;
    public TrieNode() {}
  }

  class Trie {
    private TrieNode root;

    public Trie() {
      root = new TrieNode();
    }

    
    // val has to be <= 2 ^ (LAYER - 1) - 1
    private void add(int val) {
      TrieNode iter = root;
      for (int i = 30; i >= 0; i--) {
        int bit = val & (1 << i);
        if (bit == 0) {
          if (iter.left == null) {
            iter.left = new TrieNode();
          }
          iter = iter.left;
        } else {
          if (iter.right == null) {
            iter.right = new TrieNode();
          }
          iter = iter.right;
        }
      }
    }

    private int queryXor(int val) {
      int res = 0;
      TrieNode iter = root;
      for (int i = 30; i >= 0; i--) {
        int bit = val & (1 << i);

        if (bit == 0) {
          if (iter.right != null) {
            iter = iter.right;
            res = 2 * res + 1;
          } else {
            iter = iter.left;
            res = 2 * res;
          }
        } else {
          if (iter.left != null) {
            iter = iter.left;
            res = 2 * res + 1;
          } else {
            iter = iter.right;
            res = 2 * res;
          }
        }
      }
      return res;
    }
  }

  class Pair implements Comparable<Pair> {
    int[] query;
    int idx;
    public Pair(int[] query, int idx) {
      this.query = query;
      this.idx = idx;
    }

    @Override
    public int compareTo(Pair o) {
      if (query[1] != o.query[1]) {
        return Integer.compare(query[1], o.query[1]);
      }
      return Integer.compare(idx, o.idx);
    }

    @Override
    public String toString() {
      return Arrays.toString(query) + " " + idx;
    }
  }

  public int[] maximizeXor(int[] nums, int[][] queries) {
    List<Pair> ls = new ArrayList<>();
    for (int i = 0; i < queries.length; i++) {
      ls.add(new Pair(queries[i], i));
    }
    Collections.sort(ls);
    Arrays.sort(nums);

    int[] res = new int[ls.size()];
    Arrays.fill(res, -1);
    int iter = 0;

    Trie ins = new Trie();
    for (int i = 0; i < ls.size(); i++) {
      Pair p = ls.get(i);
      while (iter < nums.length && nums[iter] <= p.query[1]) {
        ins.add(nums[iter]);
        iter++;
      }

      if (iter != 0) {
        res[p.idx] = ins.queryXor(p.query[0]);
      }
    }
    return res;
  }
}
