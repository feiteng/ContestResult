class Solution {
    public int[] maximizeXor(int[] nums, int[][] queries) {
        Arrays.sort(nums);
        int[] answer = new int[queries.length];
        for (int i = 0; i < queries.length; ++i) {
            int num = queries[i][0];
            int max = queries[i][1];
            int start = 0;
            int end = bs(nums, max);
            
            if (nums[0] > max) answer[i] = -1;
            else {
            for (int bit = 30; bit >= 0; --bit) {
                
                // System.out.println(start + " " + end);
                if ((num & (1 << bit)) != 0) {
                    int st = bsunsetend(nums, start, end, bit);
                    if (st >= start &&  ((nums[st] & (1 << bit)) == 0)) {
                        end = st;
                    }
                } else {
                    int st = bssetstart(nums, start, end, bit);
                    if (st <= end && ((nums[st] & (1 << bit)) != 0)) {
                        start = st;
                    }
                }
            }
            answer[i] = nums[start] ^ num;
            }
            
                // System.out.println("------------");
        }
        
        return answer;
    }
    
    int bs(int[] nums, int num) {
        int s = 0;
        int e = nums.length-1;
        while (s < e) {
            int m = (s + e + 1) / 2;
            if (nums[m] > num) {
                e = m-1;
            } else if (nums[m] <= num) {
                s = m;
            } 
        }
        return s;
    }
    
    int bssetstart(int[] nums, int st, int en, int bit) {
        int s = st;
        int e = en;
        while (s < e) {
            int m = (s + e) / 2;
            if ((nums[m] & (1 << bit)) != 0) {
                e = m;
            } else {
                s = m+1;
            } 
        }
        return s;
    }
    
    int bsunsetend(int[] nums, int st, int en, int bit) {
        int s = st;
        int e = en;
        while (s < e) {
            int m = (s + e + 1) / 2;
            if ((nums[m] & (1 << bit)) != 0) {
                e = m-1;
            } else {
                s = m;
            } 
        }
        return s;
    }
}
