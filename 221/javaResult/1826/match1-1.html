class Solution {
    public int[] maximizeXor(int[] nums, int[][] q) {
        // System.out.println();
        Trie root = new Trie();
        int[][] queries = new int[q.length][q[0].length + 1];
        for(int i = 0; i < q.length; i++) {
            queries[i][0] = q[i][0];
            queries[i][1] = q[i][1];
            queries[i][2] = i;
        }
        Arrays.sort(nums);
        Arrays.sort(queries, (a, b) -> {
            return a[1] - b[1];
        });
        int[] output = new int[q.length];
        Arrays.fill(output, -1);
        
int p = 0;
        for(int i = 0; i < q.length; i++) {
            // System.out.println(nums[p] +","+ queries[i][1]);
            while(p < nums.length && nums[p] <= queries[i][1]) {
                insert(root, nums[p]);
                p++;
            }
            Trie curr = root;
            int res = 0;
            if(root.left == null && root.right == null) {
                // System.out.println(p);
                continue;
            }
            
for(int j = 31; j >= 0; j--) {
                if(((1 << j) & queries[i][0]) == 0) {
                    if(curr.right != null) {
                        res |= (1 << j);
                        curr = curr.right;
                    } else {
                        curr = curr.left;
                    }
                } else {
                    if(curr.left != null) {
                        res |= (1 << j);
                        curr = curr.left;
                    } else {
                        curr = curr.right;
                    }
                }
            }
            output[queries[i][2]] = res;
        }
        return output;
    }
    
    void insert(Trie root, 
int val) {
        Trie curr = root;
        for(int i = 31; i >= 0; i--) {
            if(((1 << i) & val) == 0) {
                if(curr.left == null) curr.left = new Trie();
                curr = curr.left;
            } else {
                if(curr.right == null) curr.right = new Trie();
                curr = curr.right;
            }
        }
    }
    
    static class Trie {
        Trie left;
        Trie right;
    }
    
}
