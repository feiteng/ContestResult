class Solution {
        public int[] maximizeXor(int[] nums, int[][] queries2) {
            int n = nums.length;
            int m = queries2.length;

            Arrays.sort(nums);
            int[][]queries = new int[m][3];
            for (int i = 0; i < queries2.length; i++) {
                queries[i][0] = queries2[i][0];
                queries[i][1] = queries2[i][1];
                queries[i][2] = i;
            }
            Arrays.sort(queries,(x1,x2)->x1[1]-x2[1]);
            int i,j;
            i = j = 0;
            int[] answer = new int[m];
            Trie trie = new Trie();

            for(int[] query:queries){
                while(j<n&&nums[j]<=query[1]){
                    trie.insert(nums[j++]);
                }
                if(j==0){
                    answer[query[2]] = -1;
                    continue;
                }
                answer[query[2]] = trie.search(~query[0]);
            }
            return answer;

        }
        public class Trie {
            private boolean is_string=false;
            private Trie next[]=new Trie[26];

            public Trie(){}

            public void insert(String word){//插入单词
                Trie root=this;
                char w[]=word.toCharArray();
                for(int i=0;i<w.length;++i){
                    if(root.next[w[i]-'a']==null)root.next[w[i]-'a']=new Trie();
                    root=root.next[w[i]-'a'];
                }
                root.is_string=true;
            }
            public void insert(int word){//插入单词
                Trie root=this;

                for(int i=30;i>=0;--i){
                    if(root.next[(word>>i)&1]==null) {
                        root.next[(word>>i)&1] = new Trie();
                    }
                    root=root.next[(word>>i)&1];
                }
                root.is_string=true;
            }

            public int search(int word){//查找单词
                int ret = 0;
                Trie root=this;
//                char w[]=word.toCharArray();
                boolean flag = false;
                for(int i=30;i>=0;--i){
                    // if(word>=(1>>i)){
                    //     flag = true;
                    // }
                    if(root.next[(word>>i)&1]==null) {
                        root = root.next[1 - ((word >> i) & 1)];
                    }
                    else{
                        root=root.next[(word>>i)&1];
                        ret+=1<<i;
                    }
                }
//                return root.is_string;
                return ret;
            }

            public boolean search(String word){//查找单词
                Trie root=this;
                char w[]=word.toCharArray();
                for(int i=0;i<w.length;++i){
                    if(root.next[w[i]-'a']==null)return false;
                    root=root.next[w[i]-'a'];
                }
                return root.is_string;
            }

            public boolean startsWith(String prefix){//查找前缀
                Trie root=this;
                char p[]=prefix.toCharArray();
                for(int i=0;i<p.length;++i){
                    if(root.next[p[i]-'a']==null)return false;
                    root=root.next[p[i]-'a'];
                }
                return true;
            }
        }
    }
