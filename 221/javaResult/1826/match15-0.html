class Solution {
  class Node {
    public Node left, right;
    public int cntL, cntR, bit;
    private Node(){}
    public Node(int val) {
      left = right = null;
      cntL = cntR = 0;
      bit = val;
    }
  }
  
  void add(Node root, int val) {
    Node x = root;
    int bit = root.bit;
    while(bit > 0) {
      if((x.bit & val) > 0) { // right
        if(x.right == null) {
          x.right = new Node(bit/2);
        }
        x.cntR++;
        bit/=2;
        x = x.right;        
      } else { // left
        if(x.left == null) {
          x.left = new Node(bit/2);
        }
        x.cntL++;
        bit/=2;
        x = x.left;
      }
    }
  }
  
  public int[] maximizeXor(int[] arr, int[][] qq) {
    int n = qq.length;
    int[] ans = new int[n];
    int[][] data = new int[n][3];
    for(int i = 0; i < n; i++) {
      data[i][0] = qq[i][0];
      data[i][1] = qq[i][1];
      data[i][2] = i;
    }
    Arrays.sort(arr);
    Arrays.sort(data, (a, b) -> a[1] - b[1]);
    int next = 0, m = arr.length;

    Node root = new Node(1<<30);
    
    for(int i = 0; i < n; i++) {
      int limit = data[i][1];
      while(next < m && arr[next] <= limit) {
        // System.out.format("adding %d\n", arr[next]);
        add(root, arr[next++]);
      }
      // use arr[0 .. next -1]
      int result = data[i][0], val = data[i][0];
      if(root.cntL + root.cntR == 0) ans[data[i][2]] = -1;
      else {
        Node x = root;
        int bit = x.bit;
        while(bit > 0) {
          // System.out.format("val = %d, x.bit = %d, x.left, right = %d, %d ", val, x.bit, x.cntL, x.cntR);
          if((x.bit & val) > 0) { // '1'
            if(x.left != null) { // go left
              x = x.left;
            } else {
              result ^= x.bit;
              x = x.right;
            }
          } else { // '0'
            if(x.right != null) {
              result ^= x.bit;
              x = x.right;
            } else {
              x = x.left;
            }
          }
          // System.out.format("--> result = %d\n", result);
          bit/=2;
        }
        ans[data[i][2]] = result;
      }
    }
    return ans;    
  }
}
