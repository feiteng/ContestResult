class Solution {
    public int[] maximizeXor(int[] nums, int[][] queries) {
        int n = queries.length;
        int[] o = new int[n];
        
TrieNode root = new TrieNode();
        TreeSet<Integer> set = new TreeSet<>();
        for (int i : nums) {
            set.add(i);
            TrieNode node = root;
            for (int j = 30; j >= 0; j--) {
                int bit = (i >> j) & 1;
                if (node.children[bit] == null) {
                    node.children[bit] = new TrieNode();
                }
                node = node.children[bit];
            }
        }
        
for (int i = 0; i < n; i++) {
            TrieNode node = root;
            int x = queries[i][0];
            int m = queries[i][1];
            int y = 0;
            for (int j = 30; j >= 0; j--) {
                int bit = (1 << j);
                int xbit = (x >> j) & 1;
                if (m < bit) {
                    if (node.children[0] != null) {
                        node = node.children[0];
                    } else {
                        o[i] = -1;
                        break;
                    }
                } else {
                    if (xbit == 0) {
                        Set<Integer> subset = set.subSet(y + bit, true, y + m, true);
                        if (node.children[1] != null&& subset != null && (!subset.isEmpty())) {
                            node = node.children[1];
                            m -= bit;
                            y |= bit;
                        } else if (node.children[0] != null) {
                            node = node.children[0];
                        } else {
                            o[i] = -1;
                            break;
                        }
                    } else {
                        if (node.children[0] != null) {
                            node = node.children[0];
                        } else if (node.children[1] != null) {
                            node = node.children[1];
                            m -= bit;
                            y |= bit;
                        } else {
                            o[i] = -1;
                            break;
                        }
                    }
                }
            }
            if (o[i] != -1) 
{
                o[i] = (x ^ y);
            }
        }
        return o;
    }

    class TrieNode {
        TrieNode[] children;

        TrieNode() {
            this.children = new TrieNode[2];
        }
    }
}
