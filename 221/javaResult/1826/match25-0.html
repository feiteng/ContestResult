class Node {
    Node[] next = new Node[2];
}
class Solution {
    public int[] maximizeXor(int[] nums, int[][] queries) {
        TreeSet<Integer> set = new TreeSet<>();
        Node root = new Node();
        for (int num : nums) {
            set.add(num);
            Node node = root;
            for (int i = 30; i >= 0; i--) {
                int bit = (num & (1 << i)) >> i;
                if (node.next[bit] == null) {
                    node.next[bit] = new Node();
                }
                node = node.next[bit];
            }
        }
        int[] res = new int[queries.length];
        for (int k = 0; k < queries.length; k++) {
            int x = queries[k][0];
            int m = queries[k][1];
            Node node = root;
            int path = 0;
            for (int i = 30; i >= 0; i--) {
                int bit = (x & (1 << i)) >> i;
                int lower = path | (1 << i);
                int upper = Math.min(m, path | ( (1 << (i+1)) - 1 ));
                if (node.next[0] == null || bit == 0 && lower <= upper && !set.subSet(lower, true, upper, true).isEmpty()) {
                    node = node.next[1];
                    path |= (1 << i);
                } else {
                    node = node.next[0];
                }
                if (node == null) {
                    res[k] = -1;
                    break;
                }
            }
            System.out.println(path);
            if (res[k] != -1 && path <= m) {
                res[k] = x ^ path;
            } else {
                res[k] = -1;
            }
        }
        return res;
    }
}
