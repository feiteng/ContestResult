class Solution {
    class Node{
        Node l, r;
    }
    void add(Node root, int val){
        int i = 1, k = 31;
        while (k >= 0) {
            
if (((i << k) & val) != 0) {
                if (root.l == null) {
                    root.l = new Node();
                }
                root = root.l;
            }
            else {
                if (root.r == null) {
                    root.r = new Node();
                }
                root = root.r;
            }
            k--;
        }
    }
    int find(Node root, int val) {
        int i = 1, k = 31, res = 0;
        while (k >= 0) {
            
if (((i << k) & val) != 0) {
                if (root.r != null) {
                    root = root.r;
                    res = res * 2;
                } else {
                    root = root.l;
                    res = res * 2 + 1;
                }
            } else {
                if (root.l != null) {
                    root = root.l;
                    res = res * 2 + 1;
                } else {
                    root = root.r;
                    res = res * 2;
                }
            }
            k--;
        }
        return res;
    }
    public int[] maximizeXor(int[] nums, int[][] queries) {
        Integer[] pos = new Integer[queries.length];
        Node root = new Node();
        for (int i = 0; i < pos.length; i++)
            pos[i] = i;
        Arrays.sort(pos, Comparator.comparingInt(o -> queries[o][1]));
        Arrays.sort(nums);
        int l = -1, r = 0;
        int[] res = new int[pos.length];
        while (r < pos.length) {
            while (l < nums.length - 1 && nums[l + 1] <= queries[pos[r]][1]) {
                l++;
                add(root,nums[l]);
            }
            if (l == -1) {
                res[pos[r]] = -1;
            } else {
                int t = queries[pos[r]][0];
                res[pos[r]] = t ^ find(root,t);
            }
            r++;
        }
        return res;
    }
}
