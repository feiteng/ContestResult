/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {

    public boolean isEvenOddTree(TreeNode root) {
        if (root == null) {
            return true;
        }
        Deque<TreeNode> q = 
new ArrayDeque<>();
        q.addLast(root);
        boolean evenFlag = true;
        while (!q.isEmpty()) {
            int size = q.size();
            evenFlag = !evenFlag;
            int last = -1;
            
for (int i = 0; i < size; i++) {
                TreeNode node = q.pollFirst();
                if (evenFlag) {
                    if ((node.val & 1) != 0) {
                        return false;
                    }
                    if (i > 0 && !(node.val < last)) {
                        return false;
                    }
                } else {
                    if ((node.val & 1) == 0) {
                        return false;
                    }
                    if (i > 0 && !(node.val > last)) {
                        return false;
                    }

                }
                last = node.val;
                //    add
                if (node.left != null) {
                    q.add(node.left);
                }
                if (node.right != null) {
                    q.add(node.right);
                }
            }
        }
        return true;
    }

}
