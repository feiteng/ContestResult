/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        if (root == null) {
            return false;
        }
        boolean evenLevel = true;
        Queue<TreeNode> queue = new LinkedList<> ();
        queue.offer(root);
        Integer prev = null;
        while (!queue.isEmpty()) {
            int size = queue.size();
            prev = null;
            for (int i = 0; i < size; i++) {
                TreeNode curt = queue.poll();
                if (evenLevel) {
                    if (curt.val % 2 == 0) {
                        return false;
                    }
                    if (prev != null && curt.val <= prev) {
                        return false;
                    }
                } else {
                    if (curt.val % 2 != 0) {
                        return false;
                    }
                    if (prev != null && curt.val >= prev) {
                        return false;
                    }
                }
                prev = curt.val;
                if (curt.left != null) {
                    queue.offer(curt.left);
                }
                if (curt.right != null) {
                    queue.offer(curt.right);
                }
            }
            evenLevel = !evenLevel;
        }
        return true;
    }
}
