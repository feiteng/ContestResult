/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        if (root == null)
            return true;
        Queue<TreeNode> q = new ArrayDeque<TreeNode>();
        q.add(root);
        boolean flag = true;// true奇数，false偶数
        while (!q.isEmpty()) {
            int size = q.size();
            int lastNums = flag ? Integer.MIN_VALUE : Integer.MAX_VALUE;
            for (int i = 0; i < size; i++) {
                TreeNode curr = q.poll();
                int currVal = curr.val;
                if (flag) {
                    // 奇数且递增
                    if (!(currVal > lastNums && (currVal & 1) == 1)) {
                        return false;
                    }
                } else {
                    // 偶数且递减
                    if (!(currVal < lastNums && (currVal & 1) != 1)) {
                        return false;
                    }
                }
                lastNums = currVal;
                if (curr.left != null)
                    q.add(curr.left);
                if (curr.right != null)
                    q.add(curr.right);
            }
            flag = !flag;
        }
        return true;
    }
}
