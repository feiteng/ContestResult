/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        int d=-1;
        List<TreeNode> queue=new LinkedList<>();
        
queue.add(root);
        while(queue.size()!=0){
            d++;
            int size=queue.size();
            if(d%2==1){
                TreeNode last=queue.get(0);
                queue.remove(0);
                if(last.val%2==1) return false;
                if(last.left!=null)queue.add(last.left);
                if(last.right!=null)queue.add(last.right);
                
for(int i=1;i<size;++i){
                    TreeNode cur=queue.get(0);
                    if(cur.val>=last.val) return false;
                    last=cur;
                    queue.remove(0);
                    if(last.val%2==1) return false;
                    
if(last.left!=null)queue.add(last.left);
                    if(last.right!=null)queue.add(last.right);
                
                }
            }
            else{
                    TreeNode last=queue.get(0);
                    queue.remove(0);
                    if(last.val%2==0) return false;
                    if(last.left!=null)queue.add(last.left);
                    if(last.right!=null)queue.add(last.right);
                    
for(int i=1;i<size;++i){
                        TreeNode cur=queue.get(0);
                        if(cur.val<=last.val) return false;
                        last=cur;
                        queue.remove(0);
                        if(last.val%2==0) return false;
                        
if(last.left!=null)queue.add(last.left);
                        if(last.right!=null)queue.add(last.right);
                    }
            }
            
        }
        return true;
    }
    

}
