/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    //层次遍历即可
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> que=new LinkedList<>();
        que.add(root);
        
        boolean odd=false;//要求递减
        while(!que.isEmpty()){
            int size=que.size();
            int prv=odd?Integer.MAX_VALUE:-1;
            for(int i=0;i<size;i++){
                TreeNode curNode=que.poll();
                int cur=curNode.val;
                if(odd){
                    if(cur%2!=0){
                        return false;
                    }
                    if(cur>=prv){
                        // System.out.println(cur+">="+prv);
                        return false;
                    }
                }else{
                    if(cur%2==0){
                        return false;
                    }
                    if(cur<=prv){
                        // System.out.println(cur+"<="+prv);
                        return false;
                    }
                }
                prv=cur;
                if(curNode.left!=null){
                    que.add(curNode.left);
                }
                if(curNode.right!=null){
                    que.add(curNode.right);
                }
            }
            odd=!odd;
        }
        return true;
    }
}
