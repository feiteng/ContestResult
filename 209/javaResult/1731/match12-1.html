/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        if (root == null) {
            return true;
        }
        Queue<TreeNode> queue = new ArrayDeque<>();
        queue.offer(root);
        int level = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            TreeNode prev = null;
            while (size > 0) 
{
                size--;
                
                TreeNode cur = queue.poll();
                //even
                if (level % 2 == 0) {
                    if (cur.val % 2 != 1) {
                        return false;
                    }
                    if (prev != null && prev.val >= cur.val) {
                        return false;
                    }
                } else {
                    if (cur.val % 2 != 0) {
                        return false;
                    }
                    if (prev != null && prev.val <= cur.val) {
                        return false;
                    }
                }
                
if (cur.left != null) {
                    queue.offer(cur.left);
                }
                if (cur.right != null) {
                    queue.offer(cur.right);
                }
                prev = cur;
            }
            level++;
        }
        return true;
    }
}
