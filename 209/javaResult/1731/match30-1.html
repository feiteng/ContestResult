/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        LinkedList<TreeNode> ll=new LinkedList();
        ll.add(root);
        int lv=0;
        while(!ll.isEmpty()){
            int l=ll.size();
            int pre=-1;
            
boolean f = lv%2==0;
            while(l>0){
                TreeNode n=ll.poll();
                if(f){
                    if(n.val%2==0 || (pre!=-1 && n.val<=pre))return false;
                }else{
                    if(n.val%2==1 || (pre!=-1 && n.val>=pre))return false;
                }
                
pre=n.val;
                if(n.left!=null)ll.add(n.left);
                if(n.right!=null)ll.add(n.right);
                --l;
            }
            ++lv;
        }
        return true;
    }
}
