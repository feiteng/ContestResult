/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        boolean even = true;
        List<TreeNode> q = new ArrayList<>();
        q.add(root);
        while (!q.isEmpty()) {
            int prev = even ? Integer.MIN_VALUE : Integer.MAX_VALUE;
            List<TreeNode> nq = new ArrayList<>();
            for (TreeNode r : q) {
                if (even && (r.val % 2 == 0)) return false;
                
if (!even && (r.val % 2 == 1)) return false;
                if (even && r.val <= prev) return false;
                if ((!even) && r.val >= prev) return false;
                prev = r.val;
                if (r.left != null) nq.add(r.left);
                if (r.right != null) nq.add(r.right);
            }
            even = !even;
            q = nq;
        }
        return true;
    }
}
