class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> level = new LinkedList<>();
        level.offer(root);
        int l = 0;
        while (!level.isEmpty()) {
            int sz = level.size();
            int last = l%2==0 ? -1 : (int) 1e7;
            for (int i=0; i<sz; i++) {
                TreeNode cur = level.poll();
                if ((l%2==0 && (cur.val<=last || cur.val%2==0)) || (l%2==1 && (cur.val>=last || cur.val%2==1))) return false;
                last = cur.val;
                if (cur.left!=null) level.offer(cur.left);
                if (cur.right!=null) level.offer(cur.right);
            }
            l++;
        }
        return true;
    }
}
