/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> que = new LinkedList<>();
        que.offer(root);
        boolean isEven = false;
        while (!que.isEmpty()) {
            int size = que.size();
            int num = 0;
            if (isEven) num = Integer.MAX_VALUE;
            else num = Integer.MIN_VALUE;
            
for (int i = 0; i < size; ++i) {
                TreeNode node = que.poll();
                if (isEven) {
                    if (num <= node.val || node.val % 2 == 1) return false;
                } else {
                    if (num >= node.val || node.val % 2 == 0) return false;
                }
                num = node.val;
                if (node.left != null) que.offer(node.left);
                if (node.right != null) que.offer(node.right);
            
}
            isEven = !isEven;
        }
        return true;
    }
}
