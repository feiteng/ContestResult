/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Deque<TreeNode> queue = new ArrayDeque<>();
        queue.offer(root);
        boolean isOdd = false;
        while (!queue.isEmpty()) {
            int size = queue.size();
            int pre = 0;
            if (isOdd) {
                pre = Integer.MAX_VALUE;
            } else {
                pre = Integer.MIN_VALUE;
            }
            
for (int i = 0; i < size; i++) {
                TreeNode cur = queue.poll();
                if (isOdd) {
                    if (pre <= cur.val) {
                        return false;
                    }
                    if (cur.val % 2 == 1) {
                        return false;
                    }
                } else {
                    if (pre >= cur.val) {
                        return false;
                    }
                    if (cur.val % 2 == 0) {
                        return false;
                    }
                }
                
if (cur.left != null) {
                    queue.offer(cur.left);
                }
                if (cur.right != null) {
                    queue.offer(cur.right);
                }
                pre = cur.val;
            }
            isOdd = !isOdd;
        }
        return true;
    }
}
