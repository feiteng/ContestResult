/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        if (root == null) {
            return true;
        }
        
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        int level = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            int limit = 0;
            if (level % 2 != 0) {
                limit = Integer.MAX_VALUE;
            }
            
while (size-- > 0) {
                TreeNode temp = queue.poll();
                if (level % 2 == 0) {
                    // even
                    if (temp.val % 2 == 0 || temp.val <= limit) {
                        return false;
                    }
                    
limit = temp.val;
                } else {
                    // odd
                    if (temp.val % 2 != 0 || temp.val >= limit) {
                        return false;
                    }
                    limit = temp.val;
                
}
                
                if (temp.left != null) {
                    queue.add(temp.left);
                }
                if (temp.right != null) {
                    queue.add(temp.right);
                }
            }
            level++;
        }
        return true;
    }
}
