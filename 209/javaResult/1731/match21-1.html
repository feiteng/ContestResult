/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {        
        Queue<TreeNode> queue=new LinkedList<>();
        queue.offer(root);
        boolean oddIncreasing=true;
        while(queue.isEmpty()==false){
            int size=queue.size();
            int prev=Integer.MAX_VALUE;
            if(oddIncreasing) prev=0;
            while(size-->0){
                TreeNode node=queue.poll();
                int polled = node.val;
                
if(oddIncreasing){
                    if(polled % 2==0) return false;
                    if(polled<=prev) return false;                    
                }
                else{
                    if(polled % 2==1) return false;
                    if(polled>=prev) return false;                    
                }
                prev=polled;
                if(node.left!=null) queue.offer(node.left);
                if(node.right!=null) queue.offer(node.right);
            }      
            oddIncreasing = !oddIncreasing;
        }
        return true;
    }
}
