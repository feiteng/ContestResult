/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        
        boolean level = true;
        while(!q.isEmpty()) {
            int size = q.size();
            
            Integer pre = null;
            // System.out.println(level);
            for(int i = 0; i < size; i++) {
                TreeNode cur = q.poll();
                // System.out.print(cur.val + " ");
                if(level) {
                    if(cur.val % 2 == 0) {
                        return false;
                    }
                    
                    if(pre == null) {
                        pre = cur.val;
                    } else if(cur.val <= pre) {
                        return false;
                    } else {
                        pre = cur.val;
                    }
                } else {
                    if(cur.val % 2 != 0) {
                        return false;
                    }
                    
                    if(pre == null) {
                        pre = cur.val;
                    } else if(cur.val >= pre) {
                        return false;
                    } else {
                        pre = cur.val;
                    }
                
}
                
                if(cur.left != null) {
                    q.offer(cur.left);
                }
                
                if(cur.right != null) {
                    q.offer(cur.right);
                }
            
}
            
            // System.out.println();
            level = !level;
        }
        
        return true;
    }
}
