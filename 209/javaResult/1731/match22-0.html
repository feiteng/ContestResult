/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
      
    
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        ((LinkedList<TreeNode>) queue).add(root);
        boolean flag = false;
        while (!queue.isEmpty()) {
            int size = queue.size();
            int c = 0;
            if (!flag) {
                c = Integer.MIN_VALUE;
            } else c = Integer.MAX_VALUE;
            while (size != 0) {
                TreeNode middle = queue.poll();
                if (!flag) {
                    if (middle.val % 2 == 0) return false;
                    if (middle.val <= c) return false;
                } else {
                    if (middle.val % 2 == 1) return false;
                    if (middle.val >= c) return false;
                }
                c = middle.val;
                if (middle.left != null) queue.add(middle.left);
                if (middle.right != null) queue.add(middle.right);
                size--;
            }
            flag = !flag;
        }
        return true;
    }

}
