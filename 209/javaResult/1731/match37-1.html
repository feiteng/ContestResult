/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        if (root == null) return true;
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        boolean isEven = true;
        while (!q.isEmpty()) {
            int size = q.size();
            int prev = -1;
            for (int i = 0; i < size; i++) {
                TreeNode node  = q.poll();
                
if (isEven) {
                    if (node.val %2 == 0) {
                        return false;
                    }
                } else {
                    if (node.val % 2 == 1) {
                        return false;
                    }
                }
                if (i != 0) {
                    if (isEven) {
                        if (node.val <= prev) {
                            return false;
                        }
                    }
                    else {
                        if (node.val >= prev) {
                            return false;
                        }
                    
}
                }
                prev = node.val;
                if (node.left != null) q.offer(node.left);
                if (node.right != null) q.offer(node.right);
            }
            isEven = !isEven;
        }
        
        return true;
    }
}
