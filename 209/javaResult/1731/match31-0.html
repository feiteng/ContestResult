/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> q = new LinkedList<>();
        q.offer(root);
        int level = 0;
        while(!q.isEmpty()) {
            int sz = q.size();
            Integer pre = null;
            if(level % 2 == 0) {
                for(int i = 0; i < sz; i++) {
                    TreeNode cur = q.poll();
                    if(cur.val % 2 == 0 || pre != null && cur.val <= pre) return false;
                    pre = cur.val;
                    if(cur.left != null) q.offer(cur.left);
                    if(cur.right != null) q.offer(cur.right);
                }
            } else {
                for(int i = 0; i < sz; i++) {
                    TreeNode cur = q.poll();
                    if(cur.val % 2 != 0 || pre != null && cur.val >= pre) return false;
                    pre = cur.val;
                    if(cur.left != null) q.offer(cur.left);
                    if(cur.right != null) q.offer(cur.right);
                }
            }
            level++;
        }
        return true;
    }
}
