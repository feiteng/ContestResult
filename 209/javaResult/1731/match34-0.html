/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        if (root == null) {
            return true;
        }
        Queue<TreeNode> que = new LinkedList<>();
        que.add(root);
        int level = 0;
        
        while (!que.isEmpty()) {
            int pre = -1;
            int qlen = que.size();
            while (qlen > 0) {
                TreeNode p = que.poll();
                qlen--;
                // 偶数层
                if ((level & 1) == 0) {
                    if ((p.val & 1) == 0) {
                        return false;
                    }
                    if (pre == -1) {
                        pre = p.val;
                    } else {
                        if (p.val <= pre) {
                            return false;
                        }
                        pre = p.val;
                    }
                } else {
                    // 奇数层
                    if ((p.val & 1) == 1) {
                        return false;
                    }
                    if (pre == -1) {
                        pre = p.val;
                    } else {
                        if (p.val >= pre) {
                            return false;
                        }
                        pre = p.val;
                    }
                }
                
                if (p.left != null) {
                    que.add(p.left);
                }
                if (p.right != null) {
                    que.add(p.right);
                }
            }
            level++;
        }
        
        return true;
    }
}
