/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode > bfs = new LinkedList();
        bfs.offer(root);
        int level=0;
        while(!bfs.isEmpty()){
            int size= bfs.size();
            if(level++%2==0){
                int prev = -1;
                for(int i=0;i<size;i++){
                    TreeNode temp = bfs.poll();
                    // System.out.println(temp.val+"  "+prev);
                    if(temp.val%2!=1 || temp.val<=prev) return false
;
                    if(temp.left!=null) bfs.offer(temp.left);
                    if(temp.right!=null) bfs.offer(temp.right);
                    prev = temp.val;
                }
            }else{
                int prev = 100000000;
                for(int i=0;i<size;i++){
                    TreeNode temp = bfs.poll();                    
                    if(temp.val%2!=0 || temp.val>=prev) return false
;
                    if(temp.left!=null) bfs.offer(temp.left);
                    if(temp.right!=null) bfs.offer(temp.right);
                    prev = temp.val;
                }
            }
        }
        return true;
    }
}
