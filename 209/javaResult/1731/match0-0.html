/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    
    private static class Node {
        public TreeNode treeNode;
        public int level;
        
        public Node(TreeNode treeNode, int level) {
            this.treeNode = treeNode;
            this.level = level;
        }
    }
    
    private List<List<Integer>> lst = new ArrayList<>();
    
    public boolean isEvenOddTree(TreeNode root) {
        int prev = -1;
        Queue<Node> q = new LinkedList<>();
        q.add(new Node(root, 0));
        Node p;
        while (!q.isEmpty()) {
            p = q.poll();
            if (p.level != prev) {
                prev = p.level;
                lst.add(new ArrayList<Integer>());
            }
            lst.get(lst.size()-1).add(p.treeNode.val);
            if (p.treeNode.left != null) {
                q.add(new Node(p.treeNode.left, prev+1));
            }
            if (p.treeNode.right != null) {
                q.add(new Node(p.treeNode.right, prev+1));
            }
        }
        List<Integer> temp;
        for (int i = 0; i <= prev; i++) {
            temp = lst.get(i);
            if (i % 2 == 1) {
                if (temp.get(0) % 2 == 1) {
                    return false;
                }
                for (int j = 1; j < temp.size(); j++) {
                    if (temp.get(j) >= temp.get(j-1)) {
                        return false;
                    }
                    if (temp.get(j) % 2 == 1) {
                        return false;
                    }
                }
            } else {
                if (temp.get(0) % 2 == 0) {
                    return false;
                }
                for (int j = 1; j < temp.size(); j++) {
                    if (temp.get(j) <= temp.get(j-1)) {
                        return false;
                    }
                    if (temp.get(j) % 2 == 0) {
                        return false;
                    }
                }
            }
        }
        return true;
    }
}
