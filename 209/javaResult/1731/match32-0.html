/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        Queue<TreeNode> queue = new LinkedList<>();
        queue.add(root);
        boolean even = true;
        while (!queue.isEmpty()) {
            int n = queue.size();
            int prev = -1;
            for (int i = 0; i < n; i++) {
                TreeNode node = queue.poll();
                if (i > 0) {
                    if ((even && node.val <= prev) || (!even && node.val >= prev)) {
                        return false;
                    }
                }
                if ((even && node.val % 2 == 0) || (!even && node.val % 2 == 1)) {
                    return false;
                }
                if (node.left != null) {
                    queue.add(node.left);
                }
                if (node.right != null) {
                    queue.add(node.right);
                }
                prev = node.val;
            }
            even = !even;
        }
        return true;
    }
}
