/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isEvenOddTree(TreeNode root) {
        // dfs or bfs
        Queue<TreeNode> queue = new LinkedList<>();
        queue.offer(root);
        int level = 0;
        
        while (!queue.isEmpty()) {
            int curSize = queue.size();
            int prev = -1;
            for (int i = 0; i < curSize; i++) {
                TreeNode cur = queue.poll();
                
                if ((level & 1) == 0) {
                    if ((cur.val & 1) == 0) {
                        return false;
                    }
                    if (prev != -1 && cur.val <= prev) {
                        return false;
                    }
                } else {
                    if ((cur.val & 1) == 1) {
                        return false;
                    }
                    if (prev != -1 && cur.val >= prev) {
                        return false;
                    }
                }
                prev = cur.val;
                if (cur.left != null) {
                    queue.offer(cur.left);
                }
                if (cur.right != null) {
                    queue.offer(cur.right);
                }
            }
            level++;
        }
        return true;
    }
}
