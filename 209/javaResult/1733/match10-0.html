class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        if(angle == 360){
            return points.size();
        }
        List<Double> pointAngles = new ArrayList();
        int given = 0;
        for(var point : points){
            if(point.get(0) == location.get(0) && point.get(1) == location.get(1)){
                given++;
            }
            else{
                pointAngles.add(getAngle(point, location));
            }
        }
        Collections.sort(pointAngles);
        int max = 0;
        // for(double num : pointAngles){
        //     System.out.println(num);
        // }
        
        for(int i = 0; i < pointAngles.size(); i++){
            double endAngle = (pointAngles.get(i) + angle) % 360;
            int endIndex = binarySearch(pointAngles, endAngle);
            int length = 0;
            if(i <= endIndex){
            // System.out.println("a1 ");
                length = endIndex - i;
                if(endIndex != i && pointAngles.get(endIndex) <= endAngle){
                // System.out.println("a2 ");
                    length++;
                }
            }
            else{
                length = pointAngles.size() - 1 - i;
                length += endIndex + 1;
            }
            double next = pointAngles.get((i + 1) % pointAngles.size());
            if(endIndex == i && pointAngles.get(i) <= next && next <= endAngle){
                return points.size();
            }
            // System.out.println("a " + pointAngles.get(i) + " " + endAngle + " " + length);
            max = Math.max(max, length);
        }
        return max + given;
    }
    public double getAngle(List<Integer> target, List<Integer> location) {
        double angle = Math.toDegrees(Math.atan2(target.get(1) - location.get(1), target.get(0) - location.get(0)));

        if(angle < 0){
            angle += 360;
        }
        return angle;
    }
    
    int binarySearch(List<Double> list, double target){
        int left = 0;
        int right = list.size() - 1;
        
        while(left < right){
            int mid = left + ((right - left) / 2);
            if(target < list.get(mid)){
                right = mid;
            }
            else{
                left = mid + 1;
            }
        }
        return right;
    }
}
