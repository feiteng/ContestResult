class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int n = points.size();
        int x0 = location.get(0);
        int y0 = location.get(1);
        
        int dup = 0;
        List<Double> angles = new ArrayList<>();
        
        for (List<Integer> point : points) {
            int x = point.get(0);
            int y = point.get(1);
            if (x == x0 && y == y0) {
                dup++;
            } else {
                angles.add(calcAngle(x0, y0, x, y));
            }
        }
        
        Collections.sort(angles);
        Deque<Double> deque = new LinkedList<>();
        int res = 0;
        for (double currAngle : angles) {
            while (!deque.isEmpty() && deque.peekFirst() < currAngle - angle) deque.pollFirst();
            deque.offerLast(currAngle);
            res = Math.max(res, deque.size());
        }
        
        for (double currAngle : angles) {
            currAngle += 360;
            while (!deque.isEmpty() && deque.peekFirst() < currAngle - angle) deque.pollFirst();
            deque.offerLast(currAngle);
            res = Math.max(res, deque.size());
        }
        
        return res + dup;
    }
    
    private double calcAngle(int x1, int y1, int x2, int y2) {
        return Math.toDegrees(Math.atan2(x2 - x1, y2 - y1));
    }
}
