class Solution {

    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        ArrayList<Double> list = new ArrayList<>(), list2 = new ArrayList<>();
        int left = 0, max = 0, count = 0;
        for (List<Integer> point : points) {
            if (point.equals(location)) {
                count++;
            } else 
{
                list.add(getAngle(point, location));
            }
        }
        Collections.sort(list);
        for (int right = 0; right < list.size(); right++) {
            while (list.get(right) - 
list.get(left) > angle) {
                left++;
            }
            max = Math.max(max, right - left + 1);
        }
        for (double i : list) {
            list2.add((i + 180) % 360);
        }

        Collections.sort(list2);
        left = 0;
        for (int right = 0; right < list2.size(); right++) {
            
while (list2.get(right) - list2.get(left) > angle) {
                left++;
            }
            max = Math.max(max, right - left + 1);
        }
        return max + count;
    }

    private double getAngle(List<Integer> point, List<Integer> location) {
        double dx = point.get(0) - location.get(0);
        double dy = -(point.get(1) - location.get(1));
        double inRads = Math.atan2(dy, dx);
        if (inRads < 0)
            inRads = Math.abs(inRads);
        else
            inRads = 2 * Math.PI - inRads;
        return Math.toDegrees(inRads);
    }
}
