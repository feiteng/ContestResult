class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        List<Double> list = new ArrayList<>();
        int same = 0;
        for (List<Integer> p : points) {
            double dx = (double)(p.get(0) - location.get(0));
            double dy = (double)(p.get(1) - location.get(1));
            if (dx == 0 && dy == 0) {
                same++;
            } else {
                double ang = Math.atan2(dy, dx);
                list.add(ang);
                list.add(ang + Math.PI * 2);
            }
        }
        
        double angd = Math.PI * 2 * (double) angle / 360.0;
        Collections.sort(list);
        
        int l = 0, r = 0;
        
        int res = 0;
        while (r < list.size()) {
            double end = list.get(l) + angd;
            while (r < list.size() && list.get(r) <= end) {
                r++;
            }
            res = Math.max(res, r - l + same);
            l++;
        }
        return res;
    }
    
}
