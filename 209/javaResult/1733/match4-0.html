class Solution {
    
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        
        List<Float> list=new ArrayList<>();
        int curr=0;
        
        for(List<Integer> point: points) {
            if(point.equals(location))
                curr+=1;
            else
                list.add(getAngle(point, location));
        }
        
        Collections.sort(list);
        // System.out.println(list);
        int n=list.size();
        list.addAll(list);
        // System.out.println(list);
        
        int max=0;
        for(int i=0, j=0; i<n; i++) {
            j=Math.max(j, i);
            while(j<i+n && Math.abs((list.get(j)-list.get(i)+360)%360)<=angle)
                j+=1;
            max=Math.max(max, j-i+curr);
        }
        
        return max;
    }
    
    public float getAngle(List<Integer> point, List<Integer> location) {
        float angle = (float) Math.toDegrees(Math.atan2(point.get(1) - location.get(1), point.get(0) - location.get(0)));

        if(angle < 0){
            angle += 360;
        }

        return angle;
    }
}
