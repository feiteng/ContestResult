class Solution {
    public int visiblePoints(List<List<Integer>> points, int a, List<Integer> location) {
        // List<Integer> test = new ArrayList<>();
        // test.add(0);
        // test.add(0);
        // System.out.println(angle(test, location)) ;
        List<Double> angles = new ArrayList<>();
        int same = 0;
        for (List<Integer> p: points) {
            if (p.get(0) == location.get(0) && p.get(1) == location.get(1)) {
                same++;
            } else {
                angles.add(angle(p, location));
                angles.add(angle(p, location) + 360);
            }
        }
        Collections.sort(angles);
        // System.out.println(angles);
        
        int start = 0;
        int ret = 0;
        for (int i = 0; i < angles.size(); i++) {
            while (i - start >= angles.size() / 2 || angles.get(i) - angles.get(start) > a) {
                start++;
            }
            ret = Math.max(ret, i - start + 1);
        }
        
        return ret + same;
    }
    
    private double angle(List<Integer> point, List<Integer> location) {
        long x1 = point.get(0) - location.get(0);
        long y1 = point.get(1) - location.get(1);
        long x2 = 1;
        long y2 = 0;
        double value = (x1 * x2 + y1 * y2) / (Math.sqrt(x1 * x1 + y1 * y1) * Math.sqrt(x2 * x2 + y2 * y2));
        double angle = Math.toDegrees(Math.acos(value));
        if (y1 < 0) angle = 360 - angle;
        return angle;
    }
}
