class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        List<Double> list = new ArrayList<>();
        int same = 0;
        for (List<Integer> point : points) {
            if (point.get(0) == location.get(0) && point.get(1) == location.get(1)) same++;
            else list.add(getAngle(location.get(0), location.get(1), point.get(0), point.get(1)));
        }
        Collections.sort(list);
        
        int res = 0;
        int left = 0;
        List<Double> dList = new ArrayList<>();
        for (int i = 0; i < list.size(); i++) {
            dList.add(list.get(i));
        }
        for (int i = 0; i < list.size(); i++) {
            dList.add(list.get(i) + 360);
        }
        for (int i = 0; i < dList.size(); i++) {
            while (Math.abs(dList.get(i) - dList.get(left)) > (double) angle) {
                left++;
            }
            res = Math.max(res, i - left + 1);
        }
        return res + same;
    }
    
    public double getAngle(int x1, int y1, int x2, int y2) {
        double angle = (double) Math.toDegrees(Math.atan2(y2 - y1, x2 - x1));
        if(angle < 0){
            angle += 360;
        }
        return angle;
    }
}
