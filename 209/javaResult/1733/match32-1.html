import java.awt.Point;

class Solution {
  private int orientation(Point p, Point q, Point r) {
    int val = (q.y - p.y) * (r.x - q.x) - (q.x - p.x) * (r.y - q.y);
    if (val == 0)
      return 0;
    return (val > 0) ? 1 : 2;
  }
  boolean onSegment(Point p, Point q, Point r) {
    if (q.x <= Math.max(p.x, r.x) && q.x >= Math.min(p.x, r.x) && q.y <= Math.max(p.y, r.y)
        && q.y >= Math.min(p.y, r.y))
      return true;
    return false;
  }
  
public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
    int cx = location.get(0), cy = location.get(1);
    int res = 0;
    List<Double> ptgles = new ArrayList<Double>();
    for (List<Integer> p : points) {
      if (cx == p.get(0) && cy == p.get(1)) {
        res++;
        continue;
      }
      final double tmp = Math.toDegrees(Math.atan2(p.get(1) - cy, p.get(0) - cx));
      ptgles.add(tmp < 0 ? tmp + 360d : tmp);
    
}
    Collections.sort(ptgles);
    int max = 0;
    int l = 0;
    for (int r = 0; r < ptgles.size(); r++) {
      while (ptgles.get(r) - ptgles.get(l) > angle)
        l++;
      max = Math.max(max, r + 1 - l);
    }
    for (int r = 0; r != ptgles.size(); r++) {
      while (l != ptgles.size() && ptgles.get(r) + 360 - ptgles.get(l) > angle) {
        l++;
      }
      if (l == ptgles.size())
        break;
      max = Math.max(max, ptgles.size() + 1 + r - l);
    }
    res += max;
    return res;
  }
}
