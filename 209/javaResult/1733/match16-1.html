class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int N = points.size();
        
        
List<Double> angles = new ArrayList<>(N);
        
        int current = 0;
        for (List<Integer> point: points) {
            if (point.get(0) == location.get(0) && point.get(1) == location.get(1)) {
                current++;
                
continue;
            }
            
            angles.add(toAngle(point, location));
            
            // System.out.println(location + "->" + toAngle(point, location) + " ->" + point);
        }
        
        Collections.sort(angles);
        int maxSize = 0;
        
        Queue<Double> queue = new LinkedList<>();
        for (double currentAngle: angles) {
            while (!queue.isEmpty() && queue.peek() < currentAngle-angle) {
                queue.poll();
            }
            
            queue.add(currentAngle);
            maxSize = Math.max(maxSize, queue.size());
        }
        
        for (double currentAngle: angles) {
            while (!queue.isEmpty() && queue.peek() < currentAngle+360-angle) {
                queue.poll();
            }
            
            queue.add(currentAngle+360);
            maxSize = Math.max(maxSize, queue.size());
        }
        return Math.min(angles.size(), maxSize) + current;
    
}
    
    public double toAngle(List<Integer> point, List<Integer> location) {
        double angle = Math.toDegrees(Math.atan2(location.get(1) - point.get(1), location.get(0) - point.get(0)));
        if (angle < 0) {
            angle += 360;
        }
        return angle;
    }
}
