class Solution {
public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        if (angle == 360) return points.size();
        List<int[]> p = new ArrayList<>();
        List<Double> ang = new ArrayList<>();
        int x = location.get(0);
        int y = location.get(1);

        int count = 0;
        for (int i = 0; i < points.size(); i++) 
{
            int tx = points.get(i).get(0);
            int ty = points.get(i).get(1);
            if (tx == x && ty == y) {
                count++;
            } else {
                p.add(new int[] {tx, ty});
                ang.add(Math.atan2(ty - y, tx - x));
            }
        }
        
        int size = ang.size();
        for (int i = 0; i < size; i++) {
            ang.add(ang.get(i) + 2 * Math.PI);
        }


        double wide = angle * 2 * Math.PI / 360.0;
        ang.sort((xx, yy) -> Double.compare(xx, yy));
        
        
int ans = 0;
        for (int i = 0, j = 0; i < ang.size(); i++) {
            while (j < ang.size() && ang.get(i) + wide >= ang.get(j)) j++
;
            ans = Math.max(ans, j - i);
        }
        return ans + count;

    }
}
