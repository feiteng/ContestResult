class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int ans = 0;
        List<Double> angles = new ArrayList<>();
        for(List<Integer> point : points) {
            if(point.get(0) == location.get(0) && point.get(1) == location.get(1)) {
                ans++;
            }else{
                int dx = point.get(0) - location.get(0), dy = point.get(1) - location.get(1);
                if(dx == 0) {
                    double arc = dy > 0 ? 1.5 * Math.PI : 0.5 * Math.PI;
                    angles.add(arc*180/Math.PI);
                }else{
                    double tan = (double)dy / dx;
                    double arc = Math.atan(tan) + (dx > 0 ?  Math.PI : 0);
                    angles.add(arc*180/Math.PI);
                }
            }
        }
        Collections.sort(angles);
        int max = 0;
        int N = angles.size();
        for(int i = 0, j = 0; i < N; ++i) {
            double cur = angles.get(j % N) - angles.get(i) + (angles.get(j % N) - angles.get(i) >= 0 ? 0 : 360);
            while(j - i < N && cur <= angle) {
                j++;
                cur = angles.get(j % N) - angles.get(i) + (angles.get(j % N) - angles.get(i) >= 0 ? 0 : 360);
            
}
            max = Math.max(max, j - i);
        }
        return ans + max;
    }
}
