class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int res = 0;
        List<Double> angles = new ArrayList<>();
        for (List<Integer> p : points) {
            if (p.get(0).equals(location.get(0)) && p.get(1).equals(location.get(1))) {
                res++;
            } else {
                angles.add(Math.atan2(p.get(1) - location.get(1), p.get(0) - location.get(0)) / Math.PI * 180);
            }
        }

        Collections.sort(angles);
        int size = angles.size();

        for (int i = 0; i < size; i++) {
            angles.add(angles.get(i) + 360);
        }

        int max = 0;
        int fp = 0;
        int bp = 0;
        while (fp < size) {
            while (angles.get(bp + 1) - angles.get(fp) <= angle) bp++;
            max = Math.max(max, bp - fp + 1);
            fp++;
            if (bp == fp) bp ++;
        }

        return res + max;
    }
}
