class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int n = points.size();
        List<Double> angles = new ArrayList<>();
        int samePoint = 0;
        for(int i = 0; i < n; i++) {
            List<Integer> point = points.get(i);
            if(point.get(0) == location.get(0) && point.get(1) == location.get(1)) {
                samePoint++;
                continue;
            }
            double _angle = calcAngle(point, location);
            angles.add(_angle);
            angles.add(_angle + 360);
        }
        Collections.sort(angles);
        int i = 0, curIdx = 0;
        Queue<Double> q = new LinkedList<>();
        while(i < angles.size() && angles.get(i) - angles.get(curIdx) <= angle) {
            q.offer(angles.get(i));
            i++;
        }
        int res = i;
        while(i < angles.size()) {
            curIdx++;
            while(!q.isEmpty() && !q.peek().equals(angles.get(curIdx))) q.poll();
            while(i < angles.size() && angles.get(i) - angles.get(curIdx) <= angle) {
                q.offer(angles.get(i));
                i++;
            }
            res = Math.max(res, q.size());
        }
        return res + samePoint;
    }

    private double calcAngle(List<Integer> p1, List<Integer> p2) {
        double angle = Math.toDegrees(Math.atan2(p1.get(1) - p2.get(1), p1.get(0) - p2.get(0)));
        if(angle < 0) angle += 360;
        return angle;
    }
}
