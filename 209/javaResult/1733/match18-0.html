class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        
        List<Double> degrees = new ArrayList<>();
        boolean self = false;
        for (List<Integer> point: points) {
            if (point.get(0) == location.get(0) && point.get(1) == location.get(1)) {
                self = true;
                continue;
            }
            degrees.add(getDegree(point, location));
        }
        Collections.sort(degrees);
        // System.out.println(degrees.size());
        // for (int i = 0; i < degrees.size(); i++) {
        //     System.out.println(degrees.get(i));
        // }
        
        int max = 0;
//         int left = 0;
//         int right = 0;
//         while (left < degrees.size()) {
//             // System.out.println(degrees.get(left)+","+degrees.get(right));
//             while (left < degrees.size() && degrees.get(right) <= degrees.get(left) + (double)angle) {
//                 right++;
//                 if (right == degrees.size()) {
//                     right = 0;
//                 }
//             }
            
//             max = Math.max(max, right - left);
//             left++;
//         }
        
        int right = 0;
        int left = 0;
        for (int i = 0; i < degrees.size(); i++) {
            double high = degrees.get(i) + angle;
            while (right < degrees.size() && degrees.get(right) <= high) {
                right++;
            }
            while (left <= i && high >= 360 && degrees.get(left) <= high - 360) {
                left++;
            }
            
            // System.out.println(i+","+cnt);
            max = Math.max(max, right - i + left);
        }
        
        if (self) {
            max++;
        }
        return max;
    }
    
    private double getDegree(List<Integer> point, List<Integer> loc) {
        double degree = Math.toDegrees(Math.atan2(point.get(0) - loc.get(0), point.get(1) - loc.get(1)));
        if (degree < 0) {
            degree += 360;
        }
        // System.out.println(degree);
        return degree;
    }
}
