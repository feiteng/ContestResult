class Solution {
   public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        List<Double> l = new ArrayList<>();
       boolean origin = false;
        for (List<Integer> p : points) {
            if (p.get(0) == location.get(0) && p.get(1) == location.get(1)) {
                origin = true;
                continue;
            }
            l.add(calcAngle(p, location));
            if (calcAngle(p,location) > 150) System.out.println(p);
        }
        Collections.sort(l);
        for (int i = 0; i < points.size(); i++) {
            l.add(l.get(i) + 360);
        }
        int last = 0; int max = 0;
        for (int i = 1; i < l.size(); i++) {
            while (last < i && l.get(i) - l.get(last) > angle) {
                last++;
            }
            max = Math.max(max, i-last+1);
        }
       if (origin) max += 1;
        return max;
    }

    public double calcAngle(List<Integer> point, List<Integer> loc) {
        double deg = Math.toDegrees(Math.atan((double) (point.get(1) - loc.get(1)) / (point.get(0) - loc.get(0))));
        if (point.get(1) - loc.get(1) <= 0 && point.get(0) - loc.get(0) < 0) {
            return -180 + deg;
        }
        if (point.get(1) - loc.get(1) > 0 && point.get(0) - loc.get(0) < 0) {
            return 180+deg;
        }
        if (point.get(0) - loc.get(0) == 0) {
            if (point.get(1) - loc.get(1) > 0) {
                return 90;
            } else if (point.get(1) - loc.get(1) < 0){
                return -90;
            } else {
                return 0;
            }
        }
        return deg;
    }
}
