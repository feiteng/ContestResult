class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int x = location.get(0);
        
int y = location.get(1);
        List<Double> angles = new ArrayList<>();
        int count = 0;
        for (List<Integer> list : points) {
            int x0 = list.get(0) - x;
            int y0 = list.get(1) - y;
            if (x0 == 0 && y0 == 0) {
                count++;
            } else {
                // System.out.println(Math.toDegrees(Math.atan2(y0,x0)));
                angles.add(Math.toDegrees(Math.atan2(y0,x0)));
            }   
        }
        if (angles.size() == 0) {
            return count;
        }
        if (angle == 360) {
            return points.size();
        }
        int size = angles.size();
        for (int i = 0; i < size; i++) {
            double d = angles.get(i);
            if (d <= angle) {
                
angles.add(d + 360d);
            }
        }
        int start = 0;
        int end = 0;
        int max = 1;
        Collections.sort(angles);
        //  size = angles.size();
        // for (int i = 0; i < size; i++) {
        //     double d = angles.get(i);
        //     System.out.println("dd:" + d);
        // }
        
while (start < angles.size()) {
            while (end < angles.size() && angles.get(end) - angles.get(start) <= angle) {
                end++;
            }
            max = Math.max(max, end - start);
            // System.out.println("s:" + start +":" + end);
            start++;
        }
        
        // System.out.println("m:" + max +":c" + count);
        return count + max;
    }
}
