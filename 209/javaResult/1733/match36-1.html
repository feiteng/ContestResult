class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int fixed = 0;
        List<Double> angles = new ArrayList<>();
        
for (List<Integer> L : points) {
            int x = L.get(0) - location.get(0);
            int y = L.get(1) - location.get(1);
            
            if (x == 0 && y == 0) {
                fixed++;
                continue;
            }
            
            double theta = Math.toDegrees(Math.atan2(x * 0 - y * 1, x * 1 + y * 0));
            if (theta < 0.0) theta += 360.0;
            
            angles.add(theta);
            angles.add(theta + 360.0);
        }
        
        Collections.sort(angles);
        LinkedList<Double> L = new LinkedList<>();
        
        int max = 0;
        for (double x : angles) {
            while (!L.isEmpty() && L.peekFirst() + angle < x) L.pollFirst();
            L.offerLast(x);
            
max = Math.max(max, L.size());
        }
        
        return fixed + max;
    }
}
