class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        double angleAsDouble = angle * Math.PI / 180;
        
var angles = new ArrayList<Double>();
        int always = 0;
        for (List<Integer> point : points) {
            int dx = point.get(0) - location.get(0);
            int dy = point.get(1) - location.get(1);
            if (dx == 0 && dy == 0)
                always++;
            else 
{
                double a = Math.atan2(dx, dy);
                angles.add(a);
                angles.add(a + 2 * Math.PI);
            }
        }
        Collections.sort(angles);
        
        int n = angles.size();
        int maxVisible = 0;
        int to = 0;
        for (int from=0; from<n; from++) {
            while (to < n && angles.get(from) + angleAsDouble >= angles.get(to))
                to++;
            if (to - from > maxVisible)
                maxVisible = to - from;
        }
        return maxVisible + always;
    }
}
