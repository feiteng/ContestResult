class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> l) {
        
        int zc=0;
        ArrayList<Double> list=new ArrayList<>();
        for(List<Integer> i:points)
        {
            if(
i.get(0)==l.get(0)&&i.get(1)==l.get(1))
            {
                zc++;
            }
            else
            {
                list.add(getAngle(i.get(0),i.get(1),l.get(0),l.get(1)));
            }
        }
        
        Collections.sort(list);
        
        int size=list.size();
        for(int i=0;i<=size;i++)
        {
            list.add(list.get(i)+360);
        }
        
        int max=0;
        Queue<Double> q=new LinkedList<>();
        for(double d:list)
        {
            
            while(!q.isEmpty()&&q.peek()<d-angle)
            {
                q.poll();
            }
            
            q.add(d);
            max=Math.max(q.size(),max);
        }
        
        return max+zc;
        
    }
    
    
    public double getAngle(int x,int y,int centerx,int centery) {
    double angle = (double) Math.toDegrees(Math.atan2(y - centery,x - centerx));

    if(angle < 0){
        angle += 360;
    }

    return angle;
    }
    
}
