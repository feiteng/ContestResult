class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        // calculate angle formed by each point
        // sort by angle
        // group points in sliding window fashion
        int px = location.get(0), py = location.get(1);
        List<Double> list = new ArrayList<>();
        int special = 0;
        for(List<Integer> p : points)
        {
            int x = p.get(0), y = p.get(1);
            if(x == px && y == py)
            {
                special++;
                continue;
            }
            double ang = getAngle(x, y, px, py);
            // System.out.println(ang);
            list.add(ang);
        }
        Collections.sort(list);
        int size = list.size();
        // if(list.get(0) + angle >= list.get(size - 1)) return size;
        int j = 0;
        int ans = 0;
        for(int i = 0; i < size; i++)
        {
            j = findNext(list, i, j, angle);
            // System.out.printf("%d %d\n", i, j);
            if(j < 0) return size + special
;
            ans = Math.max(ans, (j - i + size) % size);
        }
        return ans + special;
        
    }
    
    int findNext(List<Double> list, int cur, int j, double ang)
    {
        int size = list.size();
        int count = 0;
        while((list.get(j) - list.get(cur) + 360) % 360 <= ang && count < size) 
        {
            j = (j + 1) % size;
            count++;
        }
        if(count == size) return -1
;
        return j;
    }
    
    double getAngle(int x, int y, int px, int py)
    {
        int dx = x - px, dy = y - py;
        double ang = Math.toDegrees(Math.atan2(dy, dx));
        if(ang < 0) ang += 360.0;
        return ang;
    }
}
