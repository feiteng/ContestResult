class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        if (angle >= 360) {
            return points.size();
        }
        int n = points.size();
        List<Double> theta = new ArrayList<>();
        int atLocation = 0;
        for (int i = 0; i < n; i++) {
            List<Integer> point = points.get(i);
            int x = point.get(0) - location.get(0);
            int y = point.get(1) - location.get(1);
            if (x == 0 && y == 0) {
                atLocation ++;
            } else {
                theta.add(Math.atan2((double) x, (double) y));
            }
        }
        
        double radian = Math.PI * (double) angle / 180d;
        
        int fast = 0;
        int slow = 0;
        
        Collections.sort(theta);
        for (Double d : theta) {
            // System.out.print(d + " ");
        }
        // System.out.println(theta.size());
        int tn = theta.size();
        for (int i = 0; i < tn; i++) {
            theta.add(Math.PI * 2 + theta.get(i));
        }
        int max = 0;
        while (fast < theta.size()) {
            while (fast < theta.size() && theta.get(fast) <= theta.get(slow) + radian) {
                fast++;
            }
            
            max = Math.max(max, fast - slow);
            slow++;
        }
        return max + atLocation;   
    }
}
