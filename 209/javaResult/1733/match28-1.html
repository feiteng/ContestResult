class Solution {
    public int visiblePoints(List<List<Integer>> points, int a, List<Integer> location) {
        int n = points.size();
        int res = 0, max = 0;
        ArrayList <Double> angles = new ArrayList();
        double X = location.get(0), Y = location.get(1);
        
for(List <Integer> point : points) {
            double x = point.get(0), y = point.get(1);
            if(x == X && y == Y){
                res++;
                continue;
            }
            double angle = Math.toDegrees(Math.atan2(x - X, y - Y));
            angles.add(angle);
            angles.add(angle + 360);
        }
        Collections.sort(angles);
        //System.out.println(angles);
        ArrayDeque <Double> dq = new ArrayDeque();
        for(double angle : angles){
            while(!dq.isEmpty() && dq.peek() < angle - a){
                dq.poll();
            }
            dq.offer(angle);
            max = Math.max(max, dq.size());
        }
        //System.out.println(res + " " + max);
        return res + max;
    }
}
