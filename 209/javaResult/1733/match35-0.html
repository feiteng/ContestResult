class Solution {
     double pi = Math.PI;
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
       
        List<Double> degrees = new ArrayList<>();
        int count = 0;
        for(List<Integer> point : points) {
            if(point.get(0) == location.get(0) && point.get(1) == location.get(1)) {
                count++;
            }
            else {
                double degree = getDegree(point, location);
                degrees.add(degree);
                degrees.add(degree+2*pi);
                //System.out.println(degree);
            }
        }
        
        double a = (pi * angle) / 180;
       // System.out.println("a=" + a );
        //System.out.println(degrees.toString());
        Collections.sort(degrees);
        //System.out.println(degrees.toString());
        
        int size = 0;
        double max = degrees.get(0);
        double min = degrees.get(0);
        int i = 0;
        int j = 0;
        int n = degrees.size();
        while(i < n) {
            max = degrees.get(i);
            while(j < i && max - min - a > 0.00000001) {
                j++;
                min = degrees.get(j);
            }
            
            if(i - j + 1 > size){
                size = i - j +1;
            }
            i++;
        }
    
        return count + size;
    }
    
    private double getDegree(List<Integer> p, List<Integer> l) {
        int x= p.get(0) - l.get(0);
        int y = p.get(1) - l.get(1);
        
        if(x == 0) {
            if(y >0){
                return pi/2;
            }
            else{
                return -1.0*pi/2 + 2* pi;
            }
        }
       // System.out.format("x = %d, y = %d \n", x, y);
        double d = Math.atan((double)y/(double)x);
        
        if( x < 0){
            d += pi;
        }
        if(d < 0) {
            d += 2 * pi;
        }
        
        return d;
    }
}
