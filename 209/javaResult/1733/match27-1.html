class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        
        int x = location.get(0), y = location.get(1);
        
        int extra = 0;
        List<Double> list = new ArrayList<>();
        for (List<Integer> point : points) {
            int x2 = point.get(0), y2 = point.get(1);
            if (x2 == x && y2 == y) ++extra;
            else list.add(getAngle(x, y, x2, y2));
        
}
        
        Collections.sort(list);
        
        int res = 0;
        int j = 0;
        for (int i = 0; i < list.size() * 2; ++i) {
            
            double degree = list.get(i % list.size());
            
            while ((degree - list.get(j % list.size()) + ((i % list.size()) < (j % list.size()) ? 360 : 0)) > angle) {
                ++j;
            }
            
            
            
            // while (degree - list.get(j % list.size()) + ((i % list.size()) < (j % list.size()) ? 360 : 0) > angle) {
            //     ++j;
            // }
            
            if ((i % list.size()) < (j % list.size())) {
                res = Math.max(res, (i % list.size()) + 1 + (list.size() - (j % list.size())));
            } else {
                res = Math.max(res, ((i % list.size()) - (j % list.size()) + 1));
            }
        }
        
        return res + extra;
    }
    
    private double getAngle(int x1, int y1, int x2, int y2) {
        
        double angle = Math.toDegrees(Math.atan2(y2 - y1, x2 - x1));
        if (angle < 0) angle += 360;
        return angle;
    }
}
