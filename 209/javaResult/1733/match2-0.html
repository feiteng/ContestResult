class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int px = location.get(0), py = location.get(1);
        List<Double> list = new ArrayList<>();
        int sp = 0;
        for(List<Integer> p : points)
        {
            int x = p.get(0), y = p.get(1);
            if(x == px && y == py)
            {
                sp++;
                continue;
            }
            double calculatedAngle = getAngle(x, y, px, py);
            list.add(calculatedAngle);
        }
        int size = list.size();
        int j = 0;
        int ans = 0;
        
        Collections.sort(list);
        
        for(int i = 0; i < size; i++)
        {
            j = findNextPos(list, i, j, angle);
            if(j < 0)
            {
                return size + sp;
            }
            ans = Math.max(ans, (j - i + size) % size);
        }
        return ans + sp;
        
    }
    
    int findNextPos(List<Double> list, int cur, int j, double ang)
    {
        int size = list.size();
        int count = 0;
        while((list.get(j) - list.get(cur) + 360) % 360 <= ang && count < size) 
        {
            j = (j + 1) % size;
            count++;
        }
        if(count == size) 
        {
            return -1;
        }
        return j;
    }
    
    double getAngle(int x, int y, int px, int py)
    {
        int dx = x - px, dy = y - py;
        double ang = Math.toDegrees(Math.atan2(dy, dx));
        if(ang < 0) 
        {
            ang += 360.0;
        }
        return ang;
    }
}
