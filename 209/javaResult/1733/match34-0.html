class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        List<Double> angles = new ArrayList();
        int zeroPoint = 0;
        for (List<Integer> p : points) {
            int x = p.get(0) - location.get(0);
            int y = p.get(1) - location.get(1);
            if (x == 0 && y == 0) {
                zeroPoint++;
            } else {
                angles.add(getAngle(x, y));
            }
        }
        Collections.sort(angles);
        int max = zeroPoint;
        int le = 0, ri = le + 1;
        while (le < angles.size()) {
            ri = Math.max(ri, le + 1);
            while (ri % angles.size() != le) {
                if (ri < angles.size()) {
                    if ((angles.get(le) + angle * 1.0) < angles.get(ri)) {
                        break;
                    }
                } else {
                    if ((angles.get(le) + angle * 1.0) < (angles.get(ri - angles.size()) + 360.0)) {
                        break;
                    }
                }
                ri++;
            }
            max = Math.max(max, zeroPoint + ri - le);
            le++;
        }
        return max;
    }
    
    private double getAngle(int x, int y) {
        if (x == 0) {
            if (y > 0) {
                return 90.0;
            } else {
                return 270.0;
            }
        }
        double degree = Math.toDegrees(Math.atan(1.0 * Math.abs(y) / Math.abs(x)));
        if (x < 0) {
            if (y > 0) {
                degree = 180.0 - degree;            
            } else {
                degree = 180.0 + degree;
            }
        } else if (y < 0) {
            degree = 360.0 - degree;
        }
        return degree;
    }
}
