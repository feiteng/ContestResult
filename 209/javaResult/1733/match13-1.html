class Solution {
    public int visiblePoints(List<List<Integer>> _points, int angle, List<Integer> location) {
        int fix = 0;
        List<Double> temp = new ArrayList<>();
        for (List<Integer> p : _points) {
            if (p.get(1) == location.get(1) && p.get(0) == location.get(0)) {
                fix++;
            } else {
                temp.add(
Math.toDegrees(Math.atan2(p.get(1) - location.get(1), p.get(0) - location.get(0))));
            }
        }
        Collections.sort(temp);
        double[] angles = new double[temp.size() * 2];
        for (int i = 0; i < temp.size(); i++) {
            angles[i] = temp.get(i);
            angles[i + temp.size()] = angles[i] + 360;
        }
        int left = 0, right = 0, ans = 0;
        for (left = 0; left < angles.length; left++) {
            while (right + 1 < angles.length && angles[right + 1] <= angles[left] + angle) {
                right++;
            }
            ans = Math.max(ans, right - left + 1 + fix);
        }
        return ans;
    }
}
