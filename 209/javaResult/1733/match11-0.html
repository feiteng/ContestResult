class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        double alpha = 0.0000001;
        angle = Math.min(angle, 360);
        int add = 0;
        int max = 0;
        int n = points.size();
        List<Double> angles = new ArrayList<>();
        double curAngle;
        for (List<Integer> point : points) {
            if (point.get(0) == location.get(0) && point.get(1) == location.get(1)) {
                add++;
                n--;
            } else {
                curAngle = getAngle(location, point);
                angles.add(curAngle);
                if (curAngle < angle) angles.add(curAngle + 360);
            }
        }
        Collections.sort(angles);        
        int j = 0;
        for (int i = 0; i < n; i++) {
            double curMax = angles.get(i) + angle;
            while (j < angles.size() && (angles.get(j) + alpha < curMax || angles.get(j) - alpha < curMax)) j++;
            max = Math.max(max, j-i);
        }
        return max + add;
    }
    private double getAngle(List<Integer> origin, List<Integer> to) {
        double ans = 180 * Math.atan2(to.get(0) - origin.get(0), to.get(1) - origin.get(1)) / Math.PI;
        if (ans < 0) {
            ans += 360;
        }
        return ans;
    }
}
