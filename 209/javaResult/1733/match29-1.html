class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int n = points.size();
        List<Double> ps = new ArrayList<Double>();
        int extra = 0;
        for(int i = 0; i < n; i++){
            // System.out.println(points.get(i));
            if(points.get(i).get(0) == location.get(0) && points.get(i).get(1) == location.get(1)){
                extra ++;
                continue;
            }
            
ps.add(getAngle(points.get(i), location));
        }
        Collections.sort(ps);
        
        int pre = 0;
        int end = 0;
        int res = 0;
        
        // System.out.println(ps);
        int len = ps.size();
        while(pre < len){
            // System.out.println("end " + end);
            while(end < len || (end - len < pre)){
                if((end / len) * 360 + ps.get(end % len) - ps.get(pre) <= angle){
                    res = Math.max(res, end - pre + 1); 
                    end ++;
                }else{
                    break;
                }
            }
            // System.out.println("pre end " + pre + " " + end);
            while(pre <= end && pre < len){
                if((end / len) * 360 + ps.get(end % len) - ps.get(pre) > angle){
                    pre ++;
                }else{
                    break;
                }
            }
        
}
        return res + extra;
    }
    
    public double getAngle(List<Integer> point, List<Integer> location){
        double temp = Math.atan2(point.get(1) - location.get(1), point.get(0) - location.get(0));
        double ans = Math.toDegrees(temp);
        // System.out.println(temp + " " + ans);
        return ans;
    }
}
