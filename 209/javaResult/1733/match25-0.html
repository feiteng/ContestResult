class Solution {
    public int visiblePoints(List<List<Integer>> points, int degree, List<Integer> location) {
        int base = 0;
        List<List<Integer>> aPoints = new ArrayList<>();
        for (int i = 0; i < points.size(); ++i) {
            if (isPointsEqual(points.get(i), location)) {
                base += 1;
            } else {
                aPoints.add(points.get(i));
            }
        }
        
        double radian = Math.toRadians(degree);
        List<Double> angleList = new ArrayList<>();
        for (int i = 0; i < aPoints.size(); ++i) {
            List<Integer> point = aPoints.get(i);
            double angle = Math.atan2(point.get(1) - location.get(1), point.get(0) - location.get(0));
            angleList.add(angle);
            angleList.add(angle + Math.PI*2);
        }
        Collections.sort(angleList);
        
        int count = 0;
        int j = 0;
        for (int i = 0; i < angleList.size(); ++i) {
            double start = angleList.get(i);
            if (start > Math.PI*2) break;
            for (j = Math.max(i + 1, j); j < angleList.size() && angleList.get(j) - start <= radian; ++j);
            count = Math.max(j - i, count);
        }
        
        return count + base;
    }
    
    boolean isPointsEqual(List<Integer> p1, List<Integer> p2) {
        return p1.get(0) == p2.get(0) && p1.get(1) == p2.get(1);
    }
}
