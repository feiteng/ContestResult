class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        int len = points.size();
        int base = 0;
        List<List<Integer>> ps = new ArrayList<>();
        for (int i = 0; i < len; i++) {
            if (points.get(i).get(0) == location.get(0) && 
points.get(i).get(1) == location.get(1)) {
                base++;
            } else {
                ps.add(points.get(i));
            }
        }
        

        
        
        len -= base;
        // System.out.println(base);
        
        double[] angles = new double[len * 2];
        for (int i = 0; i < ps.size(); i++) {
            angles[i] = getAngle(ps.get(i).get(0) - location.get(0), ps.get(i).get(1)-location.get(1));
            angles[i + len] = 
getAngle(ps.get(i).get(0) - location.get(0), ps.get(i).get(1)-location.get(1)) + 360;
        }
        Arrays.sort(angles);
        int l = 0;
        int max = 0;
        for (int i = 0; i < len * 2 - 1; i++) {
            // System.out.println(angles[i]);
            while ( i - len == l || angles[i] - angles[l] > angle) {
                l++;
            }
            
max = Math.max(max, i - l + 1);
        }
        return max + base;
        
    }
    
    public double getAngle(int x, int y) {
        
    double angle = Math.toDegrees(Math.atan2(x, y));

    if(angle < 0){
        angle += 360;
    }
// System.out.println("x = " + x);
//         System.out.println("y = " + y);
//         System.out.println("angle = " + angle);
    return angle;
    }
}
