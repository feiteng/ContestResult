class Solution {
    public int visiblePoints(List<List<Integer>> points, int angle, List<Integer> location) {
        List<Double> pointAngles = new ArrayList<>();
        int posX = location.get(0);
        int posY = location.get(1);
        int numEqualPoints = 0;
        for (List<Integer> point: points) {
            int x = point.get(0);
            int y = point.get(1);
            if (posX == x && posY == y) numEqualPoints++;
            else {
                pointAngles.add(getAngle(posX, posY, x, y));
            }
        }
        Collections.sort(pointAngles);
        //System.out.println(pointAngles);
        //System.out.println(pointAngles.size());
        int maxVisiblePoints = 0;
        int windowStart = 0;
        int numPoints = 0;
        for (int windowEnd = 0; windowEnd < pointAngles.size(); windowEnd++) {
            double newPointAngle = pointAngles.get(windowEnd);
            numPoints++;
            while (windowStart < windowEnd && newPointAngle - pointAngles.get(windowStart) > (double) angle) {
                windowStart++;
                numPoints--;
            }
            //System.out.println(numPoints);
            maxVisiblePoints = Math.max(maxVisiblePoints, numPoints);
        }
        for (int windowEnd = 0; windowEnd < pointAngles.size() && windowStart < pointAngles.size(); windowEnd++) {
            double newPointAngle = pointAngles.get(windowEnd);
            numPoints++;
            while (windowStart < pointAngles.size() && newPointAngle + 360 - pointAngles.get(windowStart) > (double) angle) {
                windowStart++;
                numPoints--;
            }
            //System.out.println(numPoints);
            maxVisiblePoints = Math.max(maxVisiblePoints, numPoints);
        }
        return maxVisiblePoints + numEqualPoints;
    }
    
    public double getAngle(int posX, int posY, int x, int y) {
        double angle = Math.toDegrees(Math.atan2(y - posY, x - posX));

        if (angle < 0) {
            angle += 360;
        }

        return angle;
    }
}
