class Solution {
    public int specialArray(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int magic = n;
        for(int i=0; i<n; i++) {
            
if(nums[i] >= magic && (i==0 || nums[i-1] < magic)) return magic;
            magic--;
        }
        return -1;
    }
}
