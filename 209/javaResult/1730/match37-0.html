class Solution {
   
    public int specialArray(int[] nums) {
        Arrays.sort(nums);
        int max = nums[nums.length - 1];
        int res = -1;
        for (int i = 1; i <= max; i++) {
            int count = 0;
            for (int j = nums.length - 1; j >= 0; j--) {
                if (nums[j] >= i) count++;
                else break;
            }
            if (count == i) return i;
        }
        return -1;
    }
}
