class Solution {
    public int specialArray(int[] nums) {
        
        for(int res = 0; res <=1000; res ++) {
            int count = 0;
            for(int num: nums) {
                if(num >= res )
                    count++;
            }
            
            if(count == res)
                return res;
        }
        return -1;
    }
}
