	class Solution {
	    public int specialArray(int[] a) {
	        int n = a.length;
	        for(int i = 0;i <= n;i++) {
	        	int nu = 0;
	        	for(int u : a) {
	        		if(u >= i) {
	        			nu++;
	        		}
	        	}
	        	if(i == nu)return i;
	        }
	        return -1;
	    }
	}
