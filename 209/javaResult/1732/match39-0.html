class Solution {
    long[] dp;
    public int minimumOneBitOperations(int n) {
        if (n == 0) {
            return 0;
        }
        dp = new long[32];
        dp[0] = 1;
        dp[1] = 3;
        long add = 2;
        for (int i = 2; i < 32; i++) {
            add <<= 1;
            dp[i] = dp[i - 1] + add;
        }

        return minimumOneBitOperationsInner(n);
    }

    private int minimumOneBitOperationsInner(int n) {
        if (n == 0) {
            return 0;
        }

        if (n == 1) {
            return 1;
        }

        int x = getHighestOne(n) - 1;

        return (int)(dp[x] - minimumOneBitOperationsInner(n & (~(1 << x))));
    }

    private int getHighestOne(int n) {
        int count = 31;
        int tag = 1 << 30;
        while (n < tag) {
            count--;
            tag >>= 1;
        }
        return count;
    }
}
