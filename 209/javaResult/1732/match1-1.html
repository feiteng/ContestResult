import java.math.BigInteger;
class Solution {
    public int minimumOneBitOperations(int n) {
        int bit = n < 0 ? 1 : 0;
        BigInteger result = BigInteger.valueOf(bit);
        for (int i = 0; i < 31; i++) {
            n = n << 1;
            int sign = n < 0 ? 1 : 0;
            bit = (bit + sign) % 2;
            result = result.shiftLeft(1).add(BigInteger.valueOf(bit));
        }
        return result.intValue();
    }
}
