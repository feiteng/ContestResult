class Solution {
    public int minimumOneBitOperations(int n) {
        int count = 2;
        if(n == 1) return 1;
        while(count < n){
            count *= 2;
        }
        if(count == n) return 2 * n - 1;
        count /= 2;
        int res = 0;
        boolean p = true;
        while(n > 0){
            while(n < count)count/=2;
            if(p){
                n -= count;
                res += (2 * count - 1);
                p = !p;
            } 
            else if(!p){
                n -= count;
                res -= (2 * count - 1);
                p = !p;
            
}
        }
        return res;
    }
}
