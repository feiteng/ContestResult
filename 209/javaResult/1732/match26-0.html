class Solution {
    public int minimumOneBitOperations(int n) {
        String str = Integer.toString(n,2);
        int digitCnt = str.length();
        double operations = 0;
        int sign = 1;
        
        for(int i =digitCnt-1; i >= 0; i--){
            int digit = str.charAt(i) - '0';
            double power = digit > 0 ? Math.pow(2, digitCnt - (i+1)) : 0;
            double steps = digit * (power*2-1);
            operations += (steps * sign);
            sign = sign * (digit ==0 ? 1 : -1);
        }
        int result = (int)operations;
        return result > 0 ? result : -result;
    }
}
