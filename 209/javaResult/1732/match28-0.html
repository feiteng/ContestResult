class Solution {
    public int minimumOneBitOperations(int n) {
        int count = 2;
        if(n == 1)return 1;
        while(count < n){
            count *= 2;
        }
        if(count == n)return n + n - 1;
        count /= 2;
        int res = 0;
        int p = 0;
        while(n > 0){
            while(n < count)count/=2;
            if(p % 2 == 0){
                n -= count;
                res += (count + count - 1);
                p++;
            } 
            else if(p % 2 == 1){
                n -= count;
                res -= (count + count - 1);
                p++;
            }
        }
        return res;
    }
}
