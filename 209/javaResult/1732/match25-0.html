import java.math.BigInteger;
class Solution {
    public int minimumOneBitOperations(int number) {
        String binary = Integer.toBinaryString(number);
        int digitCount = binary.length();
        int sign = 1;
        int result = 0;
        for (int i = 0; i < digitCount; i++) {
            int digit = binary.charAt(i)-'0';
            int power = digit==1 ? (int)Math.pow(2, digitCount-i-1 ) : 0;
            
            int steps = digit * (power*2-1);           
            result += steps*sign;
            sign = sign * (digit == 0 ? 1 : -1);
        }
        return result;
    }
    
}
