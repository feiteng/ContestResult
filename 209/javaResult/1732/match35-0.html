class Solution {
    public int minimumOneBitOperations(int n) {
        if (n == 0) {
            return 0;
        }
        int p = power(n);
        return (int)Math.pow(2, p + 1) - 1 - minimumOneBitOperations(n - (int)Math.pow(2, p));
    }
    
    private int power(int n) {
        int p = 0;
        while (n != 1) {
            n /= 2;
            p++;
        }
        return p;
    }
}
