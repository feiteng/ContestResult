class Solution {
    public int count(int k){
        int s=0;
        while(k>=0){
            s+=1<<k;
            k--;
        }
        return s;
    }
    public int minimumOneBitOperations(int n) {
        boolean flag=false;
        int ans=0;
        for(int k=30;k>=0;--k){
            if((n&(1<<k))!=0){
                ans+=flag==false?count(k):-count(k);
                flag=!flag;
            }
      
            
        }
          return ans;
    }
}
