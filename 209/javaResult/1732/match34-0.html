class Solution {
    public int minimumOneBitOperations(int n) {
        int[] lowbit = new int[31];
        
        lowbit[0] = 1;
        for (int i = 1; i < 30; i++) {
            lowbit[i] = lowbit[i - 1] * 2 + 1;
        }
        
        int res = 0;
        while (n > 0) {
            int lowb = n & (-n);
            n -= lowb;
            // System.out.println(lowb + "," + getIdx(lowb));
            res = lowbit[getIdx(lowb)] - res;
        }
        return res;
    }
    
    private int getIdx(int lowbit) {
        int ret = 0;
        while (lowbit > 0) {
            lowbit >>= 1;
            ret++;
        }
        return ret - 1;
    }
}
