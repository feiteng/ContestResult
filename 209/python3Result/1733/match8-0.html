import numpy as np
class Solution:
    def visiblePoints(self, points: List[List[int]], ang: int, location: List[int]) -> int:
        P = []
        res = 0
        ans = 0
        for x, y in points:
            if x == location[0] and y == location[1]:
                res += 1
            else:
                P.append(np.angle(x - location[0] + (y - location[1]) * 1j) / 2 / np.pi * 360)
        P.sort()
        dq = deque()
        for p in P:
            dq.append(p)
            while dq[-1] - dq[0] > ang:
                dq.popleft()
            ans = max(ans, len(dq))
        for p in P:
            dq.append(p + 360)
            while dq[-1] - dq[0] > ang:
                dq.popleft()
            ans = max(ans, len(dq))
        return ans + res
