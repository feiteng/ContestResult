class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        import math
        h = []
        vis = 0
        x0, y0 = location
        for x, y in points:
            if (x, y) == (x0, y0):
                vis += 1
            else:
                dx = x - x0
                dy = y - y0
                a = math.atan2(dy, dx) * 180 / math.pi
                h.append(a)
        # print(h)
        if not h:
            return vis
        h
.sort()
        h = h + [360 + c for c in h]
        maxpts = 0
        i, j = 0, 0
        while j < len(h):
            while h[j] - h[i] > angle:
                i += 1
            maxpts = max(maxpts, j - i + 1)
            j += 1
        return vis + maxpts
