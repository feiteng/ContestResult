import numpy as np
class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        def cart2pol(x, y):
            return np.arctan2(y, x) * 180 / np.pi

        x0, y0 = location
        cnt = sum(x==x0 and y==y0 for x, y in points)
        points = [cart2pol(x-x0, y-y0) for x, y in points if x!=x0 or y!=y0]
        
        points.sort()
        points += [i+360 for i in points]
        # print(points, angle)
        # points = points+points
        ans 
= 0
        lo = hi = 0
        for hi in range(len(points)):
            while (points[hi] - points[lo])> angle:
                lo += 1
            ans = max(ans, hi-lo+1)
        return ans + cnt
