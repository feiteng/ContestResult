class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        if angle>=360:
            return len(points)
        count=0
        c,d=location[0],location[1]
        ang=[]
        for a,b in points:
            if a==c and d==b:
                count+=1
            else:
                ang.append((math.atan2(b-d,a-c)/3.14159265359*180+360)%360)
        ang.sort()
        ang=ang+ang
        for m in range(len(ang)//2,len(ang)):
            ang[m]+=360
        p=0
        maxn=0
        for m,i in enumerate(ang[:len(ang)//2]):
            while p<len(ang) and ang[p]-i<angle+0.0001:
                p+=1
            maxn=max(maxn,p-m)
        return maxn+count
