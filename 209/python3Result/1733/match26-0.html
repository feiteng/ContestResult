from math import atan2, pi

def atan3(y, x):
    r = atan2(y, x) * 180 / pi
    if r < 0:
        r += 360
        assert(r >= 0)
    return r
    
class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        [x0,y0] = location
        
        angles = []
        in_points = 0
        
        for [x,y] in points:
            if [x,y] == [x0,y0]:
                in_points += 1
            else:
                angles += [atan3(y-y0,x-x0)]
        
        
        angles = sorted(angles)
        
        angles += [360 + a for a in angles]
        
        i = 0
        j = 0
        
        ans = -1
        
        while i < len(angles):
            while j < len(angles) and angles[j] - angles[i] <= angle + 1e-6 :
                j += 1
            
            ans = max(ans, j-i)
            i += 1

        return ans + in_points
