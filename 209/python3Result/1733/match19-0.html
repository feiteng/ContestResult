class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        a = []
        is_loc = 0
        xx, yy = location
        for x, y in points:
            if x==xx and y==yy:
                is_loc += 1
                continue
            a.append(math.atan2(y-yy, x-xx)*180/math.pi)
        a.sort()
        q = collections.deque()
        res = 0
        for p in a+[x+360 for x in a]:
            q.append(p)
            while q[0] < p-angle:
                q.popleft()
            res = max(res, len(q))
        return res + is_loc
