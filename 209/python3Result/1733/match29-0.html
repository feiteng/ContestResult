import numpy as np
class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        def getSlope(dot_x, dot_y, base_x, base_y):
            return (dot_y - base_y) / (dot_x - base_x)
        def getAngle(dot_x, dot_y, base_x, base_y):
            return np.arctan2(dot_y - base_y, dot_x - base_x)
        nums = []
        carry = 0
        for x, y in points:
            if x == location[0] and y == location[1]:
                carry += 1
            else:
                nums.append(getAngle(x, y, location[0], location[1]) * 180 / np.pi)
        nums.sort()

        res = 0

            deque.append(num)
            while deque and deque[-1] - deque[0] > angle:
                deque.popleft()
            res = max(res, len(deque))
        nums2 = []
        for num in nums:
            if num >= 0:
                nums2.append(-(180 - num))
            else:
                nums2.append(180 + num)
        nums2.sort()
        #print(nums2)
        deque = collections.deque()
        for num in nums2:
            deque.append(num)
            while deque and deque[-1] - deque[0] > angle:
                deque.popleft()
            #print(len(deque))
