class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        X,Y = location
        PI = math.pi
        FOV = math.radians(angle)
        
        bonus 
= 0
        angles = []
        for x,y in points:
            if (x,y) == (X,Y):
                bonus += 1
                continue
            pos = math.atan2(y-Y, x-X)
            if (pos < 0.0): pos += 2 * PI
            angles.append(pos)
        
        angles.sort()
        N = len(angles)
        angles = angles + [(2*PI)+ a for a in angles]
        r = 0
        res = 0
        
        for l in range
(N):
            while angles[r] - angles[l] <= FOV:
                r += 1
            res = max(res, r - l)
        
        return res + bonus
