class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        lists = []
        res = 0
        
        x = location[0]
        y = location[1]
        
        curr_res = 0
       # print("points " , points)

        for point in points:
    #        print("point " , point)
            if (point[0] == x):
                if(point[1] == y):
                    curr_res += 1
                elif(point[1] > y):
                    lists.append(90)
                else:
                    lists.append(-90)
                continue
                
            curr = math.atan((point[1]-y) / (point[0] - x)) /math.pi
       #     print(curr)
            curr *= 180
            
            if(point[0] < x):
                lists.append(180+curr)
            else:
                lists.append(curr)
                
        lists.sort()
        for i in range(len(lists)):
            lists.append(lists[i]+360)
 #       print(lists, res)
        p1 = 0
        p2 = 0
        while (p1 < len(lists)):
            while (p2 <len(lists)) and lists[p2] - lists[p1] <= angle:
                p2 += 1
            res = max(res, p2-p1)
            p1 +=1
        return res + curr_res
