class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        import math
        ansadd = len([x for x, y in points if [x,y]==location])
        dg = [ (math.atan2(x-location[0],y-location[1])/math.pi)*180 for x, y in points if [x,y]!=location]
        dg
.sort()
        dg2 = dg + [d + 360 for d in dg]
        ans = 1
        i = j = 0
        # print(dg2)
        while i < len(dg2):
            # print(i,j)
            while j < len(dg2) and (dg2[j]-dg2[i]) - angle <= 1e-6:
                j += 1
            # print(i,j)
            ans = max(ans, j-i)
            i += 1
        return ans + ansadd
