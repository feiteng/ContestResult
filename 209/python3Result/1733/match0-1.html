import math


class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        # math.atan(1) / math.pi
        
        list1 = []
        ans = 0
        
        x0 = location[0]
        y0 = location[1]
        
        sofar = 0
        
        for point in points:
            if (point[0] == x0):
                if (point[1] == y0):
                    sofar += 1
                elif (point[1] > y0):
                    list1.append(90)
                else:
                    list1.append(-90)
                continue
            
            curr = math.atan((point[1] - y0) / (point[0] - x0)) / math.pi
            curr *= 180
            
            if (point[0] < x0):
                list1.append(180 + curr)
            else:
                list1.append(curr)
                
        
        list1.sort()
        size = len(list1)
        for i in range(size):
            list1.append(list1[i] + 360)
        
        # print(list1, ans)
        
        i1 = 0
        i2 = 0
        ans = 0
        while(i1 < size):
            
while(i2 < len(list1) and list1[i2] - list1[i1] <= angle):
                i2 += 1
            ans = max(ans, i2-i1)
            i1 += 1
        return ans + sofar
