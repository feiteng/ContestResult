class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        ans = 0
        for p in points:
            p[0] -= location[0]
            p[1] -= location[1]
            if p[0] == 0 and p[1] == 0:
                ans += 1
        points = [math.atan2(p[1], p[0]) * 180 / math.pi for p in points if not (p[0] == 0 and p[1] == 0)]
        
        points.sort()
        points = points + [i + 360 for i in points]
        # print(points)
        l = 0
        r = 0
        res = 0
        while r < len(points):
            while r < len(points) and points[r] - points[l] <= angle:
                r += 1
            l += 1
            res = max(res, r - l + 1)
        return res + ans
