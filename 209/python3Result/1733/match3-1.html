import math
import bisect


class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        lx, ly = location
        num_always_visible = 0
        point_angles = []
        for px, py in points:
            if px == lx and py == ly:
                num_always_visible += 1
            else:
                dx, dy = px - lx, py - ly
                pt_deg = math.degrees(math.atan2(dy, dx))
                point_angles.append(pt_deg)
                if pt_deg <= 0:
                    point_angles.append(pt_deg + 360)
        point_angles.sort()
        
        idx = bisect.bisect_left(point_angles, 0)
        N = len(point_angles)
        max_in_view = 0
        while idx < N:
            left_idx = bisect.bisect_left(point_angles, point_angles[idx]-angle)
            max_in_view = max(max_in_view, idx - left_idx + 1)
            idx += 1
        return num_always_visible + max_in_view
