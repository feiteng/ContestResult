class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        
        same = 0
        ds = []
        for p in points:
            if p != location:
                d = math.atan2((p[1] - location[1]), (p[0] - location[0])) * 180/math.pi
                ds.append(d)
            else:
                same += 1
                
        ds.sort()
        n = len(ds)
        for i in range(n):
            ds.append(ds[i]+360)
        
        # print(ds)
        l = 0
        r = 0
        ans = 0
        while r < len(ds):
            if ds[r] - ds[l] <= angle:
                # print(ds[r], ds[l], angle)
                ans = max(ans, (r-l)+1)
                r += 1
            else:
                l += 1
                
        return ans + same
