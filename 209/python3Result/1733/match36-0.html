class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, l: List[int]) -> int:
        def degree(x, y):
            # print(x,y)
            return (math.atan2(y,x) / math.pi * 180 + 360) % 360
        
        # if angle == 0:
        #     return 0

        degrees = []
        origin = 0
        for p in points:
            if p[0]==l[0] and p[1]==l[1]:
                origin += 1
            else:
                degrees.append(degree(p[0]-l[0], p[1]-l[1]))
        # print(degrees)
        degrees.sort()
        # print(degrees)
        degrees = degrees + list(map(lambda x:x+360, degrees))
    
        # print(degrees)
        
        i = 0
        j = 0
        res = 0
        
        while j < len(degrees):
            i += 1
            while j < len(degrees) and degrees[j]-degrees[i] <= angle:
                j +=1
            res = max(res, j-i)
        
        return res + origin
