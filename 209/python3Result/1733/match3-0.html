class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        ptx, pty = location
        atpt = 0
        pt_angles = []
        
        for x, y in points:
            if x == ptx and y == pty:
                atpt += 1
            else:
                lx = x - ptx
                ly = y - pty
                pt_angles.append(math.degrees(math.atan2(ly, lx)))
                
        pt_angles.sort()
        pt_angles += [a + 360 for a in pt_angles]
        
        #print(pt_angles)
        
        best = 0
        i = 0
        for j, a in enumerate(pt_angles):
            while a - pt_angles[i] > angle:
                i += 1
            best = max(best, j - i + 1)
        
        return best + atpt
