class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        
        points_in_degrees = []
        
        second_round_points_in_degrees = []
        
        base = 0
        for point in points:
            if point == location:
                base += 1
                continue
            
            radians = math.atan2(point[1] - location[1], point[0] - location[0])
            degrees = math.degrees(radians)
            points_in_degrees.append(degrees)
        
        points_in_degrees.sort()
        for point in points_in_degrees:
            second_round_points_in_degrees.append(point + 360)
        
        points_in_degrees += second_round_points_in_degrees
        
        res 
= 0
        l = 0
        for r in range(len(points_in_degrees)):
            while points_in_degrees[r] - points_in_degrees[l] > angle:
                l += 1
            res = max(res, r - l + 1 + base)
        
        return res
