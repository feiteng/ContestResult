class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        nloc=[(p[0]-location[0],p[1]-location[1]) for p in points] # (x,y)
        ang=[]
        same=0
        for x,y in nloc:
            if x==0:
                if y==0:
                    same=1
                elif y>0:
                    ang.append(90)
                else:
                    ang.append(270)
            else:
                ang
.append(math.degrees(math.atan2(y,x))%360)
        ang.sort()
        #print (ang)
        
        ang2=ang+[x+360 for x in ang]
        j=0
        maxcnt=0
        #print (ang2)
        for i,val in enumerate(ang2):
            while ang2[j]<val-angle:
                j+=1
            #print (i,j)
            maxcnt=max(maxcnt,i-j+1)
                
            
        return maxcnt+same
