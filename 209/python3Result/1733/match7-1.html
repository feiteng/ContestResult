class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        ps = []
        d = 0
        for p in points:
            if p == location:
                d += 1
            else:
                ps.append(math.degrees(math.atan2(p[1] - location[1], p[0] - location[0])))
        ps.sort()
        ps.extend(list(map(lambda x: x + 360, ps)))
        res, i, j, n = 0, 0, 0, len(ps)
        
while j < n:
            while ps[j] - ps[i] > angle:
                i += 1
            res = max(res, j - i + 1)
            j += 1
        return res + d
