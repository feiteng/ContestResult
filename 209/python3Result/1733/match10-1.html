class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        from bisect import bisect
        directions = []
        rst = 0
        for point in points:
            if point[0] == location[0] and point[1] == location[1]:
                rst += 1
            else:
                anglex = math.atan2(point[1] - location[1], point[0] - location[0]) * 180 / math.pi
                directions.append(anglex if anglex >= 0 else 360 + anglex)
        directions.sort()
        directions.extend([x+360 for x in directions])
        # print(directions)
        count = 0
        for i, start in enumerate(directions):
            if start >= 360:
                break
            end = bisect(directions, start + angle)
            # print(start, start+angle, end)
            count = max(count, end - i)
        return rst + count
