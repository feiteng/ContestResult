class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        res = 0
        a = []
        for p in points:
            if p == location:
                res += 1
            else:
                v1 = p[0] - location[0]
                v2 = p[1] - location[1]
                a.append(math.atan2(v2, v1)*180/math.pi)
                
        a.sort()
        b = [i + 360 for i in a]
        a.extend(b)
        ans = 0
        i = j = 0
        while j < len(a):
            while j < len(a) and a[j] - a[i] <= angle:
                j += 1
            ans = max(ans, j - i)
            i += 1
        return ans + res
