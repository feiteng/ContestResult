class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        x0, y0 = location
        ans = 0
        degree = []
        pi = math.pi
        
        for x1, y1 in points:
            if x1 == x0:
                if y1 > y0: degree.append(90)
                elif y1 == y0: ans += 1
                else: degree.append(270)
            
            elif x1 > x0:
                if y1 >= y0:
                    tan_degree = (y1 - y0)/(x1 - x0)
                    deg = math.atan(tan_degree) * 180 / pi
                else:
                    tan_degree = (y0 - y1)/(x1 - x0)
                    deg = 360 - math.atan(tan_degree) * 180 / pi
                degree.append(deg)
            else:
                if y1 >= y0:
                    tan_degree = (y1 - y0)/(x0 - x1)
                    deg = 180 - math.atan(tan_degree) * 180 / pi
                else:
                    tan_degree = (y0 - y1)/(x0 - x1)
                    deg = 180 + math.atan(tan_degree) * 180 / pi
                degree.append(deg)
        
        degree.sort()
        
        temp = []
        for deg in degree:
            temp.append(deg + 360)
        degree += temp
            
        temp 
= 0
        for idx, deg in enumerate(degree):
            end = deg + angle
            loc = bisect.bisect_right(degree, end)
            temp = max(temp, loc - idx)
        return ans + temp
