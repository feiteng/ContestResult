class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        angles = []
        ans = 0
        x0, y0 = location
        for x, y in points:
            if x == x0 and y == y0:
                ans += 1
            else:
                phi = math.atan2(y-y0, x-x0)*180/math.pi
                angles.append(phi)
                angles.append(phi+360)
        angles.sort()
        max_int 
= 0
        left = 0
        for i, a in enumerate(angles):
            while left < i and a > angles[left] + angle:
                left += 1
            max_int = max(max_int, i-left+1)

        return max_int+ans
