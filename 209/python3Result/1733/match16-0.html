from math import atan2, degrees

class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        base=0
        aa = []
        x0, y0 = location
        for x1, y1 in points: 
            if (x1,y1)==(x0,y0):
                base+=1
                continue
            aa.append(degrees(atan2(y1-y0, x1-x0)))
        aa.sort()
        aa += [a+360 for a in aa]
        ans = 0
        j = 0
        for i in range(len(aa)-1):
            while aa[j] < aa[i]-angle: j += 1
            ans = max(ans, i-j+1)
        return ans+base
