class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        arcs = []
        cnt = 0
        for point in points:
            if point[0] == location[0] and point[1] == location[1]:
                cnt += 1
            else:
                if point[0] == location[0]:
                    if point[1] > location[1]:
                        arcs.append(90)
                    else:
                        arcs.append(270)
                elif point[1] == location[1]:
                    if point[0] > location[0]:
                        arcs.append(0)
                    else:
                        arcs.append(180)
                else:
                    if point[0] > location[0] and point[1] > location[1]:
                        arcs.append(math.degrees(math.atan((point[1] - location[1]) / (point[0] - location[0]))))
                    elif point[0] < location[0] and point[1] > location[1]:
                        arcs.append(math.degrees(math.atan((location[0] - point[0]) / (point[1] - location[1]))) + 90)
                    elif point[0] < location[0] and point[1] < location[1]:
                        arcs.append(math.degrees(math.atan((location[1] - point[1]) / (location[0] - point[0]))) + 180)
                    else:
                        arcs.append(math.degrees(math.atan((point[0] - location[0]) / (location[1] - point[1]))) + 270)
        #print(sorted(arcs))
        arcs.sort()
        extra = []
        for arc in arcs:
            extra.append(arc + 360)
        arcs.extend(extra)
        l = 0
        ans = cnt
        for i in range(len(arcs)):
            cnt += 1
            if arcs[i] - arcs[l] <= angle:
                ans = max(ans, cnt)
            else:
                while arcs[i] - arcs[l] > angle:
                    cnt -= 1
                    l += 1
        return ans
