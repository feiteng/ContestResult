import math
class Solution:
    def visiblePoints(self, points: List[List[int]], Angle: int, location: List[int]) -> int:
        # get a sliding window
        cnt = 0
        angles = []
        for p, q in points:
            if [p, q] == location:
                cnt += 1
            else:
                x, y = p - location[0], q - location[1]
                angle = atan2(y, x)
                angles.append(math.degrees(angle))
        
        angles.sort()
        angles = angles + [a+360 for a in angles]
        # print(angles)
        
        p = 0
        res = 0
        for q in range(len(angles)):
            while angles[q] - angles[p] > Angle:
                p += 1
            res = max(res, q-p+1)
        return res + cnt
