class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, loc: List[int]) -> int:
        countSame = 0
        listDegree = [];
        for pt in points:
            if pt[0] == loc[0] and pt[1] == loc[1]: 
                countSame += 1;
                continue;
            
            if pt[0] == loc[0]:
                if pt[1] > loc[1]:
                    listDegree.append(90.0);
                    #// listDegree.add(90.0 + 360);
                else:
                    listDegree.append(270.0);
                    #// listDegree.add(270.0 + 360);
            elif pt[1] == loc[1]:
                if pt[0] > loc[0]:
                    listDegree.append(0.0);
                    # // listDegree.add(0.0 + 360);
                else:
                    listDegree.append(180.0);
                    # // listDegree.add(180.0 + 360);
            else:
                y = pt[1] - loc[1] + 0.0;
                x = pt[0] - loc[0] + 0.0;
                tanVal = y / x;
                degree = math.atan2(y, x) * 180.0 / math.pi;
                # if (degree > 0 && x < 0) {
                #     degree += 180.0;
                # } else if (degree < 0) {
                #     degree += 180.0;
                #     if (x > 0) {
                #         degree += 180.0;
                #     }
                # }
                if degree < 0:
                    degree += 360.0
                listDegree.append(degree);
                # // listDegree.add(degree + 360.0);

        listDegree.sort();
        left = 0;
        right = 0;
        maxCnt = 0;
        for left in range(len(listDegree)):
            while right < len(listDegree) and listDegree[right] - listDegree[left] <= angle:
                right += 1;
                maxCnt = max(maxCnt, right - left);
            
            
            while right >= len(listDegree) and right < 2 * len(listDegree) and listDegree[right - len(listDegree)] + 360.0 - listDegree[left] <= angle:
                right += 1;
                maxCnt = max(maxCnt, right - left);
        
        
        return maxCnt + countSame;
