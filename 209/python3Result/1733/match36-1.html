class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        def ang(x1, y1, x2, y2):
            return math
.atan2((y2-y1), (x2-x1)) / math.pi * 180
        a = []
        flag = 0
        for p in points:
            if location[0] == p[0] and location[1] == p[1]:
                flag += 1
            else:
                a.append(ang(location[0], location[1], p[0], p[1]))
        
        a.sort()
        # print(a)
        n = len(a)
        for i in range(n):
            a.append(a[i] + 360)
        i, j = 0, 0
        ans = 0
        def check(i, j):
            if abs(a[j] - a[i]) < angle + 1e-6:
                return True
            else:
                return False
        for i in range(n):
            ang = a[i]
            l, r = i, i + n
            while l < r:
                mid  = (l + r + 1) >> 1
                if check(i, mid):
                    l = mid
                else:
                    r = mid - 1
            ans = max(ans, l - i + 1)
        
        return ans + flag
