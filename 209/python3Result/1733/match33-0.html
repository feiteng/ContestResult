class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        angle_new = []
        origin = 0
        for point in points:
            x, y = point
            rel_x, rel_y = x - location[0], y - location[1]
            if rel_x == 0 and rel_y == 0:
                origin += 1
                continue
            if rel_x > 0 and rel_y >= 0:
                angle_new.append(math.atan((y - location[1]) / (x - location[0])) * 180 / math.pi)
            elif rel_x == 0 and rel_y > 0:
                angle_new.append(90)
            elif rel_x < 0 and rel_y >= 0:
                angle_new.append(180 + math.atan((y - location[1]) / (x - location[0])) * 180 / math.pi)
            elif rel_x < 0 and rel_y < 0:
                angle_new.append(180 + math.atan((y - location[1]) / (x - location[0])) * 180 / math.pi)
            elif rel_x == 0 and rel_y < 0:
                angle_new.append(270)
            elif rel_x > 0 and rel_y < 0:
                angle_new.append(360 + math.atan((y - location[1]) / (x - location[0])) * 180 / math.pi)
        angle_new.sort()
        angle_new = angle_new + [an + 360 for an in angle_new]
        left = 0
        right = 0
        max_point = 0
        while right < len(angle_new) - 1:
            while right < len(angle_new) - 1 and angle_new[right+1] - angle_new[left] <= angle:
                right += 1
            max_point = max(max_point, right - left + 1 + origin)
            left += 1
        return max_point
