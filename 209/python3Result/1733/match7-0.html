import numpy as np

class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        cnt = 0
        ang = []
        for p in points:
            if p == location:
                cnt += 1
            else:
                ang.append(np.angle((p[0] - location[0]) + 1j * (p[1] - location[1]), deg=True))                    
        
        ang.sort()
        ang = ang + [x + 360 for x in ang]
        
        ans = 0
        n = len(ang) // 2
        i, j = 0, 0
        while i < n:
            while j < 2 * n and ang[j] - ang[i] <= angle:
                j += 1
            
            ans = max(ans, j - i)
            i += 1
        
        return cnt + ans
