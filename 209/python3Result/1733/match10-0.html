class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        import math
        angles = []
        same = 0
        for p in points:
            if p[0] == location[0] and p[1] == location[1]:
                same += 1
            else:
                a = math.atan2(p[0]-location[0], p[1]-location[1]) / (2 * math.pi) * 360
                angles.append(a)
        
        angles.sort()
        i = j = count = 0
        while j < 2 * len(angles):
            front, back = angles[j%len(angles)], angles[i%len(angles)]
            if j%len(angles) < i%len(angles):
                front += 360
            if j == i + len(angles):
                break
            if front - back <= angle:
                count = max(count, j-i+1)
                j += 1           
            else:
                i += 1
        return count + same
