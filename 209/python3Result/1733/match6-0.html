class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        points = [[x[0]-location[0], x[1]-location[1]] for x in points]
        org = 0
        pNew = []
        for x, y in points:
            if x == y == 0:
                org += 1
            else:
                pNew.append([x,y])                

        points = pNew
        dgs = [degrees(atan2(p[1], p[0])) for p in points]
        dgs.sort()
        n = len(dgs)
        dgs = dgs + [d + 360 for d in dgs]
        i = 0
        j = 0
        res = 0
        for i in range(n):
            while j-i < n and dgs[j] <= dgs[i] + angle:
                j+=1
            res = max(res, j-i)
        return res + org
