class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        arr = []
        pi = 3.141592653589793
        epsilon = 0.000001
        res, base = 0, 0
        for x, y in points:
            if x == location[0] and y == location[1]:
                base += 1
                continue
            cur = atan2(y-location[1],x-location[0])
            cur = cur/pi*180
            arr.append(cur)
        arr.sort()
        # print(arr)
        q = deque()
        
for a in arr:
            q.append(a)
            while q and q[-1]-q[0] > angle + epsilon:
                q.popleft()
            res = max(res,len(q))
        
for a in arr:
            q.append(a+360)
            while q and q[-1]-q[0] > angle + epsilon:
                q.popleft()
            res = max(res,len(q))
        return res + base
