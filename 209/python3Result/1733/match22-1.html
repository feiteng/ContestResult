import math

class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        base = 0
        angles = []
        mx, my = location
        for x, y in points:
            x = x - mx
            y = y - my
            if x == 0 and y == 0:
                base += 1
                continue
            angles.append((180 / math.pi) * math.atan2(y, x))
        angles.sort()
        angles
.extend([a + 360 for a in angles]) # cheat a little
        #print(angles)
        max_n = 0
        i = 0
        j = 0 # excl
        while j < len(angles):
            while j < len(angles) and angles[i] + angle >= angles[j]:
                j += 1
            max_n = max(max_n, j - i)
            i += 1
        return base + max_n
