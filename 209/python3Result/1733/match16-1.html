from math import atan, pi
class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        vectors = [(x - location[0], y - location[1]) for x, y in points]
        
        def to_angle(vector):
            if vector[0] == 0:
                if vector[1] > 0:
                    return 90
                else:
                    return 270
            degree = atan(vector[1] / vector[0]) / (pi) * 180
            if vector[0] < 0:
                degree += 180
            return degree % 360
        
        overlap 
= 0
        angles = []
        for vec in vectors:
            if vec[0] == 0 and vec[1] == 0:
                overlap += 1
                continue
            angles.append(to_angle(vec))
            
        
        angles.sort()
        # print(angles)
        
        
        ans 
= 0
        q = 0

        for p in range(len(angles)):
            while q < p and (angles[p] - angles[q]) % 360 > angle:
                q += 1
            ans = max(ans, p - q + 1)
            
        for p in range(len(angles)):
            while (angles[p] - angles[q]) % 360 > angle:
                # print(p, q, angles[p], angles[q])
                q = (q + 1) % len(angles)
            # print(p, q)
            ans = max(ans, (p - q) % len(angles) + 1)
        
        return ans + overlap
