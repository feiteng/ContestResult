class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        equal = 0
        a, b = location
        Ang = []
        f = lambda: math.acos ( (a-x) / math.sqrt((x-a)**2+(y-b)**2) ) * 180 / math.pi
        for x,y in points:
            if x==a and y==b:
                equal += 1
            else:
                bisect.insort( Ang,  f() if b>=y else 360-f() )
                
        Ang = [x-360 for x in Ang] + Ang
        
        start = 0
        ans = 0
        for end in range(len(Ang)):
            while Ang[end]-Ang[start]>angle: start += 1
            ans = max(ans, end-start+1)
            
        return ans+equal
