import math
import bisect

class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        num_add = 0
        lx, ly = location
        angles = []
        aglr = angle * math.pi / 180
        for x, y in points:
            if x == lx and y == ly:
                num_add += 1
            else:
                agl = math.atan2(y - ly, x - lx)
                if agl < 0:
                    agl = agl + 2 * math.pi
                angles.append(agl)
                if agl < aglr:
                    angles.append(agl + 2 * math.pi)
        angles.sort()
        # print(angles)
        # now ready to search over a range...
        max_n = 0
        for i, agl_ in enumerate(angles):
            ttt = agl_ + aglr
            idx = bisect.bisect_right(angles, ttt)
            # print(i, idx)
            max_n = max(max_n, idx - i)
        return max_n + num_add
