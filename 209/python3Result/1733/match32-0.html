from math import degrees, atan2

class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        x0, y0 = location
        
        same_location = sum([1 if point == location else 0 for point in points])
        
        D = [degrees(atan2(y-y0, x-x0)) for x, y in points if [x, y] != location]
        D += [d + 360 for d in D]
        D.sort()
        # print(D)
        l = r = 0
        max_in_view = 0
        while r < len(D):
            # might be <=. Is the angle inclusive?
            
            while D[r] - D[l] > angle and l < r:
                l += 1
                
            max_in_view = max(max_in_view, r - l + 1)
            
            r += 1
        
            
        return same_location + max_in_view
