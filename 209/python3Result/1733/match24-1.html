class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        if angle >=360: return len(points)
        
        free = len([p for p in points if p == location])
        
        other = [p for p in points if p!=location]
        
        px,py = location
        
        def anglef(x,y):
            if x == px:
                return 90 if y>py else 270
            ang = math.atan((y-py)/(x-px))*180/math.pi
            return ang if x>px else ang+180
        
        angs = sorted([anglef(*p) for p in other] + [anglef
(*p)+360 for p in other])
        
        fo = 0
        ba = 0
        rec = 0
        while fo<len(angs):
            while fo<len(angs) and angs[fo] - angs[ba] <= angle:
                fo+=1
            rec = max(rec, fo-ba)
            ba+=1
        
        return rec+free
