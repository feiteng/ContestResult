class Solution:
    def visiblePoints(self, points: List[List[int]], angle_t: int, location: List[int]) -> int:
        angle = []
        count = 0
        for pos_x, pos_y in points:
            if pos_x == location[0] and pos_y == location[1]:
                count += 1
                continue
            cur_angle = math.atan2(pos_x - location[0], pos_y - location[1])* 180 / math.pi
            angle.extend([cur_angle - 360, cur_angle, cur_angle + 360]) 
            
        result = -1
        angle.sort()
        left = 0
        for right in range(len(angle)):
            while angle[right] - angle[left] > angle_t:
                left += 1
                
            result = max(result, right - left + 1)
        
        return result + count
