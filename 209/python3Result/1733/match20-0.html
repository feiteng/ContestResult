class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        offset = 0
        q = []
        for p in points:
            y, x = p[1] - location[1], p[0] - location[0]
            if x == 0 and y == 0:
                offset += 1
                continue
            if x == 0:
                rad = math.pi / 2 if y > 0 else -math.pi / 2
            else:
                rad = math.atan(y / x)
            if x < 0:
                rad += math.pi
            elif x >= 0 and y < 0:
                rad += 2 * math.pi
            q.append(rad)
        # print(q)
        q.sort()
        
        q = q + [x + math.pi * 2 for x in q]
        a = angle / 180 * math.pi
        ans = 0
        l = 0
        for r in range(len(q)):
            while q[r] - q[l] > a:
                l += 1
            ans = max(ans, r - l + 1)
        return ans + offset
