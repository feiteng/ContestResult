import math
class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        x0 = location[0]
        y0 = location[1]
        
        
        def checkAngle(x1, y1):
            deltaX = x1-x0
            deltaY = y1-y0
            angleNow = math.atan2(deltaY, deltaX) 
            return angleNow/math.pi*180
        
        base = 0
        angleList = []
        for x, y in points:
            if x == x0 and y == y0:
                base += 1
                continue
            angleList.append(checkAngle(
x, y))
        
        angleList.sort()
        # print(angleList)
        angleListLen = len(angleList)
        for i in range(angleListLen):
            angleList.append(angleList[i] + 360)
        # print(angleList)
        
        l = 0
        r = 1
        ans = 1
        while r < 2*angleListLen:
            
if angleList[r] - angleList[l] <= angle:
                ans = max(ans, r-l + 1)
                r += 1
            else:
                l += 1
        return ans + base
