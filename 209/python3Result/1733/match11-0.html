class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        x, y = location
        degrees = []
        ziger = 0
        for i, j in points:
            if i == x and j == y:
                ziger += 1
                continue
            if i != x:
                if i > x:
                    degrees.append(math.atan((j - y) / (i - x)) * 180 / math.pi)
                else:
                    degrees.append(180 + math.atan((j - y) / (i - x)) * 180 / math.pi)
            elif j > y:
                degrees.append(90)
            else:
                degrees.append(-90)
        #print(degrees)
        degrees = sorted(degrees)
        #print(degrees)
        
        for d in degrees:
            if d <= angle:
                degrees.append(d + 360)
        res = 0
        s = 0
        for e in range(len(degrees)):
            while degrees[e] - degrees[s] > angle:
                s += 1
            res = max(res, e - s + 1)
        return res + ziger
