class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        angles = []
        other = 0
        for x, y in points:
            if x == location[0] and y == location[1]:
                other += 1
                continue
            temp = math.atan2(y-location[1], x-location[0]) * 180 / math.pi
            if temp < 0:
                temp = temp + 360
            angles.append(temp)
        angles.sort()
        # print(angles)
        res = 0
        n = len(angles)
        l = 0
        r = 0

        while r < 2*n:
            diff = angles[r] - angles[l] if r < n else angles[r-n] - angles[l] + 360
            if diff > angle:
                l += 1            
            r += 1
            res = max(res, r-l)
            if l >= n:
                break
        return res+other
