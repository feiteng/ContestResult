class Solution:
    def visiblePoints(self, points: List[List[int]], angle: int, location: List[int]) -> int:
        angles = []
        free = 0
        location2 = [location[0], location[1]+1]
        for point in points:
            if point[0] == location[0] and point[1] == location[1]:
                free+=1 
                continue
            a = [location[0] - location2[0], location[1] - location2[1]]
            b = [location[0] - point[0], location[1] - point[1]]
            a_dot_b = a[0] * b[0] + a[1] * b[1]
            abs_a = math.sqrt(a[0] * a[0] + a[1] * a[1])
            abs_b = math.sqrt(b[0] * b[0] + b[1] * b[1])
            #print(location, point, a_dot_b, abs_a, abs_b)
            theta = math.acos(a_dot_b / (abs_a * abs_b) )
            if (point[0] < location[0]):
                angles.append(360 - math.degrees(theta))
            else:
                angles.append(math.degrees(theta))
        angles.sort()
        #print(angles)
        start = 0
        ans = 0
        l = len(angles)
        for i in range(l):
            angles.append(360 + angles[i])
            
        for i in range(len(angles)):
            while angles[i] - angles[start] - 0.000000000001 > angle:
                start+=1
            ans = max(ans, i - start + 1)
        return ans + free
