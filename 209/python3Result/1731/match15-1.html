# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
  def isEvenOddTree(self, root: TreeNode) -> bool:
    levels = defaultdict
(list)
    def visit(node, level):
      if node is None:
        return
      levels[level].append(node.val)
      visit(node.left, level+1)
      visit(node.right, level+1)
      
    visit(root, 0)
    
    for level, lst in levels.items():
      
if level % 2 == 0:
        if lst[0] % 2 == 0:
          return False
        for i in range(1, len(lst)):
          if lst[i] <= lst[i-1] or lst[i] % 2 == 0:
            return False
      else:
        if lst[0] % 2 == 1:
          return False
        for i in range(1, len(lst)):
          if lst[i] >= lst[i-1] or lst[i] % 2 == 1:
            return False        
          
    return True
