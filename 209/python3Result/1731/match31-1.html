# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        idx = 0
        nodes = [root]
        while nodes:
            l = len(nodes)
            pre = - float('inf') if idx % 2 == 0 else float('inf')
            
for _ in range(l):
                node = nodes.pop(0)
                if idx % 2 == 0:
                    if node.val <= pre or node.val % 2 == 0:
                        return False
                else:
                    if node.val >= pre or node.val % 2 == 1:
                        return False
                pre = node.val
                if node.left:
                    nodes.append(node.left)
                if node.right:
                    nodes.append(node.right)
            idx += 1
        return True
