# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        if root.val % 2 == 0:
            return False
        queue = [root]
        i = 1
        while queue:
            tem = []
            for r in queue:
                if r.left:
                    tem += [r.left]
                if r.right:
                    tem += [r.right]
            if not tem:
                return True

            
if i % 2:
                if tem[0].val % 2:
                    return False
                for j in range(1,len(tem)):
                    if tem[j].val % 2 or tem[j].val >= tem[j-1].val:
                        return False
            else:
                if tem[0].val % 2 == 0:
                    return False
                for j in range(1,len(tem)):
                    if tem[j].val % 2 == 0 or tem[j].val <= tem[j-1].val:
                        return False
            queue = tem
            i += 1
