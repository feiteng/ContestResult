# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        level = 0
        q = deque()
        q.append(root)
        while q:
            size = len(q)
            flag = 1 if level % 2 == 0 else -1
            prev =  0 if flag == 1 else 200000000
            
for i in range(size):
                node = q.popleft()
                if flag == 1:  # increase
                    if node.val <= prev or node.val % 2 == 0:
                        return False
                else:
                    if node.val >= prev or node.val % 2 == 1:
                        return False
                prev = node.val
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            level += 1
        return True
