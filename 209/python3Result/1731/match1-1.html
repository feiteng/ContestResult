# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        queue = deque
([root])
        even = True
        while queue:
            pre = float('-inf') if even else float('inf')
            for _ in range(len(queue)):
                node = queue.pop()
                if even:
                    if not node.val & 1 or node.val <= pre:
                        return False
                else:
                    if node.val & 1 or node.val >= pre:
                        return False
                pre = node.val
                if node.left:
                    queue.appendleft(node.left)
                if node.right:
                    queue.appendleft(node.right)
            even = not even
        return True
