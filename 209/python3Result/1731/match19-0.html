# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        if root.val % 2 == 0:
            return False
        cur = [root]
        inc = -1
        while cur:
            nex = []
            for node in cur:
                if node.left:
                    nex.append(node.left)
                if node.right:
                    nex.append(node.right)
            if inc == 1:
                if nex and nex[0].val % 2 == 0:
                    return False
                for i in range(1, len(nex)):
                    if nex[i - 1].val >= nex[i].val or nex[i].val % 2 == 0:
                        return False
            else:
                if nex and nex[0].val % 2 == 1:
                    return False
                for i in range(1, len(nex)):
                    if nex[i - 1].val <= nex[i].val or nex[i].val % 2 == 1:
                        return False
            cur = nex
            inc *= -1
        return True
