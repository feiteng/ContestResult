# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        q = collections.deque()
        q.append(root)
        is_even = True # Increasing
        while q:
            pre = None
            for _ in range(len(q)):
                node = q.popleft()
                if is_even:
                    if node.val % 2 == 0 or pre is not None and node.val <= pre:
                        return False
                else:
                    if node.val % 2 == 1 or pre is not None and node.val >= pre:
                        return False
                pre = node.val
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            is_even = not is_even
        return True
