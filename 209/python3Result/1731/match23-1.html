# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        res = []
        def level_order(root,level,res):
            if not root:
                return
            if level >= len(res):
                res.append([])
            res[level].append(root.val)
            level_order(root.left,level + 1,res)
            level_order(root.right,level + 1,res)
        level_order(root,0,res)
        print(res)
        for i, v in enumerate(res):
            if i%2 == 0:
                
for node in v:
                    if node%2 == 0:
                        return False
                for j in range(len(v) - 1):
                    if v[j] >= v[j + 1]:
                        return False
            if i%2 == 1:
                
for node in v:
                    if node%2 == 1:
                        return False     
                for j in range(len(v) - 1):
                    if v[j] <= v[j + 1]:
                        return False
        return True
