# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        if root.val % 2 == 0:
            return False
        curr_layer_idx = 0
        curr_layer = [root]
        while len(curr_layer) > 0:
            nxt_layer = []
            nxt_layer_idx = curr_layer_idx + 1
            for node in curr_layer:
                if node.left is not None:
                    nxt_layer.append(node.left) 
                if node.right is not None:
                    nxt_layer.append(node.right)
            # Decrease, Even
            if nxt_layer_idx % 2 == 1:
                if len(nxt_layer) != 0 and nxt_layer[0].val % 2 == 1:
                    return False
                for idx in range(1, len(nxt_layer)):
                    if nxt_layer[idx].val >= nxt_layer[idx-1].val or nxt_layer[idx].val % 2 == 1:
                        return False
            # Increase, odd
            if nxt_layer_idx % 2 == 0:
                if len(nxt_layer) != 0 and nxt_layer[0].val % 2 == 0:
                    return False
                for idx in range(1, len(nxt_layer)):
                    if nxt_layer[idx].val <= nxt_layer[idx-1].val or nxt_layer[idx].val % 2 == 0:
                        return False
            curr_layer = nxt_layer
            curr_layer_idx = nxt_layer_idx

        return True
