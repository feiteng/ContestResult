# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        q=deque([root])
        o=1
        while q:
            if o==1:
                pre=-inf
            else:
                pre=inf
            for i in range(len(q)):
                cur=q.popleft()
                #print(cur.val,o)
                if o:
                    if (cur.val^o)&1 or cur.val<=pre:
                        #print(cur.val,pre,o,cur.val^o,cur.val<=pre)
                        return False
                else:
                    if (cur.val^o)&1 or cur.val>=pre:
                        #print(cur.val,pre,o,cur.val^o,cur.val>=pre)
                        return False
                if cur.left:
                    q.append(cur.left)
                if cur.right:
                    q.append(cur.right)
                pre=cur.val
            o^=1
        return True
