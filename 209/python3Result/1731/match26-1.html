# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
from collections import deque
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        q = collections.deque([root])
        odd = False
        while q:
            l = len(q)
            prev = float
('inf') if odd else -1
            #print('*', odd)
            for i in range(l):
                node = q.popleft()
                #print(node.val)
                if odd:
                    if node.val % 2 != 0 or node.val >= prev:
                        #print(node.val, prev)
                        return False
                else:
                    if node.val % 2 == 0 or node.val <= prev:
                        #print(node.val, prev)
                        return False
                prev = node.val
                if node.left:
                    q.append(node.left)
                if node.right:
                    q.append(node.right)
            odd = not odd
        return True
