# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        pointers = [root]
        nextPointers = []
        level = 0
        while pointers:
            values = []
            if level%2 == 0:
                preValue = 0
                
for pointer in pointers:
                    if pointer.val%2 == 0 or pointer.val <= preValue:
                        return False
                    preValue = pointer.val
                    if pointer.left:
                        nextPointers.append(pointer.left)
                    if pointer.right:
                        nextPointers.append(pointer.right)
            else:
                preValue = float('inf')
                for pointer in pointers:
                    if pointer.val%2 == 1 or pointer.val >= preValue:
                        return False
                    preValue = pointer.val
                    if pointer.left:
                        nextPointers.append(pointer.left)
                    if pointer.right:
                        nextPointers.append(pointer.right)
            pointers, nextPointers = nextPointers, []
            level += 1
        return True
