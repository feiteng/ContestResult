# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        level
=0
        nodes=[root]
        while nodes:
            newnodes=[]
            for node in nodes:
                if node.left:
                    newnodes.append(node.left)
                if node.right:
                    newnodes.append(node.right)
            if level%2==0:
                if nodes[0].val%2==0:
                    return False
                for i in range(1,len(nodes)):
                    if nodes[i].val%2==0 or nodes[i].val<=nodes[i-1].val:
                        return False
            else:
                if nodes[0].val%2==1:
                    return False
                for i in range(1,len(nodes)):
                    if nodes[i].val%2==1 or nodes[i].val>=nodes[i-1].val:
                        return False
            level+=1
            nodes=newnodes
        return True
