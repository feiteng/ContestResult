# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        queue = collections.deque([root])
        level = 0
        while queue:
            n = len(queue)
            node = queue.popleft()
            curVal = node.val
            if (curVal%2 + level%2) != 1: return False
            if node.left:
                queue.append(node.left)
            if node.right:
                queue.append(node.right)
            for _ in range(n - 1):
                node = queue.popleft()
                if level % 2 == 0:
                    if node.val%2 == 0 or node.val <= curVal:
                        return False
                else:
                    if node.val%2 == 1 or node.val >= curVal:
                        return False
                curVal = node.val
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
            level += 1
        return True
