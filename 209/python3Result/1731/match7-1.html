# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        
        def valid(nodeList, even):
            
if even:
                if nodeList[0].val % 2 == 0:
                    return False
                for i in range(1, len(nodeList)):
                    if nodeList[i].val <= nodeList[i-1].val or nodeList[i].val % 2 == 0:
                        return False
            else:
                if nodeList[0].val % 2 == 1:
                    return False
                for i in range(1, len(nodeList)):
                    if nodeList[i].val >= nodeList[i-1].val or nodeList[i].val % 2 == 1:
                        return False
            return True
        
        def recursion(currNodes, even):
            if not valid(currNodes, even):
                return False
            else:
                nextRow 
= []
                for c in currNodes:
                    if c.left is not None:
                        nextRow.append(c.left)
                    if c.right is not None:
                        nextRow.append(c.right)
                if nextRow != []:
                    return recursion(nextRow, not even)
                else:
                    return True
        return recursion([root], True)
