# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isEvenOddTree(self, root: TreeNode) -> bool:
        if root is None:
            return True
        q = [root]
        level=0
        while q:
            nq = []
            prev = sys.maxsize if level % 2==1 else -sys.maxsize
            for node in q:
                if level % 2==0:
                    if not (node.val > prev and node.val % 2==1):
                        # print(node.val, prev, node.val < prev, node.val % 2)
                        return False
                else:
                    if not (node.val < prev and node.val % 2 ==0):
                        # print(node.val)
                        return False
                prev = node.val
                if node.left:
                    nq.append(node.left)
                if node.right:
                    nq.append(node.right)
            level+=1
            q = nq
        return True
