class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        
        # f(n*2)=f(n)*2+f(n)%2ï¼Œf(n*2+1)=f(n)*2+1-f(n)%2
        @lru_cache(None)
        def dp(n):
            if n <= 1:
                return n
            if n & 1 == 0:
                return 2 * dp(n//2) + dp(n//2) % 2
            else:
                return dp(n//2) * 2 + 1 - dp(n//2) % 2

        return dp(n)
