from functools import lru_cache

class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        if n == 0: return 0
        
        @lru_cache(None)
        def f(s):
            if s == '0':
                return 1
            if s == '1':
                return 0
            n = len(s)
            if s[0] == '1':
                return g(s[1:])
            s2 = '1' + '0' * (n-2)
            return 1 + f(s[1:]) + g(
s2)
            
        @lru_cache(None)   
        def g(s):
            if s == '0':
                return 0
            if s == '1':
                return 1
            n = len(s)
            if s[0] == '0':
                return g(s[1:])
            s2 = '1' + '0' * (n-2)
            return 1 + f(s[1:]) + g(s2)  
            
        s = bin(n)[2:]
        return g(s)
