class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        arr = bin(n)[2:]
        arr = [int(a) for a in arr]
        arr = list(reversed(arr))
        print(arr)
        to1z = [0] * len(arr)
        toz = [0] * len(arr)
        
        to1z[0] = 0 if arr[0] == 1 else 1
        toz[0] = 1 - to1z[0]
        
        for i in range(1, len(arr)):
            if arr[i] == 0:
                to1z[i] = 1 + to1z[i-1] + 2 ** (i) - 1 #make
                toz[i] = toz[i-1]
                pass
            else:
                to1z[i] = toz[i-1]
                toz[i] = to1z[i-1] + 1 + 2 ** (i) - 1
                pass
            pass
        #print(to1z)
        #print(toz)#
        return toz[-1]
