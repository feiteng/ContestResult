class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        if n == 0:
            return 0
        F = [1]*32
        for i in range(31):
            F[i+1] = 2*F[i] + 1
        B 
= bin(n)[2:]
        L = len(B)
        #print (B, L)
        sign = 1
        res = 0
        for i in range(L):
            ind = L - i - 1
            if B[i] == '1':
                res += sign*F[ind]
                sign *= -1
        return res
