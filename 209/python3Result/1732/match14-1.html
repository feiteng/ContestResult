class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        def nbits(n): 
            return int(math.log(n, 2))+1\
        
        def maskbits(n, bits): 
            return n & ((1 << bits) - 1)
        
        
def n2zero(n, nbits): 
            if nbits == 1: 
                return n
            if n & 1 << (nbits - 1): 
                return n2one(maskbits(n, nbits-1), nbits-1) + 2**(nbits - 1)
            else: 
                return n2zero(maskbits(n, nbits - 1), nbits - 1)
            
        def n2one(n, nbits): 
            if nbits == 1: 
                return 1 - n
            if n & 1 << (nbits-1): 
                return n2zero(maskbits(n, nbits - 1), nbits - 1) 
            else: 
                return n2one(maskbits(n, nbits - 1), nbits - 1) + 2**(nbits - 1)
        if n == 0: 
            return 0
        return n2zero(n, nbits(n))
