from functools import lru_cache
class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        @lru_cache(None)
        def zero(x):
            if x == 0:
                return 0
            if x == 1:
                return 1
            if x == 2:
                return 3
            if x == 3:
                return 2
            
            msb = 0
            for i in range(31, -1, -1):
                if (1 << i) & x:
                    msb = i
                    break
            
            ret = one(msb - 1, x ^ (1 << msb)) # turn x to 11000...
            #print(x, ret)
            ret += 1 # turn msb to 0 (x = 01000...)
            x = 1 << (msb - 1)
            return ret + zero(x)
            
        
        @lru_cache(None)
        def one(bit, x):
            msb = 0
            for i in range(31, -1, -1):
                if (1 << i) & x:
                    msb = i
                    break
            
            ret = 0
            while msb < bit:
                ret += zero(x ^ (1 << msb)) # turn x to 01000...
                x = 1 << msb
                msb += 1
                ret += 1 # turn x to 11000...
                x |= 1 << msb 
                
            return ret + zero(x ^ (1 << bit))
            
        return zero(n)
