class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        tozero = {}
        toone = {}
        tozero[0] = 1
        toone[0] = 1
        for i in range(1, 50):
            tozero[i] = toone[i - 1] + 1 + tozero[i - 1]
            toone[i] = toone[i - 1] + 1 + tozero[i - 1]

        l = bin(n)[2:][::-1]
        l = [int(i) for i in l]
        print(l)

        def one(l):
            if len(l) == 1:
                if l[0] == 1:
                    return 0
                else:
                    return 1
            t = l[:]
            if t[-1] == 1:
                return zero(t[:len(t) - 1])
            else:
                return one(t[:len(t) - 1]) + 1 + tozero[len(t) - 2]

        def zero(l):
            if len(l) == 1:
                if l[0] == 0:
                    return 0
                else:
                    return 1
            t = l[:]
            if t[-1] == 0:
                return zero(t[:len(t) - 1])
            else:
                return one(t[:len(t) - 1]) + 1 + tozero[len(t) - 2]

        return zero(l)
