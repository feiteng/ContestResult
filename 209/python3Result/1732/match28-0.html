class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        if n == 0: return 0
        if n == 1: return 1
        
        @functools.lru_cache(None)
        def adding(x):
            if x == 1: return 1
            return 2*adding(x//2) + 1
        @functools.lru_cache(None)
        def zeroing(n):
            if n == 0: return 0
            if n == 1: return 1
            bs = format(n, 'b')
            if bs.count('1') == 1:
                return adding(n)
            if bs[1] == '1':
                right = bs[2:]
                right = int(right,2) if len(right) > 0 else 0
                return zeroing(right) + 1 + adding(1 << (len(bs)-2))
            else:
                idx = bs[1:].index('1')+1
                right = bs[idx+1:]
                right = int(right,2) if len(right) > 0 else 0
                asdf = zeroing(right) + 1 + adding(1 << (len(bs)-idx-1))
                resulting = (1 << (len(bs)-1)) + (1 << (len(bs)-idx))
                return asdf + zeroing(resulting)
        return zeroing(n)
