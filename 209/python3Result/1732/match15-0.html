class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        @lru_cache(None)
        def toZero(n: int) -> int:
            if n <= 1:
                return n
            m = 1 << (len(bin(n)) - 3)
            return toPow(n ^ m, m >> 1) + 1 + toZero(m >> 1)
        
        @lru_cache(None)
        def toPow(n: int, m: int) -> int:
            if n >= m:
                return toZero(n ^ m)
            else:
                return toPow(n, m >> 1) + 1 + toZero(m >> 1)
        
        return toZero(n)
