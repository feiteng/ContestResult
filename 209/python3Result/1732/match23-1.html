from functools import lru_cache

class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        return self.to00_(n)
    
    @lru_cache
    def to10_(self, n, w):
        if w==1:
            assert n<=1
            if n==1: return 0
            return 1
        if n>=2**(w-1):
            return self.to00_(n-2**(w-1))
        else:
            return self.to10_(n,w-1)+1+self.to00_(2
**(w-2))
            
    @lru_cache
    def to00_(self, n):
        if n==0: return 0
        if n==1: return 1
        w=floor(log(n,2))
        t=2**w
        return self.to10_(n-t,w)+1+self.to00_(t//2)
