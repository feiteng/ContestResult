class Solution:
  def minimumOneBitOperations(self, num: int) -> int:
    digits = [int(t) for t in "{0:b}".format(num)]
    n = len(digits)
    
    @lru_cache(maxsize=None)
    def to_zero(num_digit):
      if num_digit == 1:
        return 1
      else:
        return to_one(num_digit - 1) + 1 + to_zero(num_digit - 1)
      
    @lru_cache(maxsize=None)
    def to_one(num_digit):
      if num_digit == 1:
        return 1
      else:
        return to_one(num_digit - 1) + 1 + to_zero(num_digit - 1)    
    
    
    @lru_cache(maxsize=None)
    def find(idx, start_with):
      assert start_with in [0, 1]
      assert idx < n
      if idx == n - 1:
        return 1 if digits[idx] != start_with else 0
      if digits[idx] == start_with:
        return find(idx+1, 0)
      else:
        return find(idx+1, 1) + 1 + to_zero(n - (idx+1))
    
    
    return find(0, 0)
