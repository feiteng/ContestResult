@functools.lru_cache(None)
def bto0(p):
    if p == 0: return 1
    return bto0(p-1) * 2 + 1
@functools.lru_cache(None)
def xto0(x):
    if x == 0: return 0
    b1 = int(log2(x))
    # if x & (x-1) == 0: return bto0(b1)
    return bto0(b1) - xto0(x - (1<<b1))

class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        return xto0(n)
