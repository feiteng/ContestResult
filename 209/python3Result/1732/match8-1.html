class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        s=bin(n)[2:]
        ret=0
        neg=1
        for i in range(len(s)):
            dig=int(s[i])
            pw= 2**(len(s)-(i+1)) if dig else 0
            steps=dig * (pw*2-1)
            ret+=steps*neg
            neg = neg*(1 if not dig else -1)
        return ret
