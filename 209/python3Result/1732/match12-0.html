class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        #dp 1 = 1 01 -> 1
        #dp 2 = 3 10 -> 11 01 00
        #dp 3 = 2 11 -> 01 00 
        #dp 4 = 7  100 ->   101  111 110 010 011 001 000
        #dp 5 = 6, 101 -> 111 
        #dp 6 = 4 -> 110
        b = bin(n)[2:]
        l = len(b)
        accumulator = 0
        sign = 1
        for i in range(l):
            power = 0
            if b[i]  == '1':
                power = 2 **(l-(i+1))    
            if b[i]  == '1':
                steps =   (power * 2 - 1)
            else:
                steps = 0
            accumulator += steps *sign
            if b[i] == '1':
                sign = -1* sign       
        return accumulator
