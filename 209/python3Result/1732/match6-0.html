class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        b = bin(n)[2:]
        temp = len(b)
        answer = (1 << temp)
        if n == answer:
            return n
        b = list(b)
        for i in range(1, temp):
            b[i] = str(int(b[i]) ^ int(b[i-1]))
        return int(''.join(b), 2)
