class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        
        num = n
        bnum = bin(num)[2:]
        n = len(bnum)
        
        dp = [[0]*4 for _ in range(n)]
        
        for i in range(n):
            if i == 0:
                if bnum[-i-1] == "0":
                    dp[i][0] = 0
                    dp[i][1] = 1
                else:
                    dp[i][0] = 1
                    dp[i][1] = 0
                dp[i][2] = 1
                dp[i][3] = 1
            else:
                
if bnum[-i-1] == "0":
                    dp[i][0] = dp[i-1][0]
                    dp[i][1] = dp[i-1][1] + 1 + dp[i-1][2]
                else:
                    dp[i][0] = dp[i-1][1] + 1 + dp[i-1][2]
                    dp[i][1] = dp[i-1][0]
                dp[i][2] = dp[i-1][3] + 1 + dp[i-1][2]
                dp[i][3] = dp[i-1][3] + 1 + dp[i-1][2]
        
        return dp[n-1][0]
