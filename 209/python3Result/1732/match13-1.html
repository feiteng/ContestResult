class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        s = bin(n)[2:]
        digitCount = len(s)
        accumulator = 0
        sign = 1
        for i, d in enumerate(s):
            d = int(d)
            if d > 0:
                power = int(math.pow(2, digitCount - (i + 1)))
            else:
                power = 0
                
            steps = d * (power * 2 - 1)
            
            accumulator += steps * sign
            
            
            if d == 0:
                sign = sign*1
            else:
                sign = -1 * sign
        
        return accumulator
