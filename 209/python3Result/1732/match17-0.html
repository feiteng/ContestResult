from functools import lru_cache
class Solution:
    def minimumOneBitOperations(self, n: int) -> int:

        @lru_cache(None)
        def dp(st):
            if st == "0":
                return 0
            
            if st == "1":
                return 1
            
            if st[0] == "0":
                return dp(st[1:])
            
            st2 = "1" + "0" * (len(st) - 2)
            return 1 + dp2(st[1:]) + dp(st2)
                

        @lru_cache(None)
        def dp2(st):
            if st == "1":
                return 0
            
            if st == "0":
                return 1
            
            if st[0] == "1":
                return dp(st[1:])
            
            st2 = "1" + "0" * (len(st) - 2)
            return 1 + dp2(st[1:]) + dp(st2)
        
        st = bin(n)[2:]
        ans = dp(st)
                
        return ans
