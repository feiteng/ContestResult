class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        num=bin(n)[2:]
        PoM=0 #if PoM is even add, else subtract
        result=0
        lnum=len(num)
        for i in range(lnum):
            if num[i]=='1':
                if PoM==0:
                    result+=(2**(lnum-i)-1)
                    PoM=1
                else:
                    result-=(2**(lnum-i)-1)
                    PoM=0
                    
        return(result)
