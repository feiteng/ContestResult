from functools import lru_cache
class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        def high_bit(x):
            y = 1
            while y <= x:
                if y == x:
                    return y
                if (y << 1) > x:
                    return y
                y <<= 1
            return 0
        
        @lru_cache(None)
        def to_one(x, y):
            if x == y:
                return 0
            if y == 1:
                return 1
            if (x & y) != 0:
                return to_zero(x ^ y)
            ans = 0
            ans += to_one(x, y >> 1)
            ans += 1
            ans += to_zero(y >> 1)
            return ans
            
        @lru_cache(None)
        def to_zero(x):
            if x == 0:
                return 0
            if x == 1:
                return 1
            y = high_bit(x)
            ans = 0
            ans += to_one(x ^ y, y >> 1)
            ans += 1
            ans += to_zero(y >> 1)
            return ans
        
        return to_zero(n)
