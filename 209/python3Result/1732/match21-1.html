class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        if n==0:
            return 0
        if n==1:
            return 1
        
        
        dp 
= [0]*32
        mp = [0] * 32
        mp[0] = 1
        dp[0] = 1
        for i in range(1,30):
            mp[i] = 1 + dp[i-1]+mp[i-1]
            dp[i] = 1 + dp[i-1]+mp[i-1]
        arr = []
        temp = n
        while temp!=0:
            arr.append(temp%2)
            temp//=2
        arr = arr[::-1]
        
        def getone(arr):
            if len(arr)==1:
                return (1+arr[0])%2
            if arr[0]==1:
                return getzero(arr[1:])
            else:
                return dp[len(arr)-2] + getone(arr[1:]) + 1
            
        def getzero(arr):
            if len(arr)==1:
                return arr[0]
            if arr[0]==0:
                return getzero(arr[1:])
            else:
                return dp[len(arr)-2] + getone(arr[1:]) + 1
            
        return getzero(arr)
