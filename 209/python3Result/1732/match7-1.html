class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        b = bin(n)[2:]
        e = '0'
        ret = 0
        l = len(b)
        for i in range(l):
            if b[i] == e:
                e = '0'
            else:
                ret += 1 << (l - 1 - i)
                e = '1'
        return ret
