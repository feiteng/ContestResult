class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        bins = [int(x) for x in bin(n)[2:]][::-1]
        if len(bins) == 1:
            return bins[0]
        
        leading1 = [0] * 32
        leading1[0] = 1
        for i in range(1,32):
            leading1[i] = 2**i + leading1[i-1]
        
        dp = [[0,0] for i in range(len(bins))]
        
        dp[0] = [0,1] if bins[0] == 0 else [1,0]
        
        for i in range(1, len(bins)):
            if bins[i] == 0:
                dp[i][0] = dp[i-1][0]
                dp[i][1] = 1 + dp[i-1][1] + leading1[i-1]
            else:
                dp[i][0] = dp[i-1][1] + 1 + leading1[i-1]
                dp[i][1] = dp[i-1][0]
        return dp[len(bins)-1][0]
