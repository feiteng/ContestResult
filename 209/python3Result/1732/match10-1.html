class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        if n==0:
            return 0
        digits=[]
        while n>0:
            digits.append(n&1)
            n>>=1
            
        from functools import lru_cache
        
        
@lru_cache(None)
        def flip(i):
            if i==0:
                return 1
            return 2*flip(i-1)+1
            
        
        @lru_cache(None)
        def turn10(i):
            if i==0:
                return 1-digits[0]
            if digits[i]==1:
                return turn0(i-1)
            else:
                return turn10(i-1)+1+flip(i-1)
            
        @lru_cache(None)
        def turn0(i):
            if i==0:
                return digits[0]
            if digits[i]==0:
                return turn0(i-1)
            else:
                return turn10(i-1)+1+flip(i-1)

        
        high=len(digits)-1
        if high==0:
            return 1
        return turn0(high)
