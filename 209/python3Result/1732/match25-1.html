class Solution(object):
    def __init__(self):
        self.cache = {}

    
def minimumOneBitOperations(self, n):
        if n == 0:
            return 0
        if n == 1:
            return 1
        if n == 3:
            return 2
        if n == 2:
            return 3

        bits = []

        while n > 0:
            if n % 2 == 0:
                bits.insert(0, 0)
            else:
                bits.insert(0, 1)
            n = n//2
        return self.find(bits)

    def find(self, bits):
        if len(bits) == 1:
            return bits[0]
        elif len(bits) == 2:
            if bits[0] == 1:
                if bits[1] == 0:
                    return 3
                else:
                    return 2
            else:
                if bits[1] == 0:
                    return 0
                else:
                    return 1

        l = len(bits)


        if bits[0] == 1:
            low = 1 << (l-1)
            high = low*2-1
            if bits[1] == 0:
                temp = self.find(bits[2:])
                return high-temp
            else:
                temp = self.find(bits[2:])
                return low+temp
        else:
            high = (1 << (l-1))-1
            low = 0
            if bits[1] == 1:
                temp = self.find(bits[2:])
                return high-temp
            else:
                temp = self.find(bits[2:])
                return low+temp
