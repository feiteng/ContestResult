class Solution {
public:
    int specialArray(vector<int>& nums) {
        for (int num = 0; num <= 1000; ++num) {
            int cnt = 0;
            for (int num2: nums) {
                if (num <= num2) ++cnt;
            }
            if (cnt == num) return num;
        }
        return -1;
    }
};
