class Solution {
public:
    int specialArray(vector<int>& nums) {
        int maxv = *max_element(nums.begin(),nums.end());
        for(int v = 0; v <= maxv; ++v) {
            int cnt = 0;
            for(int i = 0; i < nums.size(); ++i) {
                if(nums[i] >= v) {
                    cnt++;
                }
            }
            if(cnt == v) {
                return v;
            }
        }
        return -1;
    }
};
