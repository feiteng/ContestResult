class Solution {
public:
    int specialArray(vector<int>& nums) {
        int n=nums.size();
        sort(nums.begin(),nums.end());
        for(int num=0;num<=1000;++num){
            int cnt=nums.end()-lower_bound(nums.begin(),nums.end(),num);
            if(num==cnt) return num;
        }
        return -1;

    }
};
