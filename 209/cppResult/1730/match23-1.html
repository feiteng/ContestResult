class Solution {
public:
    int specialArray(vector<int>& nums) {
        int n = nums.size();
        for (int x = 0; x <= n; x++){
            int tot = 0;
            for (int i = 0; i < n; i++){
                if (nums[i] >= x){
                    tot++;
                }
            }
            if (tot == x){
                return tot;
            }
        }
        return -1;
    }
};
