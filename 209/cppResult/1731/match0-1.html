/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

const int maxn = 1e5 + 10;
class Solution {
public:
vector<int> res[maxn];
int maxnum;
struct Node
{
	int level;
	TreeNode* x;
};

void bfs(TreeNode* root)
{
	queue<Node> q;
	q.push(Node{ 0,root });
	while (!q.empty())
	{
		Node top = q.front();
		q.pop();
		res[top.level].push_back(top.x->val);
		maxnum = max(maxnum, top.level);
		if (top.x->left != nullptr) q.push(Node{ top.level + 1,top.x->left });
		if (top.x->right != nullptr) q.push(Node{ top.level + 1,top.x->right });
	}
}
bool isEvenOddTree(TreeNode* root) {
	bfs(root);
	bool flag = false;
	for (int i = 0;i <= maxnum;i++)
	{
		if (i & 1)
		{
            if(res[i].size()>0&&res[i][0]%2==1) flag=true;
			for (int j = 1;j < res[i].size();j++)
				if (res[i][j] >= res[i][j - 1]||res[i][j]%2==1) flag = true;
		}
		else
		{
             if(res[i].size()>0&&res[i][0]%2==0) flag=true;
			for (int j = 1;j < res[i].size();j++)
				if (res[i][j] <= res[i][j - 1]|| res[i][j] % 2 == 0) flag = true;
		}
	}
	if(flag) return false;
    else return true;
}
};
