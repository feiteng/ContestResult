/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
        
        queue<TreeNode*> bfs;
        bfs.push(root);

        int level = 0;
        while(!bfs.empty())
        {
            int s = bfs.size();
            int checkOrder = 0;
            for(int i=0;i<s;i++)
            {
                auto current = bfs.front();
                bfs.pop();
                
                if(level % 2 == 0)
                {
                    // odd and increasing 
                    
                    if(current->val%2 == 0)
                        return false;
                    if(checkOrder == 0)
                    {
                        checkOrder = current->val;
                    }
                    else
                    {
                        //printf(" Check order %d current val %d",current->val,checkOrder);
                        if(checkOrder >= current->val)
                        {
                            return false;
                        }
                        
                        checkOrder = current->val;
                    }
                }
                else
                {
                    // even and decreasing 
                    
                     if(current->val%2 == 1)
                        return false;
                    if(checkOrder == 0)
                    {
                        checkOrder = current->val;
                    }
                    else
                    {
                        ///printf(" Check order %d current val %d \n",current->val,checkOrder);
                        if(checkOrder <= current->val)
                        {
                            return false;
                        }
                        
                        checkOrder = current->val;
                    }
                }
                
                
                    if(current->left != NULL)
                    {
                      // printf("%d ",current->left->val);
                         bfs.push(current->left);
                    }
                    
                    if(current->right != NULL)
                    {
                       // printf("%d ",current->right->val);
                         bfs.push(current->right);
                    }
            }
            
                                   // printf(" || End of level %d \n",level);
            level++;
        }
        return true;
    }
};
