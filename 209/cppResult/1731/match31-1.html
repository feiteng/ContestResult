/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
            stack<TreeNode*>lr,rl;
        rl.push(root);
        vector<vector<int>> ans;
        if(root==NULL)
            return true;
        while(!lr.empty() || !rl.empty())
        {   vector<int> g;
            while(!rl.empty())
            {
                TreeNode *temp=rl.top();
                g.push_back(temp->val);
                rl.pop();
                if(temp->left)
                    lr.push(temp->left);
                if(temp->right)
                    lr.push(temp->right);
                
            }
            if(g.size())
             ans.push_back(g);
            vector<int> f;
            while(!lr.empty())
            {
                TreeNode *temp=lr.top();
                f.push_back(temp->val);
                lr.pop();
                if(temp->right)
                    rl.push(temp->right);
                if(temp->left)
                    rl.push(temp->left);
            }
         if(f.size())
             ans.push_back(f);
        }
        

        for(int i=0;i<ans.size();i++){
          
             
           if(i%2==0){
                for(int j=0;j<ans[i].size();j++){
               
                    if(j+1<ans[i].size() && ans[i][j]>=ans[i][j+1])
                        return false;
                    if(ans[i][j]%2==0 or (j+1<ans[i].size() && ans[i][j+1]%2==0))
                        return false;
                }
               
            }else{
                 for(int j=0;j<ans[i].size();j++){
                    if(j+1<ans[i].size() && ans[i][j]>=ans[i][j+1])
                        return false;
                    if(ans[i][j]%2==1 or (j+1<ans[i].size() && ans[i][j+1]%2==1))
                        return false;
                }
            }
        
        }
        return true;
    }
};
