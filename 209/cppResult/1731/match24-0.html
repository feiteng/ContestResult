/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
        vector<TreeNode*> q[2];
        q[0].push_back(root);
        int level = 0;
        while (q[0].size() > 0) {
            if (level) {
                for (int i = 0; i < q[0].size(); i++) {
                    if (q[0][i]->val % 2 == 1) return false;
                }
                for (int i = 1; i < q[0].size(); i++) {
                    if (q[0][i]->val >= q[0][i - 1]->val) return false;
                }
            } else {
                for (int i = 0; i < q[0].size(); i++) {
                    if (q[0][i]->val % 2 == 0) return false;
                }
                for (int i = 1; i < q[0].size(); i++) {
                    if (q[0][i]->val <= q[0][i - 1]->val) return false;
                }
            }
            for (int i = 0; i < q[0].size(); i++) {
                if (q[0][i]->left) q[1].push_back(q[0][i]->left);
                if (q[0][i]->right) q[1].push_back(q[0][i]->right);
            }
            q[0].clear();
            q[0].swap(q[1]);
            level = 1 - level;
        }
        return true;
    }
};
