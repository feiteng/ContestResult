/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void recurse(vector<vector<int>> &lorder, TreeNode* r, int level) {
        if (r) {
            if (level >= lorder.size()) {
                lorder.push_back({r->val});
            }
            else {
                lorder[level].push_back(r->val);
            }
            if (r->left) {
                recurse(lorder, r->left, level+1);
            }
            if (r->right) {
                recurse(lorder, r->right, level+1);
            }
        }
    }
    
    bool isEvenOddTree(TreeNode* root) {
        vector<vector<int>> res;
        recurse(res, root, 0);
        
        for (int i = 0; i < res.size(); ++i) {
            if (i % 2 == 0) {
                for (int j = 0; j < res[i].size(); ++j) {
                    if (res[i][j] % 2 == 0) return false;
                    if (j > 0 && res[i][j] <= res[i][j-1]) return false;
                }
            }
            else {
                for (int j = 0; j < res[i].size(); ++j) {
                    if (res[i][j] % 2 == 1) return false;
                    if (j > 0 && res[i][j] >= res[i][j-1]) return false;
                }
            }
        }
        return true;
    }
};
