/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */

class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
        queue<TreeNode*> q;
        q.push(root);
        int l = 0;
        while (!q.empty()) {
            int n = q.size();

            int prev = -1;

            while (n) {
                TreeNode* cur = q.front();
                q.pop();
                
                if (l % 2 == 0) {
                    if ((prev != -1 && prev >= cur->val) || cur->val % 2 == 0) return false;
    
                } else {
                    if ((prev != -1 && prev <= cur->val) || cur->val % 2 == 1) return false;
                }
                prev = cur->val;    
                
                if (cur->left)
                    q.push(cur->left);
                if (cur->right) 
                    q.push(cur->right);
                
                --n;
            }
            
            l++;
        }
        return true;
    }
};
