class Solution {
public:
    bool isEvenOddTree(TreeNode *root)
    {
        if (root->val % 2 == 0)
        {
            return false;
        }
        vector<TreeNode*> vT;
        vT.push_back(root);
        int layer = 0;

        while (vT.size())
        {
            ++layer;
            vector<TreeNode*> vT2;
            for (auto *n : vT)
            {
                if (n->left)
                {
                    vT2.push_back(n->left);
                }
                if (n->right)
                {
                    vT2.push_back(n->right);
                }
            }

            if (layer % 2)
            {
                for (auto *n : vT2)
                {
                    if (n->val % 2)
                    {
                        return false;
                    }
                }
                for (int i = 1; i < vT2.size(); ++i)
                {
                    if (vT2[i]->val >= vT2[i - 1]->val)
                    {
                        return false;
                    }
                }
            }
            else
            {
                for (auto *n : vT2)
                {
                    if (n->val % 2 == 0)
                    {
                        return false;
                    }
                }
                for (int i = 1; i < vT2.size(); ++i)
                {
                    if (vT2[i]->val <= vT2[i - 1]->val)
                    {
                        return false;
                    }
                }
            }
            vT.swap(vT2);
        }
        return true;
    }
};
