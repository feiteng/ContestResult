/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<int> adj[100069];
    
    void dfs(TreeNode* curr, int depth) {
        if(!curr) return;        
        adj[depth].push_back(curr->val);
        dfs(curr->left, depth + 1);
        dfs(curr->right, depth + 1);
    }
    
    bool isEvenOddTree(TreeNode* root) {
        dfs(root, 0);
        
        for(int i = 0; i < 1e5 + 20; i++) {
            if(i & 1) {
                int pv = INT_MAX;
                for(int v : adj[i]) {
                    if(v & 1 || v >= pv) return false;
                    pv = v;
                }
            }
            else {
                int pv = INT_MIN;
                for(int v : adj[i]) {
                    if(v % 2 == 0 || v <= pv) return false;
                    pv = v;
                }
            }
        }
        return true;
    }
};
