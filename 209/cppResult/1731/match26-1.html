/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> db;
    void dfs(TreeNode* node, int lv) {
        if (!node)
            return;
        if (lv == db.size())
            db.push_back(vector<int>());
        db[lv].push_back(node->val);
        dfs(node->left, lv + 1);
        dfs(node->right, lv + 1);
    }
    bool isEvenOddTree(TreeNode* root) {
        dfs(root, 0);

        bool flag = true;
        for (int i = 0; i < db.size(); i++) {
            if (i % 2 == 0) {
                for (int j = 0; j < db[i].size(); j++) {
                    if (db[i][j] % 2 == 0) {
                        flag = false;
                        break;
                    }
                    if (j > 0 && db[i][j] <= db[i][j - 1]) {
                        flag = false;
                        break;
                    }
                }
            } else {
                for (int j = 0; j < db[i].size(); j++) {
                    if (db[i][j] % 2 != 0) {
                        flag = false;
                        break;
                    }
                    if (j > 0 && db[i][j] >= db[i][j - 1]) {
                        flag = false;
                        break;
                    }
                }
            }
        }
        return flag;
    }
};
