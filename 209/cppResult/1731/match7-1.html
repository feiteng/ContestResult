/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
        bool res = false;
        queue<TreeNode*> q;
        q.push(root);

        int level = 0;
        while(!q.empty())
        {
            int sz = q.size();
            vector<int> v;
            for(int s = 0; s < sz; ++s)
            {
                auto node = q.front();
                q.pop();
                v.push_back(node->val);
                if(node->left)
                    q.push(node->left);
                if(node->right)
                    q.push(node->right);

            }
            if(level%2 == 0)
            {
                for(int i =0; i < v.size(); ++i)
                {
                    if(v[i]%2 == 0)
                        return false;
                    if(i > 0 && v[i] <= v[i-1])
                        return false;
                }
            }
            else
            {
                for(int i =0; i < v.size(); ++i)
                {
                    if(v[i]%2 != 0)
                        return false;
                    if(i > 0 && v[i] >= v[i-1])
                        return false;
                }
            }
            level++;
        }
        return true;
    }
};
