/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
        vector<int> temp;
        return helper(root, 0, temp);
    }
    
    bool helper(TreeNode *root, int level, vector<int>& values){
        if (root==nullptr)
            return true;
        
        if ((level+(root->val))%2==0)
            return false;
        if (values.size()<=level){
            values.push_back(root->val);
        }
        else{
            if (level%2==0 && root->val<=values[level])
                return false;
            if (level%2==1 && root->val>=values[level])
                return false;
            values[level] = root->val;
        }
        bool t1 = helper(root->left, level+1, values);
        bool t2 = helper(root->right, level+1, values);
        return t1 && t2; 
    }
};
