/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    bool isEvenOddTree(TreeNode* root) {
        queue<TreeNode*> q;
        bool odd = false;
        q.push(root);
        while (q.size()) {
            int sz = q.size();

            vector<int> cur;
            for (int k = 0; k < sz; ++k) {
                auto f = q.front();
                q.pop();
                cur.push_back(f->val);
                if (f->left) q.push(f->left);
                if (f->right) q.push(f->right);
            }
            if (odd) {
                for (int i = 0; i < cur.size(); i++) {
                    if (cur[i] % 2 == 1 || i && cur[i] >= cur[i - 1]) return false;
                }
            }
            else {
                for (int i = 0; i < cur.size(); i++) {
                    if (cur[i] % 2 == 0 || i && cur[i] <= cur[i - 1]) return false;
                }
            }
            odd = !odd;
        }
        return true;
    }
};
