/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
	int level;
	vector<vector<int>> nd;
    bool isEvenOddTree(TreeNode* root) {
		level=0;
		tra(root,0);
		for(int i=0;i<level;i++){
			if(i%2==0){
				for(int j=0;j<nd[i].size()-1;j++){
					if(nd[i][j]>=nd[i][j+1])return false;
				}
				for(int j=0;j<nd[i].size();j++){
					if(nd[i][j]%2==0)return false;
				}
			}
			else{
				for(int j=0;j<nd[i].size()-1;j++){
					if(nd[i][j]<=nd[i][j+1])return false;
				}
				for(int j=0;j<nd[i].size();j++){
					if(nd[i][j]%2!=0)return false;
				}
			}
		}
		return true;
    }
	void tra(TreeNode* root,int L){
		if(root==nullptr)return;
		if(L==level){
			level++;
			nd.push_back(vector<int>());
		}
		nd[L].push_back(root->val);
		tra(root->left,L+1);
		tra(root->right,L+1);
	}
};
