class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        for(auto &v : points) {
            v[0] -= location[0];
            v[1] -= location[1];
        }
        vector<double> k;
        k.reserve(points.size() * 2);
        int allsee = 0;
        for(int i = 0; i < points.size(); i++){
            auto &p = points[i];
            if(p[0] == 0 && p[1] == 0){
                allsee++;
            }else{
                double t = atan2(p[1], p[0]);
                // 0-360
                k.push_back(t * 180 / 3.1415926 + 180);
            }

        }
        sort(k.begin(), k.end());
        int n = k.size();
        for(int i = 0; i < n; ++i){
            k.push_back(k[i]+360);
        }
        int res = 0;
        int l = 0, r = 0;
        while(r < 2 * n){
            if(k[r] - k[l] <= angle){
                res = max(res, r - l + 1);
                r++;
            }else{
                l++;
            }
        }
        return res + allsee;
    }
};
