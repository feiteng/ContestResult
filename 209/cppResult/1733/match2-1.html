class Solution {
public:
    const double pi = 3.1415926;

    int visiblePoints(vector<vector<int>>& p, int ang, vector<int>& lo) {
        vector<double> a;
        //cout << ang << " asd" << endl;
        int tag = 0;
        for (auto &t : p) {
            if(t[1] == lo[1] && t[0] == lo[0]) {
                ++tag;
                continue;
            }
            a.push_back(atan2(t[1] - lo[1], t[0] - lo[0]) * 180 / pi);
            if (a.back() < 0)
                a.back() += 360;
           // cout << a.back() << endl;
        }
        sort(a.begin(), a.end());
        
        int n = a.size();
        for (int i = 0; i < n; i++)
            a.push_back(a[i] + 360);
        n *= 2;
        int res = 0;

        int pi = 0;
        for (int i = 0; i < n; i++) {
            while(pi < n && a[pi] <= a[i] + ang)
                ++pi;
            res = max(res, pi - i);
        }
        return res + tag;
    }
};
