class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> angles;
        double PI = 3.14159265;
        int same = 0;
        
        for (auto p : points) {
            int x = p[0] - location[0];
            int y = p[1] - location[1];
            
            if (x == 0 && y == 0) {
                ++same;
            } else {
                double tmp = atan2(x, y) * 180 / PI;
                if (tmp < 0) tmp += 360;
                angles.push_back(tmp);
            }
        }
        
        sort(angles.begin(), angles.end());
        
        //for (auto i : angles)
        //    cout << i << " ";
        
        //cout << endl;
        
        int n = angles.size();
        
        for (int i = 0; i < n; ++i)
            angles.push_back(angles[i] + 360);
        
        int j = 0, res = same;
        for (int i = 0; i < angles.size(); ++i) {
            while(j < angles.size() && round((angles[i] + angle)*100000) >= round(angles[j]*100000)) ++j;
            res = max(res, j - i + same);
        }
        
        return res;
        
    }
};
