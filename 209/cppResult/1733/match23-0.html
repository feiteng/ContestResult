class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector <double> ang;
        int n = points.size(), base = 0;
        for (int i = 0; i < n; ++ i) {
            int dx = points[i][0] - location[0];
            int dy = points[i][1] - location[1];
            if (dx == 0 && dy == 0) base ++;
            else ang.push_back(atan2(dx, dy));
        }
        sort(ang.begin(), ang.end());
        const double pi = acos(-1.0);
        for (int i = 0; i < n; ++ i)
            ang.push_back(ang[i] + pi * 2);
        double angle_t = 1.0 * angle / 360 * pi * 2;
        int j = 0, ans = 0;
        for (int i = 0; i < ang.size(); ++ i) {
            while (j < ang.size() && ang[j] - ang[i] <= angle_t) ++ j;
            ans = max(ans, min(j - i, n - base));
        }
        return ans + base;
    }
};
