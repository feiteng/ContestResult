class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        static double factor = 180 / M_PI;
        vector<double> angles;
        angles.reserve(points.size());
        
        int numOrigin = 0;

        for (vector<int>& point : points) {
            double x = point[0] - location[0];
            double y = point[1] - location[1];
            if (x == 0 && y == 0) {
                numOrigin++;
                continue;
            }
            double degAngle = atan2(y, x) * factor;
            angles.push_back(degAngle);
            angles.push_back(degAngle + 360);
        }


        int maxNum = 0;
        int j = 0;
        sort(angles.begin(), angles.end());
        for (int i = 0; i < angles.size(); i++) {
            while (j < angles.size() && angles[j] <= angles[i] + angle) j++;
            maxNum = max(maxNum, j - i);
            if (j == angles.size()) break;
        }
        return maxNum + numOrigin;
    }
};
