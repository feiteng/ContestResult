class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
	    const double pi = atan(1) * 4;
	    vector<double> nums;
	    int n = points.size();
		int x0 = location[0];
		int y0 = location[1];

	    int ans = 0;
		for (int i = 0; i < n; i++) {
	    	int x = points[i][0];
	    	int y = points[i][1];
	    	if (x == x0 && y == y0) {
	    		ans++;
	    		continue;
			}
			nums.push_back(atan2(x - x0, y - y0) * 180 / pi);
		}
		sort(nums.begin(), nums.end());
		int m = nums.size();
		int le = 0;
        if (m == 0) return ans;
		int mx = 1;
        vector<double> nums2(nums);
        for (int i = 0; i < m; i++)

            nums2[i] += 360;
        nums.insert(nums.end(), nums2.begin(), nums2.end());
		for (int i = 1; i < 2 * m; i++) {
			while (nums[i] - nums[le] > angle) {
				le++;	
			}
			mx = max(mx, i - le + 1);
		}
        return ans + mx;
    }
};
