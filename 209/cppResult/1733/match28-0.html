class Solution {
public:
    int visiblePoints(vector<vector<int>>& a, int angle, vector<int>& l) {
        vector<double> t;
        double pi = 3.1415926535;
        //cout<<pi<<endl;
        int pr=0;
        
        for(auto x:a){
            double x1=x[0]-l[0];
            double y1=x[1]-l[1];
            if(x1==0 && y1==0){
                pr++;
                continue;
            }
            double r=atan2(y1,x1);
            double an=(r/pi)*180.0;
            t.push_back(an);
            if(an<0.0)
            t.push_back(an+360);
            
        }
        sort(t.begin(),t.end());
        int m=t.size();
        int ans=0;
        int l1=0;
        for(int i=0;i<m;i++){
            //cout<<t[i]<<" ";
            while(t[i]-t[l1]>angle)l1++;
            ans=max(ans,i-l1+1);
        }
        
        int n=a.size();
        ans+=pr;
        ans=min(ans,n);
        return ans;
        
    }
};
