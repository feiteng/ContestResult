class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
      const double pi=acos(-1.0);
      vector<double> g;
      int res=0,idx=0,a=0;

      for(int i=0;i<points.size();++i){
        if(points[i][1]==location[1]&&points[i][0]==location[0]){
          a++;continue;
        }
        g.push_back(atan2(points[i][1]-location[1],points[i][0]-location[0])*180/pi);
        if(g.back()<0)g.back()+=360;
      }
      int n=g.size();
      for(int i=0;i<n;++i)g.push_back(360+g[i]);
      sort(g.begin(),g.end());
      for(int i=0;i<n;++i){
        while(idx<g.size()&&g[idx]-g[i]<=angle)idx++;
        res=max(idx-i,res);
      }
      return res+a;
    }
};
