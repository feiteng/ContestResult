class Solution {
    double PI = acos(0)*2;
public:

    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> angs;
        int t = 0;
        for(auto p:points) {
            if(p==location) {
                t++;
                continue;
            }
            double deg = atan2(p[1]-location[1], p[0]-location[0]) * 180 / PI;
            if(deg<0)
                deg += 360;
            angs.push_back(deg);
            if(360-deg<=angle)
                angs.push_back(deg-360);
        }

        int m = 1;
        sort(angs.begin(), angs.end());
        int j = 0;
        for(int i=0; i<angs.size(); ++i) {
            while(j<angs.size() && angs[j] <= angs[i]+angle)
                j++;
            m = max(m, j-i);
            if(j==angs.size())
                break;
        }
        return m+t;
    }
};
