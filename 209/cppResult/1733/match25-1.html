class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int ans = 0;
        double pi = 3.14159265358979323846;
        vector<double> res;
        for(auto pts: points) {
            if (pts[0] == location[0] && pts[1] == location[1]) {
                ans += 1;
            } else {
                int dx = pts[0] - location[0];
                int dy = pts[1] - location[1];
                double ang = atan2(dy, dx);
                double ang2 = ang * 180 / 3.14159265358979323846;
                res.push_back(ang2);

            }
        }
        sort(res.begin(), res.end());
        int n = res.size();
        for(int i = 0; i < n; i++)
            res.push_back(res[i]+360);
        int slow = 0;
        int tmp = 0;
        for(int i = 0; i < res.size(); i++) {
            while (res[i]-res[slow] > angle) {
                slow++;
            }
            tmp = max(tmp, i-slow+1);
        }
        return ans + tmp;
    }
};
