#define pi 3.1415926535
class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> num;
        int zero=0;

        for(int i=0;i<points.size();i++)
        {
            int x=points[i][0]-location[0];
            int y=points[i][1]-location[1];
            if(x==0&&y==0)
            {
                zero++;
                continue;
            }
            double angle_tmp = atan2(y, x)*180/pi;
            num.push_back(angle_tmp);

        }
        int sz=num.size();
        for(int i=0;i<sz;i++)
        {
            num.push_back(num[i]+360);
        }
        sort(num.begin(),num.end());
        int right=0;
        int MAX=0;
        for(int i=0;i<num.size();i++)
        {
            while(right<num.size()&&num[right]<=num[i]+angle)
            {
                right++;
            }
            MAX=max(MAX,right-i);
        }
        return MAX+zero;
    }
};
