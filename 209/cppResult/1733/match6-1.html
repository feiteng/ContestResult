class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> angles;
        int origin = 0;

        for(int i = 0; i < points.size(); ++i) {
            if(points[i][0] == location[0] && points[i][1] == location[1]) {
                origin += 1;
            } else {
                angles.push_back(atan2(points[i][1] - location[1], points[i][0] - location[0]));
                angles.push_back(angles.back() + 2*M_PI);
            }
        }
        sort(angles.begin(), angles.end());
        double diff = M_PI*((double)angle/180.0);
        sort(angles.begin(), angles.end());
        int res = 0;

        for(int i = 0, j = 0; i < angles.size(); ++i) {
            while(angles[i] - angles[j] > diff + 1e-6) {
                ++j;
            }
            res = max(res, i - j + 1);
        }
        return res + origin;
    }
};
