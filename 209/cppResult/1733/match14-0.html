class Solution {
public:
    
    int num = 0;
    
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        
        
        
        vector<double> angles;
        vector<double> angles2;
        for (int i = 0; i < points.size(); ++i)
        {
            vector<int>& a = location;
            vector<int>& b = points[i];
            
            if (a[0] == b[0] && b[1] == a[0]) 
            {
                ++num;
                continue;
            }
            
            double theta = atan2(b[1] - a[1], b[0] - a[0]);          
            angles.push_back(theta);
            if (theta < 0) theta += 2 * 3.1415926;
            angles2.push_back(theta);           
        }
        
        double phi = angle / 360.0 * 3.1415926 * 2;
        sort(angles.begin(), angles.end());
        sort(angles2.begin(), angles2.end());
        
        return max(GetRes(angles, phi), GetRes(angles2, phi));
    }
    
    int GetRes(vector<double>& angles, double phi)
    {
        int res = 0;
        int start = 0;
        for (int i = 0; i < angles.size(); ++i)
        {
          
            while (angles[i] - angles[start] > phi)
            {
                ++start;
            }
            
            int len = i - start + 1;
            res = max(res, len);
        }
        
        return res + num;
    }
};
