class Solution {
private:
    double getAngle(vector<int>& a, vector<int>& b) {
        int dx = a[0] - b[0];
        int dy = a[1] - b[1];
        return atan2(dy, dx) * 180.0 / M_PI;
    }

public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int bonus = 0;
        vector<double> angles;
        for (int i=0; i<points.size(); i++) {
            if (points[i][0]==location[0] && points[i][1]==location[1]) {
                bonus++;
                continue;
            }
            
            double a = getAngle(points[i], location);
            if (a<0.0) {
                a += 360.0;

            }
            angles.push_back(a);
        }
        
        int n = angles.size();
        sort(angles.begin(), angles.end());
        
        for (int i=0; i<n; i++) {
            angles.push_back(angles[i] + 360.0);
        }
        
        int count = 0;
        for (int i=0, j=0; i<angles.size(); i++) {
            while (angles[i]-angles[j]>angle) {
                j++;
            }
            count = max(count, i-j+1 + bonus);
        }
        
        return count;
    }
};
