#define PI 3.14159265
class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int addon = 0;
        for (int i = 0; i < points.size(); i++){
            points[i][0] -= location[0];
            points[i][1] -= location[1];
        }
        vector<double> degree;
        for (int i = 0; i < points.size(); i++){
            if (points[i][1]==0 && points[i][0]==0){
                addon++;
            } else degree.push_back(atan2(points[i][1],points[i][0])*180/PI);
        }
        sort(degree.begin(), degree.end());
        int k =  degree.size();
        for (int i = 0; i < k; i++){
            degree.push_back(degree[i]+360.0);
        }
                cout << "bug";
        //for (auto x:degree) cout << x << "\n";
        int mx = 0;
        int left = 0, right = 0;
        while (right < degree.size()){
            if (degree[left] + angle >= degree[right]){
                mx = max(mx, right-left+1);
                right++;
            } else {
                left++;
            }
            //cout << left << " " << right << "\n" ;
        }
        
        
        return mx+addon;
        
    }
};
