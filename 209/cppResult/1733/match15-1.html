#define pi 3.1415926

class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> angles;
        int c = 0;
        for (auto p : points) {
            double dx = p[0] - location[0];

            double dy = p[1] - location[1];
            if (p[0] == location[0] && p[1] == location[1]) {
                c++;
                continue;
            }
            double a = atan2(dy, dx) * 180 / pi;
            angles.push_back(a);
        }
        sort(angles.begin(), angles.end());
        int ans = 0;
        for (int i = 0; i < angles.size(); i++) {
            double right = angles[i] + angle;
            int count = 1 + c;
            if (right > 180) {
                count = count + angles.size() - i - 1;
                right = angle + angles[i] - 360;
                int p = upper_bound(angles.begin(), angles.end(), right) - angles.begin();
                count = count + p;
            } else {
                int p = upper_bound(angles.begin() + (i + 1), angles.end(), right) - angles.begin();
                count = count + p - i - 1;
            }
            ans = max(ans, count);
        }
        return ans;
    }
};
