class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int x = location[0];
        int y = location[1];
        const double pi = acos(-1);
        int ans = 0;
        vector<double> ang;
        for(int i = 0; i < points.size(); i++) {
            int u = points[i][0] - x;
            int v = points[i][1] - y;
            if(u == 0 && v == 0) {
                ans++;
                continue;
            }
            //printf("%d %d\n", u, v);
            if(u == 0) {
                if(v > 0) {
                   ang.push_back(90); 
                } else {
                    ang.push_back(270); 
                }
            } else {
                double t = 0.0;
                if(u>= 0 && v >= 0) {
                    t = atan(1.0*v/u);
                } else if(u>=0 && v <= 0) {
                    t = 2*pi-atan(abs(1.0*v/u));
                } else if(u<=0 && v <= 0) {
                    t = pi+atan(abs(1.0*v/u));
                } else {
                    t = pi-atan(abs(1.0*v/u));
                }
                ang.push_back(t*180.0/pi);
                //printf("%lf %lf\n", t*180.0/pi, pi);
            }
        }
        sort(ang.begin(), ang.end());
        int p = ang.size();
        for(int i = 0; i < p; i++) {
            ang.push_back(ang[i]+360);
        }
        int tot = 0;
        int nx = 0;
        for(int i = 0; i < ang.size(); i++) {
            nx = max(i,nx);
            while(nx+1<ang.size() && ang[nx+1]<=ang[i]+angle) {
                nx++;
            }
            //printf("%lf %d %d\n", ang[i], i, nx);
            tot = min(max(tot, nx-i+1), p);
        }
        return ans+tot;
    }
};
