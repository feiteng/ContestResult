class Solution {
public:
    #define PI 3.14159265
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> nums;
        bool flag = false;
        for(auto& cur : points){
            if(cur[1]-location[1]==0&&cur[0]-location[0]==0){
                flag = true;
                continue;
            }
            double temp = atan2(cur[1]-location[1],cur[0]-location[0])*180/PI;
            if(temp<0)temp+=360;
            nums.push_back(temp);
            //cout<<nums.back()<<endl;
        }
        int j = 0;
        int ans = 0;
        sort(begin(nums),end(nums));
        for(int i = 0; i < nums.size(); i++){
            while(j<nums.size()&&nums[j]-nums[i]<=angle)j++;
            while(j>=nums.size() &&j<i+nums.size()&& 360+nums[j-nums.size()]-nums[i]<=angle)j++;
            //cout<<j<<endl;
            ans = max(ans,j-i);
        }
        if(flag) return ans+1;
        return ans;
    }
};
