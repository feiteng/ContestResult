class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int n = points.size(), same = 0;
        vector<double> v;
        for (int i=0; i<n; ++i) {
            if (points[i] == location) {
                ++same;
                continue;
            }
            v.push_back(atan2(points[i][1] - location[1], points[i][0] - location[0]));
        }
        
        sort(begin(v), end(v));
        int sz = v.size();
        for (int i=0; i<sz; ++i) {
            v.push_back(v[i] + 2 * 3.1415926);
        }
        
        double ang = angle / 180.0 * 3.1415926;
        
        int res = 0, l = 0, r = 0;
        while (r<v.size()) {
            while (r <v.size() && v[r] - v[l] <= ang) ++r;
            res = max(res, r - l);
            while (r <v.size() && v[r] - v[l] > ang) ++l;
        }
        
        return res + same;
    }
};
