class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> pa; int must = 0;
        for (auto i : points) {
            if (i[1] == location[1] && i[0] == location[0]) { must++; continue; }
            double t = atan2(i[1] - location[1], i[0] - location[0]) * 180 / 3.1415926535;
            pa.push_back(t);
        }
        sort(pa.begin(), pa.end()); for (int i = 0; i < points.size(); i++) pa.push_back(pa[i] + 360); int j = 0, r = 0;
        for (int i = 0; i < points.size(); i++) {
            while (j < pa.size() && pa[j] - pa[i] <= angle + 1e-5) j++; r = max(r, j - i);
        }
        return r + must;
    }
};
