class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> angles;
        double conv = 90.0/acos(0.0);
        int always = 0;
        for(int i=0;i<points.size();i++) {

            if(location[1]==points[i][1] && location[0]==points[i][0]) {
                always++;
            }
            else {
                double ang = conv*atan2(points[i][1]-location[1],points[i][0]-location[0]);
                if(ang<0)
                    ang+=360.0;
                angles.push_back(ang);
            }
        }
        sort(angles.begin(),angles.end());
        int max_points = 0;
        for(int i=0;i<angles.size();i++) {
            double end_angle = angle+angles[i];
            int cur_sweep = 0;
            if(end_angle<=360) {
                cur_sweep = upper_bound(angles.begin(),angles.end(),end_angle) - angles.begin()-i;;
            }
            else {
                cur_sweep += angles.size()-i;
                end_angle-=360;
                cur_sweep += upper_bound(angles.begin(),angles.end(),end_angle) - angles.begin();
            }
            max_points = max(max_points, cur_sweep);
        }
        return max_points+always;
    }
};
