class Solution {
public:
#define M_PI       3.14159265358979323846   // pi
    int visiblePoints(vector<vector<int>>& points, int a, vector<int>& location) {
        const double pi=M_PI,pi_45=M_PI*2.0*(double(a)/(double)360),pi_360=M_PI*2.0;
        const double epsilon=1e-10;
        int n=points.size(),result=0;
        vector<double> angles;

        int f=0;
        for(int i=0;i<n;i++){
            if(location[0]==points[i][0] && location[1]==points[i][1]){f=1;continue;}
            double angle=atan2(points[i][1]-location[1],points[i][0]-location[0]);
            angles.push_back(angle);
            if(angle<pi_45-pi){
                angles.push_back(angle+pi_360);

            }
        }
        sort(angles.begin(),angles.end());
        //for(auto x:angles)cout<<x<<" ";
        //std::sort(angles.begin(), angles.end());
        for(auto it=angles.begin();it!=angles.begin()+n;++it){
            auto bound=upper_bound(it,angles.end(),*it+(pi_45+epsilon));
            int curr=distance(it,bound);
            if(curr>result){
                result=curr;
            }
        }
        return f+result;
    }
};
