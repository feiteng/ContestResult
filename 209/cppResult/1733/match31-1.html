class Solution {

public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location)
    {
        int same = 0;
        vector<double> angles;
        for (auto& p : points) {
            if (p[0] == location[0] && p[1] == location[1])
                same++;
            else
                angles.push_back(180.0 / M_PI * atan2(p[1] - location[1], p[0] - location[0]));
        }
        sort(angles.begin(), angles.end());
        int temp = angles.size();
        for (int i = 0; i < temp; i++)
            angles.push_back(angles[i] + 360);
        int maxvisible = 0;
        int i = 0, j = 0;
        while (j < angles.size()) {
            while (j < angles.size() && angles[j] - angles[i] <= angle + 1e-12) j++;
            maxvisible = max(maxvisible, j - i);
            i++;
        }
        return same + maxvisible;
    }
};
