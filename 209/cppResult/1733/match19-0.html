class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int n = points.size();
        double ang = angle * M_PI/180;
        // cout << "ang: " << ang << endl;
        vector<double> arr;
        int cnt = 0;
        for ( int i = 0; i < n; ++i ) {
            if ( points[i] == location ) { 
                ++cnt;
                continue;
            }
            
            arr.push_back(getAngle(points[i], location));
            // cout << arr.back() << endl;
        }
        sort(arr.begin(), arr.end());
        int m = arr.size();
        for ( int i = 0; i < m; ++i ) arr.push_back(arr[i] + 2 * M_PI);
        int res = 0;
        int pt = 0;
        for ( int i = 0; i < m; ++i ) {
            while ( pt < 2*m && arr[pt] <= arr[i] + ang + 0.000002 ) ++pt;
            // cout << "pt: " << pt << endl;
            res = max(res, pt - i);
        }
        return res + cnt;
    }
    
    double getAngle(vector<int>& p, vector<int>& loc) {
        double x = p[0] - loc[0], y = p[1] - loc[1];
        double c = x/sqrt(x*x + y*y);
        double ret = acos(c);
        // cout << ":" << c << " " << ret << " " << sqrt((x*x + y*y) * (xl*xl + yl*yl))<< endl;
        if ( y < 0 ) {
            ret = 2 * M_PI - ret;
        }
        return ret;
    }
};
