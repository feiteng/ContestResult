#define PI acos(-1)

class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> nums;
        int res=0;
        for (vector<int> point:points)
        {
            double x=point[0]-location[0],y=point[1]-location[1];
            //cout << x << " " << y << endl;
            if (y==0)
            {
                if (x>0)
                {
                    nums.push_back(0);
                }
                else if (x==0)
                {
                    res++;
                }
                else
                {
                    nums.push_back(180);
                }
                continue;
            }
            double tmp=y/x;
            double ans=atan(tmp)*180/PI;
            if (x<0)
                ans+=180;
            //cout << ans << endl;
            nums.push_back(ans);
        }

        int mx=0,next=0,len=nums.size();
        sort(nums.begin(),nums.end());
        for (int i=0;i<len;i++)
        {
            while (next<i+len && (nums[next%len]-nums[i]>=0 && nums[next%len]-nums[i]<=angle) || (nums[next%len]-nums[i]<0 && nums[next%len]-nums[i]+360<=angle))
                next++;
            mx=max(mx,next-i);
        }
        return mx+res;
    }
};
