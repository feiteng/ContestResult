#define PI 3.14159265
class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> cnt;
        int duplicate = 0;
        for(vector<int>& point:points) {
            int dx = point[0] - location[0];
            int dy = point[1] - location[1];
            double _angle = 0.0;
            // cout<<"dx="<<dx<<",dy="<<dy<<endl;
            if(dx == 0 && dy == 0)duplicate++;
            if(dx == 0) {
                _angle = (dy > 0)? 90.0:270.0;
            }else {
                if(dx > 0){
                    if(dy >= 0) {
                        _angle = atan (dy * 1.0 / dx) * 180 / PI;
                    } else {
                        _angle = 360.0 - atan (-dy * 1.0 / dx) * 180 / PI;
                    }
                }else{
                    if(dy >= 0) {
                        _angle = 180.0 - atan (-dy * 1.0 / dx) * 180 / PI;
                    } else {
                        _angle = 180.0 + atan (dy * 1.0 / dx) * 180 / PI;
                    }
                }
            }
            cnt.push_back(_angle);
            cnt.push_back(_angle + 360.0);
        }
        sort(cnt.begin(), cnt.end());
        // for(double x:cnt)cout<<x<<endl;
        
        int res = 0,begin = 0,len = cnt.size();
        for(int i = 0;i < len;i++) {
            while((cnt[i] - cnt[begin] - 1e-4) > angle * 1.0 && begin <= i) {
                // cout<<":"<<cnt[i] - cnt[begin]<<","<<angle<<endl;
                begin++;
            }
            res = max(res, i - begin + 1);
            // cout<<begin<<","<<i<<endl;
        }
        // cout<<endl;
        return res + duplicate;
//         [[13779926,599856510],[195766825,597976710],[119515491,575316056],[744777345,796161766],[187192636,870346582],[413112378,430889309],[436399518,387904921],[296153131,221188617],[536914240,985130562],[226391292,83241861]]
// 64
// [451961560,358354259]
    }
};
