class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int base_ans = 0, ans = 0;
        vector<double> point_angle;
        for (int i = 0; i < points.size(); ++i) {
            if (points[i][0] == location[0] && points[i][1] == location[1]) {
                ++base_ans;
            }
            else {
                point_angle.push_back(atan2(points[i][1] - location[1], points[i][0] - location[0]));

            }
        }
        sort(point_angle.begin(), point_angle.end());
        int n = point_angle.size();
        for (int i = 0; i < n; ++i) {
            point_angle.push_back(point_angle[i] + 2 * M_PI);
        }
        double range_angle = (double) angle / 360 * 2 * M_PI;
        int last_index = 0;
        for (int i = 0; i < n; ++i) {
            while (last_index < point_angle.size() && point_angle[i] + range_angle + 1e-7 > point_angle[last_index]) {
                ++last_index;
            }
            ans = max(ans, last_index - i);
        }
        return ans + base_ans;
    }
};
