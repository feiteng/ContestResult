class Solution {
public:
    double PI=acos(-1.0);
    double ang(double x,double y)
    {
        return atan2(y,x)*180.0/PI;
    }
    double sub(double x,double y)
    {
        if(x-y>=0) return x-y;
        return x-y+360;
    }
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> vec;
        int n=points.size();
        int cnt0=0;
        for(int i=0;i<n;i++)
        {
            int x=points[i][0],y=points[i][1];
            x-=location[0]; y-=location[1];
            if(x==0&&y==0) cnt0++;
            else
                vec.push_back(ang(x,y));
        }
        //cout<<"here"<<endl;
        sort(vec.begin(),vec.end());
        int m=vec.size();
        for(int i=0;i<m;i++)
            vec.push_back(vec[i]);
        int l=0;
        int ans=0;
        for(int r=0;r<m*2;r++)
        {
            while(l+m<=r||sub(vec[r],vec[l])>angle) l++;
            ans=max(ans,r-l+1);
        }
        return ans+cnt0;
    }
};
