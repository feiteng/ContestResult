class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        const double pi=M_PI,pi_360=M_PI*2.0;
        const double pi_45 = ((pi*angle)/180);
        
        const double epsilon=1e-10;
        int n=points.size(),result=0;
        
        vector<double> angles(n);
        for(int i=0;i<n;i++){
            int y = points[i][1]-location[1];
            int x = points[i][0]-location[0];
            if(y == 0 && x == 0){
                y += 1;
                x += 1;
            }
            double angle=atan2(y,x);
            angles[i]=angle;
            if(angle<pi_45-pi){
                angles.push_back(angle+pi_360);

            }
        }
        sort(angles.begin(),angles.end());

     //   cout<<"\n";
        for(auto it=angles.begin();it!=angles.begin()+n;++it){
        //    cout<<*it<<" ";
            auto bound=upper_bound(it,angles.end(),*it+(pi_45+epsilon));
            int curr=distance(it,bound);
            if(curr>result){
                result=curr;
            }
        }
        return result;
    }
};
