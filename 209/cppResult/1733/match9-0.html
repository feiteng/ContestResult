#define PI 3.14159265358979

class Solution {
public:
    int visiblePoints(vector<vector<int>>& pts, int theta, vector<int>& pos) {
        int n = pts.size();
        vector<double> angles;
        
        int res = 0;
        for (int i = 0; i < n; i++) {
            if (pts[i][0] == pos[0] && pts[i][1] == pos[1]) {
                res += 1;
                continue;
            }
            
            if (pts[i][0] == pos[0]) {
                angles.push_back((pos[1] > pts[i][1]) ? -90.0 : 90.0);
                continue;
            }
            
            double tan = ((pts[i][1] - pos[1]) * 1.0) / (pts[i][0] - pos[0]);
            double ang = atan(tan);
            if (pts[i][0] < pos[0])
                ang += PI;
            
            ang = (ang * 180) / PI;
            
            angles.push_back(ang);
        }
        
        // for (auto x: angles)
        //     cout << x << " ";
        // cout << endl;
        
        sort(angles.begin(), angles.end());
        for (int i = 0; i < n; i++)
            angles.push_back(angles[i] + 360.0);
        
        // for (auto x: angles)
        //     cout << x << " ";
        // cout << endl;
        
        int ans = 0;
        for (int i = 0; i < 2*n; i++) {
            double a = angles[i];
            int j = upper_bound(angles.begin()+i, angles.end(), angles[i]+theta) - angles.begin();
            
            ans = max(ans, j-i);
        }
        
        return (ans + res);
        
    }
};
