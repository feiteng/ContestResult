class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<pair<double,pair<int,int>>>v;
        vector<double>v2;

        int i,j,n=points.size();
        int addall=0;
        for(i=0;i<n;i++)
        {
            double x=points[i][0]-location[0];
            double y=points[i][1]-location[1];
            if(points[i][0]==location[0] && points[i][1]==location[1])
            {
                addall+=1;
                continue;
            }
            double xx = x, yy = y, result,ang; 
            result = atan2(yy, xx); 
            ang=result * 180 / 3.141592 ;
            if(ang<0)
            {
                ang=360+ang;
            }
            
            v.push_back({ang,{x,y}});
            v2.push_back(ang);
        }
        // sort(v.begin(),v.end());
        vector<double>v3;

        for(i=0;i<v2.size();i++)
        {
            v3.push_back(v2[i]);
            v3.push_back(v2[i]+360);
        }
        sort(v3.begin(),v3.end());
        int mx=0;
        for(i=0;i<v3.size();i++)
        {
            // cout<<v3[i]<<" ";
            double prev2=v3[i]-angle;
            int pos=lower_bound(v3.begin(),v3.end(),prev2)-v3.begin();
            int cnt=i-pos+1;
            mx=max(mx,cnt+addall);
        }
        return mx;
        
    }
};
