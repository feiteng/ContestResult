class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<double> angleOfPoints;
        int result = 0;
        for (int i = 0; i < points.size(); i++) {
            if (points[i][0] == location[0] && points[i][1] == location[1]) {
                result++;
            } else {
                angleOfPoints.push_back(calcAngle(points[i], location));
            }
        }
        sort(angleOfPoints.begin(), angleOfPoints.end());
        
        int n = angleOfPoints.size();
        
        for (int i = 0; i < n; i++) {
            angleOfPoints.push_back(angleOfPoints[i] + 360);
        }
        
        int maxLen = 0;
        int index = 0;
        int startIndex = 0;
        while (index < angleOfPoints.size()) {
            while (index < angleOfPoints.size() && (angleOfPoints[index] - angleOfPoints[startIndex] <= angle)) {
                index++;
            }
            maxLen = max(maxLen, index - startIndex);
            while (index < angleOfPoints.size() && startIndex < index && (angleOfPoints[index] - angleOfPoints[startIndex] > angle)) {
                startIndex++;
            }
        }

        maxLen = min(maxLen, n);
        result += maxLen;
        
        return result;
    }
    
    double calcAngle(vector<int> points, vector<int>& location) {
        int relativeY = points[0] - location[0];
        int relativeX = points[1] - location[1];
        
        return atan2(relativeY, relativeX) / M_PI * 180;
    }
};
