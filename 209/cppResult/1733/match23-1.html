class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        long double ep = 1e-7;
        long double pi = 3.141592653589793238L;
        vector <long double> v;

        int coinc = 0;
        for (int i = 0; i < points.size(); i ++)
        {
            int dx = points[i][0] - location[0];
            int dy = points[i][1] - location[1];
            if (dx == 0 && dy == 0) coinc ++;
            else v.push_back(atan2(dx, dy));
        }
        long double ang = angle * pi / 180.0;
        sort(v.begin(), v.end());
        int sz = v.size();
        for (int i = 0; i < sz; i ++) v.push_back(v[i] + pi * 2);
        int R = 0;

        int ans = 0;
        for (int L = 0; L < sz; L ++)
        {
            while (R < sz * 2 && v[R] <= v[L] + ang + ep) R ++;
            ans = max(ans, R - L);
        }
        return ans + coinc;
    }
};
