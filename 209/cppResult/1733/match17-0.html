class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<float> angles;
        int dot, det;
        float ang;
        int sameCount = 0;
        for(vector<int>& p : points){
            if(p[0] == location[0] && p[1] == location[1]){
                sameCount++;
                continue;
            }
            dot = p[0]-location[0];
            det = p[1]-location[1];
            ang = atan2(det, dot) * 180.0 / 3.14159265358979323846;
            angles.push_back(ang);
        }
        sort(angles.begin(), angles.end());
        for(int i=0; i<points.size()-sameCount; i++){
            angles.push_back(360+angles[i]);
        }
        int result = 0;
        int j = 0;
        for(int i=0; i<angles.size(); i++){
            while(j < angles.size() && angles[j] <= angles[i]+angle){
                j++;
            }
            result = max(result, j-i);
        }
        return result + sameCount;
    }
};
