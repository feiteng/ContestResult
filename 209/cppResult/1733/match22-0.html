class Solution {
public:
    int zz=0;
    int yy=1;
    vector<pair<double,pair<int,int>>>v;
    
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        
        vector<double>v2;
        vector<double>v3;
        
        int i,j;
        int n=points.size();
        int ad=zz;
        for(i=0;i<n;i++)
        {
            double x=points[i][zz]-location[zz];
            double y=points[i][yy]-location[yy];
            if(points[i][zz]==location[zz] && points[i][yy]==location[yy])
            {
                ad=ad+yy;
                continue;
            }
            double xx = x, yy = y;
            double result;
            double gg; 
            result = atan2(yy, xx); 
            gg=result * 180 / 3.141592 ;
            if(gg<zz) gg+=360;
            
            v.push_back({gg,make_pair(x,y)});
            v2.push_back(gg);
        }
        
        for(i=0;i<v2.size();i++)
        {
            v3.push_back(v2[i]);
            v3.push_back(v2[i]+359+yy);
        }
        sort(v3.begin(),v3.end());
        int ans=zz;
        for(i=0;i<v3.size();i++)
        {
    
            double prev2=v3[i]-angle;
            int pos=lower_bound(v3.begin(),v3.end(),prev2)-v3.begin();
            ans=max(ans,i-pos+yy+ad);
        }
        return ans;
        
    }
};
