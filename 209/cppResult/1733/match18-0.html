class Solution {
public:
    double eps = 1e-18;
    int visiblePoints(vector<vector<int>>& p, int ang, vector<int>& l) {
        int n = p.size();
        vector<double> a;
        int origin = 0;
        for (int i = 0; i < n; i++) {
            if (p[i][1] -l[1] ==0 && p[i][1] -l[1] == p[i][0]-l[0]) {
                origin++;
            } else {
                double a_i = atan2(p[i][1]-l[1], p[i][0]-l[0]);
                a.push_back(a_i);
                a.push_back(a_i+acos(-1)*2);
            }
        }
        sort(a.begin(), a.end());
        double angle = ang/180.0 * acos(-1);
        int lp = 0;
        int ans = 0;
        for (int i = 0; i < a.size(); i++) {
            while (a[i] - a[lp]  - eps > angle)
                lp++;
            ans = max(ans, i-lp+1);
        }
        return ans + origin;
    }
};
