class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location){
        int norig=0;
        vector<double> a;
        for(auto &p:points){

            int x=p[0]-location[0],y=p[1]-location[1];
            if(x==0&&y==0)norig++;
            else a.push_back(atan2(x,y)/M_PI*180);
        }
        sort(a.begin(),a.end());
        int n=a.size();
        for(int i=0;i<n;i++)
            a.push_back(a[i]+360);
        double limit=double(angle)+1e-9;

        int j=0,ans=0;
        for(int i=0;i<n;i++){
            while(j<2*n&&a[j]-a[i]<limit)j++;
            ans=max(ans,j-i);
        }
        return ans+norig;
    }
};
