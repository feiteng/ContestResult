typedef long long ll;
typedef double db;
const db pi=3.1415926535898;
const db dmp=1e-7;
class Solution {
public:
    struct V{
        db x,y;
    };
    db dot(V v1,V v2){return v1.x*v2.x+v1.y*v2.y;}
    db det(V v1,V v2){return v1.x*v2.y-v1.y*v2.x;}
    db rad(V p1,V p2){
        return atan2l(det(p1,p2),dot(p1,p2));
    }
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        vector<db>a;
        V p1;
        p1.x=1,p1.y=0;
        int ex=0;
        for(int i=0;i<points.size();i++){
            V p2;
            p2.x=points[i][0]-location[0];
            p2.y=points[i][1]-location[1];
            if(p2.x==0&&p2.y==0){
                ex++;
                continue;
            }
            a.push_back(rad(p1,p2));
        }
        int len=a.size();
        for(int i=0;i<len;i++){
            if(a[i]<0) a.push_back(a[i]+2*pi);
            //a.push_back(a[i]+2*pi);
           // if(a[i]<0) a.push_back(a[i]+p)
            //cout<<a[i]<<endl;
        }
        sort(a.begin(),a.end());
        
        db aa=angle*pi/180;
        int mx=0;
        int len2=a.size();
        for(int i=0;i<len2;i++){
           // cout<<a[i]<<endl;
            int p=upper_bound(a.begin(),a.end(),a[i]+aa+dmp)-a.begin();
            mx=max(mx,p-i);
        }
        return mx+ex;
    }
};
