class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& loc) {
        const double pi= M_PI, pi_360=M_PI*2.0;
        double pi_ang = pi_360 * angle / 360.0;
        const double eps = 1e-10;
        int l = points.size(), res = 0;
        vector<double> angles(l * 2);
        int base = 0, p = 0;
        for(int i = 0; i < l; i++){
            if (points[i][1] - loc[1] == 0 && points[i][0] - loc[0] == 0) {
                base++;
                continue;
            }
            double cur = atan2(points[i][1] - loc[1], points[i][0] - loc[0]);
            angles[p++] = cur;
            if(cur < pi_ang - pi){
                angles[p++] = (cur + pi_360);
            }
        }
        sort(angles.begin(), angles.begin() + p);
        for(auto it = angles.begin(); it != angles.begin() + p; ++it){
            auto up = upper_bound(it, angles.begin() + p, *it + (pi_ang + eps));
            int curr = distance(it, up);
            if(curr > res){
                res = curr;
            }
        }
        return res + base;
        
    }
};
