class Solution {
public:
    int visiblePoints(vector<vector<int>>& points, int angle, vector<int>& location) {
        int n = points.size();
        vector<double> ang;
        int num = 0;

        for(int i = 0; i < n; i++) {
            double x = points[i][0]-location[0], y = points[i][1]-location[1];
            if(points[i][0] == location[0] && points[i][1] == location[1]) {
                num++;
            } else {
                double now = atan2(y, x)*180/M_PI;
                ang.push_back(now);

            }
        }
        sort(begin(ang), end(ang));
        int k = ang.size();
        for(int i = 0; i < k; i++) {
            ang.push_back(ang[i]+360);
        }
        int ans = 0;
        for(int i = 0; i < k; i++) {
            double to = ang[i]+angle;
            int now = upper_bound(begin(ang), end(ang), to)-begin(ang)-i;
            ans = max(ans, now);
        }
        return ans+num;
    }
};
