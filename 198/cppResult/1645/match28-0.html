class Solution {

        if(r < l) return -1000000000;

        for(int i = l + 1; i <= r; i++){
            ans = ans & arr[i];

    }

        int ans = INT_MAX;
        for(int i = 0; i < arr.size(); i++){
            for(int j = i; j < arr.size(); j++){
                if(abs(fun(arr, i, j) - target) == 0) return 0;

            }
        }

    }

class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int bt = arr[0], ans = INT_MAX;
        sort(arr.begin(), arr.end());
        for (auto &v : arr) {
            bt &= v;
            ans = min(ans, abs(v - target));
            ans = min(ans, abs(bt - target));
        }
        // cout << bt << "\n";
        return min(ans, abs(bt - target));
    }
};
