class Solution {
public:
    int closestToTarget(vector<int>& arr, int t) {
        int ans = INT_MAX,n=arr.size(); 
        set<int> s;
        for(int x:arr)s.insert(x);
        if(s.size()==1){
            return abs(arr[0]-t);
        }

        for (int i = 0; i < n; i++) { 
            int temp = arr[i]; 
            for (int j = i; j < n; j++) { 
                temp &= arr[j]; 
                ans = min(ans, abs(t - temp)); 
                if (temp <= t||abs(t-temp)>ans) 
                    break; 
            } 
            
        } 
        return ans; 
    }
};
