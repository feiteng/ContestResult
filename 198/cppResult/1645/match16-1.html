class Solution {
public:
    int closestToTarget(vector<int>& b, int target) {
        vector<int> a;
        for (int i = 0; i < b.size(); ++i)
            if (i == 0 || b[i] != b[i-1])
                a.push_back(b[i]);
        

        int n = a.size(), ans = 1e9 + target;
        
        for (int i = 0; i < n; ++i) {
            int cur = a[i];
            ans = min(ans, abs(cur - target));
            for (int j = i+1; j < n; ++j) {
                cur &= a[j];
                ans = min(ans, abs(cur - target));
                if (cur < target)
                    break;
            }
        }
        return ans;
    }
};
