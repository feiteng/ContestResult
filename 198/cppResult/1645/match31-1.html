class Solution {
public:
     int closestToTarget(vector<int>& arr, int target) {
        sort(arr.begin(), arr.end());
        int bt = arr[0], ans = INT_MAX;
        for (auto &v : arr) {
            bt &= v;
            ans = min(ans, min(abs(bt - target),abs(v - target)));
        }
        return min(abs(bt - target) , ans);
    }
};
