class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int len = arr.size();
        int ans = 1000000000;
        for(int i = 0; i < len; ++i) {
            if(arr[i] <= target) {
                ans = min(ans, target - arr[i]);
            } else {
                int t = arr[i];
                ans = min(ans, t - target);
                for(int j = i + 1; j < len; ++j) {
                    t &= arr[j];
                    if(t <= target) {
                        ans = min(ans, target - t);
                        break;
                    } else {
                        ans = min(ans, t - target);
                    }
                    if(t == arr[j]) break;
                }
            }
            if(ans == 0) break;
        }
        return ans;
    }
};
