class Solution {
public:
int closestToTarget(vector<int> &arr, int target)
{
    int ans = 100000000;
    int mm = 0;
    for (int i = 0; i < arr.size(); i++)
    {
        int cur = arr[i];
        for (int j = i; j < arr.size(); j++)
        {
            cur = cur & arr[j];
            ans = min(ans, abs(cur - target));
            if (cur <= mm)
            {
                break;
            }
            else if (cur <= target)
            {
                mm = cur;
                break;
            }
        }
        if (cur >= target)
            break;
    }
    return ans;
}
};
