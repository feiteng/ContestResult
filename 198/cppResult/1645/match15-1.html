class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {

        int res=INT_MAX;
        arr=helper(arr);
        for (int i=0;i<arr.size();i++)
        {
            int cur=arr[i];
            res=min(res,abs(cur-target));
            for (int j=i+1;j<arr.size();j++)
            {
                cur=cur & arr[j];
                res=min(res,abs(cur-target));
                if (cur<target)
                    break;
            }
        }
        return res;
    }
    vector<int> helper(vector<int> &arr)
    {
        vector<int> tmp;
        for (int cur:arr)
            if (tmp.empty() || cur!=tmp.back())
                tmp.push_back(cur);
        return tmp;
    }
};
