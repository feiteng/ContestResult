class Solution {
public:
    int closestToTarget(vector<int>& a, int k) {
        int n = a.size();
        int v = a[0];
		int md = abs(k - v);
		for (int i = 0; i < n; i++)
		{
			v &= a[i];
			md = min(md, abs(k - v));
			if (v <= k)
			{
				// will decrease more
				// so make it equal to next element
				if (i + 1 < n)
					v = a[i + 1];
			}
		}
        return md;
    }
};
