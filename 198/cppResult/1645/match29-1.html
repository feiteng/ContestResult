class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int n = arr.size();
        // cout<<n<<endl;

        vector<vector<int>> dp(n, vector<int>(17, 0));
        for (int j=0; j<17; j++) {
            for (int i=0; i<n; i++) {
                if (j==0) {
                    dp[i][j] = arr[i];
                }
                else {

                    int k = 1 << (j-1);
                    if (i + k < n) {
                        dp[i][j] = dp[i][j-1] & dp[i+k][j-1];
                    }
                }
            }
        }
        
        int ans = 1e9;
        for (int i=0; i<n; i++) {
            int current = dp[i][0];
            int next = i+1;
            while (1) {
                ans = min(ans, abs(target - current));
                if (current <= target) break;
                if (next >= n) break;

                int k=0;
                while (k + 1 < 17 && (current & dp[next][k+1]) >= target) {
                    k++;
                }
                current &= dp[next][k];
                next += 1<<k;
            }
        }
        
        return ans;
    }
};
