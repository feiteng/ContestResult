class Solution {

public:
    int closestToTarget(vector<int>& arr, int target) {
        sort(arr.begin(), arr.end());
        int mindiff = INT_MAX;
        for(int i = 0; i < arr.size(); ++i)
        {
            if(i > 0 && arr[i] == arr[i-1]) continue;
            for(int j = i; j < arr.size(); ++j)
            {
                if(j > i && arr[j] == arr[j-1]) continue;
                int func_value = func(arr, i, j);
                if(abs(func_value - target) < mindiff)
                {
                    mindiff = abs(func_value - target);
                }
            }
        }
        return mindiff;
    }

    int func(vector<int>& arr, int l, int r)
    {
        if(r < l) return -1000000000;
        int ans = arr[l];
        for(int i = l + 1; i <= r; ++i)
        {
            ans = ans & arr[i];
        }
        return ans;
    }
};
