class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int f=*arr.begin();
        int ans=INT_MAX;
        sort(arr.begin(),arr.end());
        for(auto e:arr){
            f&=e;
            ans=min({ans,abs(e-target),abs(f-target)});
        }
        return min(ans,abs(f-target));
    }
};
