class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int ans = abs(target - arr[0]);
        int n = arr.size();
        for (int num : arr) {
            if (num <= target) ans = min(ans, target - num);
        }
        if (ans == 0) return ans;
        // cout << ans << endl;
        // two pointer
        int i = 0, j = 0;
        while (i < n && arr[i] < target) i ++;
        
        int now;
        while (i < n) {
            now = arr[i];
            ans = min(ans, now - target);
            // cout << arr[i] << ',' << arr[j] << endl;
            for (j = i + 1; j < n; j ++) {
                if (arr[j] < target || now < target) break;
                now &= arr[j];
                ans = min(ans, abs(target - now));
            }
            // find next bigger interval
            i = j;
            while (i < n && arr[i] < target) i ++;
        }
        
        return ans;
    }
};
