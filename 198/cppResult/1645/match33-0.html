    int closestToTarget(vector<int>& arr, int target) {
        int n = arr.size();
        int best = target + 1000000000;
        for (int i = 0; i < n; ++i) {
            int j = i;
            while (j < n) {
                int next_best = abs(myfunc(arr, i, j) - target);
                if (next_best < best) {
                    best = next_best;
                } else {
                    break;
                }
                j++;
            }
        }
        return best;
    }
    
    int myfunc(vector<int>& arr, int l, int r) {
        if (r < l) {
            return -1000000000;
        }
        int ans = arr[l];
        for (int i = l+1; i <= r; ++i) {
            ans = ans & arr[i];
        }
        return ans;
    }
};
