class Solution {
public:
    int closestToTarget(vector<int>& arr, int k) {
        int ans = INT_MAX;  
        int n = arr.size();
        vector<int > dif;
        int mn = INT_MAX;
        for(int i=0; i<n; i++){
            int v = abs(k-arr[i]);
            if(v < mn){
                mn = v;
                dif.clear();
                dif.push_back(i);
            }else if(v == mn && arr[dif.back()] != arr[i]){
                dif.push_back(i);
            }
        }
        
        int f = 0;
        for(int i=0; i<dif.size(); i++){
            if(dif[i] == 0){
                f = 1;
                break;
            }
        }
        if(f == 0){
            dif.push_back(0);
        }
        
        // Check all possible sub-arrays 

        for (int i = 0; i < dif.size(); i++) { 
            int X = arr[dif[i]]; 
            // cout<<X<<"\n";
            for (int j = dif[i]; j < n; j++) { 
                X &= arr[j]; 

                // Find the overall minimum 
                ans = min(ans, abs(k - X)); 

                // No need to perform more AND operations 
                // as |k - X| will increase 
                if (X <= k) 
                    break; 
            } 
        } 
        return ans;
    }
};
