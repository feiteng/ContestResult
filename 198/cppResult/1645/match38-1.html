struct zkw {
    vector<int> tree;
    int M;
    int N;
    void build(int n, vector<int> &arr) {
        N = n;
        for(M=1; n>=M-1; M<<=1);
        tree = vector<int>(2*M+1, 0xffffff);
        for (int i=1; i<=n; i++) {
            tree[i+M] = arr[i-1];
        }
        for (int i=M-1; i>=1; i--) {
            tree[i] = tree[i<<1] & tree[i<<1|1];
        }
        
    }
    
    int query(int s, int t) {
        // printf("q [%d, %d] :", s, t);
        s--; t++;
        int ans = 0xffffff;
        for(s+=M, t+=M; s^t^1; s>>=1, t>>=1) {
            if (~s&1)
                ans &= tree[s^1];
            if (t&1)
                ans &= tree[t^1];
        }
        // printf(" %d\n", ans);
        return ans;
    }
    
    void find(int x, int target, int &ans) {
        int first = x;
        int count = N-x+1;
        while (count) {
            int step = count/2;
            int mid = first + step;
            int ta = query(x, mid);
            ans = std::min(ans, abs(ta - target));
            if (ta >= target) {
                first = mid+1;
                count -= step+1;
            } else {
                count = step;
            }
        }
        if (x<=first && first <=N)
            ans = std::min(ans, abs(query(x, first) - target));
        if (x<=first-1 && first-1<=N)
            ans = std::min(ans, abs(query(x, first-1) - target));
        return;
    }
    
};


class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int ans = 2100000000;
        int n = arr.size();
        zkw tree;
        tree.build(n, arr);
        for (int i=0; i<n; i++) {
            ans = min(ans, abs(arr[i] - target));
        }
        for (int i=0; i<n; i++) {
            int v = arr[i];
            if (v >= target) {
                tree.find(i+1, target, ans);
                // printf("ans: %d, i+1: %d\n", ans, i+1);
            }
        }
        return ans;
        
    }
};
