class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int res = 2e9, n = arr.size(); 
        vector<int> next(n);
        int i=0;
        while (i < n) {
            int j = i;
            while (i < n && arr[i] == arr[j])
                i++;
            next[j] = i;
        }

        for (int i = 0; i < n; i = next[i]) { 
            int cur = arr[i]; 
            for (int j = i; j < n; j = next[j]) { 
                cur &= arr[j]; 
                res = min(res, abs(target - cur)); 
                if (cur <= target) 
                    break; 
            } 
        } 
        return res;
    }
};
