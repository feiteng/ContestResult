class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        set<int>se,sum;
        int n=arr.size();
        se.insert(arr[0]);
        sum.insert(arr[0]);
        for(int i=1;i<n;i++)
        {
            set<int>temp;
            temp.insert(arr[i]);
            for(auto it:se)
                temp.insert(arr[i]&it);
            for(auto it:temp)
                sum.insert(it);
            se=temp;
        }
        if(n>=2)
        se.insert(-1000000000);
        int ans=0x3f3f3f3f;
        for(auto it:sum)
            ans=min(ans,abs(it-target));
        return ans;

    }
};
