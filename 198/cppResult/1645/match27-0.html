class Solution {
public:
    int func(vector<int>& arr, int l, int r)
    {
        int ans = arr[l];
        cout<<ans<<":"<<l<<":"<<r<<endl;
        for(int i = l+1; i <= r; ++i)
        {
            ans &= arr[i];
            cout<<ans<<" ";
            if(ans == 0)
                return 0;
        }
        return ans;
    }
    int closestToTarget(vector<int>& arr, int target) {
        int res = INT_MAX;
        for(int i =0; i < arr.size(); ++i)
        {
            int ans = arr[i];
            res = min(res, abs(ans-target));
            for(int j = i+1; j < arr.size(); ++j)
            {
                ans &= arr[j];
                res = min(res, abs(ans-target));
                if(ans == arr[j])
                    break;
            }
        }
        return res;
    }
};
