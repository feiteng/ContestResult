class Solution {
public:
    int closestToTarget(vector<int>& ar, int target) {
        map<int,int>mp;
        vector<int>arr;
        for(int i=0;i<ar.size();i++){
            if(mp[ar[i]]==0){
                arr.push_back(ar[i]);
                }
            mp[ar[i]]++;
        }

        int ans = INT_MAX; 
        //cout<<arr.size()<<endl;
        for (int i = 0; i < arr.size(); i++) { 
            int X = arr[i]; 
            for (int j = i; j < arr.size(); j++) { 
                X &= arr[j]; 
                ans = min(ans, abs(target - X));
                if (X <= target) 
                    break; 
            } 
        } 
        return ans; 
    }
};
