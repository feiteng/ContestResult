class Solution {
public:
    int closestToTarget(vector<int>& a, int t)
    {
        set<int> s(a.begin(), a.end());
        vector<int> p(s.begin(), s.end());
        int ans = INT_MAX;
        for(int i = 0; i<p.size(); i++)
        {
            int temp = p[i];
            for(int j = i; j<p.size(); j++)
            {
                temp &= p[j];
                ans = min(ans, abs(temp-t));
                if(temp <= t)
                    break;
            }
        }
        return ans;
    }
};
