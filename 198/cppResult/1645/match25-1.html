class Solution {
public:
    vector<int>Q[100005];

    int closestToTarget(vector<int>& arr, int target) {
        int n = arr.size();
        int ans = 1e9;
        for(int i = 1; i <= n; i++) {
            int now = arr[i - 1] , res;
            Q[i].push_back(now);
            for(int j = 0; j < Q[i - 1].size() ; j++) {
                res = arr[i - 1] & Q[i - 1][j];
                if (res == now) continue;
                Q[i].push_back(res);
                now = res;
            }
            for (auto & g : Q[i]) {
                ans = min (ans , abs(target - g));
            }
        }
        return ans;
    }
};
