class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        vector<int> v;
        v.push_back(arr[0]);

        int res=abs(arr[0]-target);
        for (int i=1; i<arr.size(); ++i) {
            v.push_back(arr[i]);
            for (int j=v.size()-2; ~j; --j) {
                if ((v[j]&arr[i])==v[j]) {
                    break;
                }
                v[j]=v[j]&arr[i];

            }
            int x=lower_bound(v.begin(), v.end(), target)-v.begin();
            for (int j=x-1; j<=x+1; ++j) {
                if (j>=0&&j<v.size()) {
                    res=min(res, abs(v[j]-target));
                }
            }
        }
        return res;
    }
};
