class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int m = INT_MAX;
        int allAnd = INT_MAX;
        for (int a : arr) {
            allAnd &= a;
            m = min(abs(target-a), m);
            m = min(abs(allAnd - target), m);
        }
        return m;
    }
};
