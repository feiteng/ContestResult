class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int i = 0;
        int minVal = INT_MAX, ans = arr[0];
        for(int i = 0; i < arr.size(); i++)
        {
            minVal = min(minVal, abs(arr[i]-target));
            ans = ans & arr[i];
            minVal = min(minVal, abs(ans-target));
            if(ans == target) break;
            if(ans < target) ans = 1;
        }
        return minVal;
    }
};
