class Solution {
public:
    int closestToTarget(vector<int>& a, int t) {
        const int n = a.size();
        vector<array<int, 18>> d(n);
        for (int j = 0; j < 18; ++j) {
            for (int i = 0; i < n; ++i) {
                if (j == 0) {
                    d[i][j] = a[i];
                } else{
                    d[i][j] = d[i][j - 1];
                    int k = i + (1 << (j - 1));
                    if (k < n) {
                        d[i][j] &= d[k][j - 1];
                    }
                }
            }
        }
        int ans = INT_MAX;
        for (int i = 0; i < n; ++i) {
            ans = min(ans, abs(a[i] - t));
            int x = i, v = a[i];
            for (int j = 17; j >= 0 && x < n; --j) {
                ans = min(ans, abs((v & d[x][j]) - t));
                if ((v & d[x][j]) >= t) {
                    v &= d[x][j];
                    x += 1 << j;
                }
            }
        }
        return ans;
    }
};
