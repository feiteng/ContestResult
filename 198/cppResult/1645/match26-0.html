class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int n = arr.size();
        vector<int> dp(n);
        dp[0] = arr[0];
        int minval = abs(dp[0] - target);
        for(int i = 1; i < n; i++) {
            int a = dp[i - 1] & arr[i];
            //printf("%d\n", a);
            if(abs(a - target) < abs(arr[i] - target)) {
                dp[i] = a;
                minval = min(abs(a - target), minval);
            }
            else if(abs(a - target) == abs(arr[i] - target)) {
                dp[i] = min(a, arr[i]);
                minval = min(abs(a - target), minval);
            }
            else {
                dp[i] = arr[i];
                minval = min(abs(arr[i] - target), minval);
            }
            //printf("%d\n", dp[i]);
        }
        return minval;
    }
};
