class Solution {
public:
    int closestToTarget(vector<int>& a, int target) {
        int ans = INT_MAX;

        set<int> s(a.begin(), a.end());
        vector<int> arr(s.begin(), s.end());
        int n = arr.size();
        for(int i=0;i<n;i++)
        {
            int val = arr[i];
            for(int j=i;j<n;j++)
            {
                val &= arr[j];
                ans = min(ans, abs(target-val));
                if(ans == 0)
                    return ans;
                if(val < target)
                    break;
            }
        }
        return ans;
    }
};
