class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        int val = arr[0], res = 0x7f7f7f7f;
        sort(arr.begin(), arr.end());
        for (auto &v : arr) {
            val = val & v;
            res = min(res, abs(v - target));
            res = min(res, abs(val - target));
        }
        return min(res, abs(val - target));
    }
};
