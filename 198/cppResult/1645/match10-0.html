class Solution {
public:
    int closestToTarget(vector<int>& arr, int target) {
        
        int n=arr.size();
        
        int temp=arr[0],ans=INT_MAX;
        for(int i=0;i<n;i++)
        {
            if(temp<target) temp=arr[i];
            temp=temp&arr[i];
            ans=min(ans,abs(target-temp));
        }
        return ans;
        
    }
};
