class Solution {
public:
    int numWaterBottles(int numBottles, int numExchange) {
        int res=0,remain=0;
        while (numBottles>0)
        {
            res+=numBottles;
            remain+=numBottles;
            numBottles=remain/numExchange;
            remain=remain%numExchange;
        }
        return res;
    }
};
