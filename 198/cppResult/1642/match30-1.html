class Solution {
public:
    int numWaterBottles(int f, int e) {
        int ret = 0;
        int currf = f;
        int curre = 0;
        while(currf) {
          ret++;
          currf--;
          curre++;
          if(curre == e) {
            curre = 0;
            currf++;
          }
        }
        return ret;
    }
};
