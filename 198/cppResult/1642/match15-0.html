class Solution {
public:
    int numWaterBottles(int numBottles, int numExchange) {
        int ans = 0, res = 0;
        while(numBottles) {
            ans += numBottles;
            numBottles += res;
            res = numBottles % numExchange;
            numBottles /= numExchange;
        }
        return ans;
    }
};
