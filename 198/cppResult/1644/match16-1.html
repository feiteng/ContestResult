struct seg{
    int s, t, v;
};
bool cmp(seg a,seg b)
{
	return a.t < b.t;
}

class Solution {
public:
    vector<string> maxNumOfSubstrings(string s) {
        vector<seg> segs;
        vector<int> ans,mx;
        int l = s.size();

        int first[26], last[26];
        for(int i=0;i<26;i++){
            first[i] = l + 1;
            last[i] = -1;
        }
        for(int i=0;i<l;i++){
            int ch = s[i]-'a';
            first[ch] = min(first[ch], i);
            last[ch] = max(last[ch], i);
        }
        for(int i=0;i<26;i++){
            if(last[i] == -1) continue;
            while(true){
                int tf = first[i], tl = last[i];
                for(int j=tf;j<=tl;j++){
                    int ch = s[j]-'a';
                    first[i] = min(first[ch], first[i]);
                    last[i] = max(last[ch], last[i]);
                }
                if(first[i] == tf && last[i]== tl)
                    break;
            }
        }
        for(int i=0;i<26;i++){
            if(last[i] == -1) continue;
            segs.push_back((seg){first[i],last[i],10000000 - (last[i]-first[i]+1)});
            mx.push_back(0);
            ans.push_back(-1);
        }
        sort(segs.begin(),segs.end(),cmp);
        mx[0] = segs[0].v;
        for(int i=1;i<segs.size();i++){
            printf("%d %d\n", segs[i].s, segs[i].t);
            mx[i]=segs[i].v;
            ans[i]=-1;
            for(int j=i-1;j>=0;j--){
                if(segs[j].t < segs[i].s){
                    if((segs[i].v + mx[j]) > mx[i]){
                        mx[i] = segs[i].v + mx[j];
                        ans[i] = j;
                    }
                }
            }
        }

        int val = -1, pos = -1;
        for(int i=0;i<segs.size();i++){
            if(mx[i]>val){
                val = mx[i];
                pos = i;
            }
        }
        vector<string> result;
        for(int i=pos;i!=-1;i=ans[i]){
            seg &g = segs[i];
            result.push_back(s.substr(g.s, g.t-g.s+1));
        }
        return result;
    }
};
