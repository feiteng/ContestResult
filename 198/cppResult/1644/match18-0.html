class Solution {
public:
    int L[26],R[26],len[26];
    vector<int>e[26],e2[26];
    bool vis[26];
    typedef pair<int,int> pii;
    int cnt[100010][26];
    bool out[26];
    int getsum(int L,int R,int ch) {
        if (L == 0) return cnt[R][ch];
        return cnt[R][ch] - cnt[L - 1][ch];
    }
    bool canvis[26][26];
    void dfs(int x,int rt) {
        canvis[rt][x] = true;
        for (auto t:e[x]) {
            if (!canvis[rt][t]) {
                dfs(t, rt);
            }
        }
    }
    int fa[26];
    int find(int x) {
       return x == fa[x] ? x : fa[x] = find(fa[x]); 
    }
    vector<string> maxNumOfSubstrings(string s) {
        int n = s.length();
        for (int i=0;i<26;i++) L[i] = R[i] = -1;
        for (int i=0;i<n;i++) {
            for (int j=0;j<26;j++) cnt[i][j] = 0;
        }
        for (int i=0;i<n;i++) {
            if (i > 0) {
                for (int j=0;j<26;j++) {
                    cnt[i][j] = cnt[i-1][j];
                }
            }
            int ch = s[i] - 'a';
            cnt[i][ch]++;
            if (L[ch] == -1) L[ch] = i;
            R[ch] = max(R[ch], i);
        }
        for (int i=0;i<26;i++) {
            len[i] = R[i] - L[i] + 1;
            for (int j=0;j<26;j++) {
                if (L[i] != -1 && getsum(L[i], R[i], j)) {
                    e[i].push_back(j);
                }
            }
        }
        vector<string>ans;
        memset(canvis,false,sizeof(canvis));
        for (int i=0;i<26;i++) {
            fa[i] = i;
            if (L[i] == -1) continue;
            dfs(i,i);
        }
        for (int i=0;i<26;i++) {
            for (int j=i+1;j<26;j++) {
                if (canvis[i][j] && canvis[j][i]) {
                    fa[find(i)] = find(j);
                }
            }
        }
        memset(out,false,sizeof(out));
        for (int i=0;i<26;i++) {
            for (auto t:e[i]) {
                if (find(t) != find(i)) out[find(i)] = true;
            }
        }
        for (int i=0;i<26;i++) {
            if (!out[i] && L[i] != -1 && i == find(i)) {
                int minL = 1e9, maxR = -1e9;
                for (int j=0;j<26;j++) {
                    if (find(j) == i) {
                        minL = min(minL, L[j]);
                        maxR = max(maxR, R[j]);
                    }
                }
                ans.push_back(s.substr(minL, maxR - minL + 1));
            }
        }
        return ans;
    }
};
