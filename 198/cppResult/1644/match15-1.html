class Solution{
public:
    vector<string> maxNumOfSubstrings(string s) {
        vector<pair<int, int>> r(26);
        vector<bool> select(26, true);

        for (int i = 0; i < 26; i++)
        {
            r[i].first = -1;
            r[i].second = -1;
        }

        for (int i = 0; i < s.size(); i++)
        {
            int index = s[i] - 'a';
            if (r[index].first == -1)
            {
                r[index].first = i;
                r[index].second = i;
            }
            else
            {
                r[index].second = i;
            }
        }
        for (int i = 0; i < s.size(); i++)
        {

            int index = s[i] - 'a';
            for (int j = 0; j < 26; j++)
            {
                if (i > r[j].first && i < r[j].second)
                {
                    r[j].first = min(r[index].first, r[j].first);
                    r[j].second = max(r[index].second, r[j].second);
                }
            }
        }
        sort(r.begin(), r.end());

        vector<pair<int, int>> ans;
        for (int i = 0; i < r.size(); i++)
        {
            if (r[i].first == -1)
            {
                continue;
            }
            if (!ans.empty())
            {
                auto& pre = ans.back();
                if (pre.first <= r[i].first && pre.second >= r[i].second)
                {
                    ans.pop_back();
                }
            }
            ans.push_back(r[i]);
        }

        vector<string> res;
        for (auto& p : ans)
        {
            res.push_back(s.substr(p.first, p.second - p.first + 1));
        }
        return res;

    }
};
