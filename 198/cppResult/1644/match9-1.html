class Solution {
static bool cmp(const pair<string,int> &a,const pair<string,int> &b)
{
   // if(a.first.length()==b.first.length())
        return a.second+a.first.length()<b.second+b.first.length();
   
}
public:
    vector<string> maxNumOfSubstrings(string s) {
        vector<pair<string,int>> ans;
        vector<int> freq(26,0);
        vector<int> first(26,-1);
        vector<int> last(26,-1);
        int n=s.length();
        for(int i=0;i<s.length();i++)
        {
            freq[s[i]-'a']++;
            if(first[s[i]-'a']==-1)
                first[s[i]-'a']=i;
            last[s[i]-'a']=i;
        }
        
        for(int i=0;i<26;i++)
        {
            if(first[i]!=-1)
            {
                bool f1=0;
                vector<int> f=freq;
                int maxi=last[i];
                string temp;

                for(int j=first[i];j<=maxi;j++)
                {
                    if(first[s[j]-'a']<first[i])
                    {
                        f1=1;
                        break;
                    }
                    temp.push_back(s[j]);
                    maxi=max(maxi,last[s[j]-'a']);
                }
                if(f1==0)
                ans.push_back({temp,first[i]});
            }
        }
        
        sort(ans.begin(),ans.end(),cmp);
        for(int i=0;i<ans.size();i++)

            cout<<ans[i].first<<" ";
        vector<string> res;
        res.push_back(ans[0].first);
        int r1=ans[0].second+ans[0].first.length()-1;
        
        for(int i=1;i<ans.size();i++)
        {
            int l1=ans[i].second;
            int r2=ans[i].second+ans[i].first.length()-1;
            
            if(l1>r1)
            {
                res.push_back(ans[i].first);
                r1=r2;
            }
        }
        return res;        
    }
};
