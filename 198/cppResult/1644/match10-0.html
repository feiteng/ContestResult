class Solution {
public:
    vector<string> maxNumOfSubstrings(string s) {
        
        int len = s.length();
        
        vector<int> cnt(26,0);
        
        vector<vector<int>> pa(26,vector<int>(2,-1));
        
        vector<vector<int>> pre(s.length()+1,vector<int>(26,0));
        
        for(int i = 0; i < s.length(); ++i){
            int ind = s[i]-'a';
            cnt[ind] += 1;
            
            pre[i+1] = pre[i];
            
            pre[i+1][ind] = pre[i+1][ind]+1;
            
            if(pa[ind][0]==-1) {pa[ind][0] = i;pa[ind][1] = i;} else pa[ind][1] = i;
        }
        
        vector<vector<int>> q;
        
        //cout<<"ho"<<endl;
        
        for(int i = 0; i < 26; ++i){
            
            //cout<<"ho1"<<endl;
            for(int j = 0; j < 26; ++j){
            
            int st = pa[i][0]; int en = pa[j][1];
            if(st==-1||en==-1||st>en) continue;
            
            bool flag = true;
            
            for(int i = 0; i < 26; ++i){
                
                int cnt0 = pre[en+1][i]-pre[st][i];
                
                if(cnt0!=0&&cnt0!=cnt[i]){
                    flag = false;
                    break;
                }
            }
            if(flag) q.push_back({en,st});
            }
        }
        
        
        //q.push_back({len-1,0});
        
        sort(q.begin(),q.end(),[](vector<int> a1,vector<int> a2){if(a1[0]!=a2[0]) return a1[0]<a2[0]; else return a1[1]>a2[1];});
        
        vector<string> res;
        
        int ri = -1;
        
        for(int i = 0; i < q.size(); ++i){
            
            int en = q[i][0], st = q[i][1];
            //cout<<en<<" "<<st<<endl;
            if(st>ri) {res.push_back(s.substr(st,en-st+1)); ri = en;} 
        }
        
        return res;
    }
};
