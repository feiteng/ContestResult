vector<pair<string,int>> javab;
vector<int> f(26,0);
vector<int> fi(26,-1);
vector<int> la(26,-1);

bool abc(pair<string,int> a,pair<string,int> b)
{
  
        return a.second+a.first.length()<b.second+b.first.length();
   
}

class Solution {
public:
    vector<string> maxNumOfSubstrings(string s) {
        javab.clear(); 
        for(int i=0;i<26;i++)
         {
             f[i]=0;
             fi[i]=-1;
             la[i]=-1;
         }
        int n=s.length();
        for(int i=0;i<n;i++)
        {
            
            la[s[i]-'a']=i;
            f[s[i]-'a']++;
            if(fi[s[i]-'a']==-1)
                fi[s[i]-'a']=i;
        }
        string s1;
        for(int i=0;i<26;i++)
        {
            if(fi[i]!=-1)
            {   s1.clear();
                
                int mini=la[i];
                int j;
             int o=0;
                for(j=fi[i];j<=mini;j++)
                {
                    if(fi[s[j]-'a']<fi[i])
                    {   o=1;
                        break;
                    }
                    s1.push_back(s[j]);
                    mini=max(mini,la[s[j]-'a']);
                }
                if(o==0 || j==mini+1)
                javab.push_back({s1,fi[i]});
            }
        }
        sort(javab.begin(),javab.end(),abc);
        
        for(int i=0;i<(int)javab.size();i++)
        {
            if(javab[i].first=="")
                break;
        }
        
        vector<string> javab2;
        javab2.push_back(javab[0].first);
        int p1=javab[0].second+javab[0].first.length()-1;
        int cd=0;
        for(int i=1;i<javab.size();i++)
        {
            int o1=javab[i].second;
            int r2=javab[i].second+javab[i].first.length()-1;
            
            if(o1>p1)
            {
                javab2.push_back(javab[i].first);
                p1=r2;
            }
            else
            cd=1;
        }
        return javab2;        
    }
        
};
