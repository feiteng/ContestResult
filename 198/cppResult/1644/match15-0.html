class Solution {
public:
    struct dis{
        int first;
        int last;
        //int gap;
        bool operator < (const dis &b)
        {
            return last<b.last;
        }
    };
    vector<string> maxNumOfSubstrings(string s) {
        int n = s.size();
        vector<string> ans;
        dis d[30];
        for(int i=0;i<30;++i)   {d[i].first = -1;d[i].last=0x3f3f3f3f;}
        for(int i=0;i<n;++i)
        {
            int index = s[i]-'a';
            if(d[index].first==-1)
                d[index].first = i;
            else
                d[index].last = i;
        }
        for(int i=0;i<30;++i)
            if(d[i].last==0x3f3f3f3f&&d[i].first!=-1)
                d[i].last=d[i].first;
        //bool flag = true;
        //while(flag){
        for(int z=0;z<2;++z)
        for(int i=0;i<30;++i)
        {
            //flag = false;
            if(d[i].first!=-1&&d[i].last!=0x3f3f3f3f)
                for(int j=d[i].first+1;j<d[i].last;++j)
                {
                    int p = s[j]-'a';
                    d[i].first=min(d[i].first,d[p].first);
                    d[i].last = max(d[i].last,d[p].last);
                    //flag = true;
                } 
        }
          // for(int i=0;i<n;++i)
            //cout<<d[i].first<<','<<d[i].last<<endl;
        cout<<endl;
        //} //end while;
        sort(d,d+30);
        //for(int i=0;i<n;++i)
           // cout<<d[i].first<<','<<d[i].last<<endl;

        
        int pos = -1;
        for(int i=0;i<30;++i)
        {
            if(d[i].first>pos)
            {
                string tmp = s.substr(d[i].first,d[i].last-d[i].first+1);
                ans.push_back(tmp);
                pos = d[i].last;
            }
        }
        return ans;
    }
};
