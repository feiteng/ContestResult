class Solution {
public:
    int cnt[(int)(1e5)];
    
    vector<int> dfs(int start,vector<bool> &vis,vector<vector<int>> &g,string &labels,vector<int> &res)
    {
        vis[start]=1;
        vector<int> freq(26,0);
        freq[labels[start]-'a']++;
        for(int i=0;i<g[start].size();i++)
        {
            if(!vis[g[start][i]])
            {
                vector<int> temp=dfs(g[start][i],vis,g,labels,res);
                
                for(int j=0;j<26;j++)
                    freq[j]+=temp[j];
            }
        }
        res[start]=freq[labels[start]-'a'];
        return freq;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> adj(n+1);
        for(int i=0;i<edges.size();i++)
        {
             adj[edges[i][0]].push_back(edges[i][1]);
            adj[edges[i][1]].push_back(edges[i][0]);
        }
        vector<int> res(n,0);
        vector<bool> vis(n+1,0);
        dfs(0,vis,adj,labels,res);
        return res;
        
    }   
};
