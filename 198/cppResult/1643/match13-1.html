#define FOR(i, begin, end) for(int i=(begin),i##_end_=(end);i<i##_end_;i++)
#define REP(i, n) FOR(i,0,n)


class Solution {
public:
    vector<int> ret;
    vector<vector<int>> to;
    string str;
    vector<int> dfs(int now, int par)
    {
        vector<int> cnt(26);
        for (auto nxt : to[now]) if (nxt != par)
        {

            vector<int> v = dfs(nxt, now);
            REP(i, 26) cnt[i] += v[i];
        }
        cnt[str[now] - 'a']++;
        ret[now] = cnt[str[now] - 'a'];
        return move(cnt);
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        ret.assign(n, 0);
        to.resize(n);
        for (auto v : edges)
        {
            to[v[1]].emplace_back(v[0]);
            to[v[0]].emplace_back(v[1]);
        }
        str = labels;
        dfs(0, -1);
        return ret;
    }
};
