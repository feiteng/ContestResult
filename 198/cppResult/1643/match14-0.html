class Solution {
public:
  vector<vector<int>> G;
  string labels;
  vector<int> acc;
  vector<int>  dfs(int s, int f) {
    vector<int> ans(26);
    for (auto t: G[s]) if (t != f) {
      auto tmp = dfs(t, s);
      for (int i = 0; i < 26; i++) ans[i] += tmp[i];
    }
    acc[s] = ++ans[labels[s] - 'a'];
    return ans;
  }
  vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
    G.clear();
    G.resize(n);
    this->labels = labels;
    for (auto e: edges) G[e[0]].push_back(e[1]), G[e[1]].push_back(e[0]);
    acc.clear();
    acc.resize(n);
    dfs(0, -1);
    return acc;
  }
};
