class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        unordered_map<int, vector<int>> e;
        for(auto& vec: edges) {
            e[vec[0]].push_back(vec[1]);
            e[vec[1]].push_back(vec[0]);
        }
        vector<int> ans(n, 0);

        auto dfs = [&] (auto& self, int node, int parent) -> vector<int> {
            vector<int> cur(26, 0);
            cur[labels[node]-'a'] += 1;
            for(int child: e[node]) {
                if (child == parent) continue;
                auto sub = self(self, child, node);
                for(int i = 0; i < 26; i++) {
                    cur[i] += sub[i];
                }
            }
            ans[node] = cur[labels[node]-'a'];
            return cur;
        };
        dfs(dfs, 0, -1);
        return ans;
    }
};
