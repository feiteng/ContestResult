class Solution {
public:
    vector<vector<int> > g;
    vector < int > ans;
    
    int cnt[100010][26];
    
    void dfs (int v, int par, string &s) {
        cnt[v][s[v]-'a'] ++;
        for (auto to : g[v])if(to != par) {
            dfs(to, v, s);
            for (int i = 0; i < 26; ++i) cnt[v][i] += cnt[to][i];
        }
        
        ans[v] = cnt[v][s[v]-'a'];
    }
    
    vector<int> countSubTrees(int n, vector<vector<int>>& e, string l) {
        g.resize(n);
        ans.resize(n);
        
        for (auto it : e) {
            g[it[0]].push_back(it[1]);
            g[it[1]].push_back(it[0]);
        }
        
        dfs(0, -1, l);
        return ans;
    }
};
