class Solution {
public:
   vector<int> v[100050];

   vector<int> ans;
   string labels;

   vector<int> dfs(int s, int p){
      vector<int> freq;
      freq.resize(26);
      freq[labels[s]-'a']++;
      for(auto it: v[s]){
         if(it == p)
            continue;

         vector<int> v = dfs(it, s);
         for (int i = 0; i < 26; ++i)
            freq[i] += v[i];
      }
      ans[s] = (freq[labels[s] - 'a']);
      return freq;
   }

    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
         
      for (int i = 0; i < n; ++i)
      {
         v[i].clear();
      }

      for(auto it: edges){
         v[it[0]].push_back(it[1]);
         v[it[1]].push_back(it[0]);
      }

      ans.resize(n);
      this->labels = labels;
      dfs(0,-1);
      return ans;
    }
};
