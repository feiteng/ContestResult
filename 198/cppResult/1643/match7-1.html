class Solution {
public:
    
    vector<int> solve(string &label,vector<vector<int>> &A,int l,int par,vector<int> &ans){
        vector<int> alp(26,0);
        alp[label[l]-'a']++;

        for(int i=0;i<A[l].size();i++){
            if(A[l][i]!=par){
                vector<int> tmp=solve(label,A,A[l][i],l,ans);
                for(int j=0;j<26;j++)alp[j]+=tmp[j];
            }
        }
        ans[l]=alp[label[l]-'a'];
        return alp;
    }
    

    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> A(n);
        for(int i=0;i<edges.size();i++){
            A[edges[i][0]].push_back(edges[i][1]);
            A[edges[i][1]].push_back(edges[i][0]);
        }
        vector<int> ans(n,0);
        solve(labels,A,0,-1,ans);
        return ans;
    }
};
