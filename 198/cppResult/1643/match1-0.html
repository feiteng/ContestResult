class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& ed, string la) {
        vector<int> r(n);
        vector<vector<int>> e(n + 1);
        for (auto &t : ed) {
            e[t[0]].push_back(t[1]);
            e[t[1]].push_back(t[0]);
        }
        function<vector<int>(int, int)> f;
        f = [&](int u, int p) {
            vector<int> res(26);
            ++res[la[u] - 'a'];
            for (int x : e[u]) {
                if(x == p) continue;
                auto rt = f(x, u);
                for (int i = 0; i < 26; i++)
                    res[i] += rt[i];
            }
            r[u] = res[la[u] - 'a' ];
            return res;
        };
        f(0, -1);
        return r;
    }
};
