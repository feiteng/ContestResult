class Solution {
public:
    vector<int>dfs(vector<vector<int>>&adj,int st,int par,vector<int>&ans,string &labels){
        vector<int>c(26,0);
        c[labels[st]-'a']++;
        for(auto j:adj[st]){

            if(j!=par){
                vector<int>c2=dfs(adj,j,st,ans,labels);
                for(int k=0;k<26;k++)c[k]+=c2[k];
            }
        }
        ans[st]=c[labels[st]-'a'];
        return c;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>>adj(n);
        for(int i=0;i<edges.size();i++){
            adj[edges[i][0]].push_back(edges[i][1]);
            adj[edges[i][1]].push_back(edges[i][0]);
        }
        vector<int>vis(n,0);
        vector<int>c=dfs(adj,0,-1,vis,labels);
        return vis;
    }
};
