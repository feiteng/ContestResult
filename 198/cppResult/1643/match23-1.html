class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> adj(n);
		for(auto e : edges) {
			adj[e[0]].push_back(e[1]);
			adj[e[1]].push_back(e[0]);
		}

		vector<int> ans(n);
		dfs(0, 0, adj, labels, ans);
		return ans;
    }
	vector<int> dfs(int u, int pre, vector<vector<int>> &adj, string &labels, vector<int> &ans) {
		vector<int> res(26, 0);
		res[labels[u]-'a'] = 1;
		for(auto v : adj[u]) {
			if(v != pre) {

				auto sub = dfs(v, u, adj, labels, ans);
				for(int i = 0; i < 26; i++) {
					res[i] += sub[i];
				}
			}
		}
		ans[u] = res[labels[u]-'a'];
		return res;
	}
};
