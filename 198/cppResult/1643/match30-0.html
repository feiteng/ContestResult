class Solution {
public:
    vector<vector<int> > g,cnt;
    string lab;
    vector<int> ans;
    void dfs(int u,int fa)
    {
        cnt[u][lab[u]-'a']++;
        for(int v:g[u])
        {
            if(v==fa) continue;
            dfs(v,u);
            for(int i=0;i<26;i++)
                cnt[u][i]+=cnt[v][i];
        }
        ans[u]=cnt[u][lab[u]-'a'];
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        g.assign(n,vector<int>());
        cnt.assign(n,vector<int>(26,0));
        for(auto e:edges)
            g[e[0]].push_back(e[1]),g[e[1]].push_back(e[0]);
        lab=labels;
        ans.assign(n,0);
        dfs(0,-1);
        return ans;
    }
};
