class Solution {
public:

    vector<int> dp;
    vector<int> DFS(int i,vector<vector<int>> &graph,vector<int> &vis,string &labels){
        vector<int> ans(26,0);
        for(int j=0;j<graph[i].size();j++){
            if(vis[graph[i][j]]==0){
                vis[graph[i][j]]=1;

                vector<int> p5=DFS(graph[i][j],graph,vis,labels);
                for(int k=0;k<26;k++) ans[k]+=p5[k];
            }
        }
        ans[labels[i]-'a']++;
        dp[i]=ans[labels[i]-'a'];
        return ans;
    }

    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> graph(n);
        for(int i=0;i<edges.size();i++){
            graph[edges[i][0]].push_back(edges[i][1]);
            graph[edges[i][1]].push_back(edges[i][0]);
        }
        vector<int> vis(n,0);
        vis[0]=1;
        dp.clear();
        dp.resize(n,0);
        DFS(0,graph,vis,labels);
        return dp;
    }
};
