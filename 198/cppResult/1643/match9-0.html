vector<int> getans(int n, int par, vector<int> *g, string &l, vector<int> &ans){
    vector<int> freq(26);
    for(int a=0; a< g[n].size() ;a++){
        if(g[n][a] != par){
           vector<int> v  = getans(g[n][a], n, g,l,ans);
           for(int b=0; b<26; b++)
               freq[b] += v[b];
        }
    }
    freq[l[n]-'a']++ ;
    ans[n] = freq[l[n] - 'a'];
    return freq;
}
class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<int> g[n];
        for(int a=0; a<n-1; a++){
            g[edges[a][0]].push_back(edges[a][1]);
            g[edges[a][1]].push_back(edges[a][0]);
        }
        vector<int> freq(26,0);
        vector<int> ans(n);
        getans(0,-1,g,labels,ans);
        return ans;
    }
};
