class Solution {
public:
    vector<vector<int>> es;
    vector<int> ans;
    vector<int> dfs(int p, int c, string& l) {
        vector<int> r(26);
        r[l[c]-'a']++;
        for (auto nxt : es[c]) {
            if (nxt == p) continue;
            auto ret = dfs(c, nxt, l);
            for (int i=0; i<26; i++) {
                r[i] += ret[i];
            }
        }
        ans[c] = r[l[c]-'a'];
        return r;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        es.resize(n);
        ans.resize(n);
        for (auto e : edges) {
            es[e[0]].push_back(e[1]);
            es[e[1]].push_back(e[0]);
        }
        // cout << "aaa" << endl;
        dfs(-1, 0, labels);
        return ans;
    }
};
