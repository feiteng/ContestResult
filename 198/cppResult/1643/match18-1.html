class Solution {
public:
    string labels;
    vector< vector<int> > G;
    vector<int> ans;
    vector<int> dfs(int node, int parent) {
        vector<int> ret('z' - 'a' + 1, 0);
        ret[labels[node] - 'a'] ++;
        for(int child: G[node]) {

            if(child != parent) {
                vector<int> aux = dfs(child, node);
                for(int x = 0; x < ret.size(); x ++) ret[x] += aux[x]; 
            }
        }
        ans[node] = ret[labels[node] - 'a'];
        return ret;
    }
    
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        this->labels = labels;
        G.clear();
        G.resize(n);
        for(auto edge: edges) {
            G[edge[0]].push_back(edge[1]);
            G[edge[1]].push_back(edge[0]);
        }
        ans.clear();
        ans.resize(n, 0);
        dfs(0, -1);
        return ans;
    }
};
