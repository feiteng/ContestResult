class Solution {
public:
    vector<int> dfs(int x, int f, vector<vector<int>> &v, vector<int> &ans, string &labels) {
        vector<int> re(26, 0);
        re[labels[x] - 'a']++;
        for(int i : v[x]) 
            if(i != f){
                vector<int> tmp = dfs(i, x, v, ans, labels);
                for(int j = 0; j < tmp.size(); j++)
                    re[j] += tmp[j];
            }
                
        ans[x] = re[labels[x] - 'a'];
        return re;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<int> ans(n);
        vector<vector<int>> re(n);
        for(vector<int> &i : edges)
            re[i[0]].push_back(i[1]), re[i[1]].push_back(i[0]);
        dfs(0, -1, re, ans, labels);
        return ans;
    }
};
