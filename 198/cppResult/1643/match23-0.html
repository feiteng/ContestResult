class Solution {
public:
    int c = 0;
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> g(n);
        for (auto e: edges) {
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }
        vector<int> seen(n);
        seen[0] = 1;
        vector<int> ans(n);
        go(0, g, labels, ans, seen);
        return ans;
    }
    
    vector<int> go(int cur, vector<vector<int>>& g, string& labels, vector<int>& ans, vector<int>& seen) {
        vector<int> m(26);
        m[labels[cur]-'a'] = m[labels[cur]-'a'] + 1;
        for (auto next: g[cur]) {
            if (seen[next] == 0) {
                seen[next] = 1;
                auto aa = go(next, g, labels, ans, seen);
                for (int i = 0; i < 26; i ++) {
                    m[i] += aa[i];
                }
            }
        }
        ans[cur] = m[labels[cur]-'a'];
        return m;
    } 
};
