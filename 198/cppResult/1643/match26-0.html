class Solution {
    vector<vector<int>> conn;
    vector<int> ans;
    
    vector<int> dfs(int node, int from, string& labels) {
        //cout << "node: " << node << " from: " << from << endl;
        vector<int> sums(26, 0);
        sums[(int)(labels[node] - 'a')]++;
        for (int x : conn[node]) {
            if (x == from) continue;
            vector<int> tmp = dfs(x, node, labels);
            for (int i = 0; i < 26; ++i) {
                sums[i] += tmp[i];
            }
        }
        ans[node] = sums[labels[node] - 'a'];
        return sums;
    }
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        conn = vector<vector<int>>(n);
        for (auto x : edges) {
            conn[x[0]].push_back(x[1]);
            conn[x[1]].push_back(x[0]);
        }
        ans = vector<int>(n);
        vector<int> sums(26, 0);
        dfs(0, -1, labels);
        return ans;
    }
};
