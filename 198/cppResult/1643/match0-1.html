class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> g(n);
        for(auto e:edges){
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }
        vector<int> ans(n);
        function<vector<int>(int,int)> dfs=[&](int p,int pa){
            vector<int> res(26,0);
                res[labels[p]-'a']++;
                for(int ne:g[p]){
                    if(ne==pa) continue;
                    auto r=dfs(ne,p);
                    for(int i=0;i<26;i++) res[i]+=r[i];
                }
            ans[p]=res[labels[p]-'a'];
            return res;
        };
        dfs(0,-1);
        return ans;
    }
};
