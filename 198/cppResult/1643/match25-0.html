class Solution {
private:
    vector<int> solve(int u, int fa, int n, const vector<vector<int>> &graph, vector<int> &ans, const string &labels) {
        vector<int> seen(26, 0);
        seen[labels[u] - 'a']++;
        
        for (int v : graph[u])
            if (v != fa) {
                vector<int> ret = solve(v, u, n, graph, ans, labels);
                for (int i = 0; i < 26; i++)
                    seen[i] += ret[i];
            }
        
        ans[u] = seen[labels[u] - 'a'];
        
        return seen;
    }

public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> graph(n);
        
        for (const auto &e : edges) {
            graph[e[0]].push_back(e[1]);
            graph[e[1]].push_back(e[0]);
        }
        
        vector<int> ans(n);
        solve(0, -1, n, graph, ans, labels);
        
        return ans;
    }
};
