class Solution {
public:
    
    vector<int>ans;
    vector<int>edg[100000+1];
    vector<int> dfs(int d,int u,string& labels){
        vector<int>cnt(26);
        for(auto &e:edg[u]){
            if(e==d)continue;
            vector<int> tmp=dfs(u,e,labels);

            for(int i=0;i<26;++i)cnt[i]+=tmp[i];
        }
        cnt[labels[u]-'a']++;
        ans[u]=cnt[labels[u]-'a'];
        //if(labels[d]==p)cnt++;
        return cnt;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        ans.resize(n);
        for(auto &i:edges){
            edg[i[0]].emplace_back(i[1]);
            edg[i[1]].emplace_back(i[0]);
        }
        dfs(-1,0,labels);
        return ans;
    }
};
