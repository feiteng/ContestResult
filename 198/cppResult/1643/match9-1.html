#define pb push_back


vector<int> ans;
vector<vector<int> > g;
string str;


vector<int> rec(int del,int p){
    
    vector<int> cnt(26,0);
    
    for(int i=0;i<g[del].size();i++){
        
        if(g[del][i]!=p){
            vector<int> tmp;
            tmp=rec(g[del][i],del);
            for(int j=0;j<26;j++)cnt[j]+=tmp[j];
        }
        
    }
    
    cnt[str[del]-'a']++;
    ans[del]=cnt[str[del]-'a'];
    return cnt;
    
}

class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        
        ans.clear();
        ans.resize(n,0);
        
        g.clear();
        g.resize(n);
        
        str=labels;
        
        int cnt[26]={0};
        
        for(int i=0;i<edges.size();i++){
            
            g[edges[i][0]].pb(edges[i][1]);
            g[edges[i][1]].pb(edges[i][0]);
            
        }
        
        rec(0,-1);
        
        return ans;
        
    }
};
