class Solution {
    vector<int> ans;
    vector<vector<int>> sons;
    vector<int> visited;
    vector<int> dfsHelper(int idx, int n, string& labels) {
        vector<int> ret(26, 0);
        ret[labels[idx] - 'a'] = 1;

        visited[idx] = 1;
        for (int i : sons[idx]) {
            if (visited[i]) continue;
            vector<int> cnt = dfsHelper(i, n, labels);
            for (int j = 0; j < 26; j++) {
                ret[j] += cnt[j];
            }
        }
        ans[idx] = ret[labels[idx] - 'a'];
        return ret;
    }
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        visited = vector<int>(n, 0);
        ans.resize(n);
        sons.resize(n);
        for (auto& e : edges) {
            sons[e[0]].push_back(e[1]);
            sons[e[1]].push_back(e[0]);
        }
        dfsHelper(0, n, labels);
        return ans;
    }
};
