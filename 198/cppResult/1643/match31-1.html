class Solution {
public:
    vector<vector<int>> cnt, g;
    vector<int> ans;
    string labels;

    void dfs(int x, int fa) {
        cnt[x][labels[x] - 'a']++;
        for(auto &y : g[x]) {
            if(y == fa) continue;
            dfs(y, x);
            for(int i = 0; i < 26; i++) cnt[x][i] += cnt[y][i];
        }
        ans[x] = cnt[x][labels[x] - 'a'];
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string label) {
        cnt.resize(n);
        for(int i = 0; i < n; i++) cnt[i].resize(26);
        g.resize(n);
        for(auto &e : edges) {
            g[e[0]].push_back(e[1]);
            g[e[1]].push_back(e[0]);
        }
        ans.resize(n);
        labels = label;
        dfs(0, -1);
        return ans;
    }
};
