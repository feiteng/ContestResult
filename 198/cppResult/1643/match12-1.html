class Solution {
public:
    vector<vector<int>>adj;
    string l;
    vector<int>res;
    
    vector<int>counts(int cur, int parent){
        vector<int>ans(30);
        for(auto x:adj[cur]){
            if(x==parent)continue;
            vector<int>temp = counts(x,cur);
            for(int i=0;i<30;i++)ans[i]+=temp[i];
        }
        ans[l[cur]-'a']++;
        res[cur] = ans[l[cur]-'a'];
        return ans;
    }
    
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        adj = vector<vector<int>>(n);
        l=labels;
        res = vector<int>(n);
        for(auto x:edges){
            adj[x[0]].push_back(x[1]);
            adj[x[1]].push_back(x[0]);
        }
        counts(0,0);
        return res;
    }
};
