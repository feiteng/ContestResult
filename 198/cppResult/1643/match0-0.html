class Solution {
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector < int > g[n];
        for(auto v: edges) {
            g[v[0]].push_back(v[1]);
            g[v[1]].push_back(v[0]);
        }
        vector < int > ans(n,0);
        function < vector < int> (int , int ) > dfs = [&](int u, int p) {
            vector < int > cur(26, 0);
            cur[labels[u] - 'a']++;
            for(int v: g[u]) {
              if(v == p)
                  continue;
              auto tt = dfs(v,u);
             for(int i = 0; i < 26; i++)
                 cur[i] += tt[i];
          }  
            ans[u] = cur[labels[u] - 'a'];
            return cur;
        };
        auto v = dfs(0,-1);
        return ans;
    }
};
