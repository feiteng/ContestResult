class Solution {
    static const int maxn=1e5;
    vector<int> g[maxn+5];
    vector<int> ans;
    int s[maxn+5][26];
public:

    void dfs(int k,int fa,string &labels)
    {
        s[k][labels[k]-'a']=1;
        for(auto u:g[k])
            if(u!=fa)
            {
                dfs(u,k,labels);
                for(int i=0;i<26;++i) s[k][i]+=s[u][i];
            }
        ans[k]=s[k][labels[k]-'a'];
        
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        ans.resize(n);
        for(auto &x:edges)
            g[x[0]].push_back(x[1]),g[x[1]].push_back(x[0]);
        dfs(0,-1,labels);
        return ans;
    }
};
