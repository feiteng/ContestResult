const int maxn=112345;

class Solution {
public:
    string str;
    vector<int> G[maxn];
    int cnt[maxn][26],res[maxn];
    void dfs(int u,int fa) {
        cnt[u][str[u]-'a']=1;
        for (int i=0;i<(int)G[u].size();++i) {
            int v=G[u][i];

            if (v==fa) continue;
            dfs(v,u);
            for (int i=0;i<26;++i) cnt[u][i]+=cnt[v][i];
        }
        res[u]+=cnt[u][str[u]-'a'];
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        str=labels;
        for (auto v:edges) {
            G[v[0]].push_back(v[1]);
            G[v[1]].push_back(v[0]);
        }
        dfs(0,-1);
        vector<int> vec;
        for (int i=0;i<n;++i) vec.push_back(res[i]);
        return vec;
    }
};
