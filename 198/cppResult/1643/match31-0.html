const int MAXN = 1e5 + 50;
int cnt[MAXN][30];
int ans[MAXN];
vector<int> edge[MAXN];

class Solution {
public:
    
    void dfs(int x, int fa, string &la){
        cnt[x][la[x] - 'a']++;
        for (auto v: edge[x]){
            if (v == fa) continue;
            dfs(v, x, la);
            for (int c = 0; c < 26; c++) cnt[x][c] += cnt[v][c];
        }
        ans[x] = cnt[x][la[x] - 'a'];
    }
    
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        for (int i = 0; i <= n; i++){
            memset(cnt[i], 0, sizeof(cnt[i]));
            edge[i].clear();
            ans[i] = 0;
        }
        
        for (auto &e: edges){
            int a = e[0], b = e[1];
            edge[a].push_back(b);
            edge[b].push_back(a);
        }
        
        dfs(0, -1, labels);
        
        vector<int> ret;
        for (int i = 0; i < n; i++) ret.push_back(ans[i]);
        return ret;
    }
};
