class Solution {
public:
    vector<int> dfs(int u, int f, vector<vector<int>>& e, string& s, vector<int>& ans) {
      vector<int> cur(26, 0);
      cur[s[u] - 'a'] += 1;
      for (auto v : e[u]) {
        if (v == f) {
          continue;
        }

        vector<int> t = dfs(v, u, e, s, ans);
        for (int i = 0; i < 26; i++) {
          cur[i] += t[i];
        }
      }
      ans[u] = cur[s[u] - 'a'];
      return cur;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
      vector<vector<int>> e(n);
      vector<int> ans(n);
      for (auto p : edges) {
        e[p[0]].push_back(p[1]);
        e[p[1]].push_back(p[0]);
      }
      dfs(0, 0, e, labels, ans);
      return ans;
    }
};
