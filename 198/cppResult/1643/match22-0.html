class Solution {
public:
    vector<int>g[100004];
    string s;
    int a[100004];
    vector<int>dfs(int u,int fa){
        vector<int>w(26);
        w[s[u]-'a'] = 1;
        for(int i = 0;i < g[u].size();i++){
            int v=g[u][i];
            if(v==fa) continue;
            vector<int>t = dfs(v,u);
            for(int j = 0;j < 26;j++)
                w[j] += t[j];
        }
        a[u] = w[s[u]-'a'];
        return w;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        s = labels;
        for(vector<int>v:edges){
            int x = v[0],y = v[1];
            g[x].push_back(y);
            g[y].push_back(x);
        }
        dfs(0,-1);
        vector<int>p;
        for(int i = 0;i < n;i++)
            p.push_back(a[i]);
        return p;
    }
};
