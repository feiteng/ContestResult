class Solution {
    vector<int> dfs(vector<int>& res, int cur, int fa, vector<vector<int>>& E, string& labels) {
        vector<int> A(26);
        A[labels[cur] - 'a'] = 1;

        for (auto v: E[cur]) if (v != fa) {
            vector<int> temp = dfs(res, v, cur, E, labels);
            for (int i = 0; i < 26; ++i)
                A[i] += temp[i];
        }
        res[cur] = A[labels[cur] - 'a'];
        return A;
    }
public:
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<vector<int>> E(n);
        for (const auto& e: edges) {
            E[e[0]].push_back(e[1]);
            E[e[1]].push_back(e[0]);
        }
        vector<int> res(n);
        dfs(res, 0, -1, E, labels);
        return res;
    }
};
