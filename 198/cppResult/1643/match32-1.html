class Solution {
public:
    void DFS(int u, string &labels, vector<vector<int>> &count, vector<vector<int>> &E, vector<bool> &visited) {
        visited[u] = true;
        ++count[u][labels[u] - 'a'];
        for (int v : E[u]) {

            if (!visited[v]) {
                DFS(v, labels, count, E, visited);
                for (int i = 0; i < 26; ++i) {
                    count[u][i] += count[v][i];
                }
            }
        }
        return;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        vector<bool> visited(n, false);
        vector<vector<int>> E(n);
        for (auto &edge : edges) {
            E[edge[0]].push_back(edge[1]);
            E[edge[1]].push_back(edge[0]);
        }
        vector<vector<int>> count(n, vector<int>(26, 0));
        DFS(0, labels, count, E, visited);
        vector<int> ans;
        for (int i = 0; i < n; ++i) {
            ans.push_back(count[i][labels[i] - 'a']);
        }
        return ans;
    }
};
