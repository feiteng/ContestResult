class Solution {
public:
    vector<int> ans;
    vector<vector<int> > edg;
    vector<bool> vis;
    vector<int> dfs(int now,int n,vector<vector<int>>& edges, string &labels)
    {
        vis[now]=true;
        vector<int> ret(26,0);
        ret[labels[now]-'a']++;
        for(int i=0;i<edg[now].size();++i)
        {
            if(!vis[edg[now][i]])
            {
                vector<int> tmp=dfs(edg[now][i],n,edges,labels);
                for(int j=0;j<26;++j)
                    ret[j]+=tmp[j];
            }
        }
        ans[now]=ret[labels[now]-'a'];
        return ret;
    }
    vector<int> countSubTrees(int n, vector<vector<int>>& edges, string labels) {
        edg.resize(n);
        vis.resize(n,0);
        ans.resize(n);
        for(int i=0;i<n-1;++i)
        {
            edg[edges[i][0]].push_back(edges[i][1]);
            edg[edges[i][1]].push_back(edges[i][0]);
        }
        dfs(0,n,edges,labels);
        return ans;
    }
};
