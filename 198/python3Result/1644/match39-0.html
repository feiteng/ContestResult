class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        if not s: return []
        cache = {}
        for i, e in enumerate(s):
            if e not in cache:
                cache[e] = [float('inf'), float('-inf')]
            cache[e][0] = min(cache[e][0], i)
            cache[e][1] = max(cache[e][1], i)

        cands = []
        for i, e in enumerate(s):
            if i <= cache[e][0]:
                j = i
                end = i
                bad = False
                while j < len(s):
                    end = max(end, cache[s[j]][1])
                    if end == j:
                        break
                    if cache[s[j]][0] < i:
                        bad = True
                    j += 1
                if not bad: cands.append([i, j])
                    
        cands = sorted(cands, key=lambda x:x[1])
    
        res = [cands[0]]
        for c in cands[1:]:
            if c[0] > res[-1][1]:
                res.append(c)
        
        return [s[e[0]:e[1]+1] for e in res]
