class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        se = set(s)
        #b = sorted([[s.find(i), s.rfind(i), i] for i in se])
        d = {i: [s.find(i), s.rfind(i)] for i in se}
        st = []
        b = []
        r = -1
        for i, c in enumerate(s):
            x, y = d[c]
            if i == x:
                st.append([c, x, y])        
            st
[-1][1] = min(st[-1][1], x) 
            st[-1][2] = max(st[-1][2], y)
            #if s == "abcdabcd":print(st,x,y,r)
            if i == y:
                if st[-1][1] <= x and st[-1][2] <= y and st[-1][1] > r:
                    b.append(s[st[-1][1]: y + 1])
                    r = y
                    #if s == "abcdabcd":print(b)
                if len(st) > 1:
                    st
[-2][1] = min(st[-2][1], st[-1][1])
                    st[-2][2] = max(st[-2][2], st[-1][2])
                st.pop()
        return b
