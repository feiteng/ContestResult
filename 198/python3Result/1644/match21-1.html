class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        dic = {}
        dic1={}
        for i in s:
            if i in dic:
                dic[i]+=1
                dic1[i]+=1
            else:
                dic[i]=1
                dic1[i]=1
        i = 0
        ans = []
        se = set()
        start = 0
        while i<len(s):
            dic[s[i]]-=1
            if dic[s[i]]==0:
                temp = {s[i]:1}
                se.add(s[i])
                if temp[s[i]]==dic1[s[i]]:
                    se.remove(s[i])
                j = i-1
                while j>=0 and se and j>=start:
                    if s[j] in se:
                        temp[s[j]] += 1
                    else:
                        temp
[s[j]] = 1
                        se.add(s[j])
                    if temp[s[j]]==dic1[s[j]]:
                        se.remove(s[j])
                    j-=1
                if not se:
                    ans.append(s[j+1:i+1])
                    start = i+1
                se = set()
            i+=1
        return ans
