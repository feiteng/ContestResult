class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        table = dict()
        for i, c in enumerate(s):
            if c not in table:
                table[c] = [i, i]
            else:
                table[c][1] = i
        for i, c in enumerate(s):
            for key in table.keys():
                if table[key][1] > i > table[key][0]:
                    if table[c][0] < table[key][0]:
                        table[key][0] = table[c][0]
                    if table[c][1] > table[key][1]:
                        table[key][1] = table[c][1]
        res = []
        for i, c in enumerate(s):
            if i == 0:
                cur = c
                start, end = table[c]
            else:
                if c in table:
                    if table[c][0] > start and table[c][1] < end:
                        if cur in table:
                            del table[cur]
                        cur = c
                        start, end = table[c]
                    elif table[c][0] > end:
                        res.append(s[start:end+1])
                        cur = c
                        start, end = table[c]
                    else:
                        del table[c]
        res.append(s[start:end + 1])
        return res
