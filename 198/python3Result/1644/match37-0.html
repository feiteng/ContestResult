class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        first = {}
        last = {}
        for i in range(len(s)):
            if s[i] not in first:
                first[s[i]] = i
            last[s[i]] = i
        
        intvs = []
        
        pairs = []
        for k, v in first.items():
            pairs.append(v)
        pairs.sort()
        # print(pairs)
        for idx in pairs:
            ch = s[idx]
            if first[ch] == last[ch]:
                intvs.append([1,idx,last[ch]])
            else:
                lst = last[ch]
                for iidx in pairs:
                    if idx < iidx < lst:
                        lst = max(lst, last[s[iidx]])                
                for cch in s[idx:lst+1]:
                    if first[cch] < idx: break
                else:
                    intvs.append([lst-idx+1, idx, lst])
        # print(intvs)
        intvs.sort()
        res = []
        for length, fst, lst in intvs:
            for ffst, llst in res:
                if not (fst > llst or lst < ffst):                    
                    break
            else:
                res.append([fst, lst])
                    
        rres = []
        for fst, lst in res:
            rres.append(s[fst:lst+1])
        return rres
