from string import ascii_lowercase
from collections import defaultdict
class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        d = defaultdict(list)
        for i, c in enumerate(s):
            d[c].append(i)
        
        eg = defaultdict(list)
        for a in d:
            for b in d:
                for i in d[b]:
                    if d[a][0] <= i <= d[a][-1]:
                        eg[a].append(b)
                        break
        
        
        def bfs(a):
            q = [a]
            seen = {a}
            while q:
                nq = []
                for a in q:
                    for b in eg[a]:
                        if b not in seen:
                            seen.add(b)
                            nq.append(b)
                q = nq
            return seen

        v = []
        for a in d:
            seen = bfs(a)
            v.append(''.join(c for c in s if c in seen))
        
        v.sort(key=len)
        seen = set()
        ans = []
        for s in v:
            if not seen & set(s):
                seen |= set(s)
                ans.append(s)
        return ans
