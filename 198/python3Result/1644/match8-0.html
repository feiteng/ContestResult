class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        from collections import defaultdict
        key = []
        n = len(s)
        cnt = [0 for i in range(26)]
        l = [n for i in range(26)]
        r = [-1 for i in range(26)]
        for i in range(len(s)):
            idx = ord(s[i]) - ord('a')
            cnt[idx] += 1
            l[idx] = min(l[idx], i)
            r[idx] = max(r[idx], i)
        # for i in range(26):
        #     if l[i] != n:
        #         print(chr(i + ord('a')), l[i], r[i])
        def get_id(c):
            return ord(c) - ord('a')
        def deal(idx):
            dic = {}
            if l[idx] == n:
                return
            left, right = l[idx], l[idx]
            dic = defaultdict(int)
            dic[idx] = 1
            while True:
                idt = -1
                for k, v in dic.items():
                    if v < cnt[k]:
                        idt = k
                        break
                if idt == -1:
                    break
                if l[idt] < left:
                    left -= 1
                    dic[get_id(s[left])] += 1
                else:
                    right += 1
                    dic[get_id(s[right])] += 1
            key.append([left, right])
    
        for i in range(26):
            deal(i)
        # print(key)
        key.sort(key = lambda x: (x[1], -x[0]))
        ans = [s[key[0][0]:key[0][1] + 1]]
        r = key[0][1]
        for i in range(1, len(key)):
            if key[i][0] > r:
                r = key[i][1]
                ans.append(s[key[i][0]:key[i][1] + 1])
        return ans
