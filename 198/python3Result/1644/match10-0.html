class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        d = {}
        o = []
        for ch in s:
            d[ch] = []
        for k in d.keys():
            start = -1
            end = -1
            for i, ch in enumerate(s):
                if ch == k:
                    if start == -1:
                        start = i
                    end = i
            d[k] = [start, end]
        for k in d.keys():
            for j in d.keys():
                if k != j:
                    if k in s[d[j][0]:d[j][1]+1]:
                        d[j][0] = min(d[j][0], d[k][0])
                        d[j][1] = max(d[j][1], d[k][1])
        for value in d.values():
            o.append(value)
        c = sorted(o, key=lambda x:(x[1], x[0]))
        ans = [];
        temp = c[0][1];
        ans.append(s[c[0][0]:c[0][1]+1])
        for num in c[1:]:
            if (num[0] >= temp):
                temp = num[1]
                ans.append(s[num[0]:num[1]+1])
            
        return ans
