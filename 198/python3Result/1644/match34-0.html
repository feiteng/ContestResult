class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        
        dic = {}
        for i in range(len(s)):
            if s[i] in dic:
                dic[s[i]][1] = i
            else:
                dic[s[i]] = [i, i]
                
        
        def expand(x, dic, s):
            vocab = set()
            vocab.add(x)
            left, right = dic[x]
            l, r = left-1, left+1
            cnt = 0
            # print('expand init', x, cnt, left, right, l, r, vocab)
            while r < right or l > left:
                if r < right:
                    if s[r] not in vocab:
                        vocab.add(s[r])
                        right = max(right, dic[s[r]][1])
                        left = min(left, dic[s[r]][0])
                    r += 1
                if l > left:
                    if s[l] not in vocab:
                        vocab.add(s[l])
                        right = max(right, dic[s[l]][1])
                        left = min(left, dic[s[l]][0])
                    l -= 1
                # cnt += 1
                # print('expand', x, cnt, left, right, l, r, vocab)
                # if cnt > 10: 
                #     break
            
            return [left, right]
                    
        
        # print(dic)
        candi = []
        for x in dic:
            candi.append(expand(x, dic, s))
            # print(x, candi[-1])
        
        # seen = set()
        # for x in s:
        #     if x not in seen:
        #         seen.add(x)
        #         candi.append(expand(x, dic, s))
        #         print(x, candi[-1])
            
        # print(list(dic.values()))
        lst = sorted(list(candi), key=lambda x: x[1])
        # print(lst)
        
        preend = -1
        rtn = []
        for st, end in lst:
            if st > preend:
                rtn.append(s[st:end+1])
                preend = end
        
        return rtn
