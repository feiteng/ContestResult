class Solution:
    def maxNumOfSubstrings(self, s):
        d = dict()
        ret = list()
        for index, value in enumerate(s):
            if value not in d:
                d[value] = [index, index]
            else:
                d[value][1] = index
        change = True
        while(change):
            change = False
            for key, [start, end] in d.items():
                s_set = set(s[start: end + 1])
                new_start = min(d[i][0] for i in s_set)
                new_end = max(d[i][1] for i in s_set)
                if new_start != start:
                    d[key][0] = new_start
                    change = True
                if new_end != end:
                    d[key][1] = new_end
                    change = True
        new_sort_list = list()
        for key, [start, end] in d.items():
            new_sort_list.append([end, end - start + 1, start, key])
        new_sort_list.sort()
        for index, [end, length, start, key] in enumerate(new_sort_list):
            if index == 0:
                ret.append(s[start: end + 1])
                last_end = end
            else:
                if d[key][0] > last_end:
                    ret.append(s[start: end + 1])
                    last_end = end
        print(ret)
        return ret
