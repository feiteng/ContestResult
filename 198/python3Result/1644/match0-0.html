class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        d = {}
        n = len(s)
        clist = set(s)
        for c in clist:
            left = s.index(c)
            right = n - s[::-1].index(c) - 1
            # print(c, left, right)
            d[c] = (left, right)    
            
        for c in clist:
            l = set()
            l.add(c)
            q = [c]
            while q:
                cur = q.pop(0)
                cclist = set(s[d[cur][0]:d[cur][1]+1])
                for cc in cclist:
                    if cc not in l:
                        q.append(cc)
                        l.add(cc)
            
            cleft = d[c][0]
            cright = d[c][1]
            for cc in l:
                if cc != c:
                    cleft = min(cleft, d[cc][0])
                    cright = max(cright, d[cc][1])
            d[c] = (cleft, cright)
        remove = []
        for c in clist:
            for cc in clist:
                if cc != c:
                    if d[cc][0] > d[c][0] and d[cc][1] < d[c][1]:
                        remove.append(c)
        # print(clist)
        remove = set(remove)
        for c in remove:
            clist.remove(c)
        
        clist = sorted(clist, key = lambda x: d[x][0])
        
        ss = None
        res = []
        for i, c in enumerate(clist):
            if i == 0:
                ss = d[c]
            else:
                if d[c][0] < ss[1]:
                    ss = (ss[0], d[c][1])
                else:
                    res.append(ss)
                    ss = d[c]
        else:
            res.append(ss)
        
        # print(res)
        
        
        # clist = list(clist)
        # for c in clist:
        #     clist.remove(c)
        #     for cc in clist:
        #         clist.remove(cc)
        #         if cc != c:
        #             if d[cc][0] < d[c][0] < d[cc][1] < d[c][1]:
        #                 clist.append(cc+c)
        #                 d[cc+c] = (d[cc][0], d[c][1])
        #             elif d[c][0] < d[cc][0] < d[c][1] < d[cc][1]:
        #                 clist.append(c+cc)
        #                 d[c+cc] = (d[c][0], d[cc][1])
        #             else
        # print(res) 
        if len(res) == 1:
            return [s[res[0][0]:res[0][1]+1]]
        else:
            return [s[r[0]:r[1]+1] for r in res]
