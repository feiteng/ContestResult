class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        c = collections.Counter(s)
        total = [c[x] for x in string.ascii_lowercase]
        feat = [0] * 26
        met = {(0,) * 26: 0}
        order = []
        
        res = []
        
        for i, c in enumerate(s):
            if c not in order: order += c,
            feat[ord(c) - ord('a')] += 1                
            
            tmp = feat[:]
            #print(i, c)
            for x in order[::-1]:
                if tmp[ord(x) - ord('a')] < total[ord(x) - ord('a')]:
                    break
                tmp[ord(x) - ord('a')] = 0
                #print(tuple(tmp), met)
                if tuple(tmp) in met:
                    res += s[met[tuple(tmp)]: i + 1],
                    met = {(0) * 26: 0}
                    order = []
            
            met[tuple(feat)] = i + 1
        return res
