class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        start = [-1]*26
        end = [-1]*26
        for i,it in enumerate(s):
            if start[ord(it)-ord('a')]==-1:
                start[ord(it)-ord('a')]=i
                
        for i,it in enumerate(s[::-1]):
            if end[ord(it)-ord('a')]==-1:
                end[ord(it)-ord('a')]=len(s)-1-i
        
        d={(a,b) for a,b in zip(start,end)}
        # print(d)
        while True:
            d_new=set()
            for a,b in d:
                if a==-1 and b==-1:
                    continue
                st,en = a,b
                for j in range(a,b):
                    st=min(st,start[ord(s[j])-ord('a')])
                    en=max(en,end[ord(s[j])-ord('a')])
                d_new.add((st,en))
            # print(d_new)
            if d_new==d:
                break
            d=d_new
            
        l = sorted([(a,b) for a,b in d], key=lambda x:x[1])
        out=[]
        e=-1
        for a,b in l:
            if a>e:
                e=b
                out.append((a,b))
        
        return [s[a:b+1] for a,b in out]
