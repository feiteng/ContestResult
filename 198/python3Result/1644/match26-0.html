class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        visitedset = set()
        maxCharIndex = {}
        for i,c in enumerate(s):
            maxCharIndex[c] = i
        substringa = []
        #print(maxCharIndex)
        for i,c in enumerate(s):
            if c not in visitedset:
                startIndex = i
                endIndex = i
                mustGoIndex = maxCharIndex[c]
                while mustGoIndex !=endIndex:
                    endIndex+=1
                    if endIndex < len(s):
                        if s[endIndex] not in visitedset:
                            if mustGoIndex < maxCharIndex[s[endIndex]]:
                                mustGoIndex = maxCharIndex[s[endIndex]]
                        else:
                            endIndex = -1
                            break
                    else:
                        endIndex-=1
                visitedset.add(c)
                if endIndex!=-1:
                    substringa.append([startIndex,endIndex])  
        finala = []
        print(substringa)
        substringa.sort(key=lambda c:(c[0],c[1]*-1))
        for i in range(len(substringa)):
            cv = substringa[i]
            if i == len(substringa)-1:
                finala.append(cv)
                continue
            nv = substringa[i+1]
            if cv[1]> nv[0]:
                continue
            else:
                finala.append(cv)
        finals =[]
        for a in finala:
            finals.append(s[a[0]:a[1]+1])
        return finals
