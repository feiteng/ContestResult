class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        from collections import defaultdict
        
        d = defaultdict(list)
        for i, c in enumerate(s):
            d[c].append(i)
        
        res = []
        prev = -1
        for i, c in enumerate(s):
            if i == d[c][-1] and d[c][0] > prev:
                valid = True
                start, end = d[c][0], i
                while valid and start < end:
                    p_start= start
                    for j in range(start, end):
                        if d[s[j]][-1] > i:
                            valid = False
                            break
                        start = min(start, d[s[j]][0])
                        if start <= prev:
                            valid = False
                            break
                    end = p_start
                
                if valid:
                    res.append(s[start:i+1])
                    prev = i
        
        return res
