class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        m = {}       
        for i, c in enumerate(s):
            if c in m.keys():
                m[c][1] = i
            else:
                m[c] = [i, i]
                
        m = dict(sorted(m.items(), key = lambda x: x[1][0] - x[1][1]))
        #print(m)
        for c in m.keys():
            #print(c)
            idx = m[c]
            tmp_max = idx[1]
            tmp_min = idx[0]
            for i in range(idx[0], idx[1]):
                tmp_max = max(tmp_max, m[s[i]][1])
                tmp_min = min(tmp_min, m[s[i]][0])
                
            m[c][1] = tmp_max
            m[c][0] = tmp_min
        
        arr = [m[k] for k in m.keys()]
        
        #print(m)
        arr = sorted(arr, key = lambda x: (x[1], x[1] - x[0]))
        #print(arr)
        curr_idx = -1
        result = []
        for idx  in arr:
            if curr_idx < idx[0]:
                result.append(s[idx[0]: idx[1] + 1])
                curr_idx = idx[1]
        return result
