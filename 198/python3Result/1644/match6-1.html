class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        n = len(s)
        inter = [[n, -1] for i in range(26)]
        
        for i in range(n):
            idx = ord(s[i]) - ord('a')
            if i < inter[idx][0]:
                inter[idx][0] = i
            if i > inter[idx][1]:
                inter[idx][1] = i
                
        ninter = []
        for i in range(26):
            si, ei = inter[i]
            if si == n:
                continue
            st, et = si, ei
            si, ei = ei, ei
            while si != st or et != ei:
                stmp, etmp = si, ei
                si, ei = st, et
                for c in s[st:stmp]:
                    idx = ord(c) - ord('a')
                    
if inter[idx][0] < st:
                        st = inter[idx][0]
                    if inter[idx][1] > et:
                        et = inter[idx][1]
                    
                for c in s[etmp:et+1]:
                    idx = ord(c) - ord('a')
                    
if inter[idx][0] < st:
                        st = inter[idx][0]
                    if inter[idx][1] > et:
                        et = inter[idx][1]
                        
            ninter.append([st, et])
            
        ninter.sort(key = lambda x : x[1])
        #print(ninter)
        sel = []
        for si, ei in ninter:
            if not sel:
                sel.append([si, ei])
            elif si > sel[-1][0]:
                sel.append([si, ei])
                
        res = []
        for si, ei in sel:
            res.append(s[si:ei+1])
            
        return res
