class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        
        start_end_dict = {}
        for i, c in enumerate(s):
            if c not in start_end_dict:
                start_end_dict[c] = [i, i]
            else:
                start_end_dict[c][1] = i 
        
        start_end = sorted(start_end_dict.items(), key=lambda item: item[1][1])
        
        # print(start_end)
        
        res = []
        i = 0
        while i < len(start_end):
            c, se = start_end[i]
            start, end = se 
            cs = set([c])
            to_loop = True
            while to_loop:
                to_loop = False
                for j in range(start, end):
                    if s[j] not in cs:
                        cs.add(s[j])
                        if start_end_dict[s[j]][0] < start:
                            start = start_end_dict[s[j]][0]
                            to_loop = True
                        if start_end_dict[s[j]][1] > end:
                            end = start_end_dict[s[j]][1]
                            to_loop = True

            res.append((start, end))
            # print(c, start, end)
            i += 1
        
        res = sorted(res, key=lambda item: item[1])
        
        ans, curr_end = [], -1
        for start, end in res:
            if start < curr_end:
                continue 
            ans.append(s[start:end+1])
            curr_end = end
        return ans
