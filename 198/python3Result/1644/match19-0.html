class Solution:
    def maxNumOfSubstrings(self, str7: str) -> List[str]:
        start = [-1] * 26
        end = [-1] * 26
        for i, c in enumerate(str7):
            o = ord(c) - ord('a')
            if start[o] == -1:
                start[o] = i
            end[o] = i
        
        strs = {}
        for i, (s, e) in enumerate(zip(start, end)):
            if s != -1:
                strs[chr(i + ord('a'))] = (s, e)
        # print(strs)
        strs2 = []
        for k, (s, e) in strs.items():
            m1, m2 = s, e
            chars = list(str7[s:e + 1])
            while chars:
                c = chars.pop()
                if strs[c][0] < m1:
                    chars = list(str7[strs[c][0]:m1]) + chars
                    m1 = strs[c][0]
                if strs[c][1] > m2:
                    chars = chars + list(str7[m2 + 1:strs[c][1] + 1])
                    m2 = strs[c][1]
            strs[k] = (m1, m2)
            strs2.append((m2 - m1, m1, m2))
        # print(strs2)
        strs2.sort()
        # print(strs2)
        res = []
        ints = []
        for l, s, e in strs2:
            for m1, m2 in ints:
                if s <= m1 <= e or s <= m2 <= e:
                    break
            else:
                res.append(str7[s:e + 1])
                m1 = min(m1, s)
                m2 = max(m2, e)
            ints.append([s, e])
        return res
