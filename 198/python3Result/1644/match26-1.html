class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        a=[[]for _ in range(26)]
        for i,v in enumerate(s):
            a[ord(v)-ord('a')].append(i)
        a=[[i[0],i[-1]] if i else [] for i in a]
        for i in a:
            if not i:
                continue
            q=[i]
            while q:
                now=q.pop()
                for j in range(now[0],now[1]+1):
                    ch=ord(s[j])-ord('a')
                    f,t=a[ch]
                    if f<i[0]:
                        q.append([f,i[0]])
                        i[0]=f
                    if t>i[1]:
                        q.append([i[1],t])
                        i[1]=t
            
        a=[i for i in a if i]
        a.sort(key=lambda x:x[1]-x[0])
        b=[]
        def inter(x):
            for i in b:
                f=max(i[0],x[0])
                t=min(i[1],x[1])
                if f<=t:
                    return True
            return False
        for i in a:
            if not inter(i):
                b
.append(i)
        ans=[]
        for i in b:
            ans.append(s[i[0]:i[1]+1])
        return ans
