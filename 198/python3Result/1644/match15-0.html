class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        rg = [[s.find(chr(i)), s.rfind(chr(i))] for i in range(97, 97 + 26)]
        
        for i in range(26):
            if rg[i][0] < 0: continue
            checked = [False] * 26
            checked[i] = True
            l, u = rg[i]
            tocheck = [j for j in range(26) if not checked[j] and chr(97 + j) in s[l:u + 1]]
            while tocheck:
                newl = min(min(rg[j][0] for j in tocheck), l)
                newu = max(max(rg[j][1] for j in tocheck), u)
                for j in tocheck: checked[j] = True
                added = s[newl:l] + s[u+1:newu+1]
                l, u = newl, newu
                tocheck = [j for j in range(26) if not checked[j] and chr(97 + j) in added]
            rg[i] = [l, u]

        rg = [[l, u] for l, u in rg if l >= 0]
        rg.sort(key=lambda x: (x[1], -x[0]))

        res, upper = [], -1
        for l, u in rg:
            if l <= upper: continue
            res.append(s[l:u+1])
            upper = u
        
        return res
