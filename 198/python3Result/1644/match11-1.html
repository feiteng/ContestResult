class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        t = [[-1,-1,set()] for i in range(26)]
        for i in range(len(s)):
            x = ord(s[i]) - ord('a')
            if t[x][0] == -1: t[x][0] = i
            t[x][1] = i
        for i in range(len(s)):
            for j in range(26):
                if i >= t[j][0] and i <= t[j][1]:
                    t[j][2].add(s[i])
        while True:
            flag = False
            for i in range(26):
                while True:
                    tmp = t[i][2]
                    for c in tmp:
                        x = ord(c)-ord
('a')
                        t[i][0] = min(t[i][0], t[x][0])
                        t[i][1] = max(t[i][1], t[x][1])
                        t[i][2] = tmp.union(t[x][2])
                    if len(tmp) == len(t[i][2]): break
                    else: flag = True
            if flag == False: break
        t.sort()
        ans = []
        for i in range(26):
            if t[i][0] == -1: continue
            if i == 25 or t[i][1] < t[i+1][0]:
                ans.append(s[t[i][0]:t[i][1]+1])
        return ans
