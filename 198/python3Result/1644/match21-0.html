class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        C = Counter(s)
        tmpc = Counter()
        left, right = [], []
        for i in range(len(s)):
            tmpc[s[i]] += 1
            if tmpc[s[i]] == 1:
                t = tmpc.copy()
                t[s[i]] -= 1
                left.append((i, t))
            if tmpc[s[i]] == C[s[i]]:
                right.append((i, tmpc.copy()))

        guess = []
        for i, cl in left:
            for j, cr in right:
                if j < i: continue
                tmpc = cr.copy()
                tmpc.subtract(cl)
                for k, v in tmpc.items():
                    if v != 0 and v != C[k]:
                        break
                else:
                    guess.append((i, j))
                    
        #print('guess', [s[i:j+1] for i, j in guess])
        def cmp(t):
            x, y = t
            return (y, -x)
        guess.sort(key=cmp)
        #print('G', guess)
        res, r = [], -1
        for x, y in guess:
            if r < x:
                res.append((x, y))
                r = y
        return [s[i:j+1] for i, j in res]
