class Solution:
    def maxNumOfSubstrings(self, s: str) -> List[str]:
        tmp = {c: [s.index(c), s.rindex(c)+1] for c in set(s)}
        pairs = []
        for c in set(s):
            l, r = s.index(c), s.rindex(c)+1
            tmpl, tmpr = l, r
            while True:
                t = set(s[tmpl:tmpr])
                for k in t:
                    tmpl = min(tmpl, tmp[k][0])
                    tmpr = max(tmpr, tmp[k][1])
                if tmpl == l and tmpr == r:
                    break
                l, r = tmpl, tmpr
            pairs.append([l, r])
            
        print(pairs)
            
        pairs.sort()
        stack = []
        for p in pairs:
            if not stack:
                stack.append(p)
            if stack[-1][0] < p[0] < stack[-1][1] < p[1]:
                stack[-1][-1] = p[1]
            else:
                stack.append(p)
        pairs = stack
        pairs.sort
(key=lambda x: x[1])

        res = []
        last = 0
        for a, b in pairs:
            if a >= last:
                res.append(s[a:b])
                last = b
        return res
