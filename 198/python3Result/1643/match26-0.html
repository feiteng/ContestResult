class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], la: str) -> List[int]:
        ans=[0]*n
        tree=[[] for i in range(n)]
        d=defaultdict()
        for i,j in edges:
            tree[i].append(j)
            tree[j].append(i)
        visit=[False]*n
        def dfs(root):
            nonlocal ans,visit
            visit[root]=True
            c=Counter(la[root])
            for i in tree[root]:
                if not visit[i]:
                    c+=dfs(i)
            ans[root]=c[la[root]]
            return c
        dfs(0)
        return ans
