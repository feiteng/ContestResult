class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        next_nodes = collections.defaultdict(list)
        for a, b in edges:
            next_nodes[a].append(b)
            next_nodes[b].append(a)
        ans = [0] * n
        def visit(node, parent):
            counts = collections.Counter(labels[node])
            for next_node in next_nodes[node]:
                if next_node == parent:
                    continue
                for label, count in visit(next_node, node)
.items():
                    counts[label] += count
            ans[node] = counts[labels[node]]
            return counts
        visit(0, -1)
        return ans
