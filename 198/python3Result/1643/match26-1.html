class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        tree 
= [[] for _ in range(n)]
        for fr, to in edges:
            tree[fr].append(to)
            tree[to].append(fr)
        res = [1] * n
        def dfs(root, parent):
            cnt = Counter(labels[root])
            
            for children in tree[root]:
                if children != parent:
                    cnt 
+= dfs(children, root)
            res[root] = cnt[labels[root]]
            return cnt
        
        dfs(0, -1)
        return res
