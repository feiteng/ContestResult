class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        nei = collections.defaultdict(list)
        for i, j in edges:
            nei[i].append(j)
            nei[j].append(i)

        visited = set()
        sol = {}
        def search(node):
            visited.add(node)
            cc = collections.Counter()
            cc[labels[node]] = 1
            for node2 in nei[node]:
                if node2 not in visited:
                    cc += search(node2)
            sol[node] = cc[labels[node]]
            return cc

        search(0)
        sol2 = [sol[i] for i in range(n)]
        return sol2
