class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        dic = defaultdict(list)
        for a, b in edges:
            dic[a].append(b)
            dic[b].append(a)
        lst = [0]*n
        def traverse(curr, parent):
            ret = Counter()
            ret[labels[curr]] += 1
            for nxt in dic[curr]:
                if nxt == parent:
                    continue
                ret += traverse(nxt, curr)
            lst[curr] = ret[labels[curr]]
            return ret
        traverse(0, None)
        return lst
