class Solution:
    def countSubTrees(self, n: int, ed: List[List[int]], lab: str) -> List[int]:
        g = [[] for i in range(n)]
        for u,v in ed :
            g[u].append(v)
            g[v].append(u)
            
        res = [0] * n
        def dfs(u,f) :
            r1 = [0] * 26
            ir = ord(lab[u])-ord('a')
            r1[ir] += 1
            for v in g[u] :
                if v == f :
                    continue
                r2 = dfs(v,u)
                for i in range(26) :
                    r1[i] += r2[i]
            res[u] = r1[ir]
            return r1
        
        dfs(0,-1)
        
        return res
