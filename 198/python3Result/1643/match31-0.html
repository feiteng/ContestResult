class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        d = collections.defaultdict(list)
        for s, e in edges:
            d[s].append(e)
            d[e].append(s)
        ans = [1]*n
        vis = set()
        vis.add(0)

        def dfs(st):
            if not d[st]:
                return {labels[st]: 1}
            dt = collections.defaultdict(int)
            dt[labels[st]] = 1
            for c in d[st]:
                if c not in vis:
                    vis.add(c)
                    for k, v in dfs(c).items():
                        dt[k] += v

            ans[st] = dt[labels[st]]
            return dt

        dfs(0)
        return ans
