class Solution:
    def countSubTrees(self, n: int, e: List[List[int]], la: str) -> List[int]:
        tb = [set() for _ in range
(n)]
        for a, b in e:
            tb[a].add(b)
            tb[b].add(a)
        ans = [1] * n
        def dfs(p):
            nonlocal reach
            reach[p] = True
            d = collections.Counter
([la[p]])
            for q in tb[p]:
                if not reach[q]: d += dfs(q)
            ans[p] = d[la[p]]
            return d
        reach = [False] * n
        dfs(0)
        return ans
