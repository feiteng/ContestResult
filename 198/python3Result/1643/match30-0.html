from collections import defaultdict

class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        edge_map = defaultdict(list)
        
        for a, b in edges:
            edge_map[a].append(b)
            edge_map[b].append(a)
        
        parents = {}
        parents[0] = -1
        ans = [1 for _ in range(n)]

        def dfs(node):
            label_count = defaultdict(int)
            label_count[labels[node]] = 1
            for child in edge_map[node]:
                if child != parents[node]:
                    parents[child] = node
                    child_label_count = dfs(child)
                    for k in child_label_count:
                        label_count[k] += child_label_count[k]
            ans[node] = label_count[labels[node]]
            return label_count
        
        dfs(0)
        return ans
