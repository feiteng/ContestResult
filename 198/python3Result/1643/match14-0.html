class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        links = collections.defaultdict(list)
        for a, b in edges:
            links[a].append(b)
            links[b].append(a)
        visited = [False] * n
        res = [0] * n
        def dfs(node):
            if visited[node]:
                # return []
                return collections.defaultdict(int)
            visited[node] = True
            count = collections.defaultdict(int)
            # lis = [labels[node]]
            count[labels[node]] = 1
            for neigh in links[node]:
                # lis.extend(dfs(neigh))
                sub_count = dfs(neigh)
                for k, v in sub_count.items():
                    count[k] += v
            # for c in lis:
            #     if c == labels[node]:
            #         res[node] += 1
            # print(node, lis)
            res[node] = count[labels[node]]
            return count
        dfs(0)
        return res
