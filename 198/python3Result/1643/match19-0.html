class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        ans = [0] * n
        g = defaultdict(list)
        for u, v in edges:
            g[u].append(v)
            g[v].append(u)
        
        vis = set([0])
        def dfs(node):
            count = Counter([labels[node]])
            for nei in g[node]:
                if nei not in vis:
                    vis.add(nei)
                    count += dfs(nei)
            ans[node] = count[labels[node]]
            return count
        dfs(0)
        return ans
