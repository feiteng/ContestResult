from collections import Counter

class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        g = [[] for i in range(n)]
        for u, v in edges:
            g[u].append(v)
            g[v].append(u)
        
        res = [0] * n
        
        def bt(node, p):
            tmp = Counter(labels[node])
            for c in g[node]:
                if c != p:
                    tmp += bt(c, node)
                    
            res[node] = tmp[labels[node]]
            return tmp
            
        bt(0, -1)
        return res
