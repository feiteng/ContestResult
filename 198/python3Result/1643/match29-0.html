class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        d = collections.defaultdict(list)
        for x,y in edges:
            d[x].append(y)
            d[y].append(x)
        self.res = [0] * n
        seen = set()
        def dfs(node):
            if node in seen:
                return {}
            seen.add(node)
            res = collections.Counter()
            res[labels[node]] = 1
            for nxt in d[node]:
                tmp = dfs(nxt)
                # print(node, tmp)
                for k,v in tmp.items():
                    res[k] += v
            self.res[node] = res[labels[node]]
            return res
        dfs(0)
        return self.res
