class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        from collections import defaultdict
        T = defaultdict(set)
        for u, v in edges:
            T[u].add(v)
            T[v].add(u)
        
        output = [1] * n
        visited = [False] * n
        def traversal(root):
            visited[root] = True
            
            profile = [0] * 26
            for v in T[root]:
                if not visited[v]:
                    p = traversal(v)
                    for i in range(26):
                        profile[i] += p[i]
            
            idx = ord(labels[root]) - ord('a')
            profile[idx] += 1
            output[root] = profile[idx]
            return profile
            
        traversal(0)
        return output
