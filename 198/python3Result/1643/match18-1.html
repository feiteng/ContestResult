class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        grid = [[] for _ in range(n)]
        for i, j in edges:
            grid[i].append(j)
            grid[j].append(i)
        res = [0] * n
        def helper(index, pre):
            nonlocal res
            a = [0] * 26
            for i in grid[index]:
                if i == pre:
                    continue         
                b = helper(i, index)
                for j in range(26):
                    a[j] += b[
j]
            ii = ord(labels[index]) - ord('a')
            a[ii] += 1
            res[index] = a[ii]
            # print(index, a)
            return a
        helper(0, -1)
        return res
