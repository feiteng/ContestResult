from sys import *
setrecursionlimit(300000)


class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        
        g = defaultdict(list)
        
        for u,v in edges:
            g[u].append(v)
            g[v].append(u)
        
        ans = [-1] * n
        
        def recurse(u, p):
            
            ret = Counter()
            
            val = labels
[u]
            
            ret[val] += 1
            
            for v in g[u]:
                
                if v == p:
                    continue
                
                ret += recurse(v, u)
            
            ans[u] = ret[val]
            
            return ret
            
        recurse(0, -1)
        return ans
