class Solution:
    def countSubTrees(self, n: int, e: List[List[int]], s: str) -> List[int]:
        fa = [0]*n
        ans = [0]*n
        ed 
= [[] for i in range(n)]
        for ee in e:
            ed[ee[0]].append(ee[1])
            ed[ee[1]].append(ee[0])
        vis = [0]*n
        
def dfs(x):
            vis[x] = 1
            cnt = [0]*26
            cnt[ord(s[x])-97] += 1
            for i in ed[x]:
                if not vis[i]:
                    fa[i] = x
                    subcnt = dfs(i)
                    for i in range(26):
                        cnt[i] += subcnt[i
]
            ans[x] = cnt[ord(s[x])-97]
            return cnt
        dfs(0)
        return ans
