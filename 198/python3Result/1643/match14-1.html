class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        tree = collections.defaultdict(list)
        for a, b in edges:
            tree[a].append(b)
            tree[b].append(a)
        ans = [0] * n
        seen = set()
        def dfs(root: int) -> dict:
            if root in seen:
                return {}
            if root not in tree:
                ans[root] = 1
                return {labels[root]: 1}
            seen.add(root)
            mydict 
= collections.defaultdict(int)
            mydict[labels[root]] += 1
            for child in tree[root]:
                tmp_dict = dfs(child)
                for k, v in tmp_dict.items():
                    mydict[k] += v
            ans[root] = mydict[labels[root]]
            return mydict
        dfs(0)
        return ans
