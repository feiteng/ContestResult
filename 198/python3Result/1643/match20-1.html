from collections import defaultdict

class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        self
.children = defaultdict(list)
        for i,j in edges:
            self.children[i].append(j)
            self.children[j].append(i)
        self.visited = set([0])
        self.result = [0]*n
        self.labels = labels
        self.findCount(0)
        return self.result
        
    
def findCount(self, i):
        c = Counter()
        for j in self.children[i]:
            if j not in self.visited:
                self.visited.add(j)
                c = c + self.findCount(j)
        c[self.labels[i]] += 1
        self.result[i] = c[self.labels[i]]
        return c
