class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        from collections import Counter
        
        res 
= [0] * n
        
        e = defaultdict(list)
        for a, b in edges:
            e[a].append(b)
            e[b].append(a)
        
        visited = {0}
        
def helper(node):
            d = Counter([labels[node]])
            for n in e[node]:
                if n not in visited:
                    visited.add(n)
                    d += helper(n)
            res[node] = d[labels[node]]
            return d
        
        _ = helper(0)
        return res
