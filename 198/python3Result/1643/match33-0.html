class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        
        adj = defaultdict(list)
        for e in edges:
            adj[e[0]].append(e[1])
            adj[e[1]].append(e[0])
        results = [0] * n
        visited = [False] * n
        
        def hash_label(s):
            return ord(s) - ord('a')
        
        def dfs(node):
            visited[node] = True
            
            # # leave node
            # if node not in adj:
            #     stats = [0]*26
            #     stats[hash_label(labels[node])] = 1
            #     results[node] = 1
            #     return stats
            
            nei_stats = [0]*26
            nei_stats[hash_label(labels[node])] += 1
            for nei in adj[node]:
                if not visited[nei]:
                    si = dfs(nei)
                    nei_stats = [nei+sii for nei, sii in zip(nei_stats, si)]
                
            results[node] = nei_stats[hash_label(labels[node])]
                
            return nei_stats
        
        dfs(0)
        

        return results
