from collections import defaultdict
class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        tree = defaultdict(list)
        for a, b in edges:
            tree[a].append(b)
            tree[b].append(a)
        
        ans = [0] * n
        visited = [False] * n
        
        def count(i):
            visited[i] = True
            arr = [0] * 26
            for j in tree[i]:
                if visited[j]:
                    continue
                for idx, c in enumerate(count(j)):
                    arr[idx] += c
            key = ord(labels[i]) - ord('a')
            arr[key] += 1
            ans[i] = arr[key]
            # print(i, key, arr)
            return arr
        
        count(0)
                
        return ans
