class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        eg = collections.defaultdict(list)
        for a, b in edges:
            eg[a].append(b)
            eg[b].append(a)
        
        ans = [0] * n
        def dfs(a, f):
            c = collections.Counter([labels[a]])
            
for b in eg[a]:
                if b != f:
                    c += dfs(b, a)
            ans[a] = c[labels[a]]
            return c
        dfs(0, None)
        return ans
