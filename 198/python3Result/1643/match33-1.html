    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        al = [[] for i in range(n)]
        for i in edges:
            al[i[0]].append(i[1])
            al[i[1]].append(i[0])
        
        p = [0 for i in range(n)]
        done = [False for i in range(n)]
        def f(x):
            main_arr = labels[x]
            # new_path = path[:]+[x]
            done[x] = True
            for i in al[x]:
                if done[i]==False:
                    arr = f(i)
                    main_arr+
=arr
            p[x] = main_arr.count(labels[x])
            return main_arr

        f(0)
        return p
