class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        # 0 is root
        E = defaultdict(list)
        for u, v in edges:
            E[u].append(v)
            E[v].append(u)
        
        ans = [0] * n
        def dfs(u, p):
            chs = [0] * 26
            ch = labels[u]
            chs[ord(ch) - ord('a')] += 1
            
for v in E[u]:
                if v != p:
                    chs1 = dfs(v, u)
                    for j in range(26):
                        chs[j] += chs1[j]
            ans[u] = chs[ord(ch) - ord('a')]
            return chs
                    
        dfs(0, 0)
        return ans
