class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        g = [[] for i in range(n)]
        for a, b in edges:
            g[a].append(b)
            g[b].append(a)
        self.ans = [0 for i in range(n)]
        def dfs(u, fa):
            idx = ord(labels[u]) - ord('a')
            res = [0 for i in range(26)]
            res[idx] += 1
            for v in g[u]:
                if v == fa:
                    continue
                t = dfs(v, u)
                for i in range(26):
                    res[i] += t[i]
            self.ans[u] = res[idx]
            return res
        dfs(0, 0)
        return self.ans
