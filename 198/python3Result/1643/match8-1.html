class Solution:
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        graph = collections.defaultdict(list)
        for n1, n2 in edges:
            graph[n1].append(n2)
            graph[n2].append(n1)
        
        res = [1] * n
        def dfs(i, p):
            nonlocal res, labels
            ans = collections.defaultdict(int)
            for j in graph[i]:
                if j == p: continue
                cur = dfs(j, i)
                for k, v in cur.items():
                    ans[k] +=
 v
            ans[labels[i]] += 1
            res[i] = ans[labels[i]]
            return ans
        dfs(0, -1)
        return res
