from collections import Counter
class Solution:
    def dfs(self, cur):
        ct = Counter([])
        for to in self.toList[cur]:
            if to not in self.visited:
                self.visited.add(to)
                ct += self.dfs(to)
        ct[self.labels[cur]] += 1
        self.ans[cur] = ct[self.labels[cur]]
        return ct
    def countSubTrees(self, n: int, edges: List[List[int]], labels: str) -> List[int]:
        self.labels = labels
        self.ans = [0]*n
        self.toList = [[] for _ in range(n)]
        for s, t in edges:
            self.toList[s].append(t)
            self.toList[t].append(s)
        self.visited = set()
        self.visited.add(0)
        self.dfs(0)
        return self.ans
