class Solution:
    def numWaterBottles(self, numBottles: int, numExchange: int) -> int:
        ans = numBottles
        empty = numBottles
        while empty >= numExchange:
            new_bottle = empty // numExchange
            empty %= numExchange
            ans += new_bottle
            empty += new_bottle
        return ans
