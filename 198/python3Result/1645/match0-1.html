class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        func = lambda x, y: x if y in x else x + [y]
        arr = reduce(func, [[], ] + arr)
        
        ans = float
('inf')
        n = len(arr)
        for l in range(n):
            temp = arr[l]
            for r in range(l, n):
                temp &= arr[r]
                ans = min(ans, abs(temp - target))
                if ans == 0:
                    return ans
                elif temp < target:
                    break
        return ans
