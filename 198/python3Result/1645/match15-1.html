class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        if target in arr:
            return 0
        else:
            m = float('inf')
            arr = list(set(arr))
            n 
= len(arr)
            
            for i in range(n):
                v = arr[i]
                # bool chk = True
                for j in range(i,n):
                    v&=arr[j]
                    m = min(abs(v-target),m)
                    if (m==0):
                        return m
                    # print("m,i,j",m,i,j)
                    if (v<target and abs(v-target)>m):
                        # print(v,abs(v-target),m)
                        break
            return m
