class Solution:
    # O(nlogn)
    def closestToTarget(self, A: List[int], target: int) -> int:
        def compute(counter, x, op):
            k = 0
            while x:
                if op == 1:
                    x, counter[k] = x & counter[k], x ^ counter[k]
                else:
                    x, counter[k] = x & ~counter[k], x ^ counter[k]
                k += 1

        def bitwise_and(counter, l):
            k, res = 0, -1
            while l:
                if l % 2 > 0: 
                    res &= counter[k]
                else: 
                    res &= ~counter[k]
                l //= 2
                k += 1
            return res
        
        counter = [0] * 32
        n = len(A)
        res = float('inf')
        i, j, v = 0, 0, target
        while i < n:
            if v < target or j == n:
                compute(counter, A[i], -1)
                i += 1
            else:
                compute(counter, A[j], 1)
                j += 1
            if j > i: 
                v = bitwise_and(counter, j - i)
                res = min(res, abs(v - target))
            else: 
                v = target
        return res
