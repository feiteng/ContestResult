class Solution:
    def closestToTarget(self, A: List[int], P: int) -> int:
        
        import numpy as np
        
        def add(counter,x):
            k = 0
            while x :
                x, counter[k] = x & counter[k], x ^ counter[k]
                k += 1

        def sub(counter, x):
            k = 0
            while x:
                x, counter[k] = x & ~counter[k], x ^ counter[k]
                k += 1

        def val(counter, count):            
            k = 0
            res 
= -1
            while count:
                if count % 2 > 0: res &= counter[k]
                else: res &= ~counter[k]
                count //= 2
                k += 1
            return res
        
        counter = np.zeros(32, np.int64) # up to 4Go
        counterA = np.zeros(32, np.int64) # up to 4Go
        n = len(A)
        i = j = 0
        K = P
        mini = np.int64(2 ** 63 - 1)

        
while i < n:
            if K < P or j == n:
                sub(counter, A[i])
                i += 1
            else:
                add(counter, A[j])
                j += 1

            if j > i:
                K = val(counter, j - i)
                X = np.abs(K - P)
                if mini > X: mini = X
            else: K = P

        return mini
