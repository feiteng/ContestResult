from typing import List

class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        n = len(arr)
        ans = []
        for i in range(n):
            if i == 0 or arr[i] != arr[i-1]:
                ans.append(arr[i])
        arr = ans.copy()
        n = len(arr)

        p = [[0] * 20 for _ in range(n+1)]
        for i, a in enumerate(arr):
            for j in range(20):
                p[i][j] = p[i - 1][j] + (a >> j & 1)
        
        ans = float('inf')
        l = n - 1
        for i in range(n - 1, -1, -1):
            if i < n-1 and arr[i] == arr[i+1]:
                continue
            def get(j):
                ans = 0
                for k in range(20):
                    if p[i][k] - p[j - 1][k] == i - j + 1:
                       ans |= 1 << k
                return ans

            while l >= 0 and get(l) >= target:
                l -= 1

            if l >= 0:
                ans = min(ans, abs(target - get(l)))
            if l + 1 <= i:
                ans = min(ans, abs(get(l+1) - target))
        return ans
