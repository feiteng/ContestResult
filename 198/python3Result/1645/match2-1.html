class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        arr=list(set(arr))    
        mindiff 
= -1
        for i in range(len(arr)):
            ans = arr[i]
            for j in range(i,len(arr)):
                ans = ans&arr[j]
                #print(arr[i],arr[j],ans)
                if abs(target-ans)<mindiff or mindiff==-1:
                    mindiff = abs(target-ans)
                if ans<target:
                    break
                if ans == target:
                    return 0 

        return mindiff
