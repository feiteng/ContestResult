class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        if target in arr:
            return 0
        old_arr = arr
        arr = [old_arr[0]]
        for i in range(1,len(old_arr)):
            if old_arr[i] != arr[-1]:
                arr.append(old_arr[i])
        best = abs(target - arr[0]) 
        for i in range(len(arr)):
            cur = arr[i]
            best = min(best, abs(target - cur))
            for j in range(i+1,len(arr)):
                cur &= arr[j]
                best = min(best, abs(target - cur))
                if cur < target:
                    break
        return best
