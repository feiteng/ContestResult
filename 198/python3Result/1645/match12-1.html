class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        arr2 = []
        l = arr[0] -1
        for v in arr:
            if v != l:
                arr2.append(v)
                l = v
        #print(len(arr), len(arr2))
        arr = arr2
        r = abs(-1000000000 - target)
        
for i in range(len(arr)):
            ans = arr[i]
            r = min(r, abs(ans - target))
            for j in range(i+1, len(arr)):
                ans = ans & arr[j]
                r = min(r, abs(ans - target))
                if ans == 0 or r == 0:
                    break
                if target > ans and target - ans >= r:
                    break
            if r == 0:
                break
        return r
