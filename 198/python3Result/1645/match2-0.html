class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        a = []
        for i in range(len(arr)):
            if arr[i] not in a:
                a.append(arr[i])
        arr = a
        min = 1000000000000000000000000000000
        for i in range(len(arr)):
            ans = arr[i]
            for j in range(i,len(arr)):
                ans = ans & arr[j]
                if min > abs(ans - target):
                    min = abs(ans - target)
                #print(ans,min)
        return min
