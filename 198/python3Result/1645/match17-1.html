class SegmentTree:

    def __init__(self, nums):
        self.n = len(nums)
        self.array = [0] * (self.n * 2)
        for i in range(self.n):
            self
.array[i + self.n] = nums[i]

        # self.array[0] is not used.
        for i in range(self.n - 1, 0, -1):
            self.array[i] = self.array[i * 2] & self.array[i * 2 + 1]

    def range_sum(self, start, end):
        start += self.n
        end += self.n
        ans = self.array[start]
        while start <= end:
            if start & 1:
                ans = ans & self.array[start]
                start += 1

            end //= 2
        return ans


class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        n = len(arr)
        segment_tree = SegmentTree(arr)
        min_value = float("inf")
        for i in range(n):
            for j in range(i, n):
                val = abs(segment_tree.range_sum(i, j) - target)
                if val < min_value:
                    min_value = val
                else:
                    break
        return min_value
