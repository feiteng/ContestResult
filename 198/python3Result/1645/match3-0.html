NB = 24

class Solution:
    def add(self, c, x):
        k = 0
        while x:
            x, c[k] = x & c[k], x ^ c[k]
            k += 1

    def sub(self, c, x):
        k = 0
        while x:
            x, c[k] = x & ~c[k], x ^ c[k]
            k += 1

    def val(self, c, count):
        k = 0
        res = -1
        while count:
            if count % 2 > 0:
                res &= c[k]
            else:
                res &= ~c[k]

            count //= 2
            k += 1
        return res

    def closestToTarget(self, arr: List[int], target: int) -> int:
        c = [0] * NB
        n = len(arr)
        i = 0
        j = 0
        K = target
        res = 2**NB-1

        while i < n:
            if K < target or j == n:
                self.sub(c, arr[i])
                i += 1
            else:
                self.add(c, arr[j])
                j += 1

            if j > i:
                K = self.val(c, j - i)
                X = abs(K - target)
                if res > X:
                    res = X
            else:
                K = target

        return res
