class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        A = []
        for v in arr:
            if not A or v != A[-1]:
                A.append(v)
        opt = math.inf
        for i in range(len(A)):
            cur = A[i]
            opt = min(abs(cur - target), opt)
            for j in range(i-1, -1, -1):
                cur &= A[j]
                if cur <= target - opt: break
                opt = min(abs(cur - target), opt)
        return opt
