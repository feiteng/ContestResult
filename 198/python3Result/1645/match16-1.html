class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        best = 2**31
        newArr = [arr[0]]
        for a in arr:
            if a != newArr[-1]:
                newArr.append(a)
        for i in range(len(newArr)):
            s = 2**31-1
            for j in range(i, len(newArr)):
                if s==newArr[j]:
                    break
                s = s&newArr[j]
                if abs(target-s) < best:
                    best = abs(target-s)
                    
if best==0:
                        return 0
                if s==0:
                    break
        return best
