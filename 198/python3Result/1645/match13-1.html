import functools

class Solution:
    def func(self, arr, l, r):
        if (r < l):
            return -1000000000
        ans = arr[l]
        for i in range(l+1, r+1):
            ans = ans & arr[i]
        return ans
    
    def closestToTarget(self, arr: List[int], target: int) -> int:
        min_so_far = float('inf')
        
        arr = list(set(arr))
        
        for i in range(len(arr)):
            cur = arr[i]
            flag = False
            for j in reversed(range(i+1)):
                cur = cur & arr[j]
                if flag and abs(cur - target) > min_so_far:
                    break
                elif abs(cur - target) < min_so_far:
                    min_so_far = abs(cur - target)
                    flag = True
                    
        return min_so_far
