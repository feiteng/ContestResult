class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        # v = arr[0]
        # tmp = [v]
        # for n in arr[1:]:
        #     v &= n
        #     tmp.append(v)
            
        
        
        res = math.inf
        
for i in range(len(arr)):
            r = arr[i]
            res = min(res, abs(r - target))
            for j in range(i+1, len(arr)):
                r &= arr[j]
                d = abs(r - target)
                if res > d:
                    res = d
                else:
                    break
        return res
