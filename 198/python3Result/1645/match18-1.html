class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        prexor = [1]
        n = len(arr)
#         for i in range(n):
#             temp = prexor[-1] & arr[i]
#             prexor.append(temp)
        
#         print('prexor ', prexor[1:])
#         for i in range(n):
#             for j in range(i+1, n+1):
                
#                 print(i, j-1, prexor[j] & prexor[i])
#             print()
        
        dp = [0xffffffff] + [0] * (n)
        ans = float('inf')
        
for i in range(1, n+1):
            dp[i] = max([arr[i-1], dp[i-1] & arr[i-1]], key = lambda x: -abs(x-target))
            ans = min(ans, abs(dp[i] - target))
            
        print('dp ', dp)
            
        return ans
