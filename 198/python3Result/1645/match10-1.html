class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        a = -1000000000
        ans = abs(a-target)
        array = []
        s = set()
        for i in arr:
            if not i in s:
                array.append(i)
                s.add(i)
        arr = array
                
        
for i in range(len(arr)):
            k = arr[i]
            ans = min(ans,abs(k-target))
            for j in range(i+1,len(arr)):
                k &= arr[j]
                ans = min(ans,abs(k-target))
        return ans
