class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        arr = list(set(arr))
        ans =None
        n = len(arr)
        for l in range(n):
            temp = arr[l]
            for r in range(l,n):
                temp = temp & arr[r]
                temp2 = abs(temp-target)
                if ans==None or ans>temp2:
                    ans = temp2
                    if ans==0:
                        return(0)
                if temp==0:
                    break
                    
                    
        return(ans)
