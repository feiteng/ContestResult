class Solution:
    def closestToTarget(self, arr: List[int], target: int) -> int:
        res = float('inf')
        tmp = []
        for i, a in enumerate(arr):
            if i == 0 or arr[i] != arr[i - 1]:
                tmp.append(a)
        arr = tmp
        n = len(arr)

        def getMin(b, e):
            if b > e:
                return float('inf')
            if b == e:
                return abs(arr[b] - target)
            m = (b + e) >> 1
            curmin = abs(arr[m] - target)
            if curmin == 0:
                return 0
            cur = arr[m]
            l, r = m-1, m+1
            while l>=b or r <=e:
                if r>e or l>=b and arr[l] >= arr[r]:
                    cur = cur&arr[l]
                    l-=1
                else:
                    cur = cur&arr[r]
                    r+=1
                if cur < target:
                    break
                elif cur == target:
                    return 0
                curmin = min(curmin, abs(cur - target))
            return min(getMin(b, m - 1), getMin(m + 1, e), curmin)
        return getMin(0, n-1)
