class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Map<Integer,List<Integer>> hm = new HashMap<>();
        for(int[] edge : edges) {
            if(!hm.containsKey(edge[0]))
                hm.put(edge[0], new ArrayList<Integer>());
            if(!hm.containsKey(edge[1]))
                hm.put(edge[1], new ArrayList<Integer>());
            hm.get(edge[0]).add(edge[1]);
            hm.get(edge[1]).add(edge[0]);
        }
        Set<Integer> set = new HashSet<Integer>();
        int[] ret = new int[n];
        int[] som = helper(hm, labels, set, ret, 0);
        return ret;
    }
    
    private int[] helper(Map<Integer,List<Integer>> hm, String labels, Set<Integer> set, int[] ret, int curr) {
        set.add(curr);
        int[] arr = new int[26];
        int ind = labels.charAt(curr) - 'a';
        arr[ind]++;
        List<Integer> neis = hm.getOrDefault(curr, new ArrayList<Integer>());
        for(int nei : neis) {
            if(set.contains(nei))
                continue;
            int[] last = helper(hm, labels, set, ret, nei);
            for(int i = 0; i < 26; i++) {
                arr[i] += last[i];
            }
        }
        ret[curr] = arr[ind];
        return arr;
    }
}
