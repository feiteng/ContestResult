class Solution {
    char[] ls;
    int[] ress;
    Set<Integer>[] gra;
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        ls = labels.toCharArray();
        gra = new Set[n];
        ress = new int[n];
        for (int i = 0; i < n; i++) gra[i] = new HashSet<>();
        for (int[] e : edges) {
            gra[e[0]].add(e[1]);
            gra[e[1]].add(e[0]);
        }
        dfs(0, -1);
        return ress;
    }
    int[] dfs(int cur, int par) {
        char tar = ls[cur];
        int[] res = new int[26];
        res[tar - 'a']++;
        for (int nei : gra[cur]) {
            if (nei == par) continue;
            int[] chd = dfs(nei, cur);
            for (int i = 0; i < 26; i++) res[i] += chd[i];
        }
        ress[cur] = res[tar - 'a'];
        return res;
    }
}
