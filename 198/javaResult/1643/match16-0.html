import java.util.ArrayList;

class Solution {

    ArrayList<Integer>[] tree;
    char[] lb;
    
    int[] ans;

    public int[] countSubTrees(int n, int[][] edges, String labels) {

        lb = labels.toCharArray();
        tree = new ArrayList[n];
        ans = new int[n];

        for(int  i =0;i<n;++i){
            tree[i] = new ArrayList<>();
        }

        for (int[] edge : edges) {

            tree[edge[0]].add(edge[1]);
            tree[edge[1]].add(edge[0]);
        }

        dfs(0,-1);
        
        return ans;

    }

    private int[] dfs(int node, int par) {

        int[] arr = new int[26];

        for(int i:tree[node]){

            if(i==par)
                continue;

            int[] temp = dfs(i,node);

            for(int j=0;j<26;++j){
                arr[j]+=temp[j];
            }
        }

        int num = lb[node]-'a';
        arr[num]++;
        ans[node] = arr[num];
        
        return arr;

    }
}
