class Solution {

	final int A = 26;
	
    public int[] countSubTrees(int n, int[][] edges, String labels) {
    	List<Integer>[] adj = (List<Integer>[])new List[n];
    	for(int i=0; i<n; i++) {
    		adj[i] = new ArrayList<>();
    	}
    	for(int[] e : edges ) {
    		int u = e[0];
    		int v = e[1];
    		adj[u].add(v);
    		adj[v].add(u);
    	}
    	
    	char[] labelsArray = labels.toCharArray();
    	int[] result = new int[n];
    	visit(adj, 0, new boolean[n], result, labelsArray);
    	System.out.println(Arrays.toString(result));
    	return result;
    }
    
    private int[] visit(List<Integer>[] adj, int node, boolean[] visited, int[] results, char[] labels) {
    	visited[node] = true;
    	int[] subTreeResults = new int[A];
    	subTreeResults[labels[node]-'a']++;
    	for(int v : adj[node]) {
    		if( visited[v] == false ) {
    			int[] counts = visit(adj, v, visited, results, labels);
    			for(int i=0; i<A; i++) {
    				subTreeResults[i] += counts[i];
    			}
    		}
    	}
    	results[node] = subTreeResults[labels[node]-'a'];
    	return subTreeResults;
    }
}
