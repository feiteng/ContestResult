class Solution {
    
    static int[] dfs(HashMap<Integer,List<Integer>> map,String labels,int node,int[] res,HashSet<Integer> vis)
    {
        int[] arr=new int[26];
        vis.add(node);
        arr[labels.charAt(node)-'a']++;
        for(int i:map.get(node))
        {
            if(!vis.contains(i))
            {
                int[] mid=dfs(map,labels,i,res,vis);
                for(int j=0;j<26;j++)
                {
                    arr[j]+=mid[j];
                }
            }
        }
        int ind=labels.charAt(node)-'a';
        res[node]=arr[ind];
        return arr;
        
    }
    
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        HashMap<Integer,List<Integer>> map=new HashMap<>();
        for(int i=0;i<n;i++)
        {
            map.put(i,new ArrayList<>());
        }
        for(int[] arr:edges)
        {
            map.get(arr[0]).add(arr[1]);
            map.get(arr[1]).add(arr[0]);
        }
        int[] res=new int[n];
        HashSet<Integer> vis=new HashSet<>();
        int[] arr=dfs(map,labels,0,res,vis);
        return res;
    }
}
