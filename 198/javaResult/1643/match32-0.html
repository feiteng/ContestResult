class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Node[] nodes = new Node[n];
        for(int i = 0; i < n; i++){
            nodes[i] = new Node();
            nodes[i].c = labels.charAt(i) - 'a';
        }
        for(int[] e : edges){
            Node a = nodes[e[0]];
            Node b = nodes[e[1]];
            a.adj.add(b);
            b.adj.add(a);
        }
        dfs(nodes[0], null);
        int[] ans = new int[n];
        for(int i = 0; i < n; i++){
            ans[i] = nodes[i].dp[nodes[i].c];
        }
        return ans;
    }
    
    public void dfs(Node root, Node p){
        root.dp[root.c]++;
        for(Node node : root.adj){
            if(node == p){
                continue;
            }
            dfs(node, root);
            for(int i = 0; i <= 'z' - 'a'; i++){
                root.dp[i] += node.dp[i];
            }
        }
    }
}

class Node{
    List<Node> adj = new ArrayList<>();
    int c;
    int[] dp = new int['z' - 'a' + 1];
}
