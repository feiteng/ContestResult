class Solution {
    public int[][] f;
    public ArrayList<Integer>[] e;
    public String tags;

    public void dfs(int u, int fa) {
        f[u][tags.charAt(u) - 'a']++;
        for (int v : e[u]) {
            if (v == fa) continue;
            dfs(v, u);
            for (int i = 0; i < 26; ++i) f[u][i] += f[v][i];
        }
    }

    public int[] countSubTrees(int n, int[][] edges, String labels) {
        f = new int[n][26];
        e = new ArrayList[n];
        tags = labels;
        for (int i = 0; i < n; ++i) e[i] = new ArrayList<>();
        for (int i = 0; i < edges.length; ++i) {
            e[edges[i][0]].add(edges[i][1]);
            e[edges[i][1]].add(edges[i][0]);
        }
        dfs(0, -1);
        int[] ans = new int[n];
        for (int i = 0; i < n; ++i) {
            ans[i]=f[i][tags.charAt(i)-'a'];
        }
        //简化操作，流
        return ans;
    }
}
