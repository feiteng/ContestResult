import java.util.*;


class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Map<Integer, List<Integer>> linkedMap = new HashMap<>();

        for (int i = 0; i < n; i++) {
            linkedMap.put(i, new LinkedList<>());
        }
        for (int[] edge : edges) {
            linkedMap.get(edge[0]).add(edge[1]);
            linkedMap.get(edge[1]).add(edge[0]);
        }

        int[] ans = new int[n];
        backtrack(0, labels, ans, new HashSet<>(), linkedMap);
        
return ans;
    }

    private int[] backtrack(int index, String labels, int[] ans, Set<Integer> visited, Map<Integer, List<Integer>> linkedMap) {
        int[] count = new int[26];
        count[labels.charAt(index) - 'a']++;
        visited.add(index);

        for (Integer childIndex : linkedMap.get(index)) {
            if (!visited.contains(childIndex)) 
{
                int[] childCount = backtrack(childIndex, labels, ans, visited, linkedMap);
                for (int i = 0; i < childCount.length; i++) {
                    count[i] += childCount[i];
                }
            }
        }

        ans[index] = count[labels.charAt(index) - 'a'];
        return count;
    }
}
