class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Map<Integer, List<Integer>> graph = new HashMap();
        for (int i = 0; i < n; i++) graph.put(i, new ArrayList());
        
        for (int[] edge : edges) {
            graph.get(edge[0]).add(edge[1]);
            graph.get(edge[1]).add(edge[0]);
        }
        
        int[] res = 
new int[n];
        
        dfs(graph, labels, 0, -1, res);
        return res;
        
    }
    
    public int[] dfs(Map<Integer, List<Integer>> graph, String labels, int cur, int pre, int[] res) {
        int[] cnts = new int[26];
        int idx = labels.charAt(cur) - 'a';
        cnts[idx]++;
        
        for (int child : graph.get(cur)) {
            if (child == pre) 
continue;
            
            int[] cres = dfs(graph, labels, child, cur, res);
            for (int i = 0; i < 26; i++) {
                cnts[i] += cres[i];
            }
        }
        
        res[cur] = cnts[idx];
        
        return cnts;
    }
}
