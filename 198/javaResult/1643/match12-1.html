class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        List<Integer>[] adj = new List[n];
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }
        for (int[] e : edges) {
            adj[e[0]].add(e[1]);
            adj[e[1]].add(e[0]);
        }
        int[] res = new int[n];
        boolean[] visited = new boolean[n];
        visited[0] = true;
        dfs(0, adj, res, 
labels.toCharArray(), visited);
        return res;
    }
    
    private int[] dfs(int u, List<Integer>[] adj, int[] res, char[] chs, boolean[] visited) {
        int[] cnt = new int[26];

        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                int[] nextCnt = 
dfs(v, adj, res, chs, visited);
                for (int i = 0; i < 26; i++) {
                    cnt[i] += nextCnt[i];
                }                
            }

        }
        res[u] = ++cnt[chs[u] - 'a'];
        return cnt;
    }
}
