class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        HashMap<Integer,HashSet<Integer>> graph=new HashMap<>();
        for(int[] edge:edges){
            graph.putIfAbsent(edge[0],new HashSet<>());
            graph.putIfAbsent(edge[1],new HashSet<>());
            
graph.get(edge[0]).add(edge[1]);
            graph.get(edge[1]).add(edge[0]);
        }
        int[] res=new int[n];
        dfs(0,res,graph,labels,-1);
        return res;
    }
    int[] dfs(int cur,int[] res,HashMap<Integer,HashSet<Integer>> graph,String labels,int pre){
        HashSet<Integer> children=graph.get(cur);
        int[] temp=new int[26];
        temp[labels.charAt(cur)-'a']++;
        if(children!=null){
            for(int c:children)
{
                if(c==pre) continue;
                int[] next=dfs(c,res,graph,labels,cur);
                for(int i=0;i<26;i++)
                    temp[i]+=next[i];
            }
        }
        res[cur]=temp[labels.charAt(cur)-'a'];
        return temp;
    }
}
