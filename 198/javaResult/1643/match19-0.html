class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        List<List<Integer>> map = new ArrayList(n);
        for (int i = 0; i < n; i++) map.add(new ArrayList());
        for (int[] edge : edges) {
            map.get(edge[0]).add(edge[1]);
            map.get(edge[1]).add(edge[0]);
        }
        int[] ret = new int[n];
        boolean[] met = new boolean[n];
        dfs(0, map, labels.toCharArray(), met, ret);
        return ret;
    }
    int[] dfs(int cur, List<List<Integer>> map, char[] labels, boolean[] met, int[] output) {
        met[cur] = true;
        int[] ret = new int[26];
        for (Integer next : map.get(cur)) {
            if (met[next]) continue;
            int[] subret = dfs(next, map, labels, met, output);
            for (int i = 0; i < 26; i++) ret[i] += subret[i];
        }
        ret[labels[cur]-'a']++;
        output[cur] = ret[labels[cur]-'a'];
        return ret;
    }
}
