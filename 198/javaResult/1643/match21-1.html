class Solution {
    int [] res;
    
char[] label;
    ArrayList<Integer> [] neigh;
    boolean[] visited;
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        label = labels.toCharArray();
        res = new int[n];
        neigh = new ArrayList[n];
        visited = new boolean[n];
        visited[0] = true;
        
for(int i = 0; i < n; i++)
            neigh[i] = new ArrayList();
        for(int [] edge : edges)
        {
            neigh[edge[0]].add(edge[1]);
            neigh[edge[1]].add(edge[0]);
        }        
        dfs(0);
        return res;
    }
    int[] dfs(int node){
        int[] cnt = new int[26];
        cnt[label[node] - 'a']++;
        for(int child : neigh[node])
        {
            if(visited[child]) continue;
            visited[child] = true;
            
int[] tmp = dfs(child);
            for(int j = 0; j < 26; j++)
                cnt[j] += tmp[j];
        }
        res[node] = cnt[label[node] - 'a'];
        return cnt;
    }
}
