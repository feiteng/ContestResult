class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        int[] ans = new int[n];
        List<Integer>[] adjList = new List[n];
        char[] lbls = labels.toCharArray();
        for (int i = 0; i < n; i++) {
            adjList[i] = new ArrayList();
        }
        for (int[] e : edges) {
            int u = e[0], v = e[1];
            adjList[u].add(v);
            adjList[v].add(u);
        }
        boolean[] visited = new boolean[n];
        dfs(0, ans, adjList, visited, lbls);
        return ans;
    }
    
    private int[] dfs(int node, int[] ans, List<Integer>[] adjList, boolean[] visited, char[] lbls) {
        char cur = lbls[node];
        visited[node] = true;
        int[] count = new int[26];
        count[cur - 'a']++;
        for (int neigh : adjList[node]) {
            if (visited[neigh]) {
                continue;
            }
            int[] sub = dfs(neigh, ans, adjList, visited, lbls);
            for (int i = 0; i < 26; i++) {
                count[i] += sub[i];
            }
        }
        ans[node] = count[cur - 'a'];
        return count;
    }
}
