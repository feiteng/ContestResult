class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        List<Integer>[] a = new List[n];
        for (int i=0; i<n; i++)
            a[i] = new ArrayList<>();
        for (int i=0; i<edges.length; i++) {
            a[edges[i][0]].add(edges[i][1]);
            a[edges[i][1]].add(edges[i][0]);
        }
        boolean[] v = new boolean[n];
        int[] ans = new int[n];
        count(a, labels, 0, ans, v);
        return ans;
    }
    private int[] count(List<Integer>[] a, String b, int x, int[] c, boolean[] v) {
        int[] ans = new int[26];
        v[x] = true;
        int z = b.charAt(x) - 'a';
        ans[z]++;
        for (int i=0; i<a[x].size(); i++) {
            int y = a[x].get(i);
            if (!v[y]) {
                int[] t = count(a, b, y, c, v);
                for (int j=0; j<26; j++)
                    ans[j] += t[j];
            }
        }
        c[x] = ans[z];
        return ans;
    }
}
