class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Node[] nodes = new Node[n];
        for (int i = 0; i < n; ++i) {
            nodes[i] = new Node(i);
            nodes[i].label = labels.charAt(i) - 'a';
        }
        for (int[] e : edges) {
            int u = e[0];
            int v = e[1];
            nodes[u].next.add(nodes[v]);
            nodes[v].next.add(nodes[u]);
        }
        root(nodes[0], null);
        dfs(nodes[0]);
        
int[] ans = new int[n];
        for (int i = 0; i < n; ++i) {
            ans[i] = nodes[i].freq[nodes[i].label];
        }
        return ans;
    }
    
    private static void dfs(Node u) {
        for (Node v : u.next) {
            dfs(v);
        }
        u.freq = new int[26];
        ++u.freq[u.label];
        for (Node v : u.next) {
            
for (int i = 0; i < 26; ++i) {
                u.freq[i] += v.freq[i];
            }
        }
    }
    
    private static void root(Node u, Node p) {
        u.next.remove(p);
        for (Node v : u.next) {
            root(v, u);
        }
    }
    
    private static class Node {
        public int id;
        public int label;
        public int[] freq;
        public ArrayList<Node> next = new ArrayList();
        
        public Node(int i) {
            id = i;
        }
    }
}
