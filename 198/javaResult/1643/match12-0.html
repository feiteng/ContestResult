class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        List<Integer>[] lEdges = new List[n];
        for(int i=0; i < lEdges.length; i++) {
            lEdges[i] = new ArrayList<>();
        }
        for(int[] e : edges) {
            lEdges[e[0]].add(e[1]);
            lEdges[e[1]].add(e[0]);
        }
        int[] result = new int[n];
        int[][] counter = new int[n][26];
        boolean[] visited = new boolean[n];
        dfs(0, lEdges, result, counter, labels, visited);
        return result;
    }
    
    void dfs(int cur, List<Integer>[] lEdges, int[] result, int[][]counter, String labels, boolean[] visited) {
        int label = labels.charAt(cur) - 'a';
        visited[cur] = true;
        if (lEdges[cur].size() == 0) {
            result[cur] = 1;
            ++counter[cur][label];
            return ;
        }
        counter[cur][label] = 1;
        for(int e : lEdges[cur]) {
            if (! visited[e]) {
                dfs(e, lEdges, result, counter, labels, visited);
                for(int i=0; i < 26; i++) {
                    counter[cur][i] += counter[e][i];
                }
            }
        }
        result[cur] = counter[cur][label];
        
    }
}
