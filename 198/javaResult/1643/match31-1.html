class Solution {
    HashMap<Integer,ArrayList<Integer>> map = new HashMap<>();
    int[][] arr;
    int[] visited;
    public void recurse(int node, String labels){
        visited[node]=1;
        arr[node][labels.charAt(node)-'a']++;
        if(!map.containsKey(node))
            return;
        ArrayList<Integer> list = 
map.get(node);
        
        for(int child : list){
            //arr[child][labels.charAt(child)-'a']++;
            if(visited[child]==1)
                continue;
            recurse(child, labels);
            for(int j=0;j<26;j++)
                arr[node][j]+=arr[child][j];
            //arr[node][labels.charAt(child)-'a']++;
        }
    }
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        visited = new int[n];
        arr = new int[n][26];
        for(int[] edge : edges){
            if(!map.containsKey(edge[0]))
                map.put(edge[0],new ArrayList<>());
            map.get(edge[0]).add(edge[1]);
            
            if(!map.containsKey(edge[1]))
                map.put(edge[1],new ArrayList<>());
            
map.get(edge[1]).add(edge[0]);
        }
        
        recurse(0,labels);
        int res[] = new int[n];
        for(int i=0;i<n;i++){
            res[i] = arr[i][labels.charAt(i)-'a'];
        }
        return res;
    }
}
