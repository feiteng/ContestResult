class Solution {
    int[] res;
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        res = new int[n];
        List<Integer>[] tree = new List[n];
        for(int i=0; i<n; i++){
            tree[i] = new ArrayList();
        }
        for(int[] e : edges){
            
tree[e[0]].add(e[1]);
            tree[e[1]].add(e[0]);
        }
        calc(tree, 0, labels, -1);
        return res;
    }
    
    int[] calc(List<Integer>[] tree, int node, String label, int parent){
        char c = label.charAt(node);
        int[] f = new int[26];
        for(int child : tree[node])
{
            if(child==parent) continue;
            int[] cur = calc(tree, child, label, node);
            for(int i=0; i<26; i++){
                f[i]+=cur[i];
            }
        }
        res[node] = ++f[c-'a'];
        return f;
    }
    
    
}
