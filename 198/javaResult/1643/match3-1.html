class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        List<List<Integer>> graph = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            graph.add(new ArrayList<>());
        
}
        for (int[] edge : edges) {
            graph.get(edge[0]).add(edge[1]);
            graph.get(edge[1]).add(edge[0]);
        }
        int[] res = new int[n];
        dfs(graph, 0, -1, res, labels);
        return res;
    }
    private int[] dfs(List<List<Integer>> graph, int curr, int prev, int[] res, String labels) {
        int[] counts = new int[26];
        for (int next : graph.get(curr)) {
            if (next != prev) {
                int[] nextCounts = dfs(graph, next, curr, res, labels);
                for (int i = 0; i < 26; i++) {
                    counts[i] += nextCounts[i];
                }
            }
        }
        res[curr] = ++counts[labels.charAt(curr) - 'a'];
        return counts;
    }
}
