class Solution {
    
    LinkedList<Integer>[] G;
    int[] ans;
    String labels;
    
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        this.G = new LinkedList[n];
        this.ans = new int[n];
        this.labels = labels;
        
        for (int i = 0; i < n; i++) G[i] = new LinkedList<>()
;
        for (int[] e : edges) {
            G[e[0]].offer(e[1]);
            G[e[1]].offer(e[0]);
        }
        
        postorder(0, -1);
        return ans;
    }
    
    int[] postorder(int i, int p) {
        int[] counts = new int[26];
        for (int u : G[i]) {
            if (u == p) continue;
            int[] counts2 = postorder(u, i);
            
            for (int j = 0; j < counts2.length; j++) {
                counts[j] += counts2[j];
            }
        }
        
        ans[i] = ++counts[labels.charAt(i) - 'a'];
        return counts;
    }
}
