class Solution {
    public void dfs(int cur, int[][] ct, List<Integer>[] ed, boolean[] vis, String labels, int n) {
        vis[cur] = true;
        ct[cur][
labels.charAt(cur) - 'a']++;
        for(int next : ed[cur]) {
            if(vis[next]) continue;
            dfs(next, ct, ed, vis, labels, n);
            for(int i = 0; i < 26; i++) {
                ct[cur][i] += ct[next][i];
            
}
        }
    }
    
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        List<Integer>[] ed = new List[n];
        
for(int i = 0; i < n; i++) ed[i] = new ArrayList<>();
        for(int[] edge : edges) {
            ed[edge[0]].add(edge[1]);
            ed[edge[1]].add(edge[0]);
        }
        int[][] ct = new int[n][26];
        dfs(0, ct, ed, new boolean[n], labels, n);
        
int[] ans = new int[n];
        for(int i = 0; i < n; i++) {
            ans[i] = ct[i][labels.charAt(i) - 'a'];
        }
        return ans;
    }
}
