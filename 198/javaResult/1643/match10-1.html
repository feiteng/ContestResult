class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Set<Integer>[] neighbors = new Set[n];
        for (int i=0; i<n; i++) {
            neighbors[i] = new HashSet<Integer>();
        }
        char[] cc = labels.toCharArray();
        
for (int[] e : edges) {
            neighbors[e[0]].add(e[1]);
            neighbors[e[1]].add(e[0]);
        }
        
        boolean[] visited = new boolean[n];
        
        int[] stat = new int[n];
        
visit(0, neighbors, visited, cc, stat);
        return stat;
    }
    
    private int[] visit(int x, Set<Integer>[] neighbors, boolean[] visited, char[] cc, int[] stat) {
        int[] res = new int[26];
        
        visited[x] = true;
        for (int nn : neighbors[x]) {
            if (!visited[nn]) {
                
int[] tt = visit(nn, neighbors, visited, cc, stat);
                for (int i=0; i<26; i++) {
                    res[i] += tt[i];
                }
            }
        }
        res[cc[x]-'a']++;
        
        stat[x] = res[cc[x]-'a'];
        return res;
    }
}
