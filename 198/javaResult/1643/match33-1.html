class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int i=0; i<n; i++) map.put(i, new ArrayList<>());
        for(int[] e: edges){
            
map.get(e[0]).add(e[1]);
            map.get(e[1]).add(e[0]);
        }
        Set<Integer> visited = new HashSet<>();
        int[] res = new int[n];
        
dfs(map, visited, labels, res, 0);
        return res;
    }
    
    public int[] dfs(Map<Integer, List<Integer>> map, Set<Integer> visited, String labels, int[] res, int node){
        int[] count = new int[26];
        count[labels.charAt(node)-'a'] = 1;
        visited.add(node);
        
        for(int next: map.get(node))
{
            if(visited.contains(next)) continue;
            int[] sub = dfs(map, visited, labels, res, next);
            for(int i=0; i<26; i++){
                count[i] += sub[i];
            }
        }
        res[node] = count[labels.charAt(node)-'a'];
        return count;
    }
}
