class Solution {
    private int[] ans;
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        ans = new int[n];
        List<Set<Integer>> tree = new ArrayList<>();
        for(int i = 0; i < n; i++) {
            tree.add(new HashSet<>());
        }
        for(int[] e: edges) {
            tree.get(e[0]).add(e[1]);
            tree.get(e[1]).add(e[0]);
        }
        dfs(tree, 0, -1, labels);
        
        return ans;
    }
    
    private int[] dfs(List<Set<Integer>> tree, int cur, int p, String labels) {
        int c = labels.charAt(cur)-'a';
        int[] curRet = new int[27];
        curRet[c]++;
        for(Integer nxt: tree.get(cur)) {
            if (nxt == p) continue;
            int[] ret = dfs(tree, nxt, cur, labels);
            for(int i = 0; i < 27; i++) {
                curRet[i] += ret[i];
            }
        }
        ans[cur] = curRet[c];
        return curRet;
    }
}
