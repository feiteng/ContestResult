class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        int[] res = new int[n];
        char[] l = labels.toCharArray();
        
        
List<Integer>[] g = new List[n];
        for (int i = 0; i < n; i++) {
            g[i] = new ArrayList<>();
        }
        for (int[] e : edges) {
            g[e[0]].add(e[1]);
            g[e[1]].add(e[0]);
        }
        
        helper(res, 0, g, l, -1);
        return res;
    }
    
    private int[] helper(int[] r, int cur, List<Integer>[] g, char[] l, int prev) {
        
int[] res = new int[26];
        res[l[cur] - 'a']++;
        r[cur]++;
        for (int next : g[cur]) {
            if (next == prev) continue;
            int[] c = helper(r, next, g, l, cur);
            for (int i = 0; i < 26; i++) {
                res[i] += c[i];
            }
            r[cur] += c[l[cur] - 'a'];
        }

        return res;
    }
}
