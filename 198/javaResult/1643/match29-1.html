class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        
List<Integer>[] adj = new List[n];
        for (int  i =0; i < n; i++) adj[i] = new ArrayList<>();
        
        for (int[] e : edges) {
            adj[e[0]].add(e[1]);
            adj[e[1]].add(e[0]);
        }
        
        boolean[] visited = new boolean[n];
        int[] res = new int[n];
        
dfs(adj, labels, 0, visited, res);
        
        return res;
    }
    
    private int[] dfs(List<Integer>[] adj, String labels, int start, boolean[] visited, int[] res) {
        visited[start] = true;
        
        int[] counts = new int[26];
        for (int con : adj[start]) {
            if (visited[con]) continue;
            int[] t = dfs(adj, labels, con, visited, res);

            for (int i = 0; i < counts.length; i++) {
                counts[i] += t[i];
            }
        }
        
        counts[labels.charAt(start) - 'a']++;
        res[start] = counts[labels.charAt(start) - 'a'];
        
        return counts;
    }
}
