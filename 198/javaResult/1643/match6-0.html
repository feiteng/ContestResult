class Solution {
   public int[] countSubTrees(int n, int[][] edges, String labels) {
    List<List<Integer>> g = new ArrayList<>();
    for (int i = 0; i != n; i++)
      g.add(new ArrayList<>());
    for (int[] e : edges) {
      g.get(e[0]).add(e[1]);
      g.get(e[1]).add(e[0]);
    }
    int[] res = new int[n];
    helper(g, 0, -1, labels, res);
    return res;
  }

  private int[] helper(List<List<Integer>> g, int now, int from, String labels, int[] res) {
    int[] counts = new int[26];
    for (int next : g.get(now)) {
      if (next == from)
        continue;
      int[] kidz = helper(g, next, now, labels, res);
      for (int i = 0; i != 26; i++)
        counts[i] += kidz[i];
    }
    res[now] = ++counts[labels.charAt(now) - 'a'];
    return counts;
  }
}
