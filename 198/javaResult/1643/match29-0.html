class Solution {
    public int[] countSubTrees(int n, int[][] edges, String labels) {
        int[] res = new int[n];
        List<Integer>[] adj = new List[n];
        for (int i = 0; i < n; ++i) adj[i] = new LinkedList();
        for (int[] e : edges) {
            adj[e[0]].add(e[1]);
            adj[e[1]].add(e[0]);
        }
        boolean[] used = new boolean[n];
        dfs(0, res, adj, labels.toCharArray(), used);
        return res;
    }
    private int[] dfs(int root, int[] res, List<Integer>[] adj, char[] cs, boolean[] used) {
        used[root] = true;
        int[] ans = new int[26];
        for (int i : adj[root]) {
            if (used[i]) continue;
            int[] cnt = dfs(i, res, adj, cs, used);
            for (int j = 0; j < 26; ++j) ans[j] += cnt[j];
        }
        ans[cs[root] - 'a']++;
        res[root] = ans[cs[root] - 'a'];
        return ans;
    }
}
