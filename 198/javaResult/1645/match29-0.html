class Solution {
    public int closestToTarget(int[] arr, int target) {
        int[][] next = new int[arr.length][20];
        for (int i = arr.length - 1; i >= 0; i--) {
            if (i == arr.length - 1) {
                Arrays.fill(next[i], arr.length);
            } else {
                for (int j = 0; j < 20; j++) {
                    next[i][j] = next[i + 1][j];
                }
            }
            for (int j = 0; j < 20; j++) {
                if (((arr[i] >> j) & 1) == 0) {
                    next[i][j] = i;
                }
            }
        }
        int res = (int) 1e8;
        for (int i = 0; i < arr.length; i++) {
            int tmp = arr[i];
            int j = i;
            while (j < arr.length) {
                res = Math.min(res, Math.abs(tmp - target));
                int len = arr.length;
                for (int k = 0; k < 20; k++) {
                    if (((tmp >> k) & 1) == 0) {
                        continue;
                    }
                    len = Math.min(len, next[j][k]);
                }
                j = len;
                if (j < arr.length) {
                    tmp &= arr[j];
                }
            }
        }
        return res;
    }
}
