class Solution {
  public int closestToTarget(int[] arr, int target) {
    final int N = arr.length;
    final TreeNode ROOT = new TreeNode(0, N - 1);
    for (int i = 0; i != N; i++)
      ROOT.build(i, i, arr[i]);

    int res = Integer.MAX_VALUE;
    for (int j = N - 1; j != -1; j--) {
      int idx = floor(ROOT, target, j);
      res = Math.min(res, Math.abs(target - ROOT.query(idx, j)));
      if (idx + 1 <= j)
        res = Math.min(res, 
Math.abs(ROOT.query(idx + 1, j) - target));
      if (res == 0)
        return res;        
    }
    return res;
  }

  private int floor(final TreeNode ROOT, final int TAR, final int R) {
    int l = 0, r = R;
    while (l <= r) {
      int m = (l + r) >>> 1;
      int val = ROOT.query(m, R);
      if (val > TAR)
        r = m - 1;
      else
        l = m + 1;
    }
    return r;
  }

  
private static class TreeNode {
    private final int start, mid, end;
    private int and;
    private TreeNode left, right;

    private TreeNode(int s, int e) {
      start = s;
      end = e;
      mid = (s + e) >>> 1;
    
}

    private int build(int f, int t, int delta) {
      if (f > t)
        return and;
      if (f == start && t == end)
        return and = delta;
      if (left == null) {
        left = new TreeNode(start, mid);
        right = new TreeNode(mid + 1, end);
      }
      return and = left.build(Math.max(f, start), Math.min(t, mid), delta)
          & right.build(Math.max(f, mid + 1), Math.min(t, end), delta);
    }

    
private int query(int f, int t) {
      if (f > t)
        return -1;
      if (f == start && t == end)
        return and;
      if (left == null) {
        left = new TreeNode(start, mid);
        right = new TreeNode(mid + 1, end);
      }
      return left.query(Math.max(f, start), Math.min(t, mid))
          & right.query(Math.max(f, mid + 1), Math.min(t, end));
    }
  }
}
