class Solution
{
    
private void addStat(int[][] stat, int n)
    {
        for (int i = 0; i < stat[0].length; i++)
        {
            stat[(n >> i) & 1][i]++;
        }
    }

    private void subStat(int[][] stat, int n)
    {
        for (int i = 0; i < stat[0].length; i++)
        {
            stat[(n >> i) & 1][i]--;
        }
    }

    
private int getNum(int[][] stat)
    {
        int res = 0;
        for (int i = stat[0].length - 1; i >= 0; i--)
        {
            res <<= 1;
            if (stat[0][i] == 0 && stat[1][i] > 0) res |= 1;
        }
        return res;
    }

    public int closestToTarget(int[] arr, int target)
    {
        int ans = Integer.MAX_VALUE;
        int[][] stat = new int[2][29];
        for (int s = 0, e = 0; s < arr.length &&  e < arr.length; e++)
        {
            addStat(stat, arr[e]);
            int x = getNum(stat);
            ans = Math.min(ans, Math.abs(x - target));
            while (x < target && s <= e)
            {
                subStat(stat, arr[s++]);
                x = getNum(stat);
                ans = Math.min(ans, Math.abs(x - target));
            }
        }
        return ans;
    }
}
