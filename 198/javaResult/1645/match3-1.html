class Solution {
    public int closestToTarget(int[] arr, int target) {
        int m = 20;
        
int n = arr.length;
        int[][] prev = new int[m][n];
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                
if(j == 0){
                    prev[i][j] = -1;
                }else{
                    prev[i][j] = prev[i][j - 1];
                }
                if(get(arr[j], i) == 0){
                    prev[i][j] = j;
                }
            
}
        }
        
        int ans = Integer.MAX_VALUE;
        for(int i = 0; i < n; i++){
            int val = arr[i];
            int j = i;
            ans = Math.min(ans, Math.abs(val - target));
            while(j >= 0){
                int last = -1;
                for(int k = 0; k < m; k++){
                    if(get(val, k) == 0){
                        continue;
                    }
                    last = Math.max(last, prev[k][j]);
                }
                j = last;
                if(j >= 0){
                    val &= arr[j];
                    ans = Math.min(ans, Math.abs(val - target));
                
}
            }
        }
        
        return ans;
    }
    
    public int get(int x, int i){
        return (x >> i) & 1;
    }
}
