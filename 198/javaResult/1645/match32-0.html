class Solution {
    private int func(int[] arr, int l, int r) {
        if (r < l)
            return -1000000000;
        int ans = arr[l];
        for (int i = l+1; i < r; i++) {
            ans &= arr[i];
        }
        return ans;
    }
    
    private int[] modify(int[] arr) {
        LinkedList<Integer> l = new LinkedList<>();
        l.offerLast(arr[0]);
        for (int i = 1; i < arr.length; i++) {
            if (arr[i] == l.peekLast()) {
                continue;
            }
            l.offerLast(arr[i]);
        }
        int[] newArr = new int[l.size()];
        for (int i = 0; i < newArr.length; i++) {
            newArr[i] = l.pollFirst();
        }
        return newArr;
    }
    
    public int closestToTarget(int[] arr, int target) {
        arr = modify(arr);
        
        int min = Integer.MAX_VALUE;
        int[] calc = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            calc[i] = -1;
        }
        int minJ = 0;
        for (int i = 0; i < arr.length; i++) {
            for (int j = minJ; j <= i; j++) {
                if (calc[j] == -1)
                    calc[j] = arr[i];
                else {
                    calc[j] &= arr[i];
                }
                if (Math.abs(calc[j] - target) < min) {
                    min = Math.abs(calc[j] - target);
                    // minJ = j;
                }
                if (calc[j] < target) {
                    minJ = j;
                }
                if (min == 0) {
                    return 0;
                }
                //System.out.println(String.format("i=%d, j=%d, min=%d", i, j, min));
            }
        }
        return min;
    }
}
