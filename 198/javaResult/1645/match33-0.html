class Solution {
    public int closestToTarget(int[] arr, int target) {
        int l=0;
        int ans=Math.abs(target+1000000000);
        int[] cnt=new int[22];
        int sum=0;
        for(int i=0;i<22;i++){
            sum+=1<<i;
        }

        for(int r=0;r<arr.length;r++){
            int cur=arr[r];
            for(int i=0;i<22;i++){
                if(cur%2==0){
                    cnt[i]++;
                    if(cnt[i]==1){
                        sum-=1<<i;
                    }
                }
                cur/=2;
            }
            
            if(r>=l){
                ans=Math.min(ans,Math.abs(sum-target));
            }
            while(sum<target){
                cur=arr[l];l++;
                for(int i=0;i<22;i++){
                    if(cur%2==0){
                        cnt[i]--;
                        if(cnt[i]==0){
                            sum+=1<<i;
                        }
                    }
                    cur/=2;
                }
            }
            if(r>=l){
                ans=Math.min(ans,Math.abs(sum-target));
            }
        }

        return ans;
    }
