class Solution {
    public int closestToTarget(int[] arr, int target) {
        int res=Integer.MAX_VALUE;
        for(int i=0;i<arr.length;i++)
        {
            int num=arr[i];
            for(int j=i;j<arr.length;j++)
            {
                num=num&arr[j];
                if(Math.abs(num-target)>=res)break;
                res=Math.abs(num-target);
            }
        }
        return res;
    }
}
