class Solution {
    public int closestToTarget(int[] arr, int target) {
        int res = Math.abs(arr[0] - target);
        int n = arr.length;
        Set<Integer> visited = new HashSet<>();
        for (int i = 0; i < n; i++) {
            int curr = arr[i];
            for (int j = i; j < n; j++) {
                curr &= arr[j];
                if (visited.contains(curr)) {
                    break;
                }
                visited.add(curr);
                res = Math.min(Math.abs(curr - target), res);
                if (curr < target) {
                    break;
                }
            }
        }
        return res;
    }
}
