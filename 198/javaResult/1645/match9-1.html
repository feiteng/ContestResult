class Solution {
    public int closestToTarget(int[] arr, int target) {
        int abs = Integer.MAX_VALUE;
        int n = arr.length;
        for(int l = 0; l < n; l++){
            int ans = 0xffffffff;
            for(int r = l; r < n; r++){
                ans &= arr[r];
                
abs = Math.min(abs, Math.abs(target - ans));
                if(ans <= target) break;
            }
            if(ans > target) break;

        }
        return abs;
    }
}
