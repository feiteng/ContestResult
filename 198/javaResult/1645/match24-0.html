class Solution {
    public int closestToTarget(int[] arr, int target) {
        int min = Integer.MAX_VALUE;
        int[] sign = new int[30];
        and(sign, arr[0]);
        int i = 0, j = 1;
        while (j <= arr.length) {
            int res = cal(sign);
            if (i == j || res > target) {
                if (i != j)
                    min = Math.min(min, res - target);
                if (j == arr.length) break;
                and(sign, arr[j++]);
            } else if (res < target) {
                min = Math.min(min, target - res);
                removeAnd(sign, arr[i++]);
            } else return 0;
        }
        return min;
    }
    int cal(int[] a) {
        int res = 0;
        for (int i = 0; i < 30; i++) {
            if (a[i] == 0) res += (1 << i);
        }
        return res;
    }
    void and(int[] a, int k) {
        for (int i = 0; i < 30; i++) {
            a[i] +=  1 - ((k >> i) & 1);
        }
    }
    void removeAnd(int[] a, int k) {
        for (int i = 0; i < 30; i++) {
            a[i] -=  1 - ((k >> i) & 1);
        }
    }
}
