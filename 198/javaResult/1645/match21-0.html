class Solution {
    public int closestToTarget(int[] arr, int target) {
        int res = Math.abs(arr[0] - target);
        int and = arr[0];
        for (int i : arr) {
            and &= i;
        }
        if (and > target) {
            return Math.abs(and - target);
        }
        for (int i = 0; i < arr.length; i++) {
            int cur = arr[i];
            res = Math.min(res, Math.abs(target - cur));
            for (int j = i + 1; j < arr.length; j++) {
               
                if (cur <= target) {
                    res = Math.min(res, Math.abs(target - cur));
                    break;
                } else {
                    cur &= arr[j];
                }
            }
        }
        return res;
    }
}
