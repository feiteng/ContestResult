class Solution {
    public int closestToTarget(int[] arr, int target) {

        HashSet<Integer> set = new HashSet<Integer>();
   
        
        HashSet<Integer> tmp1 = new HashSet<Integer>();
        int len = arr.length;
        for(int i=0;i<len;i++){
            HashSet<Integer> tmp2 = new HashSet<Integer>();
            tmp2.add(arr[i]);
            set.add(arr[i]);
            for(int num: tmp1){
                tmp2.add(arr[i]&num);
                set.add(arr[i]&num);
            }
            
            tmp1 = tmp2;
        
}
        
        int ans = 1000000000;
        for(int num: set){
            ans = Math.min(ans, Math.abs(num-target));
        }
        
        return ans;
    }
}
