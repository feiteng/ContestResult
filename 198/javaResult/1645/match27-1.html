class Solution {

    // 1000
    
    public int closestToTarget(int[] arr, int target) {
        int min = Integer.MAX_VALUE;
        Set<Integer> set = new HashSet<>();
        
        int N = arr.length;
        for (int l = 0; l < N; l++) {
            int ans = arr[l];
            
            int prev = Math.abs(ans - target);
            
            
for (int r = l; r < N; r++) {
                ans &= arr[r];
                if (set.contains(ans)) {
                    break;
                }
                set.add(ans);
                
                int current = Math.abs(ans - target);
                if (current > prev) {
                    break;
                }
                prev = current;
                
                min = Math.min(min, current);
                if (min == 0) {
                    return min;
                
}
            }
        }
        
        return min;
    }
}
