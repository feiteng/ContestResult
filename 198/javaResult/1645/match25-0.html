class Solution {
    public int closestToTarget(int[] arr, int target) {
        int[][] prefix = new int[arr.length + 1][32];
        for (int i = 1; i <= arr.length; i++) {
            for (int j = 0; j < 32; j++) {
                prefix[i][j] = prefix[i - 1][j];
                if ((1 & (arr[i - 1] >> j)) == 0) {
                    prefix[i][j] += 1;
                }
            }
        }
        int global = Math.abs(1_000_000_000 - target);
        for (int i = 0; i < prefix.length; i++) {
            int lo = i + 1, hi = prefix.length - 1;
            while (lo < hi) {
                int mid = lo + (hi - lo) / 2;
                // System.out.println(hi + " # " + mid + " # " + lo);
                int cur = func(prefix, i, mid) - target;
                global = Math.min(global, Math.abs(cur));
                if (cur == 0) {
                    return 0;
                } else if (cur > 0) {
                    lo = mid + 1;
                } else {
                    hi = mid;
                }
            }
            // for (int j = i + 1; j < prefix.length; j++) {
            //     int cur = func(prefix, i, j);
            //     min = Math.min(min, Math.abs(cur - target));
            //     if (cur == 0) {
            //         break;
            //     }
            // }
            if (global == 0) {
                return 0;
            }
        }
       for (int i = 0; i < prefix.length; i++) {
            int lo = i + 1, hi = prefix.length;
            while (lo < hi) {
                int mid = lo + (hi - lo) / 2;
                int cur = target - func(prefix, i, mid);
                global = Math.min(global, Math.abs(cur));
                if (cur == 0) {
                    return 0;
                } else if (cur > 0) {
                    lo = mid + 1;
                } else {
                    hi = mid;
                }
            }
            if (global == 0) {
                return 0;
            }
        }
        return global;
    }
    
    private int func(int[][] prefix, int i, int j) {
        int res = 0;
        for (int k = 0; k < 32; k++) {
            if (prefix[j][k] == prefix[i][k]) {
                res |= (1 << k);
            }
        }
        return res;
    }
}
