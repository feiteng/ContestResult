class Solution {
    public int closestToTarget(int[] arr, int target) {
        int n = arr.length;
        
int ans=Integer.MAX_VALUE;
        HashSet<Integer> set = new HashSet<>();
        for (int i=0;i<n;i++){
            int val = Integer.MAX_VALUE;
            for(int j=i;j<n&&val>0;j++){
                val=val&arr[j];
                
if(set.contains(val)){
                    break;
                }
                set.add(val);
                ans=Math.min(ans,Math.abs(val-target));
            }
        }
        return ans;
    }
}
