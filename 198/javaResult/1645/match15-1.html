class Solution {
    public static int closestToTarget(int[] arr, int target) {
        int n = arr.length;

        
int bits[][] = new int[n+1][20];

        for (int i=0; i<n; i++) {
            for (int j=0; j<20; j++) {
                bits[i+1][j] = bits[i][j] + ((arr[i] >> j) & 1);
            
}
        }

        int min = Math.abs(-1000000000 - target);
        for (int i=1; i<=n; i++) {
            int s = i;
            int e = n+1;

            while (s < e) {
                int mid = (s + e) / 2;

                int mid_val = getVal(bits, s, mid);

                if (mid_val == target) {
                    return 0;
                } else {
                    min = Math.min(min, Math.abs(mid_val - target));
                    
if (mid_val > target) {
                        s = mid + 1;
                    }  else {
                        e = mid;
                    }
                }
            }
        
}

        return min;
    }

    public static int getVal(int bit[][], int s, int e) {
        int res = 0;
        for (int i=0; i<20; i++) {
            if (bit[e][i] - bit[s-1][i] == (e-s+1)) {
                res += (1<<i);
            }
        }

        return res;
    }
}
