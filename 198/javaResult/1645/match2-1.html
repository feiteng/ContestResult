class Solution {
    public int closestToTarget(int[] arr, int target) {
        int len = arr.length;
        int[][] pre = new int[len][32];
        for(int i = 0; i < len; i ++){
            for(int j = 0; j < 32; j ++){
                int d = arr[i]&1;
                pre[i][32-j-1] = d;
                arr[i] = arr[i] >> 1;
            }
        }
        //
        
int[][] dp = new int[len+1][32];
        for(int i = 1; i <= len; i ++){
            for(int j = 0; j < 32; j ++){
                dp[i][j] = dp[i-1][j]+pre[i-1][j];
            }
        }
        int min = Integer.MAX_VALUE;
        
for(int i = 0; i < len; i ++){
            int left = i;
            int right = len-1;
            //二分
            while(left <= right){
                int mid = left+(right-left)/2;
                int value = helper(dp,i,mid);
                min = Math.min(min,Math.abs(value-target));
                if(value == target) return 0;
                else 
if(value < target){
                    right = mid-1;
                }else{
                    left = mid+1;
                }
            }
        
}
        return min;

    }
    private int helper(int[][] dp, int left, int right){
        int res = 0;
        for(int i = 0; i < 32; i ++){
            int t = 0;
            int num = dp[right+1][i]-dp[left][i];
            if(num == (right-left+1)){
                t = 1;
            }
            res = (res << 1)+t;
        }
        return res;
    }
}
