class Solution {
    public int closestToTarget(int[] arr, int target) {
        int r  = 1000000000;
        for(int i=0;i<arr.length;++i){
            if(i>0&&arr[i]==arr[i-1]) continue;
            int cur = arr[i];
            int j = i;
        
            while(j<arr.length&&(arr[j]&cur)>=target){
                cur &= arr[j];
                r = Math.min(r,Math.abs(target-cur));
                j++;
            
}
            if(j<arr.length){
                cur &= arr[j];
                r = Math.min(r,Math.abs(target-cur));
                
            }
            
            
        }
        return r;
    }
}
