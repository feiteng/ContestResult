class Solution {
    public int closestToTarget(int[] arr, int target) {
        int n = arr.length;
        int ans = Integer.MAX_VALUE;
        
for (int i = 0; i < n; ++i) {
            int number = arr[i];
            ans = Math.min(ans, Math.abs(number - target));
            for (int j = i + 1; j < n; ++j) {
                int last = Math.abs(number - target);
                number &= arr[j];
                int curr = Math.abs(number - target);
                ans = Math.min(ans, curr);
                if (curr >= last) break
;
            }
        }
        return ans;
    }
}
