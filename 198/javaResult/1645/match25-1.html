class Solution {
    public int closestToTarget(int[] arr, int target) {

        int n = arr.length;
        
        int[][] pref = new int[n][32];
        for(int i = 0; i < n; i ++){
            int cur = arr[i];
            for(int j = 0; j < 32; j ++){
                int digit = cur&1;
                pref[i][32-j-1] = digit;
                cur = cur >> 1;
            }
        }
        
int[][] cal = new int[n+1][32];
        for(int i = 1; i <= n; i ++){
            for(int j = 0; j < 32; j ++){
                cal[i][j] = cal[i-1][j]+pref[i-1][j];
            }
        }
        int min = 1000000000;
        
for(int i = 0; i < n; i ++){
            int left = i;
            int right = n-1;
            while(left <= right){
                int mid = (left+right) >>> 1;
                int value = calculate(cal,i,mid);
                min = Math.min(min,Math.abs(value-target));
                if(value == target) return 0;
                else 
if(value < target){
                    right = mid-1;
                }else{
                    left = mid+1;
                }
            }
        
}
        return min;


    }
    private int calculate(int[][] cal, int left, int right){
        int ret = 0;
        for(int i = 0; i < 32; i ++){
            int temp = 0;
            int num = cal[right+1][i]-cal[left][i];
            if(num == (right-left+1)){
                temp = 1;
            }
            ret = (ret << 1)+temp;
        }
        return ret;
    }
    
}
