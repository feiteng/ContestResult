class Solution {
    public int closestToTarget(int[] arr, int target) {
        int n = arr.length;
        int res = Integer.MAX_VALUE;
        for (int i = 0; i < n; i++) {
            int num = arr[i];
            res = Math.min(res, Math.abs(num - target));
            if (res == 0) return 0;
            if (num < target) continue;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] == arr[j-1]) break;
                num &= arr[j];
                res = Math.min(res, Math.abs(num - target));
                if (res == 0) return 0;
                
if (num < target) break;
            }
        }
        return res;
    }
    
    private int func(int[] arr, int l, int r) {
        if (r < l) return -1000000000;
        int ans = arr[l];
        for (int i = l + 1; i <= r; i++) {
            ans &= arr[i];
        }
        return ans;
    }
}
