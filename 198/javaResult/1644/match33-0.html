class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        int[] first = new int[26];
        int[] last = new int[26];
        int n = s.length();
        Arrays.fill(first, n);
        Arrays.fill(last, -1);
        for (int i = 0; i < n; i++) {
            int c = s.charAt(i) - 'a';
            first[c] = Math.min(first[c], i);
            last[c] = Math.max(last[c], i);
        }
        List<int[]> pairs = new ArrayList<>();
        for (int c = 0; c < 26; c++) {
            if (first[c] == n) {
                continue;
            }
            int start = first[c];
            int end = last[c];
            for (int i = start; i <= end; i++) {
                int d = s.charAt(i) - 'a';
                if (first[d] < start) {
                    start = first[d];
                    break;
                }
                end = Math.max(last[d], end);
            }
            if (start == first[c]) {
                pairs.add(new int[]{start, end});
            }
        }
        Collections.sort(pairs, (a, b) -> a[0] - b[0]);
        List<String> res = new ArrayList<>();
        for (int i = 0; i < pairs.size(); i++) {
            int[] curr = pairs.get(i);
            if (i == pairs.size() - 1 || curr[1] < pairs.get(i+1)[0]) {
                res.add(s.substring(curr[0], curr[1]+1));
            }
        }
        return res;
    }
}
