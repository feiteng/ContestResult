class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        int[] min=new int[26];
        int[] max=new int[26];
        Arrays.fill(min,-1);
        Arrays.fill(max,-1);
        for(int i=0;i<s.length();i++){
            if(min[s.charAt(i)-'a']==-1) min[s.charAt(i)-'a'] = i;
            max[s.charAt(i)-'a'] = i;
        }
        for(int j=0;j<26;j++){
            int l=min[j],r=max[j],ln=l,rn=r;
            do{
                if(rn==-1)break;
                int lnn=-1,rnn=-1;
                for(int i=ln;i<=rn;i++){
                    if(lnn==-1)lnn=min[s.charAt(i)-'a'];
                    else lnn=Math.min(min[s.charAt(i)-'a'],lnn);
                    rnn=Math.max(rnn,max[s.charAt(i)-'a']);
                }
                l=ln;
                r=rn;
                rn=rnn;
                ln=lnn;
            }while(l!=ln||r!=rn);
            min[j]=l;
            max[j]=r;
            // System.out.println(Arrays.toString(min));
            // System.out.println(Arrays.toString(max));
        }
        // System.out.println(Arrays.toString(min));
        // System.out.println(Arrays.toString(max));

        int p=-1;
        List<String> ans=new LinkedList();
        for(int i=0;i<s.length();i++){
            if(max[s.charAt(i)-'a']==i && min[s.charAt(i)-'a']>p){
                ans.add(s.substring(min[s.charAt(i)-'a'],i+1));
                p=i;
            }
        }
        return ans;

    }
}
