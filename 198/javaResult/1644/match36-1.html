class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        List<String> res = new ArrayList<>();
        int pos[] = new int[26];
        Arrays.fill(pos, -1);
        for (int i = 0; i < s.length(); ++i) {
            pos[s.charAt(i) - 'a'] = i;
        }
        int[] end = new int[26];
        Arrays.fill(end, -1);
        for (int i = s.length() - 1; i >= 0; --i) {
            end[
s.charAt(i) - 'a'] = i;
        }
        
        int[][] cnt = new int[26][2];
        
        for (int i = 0; i < 26; ++i) {
            cnt[i][0] = end[i];
            cnt[i][1] = pos[i];
        }
        
        for (int i = 0; i < 26 * 2; ++i) {
                int j = i % 26;
                int[] tmp = cnt[j].clone();
                for (int k = cnt[j][0] + 1; k < cnt[j][1]; ++k) {
                    char c = s.charAt(k);
                    tmp[0] = Math.min(tmp[0], cnt[c-'a'][0]);
                    tmp[1] = Math.max(tmp[1], cnt[c-'a'][1]);
                }
                cnt[j] = tmp;
        }
        Arrays.sort(cnt, (a, b)-> {
            return a[1]-b[1];
        });
        
        // for (int[] it : cnt) {
        //     for (int i : it) {
        //         System.out.print(i + " ");
        //     }
        //     System.out.println();
        // }
        
        int prev = -1;
        for (int[] it : cnt) {
            
if (prev < it[0]) {
                res.add(s.substring(it[0], it[1] + 1));
                prev = it[1];
            }
        }
        return res;
    }
}
