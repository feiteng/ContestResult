class Solution {
    class Intv {
        int s;
        int e;
    }
    public List<String> maxNumOfSubstrings(String s) {
        char[] chs = s.toCharArray();
        int[] sA = new int[26];
        int[] eA = new int[26];
        Arrays.fill(sA, -1);
        Arrays.fill(eA, -1);
        for(int i = 0; i < s.length(); ++i) {
            char x = s.charAt(i);
            if(sA[x - 'a'] == -1) sA[x - 'a'] = i;
        }
        for(int i = s.length() - 1; i >= 0; --i) {
            char x = s.charAt(i);
            if(eA[x - 'a'] == -1) eA[x - 'a'] = i;
        }
        ArrayList<Intv> lx = new ArrayList<>();
        for(int sx : sA) {
            if(sx == -1) continue;
            char cx = chs[sx];
            int ex = eA[cx - 'a'];
            int id = sx + 1;
            boolean f = true;
            while(id <= ex) {
                char cy = chs[id];
                if(sA[cy - 'a'] < sx) {
                    f = false;
                    break;
                }
                if(eA[cy - 'a'] > ex) {
                    ex = eA[cy - 'a'];
                }
                ++id;
            }
            if(f) {
                Intv ni = new Intv();
                ni.s = sx;
                ni.e = ex;
                lx.add(ni);
            }
        }
        List<String> res = new ArrayList<>();
        Collections.sort(lx, (a, b) -> a.e - b.e);
        int rb = lx.get(0).e;
        res.add(s.substring(lx.get(0).s, rb + 1));
        int id = 1;
        while(id < lx.size()) {
            Intv ix = lx.get(id);
            if(ix.s > rb) {
                rb = ix.e;
                 res.add(s.substring(ix.s, rb + 1));
            }
            ++id;
        }
        return res;
    }
}
