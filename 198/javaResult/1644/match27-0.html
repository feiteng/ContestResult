class Solution {
    int[] t = new int[26];
    int[] mi = new int[26];
    public List<String> maxNumOfSubstrings(String s) {
        Arrays.fill(mi, 5555555);
        for(int i = 0 ; i < s.length() ; i++){
            t[s.charAt(i)-'a'] = i;
            mi[s.charAt(s.length()-i-1)-'a'] = s.length()-i-1;
        }
        System.out.println(Arrays.toString(getDuration('c' , s)));
        int[][] du = new int[26][2];
        for(int i = 0 ; i < 26 ; i++){
            du[i] = getDuration((char)('a'+i) , s);
        }
        Arrays.sort(du , (a,b)->(a[1]- b[1] == 0 ? b[0] - a[0] : a[1] - b[1]));
        List<int[]> list = new ArrayList<>();
        int m = -1;
        for(int[] t : du){
            if(t[0] == -1) continue;
            System.out.println(Arrays.toString(t));
            if(t[0] > m){
                m = t[1];
                list.add(t);
            }
        }
        List<String> ans = new ArrayList<>();
        for(int[] t : list){
            ans.add(s.substring(t[0] , t[1]+1));
        }
        return ans;
    }
    public int[] getDuration(char c , String s){
        for(int i = 0 ; i < s.length() ; i++){
            if(s.charAt(i) == c){
                int j = i;
                int l = i;
                int m = t[s.charAt(j)-'a'];
                int temp = i;
                while(j < m){
                    m = Math.max(m , t[s.charAt(j)-'a']);
                    
                    l = Math.min(l , mi[s.charAt(j)-'a']);
                    while(l < temp){
                        int g = l;
                        for(int k = l ; k < temp ; k++){
                            m = Math.max(m , t[s.charAt(k)-'a']);
                            
                            l = Math.min(l , mi[s.charAt(k)-'a']);
                        }
                        temp = g;
                    }
                    j++;
                }
                return new int[]{l,m};
            }
        }
        return new int[]{-1,-1};
    }
}
