class Solution {
    
    
public List<String> maxNumOfSubstrings(String s) {
        int N = s.length();
        int[] lasts = new int[128];
        int[] firsts = new int[128];
        Arrays.fill(lasts, -1);
        Arrays.fill(firsts, -1);
        for (int i = 0; i < N; i++) lasts[s.charAt(i)] = i;
        for (int i = N - 1; i >= 0; i--) firsts[s.charAt(i)] = i
;
        
        List<int[]> intervals = new ArrayList<>();
        for (char c = 'a'; c <= 'z'; c++) {
            if (firsts[c] < 0) continue;
            
            boolean possible = true;
            int end = lasts[c];
            for (int i = firsts[c]; i <= end && i < N && possible; i++) {
                end = Math.max(end, lasts[s.charAt(i)]);
                possible &= firsts[s.charAt(i)] >= firsts[c];
            }
            
            if (possible) intervals.add(new int[] { firsts[c], end })
;
        }
        
        List<String> ans = new ArrayList<>();
        Collections.sort(intervals, (a, b) -> a[1] - b[1]);
        int end = -1;
        for (int i = 0; i < 
intervals.size(); i++) {
            int[] inv = intervals.get(i);
            if (inv[0] > end) {
                end = inv[1];
                ans.add(s.substring(inv[0], inv[1] + 1));
            }
        }
        
        return ans;
    }
}
