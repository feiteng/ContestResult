class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        int charset = 'z' - 'a' + 1;
        
int n = s.length();
        int[][] lr = new int[charset][2];
        for(int i = 0; i < charset; i++){
            for(int j = 0; j < 2; j++){
                lr[i][j] = -1;
            }
        }
        char[] chars = s.toCharArray();
        for(int i = 0; i < n; i++){
            chars[i] -= 'a';
        }
        for(int i = 0; i < n; i++){
            if(lr[chars[i]][0] == -1){
                lr[chars[i]][0] = i;
            }
            lr[chars[i]][1] = i;
        }
       // System.out.println(Arrays.deepToString(lr));
        Node[] nodes = new Node[charset];
        for(int i = 0; i < charset; i++){
            if(lr[i][0] == -1){
                continue;
            }
            nodes[i] = new Node();
            nodes[i].l = lr[i][0];
            nodes[i].r = lr[i][1];
        }
        for(int i = 0; i < charset; i++){
            if(lr[i][0] == -1){
                continue;
            }
            Node node = nodes[i];
            int edge = 0;
            for(int j = lr[i][0]; j <= lr[i][1]; j++){
                int c = chars[j];
                edge |= 1 << c;
            }
            for(int j = 0; j < charset; j++){
                if(((edge >> j) & 1) == 1){
                    node.adj.add(nodes[j]);
                }
            }
        }
        
        for(Node node : nodes){
            if(node == null){
                continue;
            }
            tarjan(node);
        }
        
        List<String> ans = new ArrayList();
        for(Node node : nodes){
            if(node == null || node.set != node){
                continue;
            }
            boolean isLeaf = true;
            
            for(Node sub : node.inset){
                
for(Node next : sub.adj){
                    if(next.set != sub.set){
                        isLeaf = false;
                    }
                }
            }
            
            if(isLeaf){
                ans.add(s.substring(node.l, node.r + 1));
            }
        }
        
        return ans;
        
    }
    
    int order = 0;
    Deque<Node> dq = new ArrayDeque();
    public void tarjan(Node root){
        if(root.dfn > 0){
            return;
        }
        root.dfn = root.low = ++order;
        dq.addLast(root);
        root.instk = true;
        
        for(Node node : root.adj){
            tarjan(node);
            if(node.instk){
                root.low = Math.min(root.low, node.low);
            }
        }
        
        if(root.low == root.dfn){
            while(true){
                Node tail = dq.removeLast();
                tail.instk = false;
                tail.set = root;
                root.inset.add(tail);
                root.l = Math.min(root.l, tail.l);
                root.r = Math.max(root.r, tail.r);
                if(tail == root){
                    break;
                }
            }
        }
    }
  
}

class Node{
    int l;
    int r;
    Node set;
    int dfn;
    int low;
    boolean instk;
    List<Node> inset = new ArrayList<>();
    List<Node> adj = new ArrayList<>();
}
