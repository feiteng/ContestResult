class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        
        int[][] intervals = new int[26][2];
        
        for(int[] it: intervals){
            Arrays.fill(it,-1);
        }
        
        for(int i = 0; i < s.length(); i++){
            if(intervals[s.charAt(i)-'a'][0] == -1 || i < intervals[s.charAt(i)-'a'][0]){
                intervals[s.charAt(i)-'a'][0] = i;
            }
            if(intervals[s.charAt(i)-'a'][1] == -1 || i > intervals[s.charAt(i)-'a'][1]){
                intervals[s.charAt(i)-'a'][1] = i;
            }
        }
        
        while(true){
            int[][] newIntervals = new int[26][2];
            for(int i = 0; i < intervals.length; i++){
                for(int j = 0; j < intervals[0].length; j++){
                    newIntervals[i][j] = intervals[i][j];
                }
            }
            for(int i = 0; i < 26; i++){
                if(intervals[i][0] != -1 && intervals[i][1] != -1){
                    int start = intervals[i][0];
                    int end = intervals[i][1];
                    for(int j = intervals[i][0]; j <= intervals[i][1]; j++ ){
                        if(intervals[s.charAt(j)-'a'][0] != -1 && intervals[s.charAt(j)-'a'][1] != -1){
                            start = Math.min(start, intervals[s.charAt(j)-'a'][0]);
                            end = Math.max(end, intervals[s.charAt(j)-'a'][1]);
                        }
                    }
                    intervals[i][0] = start;
                    intervals[i][1] = end;
                }
            }
            boolean isSame = true;
            for(int i = 0; i < intervals.length; i++){
                for(int j = 0; j < intervals[0].length; j++){
                    if(newIntervals[i][j] != intervals[i][j]){
                        isSame = false;
                        break;
                    }
                }
            }
            if(isSame == true){
                break;
            }
        }
        
        
        Comparator<int[]> comp = new Comparator<>(){
            @Override
            public int compare(int[] i1, int[] i2){
                if(i1[1] == i2[1]){
                    return i1[0]-i2[0];
                }
                return i1[1] - i2[1];
            }
        };
        
        
        
        Arrays.sort(intervals, comp);
        
        // for(int[] pr: intervals){
        //     System.out.println(Arrays.toString(pr));
        // }
        
        List<String> res = new ArrayList<>();
        int k = 0;
        while(k < intervals.length && (intervals[k][0] == -1 || intervals[k][1] == -1 ))
            k++;
        
        res.add(s.substring(intervals[k][0],intervals[k][1]+1));
        k++;
        
        for(int i = k ; i < intervals.length; i++){
            if(intervals[i][0] > intervals[i-1][1]){
                res.add(s.substring(intervals[i][0],intervals[i][1]+1));
            }
        }
        return res;
    }
}
