import java.util.ArrayList;
import java.util.List;

class Solution {
  public List<String> maxNumOfSubstrings(String paramString) {
    int[][] arrayOfInt = new int[26][2];
    int b;
    for (b = 0; b < 26; b++) {
      for (int b1 = 0; b1 < 2; b1++)
        arrayOfInt[b][b1] = -1; 
    } 
    for (b = 0; b < paramString.length(); b++) {
      if (arrayOfInt[paramString.charAt(b) - 97][0] == -1)
        arrayOfInt[paramString.charAt(b) - 97][0] = b; 
      arrayOfInt[paramString.charAt(b) - 97][1] = b;
    } 
    int[] arrayOfInt1 = new int[paramString.length()];
    int i;
    for (i = 0; i < paramString.length(); i++)
      arrayOfInt1[i] = -1; 
    for (i = 0; i < 26; i++) {
      if (arrayOfInt[i][0] != -1) {
        int k = arrayOfInt[i][1];
        for (int m = arrayOfInt[i][0]; m <= k; m++) {
          if (arrayOfInt[paramString.charAt(m) - 97][0] >= arrayOfInt[i][0]) {
            k = Math.max(k, arrayOfInt[paramString.charAt(m) - 97][1]);
          } else {
            arrayOfInt1[arrayOfInt[i][0]] = -2;
            break;
          } 
        } 
        if (arrayOfInt1[arrayOfInt[i][0]] != -2)
          arrayOfInt1[arrayOfInt[i][0]] = k; 
      } 
    } 
    i = arrayOfInt1.length;
    ArrayList<String> arrayList = new ArrayList();
    for (int j = arrayOfInt1.length - 1; j >= 0; j--) {
      if (arrayOfInt1[j] >= 0 && arrayOfInt1[j] < i) {
        arrayList.add(paramString.substring(j, arrayOfInt1[j] + 1));
        i = j;
      } 
    } 
    return arrayList;
  }
}
