class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        int[][] chars = new int[26][2];
        for(int i = 0;i<s.length();i++){
            char c = s.charAt(i);
            if(chars[c-'a'][0]==0){
                chars[c-'a'][0] = i+1;
                chars[c-'a'][1] = i+1;
            }
            else
                chars[c-'a'][1] = i+1;
        }
        for(int i = 0;i<26;i++){
            if(chars[i][0]==0)
                continue;
            int left = chars[i][0],right = chars[i][1];
            for(;left<right;left++){
                char c = s.charAt(left);
                if(chars[c-'a'][0]==0)
                    continue;
                chars[i][0] = Math.min(chars[i][0],chars[c-'a'][0]);
                chars[i][1] = Math.max(chars[i][1],chars[c-'a'][1]);
            }
        }
                for(int i = 0;i<26;i++){
            if(chars[i][0]==0)
                continue;
            int left = chars[i][0],right = chars[i][1];
            for(;left<right;left++){
                char c = s.charAt(left);
                if(chars[c-'a'][0]==0)
                    continue;
                chars[i][0] = Math.min(chars[i][0],chars[c-'a'][0]);
                chars[i][1] = Math.max(chars[i][1],chars[c-'a'][1]);
            }
        }
                for(int i = 0;i<26;i++){
            if(chars[i][0]==0)
                continue;
            int left = chars[i][0],right = chars[i][1];
            for(;left<right;left++){
                char c = s.charAt(left);
                if(chars[c-'a'][0]==0)
                    continue;
                chars[i][0] = Math.min(chars[i][0],chars[c-'a'][0]);
                chars[i][1] = Math.max(chars[i][1],chars[c-'a'][1]);
            }
        }
                for(int i = 0;i<26;i++){
            if(chars[i][0]==0)
                continue;
            int left = chars[i][0],right = chars[i][1];
            for(;left<right;left++){
                char c = s.charAt(left);
                if(chars[c-'a'][0]==0)
                    continue;
                chars[i][0] = Math.min(chars[i][0],chars[c-'a'][0]);
                chars[i][1] = Math.max(chars[i][1],chars[c-'a'][1]);
            }
        }
                for(int i = 0;i<26;i++){
            if(chars[i][0]==0)
                continue;
            int left = chars[i][0],right = chars[i][1];
            for(;left<right;left++){
                char c = s.charAt(left);
                if(chars[c-'a'][0]==0)
                    continue;
                chars[i][0] = Math.min(chars[i][0],chars[c-'a'][0]);
                chars[i][1] = Math.max(chars[i][1],chars[c-'a'][1]);
            }
        }
        // for(int i = 0;i<26;i++){
        //        System.out.println(i+" "+chars[i][0]+" "+chars[i][1]);
        //     }
        List<String> res = new ArrayList<>();
        boolean[] bools = new boolean[s.length()];
        int flag = 1;
        while(flag == 1){
            int min = s.length()+1,target = 0;
            for(int i = 0;i<26;i++){
                if(chars[i][0] == 0)
                    continue;
                if(chars[i][1] - chars[i][0]<min){
                    target = i;
                    min = chars[i][1] - chars[i][0];
                }
            }
            if(min == s.length()+1)
                break;
            for(int i = chars[target][0]-1;i<chars[target][1];i++){
                if(bools[i]==true){
                    flag = 0;
                    break;
                }
            }
            
            if(flag == 0){
                chars[target][0] = 0;
                chars[target][1] = 0;
                flag = 1;
                continue;
            }
            for(int i = chars[target][0]-1;i<chars[target][1];i++){
                bools[i]=true;
            }
            res.add(s.substring(chars[target][0]-1,chars[target][1]));
            chars[target][0] = 0;
            chars[target][1] = 0;
        }
        return res;
    }
}
