class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        int n = s.length();
        int[][] count = new int[n + 1][26];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < 26; j++) {
                count[i + 1][j] = count[i][j];
            }
            count[i + 1][s.charAt(i) - 'a']++;
        }
        List<String> out = new ArrayList<>();
        int start = 0;
        for (int i = 0; i < n; i++) {
            int index = s.charAt(i) - 'a';
            if (count[i + 1][index] == count[n][index]) {
                int size = 1;
                while (i - start + 1 >= size) {
                    boolean valid = true;
                    int diff = 0;
                    for (int j = 0; j < 26; j++) {
                        if (count[i + 1][j] - count[i + 1 - size][j] != 0 && count[i + 1][j] - count[i + 1 - size][j] < count[n][j]) {
                            valid = false;
                            diff += count[n][j] - (count[i + 1][j] - count[i + 1 - size][j]);
                        }
                    }
                    if (valid) {
                        out.add(s.substring(i + 1 - size, i + 1));
                        start = i + 1;
                        break;
                    } else {
                        size += diff;
                    }
                }
            }
        }
        return out;
    }
}
