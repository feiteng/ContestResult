class Solution {
    public List<String> maxNumOfSubstrings(String s) {
        int[][] letterRanges = new int[26][2];
        for (int[] row : letterRanges) {
            Arrays.fill(row, -1);
        }
        for (int i = 0; i < s.length(); i++) {
            char c = s.charAt(i);
            int idx = c-'a';
            if (letterRanges[idx][0] == -1) {
                letterRanges[idx][0] = i;
            }
            letterRanges[idx][1] = i;
        }
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> (a[1]-a[0])-(b[1]-b[0]));
        for (int i = 0; i < 26; i++) {
            if (letterRanges[i][0] != -1) {
                int start = letterRanges[i][0], end = letterRanges[i][1];
                while (true) {
                    int len = end-start;
                    int newStart = Integer.MAX_VALUE, newEnd = Integer.MIN_VALUE;
                    for (int j = start; j <= end; j++) {
                        int idx = s.charAt(j)-'a';
                        newStart = Math.min(newStart, letterRanges[idx][0]);
                        newEnd = Math.max(newEnd, letterRanges[idx][1]);
                    }
                    if (len == newEnd-newStart) {
                        break;
                    } else {
                        start = newStart;
                        end = newEnd;
                    }
                }
                pq.add(new int[] {start, end});
            }
        }
        int n = s.length();
        List<String> res = new ArrayList<>();
        boolean[] visited = new boolean[n];
        while (!pq.isEmpty()) {
            int[] cur = pq.poll();
            boolean isOverlap = false;
            for (int i = cur[0]; i <= cur[1]; i++) {
                if (visited[i]) {
                    isOverlap = true;
                    break;
                }
            }
            if (!isOverlap) {
                for (int i = cur[0]; i <= cur[1]; i++) {
                    visited[i] = true;
                }
                res.add(s.substring(cur[0], cur[1]+1));
            }
        }
        return res;
