class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int ans = 0;
        int bo = 0;
        
        while(numBottles > 0){
            ans+=numBottles;
            bo+=numBottles;
            
            numBottles = bo/numExchange;
            bo=bo%numExchange;
            
            
        }
        
        return ans;
    }
}
