class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
    	int result = 0;
    	int fullWaterBottles = numBottles;
    	while( fullWaterBottles > 0 ) {
    		result += fullWaterBottles;
    		fullWaterBottles = numBottles/numExchange;
    		numBottles = numBottles%numExchange + fullWaterBottles;
    	}
    	return result;
    }
}
