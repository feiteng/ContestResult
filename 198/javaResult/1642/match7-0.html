class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int res = numBottles;
        int left = 0;
        while (numBottles >= numExchange) {
            left = numBottles % numExchange;
            numBottles /= numExchange;
            res += numBottles;
            numBottles += left;
        }
        return res;
    }
}
