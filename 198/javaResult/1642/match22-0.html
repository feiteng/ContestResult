class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int res = 0, empty = 0;
        while (0 != numBottles) {
            res += numBottles;
            empty += numBottles;
            numBottles = empty / numExchange;
            empty -= numBottles * numExchange;
        }
        return res;
    }
}
