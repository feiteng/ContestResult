class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int out = 0;
        int empties = 0;
        while (numBottles + empties >= numExchange) {
// System.out.print("Bottles:");
// System.out.println(numBottles);
// System.out.print("Empties:");
// System.out.println(empties);
            out += numBottles;
            empties += numBottles;
            numBottles = empties / numExchange;
            empties = empties % numExchange;
        }
        
        return out + numBottles;
    }
}
