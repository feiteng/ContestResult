class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int res = 0;
        int left = 0;
        while (numBottles > 0) {
            res += numBottles;
            left += numBottles;
            numBottles = left / numExchange;
            left = left % numExchange;
        }
        return res;
    }
}
