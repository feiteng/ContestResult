class Solution {
    public int numWaterBottles(int numBottles, int numExchange) {
        int count = numBottles, numEmpty = 0;
        while (numBottles / numExchange != 0) {
            numEmpty = numBottles % numExchange;
            numBottles /= numExchange;
            count += numBottles;
            numBottles += numEmpty;
        }
        return count;
    }
}
