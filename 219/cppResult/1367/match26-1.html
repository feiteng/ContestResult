class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        int N = cuboids.size();
        for(int i = 0; i < N; i++){
            sort(cuboids[i].begin(), cuboids[i].end());
        }
        sort(cuboids.begin(), cuboids.end());
        int dp [N]; int ret = 0;
        for(int i = 0; i < N; i++){
            dp[i] = 0;
            for(int j = 0; j < i; j++)
                if(cuboids[j][1] <= cuboids[i][1] && cuboids[j][2] <= cuboids[i][2])
                    dp[i] = max(dp[i], dp[j]);
            dp[i] += cuboids[i][2];
            ret = max(ret, dp[i]);
        }
        return ret;
    }
};
