class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        int n=cuboids.size();
        for(int i=0;i<n;i++) sort(cuboids[i].begin(), cuboids[i].end());
        sort(cuboids.begin(), cuboids.end());
        vector<int> dp(n);
        dp[0]=cuboids[0][2];

        int ans=cuboids[0][2];
        for(int i=1;i<n;i++){
            dp[i]=cuboids[i][2];
            for(int j=i-1;j>=0;j--){
                if (cuboids[i][1]>=cuboids[j][1]&&cuboids[i][2]>=cuboids[j][2]){
                    dp[i]=max(dp[i], cuboids[i][2]+dp[j]);
                }
            }
            ans=max(ans, dp[i]);
        }
        return ans;
    }
};
