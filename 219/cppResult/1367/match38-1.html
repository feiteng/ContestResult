class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        int n = cuboids.size();
        for (auto& c : cuboids) {
            sort(c.begin(), c.end());
        }
        sort(cuboids.rbegin(), cuboids.rend());
        

        vector<int> dp(n);
        for (int i = 0; i < n; i++) {
            dp[i] = cuboids[i].back();
        }
        
        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (cuboids[i][0] <= cuboids[j][0] && cuboids[i][1] <= cuboids[j][1] && cuboids[i][2] <= cuboids[j][2]) {
                    dp[i] = max(dp[i], dp[j] + cuboids[i][2]);
                }
            }
        }
        // vector<vector<int>> dp(cuboids.begin(), cuboids.end());

        // for (int i = 1; i < n; i++) {
        //     for (int j = 0; j < 3; j++) {
        //         for (int pi = 0; pi < i; pi++) {
        //             for (int pj = 0; pj < 3; pj++) {
        //                 if () {
        //                     dp[i][j] = max(dp[i][j], dp[pi][pj] + cuboids[i][j]);
        //                 }
        //             }
        //         }
        //     }
        // }
        
        int ans = 0;
        // for (int i = 0; i < n; i++) {
        //     for (int j = 0; j < 3; j++) {
        //         ans = max(ans, dp[i][j]);
        //     }
        // }
        for (int i = 0; i < n; i++) {
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
