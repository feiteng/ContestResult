class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        
        int n = cuboids.size();
        vector<vector<int>> vec;
        for(auto cuboid: cuboids){
            sort(cuboid.begin(), cuboid.end());
            vec.push_back({cuboid[0], cuboid[1], cuboid[2]});
            if(cuboid[1]!=cuboid[2]) vec.push_back({cuboid[0], cuboid[2], cuboid[1]});
            if(cuboid[1]!=cuboid[0]) vec.push_back({cuboid[1], cuboid[2], cuboid[0]});
        }
        
        sort(vec.begin(), vec.end(), [](const vector<int> & a, const vector<int> & b){
            if(a[0] == b[0] && a[1] == b[1]) return a[2]<=b[2];
            else if(a[0] == b[0]) return a[1] <= b[1];
		    return a[0] < b[0];
        });
        //cout<<vec.size()<<endl;
		int ans = 0;
        int length = vec.size();
        vector<int> dp(length);
        
        /*
        for(int i= 0; i< length; ++i){
            cout<<vec[i][0]<<" "<<vec[i][1]<<" "<<vec[i][2]<<endl;
        }
        */
        
        
		for(int i = 0; i< length; ++i){
			dp[i] = vec[i][2];
			for(int j = 0; j < i; ++j){
				if(vec[i][0] >= vec[j][0] && vec[i][1] >= vec[j][1] && vec[i][2] >= vec[j][2]){
					dp[i] = max(dp[i],dp[j] + vec[i][2]);
				}
			}
			ans = max(ans,dp[i]);
		}
        /*
        for(int i=0; i< length; ++i){
            cout<<dp[i]<<" ";
        }
        */
	    return ans;	
	}
};
