class Solution {
 public:
  int maxHeight(vector<vector<int>> &c) {
    int n = c.size();
    for (int i = 0; i < n; i++) {
      sort(c[i].begin(), c[i].end());
    }
    sort(c.begin(), c.end());
    vector<int> f(n + 1);
    for (int i = 0; i < n; i++) {
      f[i] = c[i][2];
    }
    for (int i = n - 2; i >= 0; i--) {
      for (int j = n - 1; j > i; j--) {
        if (c[i][0] > c[j][0] || c[i][1] > c[j][1] || c[i][2] > c[j][2]) {
          continue;
        }
       // debug(j, i);
        f[i] = max(f[i], f[j] + c[i][2]);
      }
    }
    return *max_element(f.begin(), f.end());
  }
};
