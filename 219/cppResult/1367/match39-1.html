class Solution {

    int maxHeight(vector<vector<int>>& cuboids) {
        me0(dp);

        for(auto to:cuboids){
            int a[3];
            a[0]=to[0];a[1]=to[1];a[2]=to[2];
            map<pair<pair<int,int>,int>,int>mp;
            sort(a,a+3);
            int cnt = 0;
            do{
                cnt++;

                int x = a[0],y=a[1],z=a[2];
                if(mp.count({{x,y},z}))continue;
                dp[x][y][z]++;
                mp[{{x,y},z}]=1;
            } while (next_permutation(a,a+3));
        }


        fo(i,1,100){
            fo(j,1,100){
                fo(z,1,100){
                    dp[i][j][z] = max(dp[i-1][j][z],max(dp[i][j-1][z],dp[i][j][z-1]))+dp[i][j][z]*z;
                }
            }
        }
        return dp[100][100][100];

    }
