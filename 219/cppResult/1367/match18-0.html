    int maxHeight(vector<vector<int>>& a) {
        for (auto &i : a) {
            sort(i.begin(), i.end());
        }
        sort(a.begin(), a.end());
        int dp[105] = {};
        int mx = 0;
        for (int i = 0 ; i < a.size() ; i++) {
            dp[i] = a[i][2];
            for (int j = 0 ; j < i ; j++) {
                if (a[i][2] >= a[j][2] && a[i][1] >= a[j][1] && a[i][0] >= a[j][0]) {
                    dp[i] = max(dp[i], dp[j] + a[i][2]);
                }
            }
            mx = max(mx, dp[i]);
        }
        return mx;
    }
};
