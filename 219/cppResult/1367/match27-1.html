class Solution {
public:

    static bool mycompare(const vector<int>& v1, const vector<int>& v2) {
        if (v1[0] == v2[0]) {
            if (v1[1] == v2[1]) {
                return v1[2] < v2[2];
            } else {
                return v1[1] < v2[1];
            }
        } else {
            return v1[0] < v2[0];
        }
    }
    

    int maxHeight(vector<vector<int>>& cuboids) {
        int n = cuboids.size();
        for (int i = 0; i < cuboids.size(); i++) {
            sort(cuboids[i].begin(), cuboids[i].end(), greater<int>());
            swap(cuboids[i][1], cuboids[i][2]);
        }
        
        auto cu = cuboids;
        sort(cu.begin(), cu.end(), mycompare);
        reverse(cu.begin(), cu.end());

        vector<int> res(n, 0);
        res[0] = cu[0][0];
        for (int i = 1; i < n; i++) {
            res[i] = cu[i][0];
            for (int j = 0; j < i; j++) {
                if (cu[i][1] <= cu[j][1] && cu[i][2] <= cu[j][2]) {
                    res[i] = max(res[i], res[j] + cu[i][0]);
                }
            }
        }
        return *max_element(res.begin(), res.end());
    }
};
