class Solution {
public:
    int maxHeight(vector<vector<int>>& v){
        int n=v.size();
        for(int i=0;i<n;i++){
            sort(v[i].begin(),v[i].end());
        }
        sort(v.begin(),v.end());
        vector<int> p(n+1,0);
        int x=0;
        for(int i=n-1;i>=0;i--){
            p[i]=v[i][2];
            for(int j=i+1;j<n;j++){
                if(v[j][0]>=v[i][0]&&v[j][1]>=v[i][1]&&v[j][2]>=v[i][2]&&v[i][2]+p[j]>p[i])p[i]=v[i][2]+p[j];
            }
            x=max(x,p[i]);
        }
        return x;
    }
};
