bool compFunc(vector<int> & v1, vector<int> & v2) {
    if (v1[2] == v2[2]) {
       if (v1[1] == v2[1])
           return v1[0] < v2[0];
        
        return v1[1] < v2[1];
    }
    
    return v1[2] < v2[2];
}

class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        for (auto & vec : cuboids) {
            sort(vec.begin(), vec.end());
        }
        
        sort(cuboids.begin(), cuboids.end(), compFunc);
        
        vector<int> vec;
        for (auto & c : cuboids) {
            // cout << c[0] << " " << c[1] << " " << c[2] << endl;
            vec.push_back(c[2]);
        }
       
        for (int i = cuboids.size()-1; i >= 0; i--) {
            for (int j = i-1; j >= 0; j--) {
                if (cuboids[j][0] <= cuboids[i][0] && cuboids[j][1] <= cuboids[i][1] && cuboids[j][2] <= cuboids[i][2]) {
                    vec[j] = max(vec[j], vec[i] + cuboids[j][2]);
                }
            }
        }
       
        
        return *max_element(vec.begin(), vec.end());
    }
};
