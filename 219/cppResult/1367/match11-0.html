class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
     
        for(int i=0;i<cuboids.size();i++){
            sort(cuboids[i].rbegin(), cuboids[i].rend());
            // cout<<cuboids[i][0]<<" "<<cuboids[i][1]<<" "<<cuboids[i][2]<<endl;
        }
        
        sort(cuboids.begin(), cuboids.end());
        vector<int>lis(cuboids.size(),0);
        
        lis[0]=cuboids[0][0];
        int maxans=lis[0];
        for(int i=1;i<cuboids.size();i++){
            lis[i]=cuboids[i][0];
            for(int j=i-1;j>=0;j--){
                if(cuboids[i][1]>=cuboids[j][1] && cuboids[i][2]>=cuboids[j][2]){
                    lis[i]=max(lis[i], cuboids[i][0]+lis[j]);
                }
            }
            maxans=max(maxans, lis[i]);
        }
        
        
        
        return maxans;
    }
};
