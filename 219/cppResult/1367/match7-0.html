class Solution {
public:
    int maxHeight(vector<vector<int>>& A) {
        int n = A.size();
        for(int i = 0; i < n; i++) {
            sort(A[i].begin(), A[i].end());
        }
        sort(A.begin(), A.end());
        vector<int>dp(n, 1);
        int imax = 0;
        for(int i =0 ; i < n; i++) {
            dp[i] = A[i][2];
            for(int j = i - 1; j >=0; j--) {
                if(A[i][0] >= A[j][0] && A[i][1] >= A[j][1] && A[i][2] >= A[j][2]) {
                    dp[i] = max(dp[i], dp[j] + A[i][2]);
                }
            }
            imax = max(imax, dp[i]);
        }
        return imax;
    }
};
