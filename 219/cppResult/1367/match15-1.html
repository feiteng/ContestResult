class Solution {
public:
    int maxHeight(vector<vector<int>>& c) {
        for(auto& a: c) {
            sort(a.begin(), a.end());
        }  
        sort(c.begin(), c.end());
        int n = c.size();
        vector<int> dp(n);

        int ret = 0;
        for(int i=0; i<n; i++) {
            dp[i] = c[i][2];
            for(int j=i-1; j>=0; --j) {
                if(c[i][1] >= c[j][1] and c[i][2] >= c[j][2]) {
                    dp[i] = max(c[i][2]+dp[j], dp[i]);
                    //cout << i << " " << j << " " << dp[i] << endl;
                }
            }
            ret = max(ret, dp[i]);
        }
        return ret;
    }
};
