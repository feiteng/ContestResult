class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        int n=cuboids.size();
        vector<vector<int>>ans;
        for(vector<int>a:cuboids){
            sort(a.begin(),a.end());
            ans.push_back(a);
        }
        sort(ans.begin(),ans.end());
        vector<int>v(n,1);
        for(int i=0;i<n;i++){
            v[i]=ans[i][2];
        }
        for(int i=1;i<n;i++){
            for(int j=0;j<i;j++){
                if(ans[j][0]<=ans[i][0]&&ans[j][1]<=ans[i][1]&&ans[j][2]<=ans[i][2]){
                    v[i]=max(v[i],v[j]+ans[i][2]);
                }
            }
        }
        int res=0;
        for(int i=0;i<n;i++){
            res=max(res,v[i]);
        }
        return res;
    }
};
