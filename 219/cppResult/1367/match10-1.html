class Solution {
public:
    int maxHeight(vector<vector<int>>& cb) {
        int n=cb.size();
        for(int i=0;i<n;i++){
            sort(cb[i].begin(),cb[i].end());
        }
        
        sort(cb.begin(),cb.end());
        vector<int> dp(n,0);
        dp[0]=cb[0][2];

        int ans=dp[0];
        for(int i=1;i<n;i++){
            dp[i]=cb[i][2];
            for(int j=i-1;j>=0;j--){
                if(cb[j][0]<=cb[i][0]&&cb[j][1]<=cb[i][1]&&cb[j][2]<=cb[i][2]){
                    dp[i]=max(dp[i],cb[i][2]+dp[j]);
                }
            }
            ans=max(ans,dp[i]);
        }
        return ans;
    }
};
