class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        for (auto &cub: cuboids) {
            sort(cub.begin(), cub.end());
        }
        
        sort(cuboids.begin(), cuboids.end());
        int n=cuboids.size(), res=0;
        vector<int>dp(n, -1);

        for (int i=0; i<n; i++) {
            dp[i] = cuboids[i][2];
            for (int j=i-1; j>=0; j--) {
                if (cuboids[i][1]>=cuboids[j][1]&&cuboids[i][2]>=cuboids[j][2]) {
                   // cout << cuboids[i][1]<< ", " << cuboids[j][1]<< ", " << cuboids[i][0] << ", " << cuboids[j][0] << endl;
                    dp[i] = max(dp[i], cuboids[i][2]+dp[j]);
                }
            } 
            res = max(res, dp[i]);
        }
        return res;
    }
};
