class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        int n = cuboids.size();
        for(int i=0; i<n; ++i)
            sort(cuboids[i].begin(),cuboids[i].end());
        sort(cuboids.begin(),cuboids.end());
        
        // vector<vector<int>> dp1(n);
        vector<int> dp(n); dp[0] = cuboids[0][2];
        // dp[0] = {cuboids[0][2]};

        for(int i=1; i<n; ++i){
            dp[i] = cuboids[i][2];
            for(int j=i-1; j>=0; --j){
                if(cuboids[i][1]>=cuboids[j][1] && cuboids[i][2]>=cuboids[j][2]){
                    dp[i] = max(dp[i], dp[j]+cuboids[i][2]);
                }
            }
        }
            
        // int ans = 0;
        // for(int i=0; i<n; ++i){
        //     ans = max(ans, accumulate(dp[i].begin(), dp[i].end(), 0));
        // }
        int ans = *max_element(dp.begin(),dp.end());
        return ans;
    }
};
