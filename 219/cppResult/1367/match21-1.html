class Solution {
public:
    int maxHeight(vector<vector<int>>& c) {
        int res = 0;
        for (int i = 0; i < c.size(); i++) {
            sort(c[i].begin(), c[i].end());
            res = max(res, c[i][2]);

        }
        sort(c.begin(), c.end());
        vector<int> v(c.size(), 0);
        for (int i = 0; i < c.size(); i++) {
            v[i] = c[i][2];
        }
        
        for (int i = 0; i < c.size(); i++) {
            // cout << "\[" << c[i][0] << ", " << c[i][1] << ", " << c[i][2] << "\] " << v[i] << endl;
            for (int j = i+1; j < c.size(); j++) {
                if (c[i][0] <= c[j][0] && c[i][1] <= c[j][1] && c[i][2] <= c[j][2]) {
                    v[j] = max(v[j], c[j][2]+v[i]);
                    res = max(res, v[j]);
                }
            }
        }
        return res;
    }
};
