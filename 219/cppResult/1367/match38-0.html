class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        
        int res=0;
        for (auto& cuboid : cuboids) {
            sort(cuboid.begin(), cuboid.end());
        }
        res=getAnsOneRound(cuboids);
        int round=1;
        
        while (round<=6) {
            for (auto& cuboid : cuboids) {
                next_permutation(cuboid.begin(),cuboid.end());
            }
            res=max(res,getAnsOneRound(cuboids));
            round++;
        }

        return res;

        
    }
    
    int getAnsOneRound(vector<vector<int>> cuboids) {
        
        sort(cuboids.begin(), cuboids.end());
        //for (auto& c : cuboids) {
        //    cout << c[0] << " " << c[1] << " " << c[2] << endl;
        //}
        return getMaxHeight(cuboids);
    }
    
    int getMaxHeight(vector<vector<int>>& cuboids) {
        int n=cuboids.size();
        vector<int> dp(n,-1);
        for (int i=0;i<n;i++) {
            dp[i]=cuboids[i][2];
        }
        
        for (int i=n-1;i>=0;i--) {
            for (int j=i+1;j<n;j++) {
                //try to place cuboids[i] on cuboids[j];
                if (cuboids[i][0]<=cuboids[j][0] && cuboids[i][1]<=cuboids[j][1] && cuboids[i][2]<=cuboids[j][2]) {
                    dp[i]=max(dp[i],dp[j]+cuboids[i][2]);
                }
            }
        }
        
        int res=0;
        for (int height : dp) {
            res=max(res,height);
        }
        return res;
        
    }
};
