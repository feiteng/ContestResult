#define ar array

class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        vector<ar<int,3>> v;
        for (int i=0; i<cuboids.size(); i++) {
            vector<int> x = {cuboids[i][0], cuboids[i][1], cuboids[i][2]};
            sort(x.begin(), x.end());
            v.push_back({x[2], x[1], x[0]});
        }
        
        sort(v.begin(), v.end());
        reverse(v.begin(), v.end());
        vector<int> dp(cuboids.size());

        dp[0] = v[0][0];
        int ans=dp[0];
        for (int i=1; i<cuboids.size(); i++) {
            dp[i] = v[i][0];
            for (int j=0; j<i; j++) {
                if (v[j][0]>=v[i][0]&&v[j][1]>=v[i][1]&&v[j][2]>=v[i][2]) {
                    dp[i] = max(dp[i], dp[j]+v[i][0]);
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
