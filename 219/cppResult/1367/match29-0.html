struct node{
    int a;int b;int c;
};
class Solution {
public:
    int maxHeight(vector<vector<int>>& cub) {
        vector<int> dp(cub.size(),0);
        vector<node> arr;
        for(int i=0;i<cub.size();i++){
            node n;
            n.a=max(cub[i][0],max(cub[i][1],cub[i][2]));
            n.c=min(cub[i][0],min(cub[i][1],cub[i][2]));
            n.b=cub[i][0]+cub[i][1]+cub[i][2]-n.a-n.c;
            arr.push_back(n);
        }
        sort(arr.begin(),arr.end(),[&](node n1,node n2){
            if(n1.a==n2.a&&n1.b==n2.b)
                return n1.c>n2.c;
            else if(n1.a==n2.a)
                return n1.b>n2.b;
            return n1.a>n2.a;
        });
        dp[0]=arr[0].a;
        int res=dp[0];
        for(int i=1;i<arr.size();i++){
            dp[i]=arr[i].a;
            for(int j=0;j<i;j++){
                if(arr[i].a<=arr[j].a&&arr[i].b<=arr[j].b&&arr[i].c<=arr[j].c){
                    dp[i]=max(dp[i],dp[j]+arr[i].a);
                }
            }
            res=max(dp[i],res);
        }
        return res;
    }
};
