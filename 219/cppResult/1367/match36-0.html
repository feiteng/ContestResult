class Solution {
public:
    struct node {
        int width;
        int length;
        int height;
        node (int width, int length, int height) :width(width), length(length), height(height) {}
    };
    int static cmp(node p1, node p2) {
        if (p1.width != p2.width) return p1.width > p2.width;
        else if(p1.length != p2.length) return p1.length > p2.length;
        else return p1.height > p2.height;
    }
    int maxHeight(vector<vector<int>>& cuboids) {
        int n = cuboids.size();
        vector<node> nodes;
        for (int i = 0; i < n; ++i) {
            sort(cuboids[i].begin(), cuboids[i].end());
            nodes.push_back(node(cuboids[i][0], cuboids[i][1], cuboids[i][2]));
        }
        sort(nodes.begin(), nodes.end(), cmp);
        int dp[105];
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            int max_num = 0;
            for (int j = 0; j < i; ++j) {
                if (nodes[j].width >= nodes[i].width && nodes[j].length >= nodes[i].length && nodes[j].height >= nodes[i].height) {
                    max_num = max(max_num, dp[j]);
                }
            }
            dp[i] = max_num + nodes[i].height;
            ans = max(ans, dp[i]);
        }
        return ans;
    }
};
