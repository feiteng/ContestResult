bool smaller(vector<int> a, vector<int> b) {
    if(a[0] == b[0]) {
        if(a[1] == b[1]) {
            return a[2] < b[2];
        }
        return a[1] < b[1];
    }
    return a[0] < b[0];
}

bool strictSmaller(vector<int> a, vector<int> b) {
    if(a[0] <= b[0] && a[1] <= b[1] && a[2] <= b[2]) return true;
    return false;
}


class Solution {
public:
    int maxHeight(vector<vector<int>>& cuboids) {
        int n = cuboids.size();
        for(int i = 0; i < n; i++) sort(cuboids[i].begin(), cuboids[i].end());
        sort(cuboids.begin(), cuboids.end(), smaller);
        int h[n];

        int ans = 0;
        for(int i = 0; i < n; i++) {
            h[i] = cuboids[i][2];
            for(int j = 0; j < i; j++) {
                if(strictSmaller(cuboids[j], cuboids[i])) {
                    h[i] = max(h[i], h[j] + cuboids[i][2]);
                }
            }
            ans = max(ans, h[i]);
        }
        return ans;
    }
};
