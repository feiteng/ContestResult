class Solution {
public:
    int dp[1005];

    int maxHeight(vector<vector<int>>& cuboids) {
        int n = cuboids.size();
        for(int i = 0; i < n; i++){
            sort(cuboids[i].begin(), cuboids[i].end());
        }
        sort(cuboids.begin(), cuboids.end());
        int ans = 0;
        dp[0] = cuboids[0][2];

        ans = max(ans, dp[0]);
        for(int i = 1; i < n; i++){
            int mx = 0;
            for(int j = i - 1; j >= 0; j--){
                if(cuboids[i][0] >= cuboids[j][0] and cuboids[i][1] >= cuboids[j][1] and cuboids[i][2] >= cuboids[j][2]){
                    mx = max(mx, dp[j]);
                }
            }
            dp[i] = mx + cuboids[i][2];
            ans = max(ans, dp[i]);
        }
        
        return ans;
    }
};
