int dp[110][110][110];
class Solution
{
public:
    int maxHeight(vector<vector<int>> &cbs)
    {

        memset(dp, 0, sizeof(dp));
        for (auto val : cbs)
        {
            int arr[3];
            arr[0] = val[0];
            arr[1] = val[1];
            arr[2] = val[2];
            map<pair<pair<int, int>, int>, int> mp;
            int a1,a2,a3;
            sort(arr, arr + 3);
            int cnt = 0;
            do
            {
               
                 a1 = arr[0];
                 a2 = arr[1];
                 a3 = arr[2];
                 cnt++;
                if (mp.count({{a1, a2}, a3}))
                    continue;
                dp[a1][a2][a3]++;
                mp[{{a1, a2}, a3}] = 1;
            } while (next_permutation(arr, arr + 3));
        }

        fo(i, 1, 100)
        {
            fo(j, 1, 100)
            {
                fo(a3, 1, 100)
                {
                    dp[i][j][a3] = max(dp[i - 1][j][a3], max(dp[i][j - 1][a3], dp[i][j][a3 - 1])) + dp[i][j][a3] * a3;
                }
            }
        }
        return dp[100][100][100];
    }
};
