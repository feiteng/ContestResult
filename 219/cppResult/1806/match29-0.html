class Solution {
public:
    int numberOfMatches(int n) {
        int matches = 0;
        int remaining = n;
        while (remaining != 1) {
            if (remaining % 2 == 0) {
                matches += remaining / 2;
                remaining = remaining / 2;
            } else {
                matches += remaining / 2;
                remaining = (remaining + 1) / 2;
            }
        }
        return matches;
    }
};
