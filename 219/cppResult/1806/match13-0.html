class Solution {
public:
    int numberOfMatches(int n) {
        int ans = 0;
        while (n > 1) {
            if (n % 2 == 1) {
                n = (n-1) / 2 + 1;
                ans += (n-1);
            } else {
                n /= 2;
                ans +=n;
            }
            // ans += n;
        }
        return ans;
    }
};
