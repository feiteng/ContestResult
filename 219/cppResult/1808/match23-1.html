class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n=stones.size();
        vector<int> p(n+1, 0);
        for (int i=0; i<n; i++) p[i+1]=stones[i]+p[i];

        vector<vector<int>> dp(n+2, vector<int>(n+2, 0));
        for (int i=n; i>0; i--) {
            for (int j=i; j<=n; j++) {
                dp[i][j]=max(p[j]-p[i]-dp[i+1][j], p[j-1]-p[i-1]-dp[i][j-1]);
            }
        }
        return dp[1][n];
    }
};
