class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
    int n = stones.size();
    vector<int> prefix(n+1);
    for(int i=1;i<=n;i++){
        prefix[i]=prefix[i-1]+stones[i-1];
    }
    vector<vector<int>> dp(n,vector<int>(n));
    for(int i=n-1;i>=0;i--){
        for(int j=0;j<n;j++){
            if(i>=j) continue;
            if((i+j)%2!=n%2){
                dp[i][j]=max(prefix[j+1]-prefix[i+1]+dp[i+1][j],prefix[j]-prefix[i]+dp[i][j-1]);
            } else {
                dp[i][j]=min(dp[i+1][j]-prefix[j+1]+prefix[i+1],dp[i][j-1]-prefix[j]+prefix[i]);
            }
        }
    }
    return dp[0][n-1];
}
};
