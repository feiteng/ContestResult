class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<vector<int> > dp(n,vector<int>(n));
        vector<int> pre(n+1);
        for(int i = 0; i < n; ++i) pre[i+1] += pre[i]+stones[i];

        for(int k = 1; k <= n; ++k) {
            for(int i = 0; i+k < n; ++i) {
                int tmp1 = pre[i+k+1]-pre[i+1];
                int tmp2 = pre[i+k]-pre[i];
                dp[i][i+k] = max(tmp1-dp[i+1][i+k],tmp2-dp[i][i+k-1]);
            }
        }
        return dp[0][n-1];
    }
};
