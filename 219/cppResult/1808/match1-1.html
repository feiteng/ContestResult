class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<vector<int>> dp(n, vector<int>(n));
        vector<int> prefix(n + 1);
        for (int i = 0; i < n; i++) {
            dp[i][i] = 0;

            prefix[i + 1] = prefix[i] + stones[i];
        }
        for (int d = 1; d < n; d++) {
            for (int i = 0; i + d < n; i++) {
                int j = i + d;
                dp[i][j] = max(prefix[j + 1] - prefix[i + 1] - dp[i + 1][j], prefix[j] - prefix[i] - dp[i][j - 1]);
            }
        }
        return dp[0][n - 1];
    }
};
