class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n=stones.size();
        vector<int> preSum(n+1,0);
        for(int i=1;i<=n;++i)
        {
            preSum[i]=preSum[i-1]+stones[i-1];
        }
        vector<vector<int>> dp(n,vector<int>(n,0));
        /*for(int i=0;i<n;++i)
        {
            dp[i][i]=stones[i];
        }*/
        for(int i=n-2;i>=0;--i)
        {
            for(int j=i+1;j<n;++j)
            {
                dp[i][j]=max(preSum[j+1]-preSum[i+1]-dp[i+1][j],preSum[j]-preSum[i]-dp[i][j-1]);
            }
            
        }
        return dp[0][n-1];
    }
};
