class Solution {
public:
    int dp[1005][1005];
    bool c[1005][1005];
    int sa[1005];
    int get(int l, int r){
        return sa[r+1]-sa[l];
    }
    int solve(vector<int>& stones, int l, int r){
        if(l>r) return 0;
        if(c[l][r]) return dp[l][r];

        int ret = get(l+1, r)-solve(stones, l+1, r);
        ret = max(ret, get(l, r-1)-solve(stones, l, r-1));
        c[l][r]=true;
        dp[l][r]=ret;
        return ret;
    }
    int stoneGameVII(vector<int>& stones) {
        memset(c, false, sizeof(c));
        sa[0] = 0;
        for(int i=0;i<stones.size();i++){
            sa[i+1] = sa[i]+stones[i];
        }
        return solve(stones, 0, stones.size()-1);
    }
};
