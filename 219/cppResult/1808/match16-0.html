class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
      int sz = stones.size();
      vector<int> prefix_sum(sz+1, 0);
      for (int i = 0; i < sz; i++) {
        prefix_sum[i+1] = prefix_sum[i] + stones[i];
      }
      vector<vector<int>> dp(sz, vector<int>(sz, 0));
      for (int len = 1; len <= sz; len++) {
        for (int start = 0; start <= (sz - len); start++) {
          int end = start + len - 1;
          if (len == 1) {
            dp[start][end] = 0;
          } else {
            dp[start][end] = max(prefix_sum[end+1] - prefix_sum[start+1] - dp[start+1][end], prefix_sum[end] - prefix_sum[start] - dp[start][end-1]);  
          }
        }
      }
      return dp[0][sz-1];
    }
};
