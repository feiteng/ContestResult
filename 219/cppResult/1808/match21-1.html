class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<int> preSum(n + 1, 0);
        for (int i = 1; i <= n; ++i) {
            preSum[i] = preSum[i - 1] + stones[i - 1];
        }
        
        vector<vector<int>> dp(n, vector<int>(n));
        
        for (int len = 1; len < n; ++len) {
            for (int i = 0; i < n; ++i) {
                if (i + len >= n) break;
                int j = i + len;

                if (len & 1) {
                    dp[i][i + len] = max(preSum[j + 1] - preSum[i + 1] + dp[i + 1][j], preSum[j] - preSum[i] + dp[i][j - 1]);
                } else {
                    dp[i][i + len] = min(- preSum[j + 1] + preSum[i + 1] + dp[i + 1][j], -preSum[j] + preSum[i] + dp[i][j - 1]);
                }
            }
        }
        return abs(dp[0][n - 1]);
    }
};
