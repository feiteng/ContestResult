class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        int s[n + 1];
        s[0] = 0;
        for(int i = 1; i <= n; i++) s[i] = s[i - 1] + stones[i - 1];
        //dp[i][j] = max(s[j]-s[i-1]-dp[i] )

        int dp[n + 1][n + 1];
        memset(dp, 0, sizeof(dp));
        for(int i = n - 1; i >= 1; i--) {
            for(int j = i + 1; j <= n; j++) {
                dp[i][j] = max(s[j] - s[i] - dp[i + 1][j], s[j - 1] - s[i - 1] - dp[i][j - 1]);
            }
        }
        return dp[1][n];
    }
};
