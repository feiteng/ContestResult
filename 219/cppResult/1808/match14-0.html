class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<int> sum(n + 1, 0);
        for(int i = 1; i <= n; i ++)
            sum[i] = sum[i - 1] + stones[i - 1];
        
        int f[n + 1][n + 1];
        memset(f, 0, sizeof f);
        
        for(int i = 1; i <= n; i ++)
            f[i][i] = 0;
        
      
        for(int len = 2; len <= n; len ++)
        {
            for(int i = 1; i + len - 1 <= n; i ++)
            {
                int j = i + len - 1; // i - 1, j - 1
                f[i][j] = max(sum[j] - sum[i] - f[i + 1][j], sum[j - 1] - sum[i - 1] - f[i][j - 1]);
            }
        }
       
//         int a = 0, t = 0;
//         int i = 1, j = n;
//         int flag = 0;
        
//         while(i != j)
//         {
//             if(f[i][j] == sum[j] - sum[i] - f[i + 1][j])
//             {
//                 if(flag == 0)
//                     a += f[i][j];
//                 else
//                     t += f[i][j];
//                 i = i + 1;
//             }
//             else
//             {
//                 if(flag == 0)
//                     a += f[i][j];
//                 else
//                     t += f[i][j];
//                 j = j - 1;               
//             }
//             flag = 1- flag;
//         }
        
        return f[1][n];
    }
};
