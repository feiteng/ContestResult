class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int len = stones.size();
        vector<vector<int>> dp(len + 1, vector<int>(len + 1, 0));
        vector<int> sums(len + 1, 0);
        for(int i = 0; i < len; i++) {
            sums[i + 1] = sums[i] + stones[i];
        }
        for(int l = 1; l <= len; l++) {
            for(int i = 0; i + l <= len; i++) {
                int temp1 = -dp[i + 1][i + l] + sums[i + l] - sums[i + 1];
                int temp2 = -dp[i][i + l - 1] + sums[i + l - 1] - sums[i];
                dp[i][i + l] = max(temp1, temp2);
            }
        }
        return dp[0][len];
    }
};
