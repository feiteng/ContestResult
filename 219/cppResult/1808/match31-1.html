class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int N = stones.size();

        vector<int> sums(N + 1, 0);
        for (int i = 1; i <= N; ++i) {
            sums[i] = sums[i - 1] + stones[i - 1];
        }

        vector<vector<int>> dp(N + 1, vector<int>(N + 1, 0));
        for (int i = N - 2; i >= 0; --i) {
            for (int j = i + 2; j <= N; ++j) {
                dp[i][j] = max(help(sums, i + 1, j) - dp[i + 1][j], help(sums, i, j - 1) - dp[i][j - 1]);
            }
        }
        return dp[0][N];
    }

private:
    int help(const vector<int>& nums, int beg, int end) {
        return nums[end] - nums[beg];
    }
};
