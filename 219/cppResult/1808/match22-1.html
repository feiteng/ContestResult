class Solution {
public:
    int dp[1005][1005];
    int stone[1005];
    int dfs(int x,int y,bool a,int s)
    {
        if(x==y)
            return 0;
        if(dp[x][y]!=-1)
            return dp[x][y];
        int left = s-stone[x];
        int right = s-stone[y];

        if(a)
            return dp[x][y]=max(left+dfs(x+1,y,false,left),right+dfs(x,y-1,false,right));
        else
            return dp[x][y]=min(-left+dfs(x+1,y,true,left),-right+dfs(x,y-1,true,right));
        
    }
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        int s = 0;
        memset(dp,-1,sizeof(dp));
       for(int i=0;i<n;i++)
       {
           
           stone[i]=stones[i];
           s+=stones[i];
       }
        return dfs(0,n-1,1,s);
    }
};
