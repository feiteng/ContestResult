class Solution {
public:
    int pre[1010],dp[1010][1010];
    int d(int l,int r){
        if(l>=r)return 0;
        if(dp[l][r]!=-1)return dp[l][r];
        return dp[l][r] = max(pre[r]-pre[l]-d(l+1,r),pre[r-1]-pre[l-1]-d(l,r-1));
        
    }
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        memset(dp,-1,sizeof(dp));
        for(int i=1;i<=n;i++)pre[i] = pre[i-1]+stones[i-1];
        return d(1,n);
    }
};
