class Solution {
public:
    int dp[1001][1001];
    int stone[1001];
    int dfs(int i, int j, bool isA, int s)
    {
        if (i == j) return 0;
        if (dp[i][j] != 0) return dp[i][j];
        if (isA)
            return dp[i][j] = max(s-stone[i]+dfs(i+1, j, 0, s-stone[i]), s-stone[j]+dfs(i, j-1, 0, s-stone[j]));
        else
            return dp[i][j] = min(-s+stone[i]+dfs(i+1, j, 1, s-stone[i]), -s+stone[j]+dfs(i, j-1, 1, s-stone[j]));
    }
    int stoneGameVII(vector<int>& stones) {
        int l = stones.size();
        int s = 0;
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < l; i++)
        {
            stone[i] = stones[i];
            s += stone[i];
        }
        return dfs(0, l-1, 1, s);
        
    }
};
