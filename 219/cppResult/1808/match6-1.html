class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<vector<int>> dp(n, vector<int>(n, 0));
        vector<int> diff(n + 1, 0);
        for(int i = 0; i < n; ++i)
            diff[i + 1] = diff[i] + stones[i];
        for(int dif: diff)
            cout<<dif<<' ';
        cout<<endl;

        for(int k = 1; k < n; ++k){
            for(int i = 0; i + k < n; ++i){
                int tmp1 = diff[i + k] - diff[i] - dp[i][i + k - 1];
                int tmp2 = diff[i + k + 1] - diff[i + 1] - dp[i + 1][i + k];
                dp[i][i + k] = max(tmp1, tmp2);
            }
        }
        return dp[0][n - 1];
    }
};
