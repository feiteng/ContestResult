class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<int> sum(n+1,0);
        for(int i = 0; i<n; ++i){
            sum[i+1] = sum[i]+stones[i];
        }
        int as = 0,bs = 0;

        vector<vector<int>> dp(n,vector<int>(n,0));
        for(int l = 1; l<n; ++l){
            for(int i = 0; i+l<n; ++i){
                int j = i+l;
                int all = sum[j+1]-sum[i];
                int goal1 = all-stones[i];
                int goal2 = all-stones[j];
                goal1 -= dp[i+1][j];

                goal2 -= dp[i][j-1];
                if(goal1 > goal2){
                    dp[i][j] = max(dp[i][j],goal1);
                }else{
                    dp[i][j] = max(dp[i][j],goal2);
                }
            }
        }
        return dp[0][n-1];
    }
};
