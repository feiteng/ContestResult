class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<vector<int>> f(n, vector<int>(n, 0));
        vector<int> sum(n + 1, 0);
        for (int i = 0; i < n; ++i)
        {
            sum[i + 1] = sum[i] + stones[i];
        }
        for (int i = 2; i <= n; ++i)
        {
            for (int j = 0; j + i - 1 < n; ++j)
            {
                int r = j + i - 1;
                f[j][r] = max(sum[r] - sum[j] - f[j][r - 1], sum[r + 1] - sum[j + 1] - f[j + 1][r]);
            }
        }
        return f[0][n - 1];
    }
};
