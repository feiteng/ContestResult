class Solution {
public:
    int stoneGameVII(vector<int>& a) {
        int n=a.size();
        vector<int> csum(n+1);
        for(int i=0;i<n;i++)csum[i+1]=csum[i]+a[i];
        
        vector<vector<int>> dp1(n,vector<int>(n)),dp2(n,vector<int>(n));
        // for(int i=0;i<n;i++) dp1[i][i]=dp2[i][i]=0;
        for(int d=1;d<n;d++){
            for(int i=0;i+d<n;i++) {
                int j=i+d;
                dp1[i][j]=max(csum[j]-csum[i]-dp1[i][j-1],csum[j+1]-csum[i+1]-dp1[i+1][j]);
            }
        }
        return dp1[0][n-1];
    }
};
