class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
		int n = stones.size();
		vector<int> preSum(n,0);
		preSum[0]=stones[0];
		for(int i=1;i<n;i++){
			preSum[i]=preSum[i-1]+stones[i];
		}
		int sum[n+1][n+1];
		for(int i=0;i<n;i++){
			for(int j=i;j<n;j++){
				if(i!=0)
					sum[i][j]=preSum[j]-preSum[i-1];
				else
					sum[i][j]=preSum[j];
			}
		}
		int dpmax[n+1][n+1], dpmin[n+1][n+1];
		memset(dpmax, 0, sizeof(dpmax));
		// memset(dpmin, 0, (n+1)*(n+1));
		for(int len=1;len<n;len++){
			for(int i=0;i<n-len;i++){
				dpmax[i][i+len]=max(sum[i+1][i+len]-dpmax[i+1][i+len], sum[i][i+len-1]-dpmax[i][i+len-1]);
			}
		}
		return dpmax[0][n-1];
    }
};
