class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int N = stones.size();
        
        vector<int> pre(N + 1, 0);
        for (int i = 1; i < N + 1; ++i) {
            pre[i] = pre[i - 1] + stones[i - 1];
        }
        
        vector<vector<int>> dp(N);
        for (auto& d : dp) {
            d.resize(N, 0);
        }

        for (int k = 1; k < N; ++k) {
            for (int i = 0; i < N - k; ++i) {
                dp[i][i + k] = max(pre[i + 1 + k - 1] - pre[i] - dp[i][i + k - 1], 
                                   pre[i + 1 + k] - pre[i + 1] - dp[i + 1][i + k]);
            }
        }
        return dp[0][N - 1];
    }
};
