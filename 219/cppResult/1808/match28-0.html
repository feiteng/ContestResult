class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int N = stones.size();
        vector<int> preSum(N+1);
        vector<vector<int>> dp(N, vector<int>(N));
        
        for (int i=0; i<N; i++) {
            preSum[i+1] = preSum[i] + stones[i];
        }
        
        for (int len=2; len<=N; len++) {
            for (int left=0; left+len<=N; left++) {
                int right = left+len-1;
                int scores = preSum[right+1] - preSum[left];
                
                dp[left][right] = max(scores-stones[right]-dp[left][right-1], scores-stones[left]-dp[left+1][right]);
            }
        }
        
        return dp[0][N-1];
    }
};
