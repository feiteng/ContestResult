const int INF=0x3f3f3f3f;
int sum[1005];
int f[1005][1005];

class Solution {
public:
    int dfs(vector<int>& stones, int l,int r){
        if(l==r) return 0;
        if(f[l][r]!=INF) return f[l][r];
        int A = dfs(stones, l+1, r) -(sum[r] - sum[l]);
        int B = dfs(stones, l,r-1) -(sum[r-1]-sum[l-1]);

        return f[l][r]=-min(A,B);
    }
    int stoneGameVII(vector<int>& stones) {
        int n=stones.size();
        memset(f,0x3f,sizeof f);
        for(int i=1;i<=n;i++) sum[i]=sum[i-1]+stones[i-1];
        return dfs(stones, 1, n);
    }
};
