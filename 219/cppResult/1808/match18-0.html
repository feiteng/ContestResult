class Solution {
public:
    int stoneGameVII(vector<int>& a) {
        int n=a.size();
        int DP[n][n];
        for(int i=0;i<n;i++){
            DP[i][i]=0;
        }
        for(int i=0;i<n-1;i++){
            DP[i][i+1]=-max(a[i],a[i+1]);
        }
        for(int i=n-1;i>=0;i--){
            for(int j=i+2;j<n;j++){
                DP[i][j]=min(max(-a[j]+DP[i+1][j-1],-a[i+1]+DP[i+2][j]),max(-a[i]+DP[i+1][j-1],-a[j-1]+DP[i][j-2]));
            }
        }
        return -DP[0][n-1];
    }
};
