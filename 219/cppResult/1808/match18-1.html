class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        
        vector<int> scores(n + 1, 0);
        for (int i = 0; i < n; i++)
        {
            scores[i + 1] = stones[i] + scores[i];
        }
        
        vector<vector<int>> dp(n, vector<int>(n, INT_MIN));
        

        for (int i = 0; i < n; i++)
        {
            dp[i][i] = 0;
        }
        
        for (int d = 1; d < n; d++)
        {
            for (int i = 0; i < n - d; i++)
            {
                dp[i][i + d] = max(scores[i + d + 1] - scores[i + 1] - dp[i + 1][i + d], scores[i + d] - scores[i] - dp[i][i + d - 1]);       
            }
        }
        
        return dp[0][n - 1];
    }
};
