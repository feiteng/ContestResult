class Solution {
public:
    int stoneGameVII(vector<int>& stones) {
        int n = stones.size();
        vector<int> preSum(n + 1, 0);
        for (int i = 0; i < n; ++i) {
            preSum[i + 1] = preSum[i] + stones[i];
        }
        vector<vector<int>> dp(n, vector<int>(n, 0));
        for (int len = 2; len <= n; ++len) {
            for (int st = 0; st + len - 1 < n; ++st) {
                int ed = st + len - 1;
                // cout << preSum[ed + 1] - preSum[st] - dp[st + 1][ed] << endl;
                // cout << preSum[ed] - preSum[st - 1] - dp[st][ed - 1] << endl;
                dp[st][ed] = max(preSum[ed + 1] - preSum[st + 1] - dp[st + 1][ed], preSum[ed] - preSum[st] - dp[st][ed - 1]);
                // cout << "dp[" << st << "][" << ed << "] = " << dp[st][ed] << endl;
            }
        }
        return dp[0][n - 1];
    }
};
