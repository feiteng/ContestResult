class Solution {
public:
    int dp[1005][1005];
    bool c[1005][1005];
    int psum[1005];
    int ss(int l, int r){
        return psum[r+1]-psum[l];
    }
    
    int f(vector<int>& stones, int l, int r){
        if(l>r) {
            return 0;
        }
        if(c[l][r]) {
            return dp[l][r];
        }
        
        int ret = ss(l+1, r)- f(stones, l+1, r);
        ret = max(ret, ss(l, r-1) - f(stones, l, r-1));
        
        c[l][r]=true;
        dp[l][r]=ret;
        
        return ret;
    }
    
    int stoneGameVII(vector<int>& stones) {
        memset(c, false, sizeof(c));
        psum[0] = 0;
        for(int i = 0; i < stones.size(); i++){
            psum[i+1] = psum[i] + stones[i];
        }
        return f(stones, 0, stones.size() - 1);
    }
};
