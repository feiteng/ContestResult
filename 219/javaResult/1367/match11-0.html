class Solution {
    public int maxHeight(int[][] cuboids) {
        for (int[] arr : cuboids) {
            Arrays.sort(arr);
        }
        Arrays.sort(cuboids, (a, b)->{
            if (a[0] == b[0]) {
                if (a[1] == b[1])
                    return a[2]-b[2];
                return a[1]-b[1];
            }
            return a[0]-b[0];
        });
        int n = cuboids.length;
        int[] dp = new int[n];
        int max = 0;
        for (int i = n-1; i >= 0; i--) {
            dp[i] = cuboids[i][2];
            for (int k = i + 1; k <= n-1; k++) {
                if (cuboids[k][1] >= cuboids[i][1] && cuboids[k][2] >= cuboids[i][2]) {
                    dp[i] = Math.max(dp[i], cuboids[i][2] + dp[k]);
                } 
            }
            max = Math.max(max, dp[i]);
        }
        return max;
    }
}
