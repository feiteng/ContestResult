class Solution {
    public int maxHeight(int[][] boxes) {
        for (int[] box : boxes) {
            Arrays.sort(box);
        }

        Arrays.sort(boxes, (
a, b) -> {
            if (a[0] == b[0]) {
                if (a[1] == b[1]) {
                    return a[2] - b[2];
                }
                return a[1] - b[1];
            }
            return a[0] - b[0];
        });

        final int N = boxes.length;
        
int[] dp = new int[N];
        int res = 0;

        for (int i = N - 1; i >= 0; --i) {
            dp[i] = boxes[i][2];
            for (int j = i + 1; j < N; ++j) {
                if (canFit(boxes[i], boxes[j])) {
                    dp[i] = Math.max(dp[i], boxes[i][2] + dp[j]);
                }
            }
            res = Math.max(res, dp[i]);
        }

        return res;
    }

    private boolean canFit(int[] box1, int[] box2) {
        return box1[0] <= box2[0]
            && box1[1] <= box2[1]
            && box1[2] <= box2[2];
    }
}
