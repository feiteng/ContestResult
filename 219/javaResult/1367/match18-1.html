class Solution {
    public int maxHeight(int[][] cuboids) {
        int n = cuboids.length;
        for(int[] cuboid : cuboids){
            Arrays.sort(cuboid);
        }
        
        Arrays.sort(cuboids, (c1, c2) -> (c1[2] == c2[2] ? c1[1] == c2[1] ? c1[0] - c2[0] : c1[1] - c2[1] : c1[2] - c2[2]));
        
        
        int[] dp = new int[n];//前i个长方体的最大高度
        dp[0] = cuboids[0][2];
        int res = dp[0];
        for(int i = 1; i < n; i++){
            dp[i] = cuboids[i][2];
            for(int j = 0; j < i; j++){
                if(isBigger(cuboids[i], cuboids[j]))
{
                    dp[i] = Math.max(dp[i], dp[j] + cuboids[i][2]);
                }
            }
            res = Math.max(res, dp[i]);
        }
        return res;
        
    }
    
    public boolean isBigger(int[] c1, int[] c2){
        return c1[0] >= c2[0] && c1[1] >= c2[1] && c1[2] >= c2[2];
    }
}
