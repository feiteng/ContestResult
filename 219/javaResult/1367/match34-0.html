class Solution {
    private int res = 0;
    public int maxHeight(int[][] cuboids) {
        for (int[] c : cuboids) {
            Arrays.sort(c);
        }
        Arrays.sort(cuboids, (a, b) -> b[0] + b[1] + b[2] - a[0] - a[1] - a[2]);
        int n = cuboids.length;
        dfs(cuboids, 0, Integer.MAX_VALUE, Integer.MAX_VALUE, Integer.MAX_VALUE, 0, n);
        return res;
    }
    private void dfs(int[][] cuboids, int index, int l, int w, int h, int total, int n) {
        if (index == n) {
            res = Math.max(res, total);
            return;
        }
        if (cuboids[index][0] <= l && cuboids[index][1] <= w && cuboids[index][2] <= h) {
            dfs(cuboids, index + 1, cuboids[index][0], cuboids[index][1], cuboids[index][2], total + cuboids[index][2], n);
        }
        dfs(cuboids, index + 1, l, w, h, total, n);
    }
}
