class Solution {
    public int maxHeight(int[][] cuboids) {
        for (int i = 0; i < cuboids.length; i++){
            Arrays.sort(cuboids[i]);
        }
        
        Arrays.sort(cuboids, (int[] c1, int[] c2) -> c1[0] != c2[0] ? c1[0] - c2[0] : (c1[1] != c2[1] ? c1[1] - c2[1] : c1[2] - c2[2]));
        
        
int[] maxLen = new int[cuboids.length];
        for (int i = 0; i < maxLen.length; i++){
            maxLen[i] = cuboids[i][2];
        }
        
        for (int i = 1; i < cuboids.length; i++){
            for (int j = 0; j < i; j++){
                if (cuboids[j][0] <= cuboids[i][0] && cuboids[j][1] <= cuboids[i][1] && cuboids[j][2] <= cuboids[i][2]){
                    maxLen[i] = Math.max(maxLen[i], maxLen[j] + cuboids[i][2]);
                
}
            }
        }
        
        int result = 0;
        for (int i = 0; i < maxLen.length; i++){
            result = Math.max(result, maxLen[i]);
        }
        
        return result;
    }
}
