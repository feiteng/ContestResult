class Solution {
    public int maxHeight(int[][] cuboids) {
        
        
        for(int i=0;i<cuboids.length;i++)
        {
            Arrays.sort(cuboids[i]);
        }
        
        Arrays.sort(cuboids, new Compare());
        int res=cuboids[0][2];
        
        int dp[]=new int[cuboids.length];
       
        for(int i=0;i<cuboids.length;i++)
        {
            dp[i]=cuboids[i][2];
        }
        for(int i=0;i<cuboids.length;i++)
        {
            int[] prev=cuboids[i];
            for(int j=i+1;j<cuboids.length;j++)
            {
                if(isEligible(prev,cuboids[j]))
                {
                   // System.out.println("I="+i+"|J="+j);
                    dp[j]=Math.max(cuboids[j][2]+dp[i],dp[j]);
                   // prev=cuboids[i];
                }
                 res=Math.max(res,dp[j]);
            }
            res=Math.max(res,dp[i]);
        }
        
        return res;
    }
    
    private boolean isEligible(int[] item1,int[] item2)
    {
        if((item1[0]<=item2[0]) && (item1[1]<=item2[1]) && (item1[2]<=item2[2]))
            return true;
        return false;
    }
}
                       

class Compare implements Comparator<int[]>{
    
    public int compare(int[] a,int[] b)
    {
        if(a[2]==b[2])
        {
            if(a[1]==b[1])
            {
               return a[0]-b[0];
              
            }else{
                return a[1]-b[1];
            }
        }else{
            return a[2]-b[2];
        }
    }
}
