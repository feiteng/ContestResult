class Solution {
    public int maxHeight(int[][] cuboids) {
        
int n = cuboids.length;
        int[][] ary = new int[n * 6][];
        int idx = 0;
        for(int i = 0; i < n; i++)
        {
            int w = cuboids[i][0], h = cuboids[i][1], l = cuboids[i][2];
            ary[idx++] = new int[]{w, h, l, i, w * h};
            ary[idx++] = new int[]{h, w, l, i, w * h};
            ary[idx++] = new int[]{w, l, h, i, w * l};
            ary[idx++] = new int[]{l, w, h, i, w * l};
            ary[idx++] = new int[]{h, l, w, i, h * l};
            ary[idx++] = new int[]{l, h, w, i, h * l};
        }
        Arrays.sort(ary, (a, b)->(a[4] == b[4] ? a[2] - b[2] : a[4] - b[4]));
        
        int[] dp = new int[n * 6 + 1];
        int re = 0;
        n = ary.length;
        
for(int i = 0; i < n; i++)
        {
            dp[i + 1] = ary[i][2];
            for(int j = 0; j < i; j++)
            {
                if(ary[j][0] <= ary[i][0] && ary[j][1] <= ary[i][1] && ary[j][2] <= ary[i][2] && ary[j][3] != ary[i][3])
                {
                    dp[i + 1] = Math.max(dp[i + 1], dp[j + 1] + ary[i][2]);
                }
            }
            re = Math.max(re, dp[i + 1]);
        }
        // System.out.println(Arrays.toString(dp));
        return re;
    }
}
