class Solution {
    public int maxHeight(int[][] arr) {
        
        for(int ar[]:arr){
            Arrays.sort(ar);
        }
        int n = arr.length;
        int area[][] = new int[n][4];
        
        for(int i=0;i<n;i++){
            area[i][0] = arr[i][0];
            area[i][1] = arr[i][1];
            area[i][2] = arr[i][2];
            area[i][3] = arr[i][0]*arr[i][1];
        }
        Arrays.sort(area, (a,b)->{
            if(b[3]==a[3]) {
                if(b[2]==a[2]) {
                    return b[1]-a[1];
                }
                return b[2]-a[2];
            }
            return b[3]-a[3];
        });
        int dp[] = new int[n];
        for(int i=0;i<n;i++){
            dp[i] = area[i][2];
        }
        
        for(int i=0;i<n;i++){
            for(int j=0;j<=i;j++){
                if(area[i][0]<=area[j][0] && area[i][1]<=area[j][1] && area[i][2]<=area[j][2])
                dp[i] = Math.max(dp[i], area[i][2] + (i==j?0:dp[j]));
            }
        }
        int ans=0;
        for(int i=0;i<n;i++){
            ans = Math.max(dp[i], ans);
        }
        return ans;
    }
}
