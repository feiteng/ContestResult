class Solution {
    public int maxHeight(int[][] cs) {
        
        for(int c[]:cs){
            Arrays.sort(c);
        }
        Arrays.sort(cs,(x,y)->{
            if(x[2]!=y[2]){
                return Integer.compare(x[2],y[2]);
            }else if(x[1]!=y[1]){
                return Integer.compare(x[1],y[1]);
            }else{
                return Integer.compare(x[0],y[0]);
            
}
        });
        
        int dp[] = new int[cs.length];
        int r=  0;
       for(int i=0;i<cs.length;++i){
           dp[i] = cs[i][2];
           for(int j=0;j<i;++j){
               int ok = 1;
               for(int cc=0;cc<3;++cc){
                   if(cs[j][cc]>cs[i][cc]){
                       ok = 0;break;
                   }
               }
               
if(ok==1){
                   dp[i] = Math.max(dp[i],dp[j]+cs[i][2]);
               }
               
               
           }
           r=  Math.max(r,dp[i]);
       }
        return r;
        
        
    }
}
