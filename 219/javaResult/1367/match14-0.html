class Solution {
    public int maxHeight(int[][] cuboids) {
        for(int i=0;i<cuboids.length;i++){
            Arrays.sort(cuboids[i]);
        }
        Arrays.sort(cuboids, new Comparator<int[]>(){
            public int compare(int[] a, int[] b){
                if(a[2]-b[2]!=0) return a[2]-b[2];
                else if(a[1]-b[1]!=0) return a[1]-b[1];
                else return a[0]-b[0];
            }
        });
        int[] a = new int[cuboids.length];
        a[0] = cuboids[0][2];
        int ans = cuboids[0][2];
        for(int i=1;i<cuboids.length;i++){
            a[i] = cuboids[i][2];
            for(int j=0;j<i;j++){
                if(cuboids[j][0]<=cuboids[i][0] && cuboids[j][1]<=cuboids[i][1] && cuboids[j][1]<=cuboids[i][1] && a[i]<a[j]+cuboids[i][2]){
                    a[i] = a[j]+cuboids[i][2];
                }
            }
            ans = Math.max(ans, a[i]);
        }
        return ans;
    }
}
