class Solution {
    public int maxHeight(int[][] cuboids) {
        int n = cuboids.length;
        int re = 0;
        int[][] pair = new int[n * 6][];
        int idx = 0;
        for(int i = 0; i < n; i++)
        {
            int w = cuboids[i][0], h = cuboids[i][1], l = cuboids[i][2];
            pair[idx++] = new int[]{w, h, l, i};
            pair[idx++] = new int[]{w, l, h, i};
            pair[idx++] = new int[]{h, w, l, i};
            pair[idx++] = new int[]{h, l, w, i};
            pair[idx++] = new int[]{l, w, h, i};
            pair[idx++] = new int[]{l, h, w, i};
        }
        Arrays.sort(pair, (a, b)->(a[0] * a[1] == b[0] * b[1] ? a[2] - b[2] : a[0] * a[1] - b[0] * b[1]));
        n *= 6;
        int[] dp = new int[n];
        for(int i = 0; i < n; i++)
        {
            dp[i] = pair[i][2];
            for(int j = 0; j < i; j++)
            {
                if(pair[j][0] <= pair[i][0] && pair[j][1] <= pair[i][1] && pair[j][2] <= pair[i][2] && pair[j][3] != pair[i][3])
                {
                    dp[i] = Math.max(dp[i], dp[j] + pair[i][2]);
                }
            }
            re = Math.max(re, dp[i]);
        }
        return re;
    }
}
