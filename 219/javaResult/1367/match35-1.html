class Solution {
    public int maxHeight(int[][] cuboids) {
        int n = cuboids.length;
        for (int i = 0; i < n; i++) {
            Arrays.sort(cuboids[i]);
        }
        Arrays.sort(cuboids, (a, b) -> b[0] + b[1] + b[2] - a[0] - a[1] - a[2]);
        
int[][] dp = new int[n + 1][3];
        int maxHeight = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                for (int k = 2; k >= 0; k--) {
                    for (int l = 2; l >= 0; l--) {
                        // System.out.println(i + ", " + j + ", " + k + ", " + l);
                        if (canPlace(cuboids[i - 1], k, j > 0 ? cuboids[j - 1] : new int[] {0, 0, 0}, l)) 
{
                            dp[i][k] = Math.max(dp[i][k], dp[j][l] + cuboids[i - 1][k]);
                        }
                    }
                    maxHeight = Math.max(maxHeight, dp[i][k]);
                }
            }
        }
        return maxHeight;
        
    }
    
    private boolean canPlace(int[] top, int o1, int[] base, int o2) {

        if (base[0] == 0 && base[1] == 0 && base[2] == 0) {
            return true;
        }
        int[] newTop = top;
        if (o1 == 0) {
            newTop = new int[] {top[1], top[2], top[0]};
        }
        else if (o1 == 1) {
            newTop = new int[] {top[0], top[2], top[1]};
        }
        
        int[] newBase = base;
        
        if (o1 == 0) {
            newBase = new int[] {base[1], base[2], base[0]};
        }
        else if (o1 == 1) {
            newBase = new int[] {base[0], base[2], base[1]};
        }
        
        return newTop[0] <= newBase[0] && newTop[1] <= newBase[1] && newTop[2] <= newBase[2];

    }
}
