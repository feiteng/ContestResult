class Solution {
    public int maxHeight(int[][] cuboids) {
        if(cuboids.length == 0) return 0;
        int len = cuboids.length;
        for(int i = 0 ; i < cuboids.length; i++) {
            Arrays.sort(cuboids[i]);
        }
        Arrays.sort(cuboids, (a, b) -> (a[2] == b[2]) ? (a[1] == b[1]?(b[0] - a[0]):(b[1] - a[1])):(b[2] - a[2]));
        int[] dp = new int[len+1];
        
        dp[1] = cuboids[0][2];
        int ret = dp[1];
        for(int i = 1; i < len; i++) {
            dp[i+1] = cuboids[i][2];
            for(int j = 0; j < i; j++) {
                if (cuboids[i][0] <= cuboids[j][0] && cuboids[i][1] <= cuboids[j][1]) {
                    dp[i+1] = Math.max(dp[i+1], dp[j+1] + cuboids[i][2]);
                }
            }
            ret = Math.max(ret, dp[i+1]);
        }

        return ret;
    }
}
