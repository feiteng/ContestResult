class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
		int[] sum = new int[n + 1];
		for(int i = 1; i <= n; ++i) {
			sum[i] = sum[i - 1] + stones[i - 1];
		}

		int[][] dp = new int[n + 1][n + 1];
		for(int i = n; i >= 1; --i) {
			for(int j = i; j <= n; ++j) {
				if(i == j) {

					dp[i][j] = 0;
				} else {
					dp[i][j] = Math.max(sum[j] - sum[i] - dp[i + 1][j], sum[j - 1] - sum[i - 1] - dp[i][j - 1]); 
				}
			}
		}
		return dp[1][n];
    }
}
