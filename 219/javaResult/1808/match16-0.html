class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        int[][] sum = new int[n][n];
        
        for (int i = 0; i < n; ++i) {
            dp[i][i] = 0;
            sum[i][i] = stones[i];
        }
        
        for (int i = 1; i < n; ++i) {
            for (int j = 0; j < n - i; ++j) {
                int k = j + i;
                sum[j][k] = sum[j][k - 1] + stones[k];
                dp[j][k] = Math.max(sum[j][k] - stones[j] - dp[j + 1][k], sum[j][k] - stones[k] - dp[j][k - 1]);
            }
        }
        
        return dp[0][n - 1];
    }
}
