class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] pre = new int[n+1];
        for (int i=0; i<n; i++) {
            pre[i+1] = stones[i] + pre[i];
        }
        
        int[][] dp = new int[n][n];
        for (int length=0; length<=n; length++) {
            for (int start=0; start<n; start++) {
                int end = start+length;
                if (end >= n) {
                    continue;
                }
                if (start==end) {
                    dp[start][end] = 0;
                }
                else {
                    dp[start][end] = Math.max(pre[end+1]-pre[start+1]-dp[start+1][end], pre[end]-pre[start]-dp[start][end-1]);
                }
            }
        }
        return dp[0][n-1];
    }
}
