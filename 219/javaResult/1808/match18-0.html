class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        int[] sums = new int[n+1];
        sums[0] = 0;
        for (int i = 1; i <= n; i++) {
            sums[i] = sums[i-1] + stones[i-1];
        }

        for (int i = 1; i < n; i++) {
            for (int j = 0; i+j <n; j++) {
                dp[j][j+i] = Math.max(-dp[j][j+i-1]+sums[j+i]-sums[j],-dp[j+1][j+i]+sums[j+i+1]-sums[j+1]);
                //System.out.println("j="+j+",dp[j][j+1]="+dp[j][j+1]);
            }
        }
        
        return dp[0][n-1];



    }
}
