class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        int[] sum = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + stones[i - 1];
        }
        for (int len = 1; len <= n; len++) {
            for (int i = 0; i + len - 1 < n; i++) {
                int r = i + len - 1;
                if (i == r) {
                    dp[i][r] = 0;
                } else {
                    dp[i][r] = Math.max(sum[r] - sum[i] - dp[i][r - 1], sum[r + 1] - sum[i + 1] - dp[i + 1][r]);
                }
            }
        }
        return dp[0][n - 1];
    }

}
