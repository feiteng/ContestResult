class Solution {
    public int stoneGameVII(int[] stones) {

       
        int n =stones.length;
        int[][] dp =new int[n][n];
        int[] presum = new int[n];
        presum[0] = stones[0];
        for (int i=1;i<n;i++){
            presum[i] = presum[i-1]+stones[i];
        }
        for (int i=0;i<n;i++){
            for (int j=i;j>=0;j--){
                if (i==j) dp[j][i]= 0;
                else 
{
                    dp[j][i] = Math.max(presum[i]-presum[j]-dp[j+1][i],presum[i-1]-(j==0?0:presum[j-1])-dp[j][i-1]);
                }
                //System.out.println("i="+i+" j="+j+" "+dp[j][i]);
            }
        }
        return dp[0][n-1];
    }
}
