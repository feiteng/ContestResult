class Solution {
    public int stoneGameVII(int[] stones) {
        int len = stones.length;
        int[][] arr = new int[len][len];
        int[] sum = new int[len];
        sum[0] = stones[0];
        for(int i=1;i<len;i++){
            sum[i] = sum[i-1] + stones[i];
        }
        for(int i= len-1;i>=0;i--){
            for(int j= i;j<len;j++){
                if(j==i){
                    arr[i][j] = 0;
                }else{
                    arr[i][j] = Math.max(sum[j-1]-(i==0?0:sum[i-1])-arr[i][j-1],sum[j]-sum[i]-arr[i+1][j]);
                }
            }
        }
        return arr[0][len-1];
    }
}
