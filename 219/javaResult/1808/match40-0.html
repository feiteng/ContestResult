class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] prefixSums = new int[n + 1];
        for (int i = 1; i <= n; i++) prefixSums[i] = prefixSums[i - 1] + stones[i - 1];
        var dp = new int[n][n];
        for (int i = n - 2; i >= 0; i--) {
            for (int j = i + 1; j < n; j++) {
                dp[i][j] = Math.max(prefixSums[j + 1] - prefixSums[i + 1] - dp[i + 1][j], prefixSums[j] - prefixSums[i] - dp[i][j - 1]);
            }
        }
        return dp[0][n - 1];
    }
}
