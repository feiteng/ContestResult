class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        int[] prefix = new int[n+1];
        for(int i=1; i<=n; i++){
            prefix[i] = prefix[i-1] + stones[i-1];
        }
        for(int L = 2; L <= n; L++){
            
            for(int left=0; left<n; left++){
                if(left+ L - 1 >= n) break;
                
                int right = left + L - 1;
                int remove_left = prefix[right+1] - prefix[left+1];
                int remove_right = prefix[right] - prefix[left];
                
                
dp[left][left+L-1] = Math.max(remove_left - dp[left+1][right], remove_right - dp[left][right-1]);
                
                
            }
            
        }
        return dp[0][n-1];
    }
}
