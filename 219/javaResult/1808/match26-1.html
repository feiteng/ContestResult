class Solution {
    
    int[] sum;
    int[][] memo;
    
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        memo = new int[n][n];
        sum = new int[n + 1];
        for(int i = 0; i < n; i++)
            sum[i + 1] = sum[i] + stones[i]
;
        
        return solve(0, stones.length - 1, true);
    }
    
    public int solve(int i, int j, boolean isAlice){
        if(i > j) return 0;
        if(memo[i][j] != 0) return memo[i][j];
        if(isAlice){
            return memo[i][j] = Math.max(sum[j + 1] - sum[i + 1] + solve(i + 1, j, false), sum[j] - sum[i] + solve(i, j - 1, false));
        }else{
            return memo[i][j] = Math.min(solve(i + 1, j, true) - sum[j + 1] + sum[i + 1], solve(i, j - 1, true) - sum[j] + sum[i]);
        }
    }
}
