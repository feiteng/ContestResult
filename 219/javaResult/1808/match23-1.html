class Solution {
 public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] sum = new int[n+1];
        for(int i = 1; i  <= n; ++i){
            sum[i] = sum[i-1] + stones[i-1];
        }
        
        int[][] dp = new int[n][n];
        
        for(int d = 1; d <n; ++d){
            
            for(int i = 0; i + d < n; ++i){
                int j = i+d;
                
dp[i][j] = Math.max(sum[j] - sum[i] - dp[i][j-1], sum[j+1] - sum[i+1] - dp[i+1][j]);
                
                
            }
        }
        
        return dp[0][n-1];
    }
}
