class Solution {
    int[] sum;
    int[][] memo;
    
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        sum = new int[n + 1];
        memo = new int[n + 1][n + 1];
        for (int i = 1; i <= n; ++i) {
            sum[i] = sum[i - 1] + stones[i - 1];
        }
        
        return solve(1, n, 0);
    }
    
    private int solve(int l, int r, int t) {
        if (l >= r) return 0;
        if (memo[l][r] > 0) return memo[l][r];
        
        if (t % 2 == 0) {
            return memo[l][r] = Math.max(
                sum[r] - sum[l] + solve(l + 1, r, t + 1),
                sum[r - 1] - sum[l - 1] + solve(l, r - 1, t + 1)
            );
        } else {
            return memo[l][r] = Math.min(
                (sum[r] - sum[l]) * -1 + solve(l + 1, r, t + 1),
                (sum[r - 1] - sum[l - 1]) * -1 + solve(l, r - 1, t + 1)
            );
        }
    }
}
