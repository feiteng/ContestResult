class Solution {
    public int stoneGameVII(int[] stones) {
        int len = stones.length;
        int[] cum = new int[len + 1];
        for(int i = 1 ; i <= len ; i++)
            cum[i] = cum[i - 1] + stones[i - 1];
        int[][] dp = new int[len][len];
        for(int i = 0 ; i < len ; i++)
            dp[0][i] = stones[i];
        for(int i = 1 ; i < len ; i++) {
            dp[1][i] = Math.max(dp[0][i - 1], dp[0][i]);
        }
        
for(int i = 2 ; i < len ; i++) {
            for(int j = i ; j < len ; j++) {
                dp[i][j] = Math.max(cum[j] - cum[j - i] - dp[i - 1][j - 1], cum[j + 1] - cum[j - i + 1] - dp[i - 1][j]);
            }
        }
        /*for(int i = 0 ; i < len ; i++) {
            for(int j = 0 ; j < len ; j++) {
                System.out.print(dp[i][j] + "\t");
            }
            System.out.print("\n");
        }*/
        return dp[len - 1][len - 1];
    }
}
