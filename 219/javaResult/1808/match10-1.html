class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] prefix = new int[n + 1];
        for(int i = 0; i < n; i++) {
            prefix[i + 1] = prefix[i] + stones[i];
        }
        
        int[][] dp = new int[n][n];
        for(int d = 2; d <= n; d++) {
            for(int i = 0; i + d <= n; i++) {
                int j = i + d - 1;
                int left = prefix[j + 1] - prefix[i + 1] - dp[i + 1][j];
                int right = prefix[j] - prefix[i] - dp[i][j - 1];
                
dp[i][j] = Math.max(left, right);
            }
        }
        return dp[0][n - 1];
    }
}
