class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] sums = new int[n + 1];
        for (int i = 0; i < n; i++) {
            sums[i + 1] = sums[i] + stones[i];
        }
        int[][] dp = new int[n][n];
        for (int k = 1; k < n; k++) {
            for (int i = 0, j = k; j < n; i++, j++) {
                dp[i][j] = Math.max(sums[j + 1] - sums[i + 1] - dp[i + 1][j], sums[j] - sums[i] - dp[i][j - 1]);
            }
        }
        return dp[0][n - 1];
    }
}
