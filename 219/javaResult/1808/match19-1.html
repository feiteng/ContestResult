class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        int[] preSum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            preSum[i + 1] = preSum[i] + stones[i];
        }
        for (int l = 2; l <= n; l++) {
            for (int i = 0; i + l <= n; i++) {
                int j = i + l - 1;
                int pl = preSum[j + 1] - preSum[i + 1];
                int pr = preSum[j] - preSum[i];
                
dp[i][j] = Math.max(pl - dp[i + 1][j], pr - dp[i][j - 1]);
            }
        }
        return dp[0][n - 1];
    }
}
