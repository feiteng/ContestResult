class Solution {
  public int stoneGameVII(int[] stones) {
    final int N = stones.length;
    int[] prefix = new int[N + 1];
    for (int i = 0; i < N; i++) {
      prefix[i + 1] = prefix[i] + stones[i];
    }
    
    int[][] M = new int[N][N];
    for (int i = N - 1; i >= 0; i--) {
      for (int j = i + 1; j < N; j++) {
        
M[i][j] = Math.max(prefix[j + 1] - prefix[i + 1] - M[i + 1][j],
                           prefix[j] - prefix[i] - M[i][j - 1]);
      }    
    }
    return M[0][N - 1];
  }
}
