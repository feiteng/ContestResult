class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] preSum = new int[n + 1];
        for (int i = 0; i < n; i++){
            preSum[i + 1] = stones[i] + preSum[i];
        }
        int[][] dp = new int[n][n];
        for (int i = n - 1; i >= 0; i--){
            for (int j = i + 1; j < n; j++){
                int sum = preSum[j + 1] - preSum[i];
                dp[i][j] = Math.max(sum - stones[i] - dp[i + 1][j], sum - stones[j] - dp[i][j - 1]);
                
            }
        }
        return dp[0][n - 1];
    }
}
