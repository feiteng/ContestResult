class Solution {
    public static int stoneGameVII(int[] stones) {

        int len = stones.length;
        int[] sum = new int[len];
        sum[0] = stones[0];
        for (int i = 1; i < len; i++) {
            sum[i] = sum[i - 1] + stones[i];
        }
        int[][] dp = new int[len][len];

        for (int i = len - 2; i >= 0; i--) {
            for (int j = i + 1; j < len; j++) {
                if (i == 0) {
                    dp[i][j] = Math.max(sum[j] - sum[i] - dp[i + 1][j], sum[j - 1] - dp[i][j - 1]);
                }else {
                    dp[i][j] = Math.max(sum[j] - sum[i] - dp[i + 1][j], sum[j - 1] - sum[i - 1] - dp[i][j - 1]);
                }
            }
        }
        return dp[0][len - 1];
    }
}
