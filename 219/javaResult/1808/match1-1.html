class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        // diff is n - 1 then increase
        int[] sum = new int[n + 1];
        for (int i = 0; i < n; i++) {
            // sum[i + 1] = sum[0..i]
            sum[i + 1] = sum[i] + stones[i];
        }
        
        // for (int i = 0; i < n; i++) {
        //     dp[i][i] = ((n - 1) % 2 == 0) ? stones[i] : -stones[i];
        // }
        for (int len = 1; len < n; len++) {
            for (int i = 0; i + len < n; i++) {
                int j = i + len;
                if ((n - 1 - len) % 2 == 0) {
                    dp[i][j] = Math.max(dp[i + 1][j] + sum[j + 1] - sum[i + 1], dp[i][j - 1] + sum[j] - sum[i]);
                
} else {
                    dp[i][j] = Math.min(dp[i + 1][j] - sum[j + 1] + sum[i + 1], dp[i][j - 1] - sum[j] + sum[i]);
                }
            }
        }
        
        return dp[0][n - 1];
    }
}
