class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[][] dp = new int[n][n];
        int[] sum = new int[n + 1];
        for (int i = 1; i <= n; i++) {
            sum[i] = sum[i - 1] + stones[i - 1];
        }
        for (int j = 1; j < n; j++) {
            for (int i = j - 1; i >= 0; i--) {
                dp[i][j] = Math.max(sum[j] - sum[i] - dp[i][j - 1], sum[j + 1] - sum[i + 1] - dp[i + 1][j]);
            }
        }
        return dp[0][n - 1];
    }
}
