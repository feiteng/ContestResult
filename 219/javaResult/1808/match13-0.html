class Solution {
    public int stoneGameVII(int[] stones) {
        int n = stones.length;
        int[] preSum = new int[n+1];
        int[][] dp = new int[n][n];
        for (int i=1; i<=n; ++i) {
            preSum[i] = stones[i-1] + preSum[i-1];
        }
        for (int len=2; len<=n; ++len) {
            int bound = n - len;
            for (int i=0; i<=bound; ++i) {
                int j = i + len - 1;
                // System.out.println("left: " + (preSum[j] - preSum[i] - dp[i][j-1]));
                // System.out.println("preSum[" + (j+1) + "] = " + preSum[j+1]);
                // System.out.println("preSum[" + (i+1) + "] = " + preSum[i+1]);
                // System.out.println("dp[" + (i+1) + "][" + j + "] = " + dp[i+1][j]);
                // System.out.println("right: " + (preSum[j+1] - preSum[i+1] - dp[i+1][j]));
                dp[i][j] = Math.max(preSum[j] - preSum[i] - dp[i][j-1], preSum[j+1] - preSum[i+1] - dp[i+1][j]);
                // System.out.println(i + ", " + j + ": " + dp[i][j]);
            }
        }
        return dp[0][n-1];
    }
}
