class Solution {
    public int stoneGameVII(int[] stones) {
        
        int n = stones.length;
        //int[] pre = new int[n + 1];
        
        //for(int i = 1; i < n; i++) {
          //  pre[i] = pre[i - 1] + stones[i - 1];
        //}
        int[][] s = new int[n][n];
        
        for(int i= 0 ;i<n;i++){
            s[i][i] = stones[i];
            for(int j=i+1;j<n;j++){
                s[i][j] = s[i][j-1] + stones[j];
            }
        
}

        int[][] dp = new int[n][n];
        
        
        for(int i = n - 1; i >= 0; i--) {
            for(int j = i; j < n; j++) {
                if(i == j) {
                    dp[i][j] = 0;
                } else {
                    dp[i][j] = Math.max(s[i + 1][j] - dp[i + 1][j], s[i][j - 1] - dp[i][j - 1]);
                }
                
            }
        }
        
        
        return dp[0][n - 1];
        
        
        
        

    }
}
