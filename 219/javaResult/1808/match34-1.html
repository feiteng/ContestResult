class Solution {
    public int stoneGameVII(int[] st) {
        int n = st.length;
        int[][] sums = new int[n][n];
        for (int i = 0; i < n; ++i) {
            sums[i][i] = st[i];
            for (int j = i+1; j < n; ++j)
                sums[i][j] = sums[i][j-1]+st[j];
        
}
        long[][] dp = new long[n][n];
        for (int z = 1; z < n; ++z) {
            for (int i = 0; i+z < n; ++i) {
                int j = i+z;
                
dp[i][j] = Math.max(sums[i][j-1]-dp[i][j-1], sums[i+1][j]-dp[i+1][j]);
            }
        }
        return (int)dp[0][n-1];
    }
}
