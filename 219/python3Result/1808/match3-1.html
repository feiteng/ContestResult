class Solution:
    def stoneGameVII(self, a: List[int]) -> int:
        n=len(a)
        s=[[0]*n for _ in range(n)]
        for i in range(1,len(a)): a[i]+=a[i-1]
        a.append(0)
        for j in range(1,n):
            for i in range(n-j): s[i][i+j]=max(a[i+j]-a[i]-s[i+1][i+j],a[i+j-1]-a[i-1]-s[i][i+j-1])
        return s[0][-1]
