class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        sums = [0]
        for stone in stones:
            sums.append(sums[-1]+stone)
        n = len(stones)
        dp = [[0 for j in range(n)] for i in range(n)]
        for l in range(2, n+1):
            for a in range(n-l+1):
                b = a + l - 1
                if b - a == 1:
                    dp[a][b] = max(stones[a], stones[b])
                else:
                    total = sums[b+1]-sums[a]
                    dp[a][b] = max(total-stones[a]-dp[a+1][b], total-stones[b]-dp[a][b-1])
        return dp[0][-1]
