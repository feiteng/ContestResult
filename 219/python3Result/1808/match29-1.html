class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        sums = [0]
        for s in stones:
            sums.append(sums[-1] + s)
        n = len(stones)
        dp = [[0] * (n + 1) for _ in range(n + 1)]
        for diff in range(2, n + 1):
            for l in range(0, n + 1 - diff):
                r = l + diff
                t = (n - diff) % 2
                slr = sums[r] - sums[l]
                if t:
                    dp
[l][r] = min(dp[l+1][r] - (slr - stones[l]), dp[l][r-1] - (slr - stones[r-1]))
                else:
                    dp[l][r] = max(dp[l+1][r] + slr - stones[l], dp[l][r-1] + slr - stones[r-1])
        return dp[0][n]
