class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        prefix = [0]
        for s in stones:
            prefix.append(prefix[-1]+s)
        total = sum(stones)
        # @lru_cache(None)
        # def dp(i,j):
        #     if i == j:
        #         return 0
        #     else:
        #         return max(prefix[j+1]-prefix[i+1]-dp(i+1,j),prefix[j]-prefix[i]-dp(i,j-1))
        # return dp(0,len(stones)-1)
        n = len(stones)
        dp = [[0 for _ in range(n)] for _ in range(n)]
        for diff in range(1,n):
            for i in range(n):
                if i+diff < n:
                    j = i+diff
                    dp[i][j] = max(prefix[j+1]-prefix[i+1]-dp[i+1][j],prefix[j]-prefix[i]-dp[i][j-1])
        return dp[0][n-1]
