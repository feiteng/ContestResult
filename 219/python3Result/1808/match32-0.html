class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n=len(stones)
        p=[0]
        for num in stones:
            p.append(p[-1]+num)
        dp=[[float('-inf') for _ in range(n)] for _ in range(n)]
        for i in range(n):
            dp[i][i]=0

        for i in range(n-1,-1,-1):
            for j in range(i+1,n):
                dp[i][j]=max(p[j+1]-p[i+1]-dp[i+1][j],p[j]-p[i]-dp[i][j-1])

        return dp[0][n-1]
