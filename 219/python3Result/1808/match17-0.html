class Solution:
    def stoneGameVII(self, stones):
        lens = len(stones)
        cursum = [0] * (lens + 1)
        for i in range(1, lens + 1):
            cursum[i] = cursum[i - 1] + stones[i - 1]
        dp = [[0 for _ in range(lens)] for _ in range(lens)]
        for l in range(1, lens):
            for i in range(lens - l):
                dp[i][i + l] = max(-dp[i][i + l - 1] + cursum[i + l] - cursum[i], -dp[i + 1][i + l] + cursum[i + l + 1] - cursum[i + 1])
        return dp[0][-1]
