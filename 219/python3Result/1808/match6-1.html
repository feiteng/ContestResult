class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        summ = [0]*(n+1)
        for i in range(1, n+1):
            summ[i] = stones[i-1] + summ[i-1]
        
        dp = [[0]*n for _ in range(n)]
        for i in range(n):
            dp[i][i] = 0
        for j in range(n):
            for i in range(j-1, -1, -1):
                dp
[i][j] = max(summ[j+1] - summ[i+1] - dp[i+1][j],
                               summ[j] - summ[i] - dp[i][j-1])
        # print(dp)
        return dp[0][n-1]
