class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        dp = [[0]*n for i in range(n)]
        psum = [0]*(n+1)
        # print(dp)
        for i,num in enumerate(stones):
            psum[i+1] = psum[i]+num
        for j in range(n):
            for i in range(j-1,-1,-1):
                dp[i][j] = max(psum[j+1]-psum[i+1]-dp[i+1][j], psum[j]-psum[i]-dp[i][j-1])
                # print(i,j,dp[i][j])
        return dp[0][-1]
