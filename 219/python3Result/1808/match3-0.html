class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n=len(stones)
        s=[[0]*n for _ in range(n)]
        for i in range(1,len(stones)): stones[i]+=stones[i-1]
        stones.append(0)
        for j in range(1,n):
            for i in range(n-j):
                s[i][i+j]=max(stones[i+j]-stones[i]-s[i+1][i+j],stones[i+j-1]-stones[i-1]-s[i][i+j-1])
        return s[0][-1]
