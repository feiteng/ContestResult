class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        prefix = [0] * (n + 1)
        for i, _ in enumerate(stones):
            prefix[i+1] = prefix[i] + stones[i]
        
        dp_alice = [[0]*n for i in range(n)]
        dp_bob = [[0]*n for i in range(n)]
        
        for interval in range(1, n):
            for left in range(n-interval):
                right = left + interval
                dp_alice[left][right] = max(dp_bob[left+1][right] + (prefix[right+1] - prefix[left+1]), 
                                            dp_bob[left][right-1] + (prefix[right] - prefix[left]))
                dp_bob
[left][right] = min(dp_alice[left+1][right] - (prefix[right+1] - prefix[left+1]), 
                                          dp_alice[left][right-1] - (prefix[right] - prefix[left]))
        
        return dp_alice[0][n-1]
