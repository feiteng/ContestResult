class Solution:
    def stoneGameVII(self, nums: List[int]) -> int:
        n = len(nums)
        dp = [[0] * n for _ in range(n)]
        prefix = [0] * (n + 1)
        for i in range(n):
            prefix[i + 1] = prefix[i] + nums[i]
            
        for i in range(1, n):
            dp[i - 1][i] = max(nums[i - 1], nums[i])

        for span in range(2, n):
            for left in range(n - span):
                right = left + span
                dp
[left][right] = max(prefix[right + 1] - prefix[left + 1] - dp[left + 1][right],
                                      prefix[right] - prefix[left] - dp[left][right - 1])
        return dp[0][-1]
