class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        presum = [0]
        for s in stones:
            presum.append(presum[-1]+s)
        n = len(stones)
        dp = [[0]*n for i in range(n)]
        
        for gap in range(1,n):
            for i in range(n-gap):
                j = i+gap
                curr 
= presum[j+1]-presum[i]
                dp[i][j] = max(curr-stones[i]-dp[i+1][j],curr-stones[j]-dp[i][j-1])
                
        return dp[0][-1]
