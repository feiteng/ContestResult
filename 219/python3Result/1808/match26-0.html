class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:

        presum = [0]

        for v in stones:
            presum.append(presum[-1] + v)

        n = len(stones)
        dp = [[0 for _ in range(n)] for _ in range(n)]

        for k in range(1, n):
            for i in range(n):
                if i + k >= n:
                    break

                s = presum[i+k+1] - presum[i]
                dp[i][i + k] = s - min(stones[i] + dp[i+1][i+k], stones[i+k] + dp[i][i+k-1])

        return dp[0][n-1]
