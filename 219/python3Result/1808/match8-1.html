class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        dp = [[0] * n for _ in range(n)]
        prefix_sum = [0] * (n+1)
        
        for i in range(1, n+1):
            prefix_sum[i] = prefix_sum[i-1] + stones
[i-1]
            
        
        for d in range(1, n):
            for i in range(n-d):
                dp[i][i+d] = max(prefix_sum[i+d+1] - prefix_sum[i+1] - dp[i+1][i+d], prefix_sum[i+d] - prefix_sum[i] - dp[i][i+d-1])
        
        return dp[0][-1]
