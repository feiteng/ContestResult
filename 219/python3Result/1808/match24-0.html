class Solution:
    def stoneGameVII(self, ss: List[int]) -> int:
        n=len(ss)
        dp=[[0]*(n+1) for _ in range(n+1) ]
        acc=[0]*(n+1)
        for i in range(n):
            acc[i+1]=acc[i]+ss[i]
        for L in range(2, n+1):
            for i in range(0, n-L+1):
                if L==2:
                    dp[L][i]=max(ss[i], ss[i+1])
                else:
                    #print("select from (%d,%d)"%(acc[i+L]-acc[i+1]-dp[L-1][i+1],acc[i+L-1]-acc[i]-dp[L-1][i+1]))
                    dp[L][i]=max(acc[i+L]-acc[i+1]-dp[L-1][i+1],acc[i+L-1]-acc[i]-dp[L-1][i])
                #print("dp[%d][%d]:%d"%(L,i,dp[L][i]))
        return dp[n][0]
