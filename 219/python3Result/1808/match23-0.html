class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n=len(stones)
        sumS=[0]
        for i in stones:
            sumS.append(sumS[-1]+i)
        # print(sumS)
        dp=[[0]*(n+1) for i in range(n+1)]

        for j in range(1,n+1):
            for i in range(n-j):
                s=sumS[i+j+1]-sumS[i]
                dp[i][i+j]=max(s-stones[i]-dp[i+1][i+j], s-stones[i+j]-dp[i][i+j-1])
        # print(dp)
        return dp[0][n-1]
