class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        prefix = [0]*(n+1)
        for i in range(n):
            prefix[i+1] = prefix[i]+stones[i]
        
        dp = [[0]*n for _ in range(n)]
        for j in range(n):
            for i in range(j-1, -1, -1):
                dp
[i][j] = max(prefix[j+1] - prefix[i+1] - dp[i+1][j],prefix[j] - prefix[i] - dp[i][j-1])
        return dp[0][n-1]
