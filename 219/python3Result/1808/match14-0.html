class Solution:
    def stoneGameVII(self, piles: List[int]) -> int:
        N = len(piles)
        presum = [0] * (N+1)
        dp = [[0] * N for _ in range(N)]
        for i, p in enumerate(piles):
            presum[i+1] = presum[i] + p 
            # dp[i][i] = piles[i]
        # print(presum)

#         @lru_cache(None)
#         def dp(i, j):
#             # The value of the game [piles[i], piles[i+1], ..., piles[j]].
#             if i > j: 
#                 return 0
#             parity = (j - i - N) % 2
#             if parity == 1:  # first player
#                 return max(presum[j+1] - presum[i+1] + dp(i+1,j), presum[j] - presum[i] + dp(i,j-1))
#             else:
#                 return min(-presum[j+1] + presum[i+1] + dp(i+1,j), -presum[j] + presum[i] + dp(i,j-1))

#         return dp(0, N - 1)
            
        for j in range(1, N):
            for i in range(0, N-j):
                if (j-N)%2 == 1:
                    dp[i][i+j] = max(presum[j+i+1] - presum[i+1] + dp[i+1][j+i], presum[j+i] - presum[i] + dp[i][j+i-1])
                else:
                    dp[i][i+j] = min(-presum[j+i+1] + presum[i+1] + dp[i+1][j+i], -presum[j+i] + presum[i] + dp[i][j+i-1])
            # print(i, j, dp)        
        return dp[0][N-1]
