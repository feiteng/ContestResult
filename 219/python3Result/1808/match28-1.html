class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        if n < 1: return 0
        if n < 2: return stones[0]
        psum 
= [0] * (n + 1)
        for i in range(n):
            psum[i] = psum[i-1] + stones[i]
            
        dp = [[0] * (n + 1) for _ in range(n+1)]
        for i in range(n-2, -1, -1):
            for j in range(i+1, n):
                ssum = psum[j] - psum[i-1]
                dp[i][j] = ssum - min(stones[i] + dp[i+1][j], stones[j] + dp[i][j-1])
        return dp[0][n-1]
