class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        s = n%2
        cnt = [0 for i in range(n+1)]
        for i in range(n):
            cnt[i+1] = cnt[i]+stones[i]
        f = [[0 for i in range(n)] for j in range(n)]
        
for i in range(1, n):
            for j in range(n-i):
                if i%2==s:
                    f[j][j+i] = min(f[j][j+i-1]-(cnt[j+i]-cnt[j]), f[j+1][j+i]-(cnt[j+i+1]-cnt[j+1])) 
                else:
                    f[j][j+i] = max(f[j][j+i-1]+cnt[j+i]-cnt[j], f[j+1][j+i]+cnt[j+i+1]-cnt[j+1]) 
        return f[0][n-1]
