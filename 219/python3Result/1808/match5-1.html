class Solution:
    def stoneGameVII(self, nums: List[int]) -> int:
        n = len(nums)
        pre = [0] * (n+1)
        for i in range(n):
            pre[i+1] = pre[i] + nums[i]
        dp = [[0]*n for _ in range(n)]

        for i in range(n-1,-1,-1):
            for j in range(i+1,n):
                sm = pre[j+1]-pre[i]
                dp[i][j] = max(
                    sm - nums
[i] - dp[i+1][j],
                    sm - nums[j] - dp[i][j-1]
                )
        
        return dp[0][n-1]
