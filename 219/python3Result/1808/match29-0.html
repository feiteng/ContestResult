class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        pre = [0]
        for stone in stones:
            pre.append(pre[-1]+stone)
            
        n = len(stones)
        dp = [[float("-inf") for _ in range(n+1)] for _ in range(n+1)]
        for length in range(1, n+1):
            for i in range(n-length+1):
                if length==1:
                    dp[i][i+length]=0
                else:
                    dp[i][i+length] = max(dp[i][i+length], pre[i+length]-pre[i+1]-dp[i+1][i+length])
                    dp[i][i+length] = max(dp[i][i+length], pre[i+length-1]-pre[i]-dp[i][i+length-1])

        
        return dp[0][n]
