class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        d = [0]
        n = len(stones)
        for num in stones:
            d.append(d[-1]+num)
        dp = [[0] * n for _ in range(n)]
        for i in range(n):
            for j in reversed(range(i)):
                dp[i][j] = max(d[i]-d[j]-dp[i-1][j], d[i+1]-d[j+1]-dp[i][j+1])
            # print(dp)
        return dp[-1][0]
