class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        S = [0]
        n = len(stones)
        for s in stones:
            S.append(S[-1]+s)
        dp = [[-float('inf')]*n for _ in stones]
        for i in range(n):
            dp[i][i]=0
        for d in range(1,n):
            for i in range(n-d):
                j = i+d
                dp[i][j]=max(S[j]-S[i]-dp[i][j-1],
                            S[j+1]-S[i+1]-dp[i+1][j])
        #for row in dp:
        #   print(row)
        return dp[0][-1]
