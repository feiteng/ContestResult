class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        cumsum = [0]
        for x in stones:
            cumsum.append(cumsum[-1]+x)
        n = len(stones)
        dp = [[0]*n for _ in range(n)]
        
for d in range(n):
            for i in range(n-d):
                j = d + i
                if i!=j:
                    dp[i][j] = max(cumsum[j+1] - cumsum[i+1] - dp
[i+1][j], cumsum[j] - cumsum[i] - dp[i][j-1])

        # @functools.lru_cache(None)
        # def dp(i, j):
        #     if i == j:
        #         return 0
        #     return max(cumsum[j+1] - cumsum[i+1] - dp(i+1, j), cumsum[j] - cumsum[i] - dp(i, j-1))
        return dp[0][n-1]
