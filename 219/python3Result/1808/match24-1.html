class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        dp = [[0] * n for _ in range(n)]
        pre_sum = [0] * (n+1)
        for i, s in enumerate(stones):
            pre_sum[i+1] = pre_sum
[i] + s
        # print(pre_sum)
        # dp[i][j] 
        # dp[i][j] = sum(j) - sum(i) - (dp[i+1][j])
        # dp[i][j] = sum(j-1) - sum(i-1) - dp[i][j-1]
        for i in range(n-1, -1, -1):
            for j in range(i, n):
                if i + 1 < n:
                    dp[i][j] = max(dp[i][j], pre_sum[j+1] - pre_sum[i+1] - dp[i+1][j])
                if j-1 >= 0:
                    dp[i][j] = max(dp
[i][j], pre_sum[j] - pre_sum[i] - dp[i][j-1])
        # print(dp)
        return dp[0][n-1]
