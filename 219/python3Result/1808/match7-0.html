class Solution:
    def stoneGameVII(self, stones: List[int]) -> int:
        n = len(stones)
        cumsum = [0] * (n + 1)
        for i in range(n):
            cumsum[i + 1] = cumsum[i] + stones[i]
        
        dp = [[0] * n for i in range(n)]  # best score difference can get from stones[i:j + 1]
        for k in range(1, n):
            for i in range(n - k):
                j = i + k
                dp[i][j] = max(cumsum[j + 1] - cumsum[i + 1] - dp[i + 1][j], cumsum[j] - cumsum[i] - dp[i][j - 1])  
        return dp[0][n - 1]
