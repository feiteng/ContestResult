class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        n = len(cuboids)
        for i in range(n):
            cuboids[i].sort()
        cuboids.sort()
        # print(cuboids)

        def fit(cubo1, cubo2):
            return cubo1[0] <= cubo2[0] and cubo1[1] <= cubo2[1] and cubo1[2] <= cubo2[2]

        dp = [0 for _ in range(n+1)]
        # print(dp)
        # dp[0] = cuboids[0][2]
        for i in range(n):
            dp[i] = cuboids[i][2]
        for i in range(1, n):
            for j in range(i):
                # print(j, i, cuboids[j], cuboids[i])
                if fit(cuboids[j], cuboids[i]):
                    dp[i] = max(dp[i], dp[j] + cuboids[i][2])
        # print(dp)
        return max(dp)
