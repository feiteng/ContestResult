class Solution:
    def maxHeight(self, A: List[List[int]]) -> int:
        '''
        12 23 45
        20 45 50
        37 53 95
        '''
        N = len(A)
        for c in A:
            c
.sort()
        A.sort()
        
        dp = [c[2] for c in A]
        
        for i in range(N-2,-1,-1):
            for j in range(i+1, N):
                if A[i][0] <= A[j][0] and A[i][1] <= A[j]
[1] and A[i][2] <= A[j][2]:
                    dp[i] = max(dp[i], dp[j]+A[i][2])
        
        # print(A, dp)
        return max(dp)
