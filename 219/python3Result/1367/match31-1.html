class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        c = [sorted(x) for x in cuboids]
        c = sorted
(c)
        #print(c)
        n = len(c)
        d = [0]*n
        d[0] = c[0][2]
        for i in range(n):
            d[i] = c[i][2]
            for j in range(i):
                if all(c[j][k]<=c[i][k] for k in range(3)):
                    d
[i] = max(d[j]+c[i][2],d[i])
        return max(d)
