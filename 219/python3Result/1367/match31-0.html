class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        for c in cuboids:
            c.sort(reverse = True)
        cuboids.sort()
        n = len(cuboids)
        dp = [0] * n
        dp[0] = cuboids[0][0]
        for i in range(1, n):
            dp[i] = cuboids[i][0]
            for j in range(i):
                if cuboids[i][0]>=cuboids[j][0] and cuboids[i][1]>=cuboids[j][1] and cuboids[i][2]>=cuboids[j][2]:
                    dp[i] = max(dp[i], dp[j] + cuboids[i][0])
        return max(dp)
