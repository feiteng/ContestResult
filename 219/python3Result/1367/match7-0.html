class Solution:
    def maxHeight(self, arr: List[List[int]]) -> int:
        for i in arr:
            i.sort()
        arr.sort(key = lambda x: (x[0],x[1],x[2]))
        n = len(arr)
        #print(arr)
        dp= [ arr[i][2] for i in range(n)]
        dp[0] = arr[0][2]
        for i in range(n):
            for j in range(i):
                if arr[j][0]<=arr[i][0] and arr[j][1]<=arr[i][1]  and arr[j][2]<=arr[i][2] :
                    dp[i] = max(dp[i], dp[j]+arr[i][2])
        return max(dp)
