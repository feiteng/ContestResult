class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        n = len(cuboids)
        for cuboid in cuboids:
            cuboid.sort(reverse=True)
        cuboids.sort()
        dp = [cuboids[i][0] for i in range(n)]
        for i in range(1, n):
            for j in range(i):
                
if cuboids[i][0] >= cuboids[j][0] and cuboids[i][1] >= cuboids[j][1] and cuboids[i][2] >= cuboids[j][2]:
                    dp[i] = max(dp[i], cuboids[i][0] + dp[j])
        return max(dp)
