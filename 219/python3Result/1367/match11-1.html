class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        for x in cuboids:
            x.sort()
        cuboids.sort()
        #print(cuboids)
        dp=[cuboids[i][2] for i in range(len(cuboids))]
        ans=0
        
for i in range(len(cuboids)):
            for j in range(i):
                if(cuboids[i][1]>=cuboids[j][1] and cuboids[i][2]>=cuboids[j][2]):
                    dp[i] = max(dp[j]+cuboids[i][2],dp[i])
            ans = max(ans,dp[i])
        return ans
