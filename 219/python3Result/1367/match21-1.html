class Solution:
    def maxHeight(self, A: List[List[int]]) -> int:
        n=len(A)
        for arr in A:
            arr.sort(reverse=True)
        A.sort(reverse=True)
        dp=[0]*n
        for i in range(n):
            dp[i]=A[i][0]
            for j in range(i):
                if A
[i][1]<=A[j][1] and A[i][2]<=A[j][2]:
                    dp[i]=max(dp[i],A[i][0]+dp[j])
        return max(dp)
