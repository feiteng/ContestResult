class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        res
=[]
        for i,(a,b,c) in enumerate(cuboids):
            res.append([a,b,c,i])
            res.append([a,c,b,i])
            res.append([b,a,c,i])
            res.append([b,c,a,i])
            res.append([c,a,b,i])
            res.append([c,b,a,i])
        res.sort()
        n=len(res)
        dp=[0]*n
        dp[0]=res[0][2]
        for i in range(1,n):
            dp[i]=res[i][2]
            for j in range(i):
                if res
[i][3]!=res[j][3] and res[j][1]<=res[i][1] and res[j][2]<=res[i][2]:
                    dp[i]=max(dp[i],dp[j]+res[i][2])
        return max(dp)
