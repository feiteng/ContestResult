class Solution:
    def maxHeight(self, cube: List[List[int]]) -> int:
        cube = [[0,0,0]] + sorted([sorted(i)[::-1] for i in cube]) # Sorted descending, rotate so max height is first.
        
        dp = [0]*(len
(cube)+1)
        for i in range(len(cube)):
            for j in range(i):
                if(cube[j][0] <= cube[i][0] and cube[j][1] <= cube[i][1] and cube[j][2] <= cube[i][2]):
                    dp[i] = max(dp[i], dp[j]+cube[i][0])
        return (max(dp))
