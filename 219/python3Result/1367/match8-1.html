class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        
        for i, cuboid in enumerate(cuboids):
            cuboids[i].sort()
        cuboids.sort(key = lambda x: (x[2], x[0], x[1]))
        n = len
(cuboids)
        #print(cuboids)
        #dp[i] = height get using i th cuboid as the bottom one.
        dp = [cuboids[i][2] for i in range(n)]

        for i in range(1, n):
            for j in range(i):
                if cuboids[j][0] <= cuboids[i][0] and cuboids[j][1] <= cuboids[i][1] and cuboids[j][2] <= cuboids[i][2]:
                    dp[i] = max(dp[i], dp[j] + cuboids[i][2])
        
        #print(dp)
        return max(dp)
