class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        length = len(cuboids)
        for i in cuboids:
            i.sort(reverse = True)
        cuboids.sort(key = lambda x : (x[0], x[1], x[2]), reverse = True)
        print(cuboids)
        dp = [0] * (length + 1)
        dp[1] = cuboids[0][0]
        for i in range(1, length):
            j = i - 1
            dp[i + 1] = cuboids[i][0]
            while j >= 0:
                if cuboids[i][0] <= cuboids[j][0] and cuboids[i][1] <= cuboids[j][1] and cuboids[i][2] <= cuboids[j][2]:
                    dp[i + 1] = max(dp[i + 1], dp[j + 1] + cuboids[i][0])
                j -= 1
            
            
        print(dp)
        ans = float('-inf')
        for i in dp:
            ans = max(ans, i)
        return ans
