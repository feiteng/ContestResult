class Solution:
    def maxHeight(self, a: List[List[int]]) -> int:
        n = len(a)
        for i in range(n):
            a[i] = sorted(a[i], reverse=True)
        a.sort(reverse=True)
        ans 
= [0] * n
        ans[0] = a[0][0]
        for i in range(1, n):
            ans[i] = a[i][0]
            for j in range(0, i):
                if a[i][1] <= a[j][1] and a[i][2] <= a[j][2]:
                    ans[i] 
= max(ans[i], ans[j] + a[i][0])
        # print(ans)
        return max(ans)
