class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        n = len(cuboids)
        t = []
        for i,(x,y,z) in enumerate(cuboids):
            t.append([x,y,z,i])
            t.append([x,z,y,i])
            t.append([y,x,z,i])
            t.append([y,z,x,i])
            t.append([z,x,y,i])
            t.append([z,y,x,i])
        t.sort();
        ans = 0
        dp = [0]*len(t)
        for i in range(len(t)):
            dp[i] = t[i][2]
            for j in range(i):
                if t[j][0]<=t[i][0] and t[j][1]<=t[i][1] and t[j][2]<=t[i][2] and t[j][3]!=t[i][3]:
                    dp[i] = max(dp[i],t[i][2]+dp[j])
        return max(dp)
