class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        cuboids = [sorted(cub) for cub in cuboids]
        cuboids.sort(key = lambda f: (-f[0], -f[1], -f[2]))
        
        #print(cuboids)
        
        heights = [cub[2] for cub in cuboids]
        
        #print(heights)
        
        for i in range(len(cuboids)):
            for j in range(0, i):
                if cuboids[j][1] >= cuboids[i][1] and cuboids[j][2] >= cuboids[i][2]:
                    heights[i] = max(heights[i], heights[j] + cuboids[i][2])
        
        #print(heights)
        
        return max(heights)
