class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        n = len(cuboids)
        for i in range(n):
            cuboids[i].sort()
        cuboids.sort()
        # dp[i] is the max height of the cuboids with the ith cuboids placed 
        dp = [c[2] for c in cuboids] 
        for i in range(1, n):
            for j in range(i-1, -1, -1):
                if all(cuboids[i][k] >= cuboids[j][k] for k in (0,1,2)):
                    dp[i] = max(dp[i], dp[j] + cuboids[i][2])
        return max(dp)
