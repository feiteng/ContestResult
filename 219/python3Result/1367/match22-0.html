class Solution:
    def maxHeight(self, cub: List[List[int]]) -> int:
        for j in cub:
            j.sort()
        cub.sort()
        dp=[cub[j][2] for j in range(len(cub))]
        for j in range(len(cub)):
            for k in range(j):
                if cub[k][2]<=cub[j][2] and cub[k][1]<=cub[j][1]  and cub[k][0]<=cub[j][0] :
                    dp[j] = max(dp[j], dp[k]+cub[j][2])
        return max(dp)
