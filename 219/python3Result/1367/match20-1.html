class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        arr = []
        n = len(cuboids)
        for cubo in cuboids:
            cubo.sort()
            arr
.append(cubo)
        arr.sort(key = lambda x: (x[2], x[1], x[0]))
#        print(arr)
        
        dp = [arr[i][2] for i in range(n)]
        for i in range(n):
            
for j in range(i):
                if arr[j][0] <= arr[i][0] and arr[j][1] <= arr[i][1]:
                    dp[i] = max(dp[i], dp[j] + arr[i][2])
#        print(dp)
        return max(dp)
