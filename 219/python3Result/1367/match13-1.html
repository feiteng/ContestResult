class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        for c in cuboids:
            c.sort()
        
        cuboids.sort()
        
        dp 
= [c[2] for c in cuboids]
        for i in range(1, len(cuboids)):
            for j in range(i):
                if cuboids[i][1] >= cuboids[j][1] and cuboids[i][2] >= cuboids[j][2]:
                    dp[i] = max(dp[i], cuboids[i][2] + dp[j])
        
        return max(dp)
