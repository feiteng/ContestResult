class Solution:
    def maxHeight(self, cuboids: List[List[int]]) -> int:
        for c in cuboids:
            c.sort()
            
        cuboids.sort(key=lambda x: (-x[2], -x[0], -x[1]))
        
        n = len(cuboids)
        dp = [0]*n
        
        for i in range(n):
            dp[i] = cuboids[i][2]
            for j in range(i):
                if cuboids[i][0] <= cuboids[j][0] and cuboids[i][1] <= cuboids[j][1]:
                    dp[i] = max(dp[i], dp[j] + cuboids[i][2])
                    
                    
        return max(dp)
