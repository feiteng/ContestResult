class Solution:
    def maxHeight(self, nums: List[List[int]]) -> int:
        nums = sorted(sorted(num) for num in nums)
        n = len(nums)
        dp = [nums[i][2] for i in range
(n)]
        for i in range(1, n):
            for j in range(i):
                if all(a <= b for a, b in zip(nums[j], nums[i])):
                    dp[i] = max(dp[i], dp[j] + nums[i][2])
        return max(dp)
