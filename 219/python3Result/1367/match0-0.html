class Solution:
    def maxHeight(self, data: List[List[int]]) -> int:
        n = len(data)
        for i in range(n):
            data[i].sort()
        data.sort(reverse=True)
        dp = [data[i][-1] for i in range(n)]
        # print(data)
        for i in range(1, n):
            for j in range(i):
                if all(data[j][k] >= data[i][k] for k in range(3)):
                    dp[i] = max(dp[i], dp[j] + data[i][-1])
        
        # print(dp)
        return max(dp)
