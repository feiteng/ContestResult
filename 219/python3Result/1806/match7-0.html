class Solution:
    def numberOfMatches(self, n: int) -> int:
        cnt = 0
        while n!=1:
            if n%2:
                cnt += (n-1)//2
                n = (n-1)//2+1
                # cnt += 
            else:
                cnt += n//2
                n //= 2
        return cnt
