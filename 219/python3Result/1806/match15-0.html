class Solution:
    def numberOfMatches(self, n: int) -> int:
        s = 0
        # print("Case: ", n)
        while n > 1:
            # print(n, s)
            if n % 2 == 0:
                s += n // 2
                n = n // 2
            else:
                s += n // 2
                n = n // 2 + 1
        # print(n,s)
        return s
